// server/engine/expr.ts
import type { CaseState } from "./caseState";

function getPath(obj: any, path: string): any {
  const parts = path.split(".");
  let cur = obj;
  for (const p of parts) {
    if (cur == null) return undefined;
    cur = cur[p];
  }
  return cur;
}

function stripQuotes(s: string): string {
  const t = s.trim();
  if ((t.startsWith("'") && t.endsWith("'")) || (t.startsWith('"') && t.endsWith('"'))) {
    return t.slice(1, -1);
  }
  return t;
}

function parseLiteral(tok: string): any {
  const t = tok.trim();
  if (t === "true") return true;
  if (t === "false") return false;
  if (t === "null") return null;
  if (/^-?\d+(\.\d+)?$/.test(t)) return Number(t);
  if ((t.startsWith("'") && t.endsWith("'")) || (t.startsWith('"') && t.endsWith('"'))) return stripQuotes(t);
  return undefined; // not a literal
}

// Very small parser: split OR then AND then evaluate atomic conditions
export function evalExpr(expr: string, state: CaseState): boolean {
  const e = (expr || "").trim();
  if (!e || e === "true") return true;
  if (e === "false") return false;

  // OR
  const orParts = splitTopLevel(e, "||");
  if (orParts.length > 1) return orParts.some(p => evalExpr(p, state));

  // AND
  const andParts = splitTopLevel(e, "&&");
  if (andParts.length > 1) return andParts.every(p => evalExpr(p, state));

  // Atomic: "a in ['x','y']"
  const inMatch = e.match(/^\s*([A-Za-z0-9_.]+)\s+in\s+\[(.*)\]\s*$/);
  if (inMatch) {
    const leftPath = inMatch[1].trim();
    const listRaw = inMatch[2].trim();
    const items = splitCSVLike(listRaw).map(x => stripQuotes(x.trim()));
    const val = getPath(state as any, leftPath);
    return items.includes(String(val));
  }

  // Comparisons
  const ops = ["==", "!=", ">=", "<=", ">", "<"] as const;
  for (const op of ops) {
    const idx = indexOfOp(e, op);
    if (idx >= 0) {
      const left = e.slice(0, idx).trim();
      const right = e.slice(idx + op.length).trim();
      const leftVal = getPath(state as any, left);
      const lit = parseLiteral(right);
      const rightVal = lit !== undefined ? lit : getPath(state as any, right);

      switch (op) {
        case "==": return leftVal === rightVal;
        case "!=": return leftVal !== rightVal;
        case ">=": return Number(leftVal) >= Number(rightVal);
        case "<=": return Number(leftVal) <= Number(rightVal);
        case ">": return Number(leftVal) > Number(rightVal);
        case "<": return Number(leftVal) < Number(rightVal);
      }
    }
  }

  // Fallback: treat as truthy path lookup
  const v = getPath(state as any, e);
  return Boolean(v);
}

function indexOfOp(s: string, op: string): number {
  // ensure we match the operator not inside quotes (simple scan)
  let inQ: "'" | '"' | null = null;
  for (let i = 0; i <= s.length - op.length; i++) {
    const ch = s[i];
    if ((ch === "'" || ch === '"') && (i === 0 || s[i - 1] !== "\\")) {
      inQ = inQ === ch ? null : (ch as any);
    }
    if (!inQ && s.slice(i, i + op.length) === op) return i;
  }
  return -1;
}

function splitTopLevel(s: string, sep: "||" | "&&"): string[] {
  const out: string[] = [];
  let buf = "";
  let inQ: "'" | '"' | null = null;
  for (let i = 0; i < s.length; i++) {
    const ch = s[i];
    if ((ch === "'" || ch === '"') && (i === 0 || s[i - 1] !== "\\")) {
      inQ = inQ === ch ? null : (ch as any);
    }
    if (!inQ && s.slice(i, i + sep.length) === sep) {
      out.push(buf.trim());
      buf = "";
      i += sep.length - 1;
      continue;
    }
    buf += ch;
  }
  if (buf.trim()) out.push(buf.trim());
  return out;
}

function splitCSVLike(s: string): string[] {
  // split on commas not inside quotes
  const out: string[] = [];
  let buf = "";
  let inQ: "'" | '"' | null = null;
  for (let i = 0; i < s.length; i++) {
    const ch = s[i];
    if ((ch === "'" || ch === '"') && (i === 0 || s[i - 1] !== "\\")) {
      inQ = inQ === ch ? null : (ch as any);
    }
    if (!inQ && ch === ",") {
      out.push(buf);
      buf = "";
      continue;
    }
    buf += ch;
  }
  if (buf.length) out.push(buf);
  return out;
}

// server/engine/rules.ts
import type { CaseState, RedFlagHit } from "./caseState";
import type { RedFlagRuleRow, ScoringDefRow, DispositionRuleRow } from "./complaintConfig";
import { evalExpr } from "./expr";

function severityRank(s: string): number {
  switch (s) {
    case "CRITICAL": return 4;
    case "HIGH": return 3;
    case "MODERATE": return 2;
    case "LOW": return 1;
    default: return 0;
  }
}

export function runRedFlags(state: CaseState, rules: RedFlagRuleRow[]): { hits: RedFlagHit[]; anySeverity?: string } {
  const hits: RedFlagHit[] = [];
  for (const r of rules) {
    if (r.CC_ID !== state.cc_id) continue;
    if (evalExpr(r.TRIGGER_EXPR, state)) {
      hits.push({
        rf_id: r.RF_ID,
        severity: r.SEVERITY,
        action: r.ACTION,
        output_template_id: r.OUTPUT_TEMPLATE_ID,
        why: r.WHY || undefined,
      });
    }
  }
  let max: string | undefined;
  for (const h of hits) {
    if (!max || severityRank(h.severity) > severityRank(max)) max = h.severity;
  }
  return { hits, anySeverity: max };
}

/**
 * Scoring: supports "patient-reported default overridden by exam".
 * Strategy:
 * - You can define patient-reported fields in symptoms.* (e.g., symptoms.exudate_reported).
 * - Exam values live in exam.*; if exam value is set, prefer it.
 *
 * For Centor components in WHEN_EXPR, you can reference either:
 * - exam.tonsillar_exudate == true
 * - OR symptoms.tonsillar_exudate_reported == true
 *
 * We'll set a convenience overlay in state before eval:
 * - if exam.X is defined, set derived field overlay.exam_or_patient_X
 * - else derived uses symptoms reported fields.
 *
 * For v1, simplest is: store patient-reported into exam.*? No.
 * We'll add "derived" keys onto state at runtime.
 */
export function runCentor(state: CaseState, defs: ScoringDefRow[]): void {
  const components: Record<string, { points: number; hit: boolean; source?: "patient" | "exam" }> = {};
  const missing: string[] = [];

  // derived overlay: prefer exam if present
  const derived: any = (state as any).__derived || {};
  derived.exudate = state.exam.tonsillar_exudate ?? (state as any).symptoms?.tonsillar_exudate_reported;
  derived.tender_nodes = state.exam.tender_ant_cerv_nodes ?? (state as any).symptoms?.tender_ant_cerv_nodes_reported;

  (state as any).__derived = derived;

  let total = 0;
  for (const d of defs) {
    if (d.SCORING_ID !== "CENTOR_v1") continue; // or match config.registry.SCORING_ID upstream
    const pts = Number(d.POINTS || 0);

    // Detect missing fields for audit: if expression references exam.* or symptoms.* fields that are undefined
    const hit = evalExpr(
      d.WHEN_EXPR
        .replaceAll("exam.tonsillar_exudate", "__derived.exudate")
        .replaceAll("exam.tender_ant_cerv_nodes", "__derived.tender_nodes"),
      state as any
    );

    // Determine source if relevant
    let source: "patient" | "exam" | undefined;
    if (d.COMPONENT_ID === "EXUDATE") {
      source = state.exam.tonsillar_exudate != null ? "exam" : "patient";
      if (derived.exudate == null) missing.push("exudate");
    }
    if (d.COMPONENT_ID === "TENDER_ANT_CERV") {
      source = state.exam.tender_ant_cerv_nodes != null ? "exam" : "patient";
      if (derived.tender_nodes == null) missing.push("tender_ant_cerv_nodes");
    }
    if (d.COMPONENT_ID === "FEVER") {
      if (state.symptoms.fever == null) missing.push("fever");
      source = "patient";
    }
    if (d.COMPONENT_ID === "NO_Cough") {
      if (state.symptoms.cough == null) missing.push("cough");
      source = "patient";
    }

    components[d.COMPONENT_ID] = { points: pts, hit, source };
    if (hit) total += pts;
  }

  state.scores.centor_points = total;
  state.scores.centor_components = components;
  state.scores.centor_missing = Array.from(new Set(missing));
}

export function runDisposition(state: CaseState, rules: DispositionRuleRow[], anySeverity?: string): void {
  // Provide helper fields for expressions
  (state as any).red_flags_any_severity = anySeverity || null;
  (state as any).scores_centor_points = state.scores.centor_points ?? 0;

  const sorted = [...rules]
    .filter(r => r.CC_ID === state.cc_id)
    .sort((a, b) => Number(a.PRIORITY) - Number(b.PRIORITY));

  for (const r of sorted) {
    const expr = (r.WHEN_EXPR || "")
      .replaceAll("red_flags.any_severity", "red_flags_any_severity")
      .replaceAll("scores.centor_points", "scores_centor_points");

    if (evalExpr(expr, state)) {
      state.disposition = {
        level: r.DISPOSITION_LEVEL,
        rule_id: r.DISP_RULE_ID,
        rationale_template_id: r.RATIONALE_TEMPLATE_ID,
        confidence_hint: r.CONFIDENCE_HINT,
      };
      return;
    }
  }

  // fallback
  state.disposition = { level: "PCP_OR_UC", rule_id: "NONE", rationale_template_id: "OUT_DISP_DEFAULT", confidence_hint: "LOW" };
}

function parseChoices(raw: string): string[] | undefined {
  const s = (raw || "").trim();
  if (!s) return undefined;
  return s.split("|").map(x => x.trim()).filter(Boolean);
}

function coerceAnswer(answerType: string, raw: any): any {
  if (raw == null) return raw;
  const t = (answerType || "").toLowerCase();
  if (t === "yesno") return raw === true || raw === "true" || raw === "yes" || raw === 1 || raw === "1";
  if (t === "number") return typeof raw === "number" ? raw : Number(String(raw));
  if (t === "choice") return String(raw);
  return raw;
}

function setByPath(state: any, path: string, value: any) {
  if (!path) return;
  const parts = path.split(".");
  let cur = state;
  for (let i = 0; i < parts.length - 1; i++) {
    const p = parts[i];
    if (cur[p] == null || typeof cur[p] !== "object") cur[p] = {};
    cur = cur[p];
  }
  cur[parts[parts.length - 1]] = value;
}

function firstUnansweredQuestion(state: CaseState, questions: CoreQuestionRow[]): CoreQuestionRow | null {
  const sorted = [...questions].sort((a, b) => Number(a.ORDER) - Number(b.ORDER));
  for (const q of sorted) {
    if (q.CC_ID !== state.cc_id) continue;

    // ASK_IF condition
    const askIf = (q.ASK_IF || "").trim();
    if (askIf && !evalExpr(askIf, state)) continue;

    const required = (q.REQUIRED || "").toUpperCase() === "TRUE";
    const answered = Object.prototype.hasOwnProperty.call(state.answers, q.QUESTION_ID);

    if (!answered && (required || true)) {
      // We return next question regardless; the UI can keep going until required done.
      return q;
    }
  }
  return null;
}

function computeConfidence(state: CaseState): "HIGH" | "MODERATE" | "LOW" {
  // Simple v1: low if any critical missing or score missing key inputs
  const requiredCore = ["ST_DUR", "ST_FEVER", "ST_COUGH", "ST_SWALLOW", "ST_DROOL", "ST_BREATH", "ST_TRISMUS"];
  const missingRequired = requiredCore.filter(id => !Object.prototype.hasOwnProperty.call(state.answers, id));
  if (missingRequired.length > 0) return "LOW";
  if ((state.scores.centor_missing || []).length > 0) return "MODERATE";
  return "HIGH";
}

  // symptoms.tonsillar_exudate_reported, symptoms.tender_ant_cerv_nodes_reported
}

export function runSoreThroatGraph(config: ComplaintConfig, state: CaseState): RunResult {
  const t0 = Date.now();

  // Node: CORE QUESTIONS (ask next)
  const q = firstUnansweredQuestion(state, config.coreQuestions);
  if (q) {
    state.next_question = {
      question_id: q.QUESTION_ID,
      question_text: q.QUESTION_TEXT,
      answer_type: q.ANSWER_TYPE,
      choices: parseChoices(q.CHOICES),
      maps_to_field: q.MAPS_TO_FIELD || undefined,
    };

    tracePush(state, {
      node_id: "ST_CORE_QUESTIONS",
      inputs_used: ["answers", "CORE_QUESTIONS"],
      outputs: { next_question_id: q.QUESTION_ID },
      rule_refs: [],
      llm_calls: 0,
      latency_ms: Date.now() - t0,
      confidence: state.case_confidence,
    });

    return { done: false, state };
  }

  state.next_question = null;

  // Node: RED FLAGS
  const rf0 = Date.now();
  const { hits, anySeverity } = runRedFlags(state, config.redFlagRules);
  state.red_flags = hits;

  tracePush(state, {
    node_id: "ST_RED_FLAG_GATE",
    inputs_used: ["symptoms", "vitals", "patient_profile"],
    outputs: { hit_count: hits.length, any_severity: anySeverity || null },
    rule_refs: hits.map(h => h.rf_id),
    llm_calls: 0,
    latency_ms: Date.now() - rf0,
  });

  // If critical, output immediately
  if (anySeverity === "CRITICAL") {
    const templateId = hits[0]?.output_template_id || "OUT_EMERGENT_AIRWAY";
    const tpl = getTemplateById(config.templates, templateId) || getTemplateById(config.templates, "OUT_EMERGENT_AIRWAY");
    if (!tpl) throw new Error(`Missing template ${templateId}`);

    const rendered = renderTemplate(tpl, {});
    state.disposition = { level: "ED_NOW", rule_id: hits[0]?.rf_id, rationale_template_id: templateId, confidence_hint: "HIGH" };
    state.case_confidence = "HIGH";

    tracePush(state, {
      node_id: "ST_OUTPUT_COMPOSE",
      inputs_used: ["red_flags", "OUTPUT_TEMPLATES"],
      outputs: { template_id: templateId, disposition: state.disposition.level },
      rule_refs: [templateId],
      llm_calls: 0,
      latency_ms: Date.now() - t0,
      confidence: "HIGH",
      notes: "Critical red flag short-circuit",
    });

    return { done: true, state, output: { ...rendered, templateId, dispositionLevel: state.disposition.level } };
  }

  // Node: SCORING (Centor)
  const sc0 = Date.now();
  runCentor(state, config.scoringDefs);
  tracePush(state, {
    node_id: "ST_CENTOR_SCORE",
    inputs_used: ["symptoms", "exam", "SCORING_DEFS"],
    outputs: { centor_points: state.scores.centor_points, missing: state.scores.centor_missing || [] },
    rule_refs: ["CENTOR_v1"],
    llm_calls: 0,
    latency_ms: Date.now() - sc0,
  });

  // Node: DISPOSITION
  const d0 = Date.now();
  runDisposition(state, config.dispositionRules, anySeverity);
  state.case_confidence = computeConfidence(state);

  tracePush(state, {
    node_id: "ST_DISPOSITION_RULES",
    inputs_used: ["red_flags", "scores", "DISPOSITION_RULES"],
    outputs: { disposition: state.disposition.level, rule_id: state.disposition.rule_id },
    rule_refs: [state.disposition.rule_id || "NONE"],
    llm_calls: 0,
    latency_ms: Date.now() - d0,
    confidence: state.case_confidence,
  });

  // Node: OUTPUT COMPOSE (template-based; LLM can be added later for nicer phrasing)
  const templateId = state.disposition.rationale_template_id || "OUT_DISP_DEFAULT";
  const tpl = getTemplateById(config.templates, templateId) || getTemplateById(config.templates, "OUT_DISP_DEFAULT");
  if (!tpl) throw new Error(`Missing template ${templateId}`);

  const rendered = renderTemplate(tpl, {
    centor_points: state.scores.centor_points ?? 0,
    duration_days: state.symptoms.duration_days ?? "",
  });

  tracePush(state, {
    node_id: "ST_OUTPUT_COMPOSE",
    inputs_used: ["disposition", "OUTPUT_TEMPLATES"],
    outputs: { template_id: templateId, disposition: state.disposition.level },
    rule_refs: [templateId],
    llm_calls: 0,
    latency_ms: Date.now() - t0,
    confidence: state.case_confidence,
  });

  return { done: true, state, output: { ...rendered, templateId, dispositionLevel: state.disposition.level } };
}

function newCase(conversation_id: string, cc_id: string): CaseState {

router.post("/start", async (req, res) => {
  const conversation_id = String(req.body.conversation_id || "");
  const cc_id = String(req.body.cc_id || "");
  if (!conversation_id || !cc_id) return res.status(400).json({ error: "conversation_id and cc_id required" });

  const state = newCase(conversation_id, cc_id);
  CASES.set(conversation_id, state);

router.post("/answer", async (req, res) => {
  const conversation_id = String(req.body.conversation_id || "");
  const question_id = String(req.body.question_id || "");
  const answer = req.body.answer;

  const state = CASES.get(conversation_id);

  // find question row

  const result = runSoreThroatGraph(config, state);
  CASES.set(conversation_id, result.state);

  return res.json({
    done: result.done,
    next_question: result.state.next_question,
    output: result.output || null,
    state_meta: { cc_id: state.cc_id, confidence: result.state.case_confidence, disposition: result.state.disposition.level || null },
  });
});

export async function loadComplaintConfigCached(cc_id: string): Promise<ComplaintConfig> {
  const now = Date.now();
  const hit = CACHE.get(cc_id);
  if (hit && now - hit.ts < TTL_MS) return hit.config;

  const config = await loadComplaintConfig(cc_id);
  CACHE.set(cc_id, { ts: now, config });
  return config;
}

export async function loadComplaintConfig(cc_id: string): Promise<ComplaintConfig> {
  const registryRows = (await readSheetTab("COMPLAINT_REGISTRY")) as any as ComplaintRegistryRow[];
  const registry = registryRows.find(r => r.CC_ID === cc_id);
  if (!registry) throw new Error(`No COMPLAINT_REGISTRY row for ${cc_id}`);
  if ((registry.ENABLED || "").toUpperCase() !== "TRUE") throw new Error(`Complaint ${cc_id} not enabled`);

  const coreAll = (await readSheetTab("CORE_QUESTIONS")) as any as CoreQuestionRow[];
  const redAll = (await readSheetTab("RED_FLAG_RULES")) as any as RedFlagRuleRow[];
  const scoreAll = (await readSheetTab("SCORING_DEFS")) as any as ScoringDefRow[];
  const dispAll = (await readSheetTab("DISPOSITION_RULES")) as any as DispositionRuleRow[];
  const tmplAll = (await readSheetTab("OUTPUT_TEMPLATES")) as any as OutputTemplateRow[];

  const coreQuestions = coreAll.filter(r => r.CC_ID === cc_id && r.VERSION === registry.CORE_QUESTIONS_VERSION);
  const redFlagRules = redAll.filter(r => r.CC_ID === cc_id && r.RF_SET_ID === registry.RED_FLAG_SET_ID);
  const scoringDefs = scoreAll.filter(r => r.SCORING_ID === registry.SCORING_ID);
  const dispositionRules = dispAll.filter(r => r.CC_ID === cc_id && r.DISP_SET_ID === registry.DISPOSITION_SET_ID);

  // templates: include both the complaint set + shared templates
  const templates = tmplAll.filter(
    r => r.TEMPLATE_SET_ID === registry.OUTPUT_TEMPLATE_SET_ID || r.TEMPLATE_SET_ID === "OUT_SHARED"
  );

  return { registry, coreQuestions, redFlagRules, scoringDefs, dispositionRules, templates };
}
