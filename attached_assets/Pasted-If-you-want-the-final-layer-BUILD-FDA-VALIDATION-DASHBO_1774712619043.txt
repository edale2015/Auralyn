export function computeMetrics(results: any[]) {
  let TP = 0, FP = 0, FN = 0, TN = 0;

  results.forEach(r => {
    if (r.correct) TP++;
    else FP++;
  });

  const total = results.length;

  const accuracy = TP / total;
  const precision = TP / (TP + FP || 1);
  const recall = TP / (TP + FN || 1);
  const f1 = 2 * (precision * recall) / (precision + recall || 1);

  return {
    total,
    accuracy,
    precision,
    recall,
    f1
  };
}

export function stratify(results: any[]) {
  const groups = {
    pediatric: [],
    adult: [],
    highRisk: [],
    lowRisk: []
  };

  results.forEach(r => {
    if (r.input.age < 18) groups.pediatric.push(r);
    else groups.adult.push(r);

    if (r.safety === "HIGH") groups.highRisk.push(r);
    else groups.lowRisk.push(r);
  });

  return groups;
}

import { computeMetrics } from "./validationEngine";
import { stratify } from "./stratified";

export async function runValidation(cases: any[]) {
  const results = [];

  for (const c of cases) {
    const output = await runFullClinicalFlow(c.input);

    results.push({
      input: c.input,
      expected: c.expected,
      actual: output,
      correct: c.expected.diagnosis === output.diagnosis,
      safety: output.safetyLevel || "UNKNOWN"
    });
  }

  const overall = computeMetrics(results);
  const groups = stratify(results);

  const groupMetrics = {};
  for (const key in groups) {
    groupMetrics[key] = computeMetrics(groups[key]);
  }

  return {
    overall,
    groupMetrics,
    totalCases: results.length
  };
}

export function lockVersion(v: string) {
  lockedVersions.add(v);

export function isLocked(v: string) {
