// server/document/parser.ts
import fs from "fs";
import { execSync } from "child_process";

export async function parseDocument(filePath: string): Promise<string> {
  try {
    // Local parsing using docling CLI (recommended HIPAA-safe)
    const outputPath = filePath + ".md";

    execSync(`docling convert ${filePath} --output ${outputPath}`);

    const markdown = fs.readFileSync(outputPath, "utf-8");

    return markdown;
  } catch (err) {
    console.error("Local parsing failed, fallback triggered");

    // Optional fallback (cloud parsing)
    return await llamaParseFallback(filePath);
  }
}

async function llamaParseFallback(filePath: string): Promise<string> {
  const formData = new FormData();
  formData.append("file", fs.createReadStream(filePath));

  const res = await fetch("https://api.llamaparse.com/parse", {
    method: "POST",
    headers: { Authorization: `Bearer ${process.env.LLAMA_API_KEY}` },
    body: formData as any,
  });

  const data = await res.json();
  return data.markdown;
}

2. Smart Chunking Engine (Clinical-aware)

Upgrade from naive chunking → semantic + section-aware chunking

// server/document/chunker.ts
export function chunkDocument(text: string) {
  const sections = text.split(/## |### /); // preserve headings

  const chunks: string[] = [];

  for (const section of sections) {
    const words = section.split(" ");
    let buffer: string[] = [];

    for (let i = 0; i < words.length; i++) {
      buffer.push(words[i]);

      if (buffer.length >= 200) {
        chunks.push(buffer.join(" "));
        buffer = buffer.slice(100); // overlap
      }
    }

    if (buffer.length > 0) {
      chunks.push(buffer.join(" "));
    }
  }

  return chunks;
}

// server/document/embedding.ts
import { OpenAIEmbeddings } from "@langchain/openai";
import { Pinecone } from "@pinecone-database/pinecone";

const embeddings = new OpenAIEmbeddings({
  apiKey: process.env.OPENAI_API_KEY,
});

const pinecone = new Pinecone({
  apiKey: process.env.PINECONE_API_KEY!,
});

const index = pinecone.Index("medical-docs");

export async function embedChunks(chunks: string[], docId: string) {
  const vectors = [];

  for (let i = 0; i < chunks.length; i++) {
    const vector = await embeddings.embedQuery(chunks[i]);

    vectors.push({
      id: `${docId}-${i}`,
      values: vector,
      metadata: {
        text: chunks[i],
        docId,
      },
    });
  }

  await index.upsert(vectors);
}

// server/document/retriever.ts
export async function retrieveRelevant(query: string) {
  const queryVector = await embeddings.embedQuery(query);

  const results = await index.query({
    vector: queryVector,
    topK: 5,
    includeMetadata: true,
  });

  return results.matches.map((m) => m.metadata?.text).join("\n\n");
}

// server/document/knowledgeExtractor.ts
export function extractClinicalEntities(text: string) {
  const diagnoses = [];
  const medications = [];
  const redFlags = [];

  const lines = text.split("\n");

  for (const line of lines) {
    if (line.toLowerCase().includes("diagnosis")) {
      diagnoses.push(line);
    }
    if (line.toLowerCase().includes("mg") || line.includes("dose")) {
      medications.push(line);
    }
    if (line.toLowerCase().includes("emergency")) {
      redFlags.push(line);
    }
  }

  return { diagnoses, medications, redFlags };
}

// server/audit/documentAudit.ts
import crypto from "crypto";

export function logDocumentDecision(input: any, output: any) {
  const record = {
    timestamp: Date.now(),
    input,
    output,
  };

  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(record))
    .digest("hex");

  return { ...record, hash };
}
