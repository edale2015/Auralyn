function getSheetsClient() {
  const auth = new google.auth.GoogleAuth({
    scopes: ["https://www.googleapis.com/auth/spreadsheets"],
  });

  return google.sheets({ version: "v4", auth });
}

function getSpreadsheetId() {
  const id = process.env.PACKS_SPREADSHEET_ID;
  if (!id) {
    throw new Error("Missing PACKS_SPREADSHEET_ID");
  }
  return id;
}

async function getSpreadsheetMeta() {
  const sheets = getSheetsClient();
  const spreadsheetId = getSpreadsheetId();

  const meta = await sheets.spreadsheets.get({ spreadsheetId });
  return { sheets, spreadsheetId, meta };
}

async function ensureHeader(tabName: string) {
  const { sheets, spreadsheetId } = await getSpreadsheetMeta();
  const header = headerRowFor(tabName);

  await sheets.spreadsheets.values.update({
    spreadsheetId,
    range: `${tabName}!A1:Z1`,
    valueInputOption: "RAW",
    requestBody: {
      values: [header],
    },
  });
}

export async function ensureCanonicalTabs() {
  const { sheets, spreadsheetId, meta } = await getSpreadsheetMeta();

  const existingTitles = new Set(
    (meta.data.sheets || []).map(s => s.properties?.title).filter(Boolean)
  );

  const missing = CANONICAL_TABS.filter(t => !existingTitles.has(t));

  if (missing.length > 0) {
    await sheets.spreadsheets.batchUpdate({
      spreadsheetId,
      requestBody: {
        requests: missing.map(title => ({
          addSheet: {
            properties: { title },
          },
        })),
      },
    });
  }

  for (const tab of CANONICAL_TABS) {
    await ensureHeader(tab);
  }

  await appendPackAuditLog({
    entityType: "plan_template",
    entityId: "canonical_tabs",
    action: "update",
    actorId: "system",
    afterJson: JSON.stringify({ ensuredTabs: CANONICAL_TABS }),
    validationOk: true,
    validationIssuesJson: "[]",
    notes: "Canonical tabs ensured",
  });

  return {
    existing: [...existingTitles],
    created: missing,
    canonicalTabs: CANONICAL_TABS,
  };
}

async function readAllTabsAsObjects() {
  const { sheets, spreadsheetId, meta } = await getSpreadsheetMeta();
  const titles = (meta.data.sheets || [])
    .map(s => s.properties?.title)
    .filter((x): x is string => Boolean(x));

  const result: Record<string, string[][]> = {};

  for (const title of titles) {
    const res = await sheets.spreadsheets.values.get({
      spreadsheetId,
      range: `${title}!A:Z`,
    });
    result[title] = res.data.values || [];
  }

  return result;
}

export async function runDryRunMigration() {
  const workbookTabs = await readAllTabsAsObjects();

  const mapped = legacyMapWorkbookTabs(workbookTabs);

  const validations = {
    symptom: mapped.symptomRows.map(row => ({
      id: row.id,
      validation: validateAnyPackRow(row),
    })),
    modifier: mapped.modifierRows.map(row => ({
      id: row.id,
      validation: validateAnyPackRow(row),
    })),
    algorithm: mapped.algorithmRows.map(row => ({
      id: row.id,
      validation: validateAnyPackRow(row),
    })),
    question: mapped.questionRows.map(row => ({
      id: row.id,
      validation: validatePackQuestionRow(row),
    })),
  };

  return {
    summary: {
      symptomRows: mapped.symptomRows.length,
      modifierRows: mapped.modifierRows.length,
      algorithmRows: mapped.algorithmRows.length,
      questionRows: mapped.questionRows.length,
      issues: mapped.issues.length,
    },
    mapped,
    validations,
  };
}

function joinPipe(values?: string[]) {
  return (values || []).join("|");
}

async function appendRows(tabName: string, values: string[][]) {
  if (values.length === 0) return;
  const { sheets, spreadsheetId } = await getSpreadsheetMeta();

  await sheets.spreadsheets.values.append({
    spreadsheetId,
    range: `${tabName}!A:Z`,
    valueInputOption: "RAW",
    requestBody: { values },
  });
}

export async function applyMigration() {
  const dryRun = await runDryRunMigration();
  const { mapped } = dryRun;

  const validSymptomRows = mapped.symptomRows.filter(

    row => validateAnyPackRow(row).ok

    row => validatePackQuestionRow(row).ok

export async function runMigrationQa() {
  const { sheets, spreadsheetId } = await getSpreadsheetMeta();

  const tabsToCheck = [
    "Symptom_Packs",
    "Pack_Questions",
    "Modifier_Packs",
    "Clinician_Algorithms",
  ];

  const counts: Record<string, number> = {};

  for (const tab of tabsToCheck) {
    const res = await sheets.spreadsheets.values.get({
      spreadsheetId,
      range: `${tab}!A2:A`,
    });
    counts[tab] = (res.data.values || []).length;
  }

  const qa = {
    counts,
    ok:
      counts["Symptom_Packs"] > 0 &&
      counts["Pack_Questions"] > 0,
  };

  await appendPackAuditLog({
    entityType: "plan_template",
    entityId: "google_sheets_run_qa",
    action: "validate",
    actorId: "system",
    afterJson: JSON.stringify(qa),
    validationOk: qa.ok,
    validationIssuesJson: JSON.stringify([]),
    notes: "Migration QA completed",
  });

  return qa;
}

import crypto from "crypto";
import {
  SymptomPackRow,
  ModifierPackRow,
  ClinicianAlgorithmRow,
} from "../../shared/packRows";
import { PackQuestionRow } from "../../shared/packQuestionRows";

export interface LegacyMapResult {
  symptomRows: SymptomPackRow[];
  modifierRows: ModifierPackRow[];
  algorithmRows: ClinicianAlgorithmRow[];
  questionRows: PackQuestionRow[];
  issues: string[];
}

function slugify(input: string) {
  return input
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "_")
    .replace(/^_+|_+$/g, "");
}

function rowsToObjects(rows: string[][]) {
  if (!rows.length) return [];
  const header = rows[0].map(x => String(x || "").trim());
  return rows.slice(1).map(r => {
    const obj: Record<string, string> = {};
    for (let i = 0; i < header.length; i++) {
      obj[header[i]] = r[i] || "";
    }
    return obj;
  });
}

function detectSystemFromTab(tabName: string) {
  const n = tabName.toLowerCase();
  if (n.includes("ent")) return "ent";
  if (n.includes("card")) return "cardiology";
  if (n.includes("ortho")) return "ortho";
  if (n.includes("pulm")) return "pulmonary";
  if (n.includes("gi")) return "gi";
  if (n.includes("neuro")) return "neuro";
  if (n.includes("derm")) return "derm";
  if (n.includes("uro")) return "urology";
  if (n.includes("endo")) return "endocrinology";
  if (n.includes("tox")) return "toxicology";
  if (n.includes("psych")) return "psych";
  if (n.includes("eye")) return "eye";
  if (n.includes("allergy")) return "allergy_immunology";
  if (n.includes("environment")) return "environment";
  return "general";
}

function parsePipeOrCsv(text: string) {
  if (!text) return [];
  return text
    .split(/[|,]/)
    .map(x => x.trim())
    .filter(Boolean);
}

export function legacyMapWorkbookTabs(
  workbookTabs: Record<string, string[][]>
): LegacyMapResult {
  const result: LegacyMapResult = {
    symptomRows: [],
    modifierRows: [],
    algorithmRows: [],
    questionRows: [],
    issues: [],
  };

  for (const [tabName, rawRows] of Object.entries(workbookTabs)) {
    const lower = tabName.toLowerCase();
    if (!rawRows.length) continue;

    const objects = rowsToObjects(rawRows);
    const system = detectSystemFromTab(tabName);

    if (lower.includes("complaint") || lower.includes("chief") || lower.includes("symptom")) {
      for (const row of objects) {
        const title =
          row["title"] ||
          row["complaint"] ||
          row["chief_complaint"] ||
          row["symptom"] ||
          "";

        if (!title) continue;

        const id = `${system}_${slugify(title)}`;

        result.symptomRows.push({
          id,
          system,
          tier: "symptom",
          title,
          isActive: true,
          version: 1,
          tags: parsePipeOrCsv(row["tags"] || ""),
          aliases: parsePipeOrCsv(row["aliases"] || row["synonyms"] || title),
          likelyDisposition: (row["likelyDisposition"] ||
            row["disposition"] ||
            "telemed_now") as any,
          questionsJson: "[]",
          redFlags: parsePipeOrCsv(row["redFlags"] || row["red_flags"] || ""),
          autoEscalateRules: parsePipeOrCsv(
            row["autoEscalateRules"] || row["escalate_rules"] || ""
          ),
          autoReviewRules: parsePipeOrCsv(
            row["autoReviewRules"] || row["review_rules"] || ""
          ),
          planTemplateKey: row["planTemplateKey"] || id,
        });
      }
    }

    if (lower.includes("secondary") || lower.includes("question")) {
      for (const row of objects) {
        const packRef =
          row["packId"] ||
          row["complaintId"] ||
          row["chief_complaint"] ||
          row["complaint"] ||
          row["symptom"] ||
          "";

        const prompt = row["prompt"] || row["question"] || row["text"] || "";
        if (!packRef || !prompt) continue;

        const packId =
          packRef.includes("_") ? packRef : `${system}_${slugify(packRef)}`;

        const questionId = slugify(
          row["questionId"] || row["field"] || prompt.slice(0, 40)
        );

        result.questionRows.push({
          id: crypto.randomUUID(),
          packId,
          questionId,
          prompt,
          type: (row["type"] || "yes_no") as any,
          priority: Number(row["priority"] || 1),
          required: String(row["required"]).toLowerCase() === "true",
          optionsJson: row["optionsJson"] || "",
          helpText: row["helpText"] || "",
          isActive: true,
          version: 1,
        });
      }
    }

    if (lower.includes("modifier")) {
      for (const row of objects) {
        const title = row["title"] || row["modifier"] || row["name"] || "";
        if (!title) continue;

        const id = `${system}_mod_${slugify(title)}`;

        const appliesToSymptoms = parsePipeOrCsv(
          row["appliesToSymptoms"] || row["applies_to"] || row["complaints"] || ""
        ).map(x => (x.includes("_") ? x : `${system}_${slugify(x)}`));

        const triggers = parsePipeOrCsv(row["triggers"] || row["trigger_rules"] || "");

        const riskAdjustments = row["riskAdjustmentsJson"]
          ? row["riskAdjustmentsJson"]
          : JSON.stringify([
              {
                condition: triggers[0] || "ANY_RED_FLAG=true",
                action: "force_review",
                reason: title,
              },
            ]);

        result.modifierRows.push({
          id,
          system,
          tier: "modifier",
          title,
          isActive: true,
          version: 1,
          tags: parsePipeOrCsv(row["tags"] || ""),
          appliesToSymptoms,
          triggers,
          riskAdjustmentsJson: riskAdjustments,
        });
      }
    }

    if (lower.includes("triage") || lower.includes("algorithm") || lower.includes("acls")) {
      for (const row of objects) {
        const title = row["title"] || row["algorithm"] || row["name"] || "";
        if (!title) continue;

        const id = `${system}_algo_${slugify(title)}`;

        result.algorithmRows.push({
          id,
          system,
          tier: "clinician_algorithm",
          title,
          isActive: true,
          version: 1,
          tags: parsePipeOrCsv(row["tags"] || ""),
          entryCriteria: parsePipeOrCsv(row["entryCriteria"] || row["entry_criteria"] || ""),
          requiredInputs: parsePipeOrCsv(row["requiredInputs"] || row["required_inputs"] || ""),
          outputActions: parsePipeOrCsv(row["outputActions"] || row["output_actions"] || ""),
          notes: parsePipeOrCsv(row["notes"] || ""),
        });
      }
    }
  }

  if (
    result.symptomRows.length === 0 &&
    result.questionRows.length === 0 &&
    result.modifierRows.length === 0 &&
    result.algorithmRows.length === 0
  ) {
    result.issues.push("No recognizable legacy rows were mapped from workbook tabs.");
  }

  return result;
}

const baseUrl = process.env.APP_BASE_URL || "http://localhost:3001";

async function post(path: string) {
  const res = await fetch(`${baseUrl}${path}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
  });

  const json = await res.json();
  if (!res.ok) {
