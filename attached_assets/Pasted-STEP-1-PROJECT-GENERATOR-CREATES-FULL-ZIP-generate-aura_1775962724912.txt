const fs = require("fs");
const path = require("path");
const { execSync } = require("child_process");

const root = "auralyn-medical-brain";

function write(file, content) {
  const full = path.join(root, file);
  fs.mkdirSync(path.dirname(full), { recursive: true });
  fs.writeFileSync(full, content);
}

write("server/realtime/patientStream.ts", `
import { WebSocketServer } from "ws";

let clients:any[] = [];

export function initPatientStream(server:any) {
  const wss = new WebSocketServer({ server });

  wss.on("connection", ws => {
    clients.push(ws);

    ws.on("close", () => {
      clients = clients.filter(c => c !== ws);
    });
  });
}

export function broadcast(data:any) {
  clients.forEach(c => c.send(JSON.stringify(data)));
}
`);

write("server/agents/specialistCouncil.ts", `
export class SpecialistCouncil {
  async evaluate(p:any) {
    const votes = [
      this.cardio(p),
      this.id(p),
      this.icu(p)
    ];

    const final =
      votes.find(v => v.risk === "CRITICAL") ? "ICU" :
      votes.find(v => v.risk === "HIGH") ? "ED" :
      "OUTPATIENT";

    return { votes, final };
  }

  cardio(p:any) {
    return p.complaint?.includes("chest")
      ? { specialty:"cardio", risk:"HIGH" }
      : { specialty:"cardio", risk:"LOW" };
  }

  id(p:any) {
    return p.complaint?.includes("fever")
      ? { specialty:"ID", risk:"MEDIUM" }
      : { specialty:"ID", risk:"LOW" };
  }

  icu(p:any) {
    return p.redFlags?.length
      ? { specialty:"ICU", risk:"CRITICAL" }
      : { specialty:"ICU", risk:"LOW" };
  }
}
`);

write("server/learning/driftDetector.ts", `
export class DriftDetector {
  detect(a:number[], b:number[]) {
    let drift = 0;
    for (let i=0;i<a.length;i++) {
      drift += Math.abs(a[i]-b[i]);
    }
    return drift > 0.2;
  }
}
`);
