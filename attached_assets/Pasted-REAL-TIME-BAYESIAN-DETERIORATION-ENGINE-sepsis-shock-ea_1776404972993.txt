type Vitals = {
  hr: number;
  bp: number;
  spo2: number;
  temp: number;
};

type RiskState = {
  sepsisRisk: number;
  shockRisk: number;
};

export function computeDeteriorationRisk(v: Vitals): RiskState {
  let sepsis = 0;
  let shock = 0;

  // simplified Bayesian-style scoring
  if (v.temp > 38) sepsis += 0.2;
  if (v.hr > 100) sepsis += 0.2;
  if (v.bp < 90) {
    sepsis += 0.3;
    shock += 0.4;
  }
  if (v.spo2 < 92) shock += 0.3;

  return {
    sepsisRisk: Math.min(sepsis, 1),
    shockRisk: Math.min(shock, 1),
  };
}

];

  const sorted = hospitals.sort((a, b) => b.icuBeds - a.icuBeds);

  for (const h of sorted) {
    if (h.icuBeds > 0) {
      h.icuBeds -= 1;
      return h.id;
    }
  }

  return "NO_CAPACITY";

type Location = { lat: number; lon: number };

export function calculateDistance(a: Location, b: Location) {
  return Math.sqrt(
    Math.pow(a.lat - b.lat, 2) +
    Math.pow(a.lon - b.lon, 2)
  );
}

const EMS_UNITS = [
  { id: "EMS_1", location: { lat: 1, lon: 1 } },
  { id: "EMS_2", location: { lat: 5, lon: 5 } },
];

export function dispatchEMS(patientLocation: Location) {
  let best = EMS_UNITS[0];
  let bestDist = Infinity;

  for (const unit of EMS_UNITS) {
    const d = calculateDistance(unit.location, patientLocation);
    if (d < bestDist) {
      best = unit;
      bestDist = d;
    }
  }

  return best.id;
}

type Outcome = {
  diagnosis: string;
  correct: boolean;
};

export function updateModelWeights(model: any, outcomes: Outcome[]) {
  const updates: any = {};

  for (const o of outcomes) {
    if (!updates[o.diagnosis]) updates[o.diagnosis] = 0;

    updates[o.diagnosis] += o.correct ? 0.01 : -0.01;
  }

  for (const dx in updates) {
    const delta = Math.max(-0.02, Math.min(0.02, updates[dx]));
    model[dx] = (model[dx] || 0) + delta;
  }

  return model;
}

export function safeLearningLoop(model: any, outcomes: any[]) {
  const newModel = updateModelWeights(model, outcomes);

  const requiresApproval = requirePhysicianApproval(
    JSON.stringify(newModel).length
  );

  if (requiresApproval) {
    console.log("🧠 Awaiting physician approval");
    return model;
  }

  return newModel;
}
