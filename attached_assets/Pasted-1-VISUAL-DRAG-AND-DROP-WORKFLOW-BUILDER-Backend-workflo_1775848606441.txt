const steps: Record<string, StepFn> = {};

export function registerStep(name: string, fn: StepFn) {
  steps[name] = fn;
}

export function listSteps() {
  return Object.keys(steps);
}

export function getStep(name: string) {
  return steps[name];
}
server/workflows/runner.ts
import { getStep } from "./registry";

export async function runWorkflow(def: { steps: { name: string; config?: any }[] }, input: any) {
  let ctx = { ...input };
  for (const s of def.steps) {
    const fn = getStep(s.name);
    if (!fn) throw new Error(`Missing step: ${s.name}`);
    ctx = await fn({ ...ctx, config: s.config });
  }
  return ctx;
}

import { useState } from "react";

export default function WorkflowBuilder() {
  const [steps, setSteps] = useState<string[]>([]);

  function add(step: string) { setSteps([...steps, step]); }

  async function run() {
    await fetch("/api/workflows/run", {
      method: "POST",
      headers: {"Content-Type":"application/json"},
      body: JSON.stringify({ steps: steps.map(s => ({ name: s })) })
    });
    alert("Workflow executed");
  }

  return (
    <div className="p-6 bg-black text-white">
      <h2>🧩 Workflow Builder</h2>
      <div className="space-x-2">

  );
}

🌐 3) MULTI-REGION PRODUCTION (GATEWAY + FAILOVER)
🔧 Edge gateway (geo + failover)
const REGIONS = [
  { name: "us-east", url: process.env.REGION_US_EAST },
  { name: "us-west", url: process.env.REGION_US_WEST },
  { name: "eu", url: process.env.REGION_EU }
];

function pickByIP(ip: string) {
  if (ip.startsWith("172")) return REGIONS[0];
  if (ip.startsWith("10")) return REGIONS[1];
  return REGIONS[2];
}

export async function gatewayFetch(req: any, path: string) {
  const primary = pickByIP(req.ip);
  const order = [primary, ...REGIONS.filter(r => r !== primary)];

  for (const r of order) {
    try {
      const res = await fetch(r.url + path, {
        method: req.method,
        headers: {"Content-Type":"application/json"},
        body: JSON.stringify(req.body)
      });
      if (res.ok) return res.json();
    } catch {}
  }
  throw new Error("All regions failed");
}

export function requireModifiers(m: any) {
  const missing = [];
  if (!m.age) missing.push("age");
  if (!m.allergies) missing.push("allergies");
  if (!m.meds) missing.push("medications");
  return { ok: missing.length === 0, missing };
}
