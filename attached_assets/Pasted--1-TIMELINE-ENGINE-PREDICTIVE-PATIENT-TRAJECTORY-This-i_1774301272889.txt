type TimelineState = {

// server/timeline/predictor.ts

export function predictDeterioration(timeline: any[]) {
  const trend = timeline.slice(-3);

  const riskIncrease =
    trend[2].riskScore - trend[0].riskScore;

  if (riskIncrease > 0.3) {
    return {
      prediction: "HIGH_RISK_DETERIORATION",
      timeframe: "6-12h"
    };
  }

  return { prediction: "stable" };
}

// server/hospital/fullSystem.ts

export async function runHospitalSystem(input: any) {

  // 1. timeline
  const timeline = buildTimeline(input.history);
  const prediction = predictDeterioration(timeline);

  // 2. reasoning
  const decision = await clinicalReasoning(input);

  // 3. safety
  const safe = clinicalSafetyGate(decision);
  if (!safe.allowed) return { review: true };

  // 4. procedure
  if (decision.workflow) {
    await runProcedure(decision.workflow, input);
  }

  // 5. billing
  const billing = mapBilling(decision);

  // 6. learning
  trainLocal([input]);

  return {
    decision,
    prediction,
    billing
  };
}
