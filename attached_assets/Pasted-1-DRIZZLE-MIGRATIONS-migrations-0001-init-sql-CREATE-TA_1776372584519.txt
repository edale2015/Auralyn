import { diagnoses, evidenceMap, contradictionMap } from '../db/schema';

  console.log("Seeding database...");

  await db.insert(diagnoses).values([
    { id: "pe", name: "Pulmonary Embolism", prior: "0.05" },
    { id: "acs", name: "Acute Coronary Syndrome", prior: "0.05" },
    { id: "pneumonia", name: "Pneumonia", prior: "0.2" },
    { id: "sepsis", name: "Sepsis", prior: "0.03" },
    { id: "stroke", name: "Stroke", prior: "0.02" },
  ]);

  await db.insert(evidenceMap).values([
    { diagnosisId: "pe", features: ["sob", "pleuritic_chest_pain", "tachycardia"] },
    { diagnosisId: "acs", features: ["chest_pain", "radiation_left_arm", "diaphoresis"] },
    { diagnosisId: "pneumonia", features: ["fever", "cough", "sob"] },
  ]);

  await db.insert(contradictionMap).values([
    { diagnosisId: "pe", features: ["fever_high"] },
    { diagnosisId: "acs", features: ["pleuritic_chest_pain"] },
  ]);

  console.log("Seed complete.");
}

seedDatabase();
