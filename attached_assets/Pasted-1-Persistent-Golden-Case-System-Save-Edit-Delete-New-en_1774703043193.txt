// server/api/decisionTree.ts

router.get('/api/decision-tree/:complaint', async (req, res) => {
  const { complaint } = req.params;

  const questions = await getCoreQuestions(complaint);
  const rules = await getScoringRules(complaint);
  const dispositions = await getDispositions(complaint);

  const nodes = [];
  const edges = [];

  questions.forEach(q => {
    nodes.push({ id: q.id, label: q.question, type: 'question' });
  });

  rules.forEach(r => {
    nodes.push({ id: r.id, label: r.condition, type: 'rule' });
    edges.push({ from: r.questionId, to: r.id });
  });

  dispositions.forEach(d => {
    nodes.push({ id: d.id, label: d.action, type: 'disposition' });
  });

  res.json({ nodes, edges });
});
