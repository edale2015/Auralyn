import { ChatOpenAI } from "@langchain/openai";
import { RunnablePassthrough } from "@langchain/core/runnables";
import { ChatPromptTemplate } from "@langchain/core/prompts";
import { StringOutputParser } from "@langchain/core/output_parsers";
import { getClinicalRetriever } from "../../knowledge/retriever";

const model = new ChatOpenAI({
  modelName: "gpt-4o-mini",
  temperature: 0.2
});

export const buildClinicalRAG = () => {
  const retriever = getClinicalRetriever(); // your Postgres KB + embeddings

  const prompt = ChatPromptTemplate.fromTemplate(`
You are a clinical triage assistant.

Use ONLY this clinical knowledge:
{context}

Patient presentation:
{question}

Return:
- Top 3 diagnoses
- Risk level (low / moderate / high)
- Disposition (home / urgent care / ER)
- Reasoning (brief, evidence-based)
`);

  return (
    {
      context: retriever,
      question: new RunnablePassthrough()
    }
      .pipe(prompt)
      .pipe(model)
      .pipe(new StringOutputParser())
  );
};
🔁 3. LANGGRAPH: Iterative Triage Engine (Loop Until Safe)
/server/ai-orchestration/langgraph/triageGraph.ts
import { StateGraph, START, END } from "@langgraph/core";

type TriageState = {
  symptoms: string;
  riskScore: number;
  questionsAsked: string[];
  disposition?: string;
  iteration: number;
};

const MAX_ITERATIONS = 5;

const askQuestion = async (state: TriageState) => {
  return {
    questionsAsked: [...state.questionsAsked, "Ask red flag question"]
  };
};

const evaluateRisk = async (state: TriageState) => {
  const risk = Math.random() * 10; // replace with your Bayesian engine

  return {
    riskScore: risk,
    iteration: state.iteration + 1
  };
};

const decideDisposition = (state: TriageState) => {
  if (state.riskScore > 7) return "ER";
  if (state.riskScore > 4) return "UrgentCare";
  return "Home";
};

const shouldContinue = (state: TriageState) => {
  if (state.iteration > MAX_ITERATIONS) return END;
  if (state.riskScore < 4) return END;
  return "ask";
};

export const triageGraph = new StateGraph<TriageState>();

triageGraph.addNode("ask", askQuestion);
triageGraph.addNode("evaluate", evaluateRisk);

triageGraph.addEdge(START, "ask");
triageGraph.addEdge("ask", "evaluate");

triageGraph.addConditionalEdges("evaluate", shouldContinue);

triageGraph.addNode("final", (state) => ({
  disposition: decideDisposition(state)
}));

triageGraph.addEdge("evaluate", "final");

import { Agent, Task, Crew } from "crewai";

const cardiology = new Agent({
  role: "Cardiologist",
  goal: "Evaluate cardiac risk",
  backstory: "Expert in ACS, arrhythmias, CHF",
  verbose: true
});

const infectious = new Agent({
  role: "Infectious Disease Specialist",
  goal: "Evaluate infection risk",
  backstory: "Expert in sepsis, viral vs bacterial",
  verbose: true
});

const icu = new Agent({
  role: "ICU Physician",
  goal: "Assess critical deterioration risk",
  backstory: "Expert in shock, respiratory failure",
  verbose: true
});

export const runSpecialistCouncil = async (caseData: string) => {
  const tasks = [
    new Task({
      description: `Evaluate this patient:\n${caseData}`,
      expected_output: "Risk assessment + disposition",
      agent: cardiology
    }),
    new Task({
      description: `Evaluate infection risk:\n${caseData}`,
      expected_output: "Infection likelihood",
      agent: infectious
    }),
    new Task({
      description: `Assess ICU need:\n${caseData}`,
      expected_output: "Critical risk",
      agent: icu
    })
  ];

  const crew = new Crew({
    agents: [cardiology, infectious, icu],
    tasks,
    process: "sequential"
  });

  return await crew.kickoff();
};

🔄 5. n8n-STYLE WORKFLOW ENGINE (INTERNAL)
/server/ai-orchestration/events/workflowEngine.ts
type WorkflowStep = (input: any) => Promise<any>;

export class Workflow {
  steps: WorkflowStep[] = [];

  add(step: WorkflowStep) {
    this.steps.push(step);
    return this;
  }

  async run(input: any) {
    let data = input;

    for (const step of this.steps) {
      try {
        data = await step(data);
      } catch (err) {
        console.error("Workflow error:", err);
        throw err;
      }
    }

    return data;
  }
}
