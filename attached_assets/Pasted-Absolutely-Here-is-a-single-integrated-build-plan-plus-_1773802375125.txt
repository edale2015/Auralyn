import { OutcomeRecord, StructuredCase } from "../../shared/types";

const cases = new Map<string, StructuredCase>();
const outcomes: OutcomeRecord[] = [];

export const caseStore = {
  saveCase(input: StructuredCase) {
    cases.set(input.id, input);
    return input;
  },

  getCase(id: string) {
    return cases.get(id) || null;
  },

  listCases() {
    return Array.from(cases.values());
  },

  updateCase(id: string, patch: Partial<StructuredCase>) {
    const current = cases.get(id);
    if (!current) return null;
    const next = {
      ...current,
      ...patch,
      updatedAt: new Date().toISOString(),
    };
    cases.set(id, next);
    return next;
  },

  saveOutcome(record: OutcomeRecord) {
    outcomes.push(record);
    return record;
  },

  listOutcomes() {
    return outcomes;
  },
};

4. Smart intake engine

This is the intake brain that turns messy patient text into structured fields and asks only the next best question.

// server/engines/smartIntakeEngine.ts

import { PatientMessage, StructuredCase, SymptomAnswer } from "../../shared/types";

type IntakeDraft = Partial<StructuredCase>;

const RED_FLAG_PATTERNS = [
  { pattern: /chest pain|pressure in chest/i, flag: "chest_pain" },
  { pattern: /shortness of breath|can't breathe|trouble breathing/i, flag: "sob" },
  { pattern: /fainting|passed out|syncope/i, flag: "syncope" },
  { pattern: /one sided weakness|slurred speech/i, flag: "stroke_symptoms" },
  { pattern: /severe abdominal pain/i, flag: "severe_abdominal_pain" },
  { pattern: /pregnant and bleeding/i, flag: "pregnancy_bleeding" },
];

export function parseSmartIntake(
  patientId: string,
  source: "sms" | "whatsapp" | "web",
  transcript: PatientMessage[]
): IntakeDraft {
  const allText = transcript.map(t => t.text).join(" ");
  const chiefComplaint = extractChiefComplaint(allText);
  const age = extractAge(allText);
  const sex = extractSex(allText);
  const symptomDuration = extractDuration(allText);
  const redFlags = extractRedFlags(allText);

  const answers: SymptomAnswer[] = [];
  if (age != null) answers.push({ key: "age", value: age });
  if (sex) answers.push({ key: "sex", value: sex });
  if (symptomDuration) answers.push({ key: "duration", value: symptomDuration });

  const missingCriticalData = [];
  if (age == null) missingCriticalData.push("age");
  if (!chiefComplaint) missingCriticalData.push("chief_complaint");

  return {
    patientId,
    source,
    chiefComplaint: chiefComplaint || "general_medical_question",
    age,
    sex,
    symptomDuration,
    answers,
    transcript,
    redFlags,
    missingCriticalData,
  };
}

function extractChiefComplaint(text: string): string {
  const complaintMap = [
    "cough",
    "sore throat",
    "fever",
    "rash",
    "urinary burning",
    "ear pain",
    "eye redness",
    "abdominal pain",
    "vomiting",
    "diarrhea",
    "back pain",
    "headache",
    "refill",
  ];
  const lower = text.toLowerCase();
  return complaintMap.find(c => lower.includes(c)) || inferComplaint(lower);
}

function inferComplaint(text: string): string {
  if (/pee burns|burning when urinate|uti/i.test(text)) return "urinary burning";
  if (/runny nose|congestion|cold/i.test(text)) return "cough";
  if (/itchy spots|red bumps/i.test(text)) return "rash";
  return "general_medical_question";
}

function extractAge(text: string): number | undefined {
  const match = text.match(/\b(\d{1,3})\s*(yo|year old|years old)\b/i);
  if (!match) return undefined;
  return Number(match[1]);
}

function extractSex(text: string): string | undefined {
  if (/\bfemale\b|\bwoman\b|\bgirl\b/i.test(text)) return "female";
  if (/\bmale\b|\bman\b|\bboy\b/i.test(text)) return "male";
  return undefined;
}

function extractDuration(text: string): string | undefined {
  const match = text.match(/\b(\d+)\s*(day|days|week|weeks|month|months)\b/i);
  return match ? `${match[1]} ${match[2]}` : undefined;
}

function extractRedFlags(text: string): string[] {
  return RED_FLAG_PATTERNS.filter(x => x.pattern.test(text)).map(x => x.flag);
}

export function getNextBestQuestion(draft: Partial<StructuredCase>): string | null {
  if (!draft.age) return "What is the patient's age?";
  if (!draft.chiefComplaint) return "What is the main symptom today?";
  if (draft.chiefComplaint === "cough" && !hasAnswer(draft, "sob")) {
    return "Any shortness of breath, chest pain, or fever?";
  }
  if (draft.chiefComplaint === "urinary burning" && !hasAnswer(draft, "pregnancy_status")) {
    return "Any fever, back pain, vomiting, or pregnancy?";
  }
  if (draft.chiefComplaint === "rash" && !hasAnswer(draft, "rash_red_flags")) {
    return "Any mouth sores, facial swelling, trouble breathing, or severe pain?";
  }
  return null;
}

function hasAnswer(draft: Partial<StructuredCase>, key: string) {
  return (draft.answers || []).some(a => a.key === key);
}
5. Risk scoring engine

This decides who vanishes into auto-resolution and who enters the doctor queue.

// server/engines/riskScoringEngine.ts

import { StructuredCase } from "../../shared/types";

export function scoreRisk(input: Partial<StructuredCase>) {
  let score = 0;

  if ((input.redFlags || []).length > 0) score += 70;
  if ((input.missingCriticalData || []).length > 0) score += 10;
  if ((input.age || 0) < 3) score += 15;
  if ((input.age || 0) > 75) score += 12;

  const cc = (input.chiefComplaint || "").toLowerCase();

  if (cc.includes("chest")) score += 60;
  if (cc.includes("abdominal")) score += 20;
  if (cc.includes("vomiting")) score += 15;
  if (cc.includes("urinary")) score += 10;
  if (cc.includes("rash")) score += 8;
  if (cc.includes("refill")) score -= 20;

  if ((input.symptomDuration || "").includes("month")) score += 10;

  const confidenceScore = calculateConfidence(input);

  if (confidenceScore < 0.6) score += 20;
  if (confidenceScore > 0.9) score -= 8;

  score = Math.max(0, Math.min(100, score));

  let riskLevel: "low" | "medium" | "high" | "critical" = "low";
  if (score >= 80) riskLevel = "critical";
  else if (score >= 55) riskLevel = "high";
  else if (score >= 25) riskLevel = "medium";

  return {
    riskScore: score,
    riskLevel,
    confidenceScore,
  };
}

function calculateConfidence(input: Partial<StructuredCase>) {
  let confidence = 0.5;
  if (input.chiefComplaint) confidence += 0.15;
  if (input.age) confidence += 0.05;
  if (input.symptomDuration) confidence += 0.05;
  if ((input.missingCriticalData || []).length === 0) confidence += 0.1;
  if ((input.redFlags || []).length === 0) confidence += 0.1;
  if ((input.answers || []).length >= 3) confidence += 0.1;
  return Math.max(0, Math.min(0.99, confidence));
}

// server/engines/queuePrioritizationEngine.ts

import { StructuredCase } from "../../shared/types";

export function computeQueuePriority(input: Partial<StructuredCase>): number {
  let score = 0;

  score += (input.riskScore || 0) * 5;

  if (input.reviewReason === "red_flags_detected") score += 500;
  if (input.reviewReason === "medication_review_required") score += 150;
  if (input.reviewReason === "unsupported_complaint_pathway") score += 120;

  const age = input.age || 0;
  if (age < 3 || age > 75) score += 40;

  if (input.proposedDisposition === "er_now") score += 1000;
  if (input.proposedDisposition === "urgent_care") score += 300;
  if (input.proposedDisposition === "telemed_now") score += 100;

  return score;
}

export function sortQueue<T extends Partial<StructuredCase>>(cases: T[]): T[] {
  return [...cases].sort((a, b) => (b.queuePriority || 0) - (a.queuePriority || 0));
}
8. Batch approval engine

This is the physician speed booster rocket.

// server/engines/batchApprovalEngine.ts

import { BatchApprovalRequest } from "../../shared/types";
import { caseStore } from "../services/caseStore";
import { auditLog } from "../services/auditLog";

export function executeBatchApproval(req: BatchApprovalRequest) {
  const results = [];

  for (const caseId of req.caseIds) {
    const current = caseStore.getCase(caseId);
    if (!current) continue;

    let queueStatus = current.queueStatus;
    let proposedDisposition = current.proposedDisposition;
    let overrideNotes = current.overrideNotes;

    if (req.action === "approve") {
      queueStatus = "approved";
    } else if (req.action === "escalate") {
      queueStatus = "escalated";
      proposedDisposition = req.overrideDisposition || "telemed_now";
      overrideNotes = req.overridePlanNote || "Escalated during batch review";
    } else if (req.action === "request_review") {
      queueStatus = "needs_review";
      overrideNotes = req.overridePlanNote || "Additional review requested";
    }

    const updated = caseStore.updateCase(caseId, {
      queueStatus,
      proposedDisposition,
      overrideNotes,
      approvedBy: req.physicianId,
      approvedAt: new Date().toISOString(),
    });

    auditLog.write({
      actor: req.physicianId,
      entityId: caseId,
      event: `batch_${req.action}`,
      details: {
        priorStatus: current.queueStatus,
        nextStatus: queueStatus,
        overrideDisposition: req.overrideDisposition || null,
        overridePlanNote: req.overridePlanNote || null,
      },
    });

    results.push(updated);
  }

  return results;
}

import { OutcomeRecord } from "../../shared/types";
import { caseStore } from "../services/caseStore";
import { auditLog } from "../services/auditLog";

export function recordOutcome(input: OutcomeRecord) {
  const saved = caseStore.saveOutcome(input);

  const linkedCase = caseStore.getCase(input.caseId);

  auditLog.write({
    actor: "system_outcome_collector",
    entityId: input.caseId,
    event: "outcome_recorded",
    details: {
      outcomeType: input.outcomeType,
      npsScore: input.npsScore ?? null,
      patientComment: input.patientComment ?? null,
      complaint: linkedCase?.chiefComplaint ?? null,
      disposition: linkedCase?.proposedDisposition ?? null,
    },
  });

  return saved;
}

export function buildOutcomeAnalytics() {
  const outcomes = caseStore.listOutcomes();

  const total = outcomes.length || 1;
  const avgNps =
    outcomes.filter(x => typeof x.npsScore === "number")
      .reduce((sum, x) => sum + (x.npsScore || 0), 0) /
    Math.max(1, outcomes.filter(x => typeof x.npsScore === "number").length);

  const byType = outcomes.reduce<Record<string, number>>((acc, item) => {
    acc[item.outcomeType] = (acc[item.outcomeType] || 0) + 1;
    return acc;
  }, {});

  return {
    totalOutcomes: outcomes.length,
    averageNps: Number(avgNps.toFixed(2)),
    byType,
    worseningRate: Number((((byType["worsened"] || 0) / total) * 100).toFixed(2)),
    erVisitRate: Number((((byType["er_visit"] || 0) / total) * 100).toFixed(2)),
  };
}

// server/routes/intake.ts

import express from "express";
import crypto from "crypto";
import { parseSmartIntake } from "../engines/smartIntakeEngine";
import { scoreRisk } from "../engines/riskScoringEngine";
import { generateAutoPlan } from "../engines/autoPlanEngine";
import { computeQueuePriority } from "../engines/queuePrioritizationEngine";
import { caseStore } from "../services/caseStore";
import { auditLog } from "../services/auditLog";
import { StructuredCase } from "../../shared/types";

const router = express.Router();

router.post("/webhook/twilio", async (req, res) => {
  const messageBody = req.body.Body || "";
  const from = req.body.From || "unknown";
  const source = inferSource(req.body);

  const transcript = [
    {
      role: "patient" as const,
      text: messageBody,
      at: new Date().toISOString(),
    },
  ];

  const intake = parseSmartIntake(from, source, transcript);
  const risk = scoreRisk(intake);
  const plan = generateAutoPlan({ ...intake, ...risk });
  const queuePriority = computeQueuePriority({ ...intake, ...risk, ...plan });

  const newCase: StructuredCase = {
    id: crypto.randomUUID(),
    patientId: from,
    source,
    chiefComplaint: intake.chiefComplaint || "general_medical_question",
    age: intake.age,
    sex: intake.sex,
    symptomDuration: intake.symptomDuration,
    answers: intake.answers || [],
    transcript,
    redFlags: intake.redFlags || [],
    missingCriticalData: intake.missingCriticalData || [],
    riskScore: risk.riskScore,
    riskLevel: risk.riskLevel,
    confidenceScore: risk.confidenceScore,
    differential: plan.differential,
    proposedDisposition: plan.proposedDisposition,
    proposedPlan: plan.proposedPlan,
    reviewReason: plan.reviewReason,
    queuePriority,
    queueStatus:
      risk.riskLevel === "low" &&
      risk.confidenceScore >= 0.85 &&
      !plan.reviewReason &&
      plan.proposedPlan
        ? "auto_resolved"
        : "needs_review",
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
  };

  caseStore.saveCase(newCase);

  auditLog.write({
    actor: "smart_intake_engine",
    entityId: newCase.id,
    event: "case_created",
    details: {
      source,
      riskLevel: newCase.riskLevel,
      confidenceScore: newCase.confidenceScore,
      queueStatus: newCase.queueStatus,
      reviewReason: newCase.reviewReason || null,
    },
  });

  if (newCase.queueStatus === "auto_resolved" && newCase.proposedPlan) {
    return res.type("text/xml").send(twimlReply(newCase.proposedPlan.patientMessage));
  }

  return res.type("text/xml").send(
    twimlReply(
      "Thanks. Your case has been received and is being reviewed by a clinician. We'll message you shortly."
    )
  );
});

function inferSource(body: any): "sms" | "whatsapp" | "web" {
  const from = String(body.From || "");
  return from.startsWith("whatsapp:") ? "whatsapp" : "sms";
}

function twimlReply(message: string) {
  return `<?xml version="1.0" encoding="UTF-8"?>
<Response>
  <Message>${escapeXml(message)}</Message>
</Response>`;
}

function escapeXml(input: string) {
  return input
    .replace(/&/g, "&amp;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&apos;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;");
}

export default router;

// frontend/src/api/reviewApi.ts

export async function fetchReviewQueue() {
  const res = await fetch("/api/queue/review-queue");
  if (!res.ok) throw new Error("Failed to load queue");
  return res.json();
}

export async function fetchCase(id: string) {
  const res = await fetch(`/api/queue/review-queue/${id}`);
  if (!res.ok) throw new Error("Failed to load case");
  return res.json();
}

export async function batchApprove(payload: {
  caseIds: string[];
  physicianId: string;
  action: "approve" | "escalate" | "request_review";
  overrideDisposition?: string;
  overridePlanNote?: string;
}) {
  const res = await fetch("/api/review/batch-approve", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(payload),
  });
  if (!res.ok) throw new Error("Batch approval failed");
  return res.json();
}

import React from "react";

export function CaseDrawer({ item }: { item: any | null }) {
  if (!item) return <div style={{ padding: 16 }}>Select a case</div>;

  return (
    <div style={{ padding: 16, borderLeft: "1px solid #ddd", height: "100%", overflow: "auto" }}>
      <h3>{item.chiefComplaint}</h3>
      <p><strong>Age:</strong> {item.age ?? "-"}</p>
      <p><strong>Risk:</strong> {item.riskLevel} ({item.riskScore})</p>
      <p><strong>Disposition:</strong> {item.proposedDisposition}</p>
      <p><strong>Reason:</strong> {item.reviewReason || "auto-safe pattern"}</p>

      <h4>Differential</h4>
      <ul>
        {(item.differential || []).map((d: any) => (
          <li key={d.diagnosis}>
            {d.diagnosis}: {Math.round(d.probability * 100)}%
          </li>
        ))}
      </ul>

      <h4>Red flags</h4>
      <ul>
        {(item.redFlags || []).map((f: string) => <li key={f}>{f}</li>)}
      </ul>

      <h4>Plan</h4>
      {item.proposedPlan ? (
        <>
          <p>{item.proposedPlan.summary}</p>
          <h5>Meds</h5>
          <ul>
            {(item.proposedPlan.meds || []).map((m: any) => (
              <li key={m.name}>
                {m.name} {m.dose} - {m.instructions}
              </li>
            ))}
          </ul>
          <h5>Home care</h5>
          <ul>
            {(item.proposedPlan.homeCare || []).map((x: string) => <li key={x}>{x}</li>)}
          </ul>
          <h5>Return precautions</h5>
          <ul>
            {(item.proposedPlan.returnPrecautions || []).map((x: string) => <li key={x}>{x}</li>)}
          </ul>
        </>
      ) : (
        <p>No auto-plan available.</p>
      )}
    </div>
  );
}
18. Batch review page
// frontend/src/pages/BatchReviewPage.tsx

import React, { useEffect, useState } from "react";
import { fetchCase, fetchReviewQueue, batchApprove } from "../api/reviewApi";
import { ReviewQueueTable } from "../components/ReviewQueueTable";
import { BatchToolbar } from "../components/BatchToolbar";
import { CaseDrawer } from "../components/CaseDrawer";

export default function BatchReviewPage() {
  const [rows, setRows] = useState<any[]>([]);
  const [selected, setSelected] = useState<string[]>([]);
  const [activeCase, setActiveCase] = useState<any | null>(null);

  async function load() {
    const data = await fetchReviewQueue();
    setRows(data);
  }

  useEffect(() => {
    load();
  }, []);

  function toggleSelected(id: string) {
    setSelected(prev =>
      prev.includes(id) ? prev.filter(x => x !== id) : [...prev, id]
    );
  }

  async function openCase(id: string) {
    const data = await fetchCase(id);
    setActiveCase(data);
  }

  async function handleApprove() {
    if (!selected.length) return;
    await batchApprove({
      caseIds: selected,
      physicianId: "doctor_1",
      action: "approve",
    });
    setSelected([]);
    setActiveCase(null);
    await load();
  }

  async function handleEscalate() {
    if (!selected.length) return;
    await batchApprove({
      caseIds: selected,
      physicianId: "doctor_1",
      action: "escalate",
      overrideDisposition: "telemed_now",
      overridePlanNote: "Escalated from batch queue",
    });
    setSelected([]);
    setActiveCase(null);
    await load();
  }

  return (
    <div style={{ display: "grid", gridTemplateColumns: "2fr 1fr", height: "100vh" }}>
      <div style={{ padding: 16 }}>
        <h2>Physician Batch Review</h2>
        <BatchToolbar
          selectedCount={selected.length}
          onApprove={handleApprove}
          onEscalate={handleEscalate}
          onRefresh={load}
        />
        <ReviewQueueTable
          rows={rows}
          selected={selected}
          toggleSelected={toggleSelected}
          openCase={openCase}
        />
      </div>
      <CaseDrawer item={activeCase} />
    </div>
  );
}

22. One-click “approve all safe” endpoint
// add to server/routes/review.ts

import { caseStore } from "../services/caseStore";
import { executeBatchApproval } from "../engines/batchApprovalEngine";

router.post("/approve-all-safe", (req, res) => {
  const physicianId = req.body.physicianId || "doctor_1";

  const safeIds = caseStore
    .listCases()
    .filter(c =>
      c.queueStatus === "needs_review" &&
      c.riskLevel === "low" &&
      c.confidenceScore >= 0.85 &&
      c.redFlags.length === 0 &&
      c.proposedPlan &&
      c.reviewReason !== "red_flags_detected"
    )
    .map(c => c.id);

  const results = executeBatchApproval({
    caseIds: safeIds,
    physicianId,
    action: "approve",
  });

  res.json({
    approvedCount: results.length,
    results,
  });
});
