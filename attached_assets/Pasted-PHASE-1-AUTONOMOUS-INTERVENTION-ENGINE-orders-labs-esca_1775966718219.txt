type Intervention = {
  type: "lab" | "med" | "escalation";
  action: string;
  priority: "low" | "medium" | "high";
};

export function generateInterventions(patient: any): Intervention[] {
  const interventions: Intervention[] = [];

  const { hr, spo2, temp, bp } = patient.vitals;
  const systolic = parseInt(bp.split("/")[0]);

  // 🚨 SEPSIS / SHOCK
  if (temp > 100.4 && hr > 110 && systolic < 100) {
    interventions.push({
      type: "lab",
      action: "Order lactate, blood cultures x2",
      priority: "high"
    });

    interventions.push({
      type: "med",
      action: "Start IV fluids (30cc/kg)",
      priority: "high"
    });

    interventions.push({
      type: "escalation",
      action: "Alert physician + consider ICU transfer",
      priority: "high"
    });
  }

  // 🫁 HYPOXIA
  if (spo2 < 92) {
    interventions.push({
      type: "med",
      action: "Start oxygen therapy",
      priority: "high"
    });
  }

  // ❤️ TACHYCARDIA
  if (hr > 120) {
    interventions.push({
      type: "lab",
      action: "Check CBC, CMP, troponin",
      priority: "medium"
    });
  }

  return interventions;
}

type Outcome = {
  patientId: string;
  prediction: string;
  actualOutcome: string;
};

let weights: any = {
  hr: 1,
  spo2: 1,
  temp: 1,
  bp: 1
};

export function updateWeights(outcome: Outcome) {
  const correct = outcome.prediction === outcome.actualOutcome;

  if (!correct) {
    // adjust weights slightly
    weights.hr += 0.05;
    weights.spo2 += 0.05;
  }

  return weights;
}

export function getWeights() {
  return weights;
}

import os
import zipfile

PROJECT_ROOT = "auralyn_full_stack"
ZIP_NAME = "auralyn_FINAL_production.zip"

def zipdir(path, ziph):
    for root, dirs, files in os.walk(path):
        for file in files:
            full = os.path.join(root, file)
            ziph.write(full, os.path.relpath(full, path))

with zipfile.ZipFile(ZIP_NAME, 'w', zipfile.ZIP_DEFLATED) as zipf:
    zipdir(PROJECT_ROOT, zipf)

print("🚀 FINAL SYSTEM EXPORTED:", ZIP_NAME)
