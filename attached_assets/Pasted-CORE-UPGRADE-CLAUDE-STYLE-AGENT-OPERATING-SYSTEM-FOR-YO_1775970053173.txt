// server/skills/sepsisDetection.ts

export function detectSepsis(patient) {
  const { temp, hr, rr, sbp, mentalStatus } = patient.vitals;

  let qsofa = 0;

  if (sbp < 100) qsofa++;
  if (rr > 22) qsofa++;
  if (mentalStatus !== "normal") qsofa++;

  const risk = qsofa >= 2 ? "HIGH" : "LOW";

  return {
    qsofa,
    risk,
    action: risk === "HIGH"
      ? "ESCALATE_IMMEDIATELY"
      : "MONITOR"
  };
}

export class ICUAgent {
  evaluate(patient) {
    if (
      patient.vitals.sbp < 90 ||
      patient.vitals.o2 < 92 ||
      patient.mentalStatus !== "normal"
    ) {
      return {
        recommendation: "ICU_ADMISSION",
        confidence: 0.95,
        reason: "Hemodynamic instability"
      };
    }

    return null;
  }
}
🧠 Multi-Agent Voting Engine
// server/agents/multiAgentCouncil.ts

import { ICUAgent } from "./icuAgent";
import { CardiologyAgent } from "./cardiologyAgent";
import { IDAgent } from "./idAgent";

export function runAgentCouncil(patient) {
  const agents = [
    new ICUAgent(),
    new CardiologyAgent(),
    new IDAgent()
  ];

  const decisions = agents
    .map(agent => agent.evaluate(patient))
    .filter(Boolean);

  return decisions.sort((a, b) => b.confidence - a.confidence)[0];
}
