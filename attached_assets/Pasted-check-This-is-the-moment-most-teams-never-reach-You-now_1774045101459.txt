export function estimateImpact({ aiGroup, controlGroup }) {
  const aiRecovery =
    aiGroup.filter(p => p.recovered).length / aiGroup.length;

  const controlRecovery =
    controlGroup.filter(p => p.recovered).length / controlGroup.length;

  return {
    uplift: aiRecovery - controlRecovery
  };
}
