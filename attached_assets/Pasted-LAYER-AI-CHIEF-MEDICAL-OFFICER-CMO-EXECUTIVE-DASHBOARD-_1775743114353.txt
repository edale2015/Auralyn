import { auditStep } from "../audit/auditLogger";

export interface ExecutiveInput {
  metrics: {
    totalPatients: number;
    erRate: number;
    safetyMismatchRate: number;
    avgLatency: number;
    topComplaints: string[];
    drift: number;
  };
  alerts: string[];
  clusters: any[];
  timestamp: number;
}

export interface ExecutiveReport {
  summary: string;
  risks: string[];
  recommendations: string[];
  priorities: string[];
}

export async function generateExecutiveReport(
  input: ExecutiveInput,
  traceId: string
): Promise<ExecutiveReport> {

  const risks: string[] = [];
  const recommendations: string[] = [];
  const priorities: string[] = [];

  // ─────────────────────────────────────────────
  // Risk Detection
  // ─────────────────────────────────────────────
  if (input.metrics.safetyMismatchRate > 0) {
    risks.push("Safety mismatches detected — potential missed critical cases");
    priorities.push("Immediate safety audit");
  }

  if (input.metrics.avgLatency > 3000) {
    risks.push("High latency may delay urgent decisions");
    recommendations.push("Reduce cognitive budget or optimize NLP stage");
  }

  if (input.metrics.drift > 0.05) {
    risks.push("Clinical drift detected — system behavior diverging");
    priorities.push("Run meta-learning cycle");
  }

  // ─────────────────────────────────────────────
  // Recommendations
  // ─────────────────────────────────────────────
  if (input.metrics.erRate > 0.25) {
    recommendations.push("Increase triage sensitivity thresholds");
  }

  if (input.clusters.length > 0) {
    recommendations.push(`Investigate failure cluster: ${input.clusters[0][0]}`);
  }

  const summary =
    `System processed ${input.metrics.totalPatients} patients. ` +
    `ER rate ${input.metrics.erRate}. ` +
    `Latency ${input.metrics.avgLatency}ms. ` +
    `Drift ${input.metrics.drift}.`;

  const report: ExecutiveReport = {
    summary,
    risks,
    recommendations,
    priorities
  };

  await auditStep({
    traceId,
    step: "executive_report_generated",
    input,
    output: report,
    metadata: {}
  });

  return report;
}

import { useEffect, useState } from "react";

export default function ExecutiveDashboard() {
  const [report, setReport] = useState<any>(null);

  useEffect(() => {
    fetch("/api/executive/report", {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        input: window.__EXEC_METRICS__,
        traceId: "exec-dashboard"
      })
    })
      .then(res => res.json())
      .then(setReport);
  }, []);

  if (!report) return <div>Loading Executive Insights...</div>;

  return (
    <div className="p-6 bg-gray-900 text-white min-h-screen">

      <h1 className="text-2xl font-bold mb-4">🧠 Executive Dashboard</h1>

      <div className="bg-gray-800 p-4 rounded-xl mb-4">
        <h2 className="font-semibold">Summary</h2>
        <p>{report.summary}</p>
      </div>

      <div className="bg-red-900 p-4 rounded-xl mb-4">
        <h2 className="font-semibold">Risks</h2>
        {report.risks.map((r: string, i: number) => (
          <p key={i}>⚠️ {r}</p>
        ))}
      </div>

      <div className="bg-yellow-800 p-4 rounded-xl mb-4">
        <h2 className="font-semibold">Recommendations</h2>
        {report.recommendations.map((r: string, i: number) => (
          <p key={i}>➡️ {r}</p>
        ))}
      </div>

      <div className="bg-blue-800 p-4 rounded-xl">
        <h2 className="font-semibold">Priorities</h2>
        {report.priorities.map((p: string, i: number) => (
          <p key={i}>🎯 {p}</p>
        ))}
      </div>

    </div>
  );
}
