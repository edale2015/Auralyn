🔧 Launch endpoint
app.get("/smart/launch", (req, res) => {
  const iss = req.query.iss as string;
  const launch = req.query.launch as string;

  const url = new URL(`${iss}/authorize`);
  url.searchParams.set("response_type", "code");
  url.searchParams.set("client_id", process.env.SMART_CLIENT_ID!);
  url.searchParams.set("redirect_uri", process.env.SMART_REDIRECT!);
  url.searchParams.set("scope", "launch openid profile user/*.read");
  url.searchParams.set("launch", launch);

  res.redirect(url.toString());
});
🧠 3) LIGHTGBM TRAINING NOTEBOOK (SCHEMA + TRAIN)
ml/train.py
import pandas as pd
import lightgbm as lgb
import joblib

df = pd.read_csv("training_data.csv")

X = df[["age", "sbp", "spo2", "hr", "rr", "chestPain"]]
y = df["admitted"]

model = lgb.LGBMClassifier()
model.fit(X, y)

joblib.dump(model, "model.pkl")

print("Model trained and saved.")
