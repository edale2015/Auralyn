export function predictDeterioration(patient: any) {
  const { hr, spo2, temp, bp } = patient.vitals;

  let score = 0;

  // Early warning scoring (simplified NEWS2-style logic)
  if (hr > 110) score += 2;
  if (hr > 130) score += 3;

  if (spo2 < 94) score += 2;
  if (spo2 < 90) score += 3;

  if (temp > 100.4 || temp < 96) score += 2;

  const systolic = parseInt(bp.split("/")[0]);
  if (systolic < 100) score += 2;
  if (systolic < 90) score += 3;

  // Prediction tiers
  let risk = "low";
  if (score >= 6) risk = "high";
  else if (score >= 3) risk = "medium";

  return {
    score,
    risk,
    prediction:
      risk === "high"
        ? "Possible sepsis / shock developing"
        : risk === "medium"
        ? "Monitor closely for deterioration"
        : "Stable"
  };
}

import axios from "axios";

const FHIR_BASE = "https://api.hospital-fhir.com";

export async function fetchFHIRVitals(patientId: string) {
  const res = await axios.get(
    `${FHIR_BASE}/Observation?patient=${patientId}`
  );

  return res.data.entry.map((e: any) => ({
    type: e.resource.code.text,
    value: e.resource.valueQuantity?.value
  }));
}

📁 /backend/engines/triageEngine.ts
export function rankPatients(patients: any[]) {
  return patients
    .map(p => {
      const riskScore =
        (p.deterioration?.score || 0) * 2 +
        (p.status === "critical" ? 5 : 0);

      return {
        ...p,
        priorityScore: riskScore
      };
    })
    .sort((a, b) => b.priorityScore - a.priorityScore);
}

import os
import zipfile

PROJECT_NAME = "auralyn_production_system"
ZIP_NAME = "auralyn_production_system.zip"

def zipdir(path, ziph):
    for root, dirs, files in os.walk(path):
        for file in files:
            full_path = os.path.join(root, file)
            ziph.write(full_path, os.path.relpath(full_path, path))

with zipfile.ZipFile(ZIP_NAME, 'w', zipfile.ZIP_DEFLATED) as zipf:
    zipdir(PROJECT_NAME, zipf)

print("✅ FULL PRODUCTION ZIP CREATED:", ZIP_NAME)
