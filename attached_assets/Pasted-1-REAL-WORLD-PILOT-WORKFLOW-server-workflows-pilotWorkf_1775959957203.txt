import { runClinicalWorkflow } from "./clinicalWorkflowEngine";
import { generateCPT } from "../billing/cptEngine";
import { submitEncounter } from "../ehr/ehrOrchestrator";

export async function runPilotEncounter(input: any) {
  // Step 1: Clinical decision
  const clinical = await runClinicalWorkflow(input);

  // Step 2: Physician review REQUIRED
  if (!clinical.documented) {
    return {
      status: "pending_physician_review",
      clinical,
    };
  }

  // Step 3: Billing
  const billing = generateCPT(clinical);

  // Step 4: EHR submission
  await submitEncounter({
    ...clinical,
    billing,
  });

  return {
    status: "complete",
    clinical,
    billing,
  };
}

import { runEHRAction } from "../automation/ehr/playwrightEHR";

export async function submitEncounter(data: any) {
  // Decide system dynamically
  const system = process.env.EHR_SYSTEM || "athena";

  if (system === "athena") {
    return runEHRAction("athena_submit", data);
  }

  if (system === "epic") {
    return runEHRAction("epic_submit", data);
  }

  if (system === "ecw") {
    return runEHRAction("ecw_submit", data);
  }

  throw new Error("Unsupported EHR");
}
📁 /server/automation/ehr/playwrightEHR.ts (UPGRADE)
import { chromium } from "playwright";

export async function runEHRAction(action: string, payload: any) {
  const browser = await chromium.launch({ headless: false });
  const page = await browser.newPage();

  if (action === "athena_submit") {
    await page.goto("https://athenahealth.com");

    await page.fill("#diagnosis", payload.diagnosis);
    await page.fill("#plan", payload.disposition);
    await page.fill("#cpt", payload.billing.code);

    await page.click("#submit");
  }

  return { success: true };
}

📁 /server/billing/cptEngine.ts
export function generateCPT(clinical: any) {
  let code = "99213"; // default

  if (clinical.riskLevel === "high") code = "99214";
  if (clinical.riskLevel === "critical") code = "99285";

  return {
    code,
    justification: `Risk: ${clinical.riskLevel}, Diagnosis: ${clinical.diagnosis}`,
  };
}

📁 /server/services/payerContractService.ts
export class PayerContractService {
  simulateContract(volume: number) {
    const baseRate = 100;
    const bonus = volume > 1000 ? 20 : 0;

    return {
      ratePerVisit: baseRate + bonus,
      annualRevenue: (baseRate + bonus) * volume,
    };
  }

  suggestNegotiation(data: any) {
    if (data.avoidedEDVisits > 100) {
      return "Request higher reimbursement based on ED diversion savings";
    }

    return "Standard contract acceptable";
  }
}

export const payerContractService = new PayerContractService();

import { useState } from "react";

export default function PilotDashboard() {
  const [encounter, setEncounter] = useState<any>(null);
  const [contract, setContract] = useState<any>(null);

  const runEncounter = async () => {
    const res = await fetch("/api/workflow/run", {

        complaint: "cough",
      }),
      headers: { "Content-Type": "application/json" },

    setEncounter(await res.json());
  };

  const simulateContract = async () => {
    const res = await fetch("/api/payer/simulate?volume=2000");

  };

  return (
    <div style={{ padding: 20 }}>
      <h1>🏥 Pilot Dashboard</h1>

      <button onClick={runEncounter}>Run Encounter</button>
      <button onClick={simulateContract}>Simulate Contract</button>

      <h2>Encounter</h2>
      <pre>{JSON.stringify(encounter, null, 2)}</pre>

      <h2>Contract</h2>
      <pre>{JSON.stringify(contract, null, 2)}</pre>
    </div>
  );
}
