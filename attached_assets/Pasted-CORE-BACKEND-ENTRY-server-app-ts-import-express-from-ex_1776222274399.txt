🔐 SAFE RULE ENGINE (NO RCE)
server/monitoring/safeRuleEngine.ts
export type Operator = ">" | "<" | ">=" | "<=";

export interface SafeRule {
  metric: string;
  operator: Operator;
  value: number;
}

export function evaluateRule(rule: SafeRule, metrics: Record<string, number>) {
  const val = metrics[rule.metric];
  if (val === undefined) return false;

  switch (rule.operator) {
    case ">": return val > rule.value;
    case "<": return val < rule.value;
    case ">=": return val >= rule.value;
    case "<=": return val <= rule.value;
  }
}
🚨 ALERT ENGINE (FIXED + HEARTBEAT)
server/monitoring/alertEngine.ts
import { detectAnomaly } from "./anomalyDetector";
import { emitAlert } from "./alertBus";

export async function handleAnomalies() {
  const { anomalies, metrics, severity } = detectAnomaly();

  if (anomalies.length === 0) {
    emitAlert("System recovered", "info");
    return;
  }

  emitAlert(
    `Anomalies: ${anomalies.join(", ")} | latency=${metrics.avgLatency}`,
    severity === "CRITICAL" ? "critical" : "warn"
  );
}

export function startAlertEngine() {
  setInterval(handleAnomalies, 10000).unref();
}

⚖️ GOVERNED IMPROVEMENT ENGINE
server/agents/selfImprove.ts
import { db } from "../db";
import { auditStep } from "../audit/auditLogger";

export async function applyImprovement(actionId: number, reviewer: string) {
  return db.transaction(async (tx) => {

    const action = await tx.select()
      .from("improvements")
      .where({ id: actionId })
      .limit(1);

    if (!action.length) throw new Error("Not found");

    await tx.update("improvements")
      .set({ status: "applied", reviewedBy: reviewer })
      .where({ id: actionId });

    await auditStep({
      traceId: `improvement-${actionId}`,
      step: "applied",
      input: action[0],
      output: null
    });

    return action[0];
  });
}

export function startSelfHealingLoop() {
  setInterval(async () => {
    const agents = ["triage", "diagnosis", "routing"];

    for (const agent of agents) {
      try {
        await safeAdjustWeight(agent, Math.random() - 0.5);

      }
    }

📊 DATA DRIFT (FIXED)
server/monitoring/dataDrift.ts
let baseline: any = null;

export function updateBaseline(samples: any[]) {
  baseline = samples.slice(-100);
}

export function detectDrift(samples: any[]) {
  if (!baseline) return false;

  const diff = Math.abs(samples.length - baseline.length);
  return diff > 50;
}

import crypto from "crypto";

let lastHash = "";

export async function auditStep(entry: any) {
  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(entry) + lastHash)
    .digest("hex");

  lastHash = hash;

  console.log("AUDIT:", { ...entry, hash });
}
