// server/autonomy/autonomyController.ts

export function enforceAutonomyLevel(actions: any[], level: number) {
  if (level === 0) return [];

  if (level === 1) return actions.map(a => ({ ...a, mode: "SUGGEST" }));

  if (level === 2) return actions.map(a => ({ ...a, mode: "REQUIRES_APPROVAL" }));

  if (level === 3) {
    return actions.map(a => ({
      ...a,
      mode: a.risk === "LOW" ? "AUTO_EXECUTE" : "REQUIRES_APPROVAL",
    }));
  }

  return actions; // level 4 (research only)
}

🧠 4. SKILL WEIGHT EVOLUTION ENGINE

Instead of retraining models, you adjust decision weights safely.

// server/rlhf/skillWeights.ts

export type SkillWeights = {
  sepsisThreshold: number;
  icuPriority: number;
  antibioticUrgency: number;
};

export function evolveWeights(current: SkillWeights, feedback: number) {
  const cap = 0.02;

  return {
    sepsisThreshold: current.sepsisThreshold * (1 + Math.min(cap, feedback)),
    icuPriority: current.icuPriority * (1 + Math.min(cap, feedback)),
    antibioticUrgency: current.antibioticUrgency * (1 + Math.min(cap, feedback)),
  };
}
🔐 5. SAFETY GOVERNOR (CRITICAL)

Prevents dangerous learning drift.

// server/rlhf/safetyGovernor.ts

export function validateUpdate(oldWeights: any, newWeights: any) {
  for (const key in oldWeights) {
    const change = Math.abs(newWeights[key] - oldWeights[key]) / oldWeights[key];

    if (change > 0.02) {
      throw new Error("Unsafe weight change detected");
    }
  }

  return true;
}
