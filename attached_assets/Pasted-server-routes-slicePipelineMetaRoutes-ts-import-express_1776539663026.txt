server/routes/slicePipelineMetaRoutes.ts
import express from 'express';
import fs from 'fs';
import path from 'path';
import { db } from '../db/client';
import { reviewSlices, claudeSliceReviews, openaiSliceReviews, sliceProposals } from '../db/schema';
import { eq } from 'drizzle-orm';


const router = express.Router();


router.get('/history/:sliceId', async (req, res) => {
  try {
    const sliceRows = await db.select().from(reviewSlices).where(eq(reviewSlices.sliceId, req.params.sliceId));
    const slice = sliceRows[0];
    if (!slice) return res.status(404).json({ ok: false, error: 'Slice not found' });


    const claude = await db.select().from(claudeSliceReviews).where(eq(claudeSliceReviews.reviewSliceId, slice.id));
    const openai = await db.select().from(openaiSliceReviews).where(eq(openaiSliceReviews.reviewSliceId, slice.id));
    const proposals = await db.select().from(sliceProposals).where(eq(sliceProposals.reviewSliceId, slice.id));


    res.json({
      ok: true,
      slice,
      history: {
        claude,
        openai,
        proposals,
      },
    });
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});


router.get('/download/:sliceId', async (req, res) => {
  try {
    const sliceRows = await db.select().from(reviewSlices).where(eq(reviewSlices.sliceId, req.params.sliceId));
    const slice = sliceRows[0];
    if (!slice?.exportPath) return res.status(404).json({ ok: false, error: 'Slice export not found' });


    const normalized = path.normalize(slice.exportPath);
    if (!fs.existsSync(normalized)) return res.status(404).json({ ok: false, error: 'Slice file missing on disk' });


    res.download(normalized, `${slice.sliceId}.md`);
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});


export default router;
