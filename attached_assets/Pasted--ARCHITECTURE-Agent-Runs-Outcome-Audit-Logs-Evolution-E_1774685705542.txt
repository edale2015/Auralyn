🧠 PART 1 — EVOLUTION STORE (VERSIONING)
📁 evolutionStore.ts
type AgentVersion = {
  agent: string;
  version: number;
  config: any;
  metrics?: any;
  approved: boolean;
  timestamp: number;
};

const versions: AgentVersion[] = [];

export function saveVersion(v: AgentVersion) {
  versions.push(v);
}

export function getLatestVersion(agent: string) {
  return versions
    .filter(v => v.agent === agent && v.approved)
    .sort((a, b) => b.version - a.version)[0];
}

import { getRecentOutcomes } from "../learning/outcomeStore";

export function proposeEvolution() {
  const outcomes = getRecentOutcomes(100);

  const failures = outcomes.filter(o => !o.correct);

  if (failures.length < 5) return null;

  return {
    agent: "RoutingAgent",
    change: "increase_complex_case_threshold",
    newConfig: { complexityThreshold: 0.6 }
  };
}
🧪 PART 3 — SANDBOX SIMULATOR

Never deploy directly. Always test.

📁 sandboxRunner.ts
import { runGoldenCases } from "../golden/goldenRunner";

export async function runSandbox(runFlow: any, config: any) {
  const results = await runGoldenCases(runFlow);

  const passRate =
    results.filter(r => r.correct).length / results.length;

  return {
    passRate,
    results
  };
}
📊 PART 4 — VALIDATION GATE

Only allow improvements that are actually better.

📁 evolutionValidator.ts
export function validateEvolution(current: any, candidate: any) {
  if (candidate.passRate < current.passRate) return false;
  if (candidate.passRate < 0.8) return false;

  return true;
}

import { saveVersion } from "./evolutionStore";

export function promote(agent: string, config: any, metrics: any) {
  saveVersion({
    agent,
    version: Date.now(),
    config,
    metrics,
    approved: true,
    timestamp: Date.now()
  });

  console.log(`🚀 Promoted new version of ${agent}`);
}
🔁 PART 6 — EVOLUTION LOOP

Runs periodically.

import { proposeEvolution } from "./evolutionEngine";
import { runSandbox } from "./sandboxRunner";
import { validateEvolution } from "./evolutionValidator";
import { promote } from "./promotionEngine";

export async function runEvolutionCycle(runFlow: any) {

  const proposal = proposeEvolution();
  if (!proposal) return;

  console.log("🧬 Proposed:", proposal);

  const sandbox = await runSandbox(runFlow, proposal.newConfig);

  const current = { passRate: 0.85 }; // from last validation

  const approved = validateEvolution(current, sandbox);

  if (approved) {
    promote(proposal.agent, proposal.newConfig, sandbox);
  } else {
    console.log("❌ Evolution rejected");
  }
}
