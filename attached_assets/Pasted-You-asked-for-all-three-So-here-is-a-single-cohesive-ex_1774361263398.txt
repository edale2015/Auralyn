type ExamStep =
  | { type: "ask"; question: string; field: string }
  | { type: "vision"; target: "throat" | "ear" | "rash" }
  | { type: "robot"; action: any }
  | { type: "device"; device: string }
  | { type: "decision"; rule: (ctx: any) => boolean; next: string };

type ExamProtocol = {
  id: string;
  complaint: string;
  steps: ExamStep[];
};

export async function runExamProtocol(protocol: ExamProtocol, ctx: any) {
  const results: any = {};

  for (const step of protocol.steps) {
    switch (step.type) {
      case "ask":
        const answer = await ctx.ask(step.question);
        results[step.field] = answer;
        break;

      case "vision":
        const vision = await ctx.vision(step.target);
        results[step.target] = vision;

        if (vision.redFlags?.length) {
          return { escalate: true, reason: "vision_red_flag", results };
        }
        break;

      case "robot":
        await ctx.robot(step.action);
        break;

      case "device":
        const reading = await ctx.device(step.device);
        results[step.device] = reading;
        break;

      case "decision":
        if (step.rule(results)) {
          return { next: step.next, results };
        }
        break;
    }
  }

  return { complete: true, results };
}

type DeviceReading =
  | { type: "bp"; systolic: number; diastolic: number }
  | { type: "spo2"; value: number }
  | { type: "ekg"; waveform: number[] };

export async function readDevice(device: string): Promise<DeviceReading> {
  switch (device) {
    case "bp":
      return fetchDevice("/bp");

    case "spo2":
      return fetchDevice("/spo2");

    case "ekg":
      return fetchDevice("/ekg");

    default:
      throw new Error("Unknown device");
  }
}

async function fetchDevice(path: string) {
  const res = await fetch(process.env.DEVICE_API + path);
  return res.json();
}
