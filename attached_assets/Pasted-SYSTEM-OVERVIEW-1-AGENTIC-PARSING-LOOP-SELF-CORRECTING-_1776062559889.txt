// server/document/agenticParser.ts

import { parseDocument } from "./parser";
import OpenAI from "openai";

const openai = new OpenAI();

export async function agenticParse(filePath: string) {
  let markdown = await parseDocument(filePath);

  for (let i = 0; i < 3; i++) {
    const evaluation = await evaluateParse(markdown);

    if (evaluation.confidence > 0.9) {
      return markdown;
    }

    console.log(`🔁 Re-parsing attempt ${i + 1}`);

    markdown = await improveParse(markdown, evaluation.issues);
  }

  return markdown;
}

async function evaluateParse(markdown: string) {
  const res = await openai.chat.completions.create({
    model: "gpt-5.3",
    messages: [
      {
        role: "system",
        content: `
Evaluate document parsing quality.

Check:
- Missing tables
- Broken structure
- Missing headings
- Clinical data integrity

Return JSON:
{ confidence: 0-1, issues: [] }
        `,
      },
      { role: "user", content: markdown },
    ],
  });

  return JSON.parse(res.choices[0].message.content!);
}

async function improveParse(markdown: string, issues: string[]) {
  const res = await openai.chat.completions.create({
    model: "gpt-5.3",
    messages: [
      {
        role: "system",
        content: `
Fix document parsing issues.

Preserve:
- Tables
- Structure
- Clinical meaning
        `,
      },
      {
        role: "user",
        content: `
Issues:
${issues.join("\n")}

Document:
${markdown}
        `,
      },
    ],
  });

  return res.choices[0].message.content!;
}

// server/agents/docCouncil.ts

import OpenAI from "openai";
import { retrieveRelevant } from "../document/retriever";

const openai = new OpenAI();

const AGENTS = [
  { name: "ICU", focus: "critical illness and deterioration" },
  { name: "InfectiousDisease", focus: "infection diagnosis and treatment" },
  { name: "Emergency", focus: "triage and disposition" },
  { name: "Pharmacology", focus: "medications and dosing safety" },
];

export async function runDocumentCouncil(query: string) {
  const context = await retrieveRelevant(query);

  const responses = [];

  for (const agent of AGENTS) {
    const res = await openai.chat.completions.create({
      model: "gpt-5.3",
      messages: [
        {
          role: "system",
          content: `
You are a ${agent.name} specialist.

Focus: ${agent.focus}

Use ONLY the provided context.
Return structured JSON:
{
  opinion: "",
  confidence: 0-1,
  risks: [],
  recommendation: ""
}
          `,
        },
        {
          role: "user",
          content: `
Question:
${query}

Context:
${context}
          `,
        },
      ],
    });

    responses.push({
      agent: agent.name,
      ...JSON.parse(res.choices[0].message.content!),
    });
  }

  return responses;
}
🧮 3. CONSENSUS ENGINE (VOTING + SAFETY PRIORITY)
// server/agents/consensusEngine.ts

export function buildConsensus(responses: any[]) {
  let best = responses[0];

  for (const r of responses) {
    if (r.confidence > best.confidence) {
      best = r;
    }

    // Safety override
    if (r.risks?.includes("life-threatening")) {
      best = r;
    }
  }

  return {
    finalRecommendation: best.recommendation,
    confidence: best.confidence,
    dissentingOpinions: responses.filter(r => r !== best),
  };
}

import { db } from "../db";

export async function ingestToKB(consensus: any, sourceDoc: string) {
  const entry = {
    rule_type: "document-derived",
    recommendation: consensus.finalRecommendation,
    confidence: consensus.confidence,
    source: sourceDoc,
    status: "pending_review",
    created_at: new Date(),
  };

  await db.insert("kb_pending_rules", entry);

  return entry;
}

// server/kb/reviewQueue.ts

export async function getPendingRules() {
  return await db.select("*").from("kb_pending_rules")
    .where({ status: "pending_review" });
}

export async function approveRule(id: string) {
  const rule = await db.find("kb_pending_rules", id);

  await db.insert("kb_rules", {
    ...rule,
    status: "approved",
  });

  await db.update("kb_pending_rules", id, {
    status: "approved",
  });
}

// server/audit/councilAudit.ts

import crypto from "crypto";

export function auditCouncil(query: string, responses: any, consensus: any) {
  const record = {
    query,
    responses,
    consensus,
    timestamp: Date.now(),
  };

  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(record))
    .digest("hex");

  return { ...record, hash };
}
