import { v4 as uuidv4 } from "uuid";

type Task = {
  id: string;
  description: string;
  status: "pending" | "in_progress" | "done";
};

export class ClinicalDeepAgent {
  private tasks: Task[] = [];
  private tools: Record<string, Function>;

  constructor(tools: Record<string, Function>) {
    this.tools = tools;
  }

  // 🧠 Planning (write_todos equivalent)
  plan(caseData: any) {
    this.tasks = [
      { id: uuidv4(), description: "Collect modifiers", status: "pending" },
      { id: uuidv4(), description: "Check red flags", status: "pending" },
      { id: uuidv4(), description: "Run Bayesian engine", status: "pending" },
      { id: uuidv4(), description: "Evaluate deterioration risk", status: "pending" },
      { id: uuidv4(), description: "Determine disposition", status: "pending" }
    ];
  }

  async execute(caseData: any) {
    for (const task of this.tasks) {
      task.status = "in_progress";

      switch (task.description) {
        case "Collect modifiers":
          await this.tools.collectModifiers(caseData);
          break;
        case "Check red flags":
          await this.tools.checkRedFlags(caseData);
          break;
        case "Run Bayesian engine":
          caseData.differential = await this.tools.bayesian(caseData);
          break;
        case "Evaluate deterioration risk":
          caseData.risk = await this.tools.deterioration(caseData);
          break;
        case "Determine disposition":
          caseData.disposition = await this.tools.disposition(caseData);
          break;
      }

      task.status = "done";
    }

    return caseData;
  }

  getPlan() {
    return this.tasks;
  }
}

3. 📉 ADD CONTEXT COMPRESSION (CRITICAL FOR SCALE)

Deep Agents auto-summarizes context. You currently don’t.

📁 /server/context/contextCompressor.ts
export function compressContext(history: any[]) {
  if (history.length < 20) return history;

  return [
    {
      type: "summary",
      content: summarize(history.slice(0, history.length - 10))
    },
    ...history.slice(-10)
  ];
}

function summarize(entries: any[]) {
  return entries.map(e => e.event).join(" | ").slice(0, 500);
}

export async function runParallelTasks(tasks: Function[]) {
