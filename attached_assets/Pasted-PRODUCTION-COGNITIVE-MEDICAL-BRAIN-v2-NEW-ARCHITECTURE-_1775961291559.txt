CREATE TABLE cases (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  created_at TIMESTAMP DEFAULT now(),
  input JSONB,
  diagnosis TEXT,
  disposition TEXT,
  confidence FLOAT
);

CREATE TABLE cognitive_traces (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  case_id UUID REFERENCES cases(id),
  monologue JSONB,
  debate JSONB,
  strategy TEXT,
  uncertainty FLOAT,
  created_at TIMESTAMP DEFAULT now()
);

CREATE TABLE memory_graph (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  symptom TEXT,
  diagnosis TEXT,
  outcome TEXT,
  frequency INT DEFAULT 1
);

// server/cognitive/memoryGraph.ts

import { pool } from "../db/postgres";

export async function writeToMemoryGraph(caseData: any, result: any) {
  const { symptoms } = caseData;
  const diagnosis = result.bayesianResult?.topDiagnosis;

  for (const symptom of symptoms) {
    await pool.query(
      `
      INSERT INTO memory_graph (symptom, diagnosis, frequency)
      VALUES ($1, $2, 1)
      ON CONFLICT (symptom, diagnosis)
      DO UPDATE SET frequency = memory_graph.frequency + 1
      `,
      [symptom, diagnosis]
    );
  }
}
🧾 Case + Audit Write
// server/db/caseWriter.ts

import { pool } from "./postgres";

export async function persistCase(caseData, result) {
  const caseRes = await pool.query(
    `
    INSERT INTO cases (input, diagnosis, disposition, confidence)
    VALUES ($1, $2, $3, $4)
    RETURNING id
    `,
    [
      caseData,
      result.diagnosis,
      result.disposition,
      result.confidence
    ]
  );

  const caseId = caseRes.rows[0].id;

  await pool.query(
    `
    INSERT INTO cognitive_traces
    (case_id, monologue, debate, strategy, uncertainty)
    VALUES ($1, $2, $3, $4, $5)
    `,
    [
      caseId,
      result.reasoning.monologue,
      result.reasoning.debate,
      result.strategy,
      result.reasoning.monologue.uncertainty_level
    ]
  );

  return caseId;
}

🧠 PHASE 4 — FULL ORCHESTRATOR (PRODUCTION)
// server/cognitive/cognitiveOrchestrator.ts

import { generateClinicalMonologue } from "./monologueEngine";
import { runSpecialistDebate } from "./debateCouncil";
import { runBayesianEngine } from "../bayesian/engine";
import { selectStrategy } from "./strategyEngine";
import { applyBiasGuards } from "./biasEngine";
import { computeDisposition } from "./dispositionEngine";
import { writeToMemoryGraph } from "./memoryGraph";
import { persistCase } from "../db/caseWriter";
import { publishCognitiveState } from "../stream/cognitiveStream";

export async function runCognitiveBrain(caseData: any) {
  const monologue = await generateClinicalMonologue(caseData);

  const bayesianResult = await runBayesianEngine(caseData);

  const debate = await runSpecialistDebate(caseData, bayesianResult);

  const strategy = selectStrategy(monologue, debate);

  const safePlan = applyBiasGuards({
    plan: debate,
    monologue
  });

  const disposition = computeDisposition({
    confidence: debate.confidence,
    uncertainty: monologue.uncertainty_level,
    disagreement: debate.disagreementScore,
    redFlags: caseData.redFlags
  });

  const result = {
    diagnosis: safePlan.final_diagnosis,
    disposition,
    confidence: debate.confidence,
    strategy,
    reasoning: {
      monologue,
      debate
    }
  };

  // 🔥 LIVE STREAM
  await publishCognitiveState("temp", result);

  // 🔥 PERSIST
  const caseId = await persistCase(caseData, result);

  // 🔥 MEMORY
  await writeToMemoryGraph(caseData, result);

  return { caseId, ...result };
}

import { useEffect, useState } from "react";

export default function CognitivePanel() {
  const [state, setState] = useState(null);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:8080");

    ws.onmessage = (event) => {
      setState(JSON.parse(event.data));
    };

    return () => ws.close();
  }, []);

  if (!state) return <div>Waiting for brain activity...</div>;

  return (
    <div>
      <h2>🧠 Cognitive State</h2>
      <p>Diagnosis: {state.state.diagnosis}</p>

      <p>Uncertainty: {state.state.reasoning.monologue.uncertainty_level}</p>
      <p>Strategy: {state.state.strategy}</p>
    </div>
  );
}
