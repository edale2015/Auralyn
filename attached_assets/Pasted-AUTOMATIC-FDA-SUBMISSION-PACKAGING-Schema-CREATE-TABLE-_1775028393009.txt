CREATE TABLE IF NOT EXISTS fda_packages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  version TEXT,
  intended_use TEXT,
  system_description JSONB,
  validation_metrics JSONB,
  risk_analysis JSONB,
  audit_summary JSONB,
  created_at TIMESTAMP DEFAULT now()
);
⚙️ Generator
// server/fda/buildPackage.ts
export async function buildFDAPackage(db: any) {
  const metrics = await db.validation_runs.findLatest()
  const audit = await db.audit_events.findAll()

  const risk = {
    red_flag_miss_rate: metrics.false_reassurance_rate,
    mitigation: "Red flag layer + uncertainty gating"
  }

  const pkg = {
    version: "v1.0",
    intended_use: "Clinical decision support for triage",
    system_description: {
      pipeline: [
        "intake",
        "modifiers",
        "findings",
        "red flags",
        "workup",
        "diagnosis",
        "treatment",
        "disposition"
      ]
    },
    validation_metrics: metrics,
    risk_analysis: risk,
    audit_summary: {
      total_events: audit.length
    }
  }

  await db.fda_packages.insert(pkg)
  return pkg
}
🔌 Route
router.get('/fda-package', async (req, res) => {
  const pkg = await buildFDAPackage(req.db)
  res.json(pkg)
})

⚙️ Engine
// server/payer/report.ts
function avg(arr:number[]) {
  return arr.length ? arr.reduce((a,b)=>a+b,0)/arr.length : 0
}
function rate(arr:boolean[]) {
  return arr.length ? arr.filter(Boolean).length/arr.length : 0
}

export async function generatePayerReport(db:any) {
  const rows = await db.payer_outcomes.findAll()

  const report = {
    period: "monthly",
    avg_cost: avg(rows.map((r:any)=>r.cost)),
    readmission_rate: rate(rows.map((r:any)=>r.readmission)),
    avg_los: avg(rows.map((r:any)=>r.length_of_stay)),
    outcome_score: avg(rows.map((r:any)=>r.outcome))
  }

  await db.payer_reports.insert(report)
  return report
}

⚙️ Engine
// server/risk/malpractice.ts
export function computeMalpracticeRisk(caseData:any) {
  let risk = 0
  const drivers:string[] = []

  if (caseData.red_flag_missed) {
    risk += 0.5
    drivers.push("Missed red flag")
  }

  if (caseData.uncertainty > 0.4) {
    risk += 0.3
    drivers.push("High uncertainty")
  }

  if (caseData.override) {
    risk += 0.2
    drivers.push("Clinician override")
  }

  return {
    risk_score: Math.min(1, risk),
    drivers
  }
}
🔌 Hook
const risk = computeMalpracticeRisk(caseData)

await db.malpractice_risk.insert({
  patient_id: caseData.id,
  risk_score: risk.risk_score,
  drivers: risk.drivers
})

}
