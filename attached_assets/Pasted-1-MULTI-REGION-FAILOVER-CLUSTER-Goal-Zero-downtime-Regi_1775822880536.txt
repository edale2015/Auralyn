  { name: "us-east", url: "https://us-east.api", healthy: true },
  { name: "us-west", url: "https://us-west.api", healthy: true },
  { name: "eu-central", url: "https://eu.api", healthy: true }
];

export function getHealthyRegion() {
  return REGIONS.find(r => r.healthy) || REGIONS[0];
}
🔧 Failover HTTP wrapper
export async function resilientFetch(path: string, options: any) {

  for (const region of REGIONS) {
    if (!region.healthy) continue;

    try {
      const res = await fetch(region.url + path, options);
      if (res.ok) return res.json();
    } catch {
      region.healthy = false;
    }
  }

  throw new Error("All regions failed");
}

export async function sendToHospitalAPI(patient: any) {

  const res = await fetch(process.env.HOSPITAL_API!, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.HOSPITAL_TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify(patient)
  });

  return res.json();
}

export async function simulateHospital(hours = 24) {

  let state = {
    patients: 0,
    er: 0,
    telemed: 0,
    overload: false
  };

  for (let h = 0; h < hours; h++) {

    const arrivals = Math.floor(Math.random() * 50);
    state.patients += arrivals;

    const erCases = Math.floor(arrivals * 0.2);
    state.er += erCases;

    if (state.er > 200) {
      state.overload = true;
    }
  }

  return state;
}
