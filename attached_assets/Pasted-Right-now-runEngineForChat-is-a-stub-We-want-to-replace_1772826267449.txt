Right now runEngineForChat() is a stub. We want to replace it with a thin adapter, not smear engine logic all over the chat code.

Design principle

Keep one adapter seam:

chat/session layer owns:

sessions

messages

answer persistence

next-question orchestration

engine adapter owns:

translating CaseRecord to engine input

calling loadComplaintConfig

calling runGenericComplaintV1

translating engine output back to CaseEngineResult

That keeps the architecture clean.

File 2: server/services/chatAnswerNormalizer.ts

Purpose: convert raw chat text into structured answer values before feeding the engine.

Examples:

"yes", "yep", "yeah" → true

"no", "nah" → false

"5" → 5

everything else stays string

Draft
export function normalizeChatAnswer(raw: string): string | number | boolean {
  const s = raw.trim().toLowerCase();

  if (["yes", "y", "yeah", "yep", "true"].includes(s)) return true;
  if (["no", "n", "nah", "nope", "false"].includes(s)) return false;

  if (/^-?\d+(\.\d+)?$/.test(s)) {
    const n = Number(s);
    if (!Number.isNaN(n)) return n;
  }

  return raw.trim();
}

import { loadComplaintConfig } from "./complaintConfigLoader";
// import { evaluateExpr } from "./exprEval"; // if safe to use here

export async function planNextQuestion(
  complaintId: string,
  answers: Record<string, unknown>,
  unansweredCriticalQuestions: string[] = []
): Promise<{
  token?: string;
  text?: string;
  completed: boolean;
}> {
  if (unansweredCriticalQuestions.length > 0) {
    const token = unansweredCriticalQuestions[0];
    return {
      token,
      text: `Please answer: ${token.toLowerCase().replace(/_/g, " ")}?`,
      completed: false
    };
  }

  const cfg = await loadComplaintConfig(complaintId);
  const questions = [...(cfg.questions ?? [])].sort(
    (a: any, b: any) => Number(a.ASK_ORDER || 0) - Number(b.ASK_ORDER || 0)
  );

  for (const q of questions) {
    const token = q.Q_ID?.split("_").slice(2).join("_") || q.Q_ID;
    const alreadyAnswered = answers[token] !== undefined && answers[token] !== null && answers[token] !== "";

    if (alreadyAnswered) continue;

    // later: evaluate ASK_IF here
    return {
      token,
      text: q.QUESTION_TEXT,
      completed: false
    };
  }

  return { completed: true };
}

File 4: server/services/chatEngineAdapter.ts

This is the real seam.

Purpose:

translate CaseRecord → engine input

call complaint loader + generic engine

map result into CaseEngineResult

ask planner for next question

Draft structure
import type { ChatEngineRunInput, ChatEngineRunOutput } from "../types/chatEngine";
import type { CaseEngineResult } from "../types/case";
import { loadComplaintConfig } from "./complaintConfigLoader";
import { runGenericComplaintV1 } from "../engines/genericComplaintEngineV1";
import { planNextQuestion } from "./chatQuestionPlanner";

function mapAnswersToEngineInput(answers: Record<string, unknown>) {
  return {
    answers
  };
}

function mapEngineOutputToCaseEngineResult(
  complaintId: string,
  complaintLabel: string | undefined,
  engineOutput: any
): CaseEngineResult {
  return {
    complaintId,
    complaintLabel,
    recommendedDisposition: engineOutput?.disposition || "UNKNOWN",
    confidence: engineOutput?.confidence || "LOW",
    triggeredRedFlags: engineOutput?.triggeredRedFlags || [],
    winningClusterId: engineOutput?.winningClusterId,
    dxCandidates: engineOutput?.likelyDx || engineOutput?.dxCandidates || [],
    clusterScores: engineOutput?.clusterScores || [],
    ruleTrace: engineOutput?.ruleTrace || [],
    noteDraft: engineOutput?.noteDraft,
    returnPrecautions: engineOutput?.returnPrecautions || [],
    render: engineOutput?.render || {},
    engineVersion: "GENERIC_V1"
  };
}

export async function runEngineForChatAdapter(
  input: ChatEngineRunInput
): Promise<ChatEngineRunOutput> {
  const { caseRecord } = input;

  const cfg = await loadComplaintConfig(caseRecord.complaintId);

  const engineInput = mapAnswersToEngineInput(caseRecord.answers || {});

  const engineOutput = await runGenericComplaintV1(cfg, engineInput);

  const engineResult = mapEngineOutputToCaseEngineResult(
    caseRecord.complaintId,
    caseRecord.complaintLabel,
    engineOutput
  );

  const unansweredCriticalQuestions: string[] =
    engineOutput?.unansweredCriticalQuestions || [];

  const next = await planNextQuestion(
    caseRecord.complaintId,
    caseRecord.answers || {},
    unansweredCriticalQuestions
  );

  return {
    engineResult,
    nextQuestionToken: next.token,
    nextQuestionText: next.text,
    unansweredCriticalQuestions,
    completed: next.completed
  };
}
