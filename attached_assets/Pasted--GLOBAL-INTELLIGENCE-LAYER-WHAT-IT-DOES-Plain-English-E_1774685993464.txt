🌐 PART 2 — GLOBAL AGGREGATOR (CENTRAL SERVER)
📁 global/aggregator.ts
const globalStore: any[] = [];

export function receivePayload(payload: any) {
  globalStore.push(payload);
}

export function aggregate() {
  const combined = {
    totalClinics: globalStore.length,
    avgAccuracy:
      globalStore.reduce((a, b) => a + b.metrics.accuracy, 0) /
      globalStore.length
  };

  return combined;
}
🧬 PART 3 — GLOBAL MODEL BUILDER

This creates improved shared intelligence.

📁 global/modelBuilder.ts
export function buildGlobalModel(data: any[]) {
  const weights: any = {};

  for (const d of data) {
    for (const dx in d.weights) {
      weights[dx] = (weights[dx] || 0) + d.weights[dx];
    }
  }

  return weights;
}
