export function build837P(claim: any) {
  return {
    ISA: { sender: "YOUR_ID", receiver: "CLEARINGHOUSE" },
    GS: { type: "HC", version: "005010X222A1" },

    ST: {
      transactionSet: "837",
      claimId: claim.claimId
    },

    NM1: {
      patient: claim.patientName,
      provider: claim.provider
    },

    HI: {
      diagnosis: claim.icd10
    },

    SV1: {
      procedure: claim.cpt,
      charge: claim.amount || "100"
    }
  };
}
🔌 1B. Submit to Clearinghouse API
// server/billing/clearinghouseService.ts

import axios from "axios";
import { build837P } from "./x12Mapper";

export async function submitToClearinghouse(claim: any) {
  const payload = build837P(claim);

  const res = await axios.post(
    process.env.CLEARINGHOUSE_URL!,
    payload,
    {
      headers: {
        Authorization: `Bearer ${process.env.CLEARINGHOUSE_TOKEN}`,
        "Content-Type": "application/json"
      }
    }
  );

  return {
    claimId: claim.claimId,
    clearinghouseId: res.data.id,
    status: "submitted"
  };
}

// server/compliance/riskControl.ts

export function enforceRiskControls(result: any) {
  const critical = ["ACS", "PE", "Stroke"];

  if (
    result.triage === "routine" &&
    critical.includes(result.diagnosis)
  ) {
    return {
      ...result,
      blocked: true,
      reason: "Unsafe discharge blocked"
    };
  }

  if (result.triage === "ER") {
    result.requiresPhysicianReview = true;
  }

  return result;
}

🔒 3C. Rate Limiting (HIPAA + Abuse)
// server/middleware/rateLimit.ts

const requests: Record<string, number[]> = {};

export function rateLimit(req: any, res: any, next: any) {
  const ip = req.ip;
  const now = Date.now();

  if (!requests[ip]) requests[ip] = [];

  requests[ip] = requests[ip].filter(t => now - t < 60000);

  if (requests[ip].length > 100) {
    return res.status(429).send("Too many requests");
  }

  requests[ip].push(now);
  next();
}
