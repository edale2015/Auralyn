import { readEvents } from "../events/eventStream"
import { ClinicalState } from "./clinicalStateTypes"

export async function rebuildClinicalState(caseId:string){

  const events = await readEvents(caseId)

  const state: ClinicalState = {
    caseId,
    events:[]
  }

  for(const event of events){

    switch(event.type){

      case "PATIENT_MESSAGE":

        state.symptoms =
          (state.symptoms ?? "") +
          " " +
          event.payload.message

      break

      case "COMPLAINT_IDENTIFIED":
        state.complaint = event.payload.complaint
      break

      case "DIFFERENTIAL_UPDATED":
        state.differential = event.payload.differential
      break

      case "ALERTS_UPDATED":
        state.alerts = event.payload.alerts
      break

      case "MEDICATION_PLAN":
        state.medicationPlan = event.payload.medication
      break

      case "DISPOSITION_SET":
        state.disposition = event.payload.disposition
      break

      case "NOTE_READY":
        state.chartNote = event.payload.note
      break

    }

  }

  return state
}

1. Feature extractor

server/similarity/caseFeatureExtractor.ts
type CaseFeatures = {
  caseId: string
  complaint: string
  symptoms: string[]
  alerts: string[]
  disposition: string
  differential: string[]
  ageBucket?: string
  sex?: string
}

function normalizeText(s: string): string {
  return s.trim().toLowerCase().replace(/\s+/g, "_")
}

function extractSymptomTokens(symptomsText: string): string[] {
  if (!symptomsText) return []

  const knownPhrases = [
    "fever",
    "cough",
    "sore throat",
    "shortness of breath",
    "chest pain",
    "abdominal pain",
    "headache",
    "neck stiffness",
    "burning urination",
    "back pain",
    "rash",
    "ear pain",
    "difficulty swallowing",
    "vomiting",
    "diarrhea",
    "fatigue",
    "congestion",
    "sinus pressure",
    "white patches",
    "swollen glands",
  ]

  const lower = symptomsText.toLowerCase()
  const matches = knownPhrases.filter((p) => lower.includes(p))
  return matches.map(normalizeText)
}

function toAgeBucket(age?: number): string | undefined {
  if (age == null) return undefined
  if (age < 2) return "infant"
  if (age < 12) return "child"
  if (age < 18) return "adolescent"
  if (age < 50) return "adult"
  if (age < 65) return "older_adult"
  return "elderly"
}

export function extractCaseFeatures(state: any): CaseFeatures {
  const patient = state.patient ?? {}
  const differential =
    (state.differential ?? []).map((d: any) =>
      typeof d === "string" ? d : d.diagnosis
    )

  return {
    caseId: state.caseId,
    complaint: state.complaint ?? "unknown",
    symptoms: extractSymptomTokens(state.symptoms ?? ""),
    alerts: (state.alerts ?? []).map(normalizeText),
    disposition: normalizeText(state.disposition ?? "unknown"),
    differential: differential.filter(Boolean).map(normalizeText),
    ageBucket: toAgeBucket(patient.age),
    sex: patient.sex ? normalizeText(patient.sex) : undefined,
  }
}
2. Similarity scorer

Weighted similarity without ML.

server/similarity/similarityScorer.ts
type FeatureSet = {
  complaint: string
  symptoms: string[]
  alerts: string[]
  disposition: string
  differential: string[]
  ageBucket?: string
  sex?: string
}

function overlapScore(a: string[], b: string[]): number {
  if (!a.length && !b.length) return 1
  if (!a.length || !b.length) return 0

  const setA = new Set(a)
  const setB = new Set(b)
  let intersection = 0

  for (const item of setA) {
    if (setB.has(item)) intersection++
  }

  const union = new Set([...a, ...b]).size
  return union === 0 ? 0 : intersection / union
}

export function scoreCaseSimilarity(
  current: FeatureSet,
  prior: FeatureSet
): number {
  let score = 0

  if (current.complaint === prior.complaint) score += 0.35

  score += overlapScore(current.symptoms, prior.symptoms) * 0.30
  score += overlapScore(current.alerts, prior.alerts) * 0.10
  score += overlapScore(current.differential, prior.differential) * 0.10

  if (current.disposition === prior.disposition) score += 0.05
  if (current.ageBucket && current.ageBucket === prior.ageBucket) score += 0.05
  if (current.sex && current.sex === prior.sex) score += 0.05

  return Math.min(1, score)
}

4. Similarity index builder

Indexes cases from state/audit/reconciliation sources.

server/similarity/caseSimilarityIndex.ts
import fs from "fs/promises"
import path from "path"
import { extractCaseFeatures } from "./caseFeatureExtractor"
import { saveSimilarityIndex } from "./caseSimilarityStore"

const RUNTIME_DIR = path.resolve(process.cwd(), "server/data/runtime")

async function loadNdjson(fileName: string): Promise<any[]> {
  try {
    const raw = await fs.readFile(path.join(RUNTIME_DIR, fileName), "utf8")
    return raw.split(/\r?\n/).filter(Boolean).map(JSON.parse)
  } catch {
    return []
  }
}

export async function rebuildCaseSimilarityIndex() {
  const caseAudits = await loadNdjson("case_audit_log.ndjson")
  const reconciliations = await loadNdjson("case_reconciliation.ndjson")

  const reconciliationByCase = new Map<string, any>()
  for (const row of reconciliations) {
    reconciliationByCase.set(row.case_id ?? row.caseId, row)
  }

  const indexRows = caseAudits.map((audit: any) => {
    const caseId = audit.caseId ?? audit.case_id
    const recon = reconciliationByCase.get(caseId)

    const pseudoState = {
      caseId,
      complaint: audit.complaintId ?? audit.complaint_id ?? "unknown",
      symptoms: audit.rawText ?? "",
      alerts: audit.redFlagHits ?? [],
      disposition: audit.finalDisposition ?? audit.disposition ?? "unknown",
      differential: (audit.differentialTop3 ?? []).map((d: string) => ({
        diagnosis: d,
      })),
      patient: {
        age: audit.age,
        sex: audit.sex,
      },
    }

    return {
      ...extractCaseFeatures(pseudoState),
      outcome: recon
        ? {
            actualDiagnosis: recon.actualFinalDiagnosis ?? "",
            actualDisposition: recon.actualDisposition ?? "",
            topPredictionMatch: Boolean(recon.top_prediction_match),
            dispositionMatch: Boolean(recon.disposition_match),
            safetyMiss: Boolean(recon.safety_miss_flag),
          }
        : null,
    }
  })

  await saveSimilarityIndex(indexRows)
  return {
    count: indexRows.length,
  }
}
5. Similarity retrieval service

This finds nearest prior cases and summarizes them.

server/similarity/caseSimilarityService.ts
import { extractCaseFeatures } from "./caseFeatureExtractor"
import { loadSimilarityIndex } from "./caseSimilarityStore"
import { scoreCaseSimilarity } from "./similarityScorer"

export async function findSimilarCasesForState(state: any, limit = 5) {
  const current = extractCaseFeatures(state)
  const index = await loadSimilarityIndex()

  const scored = index
    .filter((row: any) => row.caseId !== current.caseId)
    .map((row: any) => ({
      ...row,
      similarityScore: scoreCaseSimilarity(current, row),
    }))
    .sort((a: any, b: any) => b.similarityScore - a.similarityScore)
    .slice(0, limit)

  const topDiagnoses: Record<string, number> = {}
  const topDispositions: Record<string, number> = {}

  for (const row of scored) {
    const dx = row.outcome?.actualDiagnosis || row.differential?.[0] || "unknown"
    const disp = row.outcome?.actualDisposition || row.disposition || "unknown"

    topDiagnoses[dx] = (topDiagnoses[dx] ?? 0) + 1
    topDispositions[disp] = (topDispositions[disp] ?? 0) + 1
  }

  return {
    query: current,
    similarCases: scored,
    summary: {
      topDiagnoses: Object.entries(topDiagnoses)
        .sort((a, b) => b[1] - a[1])
        .slice(0, 5)
        .map(([diagnosis, count]) => ({ diagnosis, count })),
      topDispositions: Object.entries(topDispositions)
        .sort((a, b) => b[1] - a[1])
        .slice(0, 5)
        .map(([disposition, count]) => ({ disposition, count })),
    },
  }
}

  const res = await fetch(url)
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

async function postJson<T>(url: string, body: any): Promise<T> {
  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  })
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

1. Outcome-Weighted Similarity
server/similarity/outcomeWeightedSimilarity.ts
import { scoreCaseSimilarity } from "./similarityScorer"

function outcomeWeight(row:any){

  if(!row.outcome) return 0.5

  let weight = 1

  if(row.outcome.topPredictionMatch)
    weight += 0.5

  if(row.outcome.dispositionMatch)
    weight += 0.5

  if(row.outcome.safetyMiss)
    weight -= 1

  return Math.max(0,weight)
}

export function scoreOutcomeWeightedSimilarity(
  current:any,
  prior:any
){

  const similarity = scoreCaseSimilarity(current,prior)

  const weight = outcomeWeight(prior)

  return similarity * weight
}

This means cases with good outcomes count more.

2. Differential Vote Aggregator

Each similar case casts a vote for its final diagnosis.

server/similarity/differentialVoteAggregator.ts
export function aggregateDifferentialVotes(similarCases:any[]){

  const votes:Record<string,number> = {}

  for(const c of similarCases){

    const diagnosis =
      c.outcome?.actualDiagnosis ||
      c.differential?.[0]

    if(!diagnosis) continue

    const weight = c.similarityScore ?? 0

    votes[diagnosis] =
      (votes[diagnosis] ?? 0) + weight
  }

  return Object.entries(votes)
    .map(([diagnosis,score])=>({
      diagnosis,
      score
    }))
    .sort((a,b)=>b.score-a.score)
}

import { extractCaseFeatures } from "./caseFeatureExtractor"
import { loadSimilarityIndex } from "./caseSimilarityStore"
import { scoreOutcomeWeightedSimilarity } from "./outcomeWeightedSimilarity"
import { aggregateDifferentialVotes } from "./differentialVoteAggregator"

export async function computeSimilarityWeightedDifferential(state:any){

  const current = extractCaseFeatures(state)

  const index = await loadSimilarityIndex()

  const scored = index
    .map((row:any)=>({

      ...row,

      similarityScore:
        scoreOutcomeWeightedSimilarity(
          current,
          row
        )

    }))
    .sort((a:any,b:any)=>
      b.similarityScore-a.similarityScore
    )
    .slice(0,10)

  const votes =
    aggregateDifferentialVotes(scored)

  return {
    similarCases:scored,
    voteDifferential:votes
  }

}
