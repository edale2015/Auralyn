type HealthStatus = "green" | "yellow" | "red" | "gray";

export interface EngineHealth {
  name: string;
  status: HealthStatus;
  lastHeartbeat: number;
  lastSuccess?: number;
  lastFailure?: number;
  latencyMs?: number;
  errorCount: number;
  notes?: string;
}

export interface SkillHealth {
  name: string;
  status: HealthStatus;
  lastCalled?: number;
  successCount: number;
  failureCount: number;
  avgLatencyMs?: number;
  lastError?: string;
}

export interface SkillTrace {
  skill: string;
  status: "success" | "failed" | "skipped";
  latencyMs?: number;
  input?: any;
  output?: any;
  reason?: string;
}

export interface CaseTrace {
  caseId: string;
  startedAt: number;
  steps: SkillTrace[];
}

const engineMap = new Map<string, EngineHealth>();
const skillMap = new Map<string, SkillHealth>();
const caseMap = new Map<string, CaseTrace>();

// ---------------- ENGINE ----------------

export function registerEngine(name: string) {
  engineMap.set(name, {
    name,
    status: "gray",
    lastHeartbeat: Date.now(),
    errorCount: 0
  });
}

export function heartbeatEngine(name: string) {
  const e = engineMap.get(name);
  if (!e) return;
  e.lastHeartbeat = Date.now();
  e.status = "green";
}

export function recordEngineSuccess(name: string, latencyMs: number) {
  const e = engineMap.get(name);
  if (!e) return;
  e.lastSuccess = Date.now();
  e.latencyMs = latencyMs;
  e.status = latencyMs > 2000 ? "yellow" : "green";
}

export function recordEngineFailure(name: string, err: string) {
  const e = engineMap.get(name);
  if (!e) return;
  e.lastFailure = Date.now();
  e.errorCount += 1;
  e.status = "red";
  e.notes = err;
}

export function getEngines() {
  return Array.from(engineMap.values());
}

// ---------------- SKILL ----------------

export function registerSkill(name: string) {
  skillMap.set(name, {
    name,
    status: "gray",
    successCount: 0,
    failureCount: 0
  });
}

export function recordSkillSuccess(name: string, latencyMs: number) {
  const s = skillMap.get(name);
  if (!s) return;

  s.lastCalled = Date.now();
  s.successCount += 1;
  s.avgLatencyMs =
    ((s.avgLatencyMs || 0) + latencyMs) / 2;

  s.status = latencyMs > 1500 ? "yellow" : "green";
}

export function recordSkillFailure(name: string, err: string) {
  const s = skillMap.get(name);
  if (!s) return;

  s.lastCalled = Date.now();
  s.failureCount += 1;
  s.status = "red";
  s.lastError = err;
}

export function getSkills() {
  return Array.from(skillMap.values());
}

// ---------------- TRACE ----------------

export function startCase(caseId: string) {
  caseMap.set(caseId, {
    caseId,
    startedAt: Date.now(),
    steps: []
  });
}

export function addTraceStep(caseId: string, step: SkillTrace) {
  const c = caseMap.get(caseId);
  if (!c) return;
  c.steps.push(step);
}

export function getCaseTrace(caseId: string) {
  return caseMap.get(caseId);
}

  });
}

import { getEngines, getSkills } from "./healthRegistry";

const lastAlertMap = new Map<string, number>();
const COOLDOWN = 5 * 60 * 1000;

export function evaluateAlerts() {
  const now = Date.now();
  const alerts: any[] = [];

  const engines = getEngines();
  const skills = getSkills();

  for (const e of engines) {
    if (e.status === "red") {
      if (!shouldAlert(e.name, now)) continue;

      alerts.push({
        type: "ENGINE_DOWN",
        name: e.name,
        severity: "critical"
      });
    }
  }

  for (const s of skills) {
    if (s.failureCount > 5) {
      if (!shouldAlert(s.name, now)) continue;

      alerts.push({
        type: "SKILL_FAILING",
        name: s.name,
        severity: "warning"
      });
    }
  }

  return alerts;
}

function shouldAlert(key: string, now: number) {
  const last = lastAlertMap.get(key) || 0;
  if (now - last < COOLDOWN) return false;

  lastAlertMap.set(key, now);
  return true;
}

const latencyHistory = new Map<string, number[]>();

export function trackLatency(name: string, latency: number) {
  const arr = latencyHistory.get(name) || [];
  arr.push(latency);

  if (arr.length > 20) arr.shift();
  latencyHistory.set(name, arr);
}

export function detectDegradation() {
  for (const [name, arr] of latencyHistory.entries()) {
    const avg = arr.reduce((a, b) => a + b, 0) / arr.length;

    if (avg > 2000) {
      console.log(`⚠️ Degrading: ${name}`);
    }
  }
}
