🧠 1. SCRIPT ENGINE (CORE LOGIC)
/server/services/communication/scriptEngine.ts
import { detectTone } from "./toneDetector";
import { getScriptVariant } from "./scriptVariants";

export interface ScriptInput {
  complaint: string;
  visitCount: number;
  durationDays: number;
  priorAntibiotics: boolean;
  patientText?: string;
}

export interface ScriptOutput {
  script: string;
  tone: string;
  variant: string;
}

export function generateCommunicationScript(input: ScriptInput): ScriptOutput {
  const {
    complaint,
    visitCount,
    durationDays,
    priorAntibiotics,
    patientText
  } = input;

  // 🔥 TRIGGER CONDITIONS
  const isRepeatVisit =
    visitCount >= 3 &&
    durationDays <= 14 &&
    ["cough", "sinus", "uri"].includes(complaint.toLowerCase());

  if (!isRepeatVisit) {
    return {
      script: "",
      tone: "neutral",
      variant: "none"
    };
  }

  // 🧠 Tone detection
  const tone = detectTone(patientText || "");

  // 🎯 Get variant
  const variant = getScriptVariant({
    tone,
    complaint,
    priorAntibiotics
  });

  return {
    script: variant.script,
    tone,
    variant: variant.name
  };
}
🎭 2. TONE DETECTION ENGINE
/server/services/communication/toneDetector.ts
export function detectTone(text: string): string {
  const lower = text.toLowerCase();

  if (
    lower.includes("nothing is helping") ||
    lower.includes("this is ridiculous") ||
    lower.includes("i've been here multiple times")
  ) {
    return "frustrated";
  }

  if (
    lower.includes("i want antibiotics") ||
    lower.includes("just give me something stronger")
  ) {
    return "demanding";
  }

  if (
    lower.includes("i'm worried") ||
    lower.includes("is this serious") ||
    lower.includes("could this be something bad")
  ) {
    return "anxious";
  }

  return "neutral";
}

🔍 1) DEMAND DETECTOR
/server/services/communication/antibioticDemandDetector.ts
export type DemandSignal = {
  isDemandingAntibiotic: boolean;
  phrasesMatched: string[];
};

const PHRASES = [
  "zpack", "z-pak", "z pak", "azithromycin",
  "i want antibiotics", "just give me antibiotics",
  "i know my body", "it always turns into",
  "this always becomes", "it turns into a sore throat",
  "antibiotics always fix it"
];

export function detectAntibioticDemand(text: string = ""): DemandSignal {
  const lower = text.toLowerCase();
  const matches = PHRASES.filter(p => lower.includes(p));
  return {
    isDemandingAntibiotic: matches.length > 0,
    phrasesMatched: matches
  };
}
🧠 2) DEMAND RESPONSE ENGINE
/server/services/communication/antibioticDemandEngine.ts
import { detectAntibioticDemand } from "./antibioticDemandDetector";

export interface AntibioticDemandInput {
  patientText?: string;
  hasBacterialCriteria: boolean; // e.g., Centor ≥ threshold, fever, exudate, etc.
  priorAntibiotics: boolean;
}

export interface AntibioticDemandOutput {
  triggered: boolean;
  script: string;
  offerDelayedRx: boolean;
  rationale: string[];
}

export function generateAntibioticDemandResponse(
  input: AntibioticDemandInput
): AntibioticDemandOutput {

  const { isDemandingAntibiotic } = detectAntibioticDemand(input.patientText);

  if (!isDemandingAntibiotic) {
    return {
      triggered: false,
      script: "",
      offerDelayedRx: false,
      rationale: []
    };
  }

  // If true bacterial criteria → treat normally
  if (input.hasBacterialCriteria) {
    return {
      triggered: true,
      script: `
I hear you—and based on your symptoms and exam today, this does look like something antibiotics can help with. Let’s go ahead and treat it appropriately.
`,
      offerDelayedRx: false,
      rationale: ["bacterial_criteria_met"]
    };
  }

  // 🔥 Core script (no bacterial criteria)
  const script = `
I hear you—you know how your symptoms usually progress, and you’ve seen this pattern before.

My goal is the same as yours—to treat this early if it’s going to turn into something that actually needs treatment.

The tricky part is that most early sore throats are viral, and antibiotics like a Z-Pak don’t actually prevent that progression—they only help in specific bacterial cases.

What I’d rather do is time it so you get treatment when it will actually help. If you develop things like fever, worsening throat pain, or other specific symptoms, that’s when antibiotics become useful.

If it would make you more comfortable, we can also set this up so you don’t have to come back—I can give you a prescription to use only if those symptoms develop.

That way we’re not just treating early—we’re treating effectively.
`;

  return {
    triggered: true,
    script,
    offerDelayedRx: true,
    rationale: ["demand_without_criteria"]
  };
}
