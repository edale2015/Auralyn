import { useEffect, useState } from "react";

export default function ControlTower() {

  const [state, setState] = useState<any>(null);

  async function refresh() {
    const res = await fetch("/api/control/state");
    setState(await res.json());
  }

  useEffect(() => {
    refresh();
    const i = setInterval(refresh, 2000);
    return () => clearInterval(i);
  }, []);

  async function runSimulation() {
    await fetch("/api/control/simulate", { method: "POST" });
  }

  async function runStress() {
    await fetch("/api/control/stress", { method: "POST" });
  }

  async function exportPackage() {
    await fetch("/api/control/export");
    alert("Exported");
  }

  if (!state) return <div>Loading...</div>;

  return (
    <div className="bg-black text-white p-6">

      <h1 className="text-2xl font-bold">🧠 Control Tower</h1>

      {/* SYSTEM STATE */}
      <div className="mt-4">
        <h2>System</h2>
        <p>Regions: {state.infrastructure.regions.join(", ")}</p>
        <p>ML Model: {state.ml.modelVersion}</p>
        <p>Automation Templates: {state.automation.templates}</p>
      </div>

      {/* SAFETY */}
      <div className="mt-4 text-green-400">
        Safety mismatch: {state.safety.mismatchRate}
      </div>

      {/* ACTIONS */}
      <div className="mt-6 space-x-4">
        <button onClick={runSimulation}>🧪 Run Simulation</button>
        <button onClick={runStress}>🔥 Stress Test</button>
        <button onClick={exportPackage}>📦 Export</button>
      </div>

    </div>
  );
}

WebSocket unified stream
import { WebSocketServer } from "ws";
import { controlBus } from "./liveBus";

export function startControlStream(server: any) {

  const wss = new WebSocketServer({ server });

  wss.on("connection", ws => {

    const handler = (data: any) => {
      ws.send(JSON.stringify(data));
    };

    controlBus.on("update", handler);

    ws.on("close", () => {
      controlBus.off("update", handler);
    });
  });
}
