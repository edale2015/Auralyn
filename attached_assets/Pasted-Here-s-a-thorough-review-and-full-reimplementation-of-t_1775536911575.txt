Here's a thorough review and full reimplementation of the audit chain block.

A) Bugs Found
1. Server restart breaks the chain silently
lastHash is initialized to "GENESIS" in memory. After a restart, the chain starts over from GENESIS even though the DB has hundreds of prior entries. Every new entry claims prevHash = "GENESIS", which is cryptographically false — the chain is silently forked and unverifiable.
2. JSON.stringify is not deterministic
Key insertion order varies across JS engines, Node versions, and object construction paths. The same logical audit entry can produce different JSON strings, and therefore different hashes, depending on how the object was built. This silently breaks verification across restarts.
3. timingSafeEqual throws on length mismatch
If claimedHash is malformed (wrong length, truncated, non-hex), Buffer.from(..., "hex") produces a buffer of a different byte length than the expected hash. timingSafeEqual throws a RangeError rather than returning false. The original try/catch handles this, but silently — you can't distinguish a thrown error from a deliberate false result.
4. Race condition on concurrent auditStep() calls
Two concurrent calls both read lastHash, both advance the chain with the same prevHash, then both write to DB. The chain now has two entries claiming the same prevHash — a fork. Neither call is wrong individually, but the chain is broken.
5. In-memory hash advances before DB write succeeds
advanceChain() mutates lastHash synchronously before the db.insert() is awaited. If the DB write fails, lastHash has already moved forward. The next call will chain off a hash that was never persisted, creating a permanently unverifiable gap.

B) Rewritten Code
typescript// server/audit/hashChain.ts
import crypto from "crypto";

/**
 * Deterministic JSON serialization — sorts keys recursively so the same
 * logical object always produces the same string regardless of insertion order
 * or JS engine. This is required for hash stability across restarts.
 */
export function stableStringify(value: unknown): string {
  if (value === null || typeof value !== "object" || Array.isArray(value)) {
    return JSON.stringify(value);
  }
  const sorted = Object.keys(value as Record<string, unknown>)
    .sort()
    .reduce<Record<string, unknown>>((acc, key) => {
      acc[key] = (value as Record<string, unknown>)[key];
      return acc;
    }, {});
  return "{" + Object.keys(sorted).map(k =>
    JSON.stringify(k) + ":" + stableStringify(sorted[k])
  ).join(",") + "}";
}

/**
 * Pure hash computation — no side effects, no state.
 * Always call this with a prevHash you trust (seeded from DB, not memory).
 */
export function computeChainHash(
  prevHash: string,
  entry: Record<string, unknown>
): string {
  const content = prevHash + stableStringify(entry);
  return crypto.createHash("sha256").update(content, "utf8").digest("hex");
}

/**
 * Timing-safe comparison with explicit length guard.
 * Returns false (never throws) if either input is malformed or lengths differ.
 */
export function verifyChainLink(
  entry: Record<string, unknown>,
  prevHash: string,
  claimedHash: string
): boolean {
  try {
    const expected = computeChainHash(prevHash, entry);
    const expectedBuf = Buffer.from(expected, "hex");
    const claimedBuf  = Buffer.from(claimedHash, "hex");

    // timingSafeEqual requires equal lengths — guard explicitly
    if (expectedBuf.length !== claimedBuf.length) return false;

    return crypto.timingSafeEqual(expectedBuf, claimedBuf);
  } catch {
    return false;
  }
}
typescript// server/audit/auditLogger.ts
import { db } from "../db";
import { auditLogs } from "../../shared/schema";
import { eq, desc } from "drizzle-orm";
import { v4 as uuidv4 } from "uuid";
import { computeChainHash } from "./hashChain";

export function createTraceId(): string { return uuidv4(); }

/**
 * Mutex to serialize concurrent auditStep() calls.
 * Prevents two calls from racing on the same prevHash.
 *
 * In a multi-process deployment (e.g. PM2 cluster), replace this with
 * a DB-level advisory lock or a SELECT FOR UPDATE on the last row.
 */
let chainMutex: Promise<void> = Promise.resolve();

/**
 * Seeds the chain head from the database.
 * Call this once at server startup — before any auditStep() calls.
 * This is what fixes the server-restart bug.
 */
export async function initChainFromDb(): Promise<string> {
  const rows = await db
    .select({ hash: auditLogs.hash })
    .from(auditLogs)
    .orderBy(desc(auditLogs.createdAt))
    .limit(1);

  return rows.length > 0 ? rows[0].hash : "GENESIS";
}

/**
 * Writes one audit step to the DB as part of the hash chain.
 *
 * Key design decisions:
 * - The chain head is fetched from DB each time (inside the mutex), not from
 *   a module-level variable. This means a restart never silently forks the chain.
 * - The hash is computed AFTER acquiring the lock and reading the DB head,
 *   so concurrent calls are serialized and each sees the true previous hash.
 * - In-memory state is never advanced until AFTER the DB write succeeds.
 *   If the write fails, no hash is committed and the error is rethrown.
 */
export async function auditStep({
  traceId,
  step,
  input,
  output,
  metadata = {},
}: {
  traceId:  string;
  step:     string;
  input:    unknown;
  output:   unknown;
  metadata?: Record<string, unknown>;
}): Promise<void> {
  // Serialize all writes — one at a time, in order
  chainMutex = chainMutex.then(async () => {
    // Always read the real chain head from DB inside the lock.
    // This is the source of truth — not a module variable.
    const rows = await db
      .select({ hash: auditLogs.hash })
      .from(auditLogs)
      .orderBy(desc(auditLogs.createdAt))
      .limit(1);

    const prevHash = rows.length > 0 ? rows[0].hash : "GENESIS";

    const entry: Record<string, unknown> = {
      traceId,
      step,
      input:    input  ?? null,
      output:   output ?? null,
      metadata,
    };

    const hash = computeChainHash(prevHash, entry);

    // Only write to DB — no in-memory hash is advanced.
    // If this throws, the chain is unmodified and the error propagates.
    await db.insert(auditLogs).values({
      traceId,
      step,
      input:    input  ?? null,
      output:   output ?? null,
      metadata,
      hash,
      prevHash,
    });
  });

  // Propagate errors to the caller rather than swallowing them
  await chainMutex;
}

/**
 * Retrieves all steps for a trace in chronological order.
 */
export async function getTraceSteps(traceId: string) {
  return db
    .select()
    .from(auditLogs)
    .where(eq(auditLogs.traceId, traceId))
    .orderBy(auditLogs.createdAt);
}
