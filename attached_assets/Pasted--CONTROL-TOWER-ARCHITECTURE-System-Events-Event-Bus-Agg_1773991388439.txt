🚀 PART 1 — EVENT BUS (THE NERVOUS SYSTEM)

Everything must emit events.

📁 server/controlTower/eventBus.ts
type Event = {
  type: string;
  payload: any;
  timestamp: number;
};

const listeners: Function[] = [];

export function emitEvent(event: Event) {
  for (const l of listeners) l(event);
}

export function subscribe(fn: Function) {
  listeners.push(fn);
}

📡 PART 2 — AGGREGATOR ENGINE

Collects system-wide state.

📁 server/controlTower/aggregator.ts
const state = {
  patients: [],
  errors: [],
  engines: {},
  alerts: []
};

export function updateState(event) {
  if (event.type === "PATIENT_FLOW") {
    state.patients.push(event.payload);
  }

  if (event.type === "ERROR") {
    state.errors.push(event.payload);
  }

  if (event.type === "ENGINE_STATUS") {
    state.engines[event.payload.name] = event.payload.status;
  }

  if (event.type === "ALERT") {
    state.alerts.push(event.payload);
  }
}

export function getState() {
  return state;
}

🌐 PART 3 — REAL-TIME STREAM (WEBSOCKET)
📁 server/controlTower/socket.ts
import { WebSocketServer } from "ws";
import { getState } from "./aggregator";
import { subscribe } from "./eventBus";

const wss = new WebSocketServer({ port: 8080 });

subscribe(() => {
  const state = JSON.stringify(getState());

  wss.clients.forEach(client => {
    if (client.readyState === 1) {
      client.send(state);
    }
  });
});
🧠 PART 4 — ANOMALY DETECTION ENGINE
📁 server/controlTower/anomalyEngine.ts
import { emitEvent } from "./eventBus";

export function detectAnomalies(state) {
  if (state.errors.length > 10) {
    emitEvent({
      type: "ALERT",
      payload: { message: "High error volume detected" },
      timestamp: Date.now()
    });
  }

  const highRisk = state.patients.filter(p => p.safety?.level === "HIGH");

  if (highRisk.length > 5) {
    emitEvent({
      type: "ALERT",
      payload: { message: "Spike in HIGH-risk patients" },
      timestamp: Date.now()
    });
  }
}

🖥️ PART 5 — FRONTEND CONTROL TOWER
📁 client/components/ControlTower.tsx
import { useEffect, useState } from "react";

export function ControlTower() {
  const [state, setState] = useState(null);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:8080");

    ws.onmessage = (msg) => {
      setState(JSON.parse(msg.data));
    };

    return () => ws.close();
  }, []);

  if (!state) return <div>Loading...</div>;

  return (
    <div>
      <h2>System Control Tower</h2>

      <h3>Patients</h3>
      <p>{state.patients.length}</p>

      <h3>Errors</h3>
      <p>{state.errors.length}</p>

      <h3>Alerts</h3>
      {state.alerts.map((a, i) => (
        <div key={i}>{a.message}</div>
      ))}

      <h3>Engines</h3>
      {Object.entries(state.engines).map(([k, v]) => (
        <div key={k}>
          {k}: {v === "healthy" ? "🟢" : "🔴"}
        </div>
      ))}
    </div>
  );
}
📊 PART 6 — GLOBAL METRICS PANEL
📁 client/components/SystemMetrics.tsx
export function SystemMetrics({ state }) {
  const avgLatency =
    state.patients.reduce((a, p) => a + (p.latency || 0), 0) /
    (state.patients.length || 1);

  return (
    <div>
      <h3>Avg Latency: {Math.round(avgLatency)} ms</h3>
      <h3>Active Patients: {state.patients.length}</h3>
    </div>
  );
}
