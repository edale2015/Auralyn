Failure Taxonomy Engine

server/simulation/failureTaxonomyEngine.ts

export interface FailureAnalysis {

 caseId:string
 category:string
 explanation:string

}

export function classifyFailure(simCase:any,prediction:any):FailureAnalysis|null{

 if(simCase.expectedDisposition==="er_now" && prediction.predictedDisposition!=="er_now"){

  return{
   caseId:simCase.caseId,
   category:"missed_red_flag",
   explanation:"Emergency case triaged lower than ER level"
  }

 }

 if(simCase.expectedDisposition==="self_care" && prediction.predictedDisposition==="er_now"){

  return{
   caseId:simCase.caseId,
   category:"over_triage",
   explanation:"Low risk case escalated to emergency"
  }

 }

 if(simCase.expectedTopDiagnosis && prediction.predictedTopDiagnosis !== simCase.expectedTopDiagnosis){

  return{
   caseId:simCase.caseId,
   category:"wrong_top_diagnosis",
   explanation:"Primary diagnosis mismatch"
  }

 }

 return null

}
Failure Aggregator

server/simulation/failureAggregator.ts

export function aggregateFailures(results:any[]){

 const stats:any={}

 results.forEach(r=>{

  if(!r.failure) return

  if(!stats[r.failure.category])
   stats[r.failure.category]=0

  stats[r.failure.category]++

 })

 return stats

}
2️⃣ Simulation → Learning Bridge

Automatically feeds simulation results back into learning engines.

server/simulation/simulationLearningBridge.ts

export function feedSimulationLearning(results:any[]){

 const updates=[]

 results.forEach(r=>{

  if(!r.dispositionCorrect){

   updates.push({

    type:"disposition_error",
    complaint:r.complaint,
    predicted:r.predictedDisposition,
    expected:r.expectedDisposition

   })

  }

  if(!r.diagnosisMatch){

   updates.push({

    type:"diagnosis_error",
    complaint:r.complaint

   })

  }

 })

 return updates

}

These updates can feed into:

physician learning
