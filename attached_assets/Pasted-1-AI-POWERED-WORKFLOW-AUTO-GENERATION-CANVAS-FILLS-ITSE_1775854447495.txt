import OpenAI from "openai";
const openai = new OpenAI();

export async function generateWorkflow(prompt: string) {
  const res = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "Return a JSON graph with nodes and edges for a clinical workflow." },
      { role: "user", content: prompt }
    ]
  });

  // Expecting: { nodes: [...], edges: [...] }
  const graph = JSON.parse(res.choices[0].message.content!);
  return graph;
}

import { assignCPT } from "./cpt";
import { scrubClaim } from "./scrubber";
import { predictDenial } from "./denialPredictor";
import { payerContract } from "./contracts";

export function processRevenue(patient: any, disposition: string) {

  let claim: any = {
    patientId: patient.patientId,
    insurance: patient.insurance,
    disposition,
    cpt: assignCPT(disposition)
  };

  const scrub = scrubClaim(claim);
  claim = scrub.claim;

  const denial = predictDenial(claim);
  if (denial.risk === "high") {
    claim.cpt = "99284";
  }

  const revenue = payerContract(claim);

  return { claim, denial, revenue };
}

export async function connectorHealth(connectors: any[]) {
  const status: any = {};

  for (const c of connectors) {
    try {
      await c.ping();
      status[c.name] = "ok";
    } catch {
      status[c.name] = "fail";
    }
  }

  return status;
}
