1A. Backend: selector inspector service
// server/templateStudio/selectorInspectorService.ts
export interface SelectorCandidateResult {
  selector: string;
  strategy: string;
  confidence: number;
  matched: boolean;
  previewText?: string;
  previewHtml?: string;
}

export interface SelectorInspectionResult {
  originalSelector: string;
  originalMatched: boolean;
  healedCandidates: SelectorCandidateResult[];
  screenshotPath?: string;
  domSummary?: {
    inputs: Array<{ name?: string; id?: string; placeholder?: string; label?: string }>;
    buttons: Array<{ text?: string; id?: string; ariaLabel?: string }>;
    links: Array<{ text?: string; href?: string }>;
  };
}

async function safeText(locator: any): Promise<string | undefined> {
  try {
    const text = await locator.innerText({ timeout: 1000 });
    return text?.slice(0, 200);
  } catch {
    return undefined;
  }
}

async function safeHtml(locator: any): Promise<string | undefined> {
  try {
    const html = await locator.evaluate((el: Element) => el.outerHTML);
    return String(html).slice(0, 500);
  } catch {
    return undefined;
  }
}

export class SelectorInspectorService {
  async inspect(page: any, originalSelector: string): Promise<SelectorInspectionResult> {
    let originalMatched = false;
    const healedCandidates: SelectorCandidateResult[] = [];

    try {
      const count = await page.locator(originalSelector).count();
      originalMatched = count > 0;
    } catch {
      originalMatched = false;
    }

    const domSummary = await this.buildDomSummary(page);

    const candidateSelectors = this.buildCandidates(originalSelector, domSummary);

    for (const candidate of candidateSelectors) {
      try {
        const locator = page.locator(candidate.selector).first();
        const count = await page.locator(candidate.selector).count();
        const matched = count > 0;

        healedCandidates.push({
          selector: candidate.selector,
          strategy: candidate.strategy,
          confidence: candidate.confidence,
          matched,
          previewText: matched ? await safeText(locator) : undefined,
          previewHtml: matched ? await safeHtml(locator) : undefined,
        });
      } catch {
        healedCandidates.push({
          selector: candidate.selector,
          strategy: candidate.strategy,
          confidence: candidate.confidence,
          matched: false,
        });
      }
    }

    const screenshotPath = `artifacts/selector-inspector-${Date.now()}.png`;
    try {
      await page.screenshot({ path: screenshotPath, fullPage: true });
    } catch {
      // ignore
    }

    return {
      originalSelector,
      originalMatched,
      healedCandidates: healedCandidates.sort((a, b) => {
        if (a.matched !== b.matched) return a.matched ? -1 : 1;
        return b.confidence - a.confidence;
      }),
      screenshotPath,
      domSummary,
    };
  }

  private buildCandidates(originalSelector: string, domSummary: SelectorInspectionResult["domSummary"]) {
    const candidates: Array<{ selector: string; strategy: string; confidence: number }> = [];

    candidates.push({
      selector: originalSelector,
      strategy: "original",
      confidence: 1.0,
    });

    for (const input of domSummary?.inputs || []) {
      if (input.id) {
        candidates.push({
          selector: `#${CSS.escape(input.id)}`,
          strategy: "id",
          confidence: 0.95,
        });
      }
      if (input.name) {
        candidates.push({
          selector: `[name="${input.name}"]`,
          strategy: "name",
          confidence: 0.9,
        });
      }
      if (input.placeholder) {
        candidates.push({
          selector: `[placeholder*="${input.placeholder.replace(/"/g, '\\"')}"]`,
          strategy: "placeholder",
          confidence: 0.75,
        });
      }
      if (input.label) {
        candidates.push({
          selector: `label:has-text("${input.label.replace(/"/g, '\\"')}")`,
          strategy: "label-nearby",
          confidence: 0.72,
        });
      }
    }

    for (const btn of domSummary?.buttons || []) {
      if (btn.id) {
        candidates.push({
          selector: `#${CSS.escape(btn.id)}`,
          strategy: "button-id",
          confidence: 0.92,
        });
      }
      if (btn.ariaLabel) {
        candidates.push({
          selector: `[aria-label="${btn.ariaLabel.replace(/"/g, '\\"')}"]`,
          strategy: "aria-label",
          confidence: 0.88,
        });
      }
      if (btn.text) {
        candidates.push({
          selector: `button:has-text("${btn.text.replace(/"/g, '\\"')}")`,
          strategy: "button-text",
          confidence: 0.82,
        });
      }
    }

    const deduped = new Map<string, { selector: string; strategy: string; confidence: number }>();
    for (const c of candidates) {
      if (!deduped.has(c.selector)) deduped.set(c.selector, c);
    }
    return [...deduped.values()];
  }

  private async buildDomSummary(page: any) {
    return page.evaluate(() => {
      function getLabelFor(el: Element): string | undefined {
        const id = el.getAttribute("id");
        if (id) {
          const explicit = document.querySelector(`label[for="${id}"]`);
          if (explicit?.textContent?.trim()) return explicit.textContent.trim();
        }
        const parentLabel = el.closest("label");
        if (parentLabel?.textContent?.trim()) return parentLabel.textContent.trim();
        return undefined;
      }

      const inputs = Array.from(document.querySelectorAll("input, textarea, select"))
        .slice(0, 50)
        .map((el: Element) => ({
          name: el.getAttribute("name") || undefined,
          id: el.getAttribute("id") || undefined,
          placeholder: el.getAttribute("placeholder") || undefined,
          label: getLabelFor(el),
        }));

      const buttons = Array.from(document.querySelectorAll("button, input[type=button], input[type=submit]"))
        .slice(0, 50)
        .map((el: Element) => ({
          text: (el.textContent || el.getAttribute("value") || "").trim() || undefined,
          id: el.getAttribute("id") || undefined,
          ariaLabel: el.getAttribute("aria-label") || undefined,
        }));

      const links = Array.from(document.querySelectorAll("a"))
        .slice(0, 50)
        .map((el: Element) => ({
          text: (el.textContent || "").trim() || undefined,
          href: el.getAttribute("href") || undefined,
        }));

      return { inputs, buttons, links };
    });
  }
}
1B. Backend route for selector inspection
// server/routes/templateSelectorRoutes.ts
import express from "express";
import { chromium } from "playwright";
import { SelectorInspectorService } from "../templateStudio/selectorInspectorService";

const router = express.Router();
const inspector = new SelectorInspectorService();

router.post("/inspect", async (req, res) => {
  const { url, selector } = req.body as { url: string; selector: string };

  if (!url || !selector) {
    return res.status(400).json({ error: "url and selector are required" });
  }

  const browser = await chromium.launch({ headless: true });
  const page = await browser.newPage();

  try {
    await page.goto(url, { waitUntil: "domcontentloaded", timeout: 30000 });
    const result = await inspector.inspect(page, selector);
    res.json({ result });
  } catch (error: any) {
    res.status(500).json({
      error: error?.message || "Selector inspection failed",
    });
  } finally {
    await browser.close();
  }
});

export default router;

// replace reorderSteps in server/templateStudio/controller.ts
export async function reorderSteps(req: Request, res: Response) {
  const version = await repo.getVersion(req.params.templateId, req.body.versionId);
  if (!version) return res.status(404).json({ error: "Version not found" });

  if (Array.isArray(req.body.steps) && req.body.steps.length > 0) {
    version.steps = req.body.steps;
    await repo.upsertVersion(version);
    return res.json({ version });
  }

  const orderedIds: string[] = req.body.orderedStepIds ?? [];
  const stepMap = new Map(version.steps.map(s => [s.id, s]));
  version.steps = orderedIds.map(id => stepMap.get(id)).filter(Boolean) as any;

  await repo.upsertVersion(version);
  res.json({ version });
}

// client/src/components/template-studio/StepListDnD.tsx
import React, { useState } from "react";

type Step = {
  id: string;
  name: string;
  action: string;
  enabled: boolean;
};

function reorder<T>(items: T[], from: number, to: number): T[] {
  const copy = [...items];
  const [moved] = copy.splice(from, 1);
  copy.splice(to, 0, moved);
  return copy;
}

export default function StepListDnD({
  steps,
  selectedStepId,
  onSelect,
  onReorder,
}: {
  steps: Step[];
  selectedStepId?: string;
  onSelect: (step: Step) => void;
  onReorder: (steps: Step[]) => void;
}) {
  const [dragIndex, setDragIndex] = useState<number | null>(null);

  return (
    <div className="space-y-2">
      {steps.map((step, index) => (
        <div
          key={step.id}
          draggable
          onDragStart={() => setDragIndex(index)}
          onDragOver={e => e.preventDefault()}
          onDrop={() => {
            if (dragIndex == null || dragIndex === index) return;
            onReorder(reorder(steps, dragIndex, index));
            setDragIndex(null);
          }}
          className={`p-3 rounded-xl border cursor-move ${
            selectedStepId === step.id ? "border-blue-500 bg-blue-50" : "bg-white"
          }`}
          onClick={() => onSelect(step)}
        >
          <div className="flex items-center justify-between gap-3">
            <div>
              <div className="font-medium">{step.name}</div>
              <div className="text-xs text-slate-500">
                {step.action} • {step.enabled ? "enabled" : "disabled"}
              </div>
            </div>
            <div className="text-slate-400 text-lg">⋮⋮</div>
          </div>
        </div>
      ))}
    </div>
  );
}

import React, { useState } from "react";

type Candidate = {
  selector: string;
  strategy: string;
  confidence: number;
  matched: boolean;
  previewText?: string;
  previewHtml?: string;
};

export default function SelectorHealingInspector({
  pageUrl,
  selector,
  onAcceptSelector,
}: {
  pageUrl: string;
  selector: string;
  onAcceptSelector: (selector: string) => void;
}) {
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<any>(null);

  async function inspect() {
    setLoading(true);
    setResult(null);
    try {
      const res = await fetch("/api/template-selector/inspect", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ url: pageUrl, selector }),
      });
      const data = await res.json();
      setResult(data.result);
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="bg-white rounded-2xl shadow p-4 space-y-4">
      <div className="flex items-center justify-between gap-3">
        <h3 className="text-lg font-semibold">Selector Healing Inspector</h3>
        <button
          className="px-4 py-2 rounded-xl bg-blue-600 text-white disabled:opacity-50"
          onClick={inspect}
          disabled={!pageUrl || !selector || loading}
        >
          {loading ? "Inspecting..." : "Inspect Selector"}
        </button>
      </div>

      <div className="text-sm text-slate-600">
        <div><strong>Page URL:</strong> {pageUrl || "-"}</div>
        <div><strong>Original selector:</strong> {selector || "-"}</div>
      </div>

      {result && (
        <>
          <div className={`rounded-xl p-3 text-sm ${result.originalMatched ? "bg-green-50 text-green-700" : "bg-red-50 text-red-700"}`}>
            Original selector {result.originalMatched ? "matched successfully" : "did not match"}
          </div>

          <div className="space-y-3">
            {(result.healedCandidates || []).map((c: Candidate, idx: number) => (
              <div key={`${c.selector}-${idx}`} className="border rounded-xl p-3">
                <div className="flex items-start justify-between gap-3">
                  <div className="min-w-0">
                    <div className="font-mono text-xs break-all">{c.selector}</div>
                    <div className="text-xs text-slate-500 mt-1">
                      {c.strategy} • confidence {c.confidence} • {c.matched ? "matched" : "not matched"}
                    </div>
                  </div>
                  <button
                    className="px-3 py-1 rounded-lg border text-sm"
                    onClick={() => onAcceptSelector(c.selector)}
                  >
                    Use
                  </button>
                </div>

                {c.previewText && (
                  <div className="mt-2 text-sm">
                    <strong>Text:</strong> {c.previewText}
                  </div>
                )}

                {c.previewHtml && (
                  <details className="mt-2">
                    <summary className="cursor-pointer text-sm text-slate-600">HTML preview</summary>

                      {c.previewHtml}
                    </pre>

                )}

        </>

  const [selectedVersion, setSelectedVersion] = useState<any>(null);
  const [selectedStep, setSelectedStep] = useState<Step | null>(null);

  const [draftChangelog, setDraftChangelog] = useState("");
  const pageUrl = useMemo(
    () => selectedStep?.url || selectedVersion?.steps?.find((s: Step) => !!s.url)?.url || "",
    [selectedStep, selectedVersion]
  );

  useEffect(() => {
    fetch("/api/template-studio/templates")
      .then(r => r.json())
      .then(data => setTemplates(data.templates || []));
  }, []);

  async function openTemplate(templateId: string) {
    const res = await fetch(`/api/template-studio/templates/${templateId}`);
    const data = await res.json();
    setSelectedTemplate(data.template);

    const current =
      data.versions?.find((v: any) => v.versionId === data.template.currentVersionId) ??
      data.versions?.[data.versions.length - 1];

    setSelectedVersion(current);
    setDraftSteps(current?.steps || []);
    setSelectedStep(current?.steps?.[0] ?? null);
  }

    if (!selectedTemplate || !selectedVersion) return;

    const res = await fetch(`/api/template-studio/templates/${selectedTemplate.id}/reorder`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        versionId: selectedVersion.versionId,
        steps: nextSteps,
      }),
    });

    const data = await res.json();
    setSelectedVersion(data.version);

        createdBy: "user",
        changelog: draftChangelog || "Updated in Template Studio",
        steps: draftSteps,
        variables: selectedVersion?.variables || [],

    setDraftChangelog("");

Part 2: Next layer I recommend
Execution Sandbox / Dry-Run Engine with DOM Snapshot Diff
Why this is the right next engine

After editing and healing selectors, the next danger is obvious:

“Will this template still work before I run it for real?”

So the best next layer is a Sandbox Runner that:

runs a template in dry-run mode
never clicks final submit unless explicitly allowed
captures before/after DOM snapshots per step
computes diffs
flags risky changes
gives a safety score before publish

This becomes your staging lab for automation templates.

What it does

For each step, the Sandbox Engine can:

execute or simulate the step
capture:
screenshot before
DOM summary before
screenshot after
DOM summary after
compute:
fields added/removed
buttons added/removed
URL changed
selector mismatch
visibility changes
mark step as:
safe
warning
risky

This pairs beautifully with Replay Inspector and Template Studio.

import { DomSnapshotSummary } from "../../shared/types/sandbox";

export class DomSnapshotService {
  async capture(page: any): Promise<DomSnapshotSummary> {
    return page.evaluate(() => {
      const inputs = Array.from(document.querySelectorAll("input, textarea, select"))
        .slice(0, 100)
        .map((el: Element) => ({
          tag: el.tagName.toLowerCase(),
          type: el.getAttribute("type") || undefined,
          name: el.getAttribute("name") || undefined,
          id: el.getAttribute("id") || undefined,
          placeholder: el.getAttribute("placeholder") || undefined,
          visibleText: (el.textContent || "").trim() || undefined,
        }));

      const buttons = Array.from(document.querySelectorAll("button, input[type=button], input[type=submit]"))
        .slice(0, 100)
        .map((el: Element) => ({
          text: (el.textContent || el.getAttribute("value") || "").trim() || undefined,
          id: el.getAttribute("id") || undefined,
          ariaLabel: el.getAttribute("aria-label") || undefined,
        }));

      const links = Array.from(document.querySelectorAll("a"))
        .slice(0, 100)
        .map((el: Element) => ({
          text: (el.textContent || "").trim() || undefined,
          href: el.getAttribute("href") || undefined,
        }));

      return {
        url: window.location.href,
        title: document.title,
        inputs,
        buttons,
        links,
      };
    });
  }
}
2C. DOM diff service
// server/templateStudio/domDiffService.ts
import { DomSnapshotSummary } from "../../shared/types/sandbox";

export class DomDiffService {
  diff(before: DomSnapshotSummary, after: DomSnapshotSummary) {
    return {
      urlChanged: before.url !== after.url,
      titleChanged: before.title !== after.title,
      inputCountDelta: after.inputs.length - before.inputs.length,
      buttonCountDelta: after.buttons.length - before.buttons.length,
      linkCountDelta: after.links.length - before.links.length,
    };
  }

  classify(action: string, diff: ReturnType<DomDiffService["diff"]>) {
    const warnings: string[] = [];

    if (diff.urlChanged && action !== "goto") {
      warnings.push("URL changed unexpectedly");
    }
    if (Math.abs(diff.inputCountDelta) > 5) {
      warnings.push("Large input count change");
    }
    if (Math.abs(diff.buttonCountDelta) > 5) {
      warnings.push("Large button count change");
    }

    let status: "safe" | "warning" | "risky" = "safe";
    if (warnings.length > 0) status = "warning";
    if (warnings.some(w => w.includes("unexpected"))) status = "risky";

    return { status, warnings };
  }
}
2D. Sandbox engine
// server/templateStudio/sandboxEngine.ts
import crypto from "crypto";
import path from "path";
import fs from "fs/promises";
import { chromium } from "playwright";
import { DomSnapshotService } from "./domSnapshotService";
import { DomDiffService } from "./domDiffService";
import { SandboxRunResult, SandboxStepResult } from "../../shared/types/sandbox";

async function ensureDir(dir: string) {
  await fs.mkdir(dir, { recursive: true });
}

export class SandboxEngine {
  private domSnapshotService = new DomSnapshotService();
  private domDiffService = new DomDiffService();

  async run(params: {
    templateId: string;
    versionId: string;
    steps: any[];
    allowSubmit?: boolean;
  }): Promise<SandboxRunResult> {
    const browser = await chromium.launch({ headless: true });
    const page = await browser.newPage();
    const sandboxRunId = crypto.randomUUID();
    const artifactDir = path.join(process.cwd(), "artifacts", "sandbox", sandboxRunId);
    await ensureDir(artifactDir);

    const startedAt = new Date().toISOString();
    const results: SandboxStepResult[] = [];

    try {
      for (const step of params.steps) {
        const beforeScreenshot = path.join(artifactDir, `${step.id}-before.png`);
        const afterScreenshot = path.join(artifactDir, `${step.id}-after.png`);

        let success = false;
        let errorMessage: string | undefined;

        const beforeDom = await this.domSnapshotService.capture(page).catch(() => ({
          url: "",
          title: "",
          inputs: [],
          buttons: [],
          links: [],
        }));

        try {
          await page.screenshot({ path: beforeScreenshot, fullPage: true }).catch(() => {});

          if (step.action === "goto") {
            await page.goto(step.url, { waitUntil: "domcontentloaded" });
          } else if (step.action === "click") {
            const looksLikeSubmit = /submit|final|send/i.test(step.name || "") || /submit|final|send/i.test(step.selector || "");
            if (looksLikeSubmit && !params.allowSubmit) {
              throw new Error("Sandbox blocked possible final submit click");
            }
            await page.click(step.selector);
          } else if (step.action === "type") {
            await page.fill(step.selector, step.value ?? "");
          } else if (step.action === "waitFor") {
            await page.waitForSelector(step.selector, { timeout: step.timeoutMs ?? 8000 });
          }

          success = true;
        } catch (err: any) {
          errorMessage = err?.message || "Sandbox step failed";
        }

        const afterDom = await this.domSnapshotService.capture(page).catch(() => ({
          url: "",
          title: "",
          inputs: [],
          buttons: [],
          links: [],
        }));

        await page.screenshot({ path: afterScreenshot, fullPage: true }).catch(() => {});

        const diff = this.domDiffService.diff(beforeDom, afterDom);
        const classification = this.domDiffService.classify(step.action, diff);

        results.push({
          stepId: step.id,
          stepName: step.name,
          action: step.action,
          success,
          status: success ? classification.status : "risky",
          beforeScreenshot,
          afterScreenshot,
          beforeDom,
          afterDom,
          diff,
          warnings: success ? classification.warnings : [...classification.warnings, errorMessage || "Execution failed"],
          errorMessage,
        });

        if (!success) break;
      }

      const overallStatus: SandboxRunResult["overallStatus"] = results.some(r => !r.success)
        ? "failed"
        : results.some(r => r.status === "risky")
        ? "risky"
        : results.some(r => r.status === "warning")
        ? "warning"
        : "safe";

      return {
        sandboxRunId,
        templateId: params.templateId,
        versionId: params.versionId,
        startedAt,
        finishedAt: new Date().toISOString(),
        overallStatus,
        steps: results,
      };
    } finally {
      await browser.close();
    }
  }
}
2E. Sandbox routes
// server/routes/templateSandboxRoutes.ts
import express from "express";
import { TemplateRepository } from "../templateStudio/repository";
import { SandboxEngine } from "../templateStudio/sandboxEngine";

const router = express.Router();
const repo = new TemplateRepository();
const sandboxEngine = new SandboxEngine();

router.post("/run", async (req, res) => {
  const { templateId, versionId, allowSubmit } = req.body as {
    templateId: string;
    versionId: string;
    allowSubmit?: boolean;
  };

  const version = await repo.getVersion(templateId, versionId);
  if (!version) {
    return res.status(404).json({ error: "Version not found" });
  }

  try {
    const result = await sandboxEngine.run({
      templateId,
      versionId,
      steps: version.steps,
      allowSubmit,
    });

    res.json({ result });
  } catch (error: any) {
    res.status(500).json({ error: error?.message || "Sandbox run failed" });
  }
});

export default router;

export default function TemplateSandboxPage() {

  const [loading, setLoading] = useState(false);

    setResult(null);

      const res = await fetch("/api/template-sandbox/run", {
        method: "POST",
