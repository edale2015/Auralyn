📁 server/billing/clearinghouseRealClient.ts
import axios from "axios";

type SubmitResult = {
  claimId: string;
  status: "accepted" | "rejected" | "pending";
  raw?: any;
};

export class ClearinghouseRealClient {
  private base = process.env.CLEARINGHOUSE_URL!;
  private key = process.env.CLEARINGHOUSE_API_KEY!;

  async submit837P(x12Payload: string): Promise<SubmitResult> {
    const res = await axios.post(
      `${this.base}/claims/837p`,
      { payload: x12Payload },
      {
        headers: {
          Authorization: `Bearer ${this.key}`,
          "Content-Type": "application/json",
        },
        timeout: 15000,
      }
    );

    return {
      claimId: res.data.claimId,
      status: res.data.status,
      raw: res.data,
    };
  }

  async checkStatus(claimId: string) {
    const res = await axios.get(
      `${this.base}/claims/${claimId}/status`,
      { headers: { Authorization: `Bearer ${this.key}` } }
    );
    return res.data; // includes payer adjudication
  }

  async fetchERA835(sinceISO: string) {
    const res = await axios.get(
      `${this.base}/era/835?since=${encodeURIComponent(sinceISO)}`,
      { headers: { Authorization: `Bearer ${this.key}` } }
    );
    return res.data; // array of 835s or normalized remits
  }
}
📁 server/billing/x12Builder.ts (minimal 837P generator)
export function build837P(encounter: any) {
  const now = new Date();
  const ts = now.toISOString().replace(/[-:TZ.]/g, "").slice(0, 14);

  const icd = encounter.coding.icd10?.[0]?.code || "R69";
  const cpt = encounter.coding.cpt?.[0]?.code || "99213";

  return [
    `ISA*00*          *00*          *ZZ*SENDERID      *ZZ*RECEIVERID    *${ts.slice(0,8)}*${ts.slice(8,12)}*^*00501*000000905*1*T*:~`,
    `GS*HC*SENDERID*RECEIVERID*${ts.slice(0,8)}*${ts.slice(8,12)}*1*X*005010X222A1~`,
    `ST*837*0001*005010X222A1~`,
    `NM1*41*2*YOUR ORG*****46*123456789~`,
    `NM1*40*2*CLEARINGHOUSE*****46*999999999~`,
    `HL*1**20*1~`,
    `NM1*85*2*PROVIDER*****XX*${process.env.NPI || "1234567890"}~`,
    `CLM*${encounter.encounter_id}*${encounter.billing.estimated_revenue}***11:B:1*Y*A*Y*Y~`,
    `HI*ABK:${icd}~`,
    `SV1*HC:${cpt}*${encounter.billing.estimated_revenue}*UN*1***1~`,
    `DTP*472*D8*${ts.slice(0,8)}~`,
    `SE*12*0001~`,
    `GE*1*1~`,
    `IEA*1*000000905~`,
  ].join("");
}
📁 server/billing/eraProcessor.ts (835 → outcome + RLHF)
import { MoneyRLHFEngine } from "../learning/moneyRLHFEngine";

export class ERAProcessor {
  constructor(private rlhf = new MoneyRLHFEngine()) {}

  // expects normalized ERA items: [{ claimId, paidAmount, denied, reasons }]
  process(eraItems: any[]) {
    for (const item of eraItems) {
      this.rlhf.update({
        encounterId: item.claimId,
        paid: !item.denied,
        revenue: item.paidAmount || 0,
        denialReasons: item.reasons || [],
        coding: item.coding || {},
      });
    }
  }
}
📁 server/routes/claims.ts
import express from "express";
import { ClearinghouseRealClient } from "../billing/clearinghouseRealClient";
import { build837P } from "../billing/x12Builder";

const router = express.Router();
const ch = new ClearinghouseRealClient();

router.post("/submit", async (req, res) => {
  const encounter = req.body;
  const x12 = build837P(encounter);
  const result = await ch.submit837P(x12);
  res.json(result);
});

router.get("/:id/status", async (req, res) => {
  const status = await ch.checkStatus(req.params.id);
  res.json(status);
});

export default router;

import AWS from "aws-sdk";

const sm = new AWS.SecretsManager({ region: process.env.AWS_REGION });

export async function loadSecrets(secretId: string) {
  const data = await sm.getSecretValue({ SecretId: secretId }).promise();
  const json = JSON.parse(data.SecretString || "{}");

  for (const [k, v] of Object.entries(json)) {
    process.env[k] = String(v);
  }
}

📁 infra/aws/rds.ts (CDK)
import * as rds from "aws-cdk-lib/aws-rds";
import * as ec2 from "aws-cdk-lib/aws-ec2";

export function createRDS(scope: any, vpc: ec2.Vpc) {
  return new rds.DatabaseInstance(scope, "ClinicalRDS", {
    engine: rds.DatabaseInstanceEngine.postgres({
      version: rds.PostgresEngineVersion.VER_14,
    }),
    vpc,
    multiAz: true,
    storageEncrypted: true,
    instanceType: ec2.InstanceType.of(
      ec2.InstanceClass.T3,
      ec2.InstanceSize.MEDIUM
    ),
    credentials: rds.Credentials.fromGeneratedSecret("dbuser"),
    databaseName: "clinical",
  });
}
