🧠 GOVERNANCE ENFORCEMENT ENGINE
// server/governance/enforcementEngine.ts

import { db } from "../db";
import { governanceItems } from "@shared/schema";

export interface EnforcementContext {
  action: string;
  clinicId: string;
  payload?: any;
}

export async function enforceGovernance(ctx: EnforcementContext) {
  // 1. Check pending high-risk governance items
  const pending = await db
    .select()
    .from(governanceItems)
    .where(eq(governanceItems.status, "pending"));

  const highRisk = pending.filter(p => p.risk === "high");

  if (highRisk.length > 0) {
    throw new Error(
      `BLOCKED: ${highRisk.length} high-risk governance items pending approval`
    );
  }

  // 2. Action-specific enforcement
  if (ctx.action === "model:update") {
    const requiresApproval = ctx.payload?.impact > 0.05;
    if (requiresApproval && !ctx.payload?.approved) {
      throw new Error("BLOCKED: model update requires approval");
    }
  }

  if (ctx.action === "clinical:override") {
    if (!ctx.payload?.physicianId) {
      throw new Error("BLOCKED: override must be physician-attributed");
    }
  }

  return true;
}
2. 🔗 GLOBAL ENFORCEMENT MIDDLEWARE
// server/middleware/enforce.ts

import { enforceGovernance } from "../governance/enforcementEngine";

export function enforce(action: string) {
  return async (req, res, next) => {
    try {
      await enforceGovernance({
        action,
        clinicId: req.provider?.clinicId || req.physician?.clinicId,
        payload: req.body,
      });
      next();
    } catch (e: any) {
      res.status(403).json({
        ok: false,
        error: e.message,
      });
    }
  };
}

// frontend/src/components/GovernancePanel.tsx

import { useQuery, useMutation } from "@tanstack/react-query";

export default function GovernancePanel() {
  const { data, refetch } = useQuery({
    queryKey: ["governance"],
    queryFn: () => fetch("/api/control-tower/governance").then(r => r.json()),
  });

  const approve = useMutation({
    mutationFn: (id: string) =>
      fetch(`/api/control-tower/governance/${id}/approve`, { method: "POST" }),
    onSuccess: refetch,
  });

  return (
    <div>
      <h2 className="font-bold mb-2">🏛 Governance Queue</h2>

      {data?.map((item: any) => (
        <div key={item.id} className="border p-3 rounded mb-2">
          <div>{item.sheet}</div>
          <div className="text-xs text-gray-500">{item.risk}</div>

          <button onClick={() => approve.mutate(item.id)}>
            Approve
          </button>
        </div>
      ))}
    </div>
  );
}
