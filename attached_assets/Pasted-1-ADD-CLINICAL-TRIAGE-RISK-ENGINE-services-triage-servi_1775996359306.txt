# services/triage_service.py

from typing import Dict, Any

RED_FLAGS = {
    "chest pain": ["shortness of breath", "radiating pain", "diaphoresis"],
    "fever": ["confusion", "hypotension", "tachycardia"],
    "cough": ["hemoptysis", "severe SOB"]
}

RISK_LEVELS = ["LOW", "MODERATE", "HIGH", "CRITICAL"]

def evaluate_risk(symptoms: str) -> Dict[str, Any]:
    symptoms_lower = symptoms.lower()

    detected_flags = []
    for condition, flags in RED_FLAGS.items():
        if condition in symptoms_lower:
            for f in flags:
                if f in symptoms_lower:
                    detected_flags.append(f)

    # Risk logic
    if len(detected_flags) >= 2:
        risk = "CRITICAL"
    elif len(detected_flags) == 1:
        risk = "HIGH"
    elif any(k in symptoms_lower for k in RED_FLAGS.keys()):
        risk = "MODERATE"
    else:
        risk = "LOW"

    return {
        "risk_level": risk,
        "red_flags": detected_flags
    }

from services.triage_service import evaluate_risk

def intake_node(state: BookingState):
    user_msg = state["messages"][-1]["content"]

    triage = evaluate_risk(user_msg)

    state["symptoms"] = user_msg
    state["risk_level"] = triage["risk_level"]
    state["red_flags"] = triage["red_flags"]

    if triage["risk_level"] in ["CRITICAL", "HIGH"]:
        state["stage"] = "escalation"
        return {
            **state,
            "messages": state["messages"] + [{
                "role": "assistant",
                "content": f"⚠️ High-risk symptoms detected: {triage['red_flags']}. Immediate evaluation recommended.",
                "options": ["Go to ER", "Call 911"]
            }]
        }

    elif triage["risk_level"] == "MODERATE":
        state["stage"] = "urgent_booking"
        return {
            **state,
            "messages": state["messages"] + [{
                "role": "assistant",
                "content": "You may need urgent care. Let's book you quickly.",
                "options": ["Book Appointment"]
            }]
        }

    else:
        state["stage"] = "routine_booking"
        return {
            **state,
            "messages": state["messages"] + [{
                "role": "assistant",

def clinical_router(state: BookingState):
    if state["risk_level"] in ["CRITICAL", "HIGH"]:
        return "escalation"
    elif state["risk_level"] == "MODERATE":
        return "select_speciality"
    else:
        return "select_speciality"

# services/audit_service.py

import hashlib
import json
from datetime import datetime

def log_event(state: dict):
    payload = json.dumps(state, sort_keys=True)
    hash_val = hashlib.sha256(payload.encode()).hexdigest()

    with open("audit.log", "a") as f:
        f.write(f"{datetime.utcnow()} | {hash_val} | {payload}\n")
