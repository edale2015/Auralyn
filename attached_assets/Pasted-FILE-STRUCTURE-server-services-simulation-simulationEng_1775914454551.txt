🧪 1) SIMULATION ENGINE (10,000 PATIENTS)
/server/services/simulation/simulationEngine.ts
import { calculateCentorScore } from "../clinical/centorEngine";
import { calculateStrepProbability } from "../clinical/bayesianStrepEngine";

export async function runSimulation(n: number = 10000) {
  const results = [];

  for (let i = 0; i < n; i++) {
    const symptoms = {
      fever: Math.random() > 0.7,
      tonsillarExudate: Math.random() > 0.8,
      tenderAnteriorCervicalNodes: Math.random() > 0.75,
      absenceOfCough: Math.random() > 0.5,
      age: Math.floor(Math.random() * 80)
    };

    const centor = calculateCentorScore(symptoms);
    const prob = calculateStrepProbability({
      fever: symptoms.fever,
      exudate: symptoms.tonsillarExudate,
      nodes: symptoms.tenderAnteriorCervicalNodes,
      cough: !symptoms.absenceOfCough
    });

    const decision = prob > 0.5 ? "ANTIBIOTIC" : "NO_ANTIBIOTIC";

    results.push({ centor, prob, decision });
  }

  return results;
}
📉 2) DRIFT DETECTION ENGINE
/server/services/monitoring/driftDetectionEngine.ts
let baseline = {
  antibioticRate: 0.3,
  returnVisitRate: 0.1
};

export function detectDrift(current: {
  antibioticRate: number;
  returnVisitRate: number;
}) {
  const alerts = [];

  if (Math.abs(current.antibioticRate - baseline.antibioticRate) > 0.1) {
    alerts.push("Antibiotic rate drift detected");
  }

  if (current.returnVisitRate > baseline.returnVisitRate + 0.1) {
    alerts.push("Return visit rate increased");
  }

  return alerts;
}

export function resetBaseline(newBaseline: any) {
  baseline = newBaseline;
}
⚖️ 3) RISK GOVERNANCE ENGINE
/server/services/monitoring/riskGovernanceEngine.ts
export function evaluateRisk(input: {
  decision: string;
  probability: number;
}) {
  const alerts = [];

  if (input.decision === "NO_ANTIBIOTIC" && input.probability > 0.7) {
    alerts.push("Possible under-treatment risk");
  }

  if (input.decision === "ANTIBIOTIC" && input.probability < 0.3) {
    alerts.push("Possible over-treatment risk");
  }

  return alerts;
}

import { useState } from "react";

export default function MonitoringDashboard() {
  const [alerts, setAlerts] = useState([]);

  async function checkDrift() {
    const res = await fetch("/api/monitoring/drift", {
      method: "POST",
      body: JSON.stringify({
        antibioticRate: 0.5,
        returnVisitRate: 0.25
      })
    });

    const data = await res.json();
    setAlerts(data.alerts);
  }

  return (
    <div className="p-6">
      <h2 className="text-xl font-bold">System Monitoring</h2>

      <button onClick={checkDrift} className="bg-blue-500 text-white px-4 py-2 rounded">
        Check Drift
      </button>

      <ul className="mt-4">
        {alerts.map((a, i) => (
          <li key={i} className="text-red-500">{a}</li>
        ))}
      </ul>
    </div>
  );
}
