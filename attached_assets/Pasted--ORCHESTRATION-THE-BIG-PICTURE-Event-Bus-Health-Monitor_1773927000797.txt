type Event = {
  type: string;
  payload: any;
  timestamp: number;
};

type Handler = (event: Event) => Promise<void> | void;

export class EventBus {
  private handlers: Record<string, Handler[]> = {};

  subscribe(type: string, handler: Handler) {
    if (!this.handlers[type]) this.handlers[type] = [];
    this.handlers[type].push(handler);
  }

  async publish(type: string, payload: any) {
    const event: Event = { type, payload, timestamp: Date.now() };
    const subs = this.handlers[type] || [];

    for (const h of subs) {
      try {
        await h(event);
      } catch (e) {
        console.error("Event handler error:", e);
      }
    }
  }
}

export type AgentStatus = "healthy" | "degraded" | "failed";

export type Agent = {
  name: string;
  status: AgentStatus;
  lastHeartbeat: number;
  metrics: {
    latency: number;
    errorRate: number;
  };
};

export class AgentRegistry {
  private agents: Map<string, Agent> = new Map();

  register(agent: Agent) {
    this.agents.set(agent.name, agent);
  }

  update(name: string, updates: Partial<Agent>) {
    const agent = this.agents.get(name);
    if (!agent) return;

    Object.assign(agent, updates, { lastHeartbeat: Date.now() });
  }

  getAll() {
    return Array.from(this.agents.values());
  }

  get(name: string) {
    return this.agents.get(name);
  }
}

import { AgentRegistry } from "./agentRegistry";

export class HealthMonitor {
  constructor(private registry: AgentRegistry) {}

  evaluate() {
    const agents = this.registry.getAll();

    return agents.map(agent => {
      let status = "healthy";

      if (agent.metrics.errorRate > 0.2) status = "degraded";
      if (agent.metrics.errorRate > 0.4) status = "failed";

      if (Date.now() - agent.lastHeartbeat > 15000) {
        status = "failed";
      }

      return {
        name: agent.name,
        status,
        metrics: agent.metrics
      };
    });
  }
}

import { EventBus } from "./eventBus";
import { AgentRegistry } from "./agentRegistry";
import { HealthMonitor } from "./healthMonitor";
import { SelfHealingEngine } from "./selfHealingEngine";
import { GovernanceGate } from "./governanceGate";

export class GlobalOrchestrator {
  private bus = new EventBus();
  private registry = new AgentRegistry();
  private monitor = new HealthMonitor(this.registry);
  private healer = new SelfHealingEngine();
  private governance = new GovernanceGate();

  constructor() {
    this.setupListeners();
  }

  registerAgent(agent: any) {
    this.registry.register(agent);
  }

  private setupListeners() {
    this.bus.subscribe("AGENT_ERROR", async (event) => {
      const agent = this.registry.get(event.payload.name);
      if (!agent) return;

      agent.metrics.errorRate += 0.1;

      const actions = await this.healer.heal(agent);

      console.log("Healing actions:", actions);
    });

    this.bus.subscribe("DEPLOYMENT_REQUEST", async (event) => {
      const decision = this.governance.validateDeployment(event.payload);

      if (!decision.approved) {
        console.log("Deployment blocked:", decision.reason);
        return;
      }

      console.log("Deployment approved");
    });
  }

  async runCycle() {
    const health = this.monitor.evaluate();

    for (const agent of health) {
      if (agent.status !== "healthy") {
        await this.healer.heal(agent);
      }
    }

    return health;
  }

  getStatus() {
    return this.registry.getAll();
  }

  getBus() {
    return this.bus;
  }
}

type MetricSnapshot = {
  latency: number[];
  errorRate: number[];
};

export class PredictiveHealingEngine {
  predictFailure(metrics: MetricSnapshot) {
    const avgLatency = this.avg(metrics.latency);
    const trend = this.trend(metrics.errorRate);

    if (avgLatency > 1500 && trend > 0.05) {
      return {
        risk: "high",
        action: "scale_up"
      };
    }

    if (trend > 0.1) {
      return {
        risk: "medium",
        action: "restart"
      };
    }

    return { risk: "low" };
  }

  private avg(arr: number[]) {
    return arr.reduce((a, b) => a + b, 0) / arr.length;
  }

  private trend(arr: number[]) {
    if (arr.length < 2) return 0;
    return arr[arr.length - 1] - arr[0];
  }
}
