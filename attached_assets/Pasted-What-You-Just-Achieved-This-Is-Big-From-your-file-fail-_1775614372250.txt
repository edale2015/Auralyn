⚖️ PART 2 — FAILURE IMPACT + ADJUSTED UNCERTAINTY
📁 server/clinical/importanceUtils.ts
import { engineRegistry } from "./engineRegistry";

export function computeFailureImpact(engineFailures: { engine: string }[]) {
  const desc = engineRegistry.getDescriptors();
  const map = new Map(desc.map(d => [d.name, d]));

  let impact = 0;

  for (const f of engineFailures) {
    const d = map.get(f.engine);
    if (!d) continue;
    impact += d.importance * (d.critical ? 2 : 1);
  }

  return impact; // higher = worse
}

export function adjustUncertainty(base: number, failureImpact: number) {
  return Math.min(1, base + failureImpact * 0.03);
}

  uncertainty?: number;

export function buildExecutionPlan(ctx: PlanContext) {
  const byPhase = (p: number) => engineRegistry.getByPhase(p).map(e => e.name);

  const plan = {
    phase2: new Set(byPhase(2)),
    phase3: new Set(byPhase(3)),
  };

  // 🔻 Skip low-value engines if high risk (favor speed + safety)
  if (ctx.riskLevel === "high") {
    plan.phase2.delete("findSimilarCasesForState");
    plan.phase2.delete("findSimilarMemoryCases");
  }

  // 🔺 Expand if high uncertainty
  if ((ctx.uncertainty ?? 0) > 0.6) {
    // ensure deeper evidence + contradictions always included
    plan.phase3.add("diagnosticEvidenceEngine");
    plan.phase3.add("evidenceAggregatorEngine");
    plan.phase3.add("contradictionEngine");
  }

  return plan;
}

export async function maybeRequery({
  traceId,

  currentUncertainty,
  currentDifferentials,

  traceId: string;
  baseInput: any;
  currentUncertainty: number;

const KEY = "engine:telemetry";

export async function logEngineTelemetry(engine: string, data: any) {
  const entry = {
    ...data,
    timestamp: Date.now(),
  };

  await redis.lpush(`${KEY}:${engine}`, JSON.stringify(entry));
  await redis.ltrim(`${KEY}:${engine}`, 0, 100); // keep last 100
}

export async function getEngineTelemetry() {
  const keys = await redis.keys(`${KEY}:*`);

  const result: any = {};

  for (const k of keys) {
    const engine = k.split(":")[2];
    const entries = await redis.lrange(k, 0, 10);

    result[engine] = entries.map(e => JSON.parse(e));
  }

  return result;
}

// id="engine_bandit_1"
import { redis } from "../infra/redisClient";

const KEY = "engine:bandit";

export class EngineBandit {

  async record(engine: string, reward: number) {
    await redis.hincrby(`${KEY}:${engine}`, "count", 1);
    await redis.hincrbyfloat(`${KEY}:${engine}`, "reward", reward);
  }

  async score(engine: string) {
    const data = await redis.hgetall(`${KEY}:${engine}`);

    const count = Number(data.count || 1);
    const reward = Number(data.reward || 0);

    return reward / count;
  }

  async rank(engines: string[]) {
    const scores = await Promise.all(
      engines.map(async e => ({
        engine: e,
        score: await this.score(e),
      }))
    );

    return scores.sort((a, b) => b.score - a.score);
  }
}

export const engineBandit = new EngineBandit();

// id="meta_learning_1"
import { redis } from "../infra/redisClient";

const META_KEY = "meta:weights";

export class MetaLearningEngine {

  async updateEngineImportance(engine: string, outcome: number) {
    // outcome: +1 good, -1 bad

    const key = `${META_KEY}:importance:${engine}`;

    const current = Number(await redis.get(key) || 3);

    // small bounded update
    let updated = current + (outcome > 0 ? 0.1 : -0.1);

    updated = Math.max(1, Math.min(5, updated));

    await redis.set(key, updated);
  }

  async getEngineImportance(engine: string) {
    const val = await redis.get(`${META_KEY}:importance:${engine}`);
    return val ? Number(val) : 3;
  }

  async adjustUncertaintyScaling(globalErrorRate: number) {
    // dynamically scale uncertainty inflation
    let scale = 0.03;

    if (globalErrorRate > 0.3) scale = 0.06;
    if (globalErrorRate > 0.5) scale = 0.1;

    await redis.set(`${META_KEY}:uncertainty_scale`, scale);
  }

  async getUncertaintyScale() {
    const val = await redis.get(`${META_KEY}:uncertainty_scale`);
    return val ? Number(val) : 0.03;
  }
}

// id="memory_1"
import { redis } from "../infra/redisClient";

const KEY = "memory:cases";

export class CognitiveMemory {

  async store(caseKey: string, data: any) {
    await redis.set(`${KEY}:${caseKey}`, JSON.stringify(data));
  }

  async retrieveSimilar(features: number[]) {
    const keys = await redis.keys(`${KEY}:*`);

    const results: any[] = [];

    for (const k of keys) {
      const data = JSON.parse(await redis.get(k) || "{}");

      const similarity = this.computeSimilarity(features, data.features || []);

      results.push({ ...data, similarity });
    }

    return results
      .sort((a, b) => b.similarity - a.similarity)
      .slice(0, 5);
  }

  computeSimilarity(a: number[], b: number[]) {
    if (!a.length || !b.length) return 0;

    let score = 0;

    for (let i = 0; i < Math.min(a.length, b.length); i++) {
      score += 1 - Math.abs(a[i] - b[i]) / (Math.abs(a[i]) + 1);
    }

    return score / a.length;
  }
}

📁 server/agents/debateEngine.ts
// id="debate_engine_1"
import type { AgentOutput } from "./types";

export interface DebateMessage {
  from: string;
  to: string;
  critique: string;
  scoreAdjustment: number; // -0.2 to +0.2
}

export class DebateEngine {

  generateCritiques(outputs: AgentOutput[]): DebateMessage[] {
    const messages: DebateMessage[] = [];

    for (const a of outputs) {
      for (const b of outputs) {
        if (a.agent === b.agent) continue;

        // Example critique logic
        const diff = Math.abs(a.confidence - b.confidence);

        if (diff > 0.3) {
          messages.push({
            from: a.agent,
            to: b.agent,
            critique: `Confidence mismatch (${a.confidence.toFixed(2)} vs ${b.confidence.toFixed(2)})`,
            scoreAdjustment: -0.1,
          });
        }

        // Risk vs treatment mismatch
        if (a.agent === "risk" && b.agent === "treatment") {
          if (a.result?.risk > 0.7 && !b.result?.recommendation) {
            messages.push({
              from: "risk",
              to: "treatment",
              critique: "High risk without actionable treatment",
              scoreAdjustment: -0.2,
            });
          }
        }
      }
    }

    return messages;
  }

  apply(messages: DebateMessage[], outputs: AgentOutput[]) {
    const adjusted = outputs.map(o => ({ ...o }));

    for (const m of messages) {
      const target = adjusted.find(a => a.agent === m.to);
      if (!target) continue;

      target.confidence = Math.max(
        0,
        Math.min(1, target.confidence + m.scoreAdjustment)
      );
    }

    return adjusted;
  }
}

// id="consensus_engine_1"
import type { AgentOutput } from "./types";

export class ConsensusEngine {

  compute(outputs: AgentOutput[]) {

    const totalWeight = outputs.reduce((sum, o) => sum + o.confidence, 0) || 1;

    const weightedRisk =
      outputs.reduce((sum, o) => {
        const r = o.result?.risk || 0;
        return sum + r * o.confidence;
      }, 0) / totalWeight;

    const disagreement =
      Math.max(...outputs.map(o => o.confidence)) -
      Math.min(...outputs.map(o => o.confidence));

    return {
      risk: weightedRisk,
      disagreement,
      confidence: totalWeight / outputs.length,
    };
  }
}

import { debateEngine } from "./debateEngine";
import { consensusEngine } from "./consensusEngine";

import { buildPharyngitisGraph } from "../reasoning/sampleGraphs/pharyngitisGraph";

const graph = buildPharyngitisGraph();
const graphAdapter = new AgentGraphAdapter(graph);

export class MultiAgentCouncil {

  async run(input: any) {

    // 1. Initial parallel reasoning
    let outputs = await Promise.all([
      diagnosticAgent(input),
      riskAgent(input),
      treatmentAgent(input),
      safetyAgent(input),
      memoryAgent(input),
    ]);

    // 2. Debate round
    const critiques = debateEngine.generateCritiques(outputs);
    outputs = debateEngine.apply(critiques, outputs);

    // 3. Consensus
    const consensus = consensusEngine.compute(outputs);

    // 4. Graph reasoning
    const graphPaths = graphAdapter.buildPaths(outputs, input.patient);

    // 5. Final synthesis
    return {
      consensus,
      agents: outputs,
      debate: critiques,
      reasoningPaths: graphPaths,
      finalDecision: this.finalize(consensus, outputs),
    };
  }

  finalize(consensus: any, outputs: any[]) {
    const treatment = outputs.find(o => o.agent === "treatment")?.result;

    if (consensus.disagreement > 0.5) {
      return {
        action: "physician_review",
        reason: "high_disagreement",
      };
    }

    if (consensus.risk > 0.7) {
      return {
        action: "urgent_care",
        treatment,
      };
    }

    return {
      action: "outpatient",
      treatment,
    };
  }
}

One honest caveat: these are bootstrap-trained synthetic models and curated benchmark cases, not validated clinical artifacts. The code path is real and runnable, but the clinical performance claims should stay in the sandbox until you mount your true trained models and benchmark against a real held-out set.

The main remaining bottleneck the comparison exposed is this: your current master activation policy still tends to select top-2 councils by default, which creates extra-council noise in lower-acuity cases. The sharpest next patch is a variable activation budget so the brain can choose one council, two councils, or three based on uncertainty and instability, instead of always inviting two specialists to every hallway whisper.
