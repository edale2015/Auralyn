    """

    result = get_llm_response(red_flag_prompt).strip()

Instead of Yes/No → score + filter

def rank_context(state):
    query = state["query"]
    candidates = state["candidates"]

    scored = []

    for doc in candidates:
        prompt = f"""
        Score relevance from 0 to 1:

        Query: {query}
        Context: {doc}

        Return number only.
        """

        score = float(get_llm_response(prompt))
        scored.append((doc, score))

    # sort by score
    scored.sort(key=lambda x: x[1], reverse=True)

    # keep top
    state["context"] = [doc for doc, s in scored if s > 0.6][:3]

    return state

Every step logged:

import hashlib
import json
import time

def audit_log(state, step_name):
    entry = {
        "step": step_name,
        "timestamp": time.time(),
        "data": state
    }

    entry_str = json.dumps(entry, sort_keys=True)
    entry_hash = hashlib.sha256(entry_str.encode()).hexdigest()

    print(f"[AUDIT] {step_name} | HASH: {entry_hash}")

    return entry_hash
