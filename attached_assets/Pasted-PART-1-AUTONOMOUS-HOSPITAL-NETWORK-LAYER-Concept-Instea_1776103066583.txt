// server/network/hospitalNetworkController.ts

import { rankPatients } from "../icu/patientCommandCenter";

interface HospitalNode {
  id: string;
  city: string;
  icuBeds: number;
  activePatients: any[];
}

export function optimizeNetwork(nodes: HospitalNode[]) {
  const networkState = nodes.map(node => ({
    ...node,
    rankedPatients: rankPatients(node.activePatients),
  }));

  return balanceLoad(networkState);
}

function balanceLoad(nodes: HospitalNode[]) {
  const transfers: any[] = [];

  const overloaded = nodes.filter(n => n.icuBeds < 2);
  const available = nodes.filter(n => n.icuBeds > 5);

  for (const o of overloaded) {
    const critical = o.rankedPatients[0];

    if (!critical) continue;

    const target = available[0];
    if (!target) break;

    transfers.push({
      patientId: critical.id,
      from: o.id,
      to: target.id,
    });

    target.icuBeds--;
    o.icuBeds++;
  }

  return { nodes, transfers };
}

// server/network/nationalRoutingEngine.ts

interface Facility {
  id: string;
  city: string;
  lat: number;
  lon: number;
  capacity: number;
}

export function routeNationwide(patient: any, facilities: Facility[]) {
  return facilities
    .map(f => ({
      ...f,
      score:
        f.capacity * 0.6 -
        distance(patient, f) * 0.4,
    }))
    .sort((a, b) => b.score - a.score)[0];
}

function distance(a: any, b: any) {
  return Math.sqrt(
    Math.pow(a.lat - b.lat, 2) +
    Math.pow(a.lon - b.lon, 2)
  );
}

// server/network/networkLearningEngine.ts

interface Outcome {
  diagnosis: string;
  disposition: string;
  outcome: "correct" | "missed";
}

let weights: Record<string, number> = {};

export function updateNetworkLearning(data: Outcome[]) {
  for (const d of data) {
    if (!weights[d.diagnosis]) weights[d.diagnosis] = 1;

    if (d.outcome === "correct") {
      weights[d.diagnosis] += 0.01;
    } else {
      weights[d.diagnosis] -= 0.02;
    }
  }

  return weights;
}

// server/payer/payerOptimizationEngine.ts

interface Claim {
  payer: string;
  cpt: string;
  approved: boolean;
  reimbursement: number;
}

export function optimizePayerStrategy(claims: Claim[]) {
  const payerStats: any = {};

  for (const c of claims) {
    if (!payerStats[c.payer]) {
      payerStats[c.payer] = {
        approvals: 0,
        total: 0,
        revenue: 0,
      };
    }

    payerStats[c.payer].total++;
    if (c.approved) payerStats[c.payer].approvals++;
    payerStats[c.payer].revenue += c.reimbursement;
  }

  return Object.entries(payerStats).map(([payer, stats]: any) => ({
    payer,
    approvalRate: stats.approvals / stats.total,
    avgRevenue: stats.revenue / stats.total,
  }));
}

// server/payer/contractNegotiator.ts

export function generateNegotiationStrategy(payerStats: any[]) {
  return payerStats.map(p => ({
    payer: p.payer,
    strategy:
      p.approvalRate < 0.8
        ? "Renegotiate — low approval rate"
        : p.avgRevenue < 100
        ? "Increase CPT mix"
        : "Maintain contract",
  }));
}

import { useEffect, useState } from "react";

export default function NationalControlTower() {
  const [network, setNetwork] = useState<any>(null);

  useEffect(() => {
    fetch("/api/network")
      .then(r => r.json())
      .then(setNetwork);
  }, []);

  if (!network) return <div>Loading...</div>;

  return (
    <div>
      <h1>🌎 National Hospital Network</h1>

      {network.nodes.map((n: any) => (
        <div key={n.id}>
          <h3>{n.city}</h3>
          <p>ICU Beds: {n.icuBeds}</p>
        </div>
      ))}

      <h2>🚑 Transfers</h2>
      {network.transfers.map((t: any, i: number) => (
        <p key={i}>
          {t.patientId} → {t.to}
        </p>
      ))}
    </div>
  );
}
