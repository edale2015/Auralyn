import express from "express";
import http from "http";
import { Server } from "socket.io";
import bodyParser from "body-parser";
import { runFullTriage } from "./engine/runFullTriage";
import { ehrWrite } from "./ehr/ehrWriter";

const app = express();
app.use(bodyParser.json());

const server = http.createServer(app);
const io = new Server(server, {
  cors: { origin: "*" }
});

// 🌐 WebSocket live patient stream
io.on("connection", (socket) => {
  console.log("Client connected");

  socket.on("subscribePatients", () => {
    setInterval(() => {
      socket.emit("patientUpdate", global.patientStore || []);
    }, 2000);
  });
});

// 🧠 Full triage endpoint
app.post("/api/triage", async (req, res) => {
  const result = await runFullTriage(req.body);

  // push to live dashboard
  global.patientStore = [...(global.patientStore || []), result];

  io.emit("patientUpdate", global.patientStore);

  // write to EHR
  await ehrWrite(result);

  res.json(result);
});

server.listen(3000, () => {
  console.log("🚀 Medical OS running on port 3000");
});
🧠 ENGINE (AGENT LOOP)
engine/runFullTriage.ts
import { runSkills } from "../skills/skillRunner";
import { runAgentCouncil } from "../agents/multiAgentCouncil";
import { preDispositionHook } from "../hooks/preDisposition";

export async function runFullTriage(patient) {
  const skills = runSkills(patient);

  const agentDecision = runAgentCouncil(patient);

  const finalDecision = preDispositionHook(patient, agentDecision);

  return {
    patient,
    skills,
    agentDecision,
    finalDecision,
    timestamp: new Date()
  };
}
🤖 MULTI-AGENT SYSTEM
agents/multiAgentCouncil.ts
import { ICUAgent } from "./icuAgent";
import { IDAgent } from "./idAgent";

export function runAgentCouncil(patient) {
  const agents = [
    new ICUAgent(),
    new IDAgent()
  ];

  const results = agents
    .map(a => a.evaluate(patient))
    .filter(Boolean);

  return results.sort((a, b) => b.confidence - a.confidence)[0];
}
⚠️ HOOK SYSTEM (SAFETY)
export function preDispositionHook(patient, decision) {
  if (patient.redFlags?.length > 0) {
    return {
      disposition: "ER_IMMEDIATE",
      reason: "Red flag override"
    };
  }

  return decision;
}
🧾 EHR AUTOMATION (ATHENA / ECW / EPIC UI + API)
ehr/ehrWriter.ts
import axios from "axios";

export async function ehrWrite(result) {
  // Example: Athena API
  try {
    await axios.post(process.env.ATHENA_API, {
      patient: result.patient,
      diagnosis: result.finalDecision,
      notes: JSON.stringify(result)
    });
  } catch (e) {
    console.log("EHR write failed, fallback to UI automation");
    await simulateUIEntry(result);
  }
}
🖥️ UI Automation (Playwright)
import { chromium } from "playwright";

export async function simulateUIEntry(result) {
  const browser = await chromium.launch();
  const page = await browser.newPage();

  await page.goto(process.env.ATHENA_URL);

  await page.fill("#patientName", result.patient.name);
  await page.fill("#diagnosis", result.finalDecision.disposition);

  await page.click("#submit");

  await browser.close();
}
📡 WEBSOCKET FRONTEND CONNECTION
frontend/src/websocket/usePatients.ts
import { useEffect, useState } from "react";
import { io } from "socket.io-client";

export function usePatients() {
  const [patients, setPatients] = useState([]);

  useEffect(() => {
    const socket = io("http://localhost:3000");

    socket.emit("subscribePatients");

    socket.on("patientUpdate", (data) => {
      setPatients(data);
    });

    return () => socket.disconnect();
  }, []);

  return patients;
}

export function validateCase(result, expected) {
  return {
    correctDiagnosis: result.finalDecision === expected,
    timestamp: new Date(),
    riskLevel: result.agentDecision?.confidence
  };
}
