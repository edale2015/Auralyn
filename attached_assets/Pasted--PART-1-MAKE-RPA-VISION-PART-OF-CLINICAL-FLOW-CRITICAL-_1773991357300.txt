import { runBrowserTask } from "../rpa/browserAgent";
import { analyzeScreenshot } from "../vision/visionEngine";

export async function executeActions(plan) {
  const results = [];

  for (const step of plan.steps || []) {
    if (step.type === "rpa") {
      const r = await runBrowserTask(step.task);
      results.push({ type: "rpa", result: r });
    }

    if (step.type === "vision") {
      const v = await analyzeScreenshot(step.image);
      results.push({ type: "vision", result: v });
    }
  }

  return results;
}

import IORedis from "ioredis";

const redis = new IORedis(process.env.REDIS_URL);

export async function shouldSendAlert(key: string) {
  const exists = await redis.get(key);

  if (exists) return false;

  await redis.set(key, "1", "EX", 300); // 5 min
  return true;
}

const sm = new AWS.SecretsManager();

export async function refreshSecrets() {
  const data = await sm.getSecretValue({

  }).promise();

  const secrets = JSON.parse(data.SecretString || "{}");

  Object.assign(process.env, secrets);
}

export function autonomyDecision({ safety, confidence }) {
  if (safety.level === "HIGH") {
    return { mode: "ESCALATE" };
  }

  if (confidence > 0.85 && safety.level === "LOW") {
    return { mode: "AUTO" };
  }

  return { mode: "REVIEW" };
}
