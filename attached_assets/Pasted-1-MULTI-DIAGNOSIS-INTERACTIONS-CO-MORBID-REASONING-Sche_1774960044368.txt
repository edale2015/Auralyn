⚙️ Engine (pairwise adjustment over base scores)
// server/engine/coMorbidityEngine.ts
type DxScore = { diagnosis: string; score: number; meta?: any }

function meetsConditions(input: any, cond?: any): boolean {
  if (!cond) return true
  for (const [k, v] of Object.entries(cond)) {
    if (typeof v === 'object' && v !== null) {
      if ('>' in v && !(input[k] > (v as any)['>'])) return false
      if ('<' in v && !(input[k] < (v as any)['<'])) return false
    } else {
      if (input[k] !== v) return false
    }
  }
  return true
}

export async function applyCoMorbidityAdjustments(
  input: any,
  baseResults: DxScore[]
): Promise<DxScore[]> {
  const interactions = await db.kb_diagnosis_interactions.findAll({ is_active: true })

  // map for quick access
  const map = new Map(baseResults.map(r => [r.diagnosis, { ...r }]))

  for (const inter of interactions) {
    if (!meetsConditions(input, inter.conditions)) continue

    const a = map.get(inter.dx_a)
    const b = map.get(inter.dx_b)
    if (!a || !b) continue

    // log-space adjustment
    const adj = inter.strength // e.g., +0.3 synergy, -0.8 exclusion
    if (inter.interaction_type === 'synergy') {
      a.score += adj
      b.score += adj
    } else if (inter.interaction_type === 'risk_boost') {
      // boost only if both present signals
      if (a.score > -5 && b.score > -5) {
        a.score += adj
        b.score += adj
      }
    } else if (inter.interaction_type === 'exclusion') {
      // penalize the weaker of the two
      if (a.score > b.score) b.score += adj // adj negative
      else a.score += adj
    } else if (inter.interaction_type === 'conditional') {
      a.score += adj
      b.score += adj
    }

    a.meta = { ...(a.meta || {}), interaction: inter }
    b.meta = { ...(b.meta || {}), interaction: inter }
  }

  return Array.from(map.values()).sort((x, y) => y.score - x.score)
}

⚙️ Engine
// server/engine/temporalEngine.ts
type Series = { t: number; v: number }[]

function detectPattern(series: Series): 'rising'|'falling'|'persistent'|'intermittent'|'acute_onset' {
  if (!series || series.length < 2) return 'persistent'
  const first = series[0].v
  const last = series[series.length - 1].v
  const diffs = series.slice(1).map((p, i) => p.v - series[i].v)

  const up = diffs.filter(d => d > 0).length
  const down = diffs.filter(d => d < 0).length

  if (last - first > 0 && up > down) return 'rising'
  if (last - first < 0 && down > up) return 'falling'
  const signChanges = diffs.reduce((c, d, i) => (i>0 && Math.sign(d) !== Math.sign(diffs[i-1]) ? c+1 : c), 0)
  if (signChanges >= 2) return 'intermittent'
  if (series.length <= 3 && Math.abs(last - first) > 0.5) return 'acute_onset'
  return 'persistent'
}

export async function applyTemporalAdjustments(
  caseId: string,
  baseResults: { diagnosis: string; score: number }[]
) {
  const patterns = await db.kb_temporal_patterns.findAll({ is_active: true })
  const seriesByFeature = await db.patient_time_series.groupByFeature(caseId) // {feature_key: Series}

  const out = baseResults.map(r => ({ ...r }))

  for (const p of patterns) {
    const series = seriesByFeature[p.feature_key]
    if (!series) continue
    const pat = detectPattern(series)

    if (pat === p.pattern_type) {
      const idx = out.findIndex(x => x.diagnosis === p.diagnosis)
      if (idx >= 0) {
        out[idx].score += Math.log(p.likelihood)
        ;(out[idx] as any).temporal = { feature: p.feature_key, pattern: pat }
      }
    }
  }

  return out.sort((a, b) => b.score - a.score)
}

// server/learning/outcomeLearningEngine.ts
export async function generateLearningEvents() {
  const rows = await db.kb_outcomes.findRecent({ limit: 10000 })

  const byDx: Record<string, { total: number; errors: number }> = {}

  for (const r of rows) {
    if (!byDx[r.predicted_dx]) byDx[r.predicted_dx] = { total: 0, errors: 0 }
    byDx[r.predicted_dx].total++
    if (!r.correct) byDx[r.predicted_dx].errors++
  }

  const events = []
  for (const [dx, stats] of Object.entries(byDx)) {
    const errRate = stats.errors / Math.max(1, stats.total)
    if (errRate > 0.15) {
      // propose down-weight base prior slightly
      events.push({
        rule_id: `DX_BAY_${dx.toUpperCase()}`,
        feature_key: '__base__',
        delta: -0.05,
        confidence: Math.min(0.95, errRate + 0.5),
        source: 'real_world'
      })
    }
  }

  for (const e of events) {
    await db.kb_learning_events.insert(e)
  }

  return events
}

// server/learning/applyApprovedChanges.ts
export async function applyApprovedLearningEvents() {
  const events = await db.kb_learning_events.findAll({ status: 'approved' })

  for (const e of events) {
    if (e.feature_key === '__base__') {
      const rule = await db.kb_diagnosis_rules.findOne({ rule_id: e.rule_id })
      const newVal = Math.max(0.001, rule.base_probability + e.delta)

      await db.kb_diagnosis_rules.update({
        rule_id: e.rule_id,
        base_probability: newVal
      })
    } else {
      const f = await db.kb_feature_models.findOne({
        rule_id: e.rule_id,
        feature_key: e.feature_key
      })
      if (!f) continue
      const newVal = Math.min(0.999, Math.max(0.001, (f.p_present || 0.5) + e.delta))

      await db.kb_feature_models.update({
        id: f.id,
        p_present: newVal
      })
    }

    await db.kb_learning_events.update({
      id: e.id,
      status: 'deployed'
    })

    await db.changeAuditLog.append({
      type: 'learning_deploy',
      payload: e
    })
  }
}
