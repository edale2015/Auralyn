2) Medium scout using RSS feeds

This avoids brittle scraping and lets you configure topic feeds.

server/research/mediumScout.ts
import Parser from 'rss-parser';
import { db } from '../db/client';
import { researchArticles } from '../db/schema';

const parser = new Parser();

const MEDIUM_FEEDS = [
  'https://medium.com/feed/tag/medical-ai',
  'https://medium.com/feed/tag/clinical-decision-support',
  'https://medium.com/feed/tag/fhir',
  'https://medium.com/feed/tag/bayesian',
  'https://medium.com/feed/tag/artificial-intelligence',
];

function extractTags(categories?: string[]): string[] {
  return (categories ?? []).filter(Boolean).slice(0, 10);
}

export async function scanMediumFeeds() {
  const inserted: number[] = [];

  for (const feedUrl of MEDIUM_FEEDS) {
    let feed;
    try {
      feed = await parser.parseURL(feedUrl);
    } catch (err) {
      console.warn('[mediumScout] feed failed:', feedUrl, err);
      continue;
    }

    for (const item of feed.items ?? []) {
      const title = item.title?.trim();
      const url = item.link?.trim();
      if (!title || !url) continue;

      const result = await db
        .insert(researchArticles)
        .values({
          source: 'medium',
          title,
          url,
          author: item.creator ?? item.author ?? null,
          publishedAt: item.pubDate ? new Date(item.pubDate) : null,
          excerpt: item.contentSnippet?.slice(0, 800) ?? null,
          tags: extractTags(item.categories),
          raw: item as any,
        })
        .onConflictDoNothing?.() as any;

      if (result?.[0]?.id) inserted.push(result[0].id);
    }
  }

  return inserted;
}
3) Article triage
server/research/articleTriage.ts
type TriageInput = {
  title: string;
  excerpt?: string | null;
  tags?: string[];
};

type TriageResult = {
  relevanceScore: number;
  trustScore: number;
  noveltyScore: number;
  actionabilityScore: number;
  verdict: 'adopt' | 'test_only' | 'ignore';
  reasons: string[];
};

const HIGH_VALUE_TERMS = [
  'clinical',
  'medical',
  'sepsis',
  'fhir',
  'audit',
  'bayesian',
  'validation',
  'hallucination',
  'calibration',
  'triage',
  'decision support',
];

const LOW_VALUE_TERMS = [
  'agi',
  'singularity',
  '10x',
  'killer',
  'replace doctors',
  'revolutionary',
];

export function triageArticle(input: TriageInput): TriageResult {
  const text = `${input.title} ${input.excerpt ?? ''} ${(input.tags ?? []).join(' ')}`.toLowerCase();
  let relevanceScore = 20;
  let trustScore = 50;
  let noveltyScore = 40;
  let actionabilityScore = 30;
  const reasons: string[] = [];

  for (const term of HIGH_VALUE_TERMS) {
    if (text.includes(term)) {
      relevanceScore += 7;
      actionabilityScore += 4;
    }
  }

  for (const term of LOW_VALUE_TERMS) {
    if (text.includes(term)) {
      trustScore -= 10;
      actionabilityScore -= 6;
      reasons.push(`Contains hype signal: ${term}`);
    }
  }

  if (text.includes('fda') || text.includes('fhir') || text.includes('calibration')) {
    trustScore += 10;
    reasons.push('Touches regulated or implementation-relevant topic.');
  }

  if (text.includes('case study') || text.includes('implementation')) {
    noveltyScore += 10;
    actionabilityScore += 10;
    reasons.push('Appears implementation-oriented.');
  }

  relevanceScore = Math.max(0, Math.min(100, relevanceScore));
  trustScore = Math.max(0, Math.min(100, trustScore));
  noveltyScore = Math.max(0, Math.min(100, noveltyScore));
  actionabilityScore = Math.max(0, Math.min(100, actionabilityScore));

  const total = relevanceScore * 0.35 + trustScore * 0.2 + noveltyScore * 0.15 + actionabilityScore * 0.3;

  let verdict: TriageResult['verdict'] = 'ignore';
  if (total >= 72) verdict = 'adopt';
  else if (total >= 50) verdict = 'test_only';

  if (!reasons.length) reasons.push('Scored by relevance, trust, novelty, and actionability.');

  return {
    relevanceScore,
    trustScore,
    noveltyScore,
    actionabilityScore,
    verdict,
    reasons,
  };
}

import { db } from '../db/client';
import { researchArticles, researchSummaries } from '../db/schema';

export function summarizeArticleLocally(title: string, excerpt: string | null, tags: string[] = []) {
  const body = excerpt ?? '';
  const summary = [
    `This article discusses: ${title}.`,
    body
      ? `Main idea: ${body.slice(0, 350)}`
      : `Main idea: the article should be reviewed directly because only limited text is available.`,
    `Likely relevance areas for Auralyn: ${tags.join(', ') || 'general medical AI'}.`,
  ].join(' ');

  const takeaways = [
    'Check whether the article contains implementable details rather than concepts only.',
    'Do not adopt clinical logic without validation harness coverage.',
    'Treat this as a candidate input to the upgrade planner, not as authoritative guidance.',
  ];

  return { summary, takeaways };
}

export async function buildArticleSummary(articleId: number) {
  const rows = await db.select().from(researchArticles).where(eq(researchArticles.id, articleId));
  const article = rows[0];
  if (!article) throw new Error(`Article ${articleId} not found`);

  const { summary, takeaways } = summarizeArticleLocally(
    article.title,
    article.excerpt,
    (article.tags as string[]) ?? [],
  );

  const inserted = await db.insert(researchSummaries).values({
    articleId,
    summary,
    takeaways,
  }).returning();

  return inserted[0];
}

      rationale: 'Article suggests techniques relevant to confidence alignment and overconfidence detection.',

      patchBundle: {

      },
      validationPlan: [
        'Run golden cases before and after change.',
        'Compare Brier score and unsafe undercall rate.',
        'Block merge if unsafe undercalls increase.',
      ],
    };
  }

  if (text.includes('fhir')) {
    return {
      upgradeTitle: 'FHIR adapter hardening',
      rationale: 'Article appears relevant to interoperability and EHR integration.',
      affectedFiles: [
        'server/ehr/fhirClient.ts',
        'server/ehr/smartAuth.ts',

      },
      validationPlan: [
        'Run sandbox integration tests.',
        'Verify no change to core disposition logic.',
      ],
    };
  }

  return {
    upgradeTitle: 'Research-derived review item',
    rationale: 'Article is interesting but not specific enough for direct production changes.',
    affectedFiles: ['server/research/upgradePlanner.ts'],
    patchBundle: {},
    validationPlan: [
      'Manual review required.',
      'Convert to test-only item unless stronger implementation evidence appears.',
    ],
  };
}

export async function proposeUpgrade(articleId: number) {
  const rows = await db.select().from(researchArticles).where(eq(researchArticles.id, articleId));
  const article = rows[0];
  if (!article) throw new Error(`Article ${articleId} not found`);

  const plan = planFromArticle(article.title, article.excerpt);

  const inserted = await db.insert(proposedUpgrades).values({
    articleId,
    title: plan.upgradeTitle,
    rationale: plan.rationale,
    affectedFiles: plan.affectedFiles,
    patchBundle: plan.patchBundle,
    validationPlan: plan.validationPlan,
    validationStatus: 'pending',
    requiresHumanApproval: true,
    approved: false,
  }).returning();

  return inserted[0];
}

export async function autoValidateUpgrade(proposedUpgradeId: number) {
  const rows = await db.select().from(proposedUpgrades).where(eq(proposedUpgrades.id, proposedUpgradeId));
  const upgrade = rows[0];
  if (!upgrade) throw new Error(`Proposed upgrade ${proposedUpgradeId} not found`);

  const cases = generateFullCaseSet();
  const result = await runFullValidation(cases);

  const status =
    result.unsafeUndercalls > 0 ? 'failed' :
    result.passRate < 0.85 ? 'failed' :
    'passed';

  await db
    .update(proposedUpgrades)
    .set({ validationStatus: status })
    .where(eq(proposedUpgrades.id, proposedUpgradeId));

  return { upgradeId: proposedUpgradeId, status, result };
}

server/integrations/githubExporter.ts
import { Octokit } from '@octokit/rest';
import { db } from '../db/client';
import { proposedUpgrades, githubExports } from '../db/schema';
import { eq } from 'drizzle-orm';

const GITHUB_OWNER = process.env.GITHUB_OWNER!;
const GITHUB_REPO = process.env.GITHUB_REPO!;
const GITHUB_BASE_BRANCH = process.env.GITHUB_BASE_BRANCH || 'main';

function getOctokit() {
  const token = process.env.GITHUB_TOKEN;
  if (!token) throw new Error('GITHUB_TOKEN missing');
  return new Octokit({ auth: token });
}

export async function exportUpgradeToGitHub(upgradeId: number) {
  const rows = await db.select().from(proposedUpgrades).where(eq(proposedUpgrades.id, upgradeId));
  const upgrade = rows[0];
  if (!upgrade) throw new Error(`Upgrade ${upgradeId} not found`);

  if (!upgrade.approved) {
    throw new Error('Upgrade must be human-approved before GitHub export');
  }

  const octokit = getOctokit();
  const branchName = `research-upgrade/${upgradeId}-${Date.now()}`;

  const repoInfo = await octokit.repos.get({
    owner: GITHUB_OWNER,
    repo: GITHUB_REPO,
  });

  const defaultBranch = repoInfo.data.default_branch || GITHUB_BASE_BRANCH;

  const baseRef = await octokit.git.getRef({
    owner: GITHUB_OWNER,
    repo: GITHUB_REPO,
    ref: `heads/${defaultBranch}`,
  });

  await octokit.git.createRef({
    owner: GITHUB_OWNER,
    repo: GITHUB_REPO,
    ref: `refs/heads/${branchName}`,
    sha: baseRef.data.object.sha,
  });

  const patchBundle = (upgrade.patchBundle || {}) as Record<string, string>;
  let lastCommitSha = baseRef.data.object.sha;

  for (const [filePath, content] of Object.entries(patchBundle)) {
    try {
      const existing = await octokit.repos.getContent({
        owner: GITHUB_OWNER,
        repo: GITHUB_REPO,
        path: filePath,
        ref: branchName,
      });

      if (!Array.isArray(existing.data) && 'sha' in existing.data) {
        const result = await octokit.repos.createOrUpdateFileContents({
          owner: GITHUB_OWNER,
          repo: GITHUB_REPO,
          path: filePath,
          message: `Research upgrade ${upgradeId}: update ${filePath}`,
          content: Buffer.from(content, 'utf8').toString('base64'),
          branch: branchName,
          sha: existing.data.sha,
        });

        lastCommitSha = result.data.commit.sha;
      }
    } catch {
      const result = await octokit.repos.createOrUpdateFileContents({
        owner: GITHUB_OWNER,
        repo: GITHUB_REPO,
        path: filePath,
        message: `Research upgrade ${upgradeId}: add ${filePath}`,
        content: Buffer.from(content, 'utf8').toString('base64'),
        branch: branchName,
      });

      lastCommitSha = result.data.commit.sha;
    }
  }

  const pr = await octokit.pulls.create({
    owner: GITHUB_OWNER,
    repo: GITHUB_REPO,
    title: `[Research Upgrade] ${upgrade.title}`,
    head: branchName,
    base: defaultBranch,
    body: [
      `Upgrade ID: ${upgrade.id}`,
      '',
      `Rationale: ${upgrade.rationale}`,
      '',
      `Validation status: ${upgrade.validationStatus}`,
      '',
      `This PR was generated by the Auralyn research pipeline and still requires review.`,
    ].join('\n'),
  });

  const inserted = await db.insert(githubExports).values({
    proposedUpgradeId: upgrade.id,
    branchName,
    commitSha: lastCommitSha,
    prNumber: pr.data.number,
    prUrl: pr.data.html_url,
    status: 'opened',
  }).returning();

  return inserted[0];
}

import express from 'express';
import { scanMediumFeeds } from '../research/mediumScout';
import { triageArticle } from '../research/articleTriage';

const router = express.Router();

router.post('/scan', async (_req, res) => {

    excerpt: article.excerpt,
    tags: (article.tags as string[]) ?? [],
  });

  const inserted = await db.insert(researchReviews).values({
    articleId,
    ...result,
  }).returning();

  res.json({ ok: true, review: inserted[0] });
});

router.post('/summary/:articleId', async (req, res) => {
  const summary = await buildArticleSummary(Number(req.params.articleId));
  res.json({ ok: true, summary });
});

router.post('/propose/:articleId', async (req, res) => {
  const upgrade = await proposeUpgrade(Number(req.params.articleId));
  res.json({ ok: true, upgrade });
});

router.post('/validate/:upgradeId', async (req, res) => {
  const result = await autoValidateUpgrade(Number(req.params.upgradeId));
  res.json({ ok: true, result });
});

router.post('/approve/:upgradeId', async (req, res) => {
  const approvedBy = String(req.body.approvedBy || '').trim();
  if (!approvedBy) return res.status(400).json({ ok: false, error: 'approvedBy required' });

  const result = await approveUpgrade(Number(req.params.upgradeId), approvedBy);
  res.json({ ok: true, result });
});

router.post('/export-github/:upgradeId', async (req, res) => {
  const result = await exportUpgradeToGitHub(Number(req.params.upgradeId));
  res.json({ ok: true, export: result });
});

export default router;

10) GitHub setup, step by step
Step 1

Create a GitHub repository for Auralyn, or confirm the existing repo you want this pipeline to target.

Step 2

Set your default branch to main.

Step 3

Create a fine-grained personal access token scoped to that repo if this is your personal automation path. GitHub recommends fine-grained PATs when possible for personal REST API use.

For this backend exporter, the token needs enough repository permission to:

read/write repository contents
create pull requests
read metadata

GitHub documents that fine-grained tokens work by granting specific repository permissions.

Step 4

Store that token as a Replit secret or backend environment variable:

GITHUB_TOKEN
GITHUB_OWNER
GITHUB_REPO
GITHUB_BASE_BRANCH=main

Do not commit the token to the repo.

Step 5

In GitHub repo settings, create branch protection on main:

require pull request review
require status checks to pass
optionally require conversation resolution before merge

GitHub branch protection supports requiring approving review and passing checks before merge.

Step 6

Add a validation workflow so PRs get checked before merge.

Step 7

If you later want GitHub Actions itself to create or approve PRs, review your repository’s Actions settings carefully. GitHub documents that workflows creating or approving pull requests can be restricted, and warns that allowing that can be a security risk if merged without oversight.

Step 8

If you want secrets for workflows, add them under Settings → Secrets and variables → Actions. GitHub documents repository, environment, and organization secrets for Actions.

Step 9

Run the pipeline in this order:

POST /api/research/scan
POST /api/research/triage/:articleId
POST /api/research/summary/:articleId
POST /api/research/propose/:articleId
POST /api/research/validate/:upgradeId
POST /api/research/approve/:upgradeId
POST /api/research/export-github/:upgradeId
Step 10

Review the PR in GitHub, run validation, then merge only after human signoff.

What I think of this design

It’s good.

The strong parts are:

no direct article-to-production jump
validation before approval
approval before GitHub export
PR-based change control
clean audit trail

The main weakness is that the planner is still heuristic. That’s fine for now, because it is only generating proposals. The important thing is that validation and human approval sit downstream like a steel gate.

For Auralyn, that is exactly the right shape.

The best next addition would be a small Research Inbox UI that shows:

article summary
triage scores
proposed files
validation result
approve/export buttons

And yes, this agent can absolutely also ask for a few-paragraph summary for you to read. In this implementation, that is what buildArticleSummary() is doing.
