import { runSimulation } from "./runSimulation";

export async function runBatch(cases: any[]) {
  const results = [];

  for (const c of cases) {
    const res = await runSimulation(c);
    results.push(res);
  }

  return results;
}

const FILE = "simulation_reviews.json";

export function saveReview(review: any) {
  let data: any[] = [];

  if (fs.existsSync(FILE)) {
    data = JSON.parse(fs.readFileSync(FILE, "utf-8"));
  }

  data.push({
    ...review,
    timestamp: new Date().toISOString(),
  });

  fs.writeFileSync(FILE, JSON.stringify(data, null, 2));
}

export function generateHighYieldCases(n: number) {
  const complaints = [
    "chest pain",
    "cough",
    "fever",
    "shortness of breath",
    "abdominal pain",
    "headache",
  ];

  const cases = [];

  for (let i = 0; i < n; i++) {
    const complaint =
      complaints[Math.floor(Math.random() * complaints.length)];

    const caseData = {
      complaint,
      data: {
        symptoms: generateSymptoms(complaint),
        age: randomAge(),
        comorbidities: randomComorbidities(),
        medications: randomMedications(),
        vitals: randomVitals(),
        duration: randomDuration(),
      },
    };

    cases.push(caseData);
  }

  return cases;
}

function generateSymptoms(complaint: string) {
  const map: any = {
    "chest pain": ["pressure", "radiation", "sweating"],
    cough: ["productive", "night cough"],
    fever: ["chills", "body aches"],
  };

  return map[complaint] || ["general discomfort"];
}

function randomAge() {
  return Math.floor(Math.random() * 90) + 1;
}

function randomComorbidities() {
  const list = ["diabetes", "htn", "asthma", "cad"];
  return list.filter(() => Math.random() > 0.7);
}

function randomMedications() {
  const meds = ["lisinopril", "metformin", "albuterol"];
  return meds.filter(() => Math.random() > 0.7);
}

function randomVitals() {
  return {
    spo2: Math.random() > 0.8 ? 88 : 98,
    hr: 60 + Math.random() * 60,
  };
}

function randomDuration() {
  return Math.floor(Math.random() * 7);
}

export function amplifyEdgeCases(baseCases: any[]) {
  return baseCases.map(c => {
    if (Math.random() > 0.5) {
      c.data.symptoms.push("atypical");
    }

    if (Math.random() > 0.7) {
      c.data.vitals.spo2 = 85;
    }

    if (Math.random() > 0.6) {
      c.data.missing = true;
    }

    return c;
  });
}
