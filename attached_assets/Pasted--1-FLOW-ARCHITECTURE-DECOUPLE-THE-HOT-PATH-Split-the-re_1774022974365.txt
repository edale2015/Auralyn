🧠 3) RESPONSE CACHE (HUGE WIN FOR REPEAT TRAFFIC)
📁 server/cache/triageCache.ts
import LRU from "lru-cache";

const cache = new LRU({
  max: 2000,
  ttl: 1000 * 60 * 5
});

export function getCached(inputKey) {
  return cache.get(inputKey);
}

export function setCached(inputKey, value) {
  cache.set(inputKey, value);
}
🔥 Use in orchestrator
const key = JSON.stringify(input);

const cached = getCached(key);
if (cached) return cached;

// compute…
setCached(key, response);

👉 This alone can reduce load 30–60%

📁 Redis rate limit
export async function rateLimit(key) {
  const count = await redis.incr(key);

  if (count === 1) {
    await redis.expire(key, 60);
  }

  return count < 100;
}
