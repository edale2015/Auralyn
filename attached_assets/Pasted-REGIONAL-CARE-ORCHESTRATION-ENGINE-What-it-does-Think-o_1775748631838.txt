🧠 1. REGIONAL ORCHESTRATOR CORE
server/regional/regionalOrchestrator.ts
import { routeRegionally } from "./geoRoutingEngine";
import { computeRegionalCapacity } from "./regionalCapacity";
import { predictAdmissionRisk } from "./admissionRisk";
import { predictBounceback } from "./bouncebackPredictor";
import { buildCallbackPlan } from "./callbackAutomation";
import { detectRegionalOutbreak } from "./outbreakDetector";

export async function runRegionalOrchestration(input: any) {

  const regionalCapacity = computeRegionalCapacity(input.facilities);

  const patientPlans = input.patients.map((p: any) => {

    const admissionRisk = predictAdmissionRisk(p);
    const bouncebackRisk = predictBounceback(p);

    const route = routeRegionally({
      patient: p,
      facilities: input.facilities,
      capacity: regionalCapacity
    });

    const callbackPlan = buildCallbackPlan({
      patient: p,
      admissionRisk,
      bouncebackRisk
    });

    return {
      patientId: p.patientId,
      route,
      admissionRisk,
      bouncebackRisk,
      callbackPlan
    };
  });

  const outbreak = detectRegionalOutbreak(input.patients);

  return {
    regionalCapacity,
    patientPlans,
    outbreak
  };
}
🧭 2. GEO ROUTING ENGINE (REAL-WORLD DECISIONS)
server/regional/geoRoutingEngine.ts
export function routeRegionally(input: any) {

  const { patient, facilities, capacity } = input;

  if (patient.safetyDisposition === "ER_NOW") {
    const bestER = facilities
      .filter((f: any) => f.type === "ER")
      .sort((a: any, b: any) =>
        (a.loadScore + a.distance) - (b.loadScore + b.distance)
      )[0];

    return {
      destination: bestER.name,
      type: "ER",
      reason: "Emergency with optimal load + distance"
    };
  }

  if (patient.riskLevel === "high") {
    const clinic = facilities
      .filter((f: any) => f.type === "CLINIC" && f.loadScore < 0.8)
      .sort((a: any, b: any) => a.distance - b.distance)[0];

    return {
      destination: clinic?.name ?? "Fallback ER",
      type: clinic ? "CLINIC" : "ER",
      reason: "High risk but non-emergent"
    };
  }

  return {
    destination: "TELEMED",
    type: "TELEMED",
    reason: "Low risk + system efficiency"
  };
}
🧠 3. ADMISSION RISK ENGINE

Predicts:

“Will this patient end up admitted?”

export function predictAdmissionRisk(p: any) {

  let score = 0;

  if (p.ageYears > 65) score += 2;
  if (p.complaint === "chest_pain") score += 2;
  if (p.vitals?.systolicBp < 100) score += 3;
  if (p.vitals?.oxygenSaturation < 92) score += 3;

  const risk =
    score >= 6 ? "high" :
    score >= 3 ? "medium" : "low";

  return {
    score,
    risk,
    recommendDirectAdmissionPath: risk === "high"
  };
}
🔁 4. BOUNCEBACK PREDICTOR (72-HOUR RISK)
server/regional/bouncebackPredictor.ts
export function predictBounceback(p: any) {

  let score = 0;

  if (p.complaint === "abdominal_pain") score += 2;
  if (p.complaint === "headache") score += 1;
  if (p.ageYears > 70) score += 2;

  const risk =
    score >= 4 ? "high" :
    score >= 2 ? "medium" : "low";

  return {
    score,
    risk,
    needsFollowup: risk !== "low"
  };
}
📞 5. CALLBACK AUTOMATION
server/regional/callbackAutomation.ts
export function buildCallbackPlan(input: any) {

  const { admissionRisk, bouncebackRisk } = input;

  if (admissionRisk.risk === "high") {
    return {
      timing: "2h",
      method: "phone",
      reason: "High admission risk"
    };
  }

  if (bouncebackRisk.risk === "high") {
    return {
      timing: "12h",
      method: "sms",
      reason: "High bounceback risk"
    };
  }

  return {
    timing: "none"
  };
}
🌡️ 6. OUTBREAK DETECTOR
server/regional/outbreakDetector.ts
export function detectRegionalOutbreak(patients: any[]) {

  const counts: Record<string, number> = {};

  for (const p of patients) {
    counts[p.complaint] = (counts[p.complaint] || 0) + 1;
  }

  const clusters = Object.entries(counts)
    .filter(([_, count]) => count > 10);

  return {
    clusters,
    alert: clusters.length > 0
  };
}
