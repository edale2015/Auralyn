export function computeMetrics(results: any[]) {
  let tp = 0, fp = 0, fn = 0, tn = 0;

  for (const r of results) {
    if (r.correct) tp++;
    else {
      fp++;
      fn++;
    }
  }

  const precision = tp / (tp + fp || 1);
  const recall = tp / (tp + fn || 1);
  const f1 = 2 * (precision * recall) / (precision + recall || 1);

  return {
    total: results.length,
    accuracy: tp / results.length,
    precision,
    recall,
    f1
  };
}

const likelihoods = new Map<string, Record<string, number>>();

export function trainBayes(dx: string, feature: string) {
  const entry = likelihoods.get(dx) || {};
  entry[feature] = (entry[feature] || 0) + 1;
  likelihoods.set(dx, entry);
}

export function computeBayesScore(dx: string, features: string[]) {
  const entry = likelihoods.get(dx) || {};
  let score = 1;

  for (const f of features) {
    score *= entry[f] || 0.1;
  }

  return score;
}

function jaccard(a: string[], b: string[]) {
  const A = new Set(a);
  const B = new Set(b);

  const intersection = [...A].filter(x => B.has(x)).length;
  const union = new Set([...a, ...b]).size;

  return intersection / union;
}

export function findSimilar(input: any) {
  return cases
    .map(c => ({
      case: c,
      score: jaccard(c.features, input.features)
    }))
    .sort((a, b) => b.score - a.score)
    .slice(0, 5);
}
