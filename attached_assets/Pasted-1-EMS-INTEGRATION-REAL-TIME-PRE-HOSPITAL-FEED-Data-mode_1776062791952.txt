Distance + ETA helper
// server/network/geo.ts
export function haversineKm(a: any, b: any) {
  const R = 6371;
  const dLat = (b.lat - a.lat) * Math.PI/180;
  const dLon = (b.lon - a.lon) * Math.PI/180;
  const lat1 = a.lat * Math.PI/180;
  const lat2 = b.lat * Math.PI/180;

  const h = Math.sin(dLat/2)**2 +
    Math.cos(lat1)*Math.cos(lat2)*Math.sin(dLon/2)**2;

  return 2 * R * Math.asin(Math.sqrt(h));
}

export function etaMinutesKm(km: number, speedKmh = 40) {
  return Math.round((km / speedKmh) * 60);
}
Smart routing (severity × capacity × distance × capability)
// server/network/router.ts
import { hospitals } from "./hospitalRegistry";
import { haversineKm, etaMinutesKm } from "./geo";

type RouteInput = {
  incidentId: string;
  triage: { sepsisRisk: "LOW"|"MEDIUM"|"HIGH" };
  etaMinutes: number;
  location: { lat: number; lon: number };
  patient: any;
};

export async function routeIncident(input: RouteInput) {
  const scored = Object.values(hospitals).map((h) => {
    const km = haversineKm(input.location, h.location);
    const eta = etaMinutesKm(km);

    let score = 0;

    // Severity weighting
    if (input.triage.sepsisRisk === "HIGH") score += 5;

    // Capacity
    score += (h.capacity.icuAvailable > 0 ? 3 : -3);
    score += (1 - h.capacity.edLoad) * 2;

    // Capability
    if (h.capabilities.sepsis) score += 2;

    // Distance penalty
    score -= Math.min(eta / 10, 3);

    return { hospital: h, score, eta };
  });

  scored.sort((a, b) => b.score - a.score);

  const chosen = scored[0];

  return {
    incidentId: input.incidentId,
    destination: chosen.hospital.id,
    eta: chosen.eta,
    alternatives: scored.slice(1, 3).map(s => s.hospital.id),
  };
}

Reward function (safety-weighted)
// server/rlhf/reward.ts
export function computeReward(o: Outcome) {
  let r = 0;

  // Reward early appropriate ICU use
  if (o.actual.icuAdmission && o.predictedRisk > 0.7) r += 2;

  // Penalize missed severity
  if (o.actual.mortality && o.predictedRisk < 0.5) r -= 5;

  // Reward timely antibiotics
  if (o.actual.timeToAntibioticsMin <= 60) r += 2;
  else r -= 1;

  // Reward physiologic improvement
  if (o.actual.lactateCleared) r += 2;

  return r;
}
Constrained weight updates (±2% guardrail)
// server/rlhf/update.ts
type Weights = {
  sepsisThreshold: number;
  icuPriorityWeight: number;
  distancePenalty: number;
};

export function updateWeights(
  current: Weights,
  gradients: Partial<Weights>
): Weights {
  const cap = 0.02; // 2% safety cap

  const next: Weights = { ...current };

  (Object.keys(gradients) as (keyof Weights)[]).forEach((k) => {
    const delta = gradients[k] || 0;
    const bounded = Math.max(-cap, Math.min(cap, delta));
    next[k] = current[k] * (1 + bounded);
  });

  return next;
}
Batch trainer
// server/rlhf/trainer.ts
import { computeReward } from "./reward";
import { updateWeights } from "./update";

export async function trainRLHF(
  outcomes: any[],
  weights: any
) {
  let agg = { sepsisThreshold: 0, icuPriorityWeight: 0, distancePenalty: 0 };

  for (const o of outcomes) {
    const r = computeReward(o);

    // crude gradient signals (replace with better estimators later)
    agg.sepsisThreshold += r * 0.001;
    agg.icuPriorityWeight += r * 0.001;
    agg.distancePenalty -= r * 0.0005;
  }

  return updateWeights(weights, agg);
}

import crypto from "crypto";

export function auditEvent(event: any) {
  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(event))
    .digest("hex");

  return { ...event, hash, ts: Date.now() };
