🧩 2. Build the Agent Scope Engine (CODE)
🔧 /server/scope/agentScopeEngine.ts
type ScopeRule = {
  role: string;
  express: string[];
  implied?: string[];
  denied?: string[];
  constraints?: Record<string, any>;
};

type ActionRequest = {
  agentRole: string;
  action: string;
  context: Record<string, any>;
};

export class AgentScopeEngine {
  private rules: Map<string, ScopeRule>;

  constructor(scopeRules: ScopeRule[]) {
    this.rules = new Map(scopeRules.map(r => [r.role, r]));
  }

  evaluate(request: ActionRequest): {
    allowed: boolean;
    reason?: string;
    requiresOverride?: boolean;
  } {
    const rule = this.rules.get(request.agentRole);
    if (!rule) return { allowed: false, reason: "No scope defined" };

    // 🚫 DENIED
    if (rule.denied?.includes(request.action)) {
      return { allowed: false, reason: "Explicitly denied" };
    }

    // ✅ EXPRESS
    if (rule.express.includes(request.action)) {
      return this.checkConstraints(rule, request);
    }

    // ⚠️ IMPLIED
    if (rule.implied?.includes(request.action)) {
      return this.checkConstraints(rule, request);
    }

    return { allowed: false, reason: "Outside scope" };
  }

  private checkConstraints(rule: ScopeRule, request: ActionRequest) {
    if (!rule.constraints) return { allowed: true };

    if (rule.constraints.requires?.includes("physician_signed")) {
      if (!request.context.physicianSigned) {
        return {
          allowed: false,
          requiresOverride: true,
          reason: "Physician signature required"
        };
      }
    }

    if (rule.constraints.requires?.includes("confidence > 0.9")) {
      if ((request.context.confidence || 0) < 0.9) {
        return {
          allowed: false,
          reason: "Confidence too low"
        };
      }
    }

    return { allowed: true };
  }
}

🔌 Example: Before ANY action
/server/execution/actionGuard.ts
import { AgentScopeEngine } from "../scope/agentScopeEngine";

export const scopeEngine = new AgentScopeEngine(scopeRules);

export function guardAction(request: ActionRequest) {
  const result = scopeEngine.evaluate(request);

  if (!result.allowed) {
    throw new Error(
      `🚫 Scope violation: ${request.action} | Reason: ${result.reason}`
    );
  }

  return true;
}

🚨 5. Scope Creep Detection Engine (HIGH VALUE)

The article’s biggest warning:

“Scope creep is the biggest risk”

Build detection:
/server/monitoring/scopeDrift.ts
export function detectScopeDrift(actionLogs: any[]) {
  const violations = actionLogs.filter(log =>
    log.actionOutsideScope === true
  );

  const expansion = actionLogs.filter(log =>
    log.newPermissionGranted === true
  );

  return {
    violations,
    expansion,
    riskLevel:
      violations.length > 0 || expansion.length > 5
        ? "HIGH"
        : "LOW"
  };
}
