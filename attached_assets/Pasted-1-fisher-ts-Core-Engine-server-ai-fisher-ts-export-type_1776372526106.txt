1. fisher.ts — Core Engine
// server/ai/fisher.ts

export type ProbDist = Record<string, number>;

/**
 * Diagonal Fisher Information Matrix approximation
 * F_i = E[(∂ log p / ∂θ_i)^2]
 */
export function computeDiagonalFisher(
  probs: ProbDist,
  gradients: ProbDist
): ProbDist {
  const fisher: ProbDist = {};

  for (const key in probs) {
    const p = probs[key] || 1e-8;
    const grad = gradients[key] || 0;

    // Fisher diagonal approximation
    fisher[key] = (grad * grad) / p;
  }

  return fisher;
}
2. naturalGradient.ts — Geometry-Aware Update
// server/ai/naturalGradient.ts

import { ProbDist } from "./fisher";

/**
 * Natural gradient step:
 * θ_new = θ + η * F⁻¹ ∇L
 */
export function naturalGradientStep(
  params: ProbDist,
  gradients: ProbDist,
  fisherDiag: ProbDist,
  learningRate = 0.1
): ProbDist {
  const updated: ProbDist = {};

  for (const key in params) {
    const grad = gradients[key] || 0;
    const fisher = fisherDiag[key] || 1e-6;

    // Natural gradient = grad / Fisher
    const natGrad = grad / fisher;

    updated[key] = params[key] + learningRate * natGrad;
  }

  return normalize(updated);
}

function normalize(dist: ProbDist): ProbDist {
  const sum = Object.values(dist).reduce((a, b) => a + b, 0);
  const out: ProbDist = {};

  for (const k in dist) {
    out[k] = dist[k] / (sum || 1);
  }

  return out;
}
3. bayesianUpdater.ts — Clinical Belief Engine
// server/ai/bayesianUpdater.ts

import { computeDiagonalFisher } from "./fisher";
import { naturalGradientStep } from "./naturalGradient";

export type Observation = {
  feature: string;
  value: number;
};

export function updateBeliefsWithFisher(
  prior: Record<string, number>,
  likelihoods: Record<string, number>,
  observations: Observation[]
) {
  const gradients: Record<string, number> = {};

  // Compute gradient of log-likelihood
  for (const dx in prior) {
    let grad = 0;

    for (const obs of observations) {
      const likelihood = likelihoods[dx] || 1e-6;
      grad += Math.log(likelihood + 1e-8) * obs.value;
    }

    gradients[dx] = grad;
  }

  const fisher = computeDiagonalFisher(prior, gradients);

  const posterior = naturalGradientStep(
    prior,
    gradients,
    fisher,
    0.05
  );

  return {
    posterior,
    fisher,
    gradients,
  };
}
4. Integration into Your Medical Engine
// server/clinical/diagnosisEngine.ts

import { updateBeliefsWithFisher } from "../ai/bayesianUpdater";

export function runDiagnosisStep(state: any, input: any) {
  const { prior, likelihoods } = state;

  const observations = extractObservations(input);

  const { posterior, fisher } = updateBeliefsWithFisher(
    prior,
    likelihoods,
    observations
  );

  return {
    ...state,
    posterior,
    fisherImportance: fisher,
  };
}

function extractObservations(input: any) {
  const obs = [];

  if (input.fever) obs.push({ feature: "fever", value: 1 });
  if (input.cough) obs.push({ feature: "cough", value: 1 });
  if (input.chestPain) obs.push({ feature: "chestPain", value: 1 });

  return obs;
}
