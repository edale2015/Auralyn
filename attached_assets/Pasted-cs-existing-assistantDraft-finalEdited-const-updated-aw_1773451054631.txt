import { sql } from "drizzle-orm";
import { db } from "../db";
import { replyDrafts } from "../db/schema.replyDrafts";
import { replyFeedback } from "../db/schema.replyFeedback";

export type TimeBucket = "day" | "week";

function bucketExpr(column: any, bucket: TimeBucket) {
  if (bucket === "week") {
    return sql`date_trunc('week', ${column})`;
  }
  return sql`date_trunc('day', ${column})`;
}

export async function getDraftTimeSeries(input?: {
  bucket?: TimeBucket;
  physicianId?: string;
  daysBack?: number;
}) {
  const bucket = input?.bucket || "day";
  const daysBack = input?.daysBack || 30;

  const rows = await db.execute(sql`
    select
      ${bucketExpr(replyDrafts.createdAt, bucket)} as bucket,
      count(*)::int as total,
      count(case when ${replyDrafts.blocked} = true then 1 end)::int as blocked,
      count(case when ${replyDrafts.status} = 'approved' then 1 end)::int as approved,
      count(case when ${replyDrafts.status} = 'sent' then 1 end)::int as sent,
      avg(${replyDrafts.confidence})::float as avg_confidence
    from ${replyDrafts}
    where ${replyDrafts.createdAt} >= now() - (${daysBack} || ' days')::interval
      ${input?.physicianId ? sql`and ${replyDrafts.physicianId} = ${input.physicianId}` : sql``}
    group by 1
    order by 1 asc
  `);

  return rows.rows;
}
