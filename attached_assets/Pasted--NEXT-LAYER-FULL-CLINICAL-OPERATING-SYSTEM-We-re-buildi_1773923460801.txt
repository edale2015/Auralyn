import React, { useEffect, useState } from "react";
import axios from "axios";

export default function PhysicianAnalytics() {
  const [stats, setStats] = useState<any>({});

  useEffect(() => {
    loadStats();
  }, []);

  const loadStats = async () => {
    const res = await axios.get("/api/analytics/summary");
    setStats(res.data);
  };

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold">Clinical Performance Dashboard</h1>

      <div className="grid grid-cols-4 gap-4 mt-4">
        <Card title="Total Cases" value={stats.totalCases} />
        <Card title="Accuracy %" value={stats.accuracy} />
        <Card title="Physician Overrides" value={stats.overrides} />
        <Card title="Avg Confidence" value={stats.avgConfidence} />
      </div>
    </div>
  );
}

function Card({ title, value }: any) {
  return (
    <div className="border p-4 rounded shadow">
      <h2 className="text-sm">{title}</h2>
      <p className="text-xl font-bold">{value}</p>
    </div>
  );
}
📁 /server/routes/analytics.ts
import express from "express";
const router = express.Router();

let logs: any[] = [];

router.get("/summary", (req, res) => {
  const total = logs.length;
  const correct = logs.filter(l => l.correct).length;

  res.json({
    totalCases: total,
    accuracy: total ? (correct / total * 100).toFixed(2) : 0,
    overrides: logs.filter(l => l.override).length,
    avgConfidence:
      total
        ? logs.reduce((a, b) => a + b.confidence, 0) / total
        : 0,
  });
});

export default router;

import { runAutonomousCase } from "../pipeline/autonomous";

const complaints = ["cough", "fever", "chest pain", "sore throat"];

export async function runSimulation(batch = 10000) {
  let results = [];

  for (let i = 0; i < batch; i++) {
    const input = {
      text: generateCase(),
    };

    const res = await runAutonomousCase(input);
    results.push(res);
  }

  return results;
}

function generateCase() {
  const complaint = complaints[Math.floor(Math.random() * complaints.length)];

  return `${complaint} for ${Math.floor(Math.random() * 10)} days`;
}

import axios from "axios";

export async function sendToEHR(caseData: any) {
  const payload = {
    resourceType: "Encounter",
    status: "finished",
    subject: {
      reference: `Patient/${caseData.patientId}`,
    },
    reasonCode: [
      {
        text: caseData.complaint,
      },
    ],
  };

  return axios.post(
    "https://your-ehr-endpoint.com/fhir/Encounter",
    payload,
    {
      headers: {
        Authorization: `Bearer ${process.env.EHR_TOKEN}`,
      },
    }
  );
}

import React, { useState } from "react";
import axios from "axios";

export default function PatientPortal() {
  const [text, setText] = useState("");
  const [response, setResponse] = useState("");

  const submit = async () => {
    const res = await axios.post("/api/patient/submit", { text });
    setResponse(res.data.reply);
  };

  return (
    <div className="p-6">
      <h1>AI Clinic</h1>

      <textarea
        className="border w-full p-2"
        placeholder="Describe your symptoms..."
        onChange={(e) => setText(e.target.value)}
      />

      <button className="mt-2 bg-blue-500 text-white px-4 py-2" onClick={submit}>
        Submit
      </button>

      <div className="mt-4">{response}</div>
    </div>
  );
}

const auth = new google.auth.GoogleAuth({
  keyFile: "credentials.json",
  scopes: ["https://www.googleapis.com/auth/spreadsheets"],
});

export async function logCase(data: any) {
  const sheets = google.sheets({ version: "v4", auth });

  await sheets.spreadsheets.values.append({
    spreadsheetId: process.env.SHEET_ID,
    range: "Cases!A1",
    valueInputOption: "RAW",
    requestBody: {
      values: [[
        new Date().toISOString(),
        data.complaint,
        data.diagnosis,
        data.confidence
      ]],
    },
  });
}

import fs from "fs";
import { runAutonomousCase } from "../pipeline/autonomous";

export async function runTests() {
  const cases = JSON.parse(fs.readFileSync("real_cases.json", "utf-8"));

  let results = [];

  for (const c of cases) {
    const res = await runAutonomousCase(c);
    results.push({
      input: c,
      output: res,
    });
  }

  fs.writeFileSync("results.json", JSON.stringify(results, null, 2));
}

        price_data: {

        },
        quantity: 1,
      },
    ],
    success_url: "http://localhost:3000/success",
    cancel_url: "http://localhost:3000/cancel",
  });

  res.json({ url: session.url });
});

export default router;
