🧠 5. AUTONOMOUS ACTION ORCHESTRATOR
/server/intervention/actionOrchestrator.ts
import { generateInterventions } from "./interventionEngine";
import { executeOrder } from "./orderExecutor";
import { sendAlert } from "./alertEngine";
import { escalatePatient } from "./escalationEngine";

export const runInterventions = async (patient: any, triage: any) => {
  const actions = generateInterventions(triage);

  for (const action of actions) {
    switch (action.type) {
      case "lab":
      case "med":
        await executeOrder(action.action, patient.id);
        break;

      case "alert":
        await sendAlert(action.action, action.priority);
        break;

      case "escalation":
        await escalatePatient(patient);
        break;
    }
  }

  return actions;
};

type Patient = {
  id: string;
  symptoms: string;
  riskScore: number;
  vitals: {
    hr: number;
    bpSys: number;
    spo2: number;
    temp: number;
  };
  trend?: {
    hrTrend?: number;     // + means rising
    spo2Trend?: number;   // - means dropping
  };
};

export const computePriorityScore = (p: Patient) => {
  let score = 0;

  // Base triage risk
  score += p.riskScore * 2;

  // Vital instability
  if (p.vitals.hr > 120) score += 3;
  if (p.vitals.spo2 < 92) score += 5;
  if (p.vitals.bpSys < 90) score += 5;
  if (p.vitals.temp > 39) score += 2;

  // Trend deterioration (THIS IS HUGE)
  if (p.trend?.spo2Trend && p.trend.spo2Trend < -2) score += 6;
  if (p.trend?.hrTrend && p.trend.hrTrend > 10) score += 4;

  return score;
};

export const rankPatientsAI = (patients: Patient[]) => {
  return patients
    .map(p => ({
      ...p,
      priorityScore: computePriorityScore(p)
    }))
    .sort((a, b) => b.priorityScore - a.priorityScore);
};

type DeteriorationResult = {
  risk: "low" | "moderate" | "high" | "critical";
  flags: string[];
};

export const predictDeterioration = (patient: any): DeteriorationResult => {
  let score = 0;
  const flags: string[] = [];

  // Sepsis indicators
  if (patient.vitals.hr > 100) {
    score += 2;
    flags.push("tachycardia");
  }

  if (patient.vitals.temp > 38 || patient.vitals.temp < 36) {
    score += 2;
    flags.push("abnormal temp");
  }

  if (patient.vitals.bpSys < 100) {
    score += 3;
    flags.push("hypotension");
  }

  if (patient.vitals.spo2 < 92) {
    score += 3;
    flags.push("hypoxia");
  }

  // Trend-based early detection (critical)
  if (patient.trend?.bpTrend < -10) {
    score += 4;
    flags.push("dropping BP");
  }

  if (patient.trend?.spo2Trend < -3) {
    score += 4;
    flags.push("rapid oxygen decline");
  }

  // Risk classification
  if (score >= 10) return { risk: "critical", flags };
  if (score >= 7) return { risk: "high", flags };
  if (score >= 4) return { risk: "moderate", flags };

  return { risk: "low", flags };
};

type CaseOutcome = {
  patientId: string;
  predictedDisposition: string;
  actualDisposition: string;
  diagnosis: string;
  outcome: "improved" | "worsened" | "hospitalized";
};

export const evaluateCase = (caseData: CaseOutcome) => {
  let reward = 0;

  // Correct disposition
  if (caseData.predictedDisposition === caseData.actualDisposition) {
    reward += 1;
  } else {
    reward -= 1;
  }

  // Outcome-based reward
  if (caseData.outcome === "improved") reward += 1;
  if (caseData.outcome === "worsened") reward -= 2;
  if (caseData.outcome === "hospitalized") reward -= 3;

  return reward;
};
