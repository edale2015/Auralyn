🔧 Circuit breaker (generic)
export class CircuitBreaker {
  private failures = 0;
  private open = false;
  private lastTrip = 0;

  constructor(private threshold = 5, private cooldownMs = 30_000) {}

  canRun() {
    if (!this.open) return true;
    if (Date.now() - this.lastTrip > this.cooldownMs) {
      this.open = false; this.failures = 0;
      return true;
    }
    return false;
  }

  success() { this.failures = 0; }

  fail() {
    this.failures++;
    if (this.failures >= this.threshold) {
      this.open = true;
      this.lastTrip = Date.now();
    }
  }
}
🔧 LRU cache for hot paths
export class LRU<K, V> {
  private map = new Map<K, V>();
  constructor(private max = 1000) {}

  get(k: K) {
    if (!this.map.has(k)) return undefined;
    const v = this.map.get(k)!;
    this.map.delete(k); this.map.set(k, v);
    return v;
  }

  set(k: K, v: V) {
    if (this.map.has(k)) this.map.delete(k);
    this.map.set(k, v);
    if (this.map.size > this.max) {
      const first = this.map.keys().next().value;
      this.map.delete(first);
    }
  }
}

const W = { age: 0.02, sbp: -0.03, spo2: -0.08, hr: 0.02, rr: 0.04, chestPain: 0.7, bias: -1.2 };

export function predictAdmission(features: any) {
  const z =
    W.bias +
    W.age * features.age +
    W.sbp * features.sbp +
    W.spo2 * features.spo2 +
    W.hr * features.hr +
    W.rr * features.rr +
    W.chestPain * features.chestPain;

  const p = 1 / (1 + Math.exp(-z));
  return { probability: p, risk: p > 0.7 ? "high" : p > 0.4 ? "medium" : "low" };
}

🔧 Drift monitor
export function dataDrift(baseline: any[], current: any[]) {
  const mean = (arr: number[]) => arr.reduce((a,b)=>a+b,0)/Math.max(1,arr.length);
  const b = mean(baseline.map(x=>x.spo2));
  const c = mean(current.map(x=>x.spo2));
  return { drift: Math.abs(c - b) > 3 };
}

export async function createEncounterFHIR(patientId: string, token: string) {
  const body = {
    resourceType: "Encounter",
    status: "in-progress",
    subject: { reference: `Patient/${patientId}` }
  };

  const res = await fetch(`${process.env.FHIR_BASE}/Encounter`, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${token}`,
      "Content-Type": "application/fhir+json"
    },
    body: JSON.stringify(body)
  });

  if (!res.ok) throw new Error("FHIR encounter failed");
  return res.json();
}

export function nextQuestion(differential: any[], questions: any[]) {
  let best = null, bestScore = -Infinity;
  for (const q of questions) {
    const ig = differential.reduce((s,d)=>s + d.probability * (q.weight ?? 1), 0);
    if (ig > bestScore) { bestScore = ig; best = q; }
  }
  return best;
}

export async function getSecret(name: string) {
  if (cache[name]) return cache[name];
  const res = await fetch(process.env.SECRETS_ENDPOINT! + `/secret/${name}`);

  cache[name] = value;

}

export function idempotent(key: string) {
  if (seen.has(key)) return false;
  seen.add(key);
  setTimeout(()=>seen.delete(key), 60_000);
  return true;
}
