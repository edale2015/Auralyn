// server/engine/hospitalRouting.ts
function distance(a, b) {
  return Math.sqrt((a.lat - b.lat)**2 + (a.lon - b.lon)**2)
}

export async function selectHospital(db, patient, neededService) {
  const hospitals = await db.kb_hospitals.findAll({ is_active: true })

  return hospitals
    .filter(h => h.services.includes(neededService))
    .map(h => ({
      ...h,
      dist: distance(patient.location, h)
    }))
    .sort((a,b)=>a.dist-b.dist)[0]
}
🎛️ UI
export default function HospitalRouting({ hospital }) {

      <p>Distance: {hospital.dist.toFixed(2)}</p>
    </div>
  )
}

export function computeETA(unit, patient) {
  const dist = Math.sqrt((unit.lat-patient.lat)**2 + (unit.lon-patient.lon)**2)
  return dist * 3 // rough minutes
}
