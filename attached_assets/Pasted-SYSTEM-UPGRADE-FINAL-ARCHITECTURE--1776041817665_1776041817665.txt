CREATE TABLE clinical_documents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT,
  source TEXT,
  uploaded_at TIMESTAMP DEFAULT now()
);

CREATE TABLE document_nodes (
  id TEXT PRIMARY KEY,
  document_id UUID REFERENCES clinical_documents(id),
  title TEXT,
  start_page INT,
  end_page INT,
  summary TEXT,
  content TEXT
);

CREATE TABLE reasoning_queries (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  document_id UUID,
  question TEXT,
  selected_node TEXT,
  answer TEXT,
  confidence FLOAT,
  created_at TIMESTAMP DEFAULT now()
);

CREATE TABLE cross_reference_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  query_id UUID,
  reference TEXT,
  resolved_node TEXT
);
⚙️ 2. KB INGESTION PIPELINE (AUTO → TREE → DB)
/server/kb/ingestClinicalDoc.ts
import { PageIndexBuilder } from "../clinical_reasoning/pageIndexBuilder";
import db from "../db";

export async function ingestClinicalDoc(pdfPath: string, title: string) {
  const builder = new PageIndexBuilder();
  const tree = await builder.buildTreeFromPDF(pdfPath);

  const doc = await db("clinical_documents")
    .insert({ title })
    .returning("*");

  for (const node of tree) {
    await db("document_nodes").insert({
      id: node.node_id,
      document_id: doc[0].id,
      title: node.title,
      start_page: node.start_page,
      end_page: node.end_page,
      summary: node.summary,
      content: node.content
    });
  }

  return doc[0].id;
}
🧠 3. REASONING ENGINE → KB + LOGGING
/server/clinical_reasoning/clinicalDocEngine.ts
import db from "../db";
import { ReasoningRetriever } from "./reasoningRetriever";
import { CrossReferenceNavigator } from "./crossReferenceNavigator";

export class ClinicalDocEngine {
  retriever = new ReasoningRetriever();
  navigator = new CrossReferenceNavigator();

  async answer(documentId: string, question: string) {
    const nodes = await db("document_nodes")
      .where({ document_id: documentId });

    const nodeId = await this.retriever.findRelevantNode(nodes, question);
    const node = nodes.find(n => n.id === nodeId);

    if (!node) throw new Error("Node not found");

    let answer = await this.retriever.extractAnswer(node.content, question);

    const queryLog = await db("reasoning_queries")
      .insert({
        document_id: documentId,
        question,
        selected_node: nodeId,
        answer,
        confidence: 0.85
      })
      .returning("*");

    const refs = this.navigator.findReferences(node.content);

    for (const ref of refs) {
      const refNode = nodes.find(n =>
        n.title.toLowerCase().includes(ref.toLowerCase())
      );

      if (refNode) {
        await db("cross_reference_logs").insert({
          query_id: queryLog[0].id,
          reference: ref,
          resolved_node: refNode.id
        });

        const extra = await this.retriever.extractAnswer(refNode.content, question);
        answer += "\n\n[Cross Reference]\n" + extra;
      }
    }

    return { answer, nodeId };
  }
}
🧪 4. GOLDEN CASE GENERATOR (AUTO FROM DOCS)
/server/goldenCases/generator.ts
import db from "../db";
import OpenAI from "openai";

const openai = new OpenAI();

export async function generateGoldenCases(documentId: string) {
  const nodes = await db("document_nodes")
    .where({ document_id: documentId });

  for (const node of nodes) {
    const prompt = `
Generate 3 clinical cases based on this guideline section.

${node.content}

Return JSON:
[
 { "case": "...", "expected_answer": "...", "severity": "low|high" }
]
`;

    const res = await openai.chat.completions.create({
      model: "gpt-4o",
      messages: [{ role: "user", content: prompt }]
    });

    const cases = JSON.parse(res.choices[0].message.content || "[]");

    for (const c of cases) {
      await db("golden_cases").insert({
        complaint: node.title,
        input: c.case,
        expected_output: c.expected_answer,
        severity: c.severity
      });
    }
  }
}

📡 6. WEBSOCKET LIVE STREAM (CONTROL TOWER)
/server/ws/reasoningStream.ts
import { Server } from "ws";

export function initReasoningStream(server: any) {
  const wss = new Server({ server });

  wss.on("connection", (ws) => {
    ws.send(JSON.stringify({ type: "connected" }));

    setInterval(async () => {
      const latest = await getLatestQueries();

      ws.send(JSON.stringify({
        type: "reasoning_update",
        data: latest
      }));
    }, 2000);
  });
}

async function getLatestQueries() {
  return await db("reasoning_queries")
    .orderBy("created_at", "desc")
    .limit(10);
}

import { useEffect, useState } from "react";

export default function ClinicalReasoningPanel() {
  const [logs, setLogs] = useState<any[]>([]);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:3000");

    ws.onmessage = (msg) => {
      const data = JSON.parse(msg.data);

      if (data.type === "reasoning_update") {
        setLogs(data.data);
      }
    };
  }, []);

  return (
    <div className="p-4">
      <h2>🧠 Clinical Reasoning Engine</h2>

      {logs.map((l) => (
        <div key={l.id} className="border p-2 my-2">
          <div><b>Q:</b> {l.question}</div>
          <div><b>Node:</b> {l.selected_node}</div>
          <div><b>Answer:</b> {l.answer}</div>
        </div>
      ))}
    </div>
  );
}
