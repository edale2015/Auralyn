import fetch from "node-fetch";

const CLAUDE_API = "https://api.anthropic.com/v1/messages";

export async function runClaudeReview(prompt: string, content: string) {
  const res = await fetch(CLAUDE_API, {
    method: "POST",
    headers: {
      "x-api-key": process.env.CLAUDE_API_KEY!,
      "content-type": "application/json",
      "anthropic-version": "2023-06-01"
    },
    body: JSON.stringify({
      model: "claude-3-opus-20240229",
      max_tokens: 2000,
      messages: [
        {
          role: "user",
          content: `${prompt}\n\n${content}`
        }
      ]
    })
  });

  const json = await res.json();
  return json.content?.[0]?.text || "";
}

import fs from "fs";
import path from "path";
import { runClaudeReview } from "./claudeClient";
import { parseFindings } from "./reviewParser";
import { saveFindings } from "./reviewStore";

export async function runFullClaudeReview(exportDir: string) {
  const files = fs.readdirSync(exportDir).filter(f => f.endsWith(".md"));

  for (const file of files) {
    const content = fs.readFileSync(path.join(exportDir, file), "utf8");

    const prompt = `
You are reviewing a medical triage system.
List the TOP 5 MOST DANGEROUS FAILURE MODES.
Be specific and clinical.
`;

    const response = await runClaudeReview(prompt, content);

    const findings = parseFindings(response);

    await saveFindings(file, findings);
  }
}
📁 server/review/reviewParser.ts
export function parseFindings(text: string) {
  return text.split("\n")
    .filter(line => line.trim().length > 10)
    .map((line, i) => ({
      id: i,
      description: line.trim(),
      severity: classifySeverity(line),
    }));
}

function classifySeverity(line: string) {
  const l = line.toLowerCase();

  if (l.includes("death") || l.includes("missed")) return "CRITICAL";
  if (l.includes("unsafe")) return "HIGH";
  if (l.includes("uncertain")) return "MEDIUM";

  return "LOW";
}

export async function getFailureDashboard(db: any) {
  const rows = await db.execute(`SELECT * FROM review_findings`);

  const summary: any = {
    CRITICAL: 0,
    HIGH: 0,
    MEDIUM: 0,
    LOW: 0
  };

  rows.rows.forEach((r: any) => {
    const findings = r.findings;

    findings.forEach((f: any) => {
      summary[f.severity]++;
    });
  });

  return summary;
}

export function detectRegressions(prev: any, current: any) {
  return {
    unsafeIncrease: current.unsafeUndercalls > prev.unsafeUndercalls,
    passRateDrop: current.passRate < prev.passRate,
    calibrationWorsened: current.calibrationError > prev.calibrationError
  };
}
📁 Alert system
export function buildRegressionAlerts(reg: any) {
  const alerts = [];

  if (reg.unsafeIncrease) {
    alerts.push("🚨 Unsafe undercalls increased");
  }

  if (reg.passRateDrop) {
    alerts.push("⚠ Pass rate dropped");
  }

  if (reg.calibrationWorsened) {
    alerts.push("⚠ Calibration worsened");
  }

  return alerts;
}
