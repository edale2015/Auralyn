// server/validation/validationGate.ts

type ValidationSummary = {
  passRate: number;
  unsafeUndercalls: number;
  calibrationError: number;
};

export function enforceValidationGate(summary: ValidationSummary) {
  const errors: string[] = [];

  if (summary.unsafeUndercalls > 0) {
    errors.push("Unsafe undercalls detected");
  }

  if (summary.passRate < 0.85) {
    errors.push("Validation pass rate too low");
  }

  if (summary.calibrationError > 0.15) {
    errors.push("Model overconfidence detected");
  }

  return {
    allowed: errors.length === 0,
    errors,
  };
}

// server/validation/validationRunner.ts

import { runGoldenCases, summarizeValidation } from './goldenCaseHarness';
import { expandAdversarialSet } from './adversarialGenerator';
import { computeBrierScore } from './calibrationMonitor';

export async function runFullValidation(cases: any[]) {
  const expanded = expandAdversarialSet(cases);

  const results = await runGoldenCases(expanded);
  const summary = summarizeValidation(results);

  const calibrationRows = results.map(r => ({
    predictedConfidence: 0.7, // plug real value
    correct: r.pass,
  }));

  const calibrationError = computeBrierScore(calibrationRows);

  return {
    ...summary,
    calibrationError,
  };
}

export function detectDrift(previous: any, current: any) {

export function detectImpossibleCombo(observations: any[]) {
  const features = new Set(observations.map(o => o.feature));

  if (features.has("hypotension") && features.has("normal_perfusion")) {
    return {
      blocked: true,
      reason: "Impossible physiologic combination"
    };
  }

  return { blocked: false };
}

🧬 3. Multi-Diagnosis Coverage Requirement
export function requireDifferentialSpread(posterior: any) {
  const top3 = Object.values(posterior)
    .sort((a: any, b: any) => b - a)
    .slice(0, 3);

  const spread = top3[0] - top3[2];

  if (spread < 0.2) {
    return {
      forceReview: true,
      reason: "Diagnosis uncertainty too high"
    };
  }

  return { forceReview: false };
}

export function ensureDangerousRuledOut(topDx: string, posterior: any) {
  const dangerous = ["pe", "acs", "stroke", "sepsis"];

  for (const dx of dangerous) {
    if ((posterior[dx] ?? 0) > 0.1 && topDx !== dx) {
      return {
        blocked: true,
        reason: `Dangerous condition ${dx} not ruled out`
      };
    }
  }

  return { blocked: false };
}
🧠 5. Consistency Across Time (MULTI-STEP MEMORY)
export function checkConsistency(history: any[], current: any) {
  if (!history.length) return { ok: true };

  const last = history[history.length - 1];

  if (last.topDiagnosis !== current.topDiagnosis &&
      Math.abs(last.confidence - current.confidence) < 0.1) {
    return {
      flag: true,
      reason: "Diagnosis flipped without evidence change"
    };
  }

  return { ok: true };
}
