type Physician = {
  id: string;
  skills: string[]; // ["derm", "cardio"]
  activeCases: number;
  maxCapacity: number;
  avgResponseTime: number;
  performanceScore: number; // 0–1
};

type Case = {
  caseId: string;
  complaint: string;
  riskScore: number;
};

const physicians: Physician[] = [];

export function registerPhysician(p: Physician) {
  physicians.push(p);
}

export function selectPhysician(c: Case): Physician | null {
  const candidates = physicians.filter(p =>
    p.skills.includes(c.complaint) &&
    p.activeCases < p.maxCapacity
  );

  if (!candidates.length) return null;

  const scored = candidates.map(p => {
    const loadFactor = p.activeCases / p.maxCapacity;
    const riskWeight = c.riskScore > 0.5 ? 1.5 : 1;

    const score =
      p.performanceScore * 0.5 +
      (1 - loadFactor) * 0.3 +
      (1 / (p.avgResponseTime + 1)) * 0.2;

    return { physician: p, score: score * riskWeight };
  });

  scored.sort((a, b) => b.score - a.score);

  return scored[0].physician;
}

type Thresholds = {
  escalate: number;
  urgent: number;
  routine: number;
};

let thresholds: Thresholds = {
  escalate: 0.7,
  urgent: 0.4,
  routine: 0.1
};

let history: any[] = [];

export function recordTriageOutcome(data: {
  predictedRisk: number;
  actualSeverity: number;
}) {
  history.push(data);
}

export function optimizeThresholds() {
  if (history.length < 50) return thresholds;

  let error = 0;

  for (const h of history) {
    error += Math.abs(h.predictedRisk - h.actualSeverity);
  }

  const avgError = error / history.length;

  // adjust thresholds
  if (avgError > 0.2) {
    thresholds.escalate -= 0.05;
    thresholds.urgent -= 0.03;
  } else {
    thresholds.escalate += 0.02;
  }

  // clamp
  thresholds.escalate = Math.max(0.5, Math.min(0.9, thresholds.escalate));

  return thresholds;
}

type RiskSignal = {
  caseId: string;
  latencyMs: number;
  errorRate: number;
  overrideRate: number;
  riskScore: number;
  complaint: string;
  redFlags: number;
};

export function computeSystemRisk(signal: RiskSignal) {
  let risk = 0;

  if (signal.latencyMs > 2000) risk += 0.1;
  if (signal.errorRate > 0.05) risk += 0.2;
  if (signal.overrideRate > 0.2) risk += 0.3;
  if (signal.redFlags > 0) risk += 0.3;

  risk += signal.riskScore * 0.5;

  return Math.min(1, risk);
}
📁 server/risk/malpracticeDetector.ts
export function detectMalpracticeRisk(data: {
  diagnosis: string;
  redFlags: string[];
  disposition: string;
  protocolDeviation: boolean;
}) {
  let risk = 0;

  if (data.redFlags.length > 0 && data.disposition !== "ER") {
    risk += 0.5;
  }

  if (data.protocolDeviation) {
    risk += 0.3;
  }

  if (data.diagnosis === "uncertain") {
    risk += 0.2;
  }

  return {
    malpracticeRisk: Math.min(1, risk),
    level:
      risk > 0.7 ? "CRITICAL" :
      risk > 0.4 ? "HIGH" :
      "LOW"
  };
}

📁 server/revenue/reimbursementOptimizer.ts
import { getAllPayers } from "./payerRegistry";

export function optimizeBilling(caseData: {
  diagnosis: string;
  cptOptions: string[];
}) {
  const payers = getAllPayers();

  let best: any = null;

  for (const payer of payers) {
    for (const cpt of caseData.cptOptions) {
      const rate = payer.reimbursementRates[cpt] || 0;

      const adjusted =
        rate *
        (1 - payer.denialRate) *
        (1 / payer.avgDaysToPay);

      if (!best || adjusted > best.score) {
        best = {
          payer: payer.name,
          cpt,
          expectedRevenue: rate,
          score: adjusted
        };
      }
    }
  }

  return best;
}
