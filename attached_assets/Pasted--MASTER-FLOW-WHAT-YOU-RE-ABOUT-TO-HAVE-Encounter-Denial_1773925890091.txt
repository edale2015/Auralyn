📁 server/billing/denialLearningEngine.ts
type ClaimOutcome = {
  encounterId: string;
  denied: boolean;
  reasons: string[];
  coding: any;
};

export class DenialLearningEngine {
  private weights: Record<string, number> = {};

  update(outcome: ClaimOutcome) {
    const key = this.generateKey(outcome.coding);

    if (!this.weights[key]) this.weights[key] = 0.5;

    if (outcome.denied) {
      this.weights[key] += 0.1;
    } else {
      this.weights[key] -= 0.05;
    }

    this.weights[key] = Math.max(0, Math.min(1, this.weights[key]));
  }

  predict(coding: any): number {
    const key = this.generateKey(coding);
    return this.weights[key] || 0.3;
  }

  private generateKey(coding: any) {
    const icd = coding.icd10?.[0]?.code || "none";
    const cpt = coding.cpt?.[0]?.code || "none";
    return `${icd}_${cpt}`;
  }
}

export class AutoFixEngine {
  fix(encounter: any, denialPrediction: any) {
    const fixes = [];

    if (denialPrediction.reasons.includes("Missing clinical reasoning")) {
      encounter.clinical.reasoning = "Auto-generated reasoning based on cluster analysis.";
      fixes.push("Added clinical reasoning");
    }

    if (denialPrediction.reasons.includes("CPT level too high for complexity")) {
      encounter.coding.cpt = [{ code: "99213" }];
      fixes.push("Downgraded CPT");
    }

    if (denialPrediction.reasons.includes("Missing ICD-10 codes")) {
      encounter.coding.icd10 = [{ code: "R69", description: "Fallback diagnosis" }];
      fixes.push("Added fallback ICD");
    }

    return {
      encounter,
      fixes
    };
  }
}
🔥 PART 3 — SMART PHYSICIAN ROUTING ENGINE

Routes cases like a hospital triage board.

📁 server/review/smartRoutingEngine.ts
type Physician = {
  id: string;
  specialty: string;
  load: number;
};

export class SmartRoutingEngine {
  route(encounter: any, physicians: Physician[]) {
    const specialty = this.detectSpecialty(encounter);

    const candidates = physicians.filter(p => p.specialty === specialty);

    if (candidates.length === 0) return physicians[0];

    return candidates.sort((a, b) => a.load - b.load)[0];
  }

  private detectSpecialty(encounter: any) {
    const code = encounter.coding.icd10?.[0]?.code || "";

    if (code.startsWith("I")) return "cardiology";
    if (code.startsWith("J")) return "pulmonary";
    if (code.startsWith("K")) return "gastro";

    return "general";
  }
}
🔥 PART 4 — REVENUE ANALYTICS ENGINE

This is your financial cockpit 📊

📁 server/analytics/revenueEngine.ts
type EncounterResult = {
  revenue: number;
  denied: boolean;
};

export class RevenueEngine {
  calculateMetrics(encounters: EncounterResult[]) {
    const total = encounters.length;

    const revenue = encounters.reduce((sum, e) => sum + e.revenue, 0);

    const denied = encounters.filter(e => e.denied).length;

    return {
      totalEncounters: total,
      totalRevenue: revenue,
      denialRate: denied / total,
      avgRevenuePerEncounter: revenue / total
    };
  }
}

import { DenialEngine } from "../billing/denialEngine";
import { DenialLearningEngine } from "../billing/denialLearningEngine";
import { AutoFixEngine } from "../billing/autoFixEngine";
import { SmartRoutingEngine } from "../review/smartRoutingEngine";
import { ReviewQueue } from "../review/reviewQueue";

const denial = new DenialEngine();
const learning = new DenialLearningEngine();
const autoFix = new AutoFixEngine();
const router = new SmartRoutingEngine();
const queue = new ReviewQueue();

export async function runIntelligentPipeline(input: any, physicians: any[]) {
  let encounter = input.encounter;

  // Step 1 — Predict denial
  let prediction = denial.predict(encounter);

  // Step 2 — Auto-fix BEFORE human sees it
  const fixResult = autoFix.fix(encounter, prediction);
  encounter = fixResult.encounter;

  // Step 3 — Re-score after fix
  prediction = denial.predict(encounter);

  // Step 4 — Smart routing
  const physician = router.route(encounter, physicians);

  // Step 5 — Add to queue
  queue.add({
    ...encounter,
    assignedPhysician: physician.id,
    denialPrediction: prediction
  });

  return {
    encounter,
    assignedPhysician: physician,
    fixes: fixResult.fixes,
    denialPrediction: prediction
  };
}
