import jwt from "jsonwebtoken";
import bcrypt from "bcrypt";

const SECRET = process.env.JWT_SECRET || "dev_secret";

export async function hashPassword(password: string) {
  return bcrypt.hash(password, 10);
}

export async function verifyPassword(password: string, hash: string) {
  return bcrypt.compare(password, hash);
}

export function generateToken(payload: any) {
  return jwt.sign(payload, SECRET, { expiresIn: "12h" });
}

export function verifyToken(token: string) {
  return jwt.verify(token, SECRET);
}
/server/middleware/authGuard.ts
import { Request, Response, NextFunction } from "express";
import { verifyToken } from "../services/auth";

export function authGuard(req: Request, res: Response, next: NextFunction) {
  const header = req.headers.authorization;

  if (!header) {
    return res.status(401).json({ error: "No token" });
  }

  const token = header.split(" ")[1];

  try {
    const decoded = verifyToken(token);
    res.locals.user = decoded;
    next();
  } catch {
    return res.status(401).json({ error: "Invalid token" });
  }
}

import express from "express";
import { hashPassword, verifyPassword, generateToken } from "../services/auth";

const router = express.Router();

let users: any[] = []; // replace with DB

router.post("/register", async (req, res) => {
  const { email, password, clinicId, role } = req.body;

  const hash = await hashPassword(password);

  const user = {
    id: Date.now().toString(),
    email,
    password: hash,
    clinicId,
    role
  };

  users.push(user);

  res.json({ success: true });
});

router.post("/login", async (req, res) => {
  const { email, password } = req.body;

  const user = users.find(u => u.email === email);
  if (!user) return res.status(401).json({ error: "Not found" });

  const valid = await verifyPassword(password, user.password);
  if (!valid) return res.status(401).json({ error: "Invalid" });

  const token = generateToken({
    id: user.id,
    clinicId: user.clinicId,
    role: user.role
  });

  res.json({ token });
});

export default router;

import { db } from "../db";
import { caseQueue } from "../db/queueSchema";

export async function enqueueCase(caseId: string, clinicId: string, priority: string) {
  await db.insert(caseQueue).values({
    caseId,
    clinicId,
    status: "pending",
    priority
  });
}

export async function getNextCases(clinicId: string, limit = 10) {
  return db.query.caseQueue.findMany({
    where: (q, { eq }) => eq(q.clinicId, clinicId),
    limit
  });
}

export async function markAssigned(caseId: string) {
  await db.update(caseQueue)
    .set({ status: "assigned" })
    .where((q, { eq }) => eq(q.caseId, caseId));
}
🔗 3️⃣ IMMUTABLE AUDIT CHAIN EXPORT

Now we make logs tamper-evident.

/server/services/auditExport.ts
import crypto from "crypto";

export function buildAuditChain(logs: any[]) {
  let previousHash = "GENESIS";
  const chain = [];

  for (const log of logs) {
    const payload = JSON.stringify({
      ...log,
      previousHash
    });

    const hash = crypto.createHash("sha256").update(payload).digest("hex");

    chain.push({
      ...log,
      hash,
      previousHash
    });

    previousHash = hash;
  }

  return chain;
}

📊 4️⃣ PHYSICIAN PERFORMANCE ENGINE

This is where clinics start paying attention.

/server/services/physicianMetrics.ts
export type PhysicianCase = {
  physicianId: string;
  reviewed: boolean;
  override: boolean;
  reviewTime: number;
  satisfaction: number;
};

export function computePhysicianMetrics(cases: PhysicianCase[]) {
  const map: any = {};

  for (const c of cases) {
    if (!map[c.physicianId]) {
      map[c.physicianId] = {
        total: 0,
        overrides: 0,
        totalTime: 0,
        satisfaction: 0
      };
    }

    const p = map[c.physicianId];

    p.total++;
    if (c.override) p.overrides++;
    p.totalTime += c.reviewTime;
    p.satisfaction += c.satisfaction;
  }

  return Object.entries(map).map(([id, p]: any) => ({
    physicianId: id,
    totalCases: p.total,
    avgReviewTime: p.totalTime / p.total,
    overrideRate: p.overrides / p.total,
    avgSatisfaction: p.satisfaction / p.total
  }));
}

        </div>
      ))}

  );
}
