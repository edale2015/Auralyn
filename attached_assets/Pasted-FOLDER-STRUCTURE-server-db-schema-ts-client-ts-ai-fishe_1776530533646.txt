// server/clinical/diagnosisEngine.ts


import { updateBeliefsWithFisher } from '../ai/bayesianUpdater';
import { runHallucinationGuards } from '../ai/hallucinationGuards';


export async function runDiagnosis(input: any, kb: any) {
  const prior: any = {};


  kb.diagnoses.forEach((d: any) => {
    prior[d.id] = Number(d.prior);
  });


  const likelihoodTable: any = {};


  kb.likelihoods.forEach((l: any) => {
    if (!likelihoodTable[l.diagnosisId]) {
      likelihoodTable[l.diagnosisId] = {};
    }


    likelihoodTable[l.diagnosisId][l.feature] = {
      present: { [l.diagnosisId]: Number(l.presentProb) },
      absent: { [l.diagnosisId]: Number(l.absentProb) },
    };
  });


  const result = updateBeliefsWithFisher(
    prior,
    likelihoodTable,
    input.observations,
  );


  const evidenceMap: any = {};
  kb.evidence.forEach((e: any) => {
    evidenceMap[e.diagnosisId] = e.features;
  });


  const contradictionMap: any = {};
  kb.contradictions.forEach((c: any) => {
    contradictionMap[c.diagnosisId] = c.features;
  });


  const guard = runHallucinationGuards({
    posterior: result.posterior,
    observations: input.observations,
    evidenceMap,
    contradictionMap,
    presentationProfile: input.presentationProfile,
  });


  return {
    ...result,
    guard,
  };
}

export async function runGoldenCases(cases: GoldenCase[]): Promise<ValidationRunResult[]> {
  const kb = await loadKB();
  const results: ValidationRunResult[] = [];


  for (const gc of cases) {
    const dx = await runDiagnosis(
      {
        observations: gc.observations,
        presentationProfile: gc.presentationProfile,
      },
      kb,
    );


    const final = runDispositionSafetyPipeline({
      posterior: dx.posterior,
      observations: gc.observations,
      proposedRuleDisposition: gc.expectedDisposition,
      proposedBayesianDisposition: gc.expectedDisposition,
      redFlagCount: gc.redFlagCount ?? 0,
      evidenceMap: Object.fromEntries((kb.evidence ?? []).map((e: any) => [e.diagnosisId, e.features])),
      contradictionMap: Object.fromEntries((kb.contradictions ?? []).map((c: any) => [c.diagnosisId, c.features])),
      presentationProfile: gc.presentationProfile,
    });


    const topDiagnosis = Object.entries(dx.posterior).sort((a, b) => Number(b[1]) - Number(a[1]))[0]?.[0];
    const diagnosisMatch = gc.expectedTopDiagnosis ? topDiagnosis === gc.expectedTopDiagnosis : true;
    const dispositionMatch = final.finalDisposition === gc.expectedDisposition;


    const minSafe = gc.minimumSafeDisposition ?? gc.expectedDisposition;
    const unsafeUndercall =
      severityOrder[final.finalDisposition] < severityOrder[minSafe];


    const reasons: string[] = [];
    if (!diagnosisMatch) reasons.push(`Top diagnosis mismatch: got ${topDiagnosis}, expected ${gc.expectedTopDiagnosis}`);
    if (!dispositionMatch) reasons.push(`Disposition mismatch: got ${final.finalDisposition}, expected ${gc.expectedDisposition}`);
    if (unsafeUndercall) reasons.push(`Unsafe undercall: got ${final.finalDisposition}, minimum safe is ${minSafe}`);


    results.push({
      caseId: gc.id,
      pass: !unsafeUndercall && dispositionMatch && diagnosisMatch,
      topDiagnosis,
      finalDisposition: final.finalDisposition,
      expectedDisposition: gc.expectedDisposition,
      expectedTopDiagnosis: gc.expectedTopDiagnosis,
      diagnosisMatch,
      dispositionMatch,
      unsafeUndercall,
      reasons,
    });
  }


  return results;
}


export function summarizeValidation(results: ValidationRunResult[]) {
  const total = results.length;
  const passed = results.filter(r => r.pass).length;
  const unsafeUndercalls = results.filter(r => r.unsafeUndercall).length;
  const dxMisses = results.filter(r => !r.diagnosisMatch).length;
  const dispMisses = results.filter(r => !r.dispositionMatch).length;


  return {
    total,
    passed,
    failed: total - passed,
    passRate: total ? passed / total : 0,
    unsafeUndercalls,
    diagnosisMisses: dxMisses,
    dispositionMisses: dispMisses,
  };
}

// server/validation/adversarialGenerator.ts


import { GoldenCase } from './goldenCaseTypes';


export function generateSparseVariant(base: GoldenCase): GoldenCase {
  return {
    ...base,
    id: `${base.id}__sparse`,
    title: `${base.title} [Sparse]`,
    observations: base.observations.slice(0, Math.max(1, Math.floor(base.observations.length / 2))),
  };
}


export function generateContradictoryVariant(base: GoldenCase): GoldenCase {
  const contradictory = [...base.observations];
  contradictory.push({ feature: 'contradiction_marker', value: true });


  return {
    ...base,
    id: `${base.id}__contradictory`,
    title: `${base.title} [Contradictory]`,
    observations: contradictory,
  };
}


export function generateMissingCriticalVariant(base: GoldenCase, missingFeature: string): GoldenCase {
  return {
    ...base,
    id: `${base.id}__missing_${missingFeature}`,
    title: `${base.title} [Missing ${missingFeature}]`,
    observations: base.observations.filter(o => o.feature !== missingFeature),
  };
}


export function expandAdversarialSet(baseCases: GoldenCase[]): GoldenCase[] {
  const out: GoldenCase[] = [];
  for (const c of baseCases) {
    out.push(c);
    out.push(generateSparseVariant(c));
    out.push(generateContradictoryVariant(c));
    if (c.observations[0]?.feature) {
      out.push(generateMissingCriticalVariant(c, c.observations[0].feature));
    }
  }
  return out;
}
4) Calibration monitor

This watches for a nasty pattern: high confidence with poor correctness.

// server/validation/calibrationMonitor.ts


export type CalibrationRow = {
  predictedConfidence: number;
  correct: boolean;
};


export function computeBrierScore(rows: CalibrationRow[]): number {
  if (!rows.length) return 0;
  return rows.reduce((sum, r) => {
    const y = r.correct ? 1 : 0;
    return sum + Math.pow(r.predictedConfidence - y, 2);
  }, 0) / rows.length;
}


export function bucketCalibration(rows: CalibrationRow[], bucketSize = 0.1) {
  const buckets: Array<{ bucket: string; avgConfidence: number; accuracy: number; count: number }> = [];


  for (let start = 0; start < 1; start += bucketSize) {
    const end = start + bucketSize;
    const inBucket = rows.filter(r => r.predictedConfidence >= start && r.predictedConfidence < end);
    if (!inBucket.length) continue;


    const avgConfidence = inBucket.reduce((a, b) => a + b.predictedConfidence, 0) / inBucket.length;
    const accuracy = inBucket.filter(r => r.correct).length / inBucket.length;
    buckets.push({
      bucket: `${start.toFixed(1)}-${end.toFixed(1)}`,
      avgConfidence,
      accuracy,
      count: inBucket.length,
    });
  }


  return buckets;
}


export function detectOverconfidence(rows: CalibrationRow[]) {
  const buckets = bucketCalibration(rows);
  return buckets
    .filter(b => b.avgConfidence - b.accuracy >= 0.15 && b.count >= 10)
    .map(b => ({
      ...b,
      flag: 'overconfident',
    }));
}

export function checkTemporalConsistency(observations: Observation[]): ExtraGuardResult {
  const reasons: string[] = [];
  const featureMap = new Map(observations.map(o => [o.feature, o.value]));


  const onset = featureMap.get('onset_hours');
  const duration = featureMap.get('duration_days');
  if (typeof onset === 'number' && typeof duration === 'number' && onset > 24 && duration < 1) {
    reasons.push('Temporal inconsistency between onset_hours and duration_days.');
  }


  return {
    blocked: reasons.length > 0,
    abstain: false,
    reasons,
  };
}


export function applyRiskFloor(
  topDiagnosis: string,
  proposedDisposition: string,
  riskFloors: Record<string, string>,
): ExtraGuardResult {
  const minimum = riskFloors[topDiagnosis];
  if (!minimum) return { blocked: false, abstain: false, reasons: [] };


  const severityOrder: Record<string, number> = {
    home: 0,
    urgent_care: 1,
    physician_review_required: 2,
    ed: 3,
    call_911: 4,
  };


  if (severityOrder[proposedDisposition] < severityOrder[minimum]) {
    return {
      blocked: true,
      abstain: false,
      reasons: [`Disposition ${proposedDisposition} below risk floor ${minimum} for ${topDiagnosis}.`],
    };
  }


  return { blocked: false, abstain: false, reasons: [] };
}


export function lowSupportAbstention(
  evidenceCoverageScore: number,
  contradictionScore: number,
  posteriorTopProb: number,
): ExtraGuardResult {
  const reasons: string[] = [];
  const abstain =
    evidenceCoverageScore < 0.2 ||
    contradictionScore > 0.5 ||
    posteriorTopProb < 0.35;


  if (evidenceCoverageScore < 0.2) reasons.push('Evidence coverage too low.');
  if (contradictionScore > 0.5) reasons.push('Contradiction burden too high.');
  if (posteriorTopProb < 0.35) reasons.push('Posterior confidence too low.');


  return {
    blocked: false,
    abstain,
    reasons,
  };
}
