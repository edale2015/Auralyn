type SafetyResult = {
  allowed: boolean;
  level: "LOW" | "MEDIUM" | "HIGH";
  reasons: string[];
};

export function runSafetyGate(input, safetyChecks): SafetyResult {
  const reasons: string[] = [];

  if (safetyChecks.pediatric?.risk === "HIGH") {
    reasons.push("Pediatric high-risk condition");
  }

  if (safetyChecks.pregnancy?.risk === "HIGH") {
    reasons.push("Pregnancy contraindication");
  }

  if (safetyChecks.drug?.length > 0) {
    reasons.push("Drug interaction risk");
  }

  if (input.chestPain && input.age > 50) {
    reasons.push("High-risk chest pain");
  }

  const level =
    reasons.length > 2 ? "HIGH" :
    reasons.length > 0 ? "MEDIUM" :
    "LOW";

  return {
    allowed: level !== "HIGH",
    level,
    reasons
  };
}

import { db } from "../db";
import { v4 as uuidv4 } from "uuid";

export function createTraceId() {
  return uuidv4();
}

export async function auditStep({
  traceId,
  step,
  input,
  output,
  metadata = {}
}) {
  await db.insert("audit_logs").values({
    trace_id: traceId,
    step,
    input,
    output,
    metadata
  });
}

🖥️ FRONTEND — TRACE VISUALIZER
export function TraceViewer({ traceId }) {
  const [steps, setSteps] = useState([]);

  useEffect(() => {
    fetch(`/api/audit/trace/${traceId}`)
      .then(r => r.json())
      .then(setSteps);
  }, [traceId]);

  return (
    <div>
      {steps.map((s, i) => (
        <div key={i}>
          <h4>{s.step}</h4>
          <pre>{JSON.stringify(s.output, null, 2)}</pre>
        </div>
      ))}
    </div>
  );
}
