type AgentPerformance = {
  agent: string;
  successRate: number;
  errors: number;
};

let performanceLog: AgentPerformance[] = [];

export function logPerformance(agent: string, success: boolean) {
  let record = performanceLog.find(p => p.agent === agent);

  if (!record) {
    record = { agent, successRate: 0, errors: 0 };
    performanceLog.push(record);
  }

  if (!success) record.errors++;

  record.successRate =
    (record.successRate + (success ? 1 : 0)) / 2;
}

export function computeBusinessMetrics(data: any[]) {
  const revenue = data.reduce((a, b) => a + b.amount, 0);

  const cost = data.length * 0.02;

  return {
    revenue,
    cost,
    profit: revenue - cost,
    margin: (revenue - cost) / revenue,
  };
}
