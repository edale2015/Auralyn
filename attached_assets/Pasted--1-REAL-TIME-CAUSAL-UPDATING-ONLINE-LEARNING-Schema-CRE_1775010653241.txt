export async function updateCausalModel(db, newRow) {
  const rows = await db.causal_stream_updates.findRecent(newRow.treatment)

  const ate = computeATE_IPW(rows)

  await db.causal_model_state.upsert({
    treatment: newRow.treatment,
    ate,
    sample_size: rows.length
  })

  return ate
}

// server/policy/adaptivePolicy.ts
export function chooseTreatment(state, options) {
  const epsilon = 0.1

  if (Math.random() < epsilon) {
    return options[Math.floor(Math.random() * options.length)]
  }

  return options.sort((a,b)=>b.reward-a.reward)[0]
}

export function computePayerMetrics(rows) {
  return {
    avg_cost: avg(rows.map(r=>r.cost)),
    readmission_rate: rate(rows.map(r=>r.readmission)),
    avg_los: avg(rows.map(r=>r.length_of_stay)),
    outcome_score: avg(rows.map(r=>r.outcome))
  }
}
