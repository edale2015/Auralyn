// server/billing/billingEngine.ts

export function generateBilling(input: {
  diagnosis: string;
  complexity: "low" | "moderate" | "high";
  timeMinutes: number;
}) {
  let cpt = "99213";

  if (input.complexity === "moderate") cpt = "99214";
  if (input.complexity === "high") cpt = "99215";

  return {
    cpt,
    icd10: input.diagnosis,
    estimatedReimbursement: {
      "99213": 75,
      "99214": 110,
      "99215": 160,
    }[cpt]
  };
}

🧠 4. FULL PIPELINE (REAL-WORLD EXECUTION)
🔥 realWorldEncounter.ts
// server/clinical/realWorldEncounter.ts

import { runFinalPipeline } from "./finalPipeline";
import { writeEHR } from "../integrations/ehrService";
import { generateBilling } from "../billing/billingEngine";

export async function handleEncounter(input: any) {
  // 1️⃣ Clinical decision
  const result = runFinalPipeline(input);

  // 2️⃣ EHR write
  await writeEHR({
    patientId: input.patientId,
    encounterId: result.encounterId,
    diagnosis: result.topDiagnosis,
    disposition: result.safetyDisposition,
    notes: result.physicianSummary,
  });

  // 3️⃣ Billing
  const billing = generateBilling({
    diagnosis: result.topDiagnosis,
    complexity: "moderate",
    timeMinutes: 20,
  });

  return {
    clinical: result,
    billing,
  };
}

📈 6. ROI ENGINE (THIS CLOSES DEALS)
🔥 roiEngine.ts
// server/business/roiEngine.ts

export function computeROI(encounters: any[]) {
  const total = encounters.length;

  const avgReimbursement = encounters.reduce(
    (sum, e) => sum + (e.billing?.estimatedReimbursement || 0),
    0
  ) / (total || 1);

  return {
    encounters: total,
    avgRevenue: avgReimbursement,
    projectedMonthly: avgReimbursement * total * 20,
  };
}
