// server/governance/override.ts

export function physicianOverrideCheck(result) {
  if (result.confidence < 0.85) {
    return {
      requireReview: true,
      reason: "Low confidence diagnosis"
    };
  }

  if (result.red_flags_present) {
    return {
      requireReview: true,
      reason: "Red flag triggered"
    };
  }

  return { requireReview: false };
}

// server/engine/multiComplaint.ts

export function fuseComplaints(results) {
  return results.reduce((acc, r) => {
    acc.risk += r.risk;
    acc.red_flags = acc.red_flags || r.red_flags;
    return acc;
  }, { risk: 0, red_flags: false });
}
