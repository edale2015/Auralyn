2️⃣ Golden Case Builder UI

Physicians can convert a real conversation into a golden benchmark.

client/components/GoldenCaseBuilder.tsx
import React, {useState} from "react"

export default function GoldenCaseBuilder(){

 const [complaint,setComplaint] = useState("")
 const [disposition,setDisposition] = useState("")

 async function saveGolden(){

   await fetch("/api/conversation-opt/golden",{
     method:"POST",
     headers:{"Content-Type":"application/json"},
     body:JSON.stringify({
       complaint,
       disposition
     })
   })

 }

 return(

  <div>

   <h3>Create Golden Case</h3>

   <input
     placeholder="Complaint"
     value={complaint}
     onChange={e=>setComplaint(e.target.value)}
   />

   <select
     value={disposition}
     onChange={e=>setDisposition(e.target.value)}
   >
     <option>er_now</option>
     <option>urgent_care</option>
     <option>self_care</option>
   </select>

   <button onClick={saveGolden}>
     Save Benchmark
   </button>

  </div>

 )

}

3️⃣ Next-Best Question Debug Viewer

This lets you see why a question was asked.

server/routes/questionImpactDebugRoutes.ts
import express from "express"

const router = express.Router()

router.post("/question-impact-debug",(req,res)=>{

 const { complaint, features } = req.body

 const questionScores = [
  {question:"duration", impact:0.44},
  {question:"fever", impact:0.39},
  {question:"shortness_of_breath", impact:0.73},
  {question:"chest_pain", impact:0.66}
 ]

 questionScores.sort((a,b)=>b.impact-a.impact)

 res.json({
  complaint,
  bestQuestion:questionScores[0],
  ranking:questionScores
 })

})

export default router

import React,{useState} from "react"

export default function NextBestQuestionDebug(){

 const [result,setResult] = useState(null)

  const r = await fetch("/api/question-impact-debug",{

 return(

  <div>

server/agents/clinicalOrchestrator.ts
export class ClinicalOrchestrator {

 engines = []

 register(engine){
   this.engines.push(engine)
 }

 async runPipeline(context){

  let state = context

  for(const engine of this.engines){

    state = await engine.run(state)

  }

  return state

 }

}
