📁 server/learning/payerRLHFEngine.ts
type Outcome = {
  payer: string;
  coding: any;
  paid: boolean;
  revenue: number;
};

export class PayerRLHFEngine {
  private weights: Record<string, Record<string, number>> = {};

  update(outcome: Outcome) {
    const payer = outcome.payer;
    const key = this.key(outcome.coding);

    if (!this.weights[payer]) this.weights[payer] = {};
    if (!this.weights[payer][key]) this.weights[payer][key] = 0.5;

    if (outcome.paid) {
      const reward = outcome.revenue / 500;
      this.weights[payer][key] += 0.1 * reward;
    } else {
      this.weights[payer][key] -= 0.2;
    }

    this.weights[payer][key] = Math.max(0, Math.min(1, this.weights[payer][key]));
  }

  score(coding: any, payer: string): number {
    return this.weights[payer]?.[this.key(coding)] || 0.4;
  }

  private key(coding: any) {
    const icd = coding.icd10?.[0]?.code || "none";
    const cpt = coding.cpt?.[0]?.code || "none";
    return `${icd}_${cpt}`;
  }
}
🧮 PART 2 — CONTRACT SIMULATION ENGINE

This lets you answer:

“If I had 1,000 cases… how much would each payer pay me?”

📁 server/analytics/contractSimulationEngine.ts
type Contract = {
  payer: string;
  cptRates: Record<string, number>;
};

type Encounter = {
  coding: any;
};

export class ContractSimulationEngine {
  simulate(encounters: Encounter[], contracts: Contract[]) {
    return contracts.map(contract => {
      let total = 0;

      for (const e of encounters) {
        const cpt = e.coding.cpt?.[0]?.code || "99213";
        const rate = contract.cptRates[cpt] || 50;

        total += rate;
      }

      return {
        payer: contract.payer,
        projectedRevenue: total,
        avgPerEncounter: total / encounters.length
      };
    });
  }
}

🎯 PART 3 — PAYER OPTIMIZATION DECISION ENGINE

This chooses the best strategy per encounter.

📁 server/analytics/payerStrategyEngine.ts
export class PayerStrategyEngine {
  chooseBest(encounter: any, payers: string[], rlhf: any, contracts: any) {
    let best = null;

    for (const payer of payers) {
      const score = rlhf.score(encounter.coding, payer);

      const rate = contracts[payer]?.[encounter.coding.cpt?.[0]?.code] || 50;

      const expectedValue = score * rate;

      if (!best || expectedValue > best.value) {
        best = {
          payer,
          value: expectedValue,
          score,
          rate
        };
      }
    }

    return best;
  }
}
