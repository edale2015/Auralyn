type Question = {
  id: string;
  text: string;
  weight: number;
  relevance: string[];
};

let questionBank: Question[] = [
  {
    id: "q1",
    text: "Do you have fever?",
    weight: 0.5,
    relevance: ["cough", "infection"],
  },
  {
    id: "q2",
    text: "Any chest pain or shortness of breath?",
    weight: 0.8,
    relevance: ["cough", "cardiac"],
  },
];

export function getNextBestQuestion(context: any) {
  const { complaint, asked } = context;

  let candidates = questionBank.filter(
    q =>
      q.relevance.includes(complaint) &&
      !asked.includes(q.id)
  );

  candidates.sort((a, b) => b.weight - a.weight);

  return candidates[0];
}
🔥 Upgrade: Reinforcement of Questions
export function updateQuestionWeight(id: string, delta: number) {
  const q = questionBank.find(q => q.id === id);
  if (q) q.weight += delta;
}

let pathways: Record<string, number> = {};

export function reinforcePathway(path: string, success: boolean) {
  if (!pathways[path]) pathways[path] = 0;

  pathways[path] += success ? 1 : -1;
}

export function getBestPath(paths: string[]) {
  return paths.sort(
    (a, b) => (pathways[b] || 0) - (pathways[a] || 0)
  )[0];
}
🔁 Connect to Outcomes
export function processOutcome(caseData: any, outcome: any) {
