import { scopeEngine } from "../execution/actionGuard";
import { auditLog } from "../audit/auditLogger";
import { broadcast } from "../ws/broadcast";

export async function evaluateAndExecuteAction(request: any) {
  const result = scopeEngine.evaluate(request);

  const logEntry = {
    timestamp: Date.now(),
    agent: request.agentRole,
    action: request.action,
    allowed: result.allowed,
    reason: result.reason || null,
    requiresOverride: result.requiresOverride || false,
    context: request.context
  };

  await auditLog("SCOPE_EVALUATION", logEntry);

  // 📡 Broadcast to Control Tower UI
  broadcast("scope_update", logEntry);

  if (!result.allowed) {
    if (result.requiresOverride) {
      return {
        status: "PENDING_OVERRIDE",
        message: "Physician approval required"
      };
    }

    throw new Error(`🚫 BLOCKED BY SCOPE: ${result.reason}`);
  }

  return { status: "APPROVED" };
}

export async function executeWithScope(request: any, handler: Function) {
  const result = await evaluateAndExecuteAction(request);

  if (result.status === "PENDING_OVERRIDE") {
    return result;
  }

  return await handler();
}

let clients: WebSocket[] = [];

export function registerClient(ws: WebSocket) {
  clients.push(ws);
}

export function broadcastScopeEvent(event: any) {
  clients.forEach(client => {
    client.send(JSON.stringify({
      type: "SCOPE_EVENT",
      payload: event
    }));
  });

import { useEffect, useState } from "react";

export default function ScopeCommandCenter() {
  const [events, setEvents] = useState<any[]>([]);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:3000/ws");

    ws.onmessage = (msg) => {
      const data = JSON.parse(msg.data);

      if (data.type === "SCOPE_EVENT") {
        setEvents(prev => [data.payload, ...prev]);
      }
    };

    return () => ws.close();
  }, []);

  return (
    <div className="p-4">
      <h1 className="text-xl font-bold mb-4">🧭 Agent Scope Command Center</h1>

      <div className="space-y-2">
        {events.map((e, i) => (
          <div key={i} className={`p-3 rounded shadow 
            ${e.allowed ? "bg-green-100" : "bg-red-100"}`}>
            
            <div><b>Agent:</b> {e.agent}</div>
            <div><b>Action:</b> {e.action}</div>
            <div><b>Status:</b> {e.allowed ? "✅ Allowed" : "🚫 Blocked"}</div>
            <div><b>Reason:</b> {e.reason}</div>

            {e.requiresOverride && (
              <button className="mt-2 bg-yellow-500 text-white px-2 py-1 rounded">
                Approve Override
              </button>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
