import express from "express";
import client from "prom-client";

const app = express();
const register = new client.Registry();

const httpLatency = new client.Histogram({
  name: "http_latency_ms",
  help: "HTTP latency",
  buckets: [50, 100, 300, 500, 1000, 3000],
});

const erRate = new client.Gauge({
  name: "er_rate",
  help: "ER rate"
});

register.registerMetric(httpLatency);
register.registerMetric(erRate);

app.get("/metrics", async (_, res) => {
  res.set("Content-Type", register.contentType);
  res.end(await register.metrics());
});

export function startMetricsServer() {
  app.listen(9100, () => console.log("📊 Metrics on 9100"));
}

export { httpLatency, erRate };

server/routes/smartRoutes.ts
import express from "express";

const router = express.Router();

router.get("/launch", (req, res) => {
  const iss = req.query.iss as string;

  const url = new URL(`${iss}/authorize`);
  url.searchParams.set("response_type", "code");
  url.searchParams.set("client_id", process.env.SMART_CLIENT_ID!);
  url.searchParams.set("redirect_uri", process.env.SMART_REDIRECT!);
  url.searchParams.set("scope", "launch openid profile user/*.read");

  res.redirect(url.toString());
});

router.get("/callback", async (req, res) => {
  const code = req.query.code as string;

  const tokenRes = await fetch(`${process.env.EPIC_ISSUER}/token`, {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: new URLSearchParams({
      grant_type: "authorization_code",
      code,
      redirect_uri: process.env.SMART_REDIRECT!,
      client_id: process.env.SMART_CLIENT_ID!
    })
  });

  const token = await tokenRes.json();

  res.json(token);
});

export default router;

export function buildRiskHeatmap(patients: any[]) {
  const map: Record<string, number> = {};

  for (const p of patients) {
    const key = p.complaint;
    map[key] = (map[key] || 0) + (p.riskScore || 1);
  }

  return map;
}

export function sortByPriority(patients: any[]) {
  return patients.sort((a, b) => {
    return (b.riskScore || 0) - (a.riskScore || 0);
  });
}

export function detectPatterns(data: any[]) {
  const counts: Record<string, number> = {};

  data.forEach(d => {
    counts[d.symptom] = (counts[d.symptom] || 0) + 1;
  });

  return Object.entries(counts)
    .filter(([_, c]) => c > 50);
}
