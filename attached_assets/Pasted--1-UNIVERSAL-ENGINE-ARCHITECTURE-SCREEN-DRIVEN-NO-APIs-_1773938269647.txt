🔹 app.py (Main Server)
from flask import Flask, request, render_template, jsonify
from engine.eligibility import determine_programs
from automation.agent import run_application_flow

app = Flask(__name__)

@app.route("/")
def home():
    return render_template("index.html")

@app.route("/submit", methods=["POST"])
def submit():
    data = request.json

    programs = determine_programs(data)

    results = []
    for program in programs:
        result = run_application_flow(program, data)
        results.append(result)

    return jsonify({
        "programs": programs,
        "results": results
    })

if __name__ == "__main__":
    app.run(debug=True)
🔹 eligibility.py
def determine_programs(user):
    programs = []

    if user.get("income", 0) < 2000:
        programs.append("SNAP")

    if user.get("children", 0) > 0:
        programs.append("Medicaid")

    return programs
🔹 agent.py (CORE RPA ENGINE)
from playwright.sync_api import sync_playwright
from automation.perception import find_field
from automation.actions import fill_field, click_button

def run_application_flow(program, user_data):
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=False)
        page = browser.new_page()

        if program == "SNAP":
            page.goto("https://example-snap-form.com")

        # Perception loop
        fields = find_field(page)

        for field in fields:
            label = field["label"]

            if label in user_data:
                fill_field(page, field, user_data[label])

        click_button(page, "Submit")

        return {"program": program, "status": "submitted"}
🔹 perception.py
def find_field(page):
    inputs = page.query_selector_all("input")

    fields = []
    for i in inputs:
        label = i.get_attribute("name") or "unknown"
        fields.append({
            "element": i,
            "label": label
        })

    return fields
🔹 actions.py
def fill_field(page, field, value):
    field["element"].fill(str(value))

def click_button(page, text):
    buttons = page.query_selector_all("button")

    for b in buttons:
        if text.lower() in b.inner_text().lower():
            b.click()
            break
