import { runChatIntakeEngine } from "../channels/chatIntakeEngine";
import { createSessionFromVoice } from "./voiceSessionStore";

export async function handleVoiceStream(stream: AsyncIterable<any>, meta: {
  userId: string;
  channel: "phone" | "web" | "app";
}) {
  const session = createSessionFromVoice(meta.userId);

  for await (const chunk of stream) {
    const transcript = chunk.text; // from STT engine

    const response = await runChatIntakeEngine({
      userId: meta.userId,
      channel: meta.channel,
      text: transcript,
      session
    });

    yield {
      audio: await textToSpeech(response.reply),
      text: response.reply
    };

    if (response.escalation) {
      break;
    }
  }
}

const router = express.Router();

router.post("/voice", async (req, res) => {

  const stream = createTwilioStream(req); // wrapper

  for await (const reply of handleVoiceStream(stream, {
    userId,
    channel: "phone"
  })) {
    await sendVoiceResponse(reply.audio);
  }

  res.sendStatus(200);
});

export async function analyzeImage(input: {
  imageUrl: string;
  context?: string;
}) {
  const result = await multimodalLLM({
    input: [
      { type: "image_url", url: input.imageUrl },
      { type: "text", text: input.context || "Analyze clinically" }
    ]
  });

  return {
    findings: result.findings,
    redFlags: result.redFlags || [],
    confidence: result.confidence || 0.7
  };
}

🎛️ PART 4 — MULTIMODAL ORCHESTRATOR

This is the fusion brain.

File: server/multimodal/orchestrator.ts
export async function runMultimodalFlow(input: {
  text?: string;
  imageUrl?: string;
  audioTranscript?: string;
  videoFrame?: string;
  session: any;
}) {
  const signals = [];

  if (input.text) signals.push({ type: "text", data: input.text });
  if (input.imageUrl) signals.push({ type: "vision", data: input.imageUrl });
  if (input.audioTranscript) signals.push({ type: "voice", data: input.audioTranscript });

  const fused = await fuseSignals(signals);

  return {
    structured: fused,
    nextStep: decideNextStep(fused)
  };
}

  | { type: "move"; axis: "x" | "y" | "z"; value: number }

  return fetch(process.env.ROBOT_API_URL + "/command", {
    method: "POST",
    body: JSON.stringify(cmd),
    headers: { "Content-Type": "application/json" }

export async function runUIAutomation(task: {
  goal: string;
  screenState: any;
}) {
  const plan = await uiLLM({
    prompt: `
    You control a UI.
    Goal: ${task.goal}
    Current Screen: ${JSON.stringify(task.screenState)}
    Return steps: click, type, scroll
    `
  });

  for (const step of plan.steps) {
    await executeStep(step);
  }

  return { success: true };
}
Example use
await runUIAutomation({
  goal: "Submit prior authorization form",
  screenState
});
