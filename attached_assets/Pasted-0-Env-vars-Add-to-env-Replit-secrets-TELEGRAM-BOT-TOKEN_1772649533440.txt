1) Firestore mapping: chat → active case

Create a tiny mapping collection channel_threads so each chat maps to the current open case.

server/services/channelThreadService.ts
import { firestore } from "../firestoreAdmin";

const THREADS = firestore.collection("channel_threads");

export async function getActiveCaseId(params: {
  channel: "telegram";
  threadId: string;
}): Promise<string | null> {
  const id = `${params.channel}:${params.threadId}`;
  const snap = await THREADS.doc(id).get();
  if (!snap.exists) return null;
  const data = snap.data() as any;
  return data?.activeCaseId ?? null;
}

export async function setActiveCaseId(params: {
  channel: "telegram";
  threadId: string;
  activeCaseId: string;
}): Promise<void> {
  const id = `${params.channel}:${params.threadId}`;
  await THREADS.doc(id).set(
    {
      channel: params.channel,
      threadId: params.threadId,
      activeCaseId: params.activeCaseId,
      updatedAt: new Date().toISOString(),
    },
    { merge: true }
  );
}
2) Telegram send helper
server/services/telegramClient.ts
export async function telegramSendMessage(params: {
  botToken: string;
  chatId: number | string;
  text: string;
}) {
  const url = `https://api.telegram.org/bot${params.botToken}/sendMessage`;
  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      chat_id: params.chatId,
      text: params.text,
      parse_mode: "HTML",
      disable_web_page_preview: true,
    }),
  });

  if (!res.ok) {
    const body = await res.text();
    throw new Error(`Telegram sendMessage failed: ${res.status} ${body}`);
  }
  return res.json();
}
3) Minimal complaint selection + required-question flow

This uses your registry to map “symptom text” → complaint slug.
If you already have a complaint alias matcher, plug it in here.

server/services/complaintMatchService.ts
import fs from "node:fs";

type RegistryRow = {
  Complaint_Slug: string;
  Display_Name?: string;
  Aliases?: string; // comma-separated
};

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQ = false;
  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"' && line[i + 1] === '"') { cur += '"'; i++; continue; }
    if (ch === '"') { inQ = !inQ; continue; }
    if (ch === "," && !inQ) { out.push(cur); cur = ""; continue; }
    cur += ch;
  }
  out.push(cur);
  return out.map(s => s.trim().replace(/^"|"$/g, ""));
}

let CACHE: RegistryRow[] | null = null;

function loadRegistry(): RegistryRow[] {
  if (CACHE) return CACHE;
  const p = "data/COMPLAINT_REGISTRY.csv"; // adjust if needed
  const text = fs.readFileSync(p, "utf8").trim();
  const lines = text.split("\n").map(l => l.trim()).filter(Boolean);
  const header = splitCsvLine(lines[0]);
  const idx = (k: string) => header.indexOf(k);

  const rows = lines.slice(1).map(line => {
    const c = splitCsvLine(line);
    return {
      Complaint_Slug: c[idx("Complaint_Slug")] ?? "",
      Display_Name: c[idx("Display_Name")] ?? "",
      Aliases: c[idx("Aliases")] ?? "",
    };
  }).filter(r => r.Complaint_Slug);

  CACHE = rows;
  return rows;
}

export function matchComplaintFromText(text: string): { slug: string; display: string } | null {
  const t = text.toLowerCase();
  const rows = loadRegistry();

  // Exact alias keyword hit
  for (const r of rows) {
    const aliases = (r.Aliases ?? "")
      .split(",")
      .map(s => s.trim().toLowerCase())
      .filter(Boolean);

    for (const a of aliases) {
      if (a && t.includes(a)) {
        return { slug: r.Complaint_Slug, display: r.Display_Name || r.Complaint_Slug };
      }
    }
  }

  // Fallback: slug contains token
  for (const r of rows) {
    if (t.includes(r.Complaint_Slug.replaceAll("_", " "))) {
      return { slug: r.Complaint_Slug, display: r.Display_Name || r.Complaint_Slug };
    }
  }

  return null;
}
Required questions

You likely have REQUIRED column in CORE_QUESTIONS. We’ll ask only required questions first.

server/services/questionFlowService.ts
import fs from "node:fs";

type QRow = {
  Complaint_Slug: string;
  Question_ID: string;
  Question_Text: string;
  Type: string;
  Required: string; // "TRUE"/"FALSE" or "1"/"0"
};

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQ = false;
  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"' && line[i + 1] === '"') { cur += '"'; i++; continue; }
    if (ch === '"') { inQ = !inQ; continue; }
    if (ch === "," && !inQ) { out.push(cur); cur = ""; continue; }
    cur += ch;
  }
  out.push(cur);
  return out.map(s => s.trim().replace(/^"|"$/g, ""));
}

let CACHE: QRow[] | null = null;

function loadQuestions(): QRow[] {
  if (CACHE) return CACHE;
  const p = "data/CORE_QUESTIONS.csv"; // adjust if needed
  const text = fs.readFileSync(p, "utf8").trim();
  const lines = text.split("\n").map(l => l.trim()).filter(Boolean);
  const header = splitCsvLine(lines[0]);
  const idx = (k: string) => header.indexOf(k);

  CACHE = lines.slice(1).map(line => {
    const c = splitCsvLine(line);
    return {
      Complaint_Slug: c[idx("Complaint_Slug")] ?? "",
      Question_ID: c[idx("Question_ID")] ?? "",
      Question_Text: c[idx("Question_Text")] ?? "",
      Type: c[idx("Type")] ?? "boolean",
      Required: c[idx("Required")] ?? "FALSE",
    };
  }).filter(r => r.Complaint_Slug && r.Question_ID);

  return CACHE!;
}

function isRequired(v: string) {
  const x = (v ?? "").toLowerCase().trim();
  return x === "true" || x === "1" || x === "yes";
}

export function getRequiredQuestions(complaintSlug: string): QRow[] {
  return loadQuestions().filter(q => q.Complaint_Slug === complaintSlug && isRequired(q.Required));
}

export function getNextRequiredQuestion(params: {
  complaintSlug: string;
  answers: Record<string, any>;
}): QRow | null {
  const req = getRequiredQuestions(params.complaintSlug);
  for (const q of req) {
    if (params.answers[q.Question_ID] === undefined) return q;
  }
  return null;
}
4) Telegram webhook route
server/routes/telegram.routes.ts
import { Router } from "express";
import { telegramSendMessage } from "../services/telegramClient";
import { getActiveCaseId, setActiveCaseId } from "../services/channelThreadService";
import { createCase, appendMessage, mergeAnswers, getCase } from "../services/caseService";
import { matchComplaintFromText } from "../services/complaintMatchService";
import { getNextRequiredQuestion } from "../services/questionFlowService";
import { runTriage } from "../services/triageService";

export const telegramRouter = Router();

function requireSecret(req: any) {
  const secret = process.env.TELEGRAM_WEBHOOK_SECRET;
  if (!secret) return true;
  return req.headers["x-telegram-bot-api-secret-token"] === secret;
}

// Minimal boolean parsing: "yes/no", "true/false"
function parseBoolean(text: string): boolean | null {
  const t = text.trim().toLowerCase();
  if (["yes", "y", "true", "1"].includes(t)) return true;
  if (["no", "n", "false", "0"].includes(t)) return false;
  return null;
}

telegramRouter.post("/webhook", async (req, res) => {
  if (!requireSecret(req)) return res.status(401).json({ error: "bad secret" });

  const botToken = process.env.TELEGRAM_BOT_TOKEN!;
  const update = req.body;

  const msg = update?.message ?? update?.edited_message;
  if (!msg) return res.json({ ok: true });

  const chatId = msg.chat?.id;
  const threadId = String(chatId);
  const text = msg.text ?? "";

  // Find or create case
  let caseId = await getActiveCaseId({ channel: "telegram", threadId });
  let c = caseId ? await getCase(caseId) : null;

  if (!c) {
    // Create new draft case with placeholder complaint until we match
    const created = await createCase({
      channel: "telegram",
      threadId,
      userId: String(chatId),
      complaintSlug: "unknown",
      complaintDisplay: "Unknown",
      engine: "GENERIC_V1",
    });
    caseId = created.caseId;
    await setActiveCaseId({ channel: "telegram", threadId, activeCaseId: caseId });
    c = created;
  }

  // Log inbound
  await appendMessage(caseId!, {
    ts: new Date().toISOString(),
    dir: "in",
    channel: "telegram",
    text,
  });

  // If complaint not chosen yet, try match from text
  if (c.complaint.slug === "unknown") {
    const match = matchComplaintFromText(text);
    if (!match) {
      await telegramSendMessage({
        botToken,
        chatId,
        text: "What’s your main symptom? (Example: “cough”, “fever”, “chest pain”, “rash”)",
      });
      return res.json({ ok: true });
    }

    // Update complaint in case doc (mergeAnswers doesn't touch complaint; so patch directly)
    // simplest: use mergeAnswers to store complaint_selected marker, and update complaint via state route if you have it.
    // For now, re-create a fresh case with correct complaint (cleanest):
    const newCase = await createCase({
      channel: "telegram",
      threadId,
      userId: String(chatId),
      complaintSlug: match.slug,
      complaintDisplay: match.display,
      engine: "GENERIC_V1",
    });
    await setActiveCaseId({ channel: "telegram", threadId, activeCaseId: newCase.caseId });

    await telegramSendMessage({
      botToken,
      chatId,
      text: `Okay. Let’s work through ${match.display}. Please answer yes/no.\n${getNextRequiredQuestion({ complaintSlug: match.slug, answers: {} })?.Question_Text ?? "Tell me more."}`,
    });

    return res.json({ ok: true });
  }

  // We have a complaint. Expect yes/no answer to next required question.
  const nextQ = getNextRequiredQuestion({
    complaintSlug: c.complaint.slug,
    answers: c.answers.structured ?? {},
  });

  if (!nextQ) {
    // Run triage (idempotent)
    const triage = await runTriage({
      complaintSlug: c.complaint.slug,
      answers: c.answers.structured ?? {},
      rulesetVersion: process.env.RULESET_VERSION ?? "local",
      dxPriorityVersion: process.env.DX_PRIORITY_VERSION ?? "local",
    });

    const needsReview =
      triage.disposition !== "ROUTINE" ||
      triage.confidence === "LOW" ||
      (triage.rfTriggered?.length ?? 0) > 0 ||
      ((triage as any).consistencyFlags?.length ?? 0) > 0;

    // Store triage + state (reuse existing route behavior via caseService if you prefer)
    // simplest: call your existing /api/cases/:id/triage endpoint, but here we can just respond message:
    const summary =
      `Disposition: <b>${triage.disposition}</b>\nTop Dx: <b>${triage.topDx}</b>\nConfidence: <b>${triage.confidence}</b>\n` +
      (needsReview ? "\nA clinician review is required before final advice is sent." : "\n");

    await telegramSendMessage({ botToken, chatId, text: summary });
    return res.json({ ok: true });
  }

  const b = parseBoolean(text);
  if (b === null) {
    await telegramSendMessage({
      botToken,
      chatId,
      text: `Please reply yes/no.\n${nextQ.Question_Text}`,
    });
    return res.json({ ok: true });
  }

  // Save answer
  const patch: Record<string, any> = {};
  patch[nextQ.Question_ID] = b ? "yes" : "no";
  const updated = await mergeAnswers(caseId!, patch);

  // Ask next question or triage
  const next2 = getNextRequiredQuestion({
    complaintSlug: updated.complaint.slug,
    answers: updated.answers.structured ?? {},
  });

  if (next2) {
    await telegramSendMessage({ botToken, chatId, text: next2.Question_Text });
  } else {
    await telegramSendMessage({ botToken, chatId, text: "Thanks. Processing your triage now…" });
    // On next message it will triage; or you can run triage immediately here.
  }

  res.json({ ok: true });
});

5) Webhook setup script
scripts/telegram-set-webhook.ts
const token = process.env.TELEGRAM_BOT_TOKEN!;
const base = process.env.PUBLIC_BASE_URL!;
const secret = process.env.TELEGRAM_WEBHOOK_SECRET!;

async function main() {
  const url = `https://api.telegram.org/bot${token}/setWebhook`;
  const webhookUrl = `${base}/telegram/webhook`;

  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      url: webhookUrl,
      secret_token: secret,
    }),
  });

  const body = await res.text();
  console.log(body);
}
main();
