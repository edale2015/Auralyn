📁 server/clinical/runFullClinicalFlow.ts
import { evaluateSafety } from "../safety/safetyGate";
import { runClinicalDebate } from "../reasoning/orchestrator";
import { logAuditEvent } from "../governance/auditLogger";
import { withTimeout } from "../utils/withTimeout";

export interface ClinicalInput {
  patientId: string;
  complaint: string;
  data: any;
}

export async function runFullClinicalFlow(input: ClinicalInput) {
  const start = Date.now();

  // 1) SAFETY GATE FIRST (hard stop if unsafe)
  const safety = await withTimeout(
    () => evaluateSafety(input),
    2500,
    { allowed: false, reason: "timeout_safety" }
  );

  if (!safety.allowed) {
    await logAuditEvent({
      type: "SAFETY_BLOCK",
      source: "runFullClinicalFlow",
      payload: { input, safety },
    });
    return {
      status: "blocked",
      reason: safety.reason,
      recommendation: "Seek immediate care",
    };
  }

  // 2) REASONING (multi-agent debate)
  const result = await withTimeout(
    () => runClinicalDebate(input.data),
    3000,
    { finalDiagnosis: "unknown", score: 0 }
  );

  // 3) AUDIT
  await logAuditEvent({
    type: "CLINICAL_DECISION",
    source: "runFullClinicalFlow",
    payload: { input, result, latencyMs: Date.now() - start },
  });

  return {
    status: "ok",
    diagnosis: result.finalDiagnosis,
    confidence: result.score,
  };
}

🧮 3) DATA ACCESS LAYER (kill duplicates)
📁 server/data/dataAccessLayer.ts
import { db } from "../db";
import { encounters, claims } from "../db/schema";

export async function getOutcomeStats() {
  const data = await db.select().from(encounters);

  const total = data.length || 1;

  return {
    avgOutcome:
      data.reduce((s, e) => s + Number(e.outcomeScore || 0), 0) / total,
    escalationRate:
      data.filter(e => e.escalated).length / total,
  };
}

export async function getRevenueStats() {
  const data = await db.select().from(claims);
  const total = data.length || 1;

  return {
    denialRate:
      data.filter(c => c.status === "denied").length / total,
    avgPaid:
      data.reduce((s, c) => s + Number(c.amountPaid || 0), 0) / total,
  };
}

import fs from "fs";
import path from "path";

const FILE = path.join(__dirname, "../../data/version_snapshots.json");

export function saveSnapshot(obj: any) {
  let data: any[] = [];
  if (fs.existsSync(FILE)) {
    data = JSON.parse(fs.readFileSync(FILE, "utf-8"));
  }

  data.push({
    id: Date.now(),
    timestamp: new Date().toISOString(),
    data: obj,
  });

  fs.writeFileSync(FILE, JSON.stringify(data, null, 2));
}

export function rollbackLast() {
  if (!fs.existsSync(FILE)) return null;

  const data = JSON.parse(fs.readFileSync(FILE, "utf-8"));
  return data[data.length - 2]?.data || null;
}

export async function runGoldenValidation() {

  for (const c of cases) {
    const res = await runFullClinicalFlow({
      patientId: "test",
      complaint: c.complaint,
      data: {},
    });

    if (
      (c.expected === "blocked" && res.status === "blocked") ||
      res.diagnosis === c.expected
    ) {
      pass++;
    }
  }

  const accuracy = pass / cases.length;

  return {
    accuracy,
    passed: accuracy >= 0.9,
  };
}
🚨 7) CIRCUIT BREAKER + TIMEOUT
📁 server/utils/withTimeout.ts
export async function withTimeout<T>(
  fn: () => Promise<T>,
  ms: number,
  fallback: T
): Promise<T> {
  return new Promise((resolve) => {
    const t = setTimeout(() => resolve(fallback), ms);

    fn()
      .then(r => {
        clearTimeout(t);
        resolve(r);
      })
      .catch(() => {
        clearTimeout(t);
        resolve(fallback);
      });
  });
}
