2) Strategy simulation lab

This compares multiple strategies against a selected pack.

/server/services/simulationLab.ts
import { SimulationCase } from "./simulationPacks";

export type SimulationStrategy = {
  name: string;
  confidenceThreshold: number;
  batchApprovalEnabled: boolean;
  mandatoryReviewMedium: boolean;
};

export type StrategyResult = {
  strategy: string;
  totalCases: number;
  autoApproved: number;
  physicianReviewed: number;
  escalated: number;
  accuracyPct: number;
  overrideRatePct: number;
  avgReviewSeconds: number;
  avgSatisfaction: number;
  costPerCase: number;
};

export function runSimulationStrategy(
  cases: SimulationCase[],
  strategy: SimulationStrategy
): StrategyResult {
  let autoApproved = 0;
  let physicianReviewed = 0;
  let escalated = 0;
  let correct = 0;
  let overrides = 0;
  let totalReviewSeconds = 0;
  let totalSatisfaction = 0;
  let totalCost = 0;

  for (const c of cases) {
    const mustReview =
      c.riskLevel === "HIGH" ||
      (strategy.mandatoryReviewMedium && c.riskLevel === "MEDIUM") ||
      !strategy.batchApprovalEnabled ||
      c.rawConfidence < strategy.confidenceThreshold;

    if (mustReview) {
      physicianReviewed++;
      totalReviewSeconds += c.reviewSeconds;
      totalCost += 7.5;
    } else {
      autoApproved++;
      totalReviewSeconds += Math.max(2, c.reviewSeconds * 0.2);
      totalCost += 2.2;
    }

    if (c.escalated) escalated++;
    if (c.wasCorrect) correct++;
    if (c.override) overrides++;
    totalSatisfaction += c.satisfaction;
  }

  const totalCases = cases.length;

  return {
    strategy: strategy.name,
    totalCases,
    autoApproved,
    physicianReviewed,
    escalated,
    accuracyPct: Number(((correct / Math.max(1, totalCases)) * 100).toFixed(2)),
    overrideRatePct: Number(((overrides / Math.max(1, totalCases)) * 100).toFixed(2)),
    avgReviewSeconds: Number((totalReviewSeconds / Math.max(1, totalCases)).toFixed(2)),
    avgSatisfaction: Number((totalSatisfaction / Math.max(1, totalCases)).toFixed(2)),
    costPerCase: Number((totalCost / Math.max(1, totalCases)).toFixed(2))
  };
}

export function runSimulationLab(
  cases: SimulationCase[],
  strategies: SimulationStrategy[]
): StrategyResult[] {
  return strategies.map(strategy => runSimulationStrategy(cases, strategy));
}

This is the automatic “closing bell” brain pass.

/server/services/nightlyControlRun.ts
import { runAdaptiveControlLoop, AdaptiveControlInput } from "./adaptiveControlLoop";

export type NightlyControlRunRecord = {
  id: string;
  clinicId: string;
  runAt: string;
  input: AdaptiveControlInput;
  output: ReturnType<typeof runAdaptiveControlLoop>;
};

const nightlyControlRunStore: NightlyControlRunRecord[] = [];

export function executeNightlyControlRun(input: AdaptiveControlInput) {
  const output = runAdaptiveControlLoop(input);

  const record: NightlyControlRunRecord = {
    id: `ncr-${Date.now()}`,
    clinicId: input.clinicId,
    runAt: new Date().toISOString(),
    input,
    output
  };

  nightlyControlRunStore.push(record);
  return record;
}

export function listNightlyControlRuns(clinicId?: string) {
  if (!clinicId) return nightlyControlRunStore;
  return nightlyControlRunStore.filter(r => r.clinicId === clinicId);
}

export type ExecutiveClinicSnapshot = {
  clinicId: string;
  totalCases: number;
  reviewedCases: number;
  escalatedCases: number;
  overrideRate: number;
  avgSatisfaction: number;
  avgCostPerCase: number;
  avgRevenuePerCase: number;
};

export type ExecutiveDashboardCard = {
  title: string;
  value: string | number;
  status: "good" | "watch" | "critical";
  note: string;
};

export function buildExecutiveDashboard(snapshot: ExecutiveClinicSnapshot) {
  const pendingCases = snapshot.totalCases - snapshot.reviewedCases;
  const marginPerCase = snapshot.avgRevenuePerCase - snapshot.avgCostPerCase;
  const marginPct = snapshot.avgRevenuePerCase
    ? (marginPerCase / snapshot.avgRevenuePerCase) * 100
    : 0;

  const cards: ExecutiveDashboardCard[] = [
    {
      title: "Total Cases",
      value: snapshot.totalCases,
      status: "good",
      note: "System-wide throughput"
    },
    {
      title: "Pending Cases",
      value: pendingCases,
      status: pendingCases > 50 ? "watch" : "good",
      note: "Queue pressure"
    },
    {
      title: "Escalation Rate",
      value: `${((snapshot.escalatedCases / Math.max(1, snapshot.totalCases)) * 100).toFixed(1)}%`,
      status: snapshot.escalatedCases / Math.max(1, snapshot.totalCases) > 0.12 ? "watch" : "good",
      note: "Safety load"
    },
    {
      title: "Override Rate",
      value: `${(snapshot.overrideRate * 100).toFixed(1)}%`,
      status: snapshot.overrideRate > 0.15 ? "critical" : snapshot.overrideRate > 0.08 ? "watch" : "good",
      note: "Clinical friction indicator"
    },
    {
      title: "Satisfaction",
      value: snapshot.avgSatisfaction.toFixed(2),
      status: snapshot.avgSatisfaction < 4.2 ? "watch" : "good",
      note: "Patient experience"
    },
    {
      title: "Margin %",
      value: `${marginPct.toFixed(1)}%`,
      status: marginPct < 25 ? "critical" : marginPct < 40 ? "watch" : "good",
      note: "Financial health"
    }
  ];

  return {
    clinicId: snapshot.clinicId,
    cards,
    headline: `Clinic ${snapshot.clinicId} is operating at ${(marginPct).toFixed(1)}% margin with ${snapshot.totalCases} total cases processed`
  };
}

type Pack = {

};

export default function SimulationLabDashboard() {
  const [packs, setPacks] = useState<Pack[]>([]);
  const [selectedPackId, setSelectedPackId] = useState<string>("");
  const [results, setResults] = useState<any[]>([]);

  useEffect(() => {
    fetch("/api/simulation-control/packs")

      })
    });

    const data = await res.json();
    setResults(data.results || []);
