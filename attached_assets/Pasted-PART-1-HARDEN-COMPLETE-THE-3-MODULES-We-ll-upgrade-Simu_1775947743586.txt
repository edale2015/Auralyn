import { calculateCentorScore } from "../clinical/centorEngine";
import { calculateStrepProbability } from "../clinical/bayesianStrepEngine";
import { evaluateRisk } from "../monitoring/riskGovernanceEngine";

interface SimulationResult {
  centor: number;
  probability: number;
  decision: string;
  riskFlags: string[];
  correctDecision: boolean;
}

export async function runSimulation(n: number = 10000) {
  const results: SimulationResult[] = [];

  let correct = 0;
  let antibiotics = 0;

  for (let i = 0; i < n; i++) {
    // 🎯 More realistic distributions
    const symptoms = {
      fever: Math.random() > 0.75,
      tonsillarExudate: Math.random() > 0.85,
      tenderAnteriorCervicalNodes: Math.random() > 0.7,
      absenceOfCough: Math.random() > 0.5,
      age: Math.floor(Math.random() * 80)
    };

    const centor = calculateCentorScore(symptoms);

    const prob = calculateStrepProbability({
      fever: symptoms.fever,
      exudate: symptoms.tonsillarExudate,
      nodes: symptoms.tenderAnteriorCervicalNodes,
      cough: !symptoms.absenceOfCough
    });

    const decision = prob > 0.5 ? "ANTIBIOTIC" : "NO_ANTIBIOTIC";

    if (decision === "ANTIBIOTIC") antibiotics++;

    const correctDecision =
      (prob > 0.6 && decision === "ANTIBIOTIC") ||
      (prob <= 0.6 && decision === "NO_ANTIBIOTIC");

    if (correctDecision) correct++;

    const riskFlags = evaluateRisk({ decision, probability: prob });

    results.push({ centor, probability: prob, decision, riskFlags, correctDecision });
  }

  return {
    results,
    summary: {
      accuracy: correct / n,
      antibioticRate: antibiotics / n
    }
  };
}

import { db } from "../../db";

export async function detectDrift(current: {
  antibioticRate: number;
  returnVisitRate: number;
}) {
  const baseline = await db.select("*").from("system_baseline").limit(1);

  const alerts: string[] = [];

  if (!baseline.length) return { alerts: ["No baseline set"] };

  const base = baseline[0];

  if (Math.abs(current.antibioticRate - base.antibiotic_rate) > 0.1) {
    alerts.push("Antibiotic rate drift");
  }

  if (current.returnVisitRate > base.return_visit_rate + 0.1) {
    alerts.push("Return visits increased");
  }

  return { alerts };
}

export async function updateBaseline(data: any) {
  await db.insert("system_baseline").values(data);
}
⚖️ 3. RISK GOVERNANCE (SEVERITY + ESCALATION)
/server/services/monitoring/riskGovernanceEngine.ts
export interface RiskAlert {
  message: string;
  severity: "LOW" | "MEDIUM" | "HIGH";
}

export function evaluateRisk(input: {
  decision: string;
  probability: number;
}): RiskAlert[] {
  const alerts: RiskAlert[] = [];

  if (input.decision === "NO_ANTIBIOTIC" && input.probability > 0.7) {
    alerts.push({
      message: "Possible under-treatment",
      severity: "HIGH"
    });
  }

  if (input.decision === "ANTIBIOTIC" && input.probability < 0.3) {
    alerts.push({
      message: "Possible over-treatment",
      severity: "MEDIUM"
    });
  }

  return alerts;
}

export async function logAudit(event: any) {

    .createHash("sha256")
    .update(JSON.stringify(event))
    .digest("hex");

  await db.insert("audit_log").values({
    ...event,
    hash,
    timestamp: new Date()
  });
}
