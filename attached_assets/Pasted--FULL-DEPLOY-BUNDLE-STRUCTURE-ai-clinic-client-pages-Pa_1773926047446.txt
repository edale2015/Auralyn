рҹ“Ғ /server/engines/triageNurse.ts
export async function triageConversation(input: any) {
  const { text, history = [] } = input;

  // Step 1: extract complaint
  let complaint = detectComplaint(text);

  // Step 2: ask next best question
  let nextQuestion = generateNextQuestion(complaint, history);

  return {
    complaint,
    nextQuestion,
    updatedHistory: [...history, text],
  };
}

function detectComplaint(text: string) {
  if (text.includes("cough")) return "cough";
  if (text.includes("chest")) return "chest pain";
  return "general";
}

function generateNextQuestion(complaint: string, history: any[]) {
  if (complaint === "cough") {
    return "Do you have fever, shortness of breath, or chest pain?";
  }
  return "Can you describe your symptoms more?";
}

import express from "express";
import { triageConversation } from "../engines/triageNurse";

const router = express.Router();

router.post("/incoming", async (req, res) => {
  const speech = req.body.SpeechResult || "";

  const result = await triageConversation({ text: speech });

  res.send(`
<Response>
  <Say>${result.nextQuestion}</Say>
  <Gather input="speech" action="/api/voice/incoming" />
</Response>
  `);
});

export default router;

const patientMemory: Record<string, any> = {};

export function updatePatientMemory(patientId: string, data: any) {
  if (!patientMemory[patientId]) {
    patientMemory[patientId] = [];
  }
  patientMemory[patientId].push(data);
}

export function getPatientContext(patientId: string) {
  return patientMemory[patientId] || [];
}
