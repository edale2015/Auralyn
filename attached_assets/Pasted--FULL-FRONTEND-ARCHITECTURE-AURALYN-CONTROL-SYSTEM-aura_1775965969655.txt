  return (
    <BrowserRouter>
      <MainLayout>
        <Routes>
          <Route path="/" element={<ControlTower />} />
          <Route path="/clinical" element={<ClinicalDashboard />} />
          <Route path="/knowledge" element={<KnowledgeHub />} />

    </BrowserRouter>
  );
}

📁 components/StatusBadge.tsx
export function StatusBadge({ status }: any) {
  const colors: any = {
    stable: "var(--color-success)",
    warning: "var(--color-warning)",
    critical: "var(--color-danger)"
  };

  return (
    <span
      style={{
        background: "#f1f5f9",
        padding: "4px 10px",
        borderRadius: "6px",
        color: colors[status]
      }}
    >
      {status}
    </span>
  );
}
