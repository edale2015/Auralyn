🔧 Backend
import { query } from "../db";

export async function addAlertRule(expr: string, target: string) {
  await query(`INSERT INTO alert_rules (expr,target) VALUES ($1,$2)`, [expr, target]);
}

export async function getAlertRules() {
  const r = await query(`SELECT * FROM alert_rules WHERE enabled=true`);
  return r.rows;
}

export async function evalAlertRules(metrics: any) {
  const rules = await getAlertRules();
  for (const r of rules) {
    if (eval(r.expr)) {
      if (r.target === "slack") await sendSlackAlert(`Rule fired: ${r.expr}`);
      if (r.target === "whatsapp") await sendWhatsAppAlert(`Rule fired: ${r.expr}`);
    }
  }
}

export default function AlertRules() {
  const [expr, setExpr] = useState("");
  const [target, setTarget] = useState("slack");

  async function save() {
    await fetch("/api/alerts/rules", {
      method: "POST",
      headers: {"Content-Type":"application/json"},
      body: JSON.stringify({ expr, target })
    });
    alert("Saved");
  }

  return (
    <div>
      <h3>🚨 Alert Rules</h3>
      <input value={expr} onChange={e=>setExpr(e.target.value)} />
      <select onChange={e=>setTarget(e.target.value)}>
        <option value="slack">Slack</option>
        <option value="whatsapp">WhatsApp</option>
      </select>
      <button onClick={save}>Save</button>
    </div>
  );
}

🏢 4) MULTI-TENANT ADMIN + RBAC + AUDIT EXPLORER
🔧 RBAC
const roles: Record<string, string[]> = {
  admin: ["*"],
  physician: ["view","override"],
  staff: ["view"]
};

export function allow(role: string, action: string) {
  return roles[role]?.includes("*") || roles[role]?.includes(action);
}
🔧 Middleware
export function requirePerm(action: string) {
  return (req:any,res:any,next:any)=>{
    const role = req.headers["x-role"] || "staff";
    if (!allow(role, action)) return res.status(403).send("Forbidden");
    next();
  };
}
🔧 Audit store + explorer
export async function writeAudit(a: any) {
  await query(`INSERT INTO audit (trace, step, data) VALUES ($1,$2,$3)`,
    [a.traceId, a.step, JSON.stringify(a)]);
}

export async function listAudit(limit=100) {
  const r = await query(`SELECT * FROM audit ORDER BY id DESC LIMIT $1`, [limit]);

import { Queue } from "bullmq";
const extQ = new Queue("external");

export async function externalSafe(fn: any, payload: any) {
  try { return await fn(payload); }
  catch {
    await extQ.add("retry", { fn: fn.name, payload }, { attempts: 5 });
    return { queued: true };
  }
}

export function sloBurnRate(errors: number, total: number) {

}
