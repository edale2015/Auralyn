        { action: "order:lactate" },
        { action: "order:blood_cultures" },
        { action: "give:fluids" }
      ],
      confidence: 0.93
    });
  }

  if (patient.vitals.sbp < 90) {
    actions.push({
      type: "HYPOTENSION_PROTOCOL",
      actions: [{ action: "give:fluids" }],
      confidence: 0.91
    });
  }

  return actions;
}

export async function replayCase(events: any[]) {
  const timeline = [];

  for (const e of events) {
    timeline.push({
      step: e.step,
      action: e.action,
      decision: e.result,
      timestamp: e.timestamp
    });
  }

  return timeline;
}

export function rerunDecision(event: any) {
  const result = scopeEngine.evaluate({
    agentRole: event.agent,
    action: event.action,
    context: event.context
  });

  return {
    original: event.result,
    replayed: result,
    match: JSON.stringify(event.result) === JSON.stringify(result)
  };
}

import { runDigitalTwin } from "../digitalTwin/digitalTwinEngine";
import { generateInterventions } from "../intervention/autonomousCopilot";

export async function runTrial(patients: any[]) {
  let outcomes = [];

  for (const p of patients) {
    const twin = runDigitalTwin(p);

    const interventions = await generateInterventions(p);

    const improvedOutcome = simulateOutcome(p, interventions);

    outcomes.push({
      patientId: p.id,
      baselineICU: twin.icuProb,
      improvedICU: improvedOutcome.icuProb
    });
  }

  return summarize(outcomes);
}

function simulateOutcome(patient: any, interventions: any[]) {
  let icuProb = 0.5;

  if (interventions.length > 0) {
    icuProb *= 0.7; // improvement effect
  }

  return { icuProb };
}

function summarize(outcomes: any[]) {
  const reduction =
    outcomes.reduce((sum, o) => sum + (o.baselineICU - o.improvedICU), 0) /
    outcomes.length;

  return {
    patients: outcomes.length,
    icuReduction: reduction
  };
}
