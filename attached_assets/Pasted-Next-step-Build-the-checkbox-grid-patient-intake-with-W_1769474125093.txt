Patch 1 (Backend): token+code generation + intake endpoints
A) Add helpers: server/intake/intakeAuth.ts

Create a new file:

// server/intake/intakeAuth.ts
import crypto from "crypto";

export function generateToken(): string {
  return crypto.randomBytes(24).toString("hex"); // 48 chars
}

export function generateCode(): string {
  return String(Math.floor(100000 + Math.random() * 900000)); // 6 digits
}

export function expiresAtMinutes(minutes: number): number {
  return Date.now() + minutes * 60 * 1000;
}

export function normalizeBaseUrl(url: string): string {
  return (url || "").replace(/\/+$/, "");
}

B) Add API routes: POST /api/intake/:token/verify and POST /api/intake/:token/submit

In server/routes.ts (or wherever you register API routes), add these handlers.

1) Verify code endpoint
app.post("/api/intake/:token/verify", async (req, res) => {
  try {
    const token = req.params.token;
    const { code } = req.body || {};
    if (!code) return res.status(400).json({ ok: false, error: "Missing code" });

    const encounter = await storage.getEncounterByIntakeToken(token);
    if (!encounter) return res.status(404).json({ ok: false, error: "Invalid token" });

    if (!encounter.intakeCode || String(code).trim() !== String(encounter.intakeCode).trim()) {
      return res.status(401).json({ ok: false, error: "Invalid code" });
    }
    if (encounter.intakeExpiresAt && Date.now() > encounter.intakeExpiresAt) {
      return res.status(401).json({ ok: false, error: "Code expired" });
    }

    return res.json({ ok: true, encounterId: encounter.id, flowId: encounter.flowId });
  } catch (e: any) {
    res.status(500).json({ ok: false, error: e?.message || String(e) });
  }
});

2) Submit answers endpoint
app.post("/api/intake/:token/submit", async (req, res) => {
  try {
    const token = req.params.token;
    const { code, answers } = req.body || {};
    if (!code || !answers) return res.status(400).json({ ok: false, error: "Missing code or answers" });

    const encounter = await storage.getEncounterByIntakeToken(token);
    if (!encounter) return res.status(404).json({ ok: false, error: "Invalid token" });

    if (!encounter.intakeCode || String(code).trim() !== String(encounter.intakeCode).trim()) {
      return res.status(401).json({ ok: false, error: "Invalid code" });
    }
    if (encounter.intakeExpiresAt && Date.now() > encounter.intakeExpiresAt) {
      return res.status(401).json({ ok: false, error: "Code expired" });
    }

    // Persist answers as object (preferred)
    const parsedAnswers = typeof answers === "string" ? JSON.parse(answers) : answers;

    // Compute proposal using existing computeProposal
    const proposal = await computeProposal(parsedAnswers);

    // Persist modifiers too (you already do this elsewhere)
    const modifiers = buildModifiersFromAnswers(parsedAnswers);

    await storage.updateEncounter(encounter.id, {
      answers: parsedAnswers,
      modifiers,
      proposal,
      status: "pending_review",
      // invalidate code after use
      intakeCode: null,
      intakeExpiresAt: null,
    });

    // Notify via WhatsApp if phone exists
    if (encounter.phoneNumber) {
      await sendWhatsAppMessage(encounter.phoneNumber,
        "Thanks. Your disposition has been sent to the physician for review.");
    }

    res.json({ ok: true, encounterId: encounter.id });
  } catch (e: any) {
    res.status(500).json({ ok: false, error: e?.message || String(e) });
  }
});
