⚙️ /server/services/router.ts
type Physician = {
  id: number;
  active: boolean;
  currentLoad: number;
  maxLoad: number;
  specialty?: string;
};

let physicians: Physician[] = [];

export function registerPhysician(p: Physician) {
  physicians.push(p);
}

export function assignCase(caseData: any) {
  const available = physicians
    .filter(p => p.active && p.currentLoad < p.maxLoad)
    .sort((a, b) => a.currentLoad - b.currentLoad);

  if (!available.length) return null;

  const selected = available[0];
  selected.currentLoad += 1;

  return selected.id;
}

let metrics = {
  totalCases: 0,
  avgTime: 0,
  overrides: 0
};

export function logCase(time: number, override: boolean) {
  metrics.totalCases++;
  metrics.avgTime =
    (metrics.avgTime + time) / 2;

  if (override) metrics.overrides++;
}

export function getMetrics() {
  return metrics;
}

import fetch from "node-fetch";

export async function pushToEHR(patient) {
  await fetch(process.env.EHR_URL + "/Patient", {
    method: "POST",
    headers: {
      Authorization: `Bearer ${process.env.EHR_TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      resourceType: "Patient",
      name: [{ text: patient.name }]
    })
  });
}
