from plugins.clinical.bayesian_engine import calculate_probability

@app.post("/diagnosis")
def diagnose(req: DiagnosisRequest):
    probs = calculate_probability(req.symptoms)

    top = max(probs, key=probs.get)

    return {
        "top": top,
        "confidence": probs[top],
        "full_distribution": probs
    }

import { useEffect, useState } from "react";

export default function ClinicalPanel() {
  const [data, setData] = useState(null);

  useEffect(() => {
    fetch("/clinical/live")
      .then(res => res.json())
      .then(setData);
  }, []);

  return (
    <div>
      <h2>Clinical Engine Status</h2>
      <pre>{JSON.stringify(data, null, 2)}</pre>
    </div>
  );
}
