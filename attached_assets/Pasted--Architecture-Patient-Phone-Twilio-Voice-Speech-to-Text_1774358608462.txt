  res.send(`
    <Response>
      <Say>Welcome. I will ask you a few questions to help triage your condition.</Say>
      <Gather input="speech" action="/api/voice/process" method="POST">
        <Say>Please briefly describe your problem.</Say>
      </Gather>
    </Response>
  `);
});

export default router;
🧠 2. Process Speech → AI → Response
// server/voice/processVoice.ts

import { runPatientFlow } from "../patient/patientFlow";

export async function processVoiceInput(transcript: string) {

  const result = await runPatientFlow({
    complaint: transcript,
    source: "voice"
  });

  return {
    response: result.status === "physician_review"
      ? "A physician will review your case shortly."
      : "Based on your answers, supportive care is recommended."
  };
}
🔁 3. Voice Response Loop
router.post("/api/voice/process", async (req, res) => {
  const speech = req.body.SpeechResult;

  robot.moveMouse(x, y);
  robot.mouseClick();
}

export function pressEnter() {
  robot.keyTap("enter");
}
🧠 2. AI Instruction Parser
// server/uiAgent/actionParser.ts

export function parseInstruction(instruction: string) {
  if (instruction.includes("type")) {
    return { action: "type", value: instruction.replace("type", "") };
  }

  if (instruction.includes("click")) {
    return { action: "click", x: 500, y: 400 };
  }

  return { action: "unknown" };
}
🔁 3. Execute Actions
// server/uiAgent/execute.ts

import { typeText, click } from "./uiController";

export function executeAction(action: any) {
  if (action.action === "type") {
    typeText(action.value);
  }

  if (action.action === "click") {
    click(action.x, action.y);
  }
}
