// server/ehr/clinicalWriteOrchestrator.ts

import { ehrWrite } from "./ehrWriter";
import { syncEncounterToFhir } from "./fhir/fhirService";
import { executeWithScope } from "../execution/executeWithScope";
import { logEvent } from "../ops/auditEvents";

export interface ClinicalWriteInput {
  clinicId: string;
  patientId: string;
  physicianId: string;

  disposition: string;
  notes: string;

  encounter?: any;
  patient?: any;

  system?: "athena" | "epic" | "ecw";

  physicianSigned: boolean;
  confidence: number;
}

export async function executeClinicalWrite(input: ClinicalWriteInput) {
  return executeWithScope(
    {
      agentRole: "clinical_writer",
      action: "write:clinical",
      context: {
        physicianSigned: input.physicianSigned,
        confidence: input.confidence,
        clinicId: input.clinicId,
      },
    },
    async () => {
      const startedAt = Date.now();

      // ── STEP 1: EHR WRITE (PRIMARY SOURCE OF TRUTH) ──
      const ehrResult = await ehrWrite({
        patientId: input.patientId,
        disposition: input.disposition,
        notes: input.notes,
        physicianId: input.physicianId,
        system: input.system,
      });

      // ── STEP 2: FHIR SYNC (SECONDARY / INTEROP) ──
      let fhirResult = null;

      if (input.encounter && input.patient) {
        try {
          fhirResult = await syncEncounterToFhir({
            clinicId: input.clinicId,
            encounter: input.encounter,
            patient: input.patient,
          });
        } catch (err: any) {
          // FHIR failure does NOT invalidate EHR success
          fhirResult = { ok: false, error: err.message };
        }
      }

      // ── STEP 3: AUDIT LOG ──
      logEvent({
        actor: input.physicianId,
        action: "clinical.write.completed",
        entityType: "patient",
        entityId: input.patientId,
        details: {
          ehrSuccess: ehrResult.success,
          fhirSuccess: fhirResult?.ok ?? null,
          latencyMs: Date.now() - startedAt,
        },
      });

      return {
        success: true,
        ehr: ehrResult,
        fhir: fhirResult,
      };
    }
  );
}

// server/audit/hashChain.ts

import crypto from "crypto";

let lastHash = "";

export function appendAudit(entry: any) {
  const payload = JSON.stringify(entry);
  const hash = crypto
    .createHash("sha256")
    .update(payload + lastHash)
    .digest("hex");

  lastHash = hash;

  return {
    ...entry,
    hash,
    previousHash: lastHash,
  };
}

const tokenCache = new Map<string, TokenCache>();

export async function getBearerToken(clinicId: string) {
  const cached = tokenCache.get(clinicId);

  if (cached && cached.expiresAt > Date.now()) {
    return cached.token;
  }

  const token = await fetchNewToken(clinicId);

  tokenCache.set(clinicId, {
    token,
    expiresAt: Date.now() + 3600 * 1000,
  });

  return token;
}
