let io: Server;

export function initPatientStream(server: any) {
  io = new Server(server, {
    cors: { origin: "*" }
  });

  io.on("connection", (socket) => {
    console.log("Client connected:", socket.id);

    socket.on("subscribe", () => {
      console.log("Subscribed to patient stream");
    });

    socket.on("disconnect", () => {
      console.log("Disconnected:", socket.id);
    });
  });
}

export function broadcastPatientUpdate(patient: any) {
  if (io) {
    io.emit("patient_update", patient);
  }
}

export function calculateNEWS2(vitals: any) {
  let score = 0;

  // Respiratory rate
  if (vitals.rr >= 25) score += 3;
  else if (vitals.rr >= 21) score += 2;
  else if (vitals.rr >= 12) score += 0;

  // Oxygen saturation
  if (vitals.spo2 <= 91) score += 3;
  else if (vitals.spo2 <= 93) score += 2;

  // Temperature
  if (vitals.temp >= 39) score += 2;
  else if (vitals.temp <= 35) score += 3;

  // Systolic BP
  if (vitals.sbp <= 90) score += 3;
  else if (vitals.sbp <= 100) score += 2;

  // Heart rate
  if (vitals.hr >= 131) score += 3;
  else if (vitals.hr >= 111) score += 2;

  return score;
}

export function predictSepsisRisk(vitals: any, trends: any[]) {
  const news = calculateNEWS2(vitals);

  const lactateTrend =
    trends.length > 1
      ? trends[trends.length - 1].lactate -
        trends[trends.length - 2].lactate
      : 0;

  let risk = news;

  if (lactateTrend > 0.5) risk += 2;
  if (vitals.temp > 38 && vitals.hr > 100) risk += 1;

  return {
    score: risk,
    level:
      risk >= 7 ? "HIGH" :
      risk >= 4 ? "MODERATE" :
      "LOW"
  };
}
🤖 AGENT FLEET + SEPSIS INTEGRATION
📁 server/agents/patientAgentRunner.ts
import { bestOfN } from "./bestOfN";
import { aggregateAgentResults } from "./agentCouncil";
import { predictSepsisRisk } from "../clinical/sepsisEngine";
import { broadcastPatientUpdate } from "../realtime/patientStream";

export async function processPatient(patient: any) {
  const sepsis = predictSepsisRisk(patient.vitals, patient.trends);

  const agentResults = await bestOfN(patient);

  const consensus = aggregateAgentResults(agentResults);

  const enrichedPatient = {
    ...patient,
    sepsis,
    agentResults,
    consensus
  };

  broadcastPatientUpdate(enrichedPatient);

  return enrichedPatient;
}

📁 frontend/src/pages/ControlTower.tsx
import { useEffect, useState } from "react";
import { io } from "socket.io-client";

const socket = io("http://localhost:3000");

export default function ControlTower() {
  const [patients, setPatients] = useState<any[]>([]);

  useEffect(() => {
    socket.emit("subscribe");

    socket.on("patient_update", (patient) => {
      setPatients((prev) => {
        const existing = prev.find(p => p.id === patient.id);
        if (existing) {
          return prev.map(p => p.id === patient.id ? patient : p);
        }
        return [...prev, patient];
      });
    });
  }, []);

  return (
    <div className="grid grid-cols-4 gap-4 p-4">
      {patients.map(p => (
        <PatientCard key={p.id} patient={p} />
      ))}
    </div>
  );
}

export function PatientCard({ patient }: any) {
  const color =
    patient.sepsis.level === "HIGH"
      ? "bg-red-500"
      : patient.sepsis.level === "MODERATE"
      ? "bg-yellow-500"
      : "bg-green-500";

  return (
    <div className={`p-4 rounded-xl shadow ${color}`}>
      <h2 className="text-xl font-bold">{patient.name}</h2>

      <div>HR: {patient.vitals.hr}</div>
      <div>BP: {patient.vitals.sbp}</div>
      <div>Temp: {patient.vitals.temp}</div>

      <div className="mt-2 font-bold">
        Sepsis Risk: {patient.sepsis.level}
      </div>

      <div className="mt-2 text-sm">
        Top Dx: {patient.consensus[0]?.dx}
      </div>
    </div>
  );
}
