import axios from "axios";

const FHIR_BASE = process.env.FHIR_BASE_URL!;
const ACCESS_TOKEN = process.env.FHIR_ACCESS_TOKEN!;

export async function getPatient(patientId: string) {
  const res = await axios.get(`${FHIR_BASE}/Patient/${patientId}`, {
    headers: { Authorization: `Bearer ${ACCESS_TOKEN}` },
  });
  return res.data;
}

export async function createObservation(patientId: string, vitals: any) {
  return axios.post(
    `${FHIR_BASE}/Observation`,
    {
      resourceType: "Observation",
      status: "final",
      subject: { reference: `Patient/${patientId}` },
      component: [
        { code: { text: "HR" }, valueQuantity: { value: vitals.hr } },
        { code: { text: "SpO2" }, valueQuantity: { value: vitals.spo2 } },
      ],
    },
    { headers: { Authorization: `Bearer ${ACCESS_TOKEN}` } }
  );
}
✅ WRITE DISPOSITION BACK TO EHR
export async function writeDisposition(patientId: string, disposition: string) {
  return axios.post(
    `${FHIR_BASE}/CarePlan`,
    {
      resourceType: "CarePlan",
      status: "active",
      subject: { reference: `Patient/${patientId}` },
      description: `Disposition: ${disposition}`,
    },
    { headers: { Authorization: `Bearer ${ACCESS_TOKEN}` } }
  );
}

3. 💰 CPT CODING ENGINE
✅ server/billing/cptEngine.ts
export function suggestCPT(encounter: any) {
  const { complexity, timeMinutes } = encounter;

  if (timeMinutes > 45) return "99215";
  if (complexity === "high") return "99214";
  if (complexity === "moderate") return "99213";
  return "99212";
}
4. 🚫 DENIAL PREDICTION ENGINE
✅ server/billing/denialPredictor.ts
export function predictDenial(claim: any) {
  let risk = 0;

  if (!claim.documentation) risk += 0.4;
  if (!claim.icdCodes?.length) risk += 0.3;
  if (!claim.cpt) risk += 0.2;

  return {
    denialRisk: risk,
    recommendation:
      risk > 0.5
        ? "Add documentation and ICD specificity"
        : "Low denial risk",
  };
}
5. 💸 REVENUE OPTIMIZATION ENGINE
✅ server/billing/revenueOptimizer.ts
export function optimizeClaim(claim: any) {
  const suggestions = [];

  if (claim.cpt === "99213" && claim.complexity === "high") {
    suggestions.push("Upgrade to 99214");
  }

  if (!claim.icdCodes.includes("R07.9") && claim.chiefComplaint === "chest pain") {
    suggestions.push("Add ICD R07.9");
  }

  return {
    optimized: suggestions.length === 0,
    suggestions,
  };
}
6. 🧪 FULL PILOT WORKFLOW ENGINE
✅ server/pilot/pilotFlow.ts
import { runFullClinicalFlow } from "../orchestrator/clinicalOrchestrator";
import { writeDisposition } from "../ehr/fhirClient";
import { suggestCPT } from "../billing/cptEngine";
import { predictDenial } from "../billing/denialPredictor";

export async function runPilotEncounter(input: any) {
  const clinical = await runFullClinicalFlow(input);

  const disposition = clinical.disposition;

  await writeDisposition(input.patientId, disposition);

  const cpt = suggestCPT(clinical);

  const denial = predictDenial({
    cpt,
    documentation: clinical.explanation,
    icdCodes: clinical.icdCodes,
  });

  return {
    clinical,
    billing: {
      cpt,
      denial,
    },
  };
}
