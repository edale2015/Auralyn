// LIVE CLINIC DEPLOYMENT + MULTI-TENANT + PAYOR OPTIMIZATION SYSTEM
const CPT_MAP: any = {
  sore_throat: "99213",
  chest_pain: "99285",
};


const ICD10_MAP: any = {
  demo_dx: "J02.9",
};


function optimizeClaim(encounter: any) {
  const diagnosis = encounter.result?.diagnosis;


  return {
    cpt: CPT_MAP[encounter.complaint] || "99213",
    icd10: ICD10_MAP[diagnosis] || "R69",
    modifier: "AI",
  };
}


app.post("/api/billing/generate", tenantMiddleware, (req: any, res) => {
  const enc = encounters[req.body.encounterId];


  const claim = optimizeClaim(enc);


  const id = uuidv4();


  claims[id] = {
    id,
    clinicId: req.clinicId,
    encounterId: enc.id,
    ...claim,
    status: "CREATED",
  };


  res.json(claims[id]);
});


// ==============================
// CLAIM SUBMISSION (SIMULATED)
// ==============================


app.post("/api/billing/submit", tenantMiddleware, (req: any, res) => {
  const claim = claims[req.body.claimId];


  claim.status = "SUBMITTED";
  claim.payerResponse = "APPROVED";


  res.json(claim);
});
