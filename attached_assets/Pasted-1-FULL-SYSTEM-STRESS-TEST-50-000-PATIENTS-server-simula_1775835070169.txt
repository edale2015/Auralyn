import { runFinalPipeline } from "../clinical/finalPipeline";

export async function runStressTest(n = 50000) {

  const start = Date.now();
  let errors = 0;
  let er = 0;

  await Promise.all(
    Array.from({ length: n }).map(async (_, i) => {
      try {
        const res = await runFinalPipeline({
          patientId: "S" + i,
          rawText: "chest_pain",
          answers: {},
          sessionId: "stress",
          actorId: "sim"
        }, "stress-" + i);

        if (res.safetyResult.finalDisposition === "ER_NOW") er++;

      } catch {
        errors++;
      }
    })
  );

  return {
    total: n,
    erRate: er / n,
    errors,
    durationMs: Date.now() - start
  };
}

🏥 2) REAL HOSPITAL PILOT INTEGRATION

export async function sendPilotCase(patient: any) {

  const payload = {
    patientId: patient.patientId,
    complaint: patient.complaint,
    vitals: patient.vitals,
    triage: patient.disposition
  };

  const res = await fetch(process.env.HOSPITAL_PILOT_API!, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.HOSPITAL_TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify(payload)
  });

  return res.json();
}
