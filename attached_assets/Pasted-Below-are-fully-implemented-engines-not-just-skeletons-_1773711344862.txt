1️⃣ Question Impact Analyzer

Determines which question reduces diagnostic uncertainty the most.

Used for:

adaptive questioning

entropy reduction

optimal next-question selection

server/engines/questionImpactAnalyzer.ts
export interface DiagnosisProb {
  diagnosis: string
  probability: number
}

export interface QuestionOutcome {
  answer: string
  probabilities: DiagnosisProb[]
}

export interface QuestionImpact {
  questionId: string
  entropyBefore: number
  entropyAfter: number
  informationGain: number
}

function entropy(probs: number[]): number {

  let e = 0

  for (const p of probs) {
    if (p > 0) {
      e -= p * Math.log2(p)
    }
  }

  return e
}

export class QuestionImpactAnalyzer {

  evaluate(
    questionId: string,
    before: DiagnosisProb[],
    outcomes: QuestionOutcome[]
  ): QuestionImpact {

    const beforeEntropy = entropy(before.map(p => p.probability))

    let weightedAfterEntropy = 0

    for (const outcome of outcomes) {

      const outcomeEntropy = entropy(
        outcome.probabilities.map(p => p.probability)
      )

      weightedAfterEntropy += outcomeEntropy / outcomes.length
    }

    return {
      questionId,
      entropyBefore: beforeEntropy,
      entropyAfter: weightedAfterEntropy,
      informationGain: beforeEntropy - weightedAfterEntropy
    }
  }
}

2️⃣ Differential Graph Builder

Creates a clinical relationship graph of diagnoses.

Used for:

differential visualization

clustering

server/engines/differentialGraphBuilder.ts
export interface DiagnosisNode {

  id: string
  name: string
  symptoms: string[]
}

export interface DiagnosisEdge {

  source: string
  target: string
  weight: number
}

export interface DifferentialGraph {

  nodes: DiagnosisNode[]
  edges: DiagnosisEdge[]
}

function jaccard(a: string[], b: string[]): number {

  const A = new Set(a)
  const B = new Set(b)

  const intersection =
    [...A].filter(x => B.has(x)).length

  const union =
    new Set([...a, ...b]).size

  return intersection / union
}

export class DifferentialGraphBuilder {

  build(nodes: DiagnosisNode[]): DifferentialGraph {

    const edges: DiagnosisEdge[] = []

    for (let i = 0; i < nodes.length; i++) {

      for (let j = i + 1; j < nodes.length; j++) {

        const similarity =
          jaccard(nodes[i].symptoms, nodes[j].symptoms)

        if (similarity > 0.3) {

          edges.push({
            source: nodes[i].id,
            target: nodes[j].id,
            weight: similarity
          })
        }
      }
    }

    return {
      nodes,
      edges
    }
  }
}

3️⃣ Clinical Drift Detector

Detects when the system changes behavior after an update.

Used for:

regression detection

deployment monitoring

safety assurance

server/engines/clinicalDriftDetector.ts
export interface CaseResult {

  caseId: string
  diagnosis: string
  disposition: string
  confidence: number
}

export interface DriftReport {

  diagnosisChangeRate: number
  dispositionChangeRate: number
  confidenceShift: number
}

export class ClinicalDriftDetector {

  detect(
    baseline: CaseResult[],
    current: CaseResult[]
  ): DriftReport {

    let diagnosisChanges = 0
    let dispositionChanges = 0
    let confidenceShift = 0

    for (let i = 0; i < baseline.length; i++) {

      if (baseline[i].diagnosis !== current[i].diagnosis) {
        diagnosisChanges++
      }

      if (baseline[i].disposition !== current[i].disposition) {
        dispositionChanges++
      }

      confidenceShift +=
        Math.abs(
          baseline[i].confidence -
          current[i].confidence
        )
    }

    return {

      diagnosisChangeRate:
        diagnosisChanges / baseline.length,

      dispositionChangeRate:
        dispositionChanges / baseline.length,

      confidenceShift:
        confidenceShift / baseline.length
    }
  }
}

4️⃣ Diagnostic Uncertainty Navigator

Uses entropy to guide next question selection.

server/engines/diagnosticUncertaintyNavigator.ts
import { questionImpactAnalyzer } from "./questionImpactAnalyzer"

export interface CandidateQuestion {

  id: string
  outcomes: any[]
}

export class DiagnosticUncertaintyNavigator {

  chooseNextQuestion(
    currentProbs: any[],
    questions: CandidateQuestion[]
  ) {

    let bestQuestion = null
    let bestGain = -Infinity

    for (const q of questions) {

      const impact =
        questionImpactAnalyzer.evaluate(
          q.id,
          currentProbs,
          q.outcomes
        )

      if (impact.informationGain > bestGain) {

        bestGain = impact.informationGain
        bestQuestion = q
      }
    }

    return bestQuestion
  }
}

5️⃣ Treatment Outcome Learning Engine

Updates diagnosis probabilities from real outcomes.

server/engines/outcomeLearningEngine.ts
export interface OutcomeRecord {

  diagnosisPredicted: string
  diagnosisActual: string
}

export class OutcomeLearningEngine {

  updateProbabilities(
    outcomes: OutcomeRecord[],
    prior: Record<string, number>
  ) {

    const counts: Record<string, number> = {}

    for (const o of outcomes) {

      if (!counts[o.diagnosisActual]) {
        counts[o.diagnosisActual] = 0
      }

      counts[o.diagnosisActual]++
    }

    const total = outcomes.length

    const updated: Record<string, number> = {}

    for (const d in counts) {

      updated[d] =
        (counts[d] / total) * 0.7 +
        (prior[d] || 0) * 0.3
    }

    return updated
  }
}

Calculates Wells, Centor, HEART.

export class ClinicalRiskScoringEngine {

  centor(data: any) {

    let score = 0

    if (data.fever) score++
    if (data.tonsillarExudate) score++
    if (data.tenderNodes) score++
    if (!data.cough) score++

    return score
  }

  wells(data: any) {

    let score = 0

    if (data.clinicalSignsDVT) score += 3
    if (data.heartRate > 100) score += 1.5
    if (data.recentImmobilization) score += 1.5

    return score
  }

  heart(data: any) {

    return (
      data.history +
      data.ecg +
      data.age +
      data.riskFactors +
      data.troponin
    )
  }
}

Shares anonymous learning signals.

export interface FederatedUpdate {

  clinicId: string
  diagnosisCounts: Record<string, number>
}

export class FederatedLearningEngine {

  aggregate(updates: FederatedUpdate[]) {

    const global: Record<string, number> = {}

    for (const u of updates) {

      for (const d in u.diagnosisCounts) {

        if (!global[d]) global[d] = 0

        global[d] += u.diagnosisCounts[d]
      }
    }

    return global
  }
}

⚠️ The Architecture Mistake to Avoid

When systems reach 150+ engines, the biggest failure is:

❌ Engine-to-engine coupling

Example:

Engine A → Engine B
Engine B → Engine C
Engine C → Engine A

This creates:

hidden dependencies

debugging nightmares

inconsistent reasoning

The Correct Pattern

Use a central orchestrator.

Clinical Brain
   │
   ├── perception engines
   ├── reasoning engines
   ├── safety engines
   ├── learning engines
   └── governance engines

Engines never call each other directly.

They only talk through:

Clinical Brain Orchestrator
Result

Your platform becomes:

Clinical Brain
      │
      ▼
Engine Layer (160+ engines)
      │
      ▼
Knowledge Graph
      │
      ▼
Clinical Intelligence Control Center

This architecture is scalable and stable.
