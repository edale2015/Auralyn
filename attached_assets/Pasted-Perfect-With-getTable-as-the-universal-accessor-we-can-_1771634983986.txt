1) DIFF_AND_CONFIDENCE node implementation (Sheets-backed)

Create:

server/complaints/nodes/diffAndConfidenceNode.ts
import { getTable } from "../../data/registry";

type Row = Record<string, any>;

function boolish(v: any): boolean | null {
  if (v === true || v === "true" || v === "yes" || v === 1 || v === "1") return true;
  if (v === false || v === "false" || v === "no" || v === 0 || v === "0") return false;
  return null;
}

function numish(v: any): number | null {
  if (v == null || v === "") return null;
  const n = Number(v);
  return Number.isFinite(n) ? n : null;
}

export async function runDiffAndConfidenceNode(state: any) {
  const cc = String(state.normalizedComplaint || "");
  const A: Record<string, any> = state.answers || {};
  const centor = numish(state.scores?.centor ?? state.scores?.CENTOR) ?? 0;

  const clusterRows: Row[] = await getTable("GLOBAL_CLUSTER_MASTER");
  const dxRows: Row[] = await getTable("CLUSTER_PRIMARY_DIAGNOSIS");

  // Expect headers like Cluster_ID, System, Primary_Chief_Complaints, Default_Disposition
  const validClusters = new Set(clusterRows.map(r => String(r.Cluster_ID || r.CLUSTER_ID || r.cluster_id)));

  // Build dx index: Cluster_ID -> [{Diagnosis_ID, Diagnosis_Name}]
  const dxByCluster = new Map<string, Array<{ diagnosisId: string; diagnosisName: string }>>();
  for (const r of dxRows) {
    const clusterId = String(r.Cluster_ID || r.CLUSTER_ID || r.cluster_id);
    const diagnosisId = String(r.Diagnosis_ID || r.DIAGNOSIS_ID || r.diagnosis_id);
    const diagnosisName = String(r.Diagnosis_Name || r.DIAGNOSIS_NAME || r.Diagnosis || r.DIAGNOSIS || "");
    if (!clusterId || !diagnosisId) continue;
    if (!dxByCluster.has(clusterId)) dxByCluster.set(clusterId, []);
    dxByCluster.get(clusterId)!.push({ diagnosisId, diagnosisName });
  }

  // Deterministic scoring for pilot: sore_throat.
  // These cluster IDs must exist in GLOBAL_CLUSTER_MASTER to be used.
  const score: Record<string, { points: number; evidence: string[] }> = {};

  function bump(clusterId: string, pts: number, evidenceField: string) {
    if (!validClusters.has(clusterId)) return;
    if (!score[clusterId]) score[clusterId] = { points: 0, evidence: [] };
    score[clusterId].points += pts;
    score[clusterId].evidence.push(evidenceField);
  }

  if (cc === "sore_throat") {
    // Viral URI / viral pharyngitis cluster
    if (boolish(A.ST_COUGH) === true) bump("CL_ST_VIRAL_URI", 2, "answers.ST_COUGH");
    if (boolish(A.ST_RUNNY) === true) bump("CL_ST_VIRAL_URI", 2, "answers.ST_RUNNY");
    if (boolish(A.ST_HOARSE) === true) bump("CL_ST_VIRAL_URI", 1, "answers.ST_HOARSE");

    // GAS cluster
    if (boolish(A.ST_FEVER) === true) bump("CL_ST_GAS", 1, "answers.ST_FEVER");
    if (boolish(A.ST_COUGH) === false) bump("CL_ST_GAS", 1, "answers.ST_COUGH");
    if (centor >= 2) bump("CL_ST_GAS", 2, "scores.centor");

    // Mono cluster (simple v1)
    if (boolish(A.ST_MONO) === true) bump("CL_ST_MONO", 2, "answers.ST_MONO");
  }

  const ranked = Object.entries(score)
    .filter(([, v]) => v.points > 0)
    .map(([clusterId, v]) => ({ clusterId, points: v.points, evidence: v.evidence }))
    .sort((a, b) => b.points - a.points)
    .slice(0, 3);

  state.clusterScores = Object.fromEntries(ranked.map(r => [r.clusterId, r.points]));
  state.clusterEvidence = Object.fromEntries(ranked.map(r => [r.clusterId, r.evidence]));
  state.activeClusters = ranked.map(r => r.clusterId);

  // Primary dx candidates (top 2 per cluster, top 3 clusters)
  const diagnosisCandidates: Array<{ diagnosisId: string; diagnosisName: string; clusterId: string; confidence: "MODERATE" | "LOW" }> = [];
  ranked.forEach((r, idx) => {
    const dxs = (dxByCluster.get(r.clusterId) || []).slice(0, 2);
    const conf: "MODERATE" | "LOW" = idx === 0 ? "MODERATE" : "LOW";
    for (const d of dxs) diagnosisCandidates.push({ ...d, clusterId: r.clusterId, confidence: conf });
  });
  state.diagnosisCandidates = diagnosisCandidates.slice(0, 6);

  // Confidence heuristic (non-breaking; you can store however you already do it)
  const missingRequired = (state.questionQueue || []).some((q: any) => q.required && !q.answered);
  const conf = missingRequired ? "LOW" : (ranked.length ? "MODERATE" : "LOW");
  state.caseConfidence = conf;

  return {
    rankedClusters: ranked,
    dxCount: state.diagnosisCandidates.length,
    confidence: conf,
    ruleRefs: [
      ...ranked.map(r => r.clusterId),
      ...state.diagnosisCandidates.map((d: any) => d.diagnosisId),
    ],
  };
}

✅ This uses your existing “old tables” (clusters + dx) and fills your already-present activeClusters. No new tables.

}

4) SPECIALIST_COUNCIL node (AutoGen-like consult, feature flag)

server/complaints/nodes/specialistCouncilNode.ts
type AgentAction = any; // import your real union type from shared/agentTypes

export interface CouncilLLM {
  callJson(system: string, payload: any): Promise<any>;
}

function shouldInvoke(state: any): boolean {
  if (state.routing?.state === "EMERGENT_ESCALATION") return false;
  if (state.redFlagGate?.gateResult === "ER_SEND") return false;

  const low = state.caseConfidence === "LOW";
  const missingRequired = (state.questionQueue || []).some((q: any) => q.required && !q.answered);
  const conflict = state.__conflictFlag === true;
  return low || missingRequired || conflict;
}

function rank(level: string): number {
  // align with your disposition strings
  const r: Record<string, number> = {
    self_care: 1,
    pcp: 2,
    urgent_care: 3,
    er_send: 4,
  };
  return r[level] ?? 2;
}

function pickKeyAnswers(state: any) {
  const A = state.answers || {};
  return {
    ST_DUR: A.ST_DUR,
    ST_FEVER: A.ST_FEVER,
    ST_COUGH: A.ST_COUGH,
    ST_RUNNY: A.ST_RUNNY,
    ST_HOARSE: A.ST_HOARSE,
    ST_SWALLOW: A.ST_SWALLOW,
    ST_DROOL: A.ST_DROOL,
    ST_BREATH: A.ST_BREATH,
    ST_TRISMUS: A.ST_TRISMUS,
    ST_ONE_SIDE: A.ST_ONE_SIDE,
    ST_MUFFLED: A.ST_MUFFLED,
    ST_EXUDATE: A.ST_EXUDATE,
    ST_TENDER_NODES: A.ST_TENDER_NODES,
    EXAM_TONSILLAR_EXUDATE: A.EXAM_TONSILLAR_EXUDATE,
    EXAM_TENDER_ANT_CERV_NODES: A.EXAM_TENDER_ANT_CERV_NODES,
  };
}

function missingRequiredIds(state: any): string[] {
  return (state.questionQueue || [])
    .filter((q: any) => q.required && !q.answered)
    .map((q: any) => String(q.questionId));
}

export async function runSpecialistCouncilNode(state: any, llm: CouncilLLM): Promise<{ pendingActions?: AgentAction[]; ruleRefs: string[]; llmCalls: number; invoked?: boolean }> {
  if (process.env.ENABLE_COUNCIL !== "1") return { ruleRefs: [], llmCalls: 0 };
  if (!shouldInvoke(state)) return { ruleRefs: [], llmCalls: 0 };

  const snapshot = {
    cc: state.normalizedComplaint,
    disposition: state.disposition,
    dispositionReasonCodes: state.dispositionReasonCodes || [],
    scores: state.scores || {},
    redFlagGate: state.redFlagGate || {},
    activeClusters: state.activeClusters || [],
    dxCandidates: (state.diagnosisCandidates || []).slice(0, 4),
    keyAnswers: pickKeyAnswers(state),
    missingRequired: missingRequiredIds(state),
  };

  const system = [
    "You are a clinical safety council with strict rules:",
    "1) Output STRICT JSON only.",
    "2) You may only escalate disposition, never de-escalate.",
    "3) You may propose missing question IDs and safety-net additions.",
    "Return keys: finalDisposition, why, mustAskQuestionIds, safetyNetAdds."
  ].join("\n");

  const clinician = await llm.callJson(system, { role: "clinician", snapshot });
  const critic = await llm.callJson(system, { role: "safety_critic", snapshot, clinician });
  const arbiter = await llm.callJson(system, { role: "rules_arbiter", snapshot, clinician, critic });

  const proposed = String(arbiter?.finalDisposition || critic?.finalDisposition || clinician?.finalDisposition || state.disposition);
  const finalDisp = rank(proposed) >= rank(state.disposition) ? proposed : state.disposition;

  if (finalDisp !== state.disposition) {
    state.disposition = finalDisp;
    state.dispositionReasonCodes = [...(state.dispositionReasonCodes || []), "COUNCIL_ESCALATION"];
    state.audit?.events?.push({ type: "COUNCIL", severity: "warn", message: String(arbiter?.why || "Escalated by safety council") });
  }

  const mustAsk: string[] = Array.isArray(arbiter?.mustAskQuestionIds) ? arbiter.mustAskQuestionIds.map(String) : [];
  const pendingActions: AgentAction[] = mustAsk.slice(0, 3).map(qid => ({
    type: "ASK_QUESTION",
    questionId: qid,
    // include any other fields your AgentAction requires; your CORE_QUESTIONS runner already knows how to load question text
  }));

  const safetyAdds: string[] = Array.isArray(arbiter?.safetyNetAdds) ? arbiter.safetyNetAdds.map(String) : [];
  if (safetyAdds.length) state.__councilSafetyNetAdds = safetyAdds;

  return { invoked: true, pendingActions, ruleRefs: ["COUNCIL"], llmCalls: 3 };
}

Key point: This doesn’t need dx→med mapping CSV, because your meds are already keyed by Indications_Cluster / DIAGNOSIS_ID in GLOBAL_MEDICATIONS_MASTER.
