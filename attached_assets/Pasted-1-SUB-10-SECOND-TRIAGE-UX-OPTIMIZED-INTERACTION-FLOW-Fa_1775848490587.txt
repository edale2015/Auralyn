import { nextSecondaryQuestion } from "../ux/secondary";
import { collectModifiers } from "../ux/modifiers";
import { fastTrack } from "../ux/fastTrack";

export async function fastTriageFlow(input: any) {
  const start = Date.now();

  const modifiers = collectModifiers(input);
  const fast = fastTrack({ ...input, ...modifiers });

  if (fast) {
    return {
      disposition: fast,
      durationMs: Date.now() - start,
      path: "fast-track"
    };
  }

  const q = nextSecondaryQuestion(input);
  if (q) {
    return {
      ask: q,
      durationMs: Date.now() - start,
      path: "progressive"
    };
  }

  // fallback to full pipeline
  const res = await runFinalPipeline(input, "fast");
  return {
    disposition: res.safetyResult.finalDisposition,
    durationMs: Date.now() - start,
    path: "full"
  };
}

💰 3) REAL PAYER CONTRACT ENGINE (MORE REALISTIC)

export function payerContract(claim: any) {

  const contracts: any = {
    Aetna: { multiplier: 1.0 },
    Medicaid: { multiplier: 0.6 },
    Medicare: { multiplier: 0.8 }
  };

  const base = {
    "99285": 500,
    "99284": 300,
    "99213": 120
  };

  const contract = contracts[claim.insurance] || { multiplier: 0.5 };

  return base[claim.cpt] * contract.multiplier;
}
