import { Pool } from "pg"
import Redis from "ioredis"
import { cosineSimilarity } from "../utils/vector"

const pool = new Pool({ connectionString: process.env.DATABASE_URL })
const redis = new Redis(process.env.REDIS_URL!)

// BM25-style keyword scoring (simplified)
function keywordScore(text: string, query: string): number {
  const words = query.toLowerCase().split(" ")
  let score = 0
  for (const w of words) {
    if (text.toLowerCase().includes(w)) score += 1
  }
  return score / words.length
}

// Vector similarity (your embedding pipeline assumed)
function vectorScore(vecA: number[], vecB: number[]): number {
  return cosineSimilarity(vecA, vecB)
}

// Reciprocal Rank Fusion
function rrf(rank: number, k = 60) {
  return 1 / (k + rank)
}

export async function hybridRetrieve(query: string, embedding: number[]) {
  const docs = await pool.query(`
    SELECT id, content, embedding
    FROM knowledge_documents
    LIMIT 200
  `)

  const scored = docs.rows.map((doc: any) => {
    const bm25 = keywordScore(doc.content, query)
    const vec = vectorScore(doc.embedding, embedding)

    return {
      ...doc,
      hybridScore: 0.5 * bm25 + 0.5 * vec
    }
  })

  return scored
    .sort((a, b) => b.hybridScore - a.hybridScore)
    .slice(0, 10)
}

⚡ 4. SEMANTIC CACHE (Redis)
server/cache/semanticCache.ts
import Redis from "ioredis"
import { cosineSimilarity } from "../utils/vector"

const redis = new Redis(process.env.REDIS_URL!)

const THRESHOLD = 0.92

export async function checkCache(query: string, embedding: number[]) {
  const keys = await redis.keys("cache:*")

  for (const key of keys) {
    const cached = JSON.parse(await redis.get(key)!)

    const sim = cosineSimilarity(embedding, cached.embedding)

    if (sim > THRESHOLD) {
      return cached.response
    }
  }

  return null
}

export async function storeCache(query: string, embedding: number[], response: string) {
  await redis.set(
    `cache:${Date.now()}`,
    JSON.stringify({ query, embedding, response })
  )
}
📊 5. RAG EVALUATION HARNESS (Clinical-grade)
server/eval/ragEvaluator.ts
export function evaluateRAG({
  question,
  answer,
  context,
  groundTruth
}: {
  question: string
  answer: string
  context: string[]
  groundTruth: string
}) {
  const faithfulness = similarity(answer, context.join(" "))
  const relevancy = similarity(answer, question)
  const precision = similarity(context.join(" "), groundTruth)

  return {
    faithfulness,
    relevancy,
    context_precision: precision
  }
}

function similarity(a: string, b: string) {
  const aWords = new Set(a.split(" "))
  const bWords = new Set(b.split(" "))

  let overlap = 0
  aWords.forEach(w => {
    if (bWords.has(w)) overlap++
  })

  return overlap / Math.max(aWords.size, 1)
}
🧬 6. CONTROL TOWER INTEGRATION
server/api/retrieval.ts
import express from "express"
import { retrievalAgent } from "../agents/retrievalAgent"
import { embed } from "../llm/embed"
import { checkCache, storeCache } from "../cache/semanticCache"

const router = express.Router()

router.post("/query", async (req, res) => {
  const { question } = req.body

  const embedding = await embed(question)

  const cached = await checkCache(question, embedding)
  if (cached) {
    return res.json({ answer: cached, cached: true })
  }

  const answer = await retrievalAgent(question)

  await storeCache(question, embedding, answer)

  res.json({ answer, cached: false })
})

export default router
