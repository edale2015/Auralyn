export function computeLiability(input: any) {
  const { ai, physician, outcome } = input;


  if (ai === physician) return { risk: "LOW" };


  if (outcome === "adverse") {
    return {
      risk: "HIGH",
      flag: "OVERRIDE_WITH_ADVERSE_OUTCOME",
    };
  }


  return { risk: "MODERATE" };
}
3. BIAS-AWARE RLHF ENGINE
learning/biasAwareRLHF.ts
export function updateWeights(input: any) {
  const { ai, physician, outcome, demographics } = input;


  // DO NOT TRUST SINGLE PHYSICIAN OVERRIDE
  if (!outcome) {
    return { action: "NO_UPDATE", reason: "no_outcome_data" };
  }


  // REQUIRE OUTCOME VALIDATION
  if (outcome === "confirmed_correct") {
    return { action: "INCREASE", weight: +0.1 };
  }


  if (outcome === "confirmed_wrong") {
    return { action: "DECREASE", weight: -0.2 };
  }


  return { action: "HOLD" };
}

let driftState = {
  locked: false,
};


export function handleDrift(drift: any) {
  if (drift.drift) {
    driftState.locked = true;
    return {
      action: "LOCK_MODEL",
      reason: "performance_drop",
    };
  }


  return { action: "NONE" };
}


export function isLocked() {
  return driftState.locked;
}

import { detectConfirmationBias } from "./confirmationBiasGuard";
import { updateWeights } from "./biasAwareRLHF";
import { handleDrift, isLocked } from "./driftControl";


export function safeLearning(input: any) {
  if (isLocked()) {
    return { status: "BLOCKED", reason: "model_locked" };
  }


  const bias = detectConfirmationBias(input);
  if (bias.flagged) {
    return { status: "SKIPPED", reason: bias.reason };
  }


  return updateWeights(input);
}

export function rateLimit(req: any, res: any, next: any) {
  if (active > 100) {
    return res.status(429).json({ error: "System overloaded" });
  }


  active++;
  res.on("finish", () => active--);


  next();
}
