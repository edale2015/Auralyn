🧾 PART 2 — AGENT REGISTRY
📁 server/agents/agentRegistry.ts
export type AgentStatus = "idle" | "busy" | "error" | "disabled";

export interface Agent {
  name: string;
  status: AgentStatus;
  lastRun?: number;
  run: (task: any) => Promise<any>;
}

const agents = new Map<string, Agent>();

export function registerAgent(agent: Agent) {
  agents.set(agent.name, agent);
}

export function getAgents() {
  return Array.from(agents.values());
}

export function getAgent(name: string) {
  return agents.get(name);
}
📦 PART 3 — TASK BUS

A simple queue for agents to pull from.

📁 server/agents/taskBus.ts
type Task = {
  id: string;
  type: string;
  payload: any;
  priority: number;
  createdAt: number;
};

const queue: Task[] = [];

export function addTask(task: Task) {
  queue.push(task);
  queue.sort((a, b) => b.priority - a.priority);
}

export function getNextTask(): Task | undefined {
  return queue.shift();
}

⚙️ PART 6 — AGENT EXECUTOR LOOP

This runs continuously.

📁 agentExecutor.ts
import { getNextTask } from "./taskBus";
import { getAgents } from "./agentRegistry";

export async function runAgentLoop() {
  const task = getNextTask();
  if (!task) return;

  const agents = getAgents();

  for (const agent of agents) {
    try {
      if (matches(agent.name, task.type)) {
        agent.status = "busy";

        const result = await agent.run(task);

        agent.status = "idle";
        agent.lastRun = Date.now();

        console.log(`✅ ${agent.name} handled ${task.type}`, result);
      }
    } catch (e) {
      agent.status = "error";
      console.error(`❌ ${agent.name}`, e);
    }
  }
}

function matches(agentName: string, taskType: string) {
  return agentName.toUpperCase().includes(taskType.split("_")[0]);
}
🔁 Start loop
setInterval(() => {
  runAgentLoop();
}, 1000);
