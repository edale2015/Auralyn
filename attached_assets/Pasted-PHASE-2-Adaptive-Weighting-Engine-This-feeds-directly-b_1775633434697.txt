import { getAgentPerformance } from "./outcomeLearningService";

export function getAgentWeight(agentId: string): number {
  const stats = getAgentPerformance();
  const agent = stats.find(a => a.agentId === agentId);

  if (!agent) return 1;

  // normalize score → weight
  return Math.max(0.2, Math.min(2, 1 + agent.score));
}

📈 PHASE 3 — Engine Reliability Dashboard Data
📁 server/monitoring/engineReliabilityService.ts
interface EngineMetrics {
  engine: string;
  calls: number;
  failures: number;
  totalLatencyMs: number;
}

const metrics: Record<string, EngineMetrics> = {};

export function trackEngineCall(engine: string, latencyMs: number, failed: boolean) {
  if (!metrics[engine]) {
    metrics[engine] = {
      engine,
      calls: 0,
      failures: 0,
      totalLatencyMs: 0,
    };
  }

  const m = metrics[engine];

  m.calls++;
  m.totalLatencyMs += latencyMs;
  if (failed) m.failures++;
}

export function getEngineMetrics() {
  return Object.values(metrics).map(m => ({
    engine: m.engine,
    calls: m.calls,
    failureRate: m.calls ? m.failures / m.calls : 0,
    avgLatency: m.calls ? m.totalLatencyMs / m.calls : 0,
  }));
}

📁 EnginePerformancePanel.tsx
import { useEffect, useState } from "react";

export default function EnginePerformancePanel() {
  const [data, setData] = useState<any[]>([]);

  useEffect(() => {
    fetch("/api/metrics/engines")
      .then(r => r.json())
      .then(setData);
  }, []);

  return (
    <div>
      <h2 className="text-xl font-bold">Engine Performance</h2>

      {data.map(e => (
        <div key={e.engine} className="border p-2">
          <div><b>{e.engine}</b></div>
          <div>Calls: {e.calls}</div>
          <div>Failure Rate: {(e.failureRate * 100).toFixed(1)}%</div>
          <div>Latency: {e.avgLatency.toFixed(0)} ms</div>
        </div>
      ))}
    </div>
  );
}
