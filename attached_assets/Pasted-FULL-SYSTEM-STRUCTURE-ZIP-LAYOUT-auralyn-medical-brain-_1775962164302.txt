🧠 SIMULATION ENGINE (LIVE PATIENTS)
server/routes/simulation.ts
import express from "express";
import { SequentialClinicalReasoner } from "../agents/sequentialClinicalReasoner";

const router = express.Router();
const reasoner = new SequentialClinicalReasoner();

const mockPatients = [
  {
    id: 1,
    complaint: "cough",
    age: 45,
    redFlags: [],
  },
  {
    id: 2,
    complaint: "chest pain",
    age: 60,
    redFlags: ["diaphoresis"],
  },
];

router.get("/patients", (req, res) => {
  res.json(mockPatients);
});

router.post("/run", async (req, res) => {
  const result = await reasoner.run(req.body);
  res.json(result);
});

export default router;
🧭 CONTROL TOWER API
server/routes/controlTower.ts
import express from "express";
import { SystemContextEngine } from "../agents/systemContextEngine";
import { EvidenceEngine } from "../agents/evidenceEngine";

const router = express.Router();

const context = new SystemContextEngine();
const evidence = new EvidenceEngine();

router.get("/system", (req, res) => {
  res.json(context.scanProject());
});

router.get("/evidence", async (req, res) => {
  const data = await evidence.searchGuidelines(req.query.q as string);
  res.json(data);
});

export default router;

  );

🧪 LIVE SIMULATION DASHBOARD
client/src/pages/Simulation.tsx
import { useEffect, useState } from "react";
import { api } from "../api";

export default function Simulation() {
  const [patients, setPatients] = useState<any[]>([]);
  const [result, setResult] = useState<any>(null);

  useEffect(() => {
    api.getPatients().then(setPatients);
  }, []);

  const run = async (p: any) => {
    const res = await api.runSimulation(p);
    setResult(res);
  };

  return (
    <div>
      <h1>🧪 Live Patient Simulation</h1>

      {patients.map(p => (
        <div key={p.id}>
          <button onClick={() => run(p)}>
            Run Patient {p.id} ({p.complaint})
          </button>
        </div>
      ))}

      <pre>{JSON.stringify(result, null, 2)}</pre>
    </div>
  );
}
