Step 2: Clinical State Confidence Scoring
Goal

Add a deterministic “how sure are we” layer so the system:

asks confirm questions when uncertain

avoids overconfident counseling

What to add to CaseState

confidence.global: HIGH | MODERATE | LOW

confidence.by_inference[]: list of { itemType, item, confidence, evidence[] }

Rules (simple and auditable)

HIGH: evidence from med + confirmed PMH/diagnosis table match
Example: Metformin + DM in PMH

MODERATE: evidence from med only, or symptom cluster only
Example: GLP-1 suggests obesity/DM but not confirmed

LOW: vague symptom text only

How it affects behavior

If LOW, agents must ask a confirm question before recommending interventions.

If MODERATE, allow UC spot interventions but emphasize “confirm” steps.

If HIGH, proceed normally.

Acceptance tests

Scenario with only “tired, overweight” yields LOW/MODERATE and asks confirmation.

Scenario with “metformin + A1c 9” yields HIGH.

/**
 * Bulk scenario CSV test harness
 *
 * What it does:
 * - Reads a CSV file with columns: scenario_id, channel, input_json, expected_gate, expected_contains
 * - Calls your existing runScenario admin endpoint with ?channel=
 * - Asserts:
 *    1) gate matches expected_gate
 *    2) response contains each token in expected_contains (semicolon-separated)
 *
 * Usage (Node 18+ / TS):
 *   1) Save as: scripts/runScenarioCsvHarness.ts
 *   2) Put your CSV at: server/data/tests/scenarios_100.csv (or any path)
 *   3) Run:
 *        npx ts-node scripts/runScenarioCsvHarness.ts --csv server/data/tests/scenarios_100.csv --base http://localhost:3000
 *
 * Notes:
 * - This harness is intentionally tolerant: it searches for expected tokens in multiple places:
 *   - response.gate
 *   - response.matchedRedFlags / response.matchedRedFlagIds / response.trace IDs (if present)
 *   - stringified response fallback
 */

import * as fs from "fs";
import * as path from "path";

type Row = {
  scenario_id: string;
  channel: string;
  input_json: string;
  expected_gate: string;
  expected_contains: string;
};

type Args = {
  csvPath: string;
  baseUrl: string;
  endpoint: string;
  concurrency: number;
  failFast: boolean;
  timeoutMs: number;
};

function parseArgs(argv: string[]): Args {
  const get = (key: string, def?: string) => {
    const idx = argv.indexOf(key);
    if (idx === -1) return def;
    return argv[idx + 1];
  };
  const csvPath = get("--csv", "")!;
  if (!csvPath) {
    throw new Error(`Missing --csv path. Example: --csv server/data/tests/scenarios_100.csv`);
  }
  return {
    csvPath,
    baseUrl: get("--base", "http://localhost:3000")!,
    endpoint: get("--endpoint", "/api/admin/runScenario")!,
    concurrency: Number(get("--concurrency", "6")!),
    failFast: (get("--failFast", "false")! === "true"),
    timeoutMs: Number(get("--timeoutMs", "20000")!),
  };
}

/** Minimal RFC4180-ish CSV parser that supports quoted fields with commas/newlines. */
function parseCsv(content: string): Row[] {
  const rows: string[][] = [];
  let curField = "";
  let curRow: string[] = [];
  let inQuotes = false;

  const pushField = () => {
    curRow.push(curField);
    curField = "";
  };
  const pushRow = () => {
    // skip empty trailing lines
    if (curRow.length === 1 && curRow[0].trim() === "") {
      curRow = [];
      return;
    }
    rows.push(curRow);
    curRow = [];
  };

  for (let i = 0; i < content.length; i++) {
    const ch = content[i];

    if (inQuotes) {
      if (ch === '"') {
        const next = content[i + 1];
        if (next === '"') {
          // escaped quote
          curField += '"';
          i++;
        } else {
          inQuotes = false;
        }
      } else {
        curField += ch;
      }
      continue;
    }

    if (ch === '"') {
      inQuotes = true;
      continue;
    }
    if (ch === ",") {
      pushField();
      continue;
    }
    if (ch === "\n") {
      pushField();
      pushRow();
      continue;
    }
    if (ch === "\r") {
      // ignore \r (Windows)
      continue;
    }
    curField += ch;
  }
  // flush final
  pushField();
  pushRow();

  if (rows.length < 2) {
    throw new Error("CSV appears empty or missing header row.");
  }

  const header = rows[0].map((h) => h.trim());
  const idx = (name: string) => {
    const j = header.indexOf(name);
    if (j === -1) throw new Error(`CSV missing required column: ${name}`);
    return j;
  };

  const out: Row[] = [];
  for (let r = 1; r < rows.length; r++) {
    const line = rows[r];
    // allow extra columns but require at least header count
    if (line.length < header.length) continue;

    out.push({
      scenario_id: (line[idx("scenario_id")] ?? "").trim(),
      channel: (line[idx("channel")] ?? "").trim(),
      input_json: (line[idx("input_json")] ?? "").trim(),
      expected_gate: (line[idx("expected_gate")] ?? "").trim(),
      expected_contains: (line[idx("expected_contains")] ?? "").trim(),
    });
  }
  return out.filter((x) => x.scenario_id && x.channel && x.input_json && x.expected_gate);
}

function splitExpectedContains(s: string): string[] {
  if (!s) return [];
  return s
    .split(";")
    .map((t) => t.trim())
    .filter(Boolean);
}

async function fetchWithTimeout(url: string, init: RequestInit, timeoutMs: number): Promise<Response> {
  const ac = new AbortController();
  const to = setTimeout(() => ac.abort(), timeoutMs);
  try {
    const res = await fetch(url, { ...init, signal: ac.signal });
    return res;
  } finally {
    clearTimeout(to);
  }
}

function extractTokensFromResponse(resp: any): Set<string> {
  const tokens = new Set<string>();

  // Common fields you might have
  const add = (v: unknown) => {
    if (!v) return;
    if (Array.isArray(v)) v.forEach(add);
    else if (typeof v === "string") tokens.add(v);
    else if (typeof v === "object") {
      // add id-like fields
      for (const [k, val] of Object.entries(v as any)) {
        if (k.toLowerCase().includes("id") && typeof val === "string") tokens.add(val);
      }
    }
  };

  add(resp?.gate);
  add(resp?.decision);
  add(resp?.status);

  add(resp?.matchedRedFlags);
  add(resp?.matchedRedFlagIds);
  add(resp?.redFlags);
  add(resp?.riskFlags);

  add(resp?.selectedInterventions);
  add(resp?.interventions);

  add(resp?.bundles);
  add(resp?.suggestedBundles);
  add(resp?.followupBundles);

  add(resp?.trace?.firedRuleIds);
  add(resp?.trace?.matchedRuleIds);
  add(resp?.trace?.ruleIds);
  add(resp?.trace);

  // also parse any list of actions
  add(resp?.actions);
  add(resp?.outputs);

  return tokens;
}

function responseContainsAll(resp: any, expectedTokens: string[]): { ok: boolean; missing: string[] } {
  if (expectedTokens.length === 0) return { ok: true, missing: [] };

  const tokenSet = extractTokensFromResponse(resp);
  const respStr = JSON.stringify(resp);

  const missing: string[] = [];
  for (const t of expectedTokens) {
    if (tokenSet.has(t)) continue;
    if (respStr.includes(t)) continue;
    missing.push(t);
  }
  return { ok: missing.length === 0, missing };
}

async function runOne(
  args: Args,
  row: Row
): Promise<{ ok: boolean; scenario_id: string; reason?: string; details?: any }> {
  let input: any;
  try {
    input = JSON.parse(row.input_json);
  } catch (e) {
    return { ok: false, scenario_id: row.scenario_id, reason: `Invalid JSON in input_json: ${e}` };
  }

  const url = new URL(args.endpoint, args.baseUrl);
  url.searchParams.set("channel", row.channel);

  const res = await fetchWithTimeout(
    url.toString(),
    {
      method: "POST",
      headers: { "content-type": "application/json" },
      body: JSON.stringify({
        scenarioId: row.scenario_id,
        input,
      }),
    },
    args.timeoutMs
  );

  if (!res.ok) {
    const text = await res.text().catch(() => "");
    return {
      ok: false,
      scenario_id: row.scenario_id,
      reason: `HTTP ${res.status} ${res.statusText}: ${text.slice(0, 500)}`,
    };
  }

  const resp = await res.json();

  const gate = resp?.gate ?? resp?.decision ?? resp?.status;
  if (String(gate).toUpperCase() !== row.expected_gate.toUpperCase()) {
    return {
      ok: false,
      scenario_id: row.scenario_id,
      reason: `Gate mismatch. expected=${row.expected_gate} got=${gate}`,
      details: { gate, expected: row.expected_gate, channel: row.channel },
    };
  }

  const expectedTokens = splitExpectedContains(row.expected_contains);
  const contains = responseContainsAll(resp, expectedTokens);
  if (!contains.ok) {
    return {
      ok: false,
      scenario_id: row.scenario_id,
      reason: `Missing expected tokens: ${contains.missing.join(", ")}`,
      details: { missing: contains.missing, channel: row.channel },
    };
  }

  return { ok: true, scenario_id: row.scenario_id };
}

async function runWithConcurrency<T, R>(
  items: T[],
  concurrency: number,
  worker: (t: T) => Promise<R>
): Promise<R[]> {
  const results: R[] = [];
  let idx = 0;

  const runners = Array.from({ length: Math.max(1, concurrency) }, async () => {
    while (true) {
      const myIdx = idx++;
      if (myIdx >= items.length) return;
      const r = await worker(items[myIdx]);
      results[myIdx] = r;
    }
  });

  await Promise.all(runners);
  return results;
}

async function main() {
  const args = parseArgs(process.argv.slice(2));
  const csvAbs = path.resolve(args.csvPath);

  const raw = fs.readFileSync(csvAbs, "utf8");
  const rows = parseCsv(raw);

  console.log(`Loaded ${rows.length} scenarios from ${csvAbs}`);
  console.log(`POST ${new URL(args.endpoint, args.baseUrl).toString()}?channel=...`);
  console.log(`Concurrency=${args.concurrency} TimeoutMs=${args.timeoutMs}`);

  const results = await runWithConcurrency(rows, args.concurrency, async (row) => runOne(args, row));

  const failures = results.filter((r: any) => !r?.ok);
  const passes = results.length - failures.length;

  console.log(`\n✅ Passed: ${passes}/${results.length}`);
  if (failures.length) {
    console.log(`\n❌ Failures (${failures.length}):`);
    for (const f of failures as any[]) {
      console.log(`- [${f.scenario_id}] ${f.reason}`);
      if (f.details) console.log(`  details: ${JSON.stringify(f.details)}`);
      if (args.failFast) break;
    }
    process.exitCode = 1;
  } else {
    console.log("All scenarios passed 🎯");
  }
}

main().catch((e) => {
  console.error("Harness crashed:", e);
  process.exitCode = 1;
});
