🌐 1. MULTI-HOSPITAL COORDINATION ENGINE
server/coordination/hospitalRegistry.ts
export const hospitals = [
  { id: "H1", name: "NYC General", icuBeds: 5 },
  { id: "H2", name: "Columbia", icuBeds: 2 },
  { id: "H3", name: "Mount Sinai", icuBeds: 8 },
];

export function getAvailableHospital() {
  return hospitals.sort((a, b) => b.icuBeds - a.icuBeds)[0];
}
server/coordination/bedAllocator.ts
import { getAvailableHospital } from "./hospitalRegistry";

export function allocateICUBed(patient: any) {
  const hospital = getAvailableHospital();

  if (!hospital || hospital.icuBeds <= 0) {
    return { assigned: false, reason: "No ICU beds available" };
  }

  hospital.icuBeds -= 1;

  return {
    assigned: true,
    hospital: hospital.name,
  };
}
⚡ 2. REAL-TIME PATIENT STREAM (WebSockets)
server/streaming/websocketServer.ts
import { WebSocketServer } from "ws";

const wss = new WebSocketServer({ port: 8080 });

export function startPatientStream() {
  wss.on("connection", (ws) => {
    console.log("Client connected");

    const interval = setInterval(() => {
      const data = generateLivePatient();
      ws.send(JSON.stringify(data));
    }, 1000);

    ws.on("close", () => clearInterval(interval));
  });
}

function generateLivePatient() {
  return {
    id: Math.random().toString(36).substring(7),
    risk: Math.random(),
    vitals: {
      hr: 60 + Math.random() * 80,
      sbp: 90 + Math.random() * 60,
    },
  };
}

🧾 4. FDA VALIDATION ENGINE
server/fda/validationEngine.ts
export function runFDAVerification(results: any[]) {
  const accuracy =
    results.filter((r) => r.correct).length / results.length;

  const meetsThreshold = accuracy >= 0.8;

  return {
    accuracy,
    meetsThreshold,
    classification: "SaMD Class II",
  };
}
server/fda/dossierGenerator.ts
import fs from "fs";

export function generateDossier(report: any) {
  const content = `
FDA SaMD REPORT

Accuracy: ${report.accuracy}
Threshold Met: ${report.meetsThreshold}

Risk Class: ${report.classification}
Timestamp: ${new Date().toISOString()}
`;

  fs.writeFileSync("FDA_DOSSIER.txt", content);

  return "FDA_DOSSIER.txt";
}
🔗 5. SMART ON FHIR WRITE-BACK
server/fhir/fhirClient.ts
import fetch from "node-fetch";

const FHIR_BASE = "https://api.epic.com/fhir";

export async function writeObservation(patientId: string, data: any) {
  const res = await fetch(`${FHIR_BASE}/Observation`, {
    method: "POST",
    headers: {
      "Content-Type": "application/fhir+json",
      Authorization: `Bearer ${process.env.FHIR_TOKEN}`,
    },
    body: JSON.stringify({
      resourceType: "Observation",
      subject: { reference: `Patient/${patientId}` },
      valueString: JSON.stringify(data),
    }),
  });

  return res.json();
}
🌐 6. COMMAND CENTER API
server/api/commandCenterFull.routes.ts
import express from "express";
import { runDigitalTwin } from "../digitalTwin/simulator";
import { simulatePatients } from "../simulation/multiPatientSimulator";
import { runFDAVerification } from "../fda/validationEngine";
import { generateDossier } from "../fda/dossierGenerator";
import { writeObservation } from "../fhir/fhirClient";

const router = express.Router();

router.get("/twin", async (_, res) => {
  const data = await runDigitalTwin();
  res.json(data);
});

router.post("/fda", async (req, res) => {
  const results = await simulatePatients(500);
  const report = runFDAVerification(results);
  const file = generateDossier(report);

  res.json({ report, file });
});

router.post("/fhir/write", async (req, res) => {
  const { patientId, data } = req.body;

  const result = await writeObservation(patientId, data);

  res.json(result);
});

export default router;

import { useEffect, useState } from "react";

export default function CommandCenter() {
  const [patients, setPatients] = useState([]);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:8080");

    ws.onmessage = (msg) => {
      const data = JSON.parse(msg.data);
      setPatients((prev) => [data, ...prev.slice(0, 20)]);
    };

    return () => ws.close();
  }, []);

  return (
    <div>
      <h1>🏥 Hospital Command Center</h1>

      {patients.map((p: any) => (
        <div key={p.id}>
          Patient {p.id} | Risk: {p.risk.toFixed(2)}
        </div>
      ))}
    </div>
  );
}
