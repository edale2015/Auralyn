import { normalizeDiagnosis } from "../ontology/diagnosisOntology";


export function normalizeChiefComplaint(text: string) {
  const lower = text.toLowerCase();


  if (lower.includes("sore throat")) return "J02.0";
  if (lower.includes("cough")) return "R05";
  if (lower.includes("chest pain")) return "R07.9";


  return "UNKNOWN";
}


export function structuredIntake(input: any) {
  return {
    ...input,
    complaintCode: normalizeChiefComplaint(input.freeText || ""),
  };
}
2. GOVERNED RLHF SYSTEM (VERSIONED + APPROVAL REQUIRED)
learning/versionedRLHF.ts
let pendingUpdates: any[] = [];
let modelVersions: any[] = [];


export function proposeWeightUpdate(update: any) {
  pendingUpdates.push(update);
}


export function approveUpdates() {
  const newVersion = {
    id: "v" + Date.now(),
    updates: pendingUpdates,
    approvedAt: new Date().toISOString(),
  };


  modelVersions.push(newVersion);
  pendingUpdates = [];


  return newVersion;
}


export function rollbackVersion(versionId: string) {
  modelVersions = modelVersions.filter(v => v.id !== versionId);
}

4. VALIDATION STUDY (PROSPECTIVE READY)
fda/prospectiveStudy.ts
export async function runProspectiveStudy(cases: any[], runFlow: any) {
  const results = [];


  for (const c of cases) {
    const output = await runFlow(c.input);


    results.push({
      input: c.input,
      predicted: output.diagnosis,
      actual: c.actualOutcome,
      correct: output.diagnosis === c.actualOutcome,
    });
  }


  return results;
}
5. BIAS / EQUITY ANALYSIS
fda/biasAnalysis.ts
export function analyzeBias(results: any[]) {
  const groups: any = {};


  results.forEach(r => {
    const key = r.demographic || "unknown";
    if (!groups[key]) groups[key] = [];
    groups[key].push(r.correct);
  });


  const output: any = {};


  for (const g in groups) {
    const arr = groups[g];
    output[g] = arr.filter(Boolean).length / arr.length;
  }


  return output;
}
