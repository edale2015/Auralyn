Purpose: generate synthetic cases based on graph gaps instead of random complaints.

Example:

If the graph gap detector reports:

dizziness → missing stroke risk coverage

The simulation engine generates stroke-like dizziness cases.

Gap Target Case Factory

Creates cases based on graph weaknesses.

server/simulation/gapTargetCaseFactory.ts

export function generateGapTargetCase(gap:any){

 if(gap.problem==="missing_protocol"){

  return{
   complaint:gap.complaint,
   features:{
     severity:"moderate",
     durationDays:3
   }
  }

 }

 if(gap.problem==="no_engine_assigned"){

  return{
   skill:gap.skill,
   features:{
     edgeCase:true
   }
  }

 }

 if(gap.problem==="no_disposition_path"){

  return{
   diagnosis:gap.diagnosis,
   features:{
     severe:true
   }
  }

 }

 return{
  complaint:"generic",
  features:{}
 }

}
Simulation Priority Planner

Prioritizes which graph gaps to simulate first.

server/simulation/simulationPriorityPlanner.ts

export function prioritizeSimulationTargets(gaps:any[]){

 return gaps.sort((a,b)=>{

   const weight = {
     missing_protocol:5,
     no_engine_assigned:4,
     no_disposition_path:3,
     missing_skill_mapping:2
   }

   return weight[b.problem]-weight[a.problem]

 })

}
Graph-Driven Simulation Engine

server/simulation/graphDrivenSimulationEngine.ts

import {detectGraphGaps} from "../knowledge/graphGapDetector"
import {generateGapTargetCase} from "./gapTargetCaseFactory"
import {prioritizeSimulationTargets} from "./simulationPriorityPlanner"

export function runGraphDrivenSimulation(){

 const gaps = detectGraphGaps()

 const prioritized = prioritizeSimulationTargets(gaps)

 const simulations:any[]=[]

 prioritized.slice(0,20).forEach(gap=>{

   const simCase = generateGapTargetCase(gap)

   simulations.push({
     gap,
     simCase
   })

 })

 return simulations

}
