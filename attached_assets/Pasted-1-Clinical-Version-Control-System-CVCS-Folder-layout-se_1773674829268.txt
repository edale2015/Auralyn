2️⃣ Version Store

Stores versions in memory (later can move to DB).

server/versioning/clinicalVersionStore.ts

import {ClinicalVersion} from "./clinicalVersionTypes"

const versionStore:ClinicalVersion[]=[]

export function addVersion(v:ClinicalVersion){
  versionStore.push(v)
}

export function listVersions(){
  return versionStore
}

export function getVersion(id:string){
  return versionStore.find(v=>v.id===id)
}
3️⃣ Version Manager

Creates a new version when sheets are ingested.

server/versioning/clinicalVersionManager.ts

import crypto from "crypto"
import {addVersion} from "./clinicalVersionStore"

export function createClinicalVersion(config:any){

 const versionId=`cv_${Date.now()}`

 const sheetsHash = crypto
   .createHash("sha256")
   .update(JSON.stringify(config.sheets))
   .digest("hex")

 const graphHash = crypto
   .createHash("sha256")
   .update(JSON.stringify(config.graph))
   .digest("hex")

 const version = {
   id:versionId,
   createdAt:Date.now(),
   createdBy:config.user || "system",
   sheetsHash,
   graphHash,
   sheetFiles:config.files || [],
   changeSummary:config.summary
 }

 addVersion(version)

 return version
}
4️⃣ Version Diff Engine

Shows what changed between versions.

server/versioning/clinicalVersionDiff.ts

import {getVersion} from "./clinicalVersionStore"

export function diffVersions(a:string,b:string){

 const v1=getVersion(a)
 const v2=getVersion(b)

 if(!v1 || !v2)
  return null

 const diff={

  from:a,
  to:b,
  added:0,
  removed:0,
  modified:0,
  affectedSheets:[]

 }

 if(v1.sheetsHash!==v2.sheetsHash)
   diff.modified++

 if(v1.graphHash!==v2.graphHash)
   diff.modified++

 return diff

}
5️⃣ Rollback Manager

Allows safe rollback.

server/versioning/clinicalRollbackManager.ts

import {getVersion} from "./clinicalVersionStore"

let currentVersion:string|null=null

export function deployVersion(id:string){

 const v=getVersion(id)

 if(!v) return null

 currentVersion=id

 return{
  deployed:id
 }

}

export function rollbackVersion(id:string){

 const v=getVersion(id)

 if(!v) return null

 currentVersion=id

 return{
  rollback:id
 }

}
6️⃣ Version Control API

server/routes/clinicalVersionRoutes.ts

import express from "express"
import {listVersions} from "../versioning/clinicalVersionStore"
import {diffVersions} from "../versioning/clinicalVersionDiff"
import {deployVersion,rollbackVersion} from "../versioning/clinicalRollbackManager"

const router=express.Router()

router.get("/clinical-versions",(req,res)=>{
 res.json(listVersions())
})

router.get("/clinical-versions/diff",(req,res)=>{
 const {from,to}=req.query
 res.json(diffVersions(from as string,to as string))
})

router.post("/clinical-versions/deploy",(req,res)=>{
 res.json(deployVersion(req.body.id))
})

router.post("/clinical-versions/rollback",(req,res)=>{
 res.json(rollbackVersion(req.body.id))
})

export default router
2. Clinical Change Timeline

This visualizes the entire history of clinical changes.

Timeline Engine

import {listVersions} from "./clinicalVersionStore"

export function buildClinicalTimeline(){

 const versions=listVersions()

 return versions.map(v=>({

  version:v.id,
  time:v.createdAt,
  description:v.changeSummary || "Sheet update"

 }))

}
