# similarity/feature_extractor.py

class FeatureExtractor:

    def extract(self, case):

        features = set()

        features.add(f"complaint:{case['complaint']}")

        for f in case.get("key_features", []):
            features.add(f"symptom:{f}")

        if "age" in case:
            age = case["age"]

            if age < 18:
                features.add("age_group:pediatric")
            elif age > 65:
                features.add("age_group:elderly")
            else:
                features.add("age_group:adult")

        if "sex" in case:
            features.add(f"sex:{case['sex']}")

        return features
Similarity engine
# similarity/case_similarity_engine.py

from similarity.feature_extractor import FeatureExtractor


class CaseSimilarityEngine:

    def __init__(self, cases):

        self.extractor = FeatureExtractor()

        self.cases = cases

        self.case_features = {
            c["case_id"]: self.extractor.extract(c)
            for c in cases
        }

    def similarity(self, f1, f2):

        intersection = len(f1 & f2)
        union = len(f1 | f2)

        if union == 0:
            return 0

        return intersection / union

    def find_similar(self, query_case, top_k=5):

        query_features = self.extractor.extract(query_case)

        scores = []

        for case in self.cases:

            cf = self.case_features[case["case_id"]]

            score = self.similarity(query_features, cf)

            scores.append((score, case))

        scores.sort(reverse=True, key=lambda x: x[0])

        return scores[:top_k]

import random
import uuid
from dataset.templates import TEMPLATES


def generate_case(complaint, template):

    return {
        "case_id": str(uuid.uuid4()),
        "complaint": complaint,
        "age": random.randint(1,90),
        "sex": random.choice(["male","female"]),
        "key_features": template["key_features"],
        "expected_differential": template["expected_differential"],
        "expected_disposition": template["expected_disposition"]
    }


def generate_dataset(n=300):

    dataset = []

    complaints = list(TEMPLATES.keys())

    while len(dataset) < n:

        c = random.choice(complaints)

        template = random.choice(TEMPLATES[c])

        dataset.append(generate_case(c, template))

    return dataset

from sentence_transformers import SentenceTransformer


class EmbeddingModel:

    def __init__(self):

        self.model = SentenceTransformer(
            "all-MiniLM-L6-v2"
        )

    def embed_case(self, case):

        text = case["complaint"]

        text += " " + " ".join(case.get("key_features", []))

        return self.model.encode(text)
Vector store (FAISS)
# similarity/vector_store.py

import faiss
import numpy as np


class VectorStore:

    def __init__(self, dimension):

        self.index = faiss.IndexFlatL2(dimension)

        self.cases = []

    def add(self, embedding, case):

        self.index.add(np.array([embedding]))

        self.cases.append(case)

    def search(self, embedding, k=5):

        distances, indexes = self.index.search(
            np.array([embedding]),
            k
        )

        results = []

        for i in indexes[0]:

            if i < len(self.cases):
                results.append(self.cases[i])

        return results
Embedding similarity engine
# similarity/embedding_similarity_engine.py

from similarity.embedding_model import EmbeddingModel
from similarity.vector_store import VectorStore


class EmbeddingSimilarityEngine:

    def __init__(self, cases):

        self.model = EmbeddingModel()

        sample = self.model.embed_case(cases[0])

        dimension = len(sample)

        self.store = VectorStore(dimension)

        for case in cases:

            emb = self.model.embed_case(case)

            self.store.add(emb, case)

    def find_similar(self, query_case, top_k=5):

        emb = self.model.embed_case(query_case)

        return self.store.search(emb, top_k)

Layer 1 — Deterministic Safety Layer

Purpose: Never miss dangerous conditions

This layer cannot be modified automatically.

# hybrid_reasoning/safety_layer.py

class SafetyLayer:

    RED_FLAGS = {
        "chest_pain": [
            "radiates_left_arm",
            "diaphoresis",
            "shortness_of_breath"
        ],
        "sore_throat": [
            "drooling",
            "muffled_voice"
        ],
        "abdominal_pain": [
            "rebound_tenderness",
            "rigidity"
        ],
        "headache": [
            "worst_headache",
            "neurologic_deficit"
        ]
    }

    def check(self, case):

        complaint = case["complaint"]

        features = set(case.get("key_features", []))

        flags = self.RED_FLAGS.get(complaint, [])

        triggered = []

        for f in flags:
            if f in features:
                triggered.append(f)

        if triggered:

            return {
                "override": True,
                "disposition": "ER_NOW",
                "reason": triggered
            }

        return {
            "override": False
        }

class SimilarityLayer:

    def __init__(self, dataset):

        self.engine = EmbeddingSimilarityEngine(dataset)

    def rank_differentials(self, case):

        similar = self.engine.find_similar(case, top_k=5)

        scores = {}

        for c in similar:

            dx = c["expected_differential"][0]

            scores[dx] = scores.get(dx, 0) + 1

        ranked = sorted(scores, key=scores.get, reverse=True)

        return {
            "similar_cases": similar,
            "ranked_differential": ranked
        }

from hybrid_reasoning.safety_layer import SafetyLayer
from hybrid_reasoning.similarity_layer import SimilarityLayer
from hybrid_reasoning.llm_reasoning_layer import LLMReasoningLayer


class HybridClinicalReasoning:

    def __init__(self, dataset, llm):

        self.safety = SafetyLayer()

        self.similarity = SimilarityLayer(dataset)

        self.llm = LLMReasoningLayer(llm)

    def evaluate(self, case):

        # Layer 1: Safety rules

        safety_result = self.safety.check(case)

        if safety_result["override"]:

            return {
                "disposition": safety_result["disposition"],
                "reason": safety_result["reason"],
                "explanation": "Safety red flag triggered"
            }

        # Layer 2: Case memory reasoning

        similarity = self.similarity.rank_differentials(case)

        differential = similarity["ranked_differential"]

        # simple disposition logic

        if differential:

            top = differential[0]

            if top in ["acute_coronary_syndrome","pulmonary_embolism","appendicitis"]:

                disposition = "ER_NOW"

            elif top in ["pneumonia","strep_pharyngitis"]:

                disposition = "urgent_care"

            else:

                disposition = "home_care"

        else:

            disposition = "uncertain"

        # Layer 3: Explanation

        explanation = self.llm.explain(case, differential)

        return {
            "disposition": disposition,
            "differential": differential,
            "explanation": explanation,
            "similar_cases": similarity["similar_cases"]
        }

1. Probabilistic Diagnosis Engine

This maintains conditional probabilities between symptoms and diagnoses.

# probabilistic/probabilistic_engine.py

import math
from collections import defaultdict


class ProbabilisticDiagnosisEngine:

    def __init__(self):
        self.symptom_given_dx = defaultdict(lambda: defaultdict(float))
        self.dx_prior = defaultdict(float)

    def train(self, case):

        dx = case["expected_differential"][0]
        symptoms = case["key_features"]

        self.dx_prior[dx] += 1

        for s in symptoms:
            self.symptom_given_dx[dx][s] += 1

    def probability(self, symptoms):

        scores = {}

        for dx in self.dx_prior:

            score = math.log(self.dx_prior[dx] + 1)

            for s in symptoms:
                score += math.log(self.symptom_given_dx[dx].get(s, 0) + 1)

            scores[dx] = score

        # normalize
        max_score = max(scores.values())
        exp_scores = {k: math.exp(v - max_score) for k, v in scores.items()}

        total = sum(exp_scores.values())

        probs = {k: v / total for k, v in exp_scores.items()}

        return probs
2. Uncertainty Estimator

If diagnosis probabilities are close together, the system should ask more questions.

# probabilistic/uncertainty_estimator.py

import numpy as np


class UncertaintyEstimator:

    def entropy(self, probs):

        values = list(probs.values())

        entropy = -sum(p * np.log(p + 1e-9) for p in values)

        return entropy

    def is_uncertain(self, probs):

        ent = self.entropy(probs)

        return ent > 1.2
3. Active Question Selection

Selects the most informative next question.

# probabilistic/question_information_gain.py

class QuestionInformationGain:

    def __init__(self, question_bank):

        self.question_bank = question_bank

    def next_question(self, symptoms):

        best_q = None
        best_score = -1

        for q in self.question_bank:

            gain = self.estimate_gain(q, symptoms)

            if gain > best_score:
                best_score = gain
                best_q = q

        return best_q

    def estimate_gain(self, question, symptoms):

        # simple heuristic
        if question not in symptoms:
            return 1
        return 0

from hybrid_reasoning.safety_layer import SafetyLayer
from hybrid_reasoning.similarity_layer import SimilarityLayer
from hybrid_reasoning.llm_reasoning_layer import LLMReasoningLayer

from probabilistic.probabilistic_engine import ProbabilisticDiagnosisEngine
from probabilistic.uncertainty_estimator import UncertaintyEstimator
from probabilistic.question_information_gain import QuestionInformationGain


class HybridClinicalReasoningV2:

    def __init__(self, dataset, llm, question_bank):

        self.safety = SafetyLayer()
        self.similarity = SimilarityLayer(dataset)
        self.llm = LLMReasoningLayer(llm)

        self.prob = ProbabilisticDiagnosisEngine()
        self.uncertainty = UncertaintyEstimator()
        self.question_engine = QuestionInformationGain(question_bank)

        for c in dataset:
            self.prob.train(c)

    def evaluate(self, case):

        safety = self.safety.check(case)

        if safety["override"]:
            return safety

        symptoms = case["key_features"]

        probs = self.prob.probability(symptoms)

        if self.uncertainty.is_uncertain(probs):

            next_q = self.question_engine.next_question(symptoms)

            return {
                "status": "need_more_info",
                "next_question": next_q,
                "probabilities": probs
            }

        ranked = sorted(probs, key=probs.get, reverse=True)

        explanation = self.llm.explain(case, ranked)

        return {
            "disposition": ranked[0],
            "probabilities": probs,
            "explanation": explanation
        }

1. Clinical Dataset Expansion Engine (1,000+ Cases)

Instead of manually writing cases, generate structured variations.

# dataset/dataset_expander.py

import random
import uuid

class DatasetExpander:

    def __init__(self, base_templates):
        self.templates = base_templates

    def expand(self, target_size=1000):

        dataset = []

        complaints = list(self.templates.keys())

        while len(dataset) < target_size:

            complaint = random.choice(complaints)

            template = random.choice(self.templates[complaint])

            case = {
                "case_id": str(uuid.uuid4()),
                "complaint": complaint,
                "age": random.randint(1, 90),
                "sex": random.choice(["male","female"]),
                "key_features": template["key_features"],
                "expected_differential": template["expected_differential"],
                "expected_disposition": template["expected_disposition"]
            }

            dataset.append(case)

        return dataset

Usage:

expander = DatasetExpander(TEMPLATES)
dataset = expander.expand(1000)
2. Symptom Timeline Modeling

Telemedicine cases evolve over time.

# clinical/symptom_timeline.py

class SymptomTimeline:

    def __init__(self):
        self.timeline = {}

    def add_event(self, case_id, day, symptom):

        if case_id not in self.timeline:
            self.timeline[case_id] = []

        self.timeline[case_id].append({
            "day": day,
            "symptom": symptom
        })

    def progression(self, case_id):

        return sorted(self.timeline.get(case_id, []), key=lambda x: x["day"])

Example:

5. Model Calibration Check

Ensures probability outputs are trustworthy.

# monitoring/calibration_checker.py

class CalibrationChecker:

    def __init__(self):
        self.records = []

    def record(self, predicted_prob, actual_outcome):

        self.records.append((predicted_prob, actual_outcome))

    def brier_score(self):

        total = 0

        for p, y in self.records:

            total += (p - y)**2

        return total / len(self.records)

Low Brier score = good calibration.

Fixing the Biggest Medical AI Weakness

The biggest weakness in medical triage systems is:

Overconfidence with incomplete data

The fix is an Incomplete Data Safety Layer.

Incomplete Data Safety Layer
# safety/incomplete_data_guard.py

class IncompleteDataGuard:

    REQUIRED_FIELDS = {
        "chest_pain":["radiation","shortness_of_breath"],
        "sore_throat":["fever","drooling"],
        "abdominal_pain":["location","fever"]
    }

    def check(self, case):

        complaint = case["complaint"]

        features = set(case.get("key_features",[]))

        required = self.REQUIRED_FIELDS.get(complaint,[])

        missing = []

        for r in required:
            if r not in features:
                missing.append(r)

        if missing:

            return {
                "status":"insufficient_data",
                "missing_questions": missing
            }

        return {"status":"sufficient"}

This prevents the system from prematurely triaging.

# hybrid_reasoning/hybrid_controller_v3.py

def evaluate(self, case):

    safety = self.safety.check(case)

    if safety["override"]:
        return safety

    data_check = self.data_guard.check(case)

    if data_check["status"] == "insufficient_data":

        return {
            "status":"need_more_information",
            "questions": data_check["missing_questions"]
        }

    probabilities = self.prob_engine.probability(case["key_features"])

    ranked = sorted(probabilities, key=probabilities.get, reverse=True)

    explanation = self.llm.explain(case, ranked)

    return {
        "differential": ranked,
        "probabilities": probabilities,
        "explanation": explanation
    }
