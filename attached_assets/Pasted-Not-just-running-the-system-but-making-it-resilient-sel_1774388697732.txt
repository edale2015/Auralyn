export function selectRegion(userRegion: string) {
  const healthy = regions.filter(r => r.health === "healthy");

  if (!healthy.length) throw new Error("NO REGIONS AVAILABLE");

  healthy.sort((a, b) => a.latencyMs - b.latencyMs);

  return healthy[0];
}

export async function auditAgents() {
  const agents = getAgents();
  const findings = [];

  for (const a of agents) {
    if (!a.lastAction) {
      findings.push({
        agent: a.id,
        issue: "NO_ACTIVITY",
        severity: "MEDIUM"
      });
    }

    if (a.health === "critical") {
      findings.push({
        agent: a.id,
        issue: "CRITICAL_HEALTH",
        severity: "HIGH"
      });
    }
  }

  return findings;
}

type MetricHistory = {
  latency: number[];
  errorRate: number[];
};

const history: MetricHistory = {
  latency: [],
  errorRate: []
};
📁 Record Metrics
export function recordMetrics(metrics: any) {
  history.latency.push(metrics.avgLatency);
  history.errorRate.push(metrics.errorRate);

  if (history.latency.length > 50) {
    history.latency.shift();
    history.errorRate.shift();
  }
}
📁 Trend Analysis
export function predictFailure() {
  const latencyTrend =
    history.latency[history.latency.length - 1] -
    history.latency[0];

  const errorTrend =
    history.errorRate[history.errorRate.length - 1] -
    history.errorRate[0];

  if (latencyTrend > 500 || errorTrend > 0.02) {
    return {
      predicted: true,
      reason: "Upward trend detected"
    };
  }

  return { predicted: false };
}
