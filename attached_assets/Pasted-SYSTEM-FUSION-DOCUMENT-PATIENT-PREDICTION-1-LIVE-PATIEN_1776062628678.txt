// server/realtime/patientStream.ts
import WebSocket from "ws";

const wss = new WebSocket.Server({ port: 8080 });

export const patients: Record<string, any> = {};

wss.on("connection", (ws) => {
  ws.on("message", (message) => {
    const data = JSON.parse(message.toString());

    const { patientId } = data;

    patients[patientId] = {
      ...patients[patientId],
      ...data,
      lastUpdated: Date.now(),
    };

    broadcastUpdate(patientId);
  });
});

function broadcastUpdate(patientId: string) {
  const payload = JSON.stringify({
    type: "PATIENT_UPDATE",
    patient: patients[patientId],
  });

  wss.clients.forEach((client) => {
    if (client.readyState === WebSocket.OPEN) {
      client.send(payload);
    }
  });
}

// server/clinical/sepsisPredictor.ts

export function calculateNEWS2(vitals: any) {
  let score = 0;

  if (vitals.respRate >= 25) score += 3;
  else if (vitals.respRate >= 21) score += 2;

  if (vitals.oxygenSat < 92) score += 3;

  if (vitals.temp > 38 || vitals.temp < 36) score += 1;

  if (vitals.sbp < 90) score += 3;
  else if (vitals.sbp < 100) score += 2;

  if (vitals.hr > 130) score += 3;
  else if (vitals.hr > 110) score += 2;

  if (vitals.confusion) score += 3;

  return score;
}

export function calculateQSOFA(vitals: any) {
  let score = 0;

  if (vitals.sbp < 100) score++;
  if (vitals.respRate >= 22) score++;
  if (vitals.confusion) score++;

  return score;
}
Trend-aware risk
export function calculateSepsisRisk(patient: any) {
  const news2 = calculateNEWS2(patient.vitals);
  const qsofa = calculateQSOFA(patient.vitals);

  const lactateTrend =
    patient.labs?.lactateHistory?.slice(-2) || [];

  let worseningLactate = false;

  if (lactateTrend.length === 2) {
    worseningLactate = lactateTrend[1] > lactateTrend[0];
  }

  let risk = "LOW";

  if (news2 >= 7 || qsofa >= 2 || worseningLactate) {
    risk = "HIGH";
  } else if (news2 >= 5) {
    risk = "MEDIUM";
  }

  return {
    risk,
    news2,
    qsofa,
    worseningLactate,
  };
}

export function generateAlerts(evaluation: any) {
  const alerts = [];

  if (evaluation.sepsis.risk === "HIGH") {
    alerts.push({
      type: "SEPSIS_ALERT",
      message: "High risk of sepsis detected",
      priority: "CRITICAL",
    });
  }

  if (evaluation.council.consensus.confidence > 0.8) {
    alerts.push({
      type: "ACTION",
      message: evaluation.council.consensus.finalRecommendation,
    });
  }

  return alerts;
}
