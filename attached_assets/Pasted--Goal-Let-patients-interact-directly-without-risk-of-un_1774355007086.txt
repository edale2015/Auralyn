🧠 4. Safe Patient Flow Wrapper
// server/patient/patientFlow.ts

import { runSystem } from "../brain/fullLoop";
import { isWithinScope } from "./scopeGuard";
import { needsEscalation } from "./escalation";

export async function runPatientFlow(input: any) {

  // 1. scope check
  if (!isWithinScope(input)) {
    return {
      status: "physician_required",
      reason: "out_of_scope"
    };
  }

  // 2. run system
  const result = await runSystem(input);

  // 3. escalation
  if (needsEscalation(result)) {
    return {
      status: "physician_review",
      result
    };
  }

  // 4. safe output
  return {
    status: "self_service_complete",
    plan: result.decision
  };
}

export function allowAutonomy(decision: any) {
  if (decision.riskScore > 0.25) return false;
  if (decision.uncertainty > 0.2) return false;
  if (!decision.workflow) return false;

  return true;
}

// server/brain/autonomousSafe.ts

import { allowAutonomy } from "../clinical/autonomyGate";

export async function runSafeAutonomous(input: any) {
  const result = await runSystem(input);

  if (!allowAutonomy(result.decision)) {
    return {
      status: "physician_review_required",
      result
    };
  }

  return {
    status: "autonomous_action",
    result
  };
}
