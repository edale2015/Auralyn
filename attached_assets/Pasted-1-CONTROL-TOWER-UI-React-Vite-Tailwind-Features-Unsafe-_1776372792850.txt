import { useEffect, useState } from "react";

type Patient = {
  patientId: string;
  risk: number;
  topDiagnosis: string;
  unsafe: boolean;
  hallucinationFlags: string[];
};

export default function ControlTower() {
  const [patients, setPatients] = useState<Patient[]>([]);

  useEffect(() => {
    fetch("/api/control-tower/rank")
      .then(res => res.json())
      .then(setPatients);
  }, []);

  return (
    <div className="p-6 bg-gray-900 text-white min-h-screen">
      <h1 className="text-3xl font-bold mb-6">🧠 Control Tower</h1>

      {/* 🔥 Heatmap */}
      <div className="grid grid-cols-4 gap-4 mb-8">
        {patients.map(p => (
          <div
            key={p.patientId}
            className={`p-4 rounded-lg ${
              p.unsafe
                ? "bg-red-600"
                : p.risk > 3
                ? "bg-orange-500"
                : "bg-green-600"
            }`}
          >
            <div className="text-lg font-bold">{p.patientId}</div>
            <div>Risk: {p.risk.toFixed(2)}</div>
            <div>{p.topDiagnosis}</div>
          </div>
        ))}
      </div>

      {/* 🚨 Hallucination Flags */}
      <div className="mb-8">
        <h2 className="text-xl font-semibold mb-2">🚨 Hallucination Alerts</h2>
        {patients
          .filter(p => p.hallucinationFlags.length)
          .map(p => (
            <div key={p.patientId} className="bg-red-800 p-3 mb-2 rounded">
              <b>{p.patientId}</b>: {p.hallucinationFlags.join(", ")}
            </div>
          ))}
      </div>

      {/* 🧠 Patient Ranking */}
      <table className="w-full text-left border border-gray-700">
        <thead>
          <tr className="bg-gray-800">
            <th className="p-2">Patient</th>
            <th>Risk</th>
            <th>Diagnosis</th>
            <th>Status</th>
          </tr>
        </thead>
        <tbody>
          {patients.map(p => (
            <tr key={p.patientId} className="border-t border-gray-700">
              <td className="p-2">{p.patientId}</td>
              <td>{p.risk.toFixed(2)}</td>
              <td>{p.topDiagnosis}</td>
              <td>
                {p.unsafe ? (
                  <span className="text-red-400">⚠ Review</span>
                ) : (
                  <span className="text-green-400">OK</span>
                )}
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

function generatePE(n: number) {
  return Array.from({ length: n }, (_, i) => ({
    id: `pe_${i}`,
    complaint: "sob",
    observations: [
      { feature: "sob", value: true },
      { feature: "tachycardia", value: Math.random() > 0.2 },
      { feature: "pleuritic_pain", value: Math.random() > 0.4 },
      { feature: "normal_o2", value: Math.random() > 0.7 }, // adversarial
    ],
    expectedDisposition: "ed",
    minimumSafeDisposition: "ed",
  }));
}

function generateStroke(n: number) {
  return Array.from({ length: n }, (_, i) => ({
    id: `stroke_${i}`,
    complaint: "neuro",
    observations: [
      { feature: "facial_droop", value: Math.random() > 0.3 },
      { feature: "speech", value: true },
      { feature: "arm_weakness", value: Math.random() > 0.2 },
    ],
    expectedDisposition: "call_911",
    minimumSafeDisposition: "call_911",
  }));
}

function generatePeds(n: number) {
  return Array.from({ length: n }, (_, i) => ({
    id: `peds_${i}`,
    complaint: "fever",
    observations: [
      { feature: "fever", value: true },
      { feature: "lethargy", value: Math.random() > 0.5 },
      { feature: "normal_activity", value: Math.random() > 0.5 }, // contradiction
    ],
    expectedDisposition: "ed",
  }));
}

import fs from "fs";

export function generateFDAReport(data: any) {
  const report = {
    timestamp: new Date().toISOString(),

    metrics: {
      passRate: data.passRate,
      unsafeUndercalls: data.unsafeUndercalls,
      calibrationError: data.calibrationError,
    },

    safety: {
      hallucinationBlocks: data.hallucinationBlocks,
      escalationRate: data.escalationRate,
    },

    audit: {
      totalCases: data.total,
      failures: data.failed,
    },

    justification: generateJustification(data),
  };

  fs.writeFileSync(
    "./fda_report.json",
    JSON.stringify(report, null, 2)
  );

  return report;
}
📁 server/fda/justification.ts
export function generateJustification(data: any) {
  const lines = [];

  if (data.unsafeUndercalls === 0) {
    lines.push("No unsafe undercalls observed.");
  }

  if (data.passRate > 0.9) {
    lines.push("High validation pass rate.");
  }

  if (data.calibrationError < 0.1) {
    lines.push("Model is well-calibrated.");
  }

  lines.push("System includes multi-layer hallucination detection.");
  lines.push("All unsafe outputs escalate to physician review.");

  return lines;
}

import crypto from "crypto";

export function buildAuditChain(entries: any[]) {
  let prevHash = "GENESIS";

  return entries.map(e => {
    const hash = crypto
      .createHash("sha256")
      .update(prevHash + JSON.stringify(e))
      .digest("hex");

    prevHash = hash;

    return {
      ...e,
      hash,
    };
  });
}
