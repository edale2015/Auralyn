import { db } from "../db";
import { contracts, claims, outcomes } from "../db/schema";

export type NegotiationStrategy =
  | "anchor_high"
  | "value_based"
  | "bundled_rate"
  | "risk_share"
  | "standard";

export interface ContractScore {
  payerId: string;
  avgReimbursement: number;
  denialRate: number;
  outcomeScore: number;
  score: number;
  recommendedStrategy: NegotiationStrategy;
}

export async function scoreContract(payerId: string): Promise<ContractScore> {
  const claimData = await db.select().from(claims).where({ payerId });
  const outcomeData = await db.select().from(outcomes).where({ payerId });

  const avgReimbursement =
    claimData.reduce((sum, c) => sum + c.amountPaid, 0) / (claimData.length || 1);

  const denialRate =
    claimData.filter(c => c.status === "denied").length / (claimData.length || 1);

  const outcomeScore =
    outcomeData.reduce((sum, o) => sum + o.qualityScore, 0) /
    (outcomeData.length || 1);

  const score =
    avgReimbursement * 0.4 +
    (1 - denialRate) * 0.3 +
    outcomeScore * 0.3;

  let strategy: NegotiationStrategy = "standard";

  if (score > 0.8) strategy = "anchor_high";
  else if (outcomeScore > 0.85) strategy = "value_based";
  else if (denialRate > 0.2) strategy = "bundled_rate";
  else if (score < 0.5) strategy = "risk_share";

  return {
    payerId,
    avgReimbursement,
    denialRate,
    outcomeScore,
    score,
    recommendedStrategy: strategy,
  };
}

export interface QualityMetric {
  name: string;
  numerator: number;
  denominator: number;
  rate: number;
}

export async function computeHEDISMetrics() {
  const data = await db.select().from(encounters);

  const total = data.length;

  const followUpCompleted = data.filter(
    e => e.followUpCompleted === true
  ).length;

  const antibioticAppropriate = data.filter(
    e => e.antibioticAppropriate === true
  ).length;

  const metrics: QualityMetric[] = [
    {
      name: "Follow-up Completion Rate",
      numerator: followUpCompleted,
      denominator: total,
      rate: followUpCompleted / (total || 1),
    },
    {
      name: "Appropriate Antibiotic Use",
      numerator: antibioticAppropriate,
      denominator: total,
      rate: antibioticAppropriate / (total || 1),
    },
  ];

  return metrics;
}

export interface ClinicianPerformance {
  clinicianId: string;
  totalCases: number;
  avgDecisionTime: number;
  escalationRate: number;
  accuracyScore: number;
  riskScore: number;
}

export async function computeClinicianPerformance(clinicianId: string) {
  const data = await db
    .select()
    .from(encounters)
    .where({ clinicianId });

  const totalCases = data.length;

  const avgDecisionTime =
    data.reduce((sum, e) => sum + e.decisionTimeMs, 0) /
    (totalCases || 1);

  const escalationRate =
    data.filter(e => e.escalated === true).length /
    (totalCases || 1);

  const accuracyScore =
    data.reduce((sum, e) => sum + e.correctDiagnosis, 0) /
    (totalCases || 1);

  const riskScore =
    escalationRate * 0.4 + (1 - accuracyScore) * 0.6;

  return {
    clinicianId,
    totalCases,
    avgDecisionTime,
    escalationRate,
    accuracyScore,
    riskScore,
  };
}

import { db } from "../db";
import { claims } from "../db/schema";

interface Prediction {
  probability: number;
  riskLevel: "low" | "medium" | "high";
  reasons: string[];
}

export async function predictDenial(payerId: string, cptCode: string): Promise<Prediction> {
  const history = await db.select().from(claims).where({ payerId, cptCode });

  const denialRate =
    history.filter(c => c.status === "denied").length / (history.length || 1);

  const reasons: string[] = [];

  if (denialRate > 0.3) reasons.push("High historical denial rate");
  if (cptCode.startsWith("9")) reasons.push("High-risk CPT family");
  if (history.length < 5) reasons.push("Low data confidence");

  let riskLevel: Prediction["riskLevel"] = "low";
  if (denialRate > 0.3) riskLevel = "high";
  else if (denialRate > 0.15) riskLevel = "medium";

  return {
    probability: denialRate,
    riskLevel,
    reasons,
  };
}

import { db } from "../db";
import { contracts } from "../db/schema";
import { predictDenial } from "../insurer/denialPredictor";

export async function optimizeReimbursement(payerId: string, cptOptions: string[]) {
  let bestOption = null;
  let bestScore = -Infinity;

  for (const cpt of cptOptions) {
    const contract = await db.select().from(contracts).where({ payerId, cptCode: cpt }).limit(1);

    if (!contract.length) continue;

    const rate = Number(contract[0].rate);
    const denial = await predictDenial(payerId, cpt);

    const expectedValue = rate * (1 - denial.probability);

    if (expectedValue > bestScore) {
      bestScore = expectedValue;
      bestOption = {
        cpt,
        expectedValue,
        denialRisk: denial,
      };
    }
  }

  return bestOption;
}

export async function computeOutcomeWeightedRevenue() {
  const data = await db.select().from(encounters);

  let totalRevenue = 0;
  let weightedRevenue = 0;

  for (const e of data) {
    const revenue = Number(e.reimbursement || 0);

    totalRevenue += revenue;
    weightedRevenue += revenue * outcome;
  }

  const avgOutcome = data.length
    ? data.reduce((sum, e) => sum + Number(e.outcomeScore || 0), 0) / data.length
    : 0;

  return {
    totalRevenue,
    weightedRevenue,
    avgOutcome,
    efficiencyScore: weightedRevenue / (totalRevenue || 1),
  };
}

📁 server/learning/rlhfEngine.ts
import { db } from "../db";
import { encounters } from "../db/schema";

interface ModelWeights {
  diagnosisWeight: number;
  escalationPenalty: number;
  outcomeWeight: number;
}

let weights: ModelWeights = {
  diagnosisWeight: 1,
  escalationPenalty: 1,
  outcomeWeight: 1,
};

export async function runRLHFUpdate() {
  const data = await db.select().from(encounters);

  let correct = 0;
  let escalations = 0;
  let outcomeSum = 0;

  for (const e of data) {
    correct += Number(e.correctDiagnosis || 0);
    escalations += e.escalated ? 1 : 0;
    outcomeSum += Number(e.outcomeScore || 0);
  }

  const total = data.length || 1;

  const accuracy = correct / total;
  const escalationRate = escalations / total;
  const avgOutcome = outcomeSum / total;

  // RLHF-style adjustment
  weights.diagnosisWeight *= 1 + (accuracy - 0.8);
  weights.escalationPenalty *= 1 + (escalationRate - 0.2);
  weights.outcomeWeight *= 1 + (avgOutcome - 0.75);

  return {
    updatedWeights: weights,
    metrics: { accuracy, escalationRate, avgOutcome },
  };
}

import { db } from "../db";
import { encounters } from "../db/schema";

export async function predictFailureRisk(agent: string): Promise<number> {
  const data = await db.select().from(encounters);

  const total = data.length || 1;

  const errorRate =
    data.filter(e => Number(e.correctDiagnosis) < 0.7).length / total;

  const escalationRate =
    data.filter(e => e.escalated).length / total;

  const latency =
    data.reduce((sum, e) => sum + Number(e.decisionTimeMs || 0), 0) / total;

  // Normalize latency (simple heuristic)
  const latencyScore = Math.min(latency / 5000, 1);

  const risk =
    errorRate * 0.4 +
    escalationRate * 0.3 +
    latencyScore * 0.3;

  return Math.min(risk, 1);
}

interface AgentOpinion {
  agent: string;
  diagnosis: string;
  confidence: number;
}

export function runDebate(opinions: AgentOpinion[]) {
  const votes: Record<string, number> = {};

  for (const op of opinions) {
    votes[op.diagnosis] = (votes[op.diagnosis] || 0) + op.confidence;
  }

  const sorted = Object.entries(votes).sort((a, b) => b[1] - a[1]);

  return {
    finalDiagnosis: sorted[0][0],
    score: sorted[0][1],
    allVotes: votes,
  };
}

import { EvolutionCandidate } from "./types";
import { simulateCandidate } from "./simulator";

import { governCandidate } from "./governor";
import { deployCandidate } from "./deployer";

export async function runEvolutionPipeline(candidate: EvolutionCandidate) {
  const simResult = await simulateCandidate(candidate);

  const score = await scoreCandidate(candidate, simResult);
  candidate.score = score;

  const decision = await governCandidate(candidate);

  if (decision === "approve") {
    await deployCandidate(candidate);
    candidate.status = "approved";
  } else {
    candidate.status = "rejected";
  }

  return candidate;
}

  if (candidate.type === "table") {

    if (config.risk === "high" && candidate.score! < 0.9) {

  const data = await db.select().from(encounters);

  const total = data.length || 1;

  const goodOutcomes =
    data.filter(e => Number(e.outcomeScore) > 0.8).length / total;

  const poorOutcomes =
    data.filter(e => Number(e.outcomeScore) < 0.5).length / total;

  const readmissions =
    data.filter(e => e.escalated === true).length / total;

  return {
    goodOutcomes,
    poorOutcomes,
    readmissions,
  };
}

export async function getFinancialSignals() {
  const data = await db.select().from(claims);

  const total = data.length || 1;

  const denialRate =
    data.filter(c => c.status === "denied").length / total;

  const avgReimbursement =
    data.reduce((sum, c) => sum + Number(c.amountPaid), 0) / total;

  return {
    denialRate,
    avgReimbursement,
  };
}

const FHIR_BASE = process.env.FHIR_BASE_URL!;
const TOKEN = process.env.FHIR_TOKEN!;

export async function fetchFHIR(resource: string, id: string) {
  const res = await fetch(`${FHIR_BASE}/${resource}/${id}`, {
    headers: {
      Authorization: `Bearer ${TOKEN}`,
    },
  });

  return res.json();
}
📁 server/integrations/ehr/fhirNormalizer.ts
export function normalizeEncounter(fhir: any) {
  return {
    patientId: fhir.subject?.reference,
    diagnosis: fhir.reasonCode?.[0]?.text,
    timestamp: fhir.period?.start,
  };
}

export function normalizeObservation(obs: any) {
  return {
    type: obs.code?.text,
    value: obs.valueQuantity?.value,
    unit: obs.valueQuantity?.unit,
    timestamp: obs.effectiveDateTime,
  };
}

export async function recordFollowUp(data: {
  patientId: string;
  outcome: "resolved" | "worse" | "same";
  revisit: boolean;
}) {
  const score =
    data.outcome === "resolved" ? 1 :
    data.outcome === "same" ? 0.6 : 0.2;

  await db.insert("encounters").values({
    patientId: data.patientId,
    outcomeScore: score,
    escalated: data.revisit,
  });

  return { score };
}

📁 server/patient/stateEngine.ts
import { db } from "../db";

export async function buildPatientState(patientId: string) {
  const encounters = await db.select().from("encounters").where({ patientId });
  const devices = await db.select().from("device_data").where({ patientId });

  const latestVitals = devices.slice(-5);

  const avgSpO2 =
    latestVitals
      .filter(d => d.type === "spo2")
      .reduce((sum, d) => sum + d.value, 0) /
    (latestVitals.length || 1);

  const escalationHistory =
    encounters.filter(e => e.escalated).length / (encounters.length || 1);

  return {
    patientId,
    avgSpO2,
    escalationHistory,
    recentDiagnosis: encounters.slice(-1)[0]?.diagnosis,
  };
}
🔮 2. OUTCOME PREDICTOR (LIGHTWEIGHT MODEL)
📁 server/prediction/outcomePredictor.ts
export function predictOutcome(state: any, treatment: string) {
  let base = 0.7;

  if (state.avgSpO2 < 94) base -= 0.2;
  if (state.escalationHistory > 0.3) base -= 0.2;

  if (treatment === "antibiotic") base += 0.1;
  if (treatment === "supportive") base += 0.05;

  return Math.max(0, Math.min(1, base));
}

export function generateTreatmentOptions(diagnosis: string) {
  if (diagnosis === "viral_uri") {
    return ["supportive", "steroid", "antibiotic"];
  }

  if (diagnosis === "pneumonia") {
    return ["antibiotic", "admit", "imaging"];
  }

  return ["supportive"];
}
⚖️ 4. TREATMENT OPTIMIZER (CORE ENGINE)
📁 server/treatment/treatmentOptimizer.ts
import { predictOutcome } from "../prediction/outcomePredictor";

export function optimizeTreatment(state: any, options: string[]) {
  let best = null;
  let bestScore = -Infinity;

  for (const option of options) {
    const outcome = predictOutcome(state, option);

    const safetyPenalty = state.avgSpO2 < 92 && option !== "admit" ? -0.3 : 0;

    const score = outcome + safetyPenalty;

    if (score > bestScore) {
      bestScore = score;
      best = {
        treatment: option,
        score,
        predictedOutcome: outcome,
      };
    }
  }

  return best;
}

import { buildPatientState } from "../patient/stateEngine";
import { generateTreatmentOptions } from "./treatmentOptions";
import { optimizeTreatment } from "./treatmentOptimizer";

export async function runPersonalizedTreatment(patientId: string) {
  const state = await buildPatientState(patientId);

  const options = generateTreatmentOptions(state.recentDiagnosis);

  const best = optimizeTreatment(state, options);

  return {
    patientId,
    state,
    options,
    recommendation: best,
  };
}
