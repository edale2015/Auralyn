⚡ PHASE 1 — LIVE PATIENT STREAM (REALTIME CORE)
WebSocket Server
// server/stream/patientStream.ts

import WebSocket, { WebSocketServer } from "ws";

interface Patient {
  id: string;
  vitals: {
    hr: number;
    bp: number;
    spo2: number;
    temp: number;
  };
}

const clients: WebSocket[] = [];

export function startPatientStream(server) {
  const wss = new WebSocketServer({ server });

  wss.on("connection", (ws) => {
    clients.push(ws);
  });

  setInterval(() => {
    const patient: Patient = generatePatient();

    clients.forEach(c => c.send(JSON.stringify(patient)));
  }, 2000);
}

function generatePatient(): Patient {
  return {
    id: Math.random().toString(36).substring(7),
    vitals: {
      hr: 60 + Math.random() * 80,
      bp: 90 + Math.random() * 60,
      spo2: 85 + Math.random() * 15,
      temp: 97 + Math.random() * 4,
    }
  };
}
🧠 PHASE 2 — MULTI-AGENT COUNCIL
Orchestrator
// server/agents/orchestrator.ts

import { RiskAgent } from "./riskAgent";
import { ICUAgent } from "./icuAgent";
import { OpsAgent } from "./opsAgent";
import { InsightAgent } from "./insightAgent";
import { ActionAgent } from "./actionAgent";

export class Orchestrator {
  agents = [
    new RiskAgent(),
    new ICUAgent(),
    new OpsAgent(),
    new InsightAgent(),
    new ActionAgent()
  ];

  async run(context: any) {
    let results = [];

    for (const agent of this.agents) {
      const res = await agent.run(context, results);
      results.push(res);
    }

    return results;
  }
}
Risk Agent
// server/agents/riskAgent.ts

export class RiskAgent {
  async run(ctx) {
    const { vitals } = ctx;

    let risk = 0;

    if (vitals.spo2 < 90) risk += 0.4;
    if (vitals.hr > 120) risk += 0.3;
    if (vitals.temp > 101) risk += 0.2;

    return {
      type: "RISK",
      score: risk
    };
  }
}
ICU Agent
// server/agents/icuAgent.ts

export class ICUAgent {
  async run(ctx, prev) {
    const risk = prev.find(p => p.type === "RISK")?.score || 0;

    return {
      type: "ICU_DECISION",
      needsICU: risk > 0.7
    };
  }
}

export function safetyCheck(decision) {
  if (decision.needsICU) {
    return {
      allowed: false,
      reason: "Requires physician approval"
    };
  }

  return { allowed: true };
}
⚙️ PHASE 4 — ACTION ENGINE
// server/agents/actionAgent.ts

import { safetyCheck } from "../safety/safetyGate";

export class ActionAgent {
  async run(ctx, prev) {
    const icu = prev.find(p => p.type === "ICU_DECISION");

    const safety = safetyCheck(icu);

    if (!safety.allowed) {
      return {
        type: "ACTION_BLOCKED",
        reason: safety.reason
      };
    }

    return {
      type: "ACTION",
      executed: "Notify ICU"
    };
  }
}
