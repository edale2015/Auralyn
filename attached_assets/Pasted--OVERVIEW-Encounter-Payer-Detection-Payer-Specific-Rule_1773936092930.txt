📁 server/billing/payerEngine.ts
type PayerRule = {
  name: string;
  preferredCPT: string[];
  requiresModifier25?: boolean;
  strictICD?: boolean;
  telehealthPolicy?: "strict" | "lenient";
};

export class PayerEngine {
  private rules: Record<string, PayerRule> = {
    "medicare": {
      name: "Medicare",
      preferredCPT: ["99213", "99214"],
      requiresModifier25: true,
      strictICD: true,
      telehealthPolicy: "strict"
    },
    "aetna": {
      name: "Aetna",
      preferredCPT: ["99213", "99214", "99215"],
      telehealthPolicy: "lenient"
    },
    "united": {
      name: "UnitedHealthcare",
      preferredCPT: ["99213", "99214"],
      strictICD: true
    }
  };

  optimize(encounter: any, payer: string) {
    const rule = this.rules[payer] || this.rules["medicare"];

    const changes: string[] = [];

    // CPT optimization
    if (!rule.preferredCPT.includes(encounter.coding.cpt[0]?.code)) {
      encounter.coding.cpt = [{ code: rule.preferredCPT[0] }];
      changes.push("Adjusted CPT for payer preference");
    }

    // Modifier 25
    if (rule.requiresModifier25) {
      encounter.coding.modifier = "25";
      changes.push("Added modifier 25");
    }

    // ICD specificity
    if (rule.strictICD && encounter.coding.icd10[0]?.code === "R69") {
      changes.push("ICD too vague for payer");
    }

    return { encounter, changes };
  }
}

This replaces simple rules with feature-based scoring.

📁 server/billing/denialClassifierV2.ts
type Features = {
  icd: string;
  cpt: string;
  payer: string;
  complexity: number;
  documentationScore: number;
};

export class DenialClassifierV2 {
  private weights = {
    icdSpecificity: 0.3,
    cptMismatch: 0.25,
    documentation: 0.2,
    payerRisk: 0.25
  };

  private payerRiskMap: Record<string, number> = {
    "medicare": 0.2,
    "aetna": 0.15,
    "united": 0.25
  };

  predict(f: Features) {
    let risk = 0;

    // ICD specificity
    if (f.icd === "R69") risk += this.weights.icdSpecificity;

    // CPT mismatch
    if (f.complexity < 0.5 && f.cpt === "99214") {
      risk += this.weights.cptMismatch;
    }

    // Documentation
    if (f.documentationScore < 0.5) {
      risk += this.weights.documentation;
    }

    // Payer baseline risk
    risk += this.payerRiskMap[f.payer] || 0.2;

    return {
      riskScore: Math.min(risk, 1),
      confidence: 1 - risk
    };
  }
}
🔗 Feature builder
export function buildFeatures(encounter: any, payer: string): Features {
  return {
    icd: encounter.coding.icd10[0]?.code,
    cpt: encounter.coding.cpt[0]?.code,
    payer,
    complexity: encounter.clinical.complexity || 0.5,
    documentationScore: encounter.clinical.reasoning ? 0.8 : 0.3
  };
}

📁 server/learning/payerRLHFEngine.ts
type Outcome = {
  payer: string;
  coding: any;
  paid: boolean;
  revenue: number;
};

export class PayerRLHFEngine {
  private weights: Record<string, Record<string, number>> = {};

  update(outcome: Outcome) {
    const payer = outcome.payer;
    const key = this.key(outcome.coding);

    if (!this.weights[payer]) this.weights[payer] = {};
    if (!this.weights[payer][key]) this.weights[payer][key] = 0.5;

    if (outcome.paid) {
      const reward = outcome.revenue / 500;
      this.weights[payer][key] += 0.1 * reward;
    } else {
      this.weights[payer][key] -= 0.2;
    }

    this.weights[payer][key] = Math.max(0, Math.min(1, this.weights[payer][key]));
  }

  score(coding: any, payer: string): number {
    return this.weights[payer]?.[this.key(coding)] || 0.4;
  }

  private key(coding: any) {
    const icd = coding.icd10?.[0]?.code || "none";
    const cpt = coding.cpt?.[0]?.code || "none";
    return `${icd}_${cpt}`;
  }
}

type Contract = {

};

  coding: any;
};

export class ContractSimulationEngine {
  simulate(encounters: Encounter[], contracts: Contract[]) {
    return contracts.map(contract => {
      let total = 0;

      for (const e of encounters) {
        const cpt = e.coding.cpt?.[0]?.code || "99213";
        const rate = contract.cptRates[cpt] || 50;

        total += rate;
      }

      return {
        payer: contract.payer,
        projectedRevenue: total,
        avgPerEncounter: total / encounters.length
      };
    });
  }
}

export class PayerStrategyEngine {
  chooseBest(encounter: any, payers: string[], rlhf: any, contracts: any) {
    let best = null;

    for (const payer of payers) {
      const score = rlhf.score(encounter.coding, payer);

      const rate = contracts[payer]?.[encounter.coding.cpt?.[0]?.code] || 50;

      const expectedValue = score * rate;

      if (!best || expectedValue > best.value) {
        best = {
          payer,
          value: expectedValue,
          score,
          rate
        };
      }
    }

    return best;
  }
}

type RoutingOption = {
  payer: string;
  expectedRevenue: number;
  denialRisk: number;
  rlhfScore: number;
};

export class MultiPayerRoutingEngine {
  choose(encounter: any, options: RoutingOption[]) {
    let best = null;

    for (const opt of options) {
      // Risk-adjusted value
      const adjustedValue =
        opt.expectedRevenue *
        opt.rlhfScore *
        (1 - opt.denialRisk);

      if (!best || adjustedValue > best.value) {
        best = {
          payer: opt.payer,
          value: adjustedValue,
          rawRevenue: opt.expectedRevenue,
          risk: opt.denialRisk
        };
      }
    }

    return best;
  }
}

type PricingInput = {
  basePrice: number;
  demandLevel: number;      // 0–1
  capacityUtilization: number; // 0–1
  payerType: "cash" | "insurance";
};

export class DynamicPricingEngine {
  calculate(input: PricingInput) {
    let price = input.basePrice;

    // Demand surge
    if (input.demandLevel > 0.7) {
      price *= 1.2;
    }

    // Capacity constraint
    if (input.capacityUtilization > 0.8) {
      price *= 1.15;
    }

    // Discount for low demand
    if (input.demandLevel < 0.3) {
      price *= 0.85;
    }

    // Cash premium
    if (input.payerType === "cash") {
      price *= 1.1;
    }

    return Math.round(price);
  }
}

type Channel = {
  name: string;
  costPerPatient: number;
  conversionRate: number;
  avgRevenue: number;
};

export class PatientRoutingEngine {
  allocate(budget: number, channels: Channel[]) {
    return channels.map(c => {
      const roi = (c.avgRevenue * c.conversionRate) / c.costPerPatient;

      return {
        channel: c.name,
        roi,
        allocation: budget * roi
      };
    }).sort((a, b) => b.roi - a.roi);
  }
}

      action.intakeLimit = 0.8;
    }

    return action;
  }
}
