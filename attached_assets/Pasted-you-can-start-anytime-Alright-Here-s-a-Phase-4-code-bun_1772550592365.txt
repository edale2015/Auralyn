import admin from "firebase-admin";

function initAdmin() {
  if (admin.apps.length) return admin;

  const raw = process.env.FIREBASE_ADMIN_JSON;
  if (!raw) {
    throw new Error("Missing FIREBASE_ADMIN_JSON. Add Firebase Admin service account JSON to env.");
  }

  const creds = JSON.parse(raw);
  admin.initializeApp({
    credential: admin.credential.cert(creds),
  });

  return admin;
}

export function sha256Hex(input: string): string {
  return crypto.createHash("sha256").update(input, "utf8").digest("hex");

4) server/services/caseService.ts
import { firestore, FieldValue } from "../firestoreAdmin";
import { CaseDoc, CaseMessage, CaseState, PhysicianReview } from "../models/caseTypes";
import { sha256Hex } from "./hash";

const CASES = firestore.collection("cases");

function nowIso() {
  return new Date().toISOString();
}

function defaultReview(): PhysicianReview {
  return {
    status: "NONE",
    reviewedAt: null,
    reviewer: null,
    notes: "",
    finalDisposition: null,
    finalDx: null,
  };
}

export async function createCase(params: {
  channel: CaseDoc["source"]["channel"];
  threadId?: string;
  userId?: string;
  complaintSlug: string;
  complaintDisplay: string;
  engine: CaseDoc["complaint"]["engine"];
}): Promise<CaseDoc> {
  const caseId = `CASE_${nowIso().replace(/[-:.TZ]/g, "")}_${Math.random().toString(16).slice(2, 8)}`;

  const createdAt = nowIso();
  const doc: CaseDoc = {
    caseId,
    createdAt,
    updatedAt: createdAt,
    state: "DRAFT",
    source: { channel: params.channel, threadId: params.threadId, userId: params.userId },
    complaint: { slug: params.complaintSlug, display: params.complaintDisplay, engine: params.engine },
    answers: { structured: {}, answerHash: sha256Hex("{}") },
    triage: null,
    physicianReview: defaultReview(),
    messages: [],
  };

  await CASES.doc(caseId).set(doc);
  return doc;
}

export async function getCase(caseId: string): Promise<CaseDoc | null> {
  const snap = await CASES.doc(caseId).get();
  if (!snap.exists) return null;
  return snap.data() as CaseDoc;
}

export async function appendMessage(caseId: string, msg: CaseMessage): Promise<void> {
  const updatedAt = nowIso();
  await CASES.doc(caseId).update({
    updatedAt,
    messages: FieldValue.arrayUnion(msg),
  });
}

export async function mergeAnswers(caseId: string, patch: Record<string, unknown>): Promise<CaseDoc> {
  const doc = await getCase(caseId);
  if (!doc) throw new Error("Case not found");

  const merged = { ...(doc.answers?.structured ?? {}), ...patch };
  const answerHash = sha256Hex(JSON.stringify(merged));

  const updatedAt = nowIso();
  await CASES.doc(caseId).update({
    updatedAt,
    "answers.structured": merged,
    "answers.answerHash": answerHash,
  });

  const next = await getCase(caseId);
  if (!next) throw new Error("Case not found after update");
  return next;
}

export async function setState(caseId: string, state: CaseState): Promise<void> {
  await CASES.doc(caseId).update({ updatedAt: nowIso(), state });
}

export async function setTriage(caseId: string, triage: CaseDoc["triage"], nextState: CaseState): Promise<void> {
  await CASES.doc(caseId).update({
    updatedAt: nowIso(),
    triage,
    state: nextState,
  });
}

export async function setPhysicianReview(caseId: string, review: Partial<PhysicianReview>, nextState: CaseState): Promise<void> {
  await CASES.doc(caseId).update({
    updatedAt: nowIso(),
    physicianReview: {
      ...review,
      reviewedAt: nowIso(),
    },
    state: nextState,
  });
}

export async function listReviewQueue(params: {
  state?: "NEEDS_REVIEW" | "TRIAGED";
  limit?: number;
}): Promise<CaseDoc[]> {
  const limit = params.limit ?? 50;
  const state = params.state ?? "NEEDS_REVIEW";

  const q = CASES.where("state", "==", state).orderBy("updatedAt", "desc").limit(limit);
  const snaps = await q.get();
  return snaps.docs.map((d) => d.data() as CaseDoc);
}

import { Router } from "express";
import {
  appendMessage,
  createCase,
  getCase,
  mergeAnswers,
  setState,
  setTriage,
} from "../services/caseService";
import { runTriage } from "../services/triageService";

export const casesRouter = Router();

casesRouter.post("/", async (req, res) => {
  const { channel, threadId, userId, complaintSlug, complaintDisplay, engine } = req.body ?? {};
  if (!channel || !complaintSlug) return res.status(400).json({ error: "missing channel/complaintSlug" });

  const doc = await createCase({
    channel,
    threadId,
    userId,
    complaintSlug,
    complaintDisplay: complaintDisplay ?? complaintSlug,
    engine: engine ?? "GENERIC_V1",
  });

  res.json(doc);
});

casesRouter.get("/:caseId", async (req, res) => {
  const doc = await getCase(req.params.caseId);
  if (!doc) return res.status(404).json({ error: "not found" });
  res.json(doc);
});

casesRouter.post("/:caseId/message", async (req, res) => {
  const { dir, channel, text, meta } = req.body ?? {};
  if (!dir || !channel || !text) return res.status(400).json({ error: "missing message fields" });

  await appendMessage(req.params.caseId, {
    ts: new Date().toISOString(),
    dir,
    channel,
    text,
    meta: meta ?? {},
  });

  res.json({ ok: true });
});

casesRouter.post("/:caseId/answers", async (req, res) => {
  const patch = req.body ?? {};
  const updated = await mergeAnswers(req.params.caseId, patch);
  res.json(updated);
});

casesRouter.post("/:caseId/triage", async (req, res) => {
  const doc = await getCase(req.params.caseId);
  if (!doc) return res.status(404).json({ error: "not found" });

  // run engine
  const triage = await runTriage({
    complaintSlug: doc.complaint.slug,
    answers: doc.answers.structured ?? {},
    rulesetVersion: req.body?.rulesetVersion ?? "local",
    dxPriorityVersion: req.body?.dxPriorityVersion ?? "local",
  });

  // decide if review needed
  const needsReview =
    triage.disposition !== "ROUTINE" ||
    triage.confidence === "LOW" ||
    (triage.rfTriggered?.length ?? 0) > 0;

  await setTriage(req.params.caseId, triage, needsReview ? "NEEDS_REVIEW" : "TRIAGED");

  res.json({ ok: true, triage, nextState: needsReview ? "NEEDS_REVIEW" : "TRIAGED" });
});

casesRouter.post("/:caseId/state", async (req, res) => {
  const { state } = req.body ?? {};
  if (!state) return res.status(400).json({ error: "missing state" });
  await setState(req.params.caseId, state);
  res.json({ ok: true });
});

export type CaseDoc = any;

async function j<T>(url: string, opts?: RequestInit): Promise<T> {
  const r = await fetch(url, {
    headers: { "Content-Type": "application/json", ...(opts?.headers ?? {}) },
    ...opts,
  });
  if (!r.ok) throw new Error(await r.text());
  return r.json();
}

export const api = {
  reviewQueue: (state = "NEEDS_REVIEW") => j<CaseDoc[]>(`/api/review/queue?state=${state}`),
  getCase: (caseId: string) => j<CaseDoc>(`/api/review/case/${caseId}`),
  reviewCase: (caseId: string, body: any) =>
    j(`/api/review/case/${caseId}`, { method: "POST", body: JSON.stringify(body) }),
};

12) client/src/pages/CaseDetail.tsx
import React from "react";
import { api } from "../api";

export default function CaseDetail({ caseId }: { caseId: string }) {
  const [c, setC] = React.useState<any | null>(null);
  const [notes, setNotes] = React.useState("");
  const [err, setErr] = React.useState<string | null>(null);

  React.useEffect(() => {
    api.getCase(caseId).then((doc) => {
      setC(doc);
      setNotes(doc?.physicianReview?.notes ?? "");
    }).catch((e) => setErr(String(e)));
  }, [caseId]);

  async function submit(status: string) {
    setErr(null);
    try {
      await api.reviewCase(caseId, {
        status,
        notes,
        finalDisposition: c?.triage?.disposition ?? null,
        finalDx: c?.triage?.topDx ?? null,
        reviewer: { id: "phys1", name: "Physician" },
      });
      const updated = await api.getCase(caseId);
      setC(updated);
    } catch (e: any) {
      setErr(String(e));
    }
  }

  if (!c) return <div style={{ padding: 16 }}>Loading...</div>;

  return (
    <div style={{ padding: 16, fontFamily: "system-ui" }}>
      <a href="/">← Back</a>
      <h2>{c.caseId}</h2>

      <div style={{ marginBottom: 12 }}>
        <b>Complaint:</b> {c.complaint?.display ?? c.complaint?.slug} <br />
        <b>State:</b> {c.state} <br />
        <b>Disposition:</b> {c.triage?.disposition ?? "-"} <br />
        <b>Top Dx:</b> {c.triage?.topDx ?? "-"} <br />
        <b>Confidence:</b> {c.triage?.confidence ?? "-"} <br />
        <b>TieBreak:</b> {c.triage?.tieBreak ?? "-"} (margin {c.triage?.margin ?? "-"}) <br />
      </div>

      <h3>Answers</h3>
      <pre style={{ background: "#fafafa", padding: 12, border: "1px solid #eee" }}>
        {JSON.stringify(c.answers?.structured ?? {}, null, 2)}
      </pre>

      <h3>Scoring Explanation</h3>
      <pre style={{ background: "#fafafa", padding: 12, border: "1px solid #eee" }}>
        {JSON.stringify(c.triage?.explanation ?? {}, null, 2)}
      </pre>

      <h3>Physician Review</h3>
      {err && <div style={{ color: "crimson" }}>{err}</div>}
      <textarea
        value={notes}
        onChange={(e) => setNotes(e.target.value)}
        rows={6}
        style={{ width: "100%", marginBottom: 8 }}
        placeholder="Physician notes..."
      />
      <div style={{ display: "flex", gap: 8 }}>
        <button onClick={() => submit("APPROVED")}>Approve</button>
        <button onClick={() => submit("MODIFIED")}>Modify</button>
        <button onClick={() => submit("ESCALATED")}>Escalate</button>
        <button onClick={() => submit("REJECTED")}>Reject</button>
      </div>

      <div style={{ marginTop: 12 }}>
        <b>Review status:</b> {c.physicianReview?.status}
      </div>
    </div>
  );
}
