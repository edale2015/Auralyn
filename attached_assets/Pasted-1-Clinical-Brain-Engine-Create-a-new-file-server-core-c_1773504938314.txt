1. Clinical Brain Engine

Create a new file:

server/core/clinicalBrainEngine.ts

:::

import { findSimilarCasesForState } from "../assistant/caseSimilarityService";
import { computeDifferentialProbabilities } from "../assistant/differentialProbabilityEngine";
import { selectNextBestQuestion } from "../assistant/nextBestQuestionEngine";
import { detectRedFlags } from "../agent/safety/redFlags";

export interface BrainInput {
  complaint: string;
  answers: Record<string, any>;
  state: any;
  differentialCandidates: { clusterId: string; score: number }[];
  availableQuestions: string[];
}

export interface BrainOutput {
  similarity?: any;
  differentials?: any[];
  nextQuestion?: string | null;
  questionRankings?: any[];
  redFlags?: any[];
  disposition?: string;
}

export async function runClinicalBrain(input: BrainInput): Promise<BrainOutput> {

  const { state, answers, differentialCandidates, availableQuestions } = input;

  const result: BrainOutput = {};

  // ----------------------------
  // 1️⃣ Case Similarity Engine
  // ----------------------------

  try {

    const sim = await findSimilarCasesForState(state, 5);

    result.similarity = sim;

    if (sim.summary?.safetyWarnings?.length > 0) {
      state.safetyWarnings = sim.summary.safetyWarnings;
    }

  } catch (err) {
    console.warn("Similarity engine failed", err);
  }

  // ----------------------------
  // 2️⃣ Bayesian Differential Engine
  // ----------------------------

  let differentials;

  try {

    differentials = computeDifferentialProbabilities(
      differentialCandidates,
      answers
    );

    result.differentials = differentials;

  } catch (err) {
    console.warn("Differential engine failed", err);
  }

  // ----------------------------
  // 3️⃣ Next Best Question Engine
  // ----------------------------

  try {

    const nbq = selectNextBestQuestion(
      differentialCandidates,
      answers,
      availableQuestions
    );

    result.nextQuestion = nbq.bestQuestion;
    result.questionRankings = nbq.rankings;

  } catch (err) {
    console.warn("Next question engine failed", err);
  }

  // ----------------------------
  // 4️⃣ Red Flag Safety Layer
  // ----------------------------

  try {

    const flags = detectRedFlags(state);

    result.redFlags = flags;

    if (flags?.length > 0) {
      result.disposition = "ER_NOW";
      return result;
    }

  } catch (err) {
    console.warn("Red flag engine failed", err);
  }

  // ----------------------------
  // 5️⃣ Disposition Logic
  // ----------------------------

  if (differentials && differentials.length > 0) {

    const topDx = differentials[0];

    if (topDx.posteriorProbability > 0.6) {
      result.disposition = "LIKELY_OUTPATIENT";
    } else if (topDx.posteriorProbability > 0.3) {
      result.disposition = "URGENT_CARE";
    } else {
      result.disposition = "NEEDS_MORE_INFO";
    }

  }

  return result;
}

3. Bayesian Upgrade (Recommendation 8)

Replace your probability engine with this improved version.

File:

server/assistant/differentialProbabilityEngine.ts

:::

export interface DifferentialCandidate {
  clusterId: string;
  priorProbability: number;
  posteriorProbability: number;
  evidenceFor: string[];
  evidenceAgainst: string[];
}

function likelihood(symptom: string, clusterId: string): number {

  const heuristicMap: Record<string, Record<string, number>> = {

    fever: { pneumonia: 3, flu: 2 },
    pleuritic_pain: { pulmonary_embolism: 4, pneumonia: 2 },
    dysuria: { uti: 4 }

  };

  return heuristicMap[symptom]?.[clusterId] || 1;
}

export function computeDifferentialProbabilities(
  dxCandidates: Array<{ clusterId: string; score: number }>,
  answers: Record<string, unknown>
): DifferentialCandidate[] {

  const totalScore = dxCandidates.reduce((s, d) => s + Math.max(0, d.score), 0) || 1;

  return dxCandidates.map((d) => {

    const prior = Math.max(0, d.score) / totalScore;

    let posterior = prior;

    for (const key of Object.keys(answers)) {
      if (answers[key]) {
        posterior *= likelihood(key, d.clusterId);
      }
    }

    return {
      clusterId: d.clusterId,
      priorProbability: prior,
      posteriorProbability: posterior,
      evidenceFor: Object.keys(answers),
      evidenceAgainst: []
    };

  }).sort((a, b) => b.posteriorProbability - a.posteriorProbability);
}
