🚨 PART 1 — CHAOS INJECTOR
📁 server/chaos/chaosEngine.ts
import { emitEvent } from "../controlTower/eventBus";

const chaosState = {
  enabled: false,
  scenarios: {}
};

export function enableChaos() {
  chaosState.enabled = true;
}

export function disableChaos() {
  chaosState.enabled = false;
  chaosState.scenarios = {};
}

export function injectChaos(type) {
  chaosState.scenarios[type] = true;

  emitEvent({
    type: "CHAOS_INJECTED",
    payload: { type },
    timestamp: Date.now()
  });
}

export function isChaosActive(type) {
  return chaosState.enabled && chaosState.scenarios[type];
}

🔄 PART 4 — AUTOMATED RECOVERY ENGINE
📁 server/recovery/recoveryEngine.ts
import { emitEvent } from "../controlTower/eventBus";

export async function runRecovery(state) {
  const actions = [];

  if (state.circuitBreakers.openai === "OPEN") {
    actions.push("Switch to fallback model");
  }

  if (state.queueDepth > 800) {
    actions.push("Enable backpressure");
  }

  if (state.errorRate > 0.2) {
    actions.push("Scale up pods");
  }

  if (state.redis === "DOWN") {
    actions.push("Switch to in-memory queue");
  }

  if (state.db === "DOWN") {
    actions.push("Switch to read-only mode");
  }

  if (actions.length > 0) {
    emitEvent({
      type: "SELF_HEAL_ACTIONS",
      payload: actions,
      timestamp: Date.now()
    });
  }

  return actions;
}
