1️⃣ Graph Gap Detector

Purpose: automatically find structural holes in the clinical knowledge graph.

Typical problems it detects:

complaint missing protocol

complaint missing red-flag skill

skill not handled by an engine

diagnosis missing disposition

Graph Gap Detector

server/knowledge/graphGapDetector.ts

import {getKnowledgeGraph} from "./knowledgeGraphStore"

export function detectGraphGaps(){

 const graph = getKnowledgeGraph()

 const gaps:any[]=[]

 const complaints = graph.nodes.filter(n=>n.type==="complaint")

 complaints.forEach(c=>{

   const edges = graph.edges.filter(e=>e.from===c.id)

   const hasProtocol = edges.some(e=>e.relation==="governed_by")

   if(!hasProtocol){

     gaps.push({
       complaint:c.label,
       problem:"missing_protocol"
     })

   }

   const hasSkill = edges.some(e=>e.relation==="requires")

   if(!hasSkill){

     gaps.push({
       complaint:c.label,
       problem:"missing_skill_mapping"
     })

   }

 })

 const skills = graph.nodes.filter(n=>n.type==="skill")

 skills.forEach(s=>{

   const handled = graph.edges.some(
     e=>e.from===s.id && e.relation==="handled_by"
   )

   if(!handled){

     gaps.push({
       skill:s.label,
       problem:"no_engine_assigned"
     })

   }

 })

 const diagnoses = graph.nodes.filter(n=>n.type==="diagnosis")

 diagnoses.forEach(d=>{

   const disposition = graph.edges.some(
     e=>e.from===d.id && e.relation==="can_lead_to"
   )

   if(!disposition){

     gaps.push({
       diagnosis:d.label,
       problem:"no_disposition_path"
     })

   }

 })

 return gaps

}

2️⃣ Question-to-Skill Coverage Scorer

Checks whether each skill has enough questions.

server/knowledge/questionCoverageEngine.ts

import {getKnowledgeGraph} from "./knowledgeGraphStore"

export function scoreQuestionCoverage(){

 const graph = getKnowledgeGraph()

 const results:any[]=[]

 const skills = graph.nodes.filter(n=>n.type==="skill")

 skills.forEach(skill=>{

  const relatedQuestions = graph.edges.filter(
    e=>e.to===skill.id && e.relation==="supports"
  )

  const score = relatedQuestions.length

  results.push({
    skill:skill.label,
    questionCount:score,
    coverage:score>2?"good":"weak"
  })

 })

 return results

}
Coverage API

3️⃣ Protocol-to-Graph Sync Engine

Automatically keeps the graph aligned with protocols.

server/knowledge/protocolSyncEngine.ts

import {addKnowledgeEdge} from "./knowledgeGraphStore"

export function syncProtocols(protocolMap:any){

 Object.entries(protocolMap).forEach(([complaint,protocol]:any)=>{

  addKnowledgeEdge({

   id:`sync_${complaint}`,
   from:`complaint:${complaint}`,
   to:`protocol:${protocol}`,
   relation:"governed_by"

  })

 })

}

This lets you load your protocolSelector mappings directly into the graph.

5️⃣ Graph-Aware Adaptive Question Engine

Uses the knowledge graph to pick the next question dynamically.

server/engines/graphAwareQuestionEngine.ts

import {getKnowledgeGraph} from "../knowledge/knowledgeGraphStore"

export function chooseNextQuestion(complaint:string,answered:any){

 const graph = getKnowledgeGraph()

 const complaintId=`complaint:${complaint}`

 const questionEdges = graph.edges.filter(
   e=>e.from===complaintId && e.relation==="asks"
 )

 const questions = questionEdges.map(
   e=>graph.nodes.find(n=>n.id===e.to)
 )

 const unanswered = questions.filter(
   q=>!answered.includes(q?.id)
 )

 return unanswered[0]

}

This engine will replace static question flows with skill-targeted questioning.
