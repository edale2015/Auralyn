export function assignCPT(disposition: string) {

  if (disposition === "ER_NOW") return "99285";
  if (disposition === "URGENT") return "99284";
  return "99213";
}
💵 Revenue estimator
export function estimateRevenue(visits: any[]) {

  const rates: any = {
    "99285": 500,
    "99284": 300,
    "99213": 120
  };

  return visits.reduce((sum, v) => {
    const code = assignCPT(v.disposition);
    return sum + (rates[code] || 0);
  }, 0);
}
