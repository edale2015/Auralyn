import express from "express";
import { db } from "../db";
import { cases, auditLogs } from "../db/schema";

const router = express.Router();

// Fetch batch of pending cases
router.get("/batch", async (req, res) => {
  const limit = Number(req.query.limit) || 20;

  const pending = await db.query.cases.findMany({
    where: (c, { eq }) => eq(c.status, "pending"),
    limit
  });

  res.json(pending);
});

// Batch approve
router.post("/approve", async (req, res) => {
  const { caseIds, userId } = req.body;

  for (const id of caseIds) {
    const c = await db.query.cases.findFirst({
      where: (cs, { eq }) => eq(cs.id, id)
    });

    await db.update(cases)
      .set({
        physicianDecision: c.aiSuggestion,
        status: "reviewed"
      })
      .where((cs, { eq }) => eq(cs.id, id));

    await db.insert(auditLogs).values({
      caseId: id,
      userId,
      action: "batch_approve",
      aiSuggestion: c.aiSuggestion,
      finalDecision: c.aiSuggestion,
      override: false
    });
  }

  res.json({ success: true });
});

export default router;
Frontend — batch UI
/frontend/pages/BatchReview.tsx
import { useEffect, useState } from "react";

export default function BatchReview() {
  const [cases, setCases] = useState<any[]>([]);
  const [selected, setSelected] = useState<number[]>([]);

  useEffect(() => {
    fetch("/api/batch")
      .then(r => r.json())
      .then(setCases);
  }, []);

  const toggle = (id: number) => {
    setSelected(prev =>
      prev.includes(id)
        ? prev.filter(i => i !== id)
        : [...prev, id]
    );
  };

  const approve = async () => {
    await fetch("/api/batch/approve", {
      method: "POST",
      headers: {"Content-Type": "application/json"},
      body: JSON.stringify({
        caseIds: selected,
        userId: 1
      })
    });
    location.reload();
  };

  return (
    <div className="p-6">
      <h1>Batch Review</h1>

      {cases.map(c => (
        <div key={c.id} className="border p-2 mt-2">
          <input
            type="checkbox"
            onChange={() => toggle(c.id)}
          />
          <span className="ml-2">
            {c.patientData?.chiefComplaint}
          </span>
        </div>
      ))}

      <button onClick={approve} className="bg-green-600 text-white p-2 mt-4">
        Approve Selected
      </button>
    </div>
  );
}

// send message
export async function sendWhatsApp(to: string, text: string) {
  await fetch(`https://graph.facebook.com/v18.0/${PHONE_ID}/messages`, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      messaging_product: "whatsapp",
      to,
      type: "text",
      text: { body: text }
    })
  });
}

// webhook
router.post("/webhook", async (req, res) => {
  const msg = req.body.entry?.[0]?.changes?.[0]?.value?.messages?.[0];

  if (!msg) return res.sendStatus(200);

  const from = msg.from;

  // simple intake flow
  await sendWhatsApp(from, "What is your main symptom?");
  await sendWhatsApp(from, `You said: ${text}`);

  res.sendStatus(200);
});

🧠 5️⃣ RISK SCORING ENGINE
/server/services/riskEngine.ts
export function calculateRisk(caseData: any, ai: any) {
  let score = 0;

  if (ai.confidence < 0.7) score += 2;
  if (ai.redFlags?.length) score += 3;
  if (ai.disposition === "er_now") score += 5;

  return {
    score,
    level:
      score >= 5 ? "HIGH" :
      score >= 3 ? "MEDIUM" :
      "LOW"
  };
}

import express from "express";

let outcomes: any[] = [];

const router = express.Router();

router.post("/submit", (req, res) => {
  outcomes.push(req.body);
  res.json({ success: true });
});

router.get("/stats", (req, res) => {
  const accuracy =
    outcomes.filter(o => o.correct).length / outcomes.length;

  res.json({
    total: outcomes.length,
    accuracy
  });
});

export default router;
