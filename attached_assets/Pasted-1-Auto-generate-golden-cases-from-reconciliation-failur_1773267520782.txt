import fs from "fs/promises";
import path from "path";

const RUNTIME_DIR = path.resolve(process.cwd(), "server/data/runtime");
const TEST_DIR = path.resolve(process.cwd(), "server/testing");

async function loadNdjson(fileName: string): Promise<any[]> {
  try {
    const raw = await fs.readFile(path.join(RUNTIME_DIR, fileName), "utf8");
    return raw.split(/\r?\n/).filter(Boolean).map(JSON.parse);
  } catch {
    return [];
  }
}

async function loadGoldenCases(): Promise<any[]> {
  try {
    const raw = await fs.readFile(
      path.join(TEST_DIR, "goldenCases.generated.json"),
      "utf8"
    );
    return JSON.parse(raw);
  } catch {
    return [];
  }
}

export async function generateGoldenCasesFromReconciliations() {
  const reconciliations = await loadNdjson("case_reconciliation.ndjson");
  const skillRuns = await loadNdjson("skill_run_log.ndjson");
  const existing = await loadGoldenCases();

  const newCases: any[] = [];

  for (const rec of reconciliations) {
    if (
      rec.top_prediction_match &&
      rec.disposition_match &&
      !rec.safety_miss_flag
    )
      continue;

    const caseId = rec.case_id ?? rec.caseId;

    const intakeRun = skillRuns.find(
      (r) => r.caseId === caseId && r.skillName === "identify_chief_complaint"
    );

    if (!intakeRun) continue;

    let rawText = "";
    try {
      const input = JSON.parse(intakeRun.inputSummary ?? "{}");
      rawText = input.rawText ?? "";
    } catch {
      continue;
    }

    const goldenCase = {
      id: `AUTO_${caseId}`,
      input: {
        rawText,
      },
      expected: {
        complaint_id: rec.predictedComplaint ?? "",
        disposition: rec.actualDisposition ?? "",
        top_differential_contains: rec.actualFinalDiagnosis
          ? [rec.actualFinalDiagnosis]
          : [],
      },
      source: {
        type: "reconciliation_failure",
        original_case_id: caseId,
        predicted: {
          diagnosis: rec.predictedTop,
          disposition: rec.predictedDisposition,
        },
        actual: {
          diagnosis: rec.actualFinalDiagnosis,
          disposition: rec.actualDisposition,
        },
      },
    };

    const exists = existing.some((e) => e.id === goldenCase.id);
    if (!exists) newCases.push(goldenCase);
  }

  const merged = [...existing, ...newCases];

  await fs.writeFile(
    path.join(TEST_DIR, "goldenCases.generated.json"),
    JSON.stringify(merged, null, 2),
    "utf8"
  );

  return {
    generated: newCases.length,
    total: merged.length,
  };
}

if (require.main === module) {
  generateGoldenCasesFromReconciliations().then(console.log);
}

import express from "express";

const router = express.Router();

  try {
    const result = await generateGoldenCasesFromReconciliations();
    res.json({ ok: true, result });

    res.status(500).json({ ok: false, error: err.message });

2. Complaint hardening queue

Failures and drift alerts should become actionable tasks.

Create:

server/learning/complaintHardeningQueue.ts
import fs from "fs/promises";
import path from "path";
import { generateComplaintDriftAlerts } from "./complaintDriftAlerts";

const RUNTIME_DIR = path.resolve(process.cwd(), "server/data/runtime");

export async function buildComplaintHardeningQueue() {
  const alerts = await generateComplaintDriftAlerts();

  const queue = alerts.map((alert) => ({
    complaint: alert.complaint,
    priority:
      alert.safetyMissRate > 0.05
        ? "critical"
        : alert.failureRate > 0.2
        ? "high"
        : "medium",
    failureRate: alert.failureRate,
    safetyMissRate: alert.safetyMissRate,
    recommendedActions: [
      "Review red flag rules",
      "Add new golden cases",
      "Audit differential cluster scoring",
      "Check disposition expressions",
    ],
    createdAt: new Date().toISOString(),
  }));

  await fs.writeFile(
    path.join(RUNTIME_DIR, "complaint_hardening_queue.json"),
    JSON.stringify(queue, null, 2),
    "utf8"
  );

  return queue;
}

It converts unstructured patient input into structured clinical reasoning through a series of specialized skills.
