⚙️ Backend: build tree from trace
// server/trace/buildDecisionTree.ts
export function buildDecisionTree(trace: any): TreeNode {
  return {
    id: 'root',
    label: 'Clinical Flow',
    type: 'rule',
    children: [
      {
        id: 'questions',
        label: 'Questions',
        type: 'question',
        children: trace.questions.map((q: any) => ({
          id: q.id,
          label: `${q.key}: ${q.value}`,
          type: 'question'
        }))
      },
      {
        id: 'findings',
        label: 'Findings',
        type: 'finding',
        children: trace.findings.map((f: any) => ({
          id: f.key,
          label: `${f.key}: ${f.value}`,
          type: 'finding'
        }))
      },
      {
        id: 'diagnosis',
        label: 'Diagnosis',
        type: 'dx',
        children: trace.differential.map((d: any) => ({
          id: d.ruleId,
          label: `${d.name} (${(d.posterior*100).toFixed(1)}%)`,
          type: 'dx',
          score: d.posterior,
          source: d.source
        }))
      },
      {
        id: 'disposition',
        label: `Disposition: ${trace.disposition}`,
        type: 'action'
      }
    ]
  }
}

❓ D. ADAPTIVE QUESTIONING PANEL (Next-Best Question)
🧾 Schema
CREATE TABLE IF NOT EXISTS kb_question_utility (
  question_key TEXT,

  info_gain FLOAT
);
⚙️ Engine
// server/engine/nextBestQuestion.ts
export async function getNextBestQuestion(currentDx: any[], answered: string[]) {
  const q = await db.kb_question_utility.findAll()

  const scores: Record<string, number> = {}

  for (const dx of currentDx) {
    for (const row of q) {
      if (row.diagnosis === dx.diagnosis && !answered.includes(row.question_key)) {
        scores[row.question_key] = (scores[row.question_key] || 0) +
          dx.posterior * row.info_gain
      }
    }
  }

  return Object.entries(scores)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5)
}
