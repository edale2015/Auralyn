type ConfidenceDecision = {
  autoSubmit: boolean;
  confidenceScore: number;
  reasons: string[];
};

export class ConfidenceGate {
  evaluate(encounter: any, denialRisk: number): ConfidenceDecision {
    let confidence = 1 - denialRisk;
    const reasons: string[] = [];

    // Strong signals
    if (!encounter.clinical.reasoning) {
      confidence -= 0.2;
      reasons.push("Missing reasoning");
    }

    if (encounter.coding.icd10[0]?.code === "R69") {
      confidence -= 0.3;
      reasons.push("Low-specificity diagnosis");
    }

    if (denialRisk > 0.5) {
      confidence -= 0.3;
      reasons.push("High denial risk");
    }

    const autoSubmit = confidence > 0.7;

    return {
      autoSubmit,
      confidenceScore: Math.max(0, Math.min(confidence, 1)),
      reasons
    };
  }
}

type Outcome = {
  encounterId: string;
  paid: boolean;
  revenue: number;
  denialReasons?: string[];
  coding: any;
};

export class MoneyRLHFEngine {
  private weights: Record<string, number> = {};

  update(outcome: Outcome) {
    const key = this.key(outcome.coding);

    if (!this.weights[key]) this.weights[key] = 0.5;

    // Reward = money
    if (outcome.paid) {
      const reward = outcome.revenue / 500; // normalize
      this.weights[key] += 0.1 * reward;
    } else {
      this.weights[key] -= 0.2;
    }

    this.weights[key] = Math.max(0, Math.min(1, this.weights[key]));
  }

  score(coding: any): number {
    return this.weights[this.key(coding)] || 0.4;
  }

  private key(coding: any) {
    const icd = coding.icd10?.[0]?.code || "none";
    const cpt = coding.cpt?.[0]?.code || "none";
    return `${icd}_${cpt}`;
  }
}

This is the decision + action layer.

import { ConfidenceGate } from "./confidenceGate";
import { BillingService } from "../billing/billingService";

export class AutonomousEngine {
  private gate = new ConfidenceGate();
  private billing = new BillingService();

  async execute(encounter: any, denialRisk: number) {
    const decision = this.gate.evaluate(encounter, denialRisk);

    if (decision.autoSubmit) {
      const billingResult = await this.billing.process(encounter);

      return {
        mode: "AUTO_SUBMITTED",
        billing: billingResult,
        confidence: decision
      };
    }

    return {
      mode: "PHYSICIAN_REVIEW_REQUIRED",
      confidence: decision
    };
  }
}
🔗 PART 4 — FULL AUTONOMOUS PIPELINE (ULTIMATE FORM)
📁 server/pipeline/autonomousFullPipeline.ts
import { DenialEngine } from "../billing/denialEngine";
import { AutoFixEngine } from "../billing/autoFixEngine";
import { AutonomousEngine } from "../autonomy/autonomousEngine";
import { MoneyRLHFEngine } from "../learning/moneyRLHFEngine";
import { ReviewQueue } from "../review/reviewQueue";

const denial = new DenialEngine();
const autoFix = new AutoFixEngine();
const autonomous = new AutonomousEngine();
const rlhf = new MoneyRLHFEngine();
const queue = new ReviewQueue();

export async function runAutonomousPipeline(input: any) {
  let encounter = input.encounter;

  // Step 1 — Predict denial
  let denialPrediction = denial.predict(encounter);

  // Step 2 — Auto-fix
  const fix = autoFix.fix(encounter, denialPrediction);
  encounter = fix.encounter;

  // Step 3 — Recalculate risk
  denialPrediction = denial.predict(encounter);

  // Step 4 — RLHF score boost
  const learnedScore = rlhf.score(encounter.coding);
  const adjustedRisk = (denialPrediction.riskScore || 0.3) * (1 - learnedScore);

  // Step 5 — Autonomous decision
  const result = await autonomous.execute(encounter, adjustedRisk);

  // Step 6 — If not autonomous → queue
  if (result.mode !== "AUTO_SUBMITTED") {
    queue.add({
      ...encounter,
      denialPrediction,
      reason: result.confidence.reasons
    });
  }

  return {
    encounter,
    decision: result,
    adjustedRisk,
    fixes: fix.fixes
  };
}
