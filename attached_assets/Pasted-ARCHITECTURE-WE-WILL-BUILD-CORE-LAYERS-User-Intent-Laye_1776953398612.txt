// server/agents/orchestrator.ts

import { RiskAgent } from "./riskAgent";
import { OpsAgent } from "./opsAgent";
import { InsightAgent } from "./insightAgent";
import { ActionAgent } from "./actionAgent";

export class AgentOrchestrator {
  private agents;

  constructor() {
    this.agents = [
      new RiskAgent(),
      new OpsAgent(),
      new InsightAgent(),
      new ActionAgent()
    ];
  }

  async runCycle(context: any) {
    const results = [];

    for (const agent of this.agents) {
      try {
        const res = await agent.run(context);
        results.push(res);
      } catch (err) {
        console.error(`Agent failed: ${agent.constructor.name}`, err);
      }
    }

    return results;
  }
}
🧠 2. Risk Detection Agent
// server/agents/riskAgent.ts

import { getPatients } from "../db/patients";
import { evaluateRisk } from "../engine/riskEngine";

export class RiskAgent {
  async run() {
    const patients = await getPatients();
    const risks = [];

    for (const p of patients) {
      const score = evaluateRisk(p);

      if (score > 0.7) {
        risks.push({
          patientId: p.id,
          riskScore: score,
          reason: "High deterioration probability"
        });
      }
    }

    return { type: "RISKS", data: risks };
  }
}

🤖 4. Action Agent (AUTOMATION)
// server/agents/actionAgent.ts

import { sendAlert } from "../integrations/alerts";

export class ActionAgent {
  async run(context: any) {
    const insights = context.find(i => i.type === "INSIGHTS")?.data || [];

    for (const i of insights) {
      if (i.priority === "HIGH") {
        await sendAlert(i.message);
      }
    }

    return { type: "ACTIONS_EXECUTED", count: insights.length };
  }
}

🧠 PART 2 — SELF-LEARNING ENGINE (RLHF LIGHT VERSION)
// server/learning/rlhf.ts

interface Feedback {
  action: string;
  success: boolean;
}

const weights: Record<string, number> = {};

export function updateWeights(feedback: Feedback) {
  const delta = feedback.success ? 0.02 : -0.02;

  weights[feedback.action] = (weights[feedback.action] || 0) + delta;

  // clamp safety
  if (weights[feedback.action] > 1) weights[feedback.action] = 1;
  if (weights[feedback.action] < -1) weights[feedback.action] = -1;
}
