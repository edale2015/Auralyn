2. Telemedicine session manager

Tracks the live visit.

telemedicineSessionService.ts
type TelemedicineSession = {
  caseId: string
  messages: string[]
  answers: Record<string,string>
  complaint?: string
  differential?: any[]
  disposition?: string
}

const sessions:Record<string,TelemedicineSession> = {}

export function getSession(caseId:string){

  sessions[caseId] ??={
    caseId,
    messages:[],
    answers:{}
  }

  return sessions[caseId]
}
3. Reasoning engine wrapper

This calls your existing skill orchestrator.

telemedicineReasoningService.ts
import { ClinicalSkillOrchestrator } from "../orchestrator/clinicalSkillOrchestrator"
import { getSession } from "./telemedicineSessionService"

export async function updateReasoning(caseId:string){

  const session = getSession(caseId)

  const orchestrator = new ClinicalSkillOrchestrator()

  const state = await orchestrator.run({
    caseId,
    rawText: session.messages.join(" "),
    modifiers:{},
    knownFacts:{},
    priorSkillOutputs:{},
    config:{
      orchestrationMode:"graph"
    }
  })

  session.complaint =
    state.skillResults?.identify_chief_complaint?.result?.complaint_id

  session.differential =
    state.skillResults?.generate_differential?.result?.differential_list

  session.disposition = state.finalDisposition

  return session
}
4. Next-question generator

This uses your question bundle skill.

telemedicineQuestionService.ts
import { getSession } from "./telemedicineSessionService"

export function getNextQuestions(caseId:string){

  const session = getSession(caseId)

  const complaint = session.complaint

  if(!complaint)
    return ["Can you describe your symptoms?"]

  if(complaint === "cough"){
    return [
      "Any shortness of breath?",
      "Chest pain?",
      "How long has the cough lasted?",
      "Any fever?"
    ]
  }

  if(complaint === "sore_throat"){
    return [
      "Any fever?",
      "Any cough?",
      "Pain when swallowing?",
      "Any rash?"
    ]
  }

  return ["Any other symptoms?"]
}

5. Safety alert engine

Uses your red-flag rules.

telemedicineSafetyService.ts
export function checkSafetyAlerts(text:string){

  const alerts:string[] = []

  const s = text.toLowerCase()

  if(s.includes("chest pain"))
    alerts.push("Possible cardiac emergency")

  if(s.includes("shortness of breath"))
    alerts.push("Possible respiratory emergency")

  if(s.includes("confusion"))
    alerts.push("Neurologic red flag")

  return alerts
}
6. Automatic note generation

Uses the skill outputs.

telemedicineNoteService.ts
import { getSession } from "./telemedicineSessionService"

export function generateTelemedicineNote(caseId:string){

  const session = getSession(caseId)

  const hpi = session.messages.join(" ")

  const differential =
    (session.differential ?? [])
      .slice(0,3)
      .map((d:any)=>d.diagnosis)
      .join(", ")

  return {

    HPI: hpi,

    Assessment:
      differential
        ? `Differential includes ${differential}`
        : "Assessment pending",

    Plan:[
      "Supportive care",
      "Return precautions provided"
    ],

    Disposition: session.disposition ?? "pending"
  }
}
7. Telemedicine assistant routes
telemedicineAssistantRoutes.ts
import express from "express"

import { getSession } from "./telemedicineSessionService"
import { updateReasoning } from "./telemedicineReasoningService"
import { getNextQuestions } from "./telemedicineQuestionService"
import { checkSafetyAlerts } from "./telemedicineSafetyService"
import { generateTelemedicineNote } from "./telemedicineNoteService"

const router = express.Router()

router.post("/api/telemed/message", async(req,res)=>{

  const { caseId, message } = req.body

  const session = getSession(caseId)

  session.messages.push(message)

  await updateReasoning(caseId)

  const alerts = checkSafetyAlerts(session.messages.join(" "))

  const questions = getNextQuestions(caseId)

  res.json({
    ok:true,
    complaint: session.complaint,
    differential: session.differential,
    disposition: session.disposition,
    alerts,
    nextQuestions: questions
  })
})

router.get("/api/telemed/note/:caseId",(req,res)=>{

  const note = generateTelemedicineNote(req.params.caseId)

  res.json({ok:true,note})

})

export default router

Register route:

app.use(telemedicineAssistantRoutes)
