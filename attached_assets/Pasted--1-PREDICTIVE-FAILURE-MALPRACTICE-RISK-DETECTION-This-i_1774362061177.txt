type RiskSignal = {
  caseId: string;
  latencyMs: number;
  errorRate: number;
  overrideRate: number;
  riskScore: number;
  complaint: string;
  redFlags: number;
};

export function computeSystemRisk(signal: RiskSignal) {
  let risk = 0;

  if (signal.latencyMs > 2000) risk += 0.1;
  if (signal.errorRate > 0.05) risk += 0.2;
  if (signal.overrideRate > 0.2) risk += 0.3;
  if (signal.redFlags > 0) risk += 0.3;

  risk += signal.riskScore * 0.5;

  return Math.min(1, risk);
}
📁 server/risk/malpracticeDetector.ts
export function detectMalpracticeRisk(data: {
  diagnosis: string;
  redFlags: string[];
  disposition: string;
  protocolDeviation: boolean;
}) {
  let risk = 0;

  if (data.redFlags.length > 0 && data.disposition !== "ER") {
    risk += 0.5;
  }

  if (data.protocolDeviation) {
    risk += 0.3;
  }

  if (data.diagnosis === "uncertain") {
    risk += 0.2;
  }

  return {
    malpracticeRisk: Math.min(1, risk),
    level:
      risk > 0.7 ? "CRITICAL" :
      risk > 0.4 ? "HIGH" :
      "LOW"
  };
}

import crypto from "crypto";

export function generateAuditEntry(entry: any) {
  const timestamp = new Date().toISOString();

  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(entry) + timestamp)
    .digest("hex");

  return {
    ...entry,
    timestamp,
    hash
  };
}

type FunnelEvent = {
  source: string;
  step: "landing" | "intake" | "completed";
  timestamp: number;
};

const funnel: FunnelEvent[] = [];

export function trackEvent(event: FunnelEvent) {
  funnel.push(event);
}

export function getConversionRates() {
  const landing = funnel.filter(f => f.step === "landing").length;
  const intake = funnel.filter(f => f.step === "intake").length;
  const completed = funnel.filter(f => f.step === "completed").length;

  return {
    landing,
    intake,
    completed,
    conversion: completed / (landing || 1)
  };
}

import { runMultimodalFlow } from "../multimodal/orchestrator";
import { runExamProtocol } from "../protocols/examProtocolEngine";

export async function runVoiceClinic(callStream: AsyncIterable<any>, userId: string) {
  const session: any = {};

  for await (const chunk of callStream) {
    const text = chunk.transcript;

    const multimodal = await runMultimodalFlow({
      text,
      session
    });

    if (!session.protocolStarted && multimodal.nextStep === "start_protocol") {
      session.protocolStarted = true;

      const result = await runExamProtocol(
        selectProtocol(multimodal),
        buildContext({})
      );

      return {
        action: "protocol_complete",
        result
      };
    }

    yield {
      reply: generateVoiceReply(multimodal)
    };
  }
}
