⚡ One-Click Deploy Script

📁 deploy/deploy.ts

id="deploy-script"
import { execSync } from "child_process";

function run(cmd: string) {
  console.log(">>", cmd);
  execSync(cmd, { stdio: "inherit" });
}

async function deploy() {
  run("npm run build");

  // Replit deploy
  run("echo Deploying to Replit...");
  run("replit deploy");

  // Fly.io deploy
  run("flyctl deploy");

  // AWS ECS deploy
  run("cd infra && cdk deploy");

  console.log("✅ FULL DEPLOY COMPLETE");
}

deploy();

🧠 PART 2 — SELF-IMPROVING PROTOCOL ENGINE

Now your system learns from every patient + every physician decision

📁 File: server/learning/protocolLearningEngine.ts
id="learning-engine"
type Outcome = {
  protocolId: string;
  predicted: string;
  actual: string;
  physicianOverride?: boolean;
  features: any;
};

const weightStore: Record<string, number> = {};

export function updateWeights(outcome: Outcome) {
  const key = outcome.protocolId + ":" + outcome.predicted;

  if (!weightStore[key]) weightStore[key] = 1;

  if (outcome.predicted === outcome.actual) {
    weightStore[key] += 0.1;
  } else {
    weightStore[key] -= 0.2;
  }

  if (outcome.physicianOverride) {
    weightStore[key] -= 0.3;
  }
}

export function getWeight(protocolId: string, decision: string) {
  return weightStore[protocolId + ":" + decision] || 1;
}

📁 Backend: server/orchestration/roomManager.ts
id="room-manager"
type Room = {
  caseId: string;
  status: "active" | "pending" | "escalated";
  riskScore: number;
  complaint: string;
};

const rooms: Record<string, Room> = {};

export function updateRoom(caseId: string, data: Partial<Room>) {
  rooms[caseId] = { ...rooms[caseId], ...data };
}

export function getAllRooms() {
  return Object.values(rooms);
}

🧭 Frontend (React Dashboard)

📁 client/src/pages/OrchestrationDashboard.tsx

id="dashboard-ui"
import { useEffect, useState } from "react";

export default function Dashboard() {
  const [rooms, setRooms] = useState([]);

  useEffect(() => {
    fetch("/api/orchestration/rooms")
      .then(r => r.json())
      .then(d => setRooms(d.rooms));
  }, []);

  return (
    <div className="grid grid-cols-3 gap-4">
      {rooms.map((room: any) => (
        <div
          key={room.caseId}
          className={`p-4 rounded-xl shadow ${
            room.riskScore > 0.7 ? "bg-red-200" :
            room.riskScore > 0.3 ? "bg-yellow-200" :
            "bg-green-200"
          }`}
        >
          <h2>{room.complaint}</h2>
          <p>Risk: {room.riskScore}</p>
          <p>Status: {room.status}</p>
        </div>
      ))}
    </div>
  );
}
