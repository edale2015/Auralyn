import { db } from "../db";
import { sql } from "drizzle-orm";

export type PatientSession = {
  id: string;
  status: string;
  riskLevel?: string | null;
  safetyFlags?: string[];
  disposition?: unknown;
  approvedBy?: string | null;
  overrideData?: unknown;
};

export async function createPatientSessionPg(session: PatientSession) {
  await db.execute(sql`
    INSERT INTO patient_sessions
      (id, status, risk_level, safety_flags, disposition, approved_by, override_data)
    VALUES
      (${session.id},
       ${session.status},
       ${session.riskLevel ?? null},
       ${JSON.stringify(session.safetyFlags ?? [])}::jsonb,
       ${JSON.stringify(session.disposition ?? null)}::jsonb,
       ${session.approvedBy ?? null},
       ${JSON.stringify(session.overrideData ?? null)}::jsonb)
    ON CONFLICT (id) DO UPDATE SET
      status = EXCLUDED.status,
      risk_level = EXCLUDED.risk_level,
      safety_flags = EXCLUDED.safety_flags,
      disposition = EXCLUDED.disposition,
      approved_by = EXCLUDED.approved_by,
      override_data = EXCLUDED.override_data,
      updated_at = NOW();
  `);
}

export async function getPatientSessionsPg() {
  const result = await db.execute(sql`
    SELECT *
    FROM patient_sessions
    ORDER BY created_at DESC
    LIMIT 200
  `);
  return result.rows;
}

export async function updatePatientSessionPg(
  id: string,
  patch: Partial<PatientSession>
) {
  await db.execute(sql`
    UPDATE patient_sessions
    SET
      status = COALESCE(${patch.status ?? null}, status),
      risk_level = COALESCE(${patch.riskLevel ?? null}, risk_level),
      safety_flags = COALESCE(${patch.safetyFlags ? JSON.stringify(patch.safetyFlags) : null}::jsonb, safety_flags),
      disposition = COALESCE(${patch.disposition ? JSON.stringify(patch.disposition) : null}::jsonb, disposition),
      approved_by = COALESCE(${patch.approvedBy ?? null}, approved_by),
      override_data = COALESCE(${patch.overrideData ? JSON.stringify(patch.overrideData) : null}::jsonb, override_data),
      updated_at = NOW()
    WHERE id = ${id}
  `);
}

3. Physician-specific auth for approve/override

Yes, absolutely. admin is too blunt. These actions need a physician claim.

JWT middleware
server/auth/requirePhysician.ts
import jwt from "jsonwebtoken";
import { Request, Response, NextFunction } from "express";

type JwtClaims = {
  sub: string;
  role?: string;
  physician?: boolean;
  physicianId?: string;
};

export function requirePhysician(
  req: Request & { user?: JwtClaims },
  res: Response,
  next: NextFunction
) {
  const auth = req.headers.authorization;
  if (!auth?.startsWith("Bearer ")) {
    return res.status(401).json({ error: "Missing bearer token" });
  }

  try {
    const token = auth.slice("Bearer ".length);
    const decoded = jwt.verify(token, process.env.JWT_SECRET!) as JwtClaims;

    if (!decoded.physician && decoded.role !== "physician") {
      return res.status(403).json({ error: "Physician access required" });
    }

    req.user = decoded;
    next();
  } catch {
    return res.status(401).json({ error: "Invalid token" });
  }
}

export function calibrateConfidence(rawScore: number) {
  const clipped = Math.max(0, Math.min(1, rawScore));
  if (clipped > 0.9) return 0.85 + (clipped - 0.9) * 0.5;
  if (clipped < 0.2) return clipped * 0.7;
  return clipped;
}

Use it before returning diagnosis confidence to the UI or validation package.

server/notifications/notifier.ts
import { sendSMS } from "../integrations/twilio";
import { db } from "../db";
import { sql } from "drizzle-orm";

type AlertPayload = {
  patientId: string;
  riskLevel: "LOW" | "MEDIUM" | "HIGH";
  reasons: string[];
  traceId?: string;
};

// in-memory dedupe (per instance). Good enough for pilot.
// later: move to Redis for multi-instance dedupe.
const recentAlerts = new Map<string, number>();

function shouldSendAlert(key: string, windowMs = 5 * 60 * 1000) {
  const now = Date.now();
  const last = recentAlerts.get(key) || 0;
  if (now - last < windowMs) return false;
  recentAlerts.set(key, now);
  return true;
}

export async function notifyOnCallPhysician(payload: AlertPayload) {
  if (payload.riskLevel !== "HIGH") return { sent: false, reason: "not_high" };

  const key =
    `${payload.patientId}|${payload.riskLevel}|${payload.reasons.join("|")}`;

  if (!shouldSendAlert(key)) {
    return { sent: false, reason: "deduped" };
  }

  const to = process.env.ON_CALL_PHYSICIAN_NUMBER;
  if (!to) {
    console.warn("ON_CALL_PHYSICIAN_NUMBER missing");
    return { sent: false, reason: "missing_number" };
  }

  const body =
    `[HIGH RISK ALERT]\n` +
    `Patient: ${payload.patientId}\n` +
    `Reasons: ${payload.reasons.join(", ")}\n` +
    `${payload.traceId ? `Trace: ${payload.traceId}` : ""}`;

  await sendSMS(to, body);

  // persist alert
  await db.execute(sql`
    INSERT INTO alerts (patient_id, risk_level, reasons, channel, trace_id)
    VALUES (
      ${payload.patientId},
      ${payload.riskLevel},
      ${JSON.stringify(payload.reasons)}::jsonb,
      ${"sms"},
      ${payload.traceId ?? null}
    )
  `);

  return { sent: true };
}

import { runFullClinicalFlow } from "../orchestrator/clinicalOrchestrator";
import {
  createOrUpsertSession,
  getSessions,
  updateSession,
} from "../patient/sessionStorePg";
import { requirePhysician } from "../auth/requirePhysician";
import { logApproval } from "../audit/approvalAudit";
import { notifyOnCallPhysician } from "../notifications/notifier";
import { createTraceId } from "../audit/auditLogger";

const router = express.Router();

// GET queue (paged)
router.get("/queue", async (req, res) => {
  const limit = Math.min(Number(req.query.limit || 50), 200);
  const offset = Math.max(Number(req.query.offset || 0), 0);
  const rows = await getSessions(limit, offset);
  res.json(rows);
});

// POST new patient (always goes through orchestrator)
router.post("/session", async (req, res) => {
  const traceId = createTraceId();
  const input = { ...req.body, traceId };

  const result = await runFullClinicalFlow(input);

  const patientId = req.body.patientId || traceId;

  // persist session
  await createOrUpsertSession({
    id: patientId,
    status: result.success ? "pending_review" : "blocked",
    riskLevel: result?.safety?.level ?? "LOW",
    safetyFlags: result?.safety?.reasons ?? [],
    disposition: result,
  });

  // notify if high risk blocked
  if (result?.blocked && result?.safety?.level === "HIGH") {
    await notifyOnCallPhysician({
      patientId,
      riskLevel: "HIGH",
      reasons: result.safety.reasons,
      traceId,
    });
  }

  res.json({ ...result, traceId, patientId });
});

// POST approve
router.post("/approve/:id", requirePhysician, async (req: any, res) => {
  const id = req.params.id;
  const physicianId = req.user.physicianId || req.user.sub;

  await updateSession(id, {
    status: "approved",
    approvedBy: physicianId,
  });

  await logApproval({
    patientId: id,
    physicianId,
    action: "approve",
  });

  res.json({ success: true });
});

  const id = req.params.id;

    status: "overridden",
    approvedBy: physicianId,
    overrideData: req.body,
  });

  await logApproval({
    patientId: id,
    physicianId,
    action: "override",
    overrideData: req.body,
  });

  res.json({ success: true });
});

  await logApproval({
