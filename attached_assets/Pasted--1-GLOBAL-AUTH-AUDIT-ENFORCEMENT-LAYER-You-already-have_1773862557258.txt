// server/middleware/requireRole.ts

import { Request, Response, NextFunction } from "express";

export function requireRole(roles: string[]) {
  return (req: any, res: Response, next: NextFunction) => {
    const user = req.user;

    if (!user || !roles.includes(user.role)) {
      return res.status(403).json({ error: "Forbidden" });
    }

    next();
  };
}

// server/middleware/auditMiddleware.ts

import { appendAuditRow } from "../repos/getPackRepository";
import { v4 as uuidv4 } from "uuid";

export function auditMiddleware(action: string) {
  return async (req: any, res: any, next: any) => {
    const start = Date.now();

    res.on("finish", async () => {
      try {
        await appendAuditRow({
          id: uuidv4(),
          user: req.user?.id || "unknown",
          role: req.user?.role || "unknown",
          action,
          path: req.originalUrl,
          method: req.method,
          status: res.statusCode,
          latency_ms: Date.now() - start,
          timestamp: new Date().toISOString()
        });
      } catch (e) {
        console.error("Audit logging failed:", e);
      }
    });

    next();
  };
}

// server/engines/highScaleSimulationEngine.ts

import pLimit from "p-limit";

const limit = pLimit(10); // concurrency control

export async function runHighScaleSimulations(repo, perPack = 1000) {
  const packs = await repo.getAll();

  const results: any[] = [];

  for (const pack of packs) {
    const simulations = Array.from({ length: perPack });

    const packResults = await Promise.all(
      simulations.map(() =>
        limit(async () => {
          const input = generateSyntheticCase(pack);
          const result = await runClinicalPipeline(input, repo);

          const groundTruth = generateGroundTruth(pack);

          return {
            predicted: result,
            groundTruth,
            correct: result.diagnosis === groundTruth.diagnosis,
            underTriage:
              groundTruth.triage === "ER" && result.triage !== "ER",
            overTriage:
              groundTruth.triage !== "ER" && result.triage === "ER"
          };
        })
      )
    );

    results.push({
      packId: pack.id,
      total: packResults.length,
      accuracy:
        packResults.filter(r => r.correct).length / packResults.length,
      underTriageRate:
        packResults.filter(r => r.underTriage).length / packResults.length,
      overTriageRate:
        packResults.filter(r => r.overTriage).length / packResults.length
    });
  }

  return results;
}

// server/monitoring/metricsStore.ts

const metrics = {
  totalRequests: 0,
  totalErrors: 0,
  latencies: [] as number[]
};

export function recordRequest(latency: number, isError: boolean) {
  metrics.totalRequests++;

  if (isError) metrics.totalErrors++;

  metrics.latencies.push(latency);

  if (metrics.latencies.length > 1000) {
    metrics.latencies.shift(); // rolling window
  }
}

export function getMetrics() {
  const avgLatency =
    metrics.latencies.reduce((a, b) => a + b, 0) /
    (metrics.latencies.length || 1);

  return {
    totalRequests: metrics.totalRequests,
    errorRate: metrics.totalErrors / (metrics.totalRequests || 1),
    avgLatency
  };
}
🔥 Monitoring Middleware
// server/middleware/metricsMiddleware.ts

import { recordRequest } from "../monitoring/metricsStore";

export function metricsMiddleware(req: any, res: any, next: any) {
  const start = Date.now();

  res.on("finish", () => {
    const latency = Date.now() - start;
    const isError = res.statusCode >= 400;

    recordRequest(latency, isError);
  });

  next();
}
