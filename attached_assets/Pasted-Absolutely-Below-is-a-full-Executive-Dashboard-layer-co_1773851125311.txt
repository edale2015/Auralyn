import { ExecutiveSnapshot } from "./executiveStore";

export function buildTrendSeries(rows: ExecutiveSnapshot[]) {
  return rows
    .slice()
    .sort(
      (a, b) =>
        new Date(a.snapshotDate).getTime() - new Date(b.snapshotDate).getTime()
    )
    .map(r => {
      const marginPct = r.avgRevenuePerCase
        ? ((r.avgRevenuePerCase - r.avgCostPerCase) / r.avgRevenuePerCase) * 100
        : 0;

      return {
        date: r.snapshotDate.slice(0, 10),
        totalCases: r.totalCases,
        overrideRatePct: Number((r.overrideRate * 100).toFixed(2)),
        avgSatisfaction: Number(r.avgSatisfaction.toFixed(2)),
        marginPct: Number(marginPct.toFixed(2))
      };
    });
}

import { ComplaintSnapshot, PhysicianSnapshot } from "./executiveStore";

export function buildComplaintDrilldown(rows: ComplaintSnapshot[]) {
  return rows
    .map(r => {
      let status: "good" | "watch" | "critical" = "good";
      if (r.overrideRate > 0.18 || r.escalationRate > 0.2) status = "critical";
      else if (r.overrideRate > 0.1 || r.escalationRate > 0.12) status = "watch";

      return { ...r, status };
    })
    .sort((a, b) => b.totalCases - a.totalCases);
}

export function buildPhysicianDrilldown(rows: PhysicianSnapshot[]) {
  return rows
    .map(r => {
      let status: "good" | "watch" | "critical" = "good";
      if (r.overrideRate > 0.18 || r.avgSatisfaction < 4.0) status = "critical";
      else if (r.overrideRate > 0.1 || r.avgReviewSeconds > 30) status = "watch";
      return { ...r, status };
    })
    .sort((a, b) => b.totalCases - a.totalCases);
}

/server/services/exportCsv.ts
export function toCsv(rows: Record<string, unknown>[]) {
  if (!rows.length) return "";

  const headers = Object.keys(rows[0]);
  const escape = (value: unknown) =>
    `"${String(value ?? "").replace(/"/g, '""')}"`;

  return [
    headers.join(","),
    ...rows.map(row => headers.map(h => escape(row[h])).join(","))
  ].join("\n");
}

        <ExecutiveAlertsTab rows={alerts} onCreate={createDemoAlert} />
      )}
      {tab === "email-preview" && (
        <ExecutiveEmailPreviewTab
          preview={emailPreview}
          onGenerate={generateEmailPreview}
        />
      )}
      {tab === "export" && <ExecutiveExportTab onDownloadCsv={downloadCsv} />}
    </div>
  );
}
