interface FusionInput {
  differential: any[];
  safetyAlerts: string[];
  urgency: { level: string; score: number };
  contradictionReport: any;
}

interface FusionOutput {
  finalPriority: "routine" | "urgent" | "emergency";
  dominantSignal: string;
  reasoningSummary: string;
}

export function runClinicalFusion(input: FusionInput): FusionOutput {
  const { safetyAlerts, urgency, differential } = input;

  // 🚨 Safety dominates everything
  if (safetyAlerts.length > 0) {
    return {
      finalPriority: "emergency",
      dominantSignal: "safety_alert",
      reasoningSummary: `Safety alerts present: ${safetyAlerts.join(", ")}`,
    };
  }

  // Urgency next
  if (urgency.level === "high" || urgency.level === "emergency") {
    return {
      finalPriority: "urgent",
      dominantSignal: "urgency_score",
      reasoningSummary: `High urgency score: ${urgency.score}`,
    };
  }

  // Otherwise differential-driven
  const topDx = differential[0]?.diagnosis ?? "unknown";

  return {
    finalPriority: "routine",
    dominantSignal: "differential",
    reasoningSummary: `Top differential: ${topDx}`,
  };
}

export function computeUncertainty({
  failures,
  differential,
}: {
  failures: string[];
  differential: any[];
}): number {
  let uncertainty = 0;

  // Service failures increase uncertainty
  uncertainty += failures.length * 0.1;

  // Weak differential increases uncertainty
  if (!differential.length) uncertainty += 0.3;

  return Math.min(1, uncertainty);
}

🧩 1. Multi-Agent Debate Engine (FULL CODE)
📁 server/telemedicine/debateEngine.ts
export interface AgentOpinion {
  agent: string;
  conclusion: string;
  confidence: number; // 0–1
  reasoning: string;
}

export interface DebateResult {
  winner: AgentOpinion;
  consensusScore: number;
  disagreement: number;
  opinions: AgentOpinion[];
}

export function runDebate(opinions: AgentOpinion[]): DebateResult {
  if (!opinions.length) {
    return {
      winner: {
        agent: "none",
        conclusion: "No decision",
        confidence: 0,
        reasoning: "No agents responded",
      },
      consensusScore: 0,
      disagreement: 1,
      opinions: [],
    };
  }

  // 🧠 Normalize weights
  const totalConfidence = opinions.reduce((sum, o) => sum + o.confidence, 0);

  const weighted = opinions.map(o => ({
    ...o,
    weight: totalConfidence > 0 ? o.confidence / totalConfidence : 0,
  }));

  // 🏆 Winner = highest weighted confidence
  const winner = weighted.reduce((a, b) => (a.weight > b.weight ? a : b));

  // 🤝 Consensus = how similar conclusions are
  const conclusionCounts: Record<string, number> = {};
  for (const o of opinions) {
    conclusionCounts[o.conclusion] = (conclusionCounts[o.conclusion] || 0) + 1;
  }

  const maxAgreement = Math.max(...Object.values(conclusionCounts));
  const consensusScore = maxAgreement / opinions.length;

  // ⚠️ Disagreement = spread of confidence
  const variance =
    opinions.reduce((sum, o) => sum + Math.pow(o.confidence - 0.5, 2), 0) /
    opinions.length;

  return {
    winner,
    consensusScore,
    disagreement: variance,
    opinions,
  };
}

interface EngineStats {
  name: string;
  successes: number;
  failures: number;
}

const engineStats: Record<string, EngineStats> = {};

export function registerEngine(name: string) {
  if (!engineStats[name]) {
    engineStats[name] = { name, successes: 1, failures: 1 }; // avoid divide by zero
  }
}

export function recordOutcome(name: string, success: boolean) {
  const stat = engineStats[name];
  if (!stat) return;

  success ? stat.successes++ : stat.failures++;
}

// 🎯 Thompson Sampling
export function selectBestEngine(): string {
  let bestEngine = "";
  let bestScore = -1;

  for (const stat of Object.values(engineStats)) {
    const score =
      Math.random() *
      (stat.successes / (stat.successes + stat.failures));

    if (score > bestScore) {
      bestScore = score;
      bestEngine = stat.name;
    }
  }

  return bestEngine;
}

import { Server } from "ws";

interface CognitivePayload {
  caseId: string;
  agents: any[];
  council: any;
  finalDecision: any;
  reasoningGraph: any[];
  timestamp: number;
}

const clients = new Set<any>();

export function initTelemedCognitiveStream(wss: Server) {
  wss.on("connection", (ws) => {
    clients.add(ws);

    ws.on("close", () => {
      clients.delete(ws);
    });
  });
}

export function broadcastCognitiveUpdate(payload: CognitivePayload) {
  const message = JSON.stringify({
    type: "telemed_cognition",
    data: payload,
  });

  for (const client of clients) {
    if (client.readyState === 1) {
      client.send(message);
    }
  }
}

import { useEffect, useState } from "react";

export default function TelemedCognitivePanel() {
  const [data, setData] = useState<any>(null);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:3000/ws/telemed");

    ws.onmessage = (event) => {
      const msg = JSON.parse(event.data);
      if (msg.type === "telemed_cognition") {
        setData(msg.data);
      }
    };

    return () => ws.close();
  }, []);

  if (!data) return <div className="p-4">Waiting for cognitive stream...</div>;

  return (
    <div className="p-4 space-y-4">

      {/* 🧠 AGENTS */}
      <div>
        <h2 className="text-xl font-bold">Agents</h2>
        {data.agents.map((a: any) => (
          <div key={a.agentId} className="border p-2 rounded">
            <div><b>{a.agentId}</b> ({a.domain})</div>
            <div>Conclusion: {a.conclusion}</div>
            <div>Confidence: {(a.confidence * 100).toFixed(1)}%</div>
          </div>
        ))}
      </div>

      {/* ⚖️ COUNCIL */}
      <div>
        <h2 className="text-xl font-bold">Council Decision</h2>
        <div>Winner: {data.council?.winner?.agentId}</div>
        <div>Confidence: {data.council?.winner?.confidence}</div>
      </div>

      {/* 🚨 FINAL */}
      <div>
        <h2 className="text-xl font-bold">Final Decision</h2>
        <div>{JSON.stringify(data.finalDecision)}</div>
      </div>

    </div>
  );
}

export interface RequeryPolicyInput {
  uncertainty: number;
  debate?: {
    consensusScore?: number;
    disagreement?: number;
  };
  subServiceFailures?: string[];
  safetyAlerts?: string[];
}

export interface RequeryDecision {
  shouldRequery: boolean;
  reason: string;
  priority: "low" | "medium" | "high";
}

export function shouldTriggerRequery(input: RequeryPolicyInput): RequeryDecision {
  const uncertainty = input.uncertainty ?? 0;
  const consensus = input.debate?.consensusScore ?? 1;
  const disagreement = input.debate?.disagreement ?? 0;
  const failures = input.subServiceFailures ?? [];
  const safetyAlerts = input.safetyAlerts ?? [];

  if (safetyAlerts.length > 0) {
    return {
      shouldRequery: false,
      reason: "Safety alert present, escalate immediately instead of re-querying",
      priority: "high",
    };
  }

  if (uncertainty >= 0.55) {
    return {
      shouldRequery: true,
      reason: `High uncertainty (${uncertainty.toFixed(2)})`,
      priority: "high",
    };
  }

  if (consensus <= 0.45) {
    return {
      shouldRequery: true,
      reason: `Low consensus (${consensus.toFixed(2)})`,
      priority: "high",
    };
  }

  if (disagreement >= 0.25) {
    return {
      shouldRequery: true,
      reason: `High disagreement (${disagreement.toFixed(2)})`,
      priority: "medium",
    };
  }

  if (failures.length >= 2) {
    return {
      shouldRequery: true,
      reason: `Multiple sub-service failures: ${failures.join(", ")}`,
      priority: "medium",
    };
  }

  return {
    shouldRequery: false,
    reason: "No re-query criteria met",
    priority: "low",
  };
}

export interface CandidateQuestion {
  id: string;
  text: string;
  target: "diagnosis" | "triage" | "safety" | "treatment";
  infoGain: number;
  blockingRisk?: boolean;
}

export interface NextBestQuestionResult {
  winner: CandidateQuestion | null;
  ranked: CandidateQuestion[];
  reason: string;
}

export function chooseNextBestQuestion(params: {
  adaptiveQuestions?: Array<{ id?: string; text?: string }>;
  uncertainty: number;
  debate?: { disagreement?: number; consensusScore?: number };
  missingServices?: string[];
}): NextBestQuestionResult {
  const adaptive = params.adaptiveQuestions ?? [];
  const disagreement = params.debate?.disagreement ?? 0;
  const missing = params.missingServices ?? [];

  const ranked: CandidateQuestion[] = adaptive
    .filter(q => q?.text)
    .map((q, idx) => {
      let infoGain = 0.3;

      if (missing.includes("adaptiveQuestions")) infoGain += 0.05;
      if (missing.includes("differential")) infoGain += 0.1;
      if (missing.includes("urgencyScore")) infoGain += 0.15;
      if (disagreement > 0.2) infoGain += 0.15;
      if (params.uncertainty > 0.5) infoGain += 0.2;

      return {
        id: q.id ?? `adaptive-${idx + 1}`,
        text: q.text!,
        target: disagreement > 0.2 ? "triage" : "diagnosis",
        infoGain: Math.min(1, infoGain),
      };
    })
    .sort((a, b) => b.infoGain - a.infoGain);

  return {
    winner: ranked[0] ?? null,
    ranked,
    reason: ranked[0]
      ? `Selected highest-information question (${ranked[0].infoGain.toFixed(2)})`
      : "No adaptive questions available",
  };
}

import { chooseNextBestQuestion } from "./nextBestQuestionEngine";
import { shouldTriggerRequery } from "./requeryPolicy";
import { buildSelectiveRerunPlan } from "./selectiveRerunEngine";

export interface RequeryExecutionResult {
  triggered: boolean;
  reason: string;
  questionAsked: string | null;
  rerunPlan: any | null;
  comparison: {
    pass1Urgency?: string;
    pass2Urgency?: string;
    changed: boolean;
  };
}

export async function runRequeryLoop(params: {
  pass1Result: any;
  answerProvider?: (question: string) => Promise<string | null>;
  rerunAssistant: (augmentedAnswers: Record<string, string>, rerunPlan: any) => Promise<any>;
}): Promise<RequeryExecutionResult> {
  const pass1 = params.pass1Result;

  const policy = shouldTriggerRequery({
    uncertainty: pass1.uncertainty ?? 0,
    debate: pass1.debate,
    subServiceFailures: pass1.subServiceFailures,
    safetyAlerts: pass1.safetyAlerts,
  });

  if (!policy.shouldRequery) {
    return {
      triggered: false,
      reason: policy.reason,
      questionAsked: null,
      rerunPlan: null,
      comparison: { changed: false },
    };
  }

  const nextQuestion = chooseNextBestQuestion({
    adaptiveQuestions: pass1.adaptiveQuestions,
    uncertainty: pass1.uncertainty ?? 0,
    debate: pass1.debate,
    missingServices: pass1.subServiceFailures,
  });

  if (!nextQuestion.winner) {
    return {
      triggered: false,
      reason: `Re-query indicated but no question available: ${policy.reason}`,
      questionAsked: null,
      rerunPlan: null,
      comparison: { changed: false },
    };
  }

  const answer = params.answerProvider
    ? await params.answerProvider(nextQuestion.winner.text)
    : null;

  if (!answer) {
    return {
      triggered: true,
      reason: `${policy.reason}; awaiting user answer`,
      questionAsked: nextQuestion.winner.text,
      rerunPlan: null,
      comparison: { changed: false },
    };
  }

  const rerunPlan = buildSelectiveRerunPlan({
    uncertainty: pass1.uncertainty ?? 0,
    subServiceFailures: pass1.subServiceFailures,
    debate: pass1.debate,
  });

  const pass2 = await params.rerunAssistant(
    {
      ...(pass1.answers ?? {}),
      [nextQuestion.winner.id]: answer,
    },
    rerunPlan
  );

  return {
    triggered: true,
    reason: `Re-query completed: ${policy.reason}`,
    questionAsked: nextQuestion.winner.text,
    rerunPlan,
    comparison: {
      pass1Urgency: pass1.triage?.level,
      pass2Urgency: pass2.triage?.level,
      changed: pass1.triage?.level !== pass2.triage?.level,
    },
  };
}

interface CaseMemoryEntry {
  id: string;
  caseId: string;
  iteration: number;
  timestamp: number;

  winnerAgent: string;
  triageLevel: string;
  uncertainty: number;

  questionAsked?: string;
  changedFromPrior: boolean;

const memoryStore: Record<string, CaseMemoryEntry[]> = {};

export function logCaseMemory(entry: Omit<CaseMemoryEntry, "id" | "timestamp">) {
  const fullEntry: CaseMemoryEntry = {
    ...entry,
    id: uuidv4(),
    timestamp: Date.now(),
  };

  if (!memoryStore[entry.caseId]) {
    memoryStore[entry.caseId] = [];
  }

  memoryStore[entry.caseId].push(fullEntry);

  return fullEntry;
}

export function getCaseMemory(caseId: string): CaseMemoryEntry[] {
  return memoryStore[caseId] || [];
}

export function getLastMemory(caseId: string): CaseMemoryEntry | null {
  const arr = memoryStore[caseId];
  return arr?.length ? arr[arr.length - 1] : null;
}

export function buildEscalationBundle(params: {
  result: any;
  requery?: any;
}): EscalationBundle | null {
  const { result } = params;

  const isEmergency =
    result.triage?.level === "emergency" ||
    (result.safetyAlerts?.length ?? 0) > 0;

  const isUrgent =
    result.triage?.level === "high" ||
    (result.uncertainty ?? 0) > 0.6;

  if (!isEmergency && !isUrgent) return null;

  return {
    caseId: result.caseId,
    priority: isEmergency ? "emergency" : "urgent",
    reason: isEmergency
      ? "Safety or emergency triage triggered"
      : "High uncertainty or high-risk triage",

    summary: `Complaint: ${result.complaint}`,
    triageLevel: result.triage?.level,

    topConcerns: [
      ...(result.safetyAlerts ?? []),
      ...(result.differential?.slice(0, 2).map((d: any) => d.diagnosis) ?? []),
    ],

    missingCriticalData: result.requery?.questionAsked
      ? [result.requery.questionAsked]
      : [],

    recommendedActions: isEmergency
      ? ["Immediate physician review", "Consider ER referral"]
      : ["Expedite telemedicine follow-up", "Collect missing data"],

    createdAt: Date.now(),

    specialty,
    winner,
    consideredAgents: filtered.map(a => a.agentId),
  };
}

interface OutcomeEvent {
  caseId: string;
  finalDecision: string;
  triageLevel: string;

  actualOutcome: string; // e.g. "admitted", "discharged", "ICU"
  label: OutcomeLabel;

  winningAgent: string;
  timestamp: number;
}

interface AgentPerformance {
  agentId: string;
  total: number;
  correct: number;
  incorrect: number;
  overtriage: number;
  undertriage: number;
  score: number;
}

const outcomeLog: OutcomeEvent[] = [];
const agentPerformance: Record<string, AgentPerformance> = {};

export function recordOutcome(event: OutcomeEvent) {
  outcomeLog.push(event);

  if (!agentPerformance[event.winningAgent]) {
    agentPerformance[event.winningAgent] = {
      agentId: event.winningAgent,
      total: 0,
      correct: 0,
      incorrect: 0,
      overtriage: 0,
      undertriage: 0,
      score: 0.5,
    };
  }

  const stat = agentPerformance[event.winningAgent];

  stat.total++;

  if (event.label === "correct") stat.correct++;
  if (event.label === "incorrect") stat.incorrect++;
  if (event.label === "overtriage") stat.overtriage++;
  if (event.label === "undertriage") stat.undertriage++;

  // 🎯 scoring function (you can tune)
  stat.score =
    (stat.correct * 1.0 +
      stat.overtriage * 0.3 -
      stat.undertriage * 1.2 -
      stat.incorrect * 1.0) /
    stat.total;

  return stat;
}

export function getAgentPerformance() {
  return Object.values(agentPerformance);
}

const metrics: Record<string, EngineMetrics> = {};

export function trackEngineCall(engine: string, latencyMs: number, failed: boolean) {
  if (!metrics[engine]) {
    metrics[engine] = {
      engine,
      calls: 0,
      failures: 0,
      totalLatencyMs: 0,
    };
  }

  const m = metrics[engine];

  m.calls++;
  m.totalLatencyMs += latencyMs;
  if (failed) m.failures++;
}

export function getEngineMetrics() {
  return Object.values(metrics).map(m => ({
    engine: m.engine,
    calls: m.calls,
    failureRate: m.calls ? m.failures / m.calls : 0,
    avgLatency: m.calls ? m.totalLatencyMs / m.calls : 0,
  }));
}

  summary: string;

export function computeCounterfactuals(params: {
  result: any;
}): CounterfactualResult {
  const { result } = params;

  const cf: Counterfactual[] = [];

  const triage = result.triage?.level;

  // 🚑 Example: chest pain risk modifiers
  if (result.complaint?.toLowerCase().includes("chest")) {
    cf.push({
      variable: "diaphoresis",
      current: "absent",
      alternative: "present",
      effect: "Would increase likelihood of ACS and elevate triage",
      impactScore: 0.9,
    });

    cf.push({
      variable: "exertional pain",
      current: "unknown",
      alternative: "present",
      effect: "Would increase urgency to high/emergency",
      impactScore: 0.85,
    });
  }

  // 🧪 Infection modifier
  if (!result.safetyAlerts?.length) {
    cf.push({
      variable: "fever",
      current: "unknown",
      alternative: "present",
      effect: "Would increase infectious differential weight",
      impactScore: 0.6,
    });
  }

  // 🫁 Respiratory
  if (!result.triage?.level || result.triage.level !== "emergency") {
    cf.push({
      variable: "shortness_of_breath",
      current: "absent",
      alternative: "present",
      effect: "Would escalate triage to emergency",
      impactScore: 0.95,
    });
  }

  const sorted = cf.sort((a, b) => b.impactScore - a.impactScore);

  return {
    keyFactors: sorted.slice(0, 5),
    summary: sorted.length
      ? `Top factor: ${sorted[0].variable} could significantly change outcome`
      : "No strong counterfactual factors identified",
  };
}

export interface TrajectoryPrediction {
  trend: "improving" | "stable" | "worsening";
  riskScore: number; // 0–1
  timeHorizon: string;
  drivers: string[];
}

export function predictTrajectory(params: {
  result: any;
  memory?: any[];
}): TrajectoryPrediction {
  const { result, memory } = params;

  let risk = 0.3;

  // 🚨 Safety signals
  if (result.safetyAlerts?.length) {
    risk += 0.5;
  }

  // 🚑 High triage
  if (result.triage?.level === "high") risk += 0.2;
  if (result.triage?.level === "emergency") risk += 0.5;

  // 🧠 Uncertainty increases instability
  if (result.uncertainty > 0.5) risk += 0.2;

  // 🧬 Trend from memory
  if (memory?.length >= 2) {
    const last = memory[memory.length - 1];
    const prev = memory[memory.length - 2];

    if (last.triageLevel !== prev.triageLevel) {
      risk += 0.15;
    }
  }

  const trend =
    risk > 0.7 ? "worsening" :
    risk > 0.4 ? "stable" :
    "improving";

  return {
    trend,
    riskScore: Math.min(1, risk),
    timeHorizon: risk > 0.7 ? "hours" : "1-2 days",
    drivers: [
      result.triage?.level,
      ...(result.safetyAlerts ?? []),
    ],
  };
}

interface DiagnosisPrior {
  diagnosis: string;
  prior: number;
}

interface Evidence {
  feature: string;
  likelihoodRatio: number; // LR+ or LR-
  present: boolean;
}

interface BayesianResult {
  diagnosis: string;
  posterior: number;
}

export function runBayesianUpdate(params: {
  priors: DiagnosisPrior[];
  evidence: Evidence[];
}): BayesianResult[] {
  const { priors, evidence } = params;

  const updated = priors.map(p => {
    let odds = p.prior / (1 - p.prior);

    for (const e of evidence) {
      if (e.present) {
        odds *= e.likelihoodRatio;
      } else {
        odds *= 1 / e.likelihoodRatio;
      }
    }

    const posterior = odds / (1 + odds);

    return {
      diagnosis: p.diagnosis,
      posterior,
    };
  });

  // normalize
  const total = updated.reduce((sum, d) => sum + d.posterior, 0);

  return updated.map(d => ({
    ...d,
    posterior: total ? d.posterior / total : d.posterior,
  })).sort((a, b) => b.posterior - a.posterior);
}

interface SimulationScenario {
  name: string;
  intervention: "none" | "treatment" | "delay";
}

interface SimulationResult {
  scenario: string;
  riskScore: number;
  outcome: string;
  timeToEvent: string;
}

export function runDigitalTwin(params: {
  result: any;
}): SimulationResult[] {
  const baseRisk = params.result.trajectory?.riskScore || 0.3;

  const scenarios: SimulationScenario[] = [
    { name: "No Action", intervention: "none" },
    { name: "Immediate Treatment", intervention: "treatment" },
    { name: "Delayed Care", intervention: "delay" },
  ];

  return scenarios.map(s => {
    let risk = baseRisk;

    if (s.intervention === "none") {
      risk += 0.2;
    }

    if (s.intervention === "treatment") {
      risk -= 0.25;
    }

    if (s.intervention === "delay") {
      risk += 0.35;
    }

    risk = Math.max(0, Math.min(1, risk));

    return {
      scenario: s.name,
      riskScore: risk,
      outcome:
        risk > 0.7 ? "High likelihood of deterioration" :
        risk > 0.4 ? "Moderate risk" :
        "Low risk",
      timeToEvent:
        risk > 0.7 ? "hours" :
        risk > 0.4 ? "1-2 days" :
        "stable",
    };
  });
}

interface QAFlag {
  type: "safety_miss" | "overtriage" | "undertriage" | "contradiction";
  severity: "low" | "medium" | "high";
  message: string;
}

interface QAResult {
  caseId: string;
  score: number; // 0–1
  flags: QAFlag[];
}

export function runQA(params: {
  result: any;
}): QAResult {
  const { result } = params;

  const flags: QAFlag[] = [];
  let score = 1;

  // 🚨 Safety miss
  if (
    result.safetyAlerts?.length &&
    result.triage.level !== "emergency"
  ) {
    flags.push({
      type: "safety_miss",
      severity: "high",
      message: "Safety alert present but not escalated",
    });
    score -= 0.5;
  }

  // ⚖️ Undertriage
  if (result.trajectory?.riskScore > 0.7 &&
      result.triage.level !== "emergency") {
    flags.push({
      type: "undertriage",
      severity: "high",
      message: "High predicted risk but triage not escalated",
    });
    score -= 0.3;
  }

  // ⚠️ Overtriage
  if (
    result.triage.level === "emergency" &&
    (result.safetyAlerts?.length ?? 0) === 0 &&
    result.uncertainty < 0.3
  ) {
    flags.push({
      type: "overtriage",
      severity: "medium",
      message: "Emergency without strong justification",
    });
    score -= 0.2;
  }

  // 🔍 Contradictions
  if (result.contradictionReport?.issues?.length) {
    flags.push({
      type: "contradiction",
      severity: "medium",
      message: "Conflicting clinical signals detected",
    });
    score -= 0.2;
  }

  return {
    caseId: result.caseId,
    score: Math.max(0, score),
    flags,
  };
}

  uncertaintyThreshold: number;

export function runMetaLearning() {

  for (const a of agents) {
    // 🛑 Penalize undertriage heavily

    // ⚖️ Too much overtriage → relax threshold

  systemThresholds.escalationThreshold = Math.max(
    0.5,
    Math.min(0.9, systemThresholds.escalationThreshold)

  return systemThresholds;
}

const complaintCounts: Record<string, number[]> = {};

export function trackComplaint(complaint: string) {
  if (!complaintCounts[complaint]) {
    complaintCounts[complaint] = [];
  }

  complaintCounts[complaint].push(Date.now());
}

export function detectOutbreaks() {
  const alerts: string[] = [];

  for (const [complaint, timestamps] of Object.entries(complaintCounts)) {
    const recent = timestamps.filter(t => Date.now() - t < 24 * 60 * 60 * 1000);

    if (recent.length > 50) {
      alerts.push(`Spike detected: ${complaint}`);
    }
  }

  return alerts;
}

async function withTimeout<T>(fn: () => Promise<T>, ms: number, fallback: T): Promise<T> {
  try {
    return await Promise.race([
      fn(),
      new Promise<T>((resolve) => setTimeout(() => resolve(fallback), ms)),
    ]);
  } catch {
    return fallback;
  }
}

import { AgentOpinion } from "./agentTypes";

export function runDebate(opinions: AgentOpinion[]) {
  if (!opinions.length) {
    return { winner: null, consensus: 0, disagreement: 1 };
  }

  const total = opinions.reduce((s, o) => s + o.confidence, 0);

  const weighted = opinions.map(o => ({
    ...o,
    weight: total ? o.confidence / total : 0,
  }));

  const winner = weighted.reduce((a, b) =>
    a.weight > b.weight ? a : b
  );

  const unique = new Set(opinions.map(o => o.conclusion)).size;

  return {
    winner,
    consensus: 1 / unique,
    disagreement: unique / opinions.length,
    opinions,
  };
}

export function computeUncertainty(params: {
  agents: any[];
  debate: any;
}) {
  const { agents, debate } = params;

  let uncertainty = 0;

  // disagreement adds uncertainty
  uncertainty += debate.disagreement * 0.5;

  // low consensus increases uncertainty
  uncertainty += (1 - debate.consensus) * 0.3;

  // weak confidence across agents
  const avgConfidence =
    agents.reduce((s, a) => s + a.confidence, 0) / (agents.length || 1);

  uncertainty += (1 - avgConfidence) * 0.2;

  return Math.min(1, uncertainty);
}

    factors.push({
      factor: "shortness_of_breath",

    factors.push({
      factor: "fever",
      effect: "Would increase infection likelihood",
      impact: 0.6,

🔮 3. TRAJECTORY ENGINE
📁 server/reasoning/trajectoryEngine.ts
export function predictTrajectory(result: any, memory: any[]) {
  let risk = result.uncertainty;

  if (result.triage.level === "high") risk += 0.2;
  if (result.triage.level === "emergency") risk += 0.5;

  if (memory.length > 1) {
    const last = memory[memory.length - 1];
    const prev = memory[memory.length - 2];

    if (last.triage !== prev.triage) {
      risk += 0.1;
    }
  }

  const trend =
    risk > 0.7 ? "worsening" :
    risk > 0.4 ? "stable" :
    "improving";

  return {
    trend,
    risk: Math.min(1, risk),
  };
}
🧬 4. DIGITAL TWIN SIMULATION

export function simulateFuture(result: any) {
  const scenarios = [];

  const base = result.uncertainty;

  scenarios.push({
    scenario: "no_action",
    risk: Math.min(1, base + 0.3),
  });

  scenarios.push({
    scenario: "treatment",
    risk: Math.max(0, base - 0.3),
  });

  scenarios.push({
    scenario: "delayed",
    risk: Math.min(1, base + 0.5),
  });

  return scenarios;
}

const weights: Record<string, number> = {};

export function updateWeights(agent: string, correct: boolean) {
  if (!weights[agent]) weights[agent] = 1;

  if (correct) {
    weights[agent] += 0.1;
  } else {
    weights[agent] -= 0.2;
  }

  weights[agent] = Math.max(0.2, Math.min(3, weights[agent]));
}

export function getWeight(agent: string) {
  return weights[agent] || 1;
}
