1. Shared types
// shared/systemPackTypes.ts

export type PackTier = "symptom" | "modifier" | "clinician_algorithm";

export type Disposition =
  | "self_care"
  | "office_followup"
  | "telemed_now"
  | "urgent_care"
  | "er_now";

export type QuestionType =
  | "yes_no"
  | "single_select"
  | "multi_select"
  | "text"
  | "number"
  | "duration"
  | "severity";

export interface IntakeQuestionOption {
  label: string;
  value: string;
  redFlag?: boolean;
}

export interface IntakeQuestion {
  id: string;
  prompt: string;
  type: QuestionType;
  priority: number;
  required?: boolean;
  options?: IntakeQuestionOption[];
  helpText?: string;
}

export interface SymptomPack {
  id: string;
  system: string;
  tier: "symptom";
  title: string;
  aliases: string[];
  likelyDisposition: Disposition;
  questions: IntakeQuestion[];
  redFlags: string[];
  autoEscalateRules: string[];
  autoReviewRules: string[];
  planTemplateKey: string;
}

export interface ModifierPack {
  id: string;
  system: string;
  tier: "modifier";
  title: string;
  appliesToSymptoms: string[];
  triggers: string[];
  riskAdjustments: Array<{
    condition: string;
    action: "raise_risk" | "force_review" | "force_escalation";
    amount?: number;
    reason: string;
  }>;
}

export interface ClinicianAlgorithmPack {
  id: string;
  system: string;
  tier: "clinician_algorithm";
  title: string;
  entryCriteria: string[];
  requiredInputs: string[];
  outputActions: string[];
  notes?: string[];
}

6. Recommended engine behavior

Use this order:

Step 1

Run universal safety screen

Step 2

Match one primary symptom pack

Step 3

Apply all relevant modifier packs

Step 4

If red flags or complexity present, surface clinician algorithm packs

Step 5

Generate:

patient-facing next questions

queue priority

review requirement

clinician checklist if needed

That gives you a layered engine instead of one giant soup pot.
