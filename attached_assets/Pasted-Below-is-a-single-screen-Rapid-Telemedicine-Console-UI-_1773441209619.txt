import React, { useEffect, useMemo, useRef, useState } from "react"

type DifferentialRow = {
  diagnosis: string
  score: number
}

type AssistantResult = {
  complaint?: string
  triage?: { level: string; reason?: string }
  differential?: DifferentialRow[]
  nextQuestions?: string[]
  resources?: { recommendedActions?: any[] }
  contradictions?: { diagnosis: string; conflict: string }[]
}

const COMPLAINT_CHIPS = [
  "cough",
  "sore throat",
  "UTI symptoms",
  "rash",
  "ear pain",
  "sinus pressure",
  "fever",
  "chest pain",
  "abdominal pain",
]

export default function RapidTelemedicineConsole() {
  const [caseId] = useState(`visit_${Date.now()}`)
  const [message, setMessage] = useState("")
  const [result, setResult] = useState<AssistantResult | null>(null)
  const [isAnalyzing, setIsAnalyzing] = useState(false)
  const [lastCopied, setLastCopied] = useState<"" | "note" | "discharge">("")
  const [status, setStatus] = useState("Ready")
  const [selectedQuestionIndex, setSelectedQuestionIndex] = useState(0)

  const textAreaRef = useRef<HTMLTextAreaElement | null>(null)
  const debounceRef = useRef<number | null>(null)

  async function analyze() {
    const trimmed = message.trim()
    if (!trimmed) return

    try {
      setIsAnalyzing(true)
      setStatus("Analyzing...")

      const res = await fetch("/api/telemed/assistant", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ caseId, message: trimmed }),
      })

      const data = await res.json()
      setResult(data.result)
      setStatus("Updated")
      setSelectedQuestionIndex(0)
    } catch (err: any) {
      setStatus(err?.message || "Analysis failed")
    } finally {
      setIsAnalyzing(false)
    }
  }

  function scheduleAnalyze() {
    if (debounceRef.current) {
      window.clearTimeout(debounceRef.current)
    }

    debounceRef.current = window.setTimeout(() => {
      analyze()
    }, 450)
  }

  function insertText(snippet: string) {
    setMessage((m) => `${m}${m.trim() ? " " : ""}${snippet}`.trim())
    setStatus(`Added: ${snippet}`)
    textAreaRef.current?.focus()
  }

  function appendSuggestedQuestion(question: string) {
    insertText(question)
  }

  async function copyText(kind: "note" | "discharge", text: string) {
    try {
      await navigator.clipboard.writeText(text)
      setLastCopied(kind)
      setStatus(`Copied ${kind}`)
      window.setTimeout(() => setLastCopied(""), 1200)
    } catch {
      setStatus(`Copy failed: ${kind}`)
    }
  }

  const note = useMemo(() => {
    return `Chief Complaint:
${result?.complaint ?? ""}

History:
${message}

Assessment:
${result?.differential?.[0]?.diagnosis ?? ""}

Plan:
${result?.resources?.recommendedActions?.map((a: any) => a.diagnosis).join("\n") ?? ""}

Disposition:
${result?.triage?.level ?? ""}`
  }, [message, result])

  const discharge = useMemo(() => {
    return `Based on our visit, your symptoms are most consistent with:

${result?.differential?.[0]?.diagnosis ?? ""}

Recommended care:
${result?.resources?.recommendedActions?.map((a: any) => a.diagnosis).join("\n") ?? ""}

Return immediately if:
- worsening shortness of breath
- severe chest pain
- persistent high fever
- worsening symptoms

Follow up as advised.`
  }, [result])

  useEffect(() => {
    if (!message.trim()) return
    scheduleAnalyze()

    return () => {
      if (debounceRef.current) {
        window.clearTimeout(debounceRef.current)
      }
    }
  }, [message])

  useEffect(() => {
    function onKeyDown(e: KeyboardEvent) {
      const isMod = e.ctrlKey || e.metaKey

      if (isMod && e.key === "Enter") {
        e.preventDefault()
        analyze()
        return
      }

      if (isMod && e.key.toLowerCase() === "n") {
        e.preventDefault()
        copyText("note", note)
        return
      }

      if (isMod && e.key.toLowerCase() === "d") {
        e.preventDefault()
        copyText("discharge", discharge)
        return
      }

      if (isMod && e.key.toLowerCase() === "l") {
        e.preventDefault()
        textAreaRef.current?.focus()
        return
      }

      if (e.altKey && e.key === "ArrowDown") {
        e.preventDefault()
        setSelectedQuestionIndex((i) =>
          Math.min(i + 1, Math.max((result?.nextQuestions?.length ?? 1) - 1, 0))
        )
        return
      }

      if (e.altKey && e.key === "ArrowUp") {
        e.preventDefault()
        setSelectedQuestionIndex((i) => Math.max(i - 1, 0))
        return
      }

      if (e.altKey && e.key === "Enter") {
        e.preventDefault()
        const q = result?.nextQuestions?.[selectedQuestionIndex]
        if (q) appendSuggestedQuestion(q)
        return
      }

      if (e.altKey && /^[1-9]$/.test(e.key)) {
        const idx = Number(e.key) - 1
        const q = result?.nextQuestions?.[idx]
        if (q) {
          e.preventDefault()
          appendSuggestedQuestion(q)
        }
      }
    }

    window.addEventListener("keydown", onKeyDown)
    return () => window.removeEventListener("keydown", onKeyDown)
  }, [result, selectedQuestionIndex, note, discharge])

  return (
    <div
      style={{
        display: "grid",
        gridTemplateColumns: "2.2fr 1.2fr 1.3fr",
        gap: 16,
        padding: 16,
        fontFamily: "Arial, sans-serif",
      }}
    >
      {/* LEFT: INPUT + QUESTIONS */}
      <div style={{ border: "1px solid #ddd", borderRadius: 12, padding: 16 }}>
        <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
          <h2 style={{ margin: 0 }}>Rapid Telemedicine Console</h2>
          <div style={{ fontSize: 12, color: "#555" }}>
            {isAnalyzing ? "Analyzing..." : status}
          </div>
        </div>

        <div style={{ marginTop: 10, fontSize: 12, color: "#666" }}>
          Shortcuts: Ctrl/Cmd+Enter analyze • Ctrl/Cmd+N copy note • Ctrl/Cmd+D copy discharge • Alt+1..9 add suggested question
        </div>

        <div style={{ marginTop: 12 }}>
          <textarea
            ref={textAreaRef}
            style={{
              width: "100%",
              height: 160,
              padding: 12,
              borderRadius: 10,
              border: "1px solid #ccc",
              fontSize: 15,
              lineHeight: 1.4,
            }}
            value={message}
            onChange={(e) => setMessage(e.target.value)}
            placeholder="Paste the patient message from Telegram / WhatsApp here"
          />
        </div>

        <div style={{ marginTop: 12, display: "flex", flexWrap: "wrap", gap: 6 }}>
          {COMPLAINT_CHIPS.map((chip) => (
            <button
              key={chip}
              onClick={() => insertText(chip)}
              style={{
                border: "1px solid #ccc",
                borderRadius: 999,
                padding: "6px 10px",
                background: "#fff",
                cursor: "pointer",
              }}
            >
              {chip}
            </button>
          ))}
        </div>

        <div style={{ marginTop: 12, display: "flex", gap: 8 }}>
          <button
            onClick={analyze}
            style={{
              background: "#2c7be5",
              color: "white",
              padding: "10px 14px",
              border: "none",
              borderRadius: 10,
              cursor: "pointer",
            }}
          >
            Analyze Now
          </button>

          <button
            onClick={() => {
              setMessage("")
              setResult(null)
              setStatus("Cleared")
              textAreaRef.current?.focus()
            }}
            style={{
              background: "#f5f5f5",
              color: "#222",
              padding: "10px 14px",
              border: "1px solid #ddd",
              borderRadius: 10,
              cursor: "pointer",
            }}
          >
            Clear
          </button>
        </div>

        <h3 style={{ marginTop: 20 }}>Next Best Questions</h3>

        <div style={{ display: "grid", gap: 8 }}>
          {(result?.nextQuestions ?? []).map((q, i) => {
            const active = i === selectedQuestionIndex
            return (
              <button
                key={q}
                onClick={() => appendSuggestedQuestion(q)}
                style={{
                  textAlign: "left",
                  padding: 10,
                  borderRadius: 10,
                  border: active ? "2px solid #2c7be5" : "1px solid #ddd",
                  background: active ? "#eef5ff" : "#fafafa",
                  cursor: "pointer",
                }}
              >
                <div style={{ fontSize: 12, color: "#666" }}>
                  Alt+{i + 1}
                </div>
                <div>{q}</div>
              </button>
            )
          })}
        </div>
      </div>

      {/* MIDDLE: REASONING */}
      <div style={{ border: "1px solid #ddd", borderRadius: 12, padding: 16 }}>
        <h3 style={{ marginTop: 0 }}>Reasoning</h3>

        <div style={{ marginBottom: 16 }}>
          <div style={{ fontSize: 12, color: "#666" }}>Chief complaint</div>
          <div style={{ fontWeight: 600 }}>{result?.complaint || "—"}</div>
        </div>

        <div style={{ marginBottom: 16 }}>
          <div style={{ fontSize: 12, color: "#666" }}>Triage</div>
          <div style={{ fontWeight: 700, color: result?.triage?.level === "critical" ? "#b00020" : "#222" }}>
            {result?.triage?.level || "—"}
          </div>
          <div style={{ fontSize: 12, color: "#666" }}>

          </div>

        <div>
          <div style={{ fontSize: 12, color: "#666", marginBottom: 6 }}>Differential</div>

server/routes/telegramWebhook.ts
import express from "express"
import { addMessage } from "../integrations/conversationStore"
import { publishEvent } from "../core/events/eventPublisher"
import { runTelemedicineAssistant } from "../assistant/telemedicineAssistantService"
import { sendTelegramMessage } from "../integrations/telegramAdapter"

const router = express.Router()

function formatAssistantReply(result: any) {
  const top = result?.differential?.[0]

  return [
    `Possible diagnosis: ${top?.diagnosis ?? "unknown"}`,
    `Urgency: ${result?.triage?.level ?? "unknown"}`,
    "",
    "Suggested questions:",
    ...(result?.nextQuestions ?? []).map((q: string) => `• ${q}`),
    "",
    "Recommended actions:",
    ...((result?.resources?.recommendedActions ?? []).map((a: any) => `• ${a.diagnosis}`)),
  ].join("\n")
}

router.post("/telegram/webhook", async (req, res) => {
  try {
    const message = req.body.message?.text ?? ""
    const chatId = String(req.body.message?.chat?.id ?? "")
    const caseId = `tg_${chatId}`

    if (!chatId || !message.trim()) {
      res.sendStatus(200)
      return
    }

    addMessage(caseId, "patient", message)

    await publishEvent(caseId, "PATIENT_MESSAGE", { message })

    const result = await runTelemedicineAssistant(caseId)
    const reply = formatAssistantReply(result)

    addMessage(caseId, "assistant", reply)

    if (process.env.TELEGRAM_BOT_TOKEN) {
      await sendTelegramMessage(process.env.TELEGRAM_BOT_TOKEN, chatId, reply)
    }

    res.sendStatus(200)
  } catch (err) {
    console.error("telegram webhook error", err)
    res.sendStatus(500)
  }
})

export default router

import fetch from "node-fetch"

export async function sendWhatsAppMessage(params: {
  endpoint: string
  token?: string
  to: string
  text: string
}) {
  const headers: Record<string, string> = {
    "Content-Type": "application/json",
  }

  if (params.token) {
    headers["Authorization"] = `Bearer ${params.token}`
  }

  await fetch(params.endpoint, {
    method: "POST",
    headers,
    body: JSON.stringify({
      to: params.to,
      text: params.text,

function formatAssistantReply(result: any) {
  const top = result?.differential?.[0]

  return [
    `Possible diagnosis: ${top?.diagnosis ?? "unknown"}`,
    `Urgency: ${result?.triage?.level ?? "unknown"}`,
    "",
    "Suggested questions:",
    ...(result?.nextQuestions ?? []).map((q: string) => `• ${q}`),
    "",
    "Recommended actions:",
    ...((result?.resources?.recommendedActions ?? []).map((a: any) => `• ${a.diagnosis}`)),
  ].join("\n")
}

router.post("/whatsapp/webhook", async (req, res) => {
  try {
    const from = String(req.body.from ?? "")
    const message = String(req.body.text ?? "")
    const caseId = `wa_${from}`

    if (!from || !message.trim()) {
      res.sendStatus(200)
      return
    }

    addMessage(caseId, "patient", message)

    await publishEvent(caseId, "PATIENT_MESSAGE", { message })

    const result = await runTelemedicineAssistant(caseId)
    const reply = formatAssistantReply(result)

    addMessage(caseId, "assistant", reply)

    if (process.env.WHATSAPP_SEND_ENDPOINT) {
      await sendWhatsAppMessage({
        endpoint: process.env.WHATSAPP_SEND_ENDPOINT,
        token: process.env.WHATSAPP_SEND_TOKEN,
        to: from,
        text: reply,
      })
    }

    res.sendStatus(200)
  } catch (err) {
    console.error("whatsapp webhook error", err)
    res.sendStatus(500)
  }
})

export default router

import React, { useEffect, useMemo, useRef, useState } from "react"

type ChatMessage = {
  role: "patient" | "doctor" | "assistant"
  text: string
  timestamp: string
}

type AssistantResult = {
  complaint?: string
  triage?: { level: string; reason?: string }
  differential?: { diagnosis: string; score: number }[]
  nextQuestions?: string[]
  resources?: { recommendedActions?: any[] }
  contradictions?: { diagnosis: string; conflict: string }[]
  pathway?: { labs?: string[]; medications?: string[]; followup?: string[] }
}

export default function TelemedicineSplitPaneConsole() {
  const [caseId, setCaseId] = useState("tg_demo_1")
  const [messages, setMessages] = useState<ChatMessage[]>([])
  const [doctorText, setDoctorText] = useState("")
  const [assistantResult, setAssistantResult] = useState<AssistantResult | null>(null)
  const [loading, setLoading] = useState(false)
  const listRef = useRef<HTMLDivElement | null>(null)

  async function loadConversation() {
    const res = await fetch(`/api/conversations/${encodeURIComponent(caseId)}`)
    const data = await res.json()
    setMessages(data.messages ?? [])
  }

  async function sendDoctorMessage(text?: string) {
    const nextText = (text ?? doctorText).trim()
    if (!nextText) return

    setLoading(true)
    try {
      const res = await fetch(`/api/conversations/${encodeURIComponent(caseId)}/doctor-message`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ text: nextText }),
      })
      const data = await res.json()

      setMessages(data.messages ?? [])
      setAssistantResult(data.result ?? null)
      setDoctorText("")
    } finally {
      setLoading(false)
    }
  }

  function appendSuggestedQuestion(q: string) {
    const text = `${q}`
    void sendDoctorMessage(text)
  }

  useEffect(() => {
    void loadConversation()
  }, [caseId])

  useEffect(() => {
    if (listRef.current) {
      listRef.current.scrollTop = listRef.current.scrollHeight
    }
  }, [messages])

  const chartNote = useMemo(() => {
    return `Chief Complaint:
${assistantResult?.complaint ?? ""}

Conversation Summary:
${messages.map((m) => `[${m.role}] ${m.text}`).join("\n")}

Assessment:
${assistantResult?.differential?.[0]?.diagnosis ?? ""}

Plan:
${assistantResult?.resources?.recommendedActions?.map((a: any) => a.diagnosis).join("\n") ?? ""}

Disposition:
${assistantResult?.triage?.level ?? ""}`
  }, [assistantResult, messages])

  const discharge = useMemo(() => {
    return `Based on our telemedicine visit, the most likely diagnosis is:

${assistantResult?.differential?.[0]?.diagnosis ?? ""}

Recommended care:
${assistantResult?.resources?.recommendedActions?.map((a: any) => a.diagnosis).join("\n") ?? ""}

Please seek urgent care immediately if symptoms worsen, severe chest pain develops, breathing worsens, or fever persists.`
  }, [assistantResult])

  async function copyText(text: string) {
    await navigator.clipboard.writeText(text)
  }

  return (
    <div
      style={{
        display: "grid",
        gridTemplateColumns: "1.2fr 1fr",
        gap: 16,
        padding: 16,
        height: "100vh",
        boxSizing: "border-box",
        fontFamily: "Arial, sans-serif",

      {/* LEFT: CHAT */}
      <div

          border: "1px solid #ddd",
          borderRadius: 12,
          display: "grid",
          gridTemplateRows: "auto 1fr auto",
          minHeight: 0,
        }}
      >
        <div style={{ padding: 12, borderBottom: "1px solid #eee", display: "flex", gap: 8 }}>
          <input
            value={caseId}
            onChange={(e) => setCaseId(e.target.value)}
            style={{ flex: 1, padding: 8, borderRadius: 8, border: "1px solid #ccc" }}
            placeholder="Case ID / tg_xxx / wa_xxx"
          />
          <button onClick={() => void loadConversation()}>Load</button>
        </div>

        <div

            background: "#fafafa",
          }}
        >
          {messages.map((m, idx) => (
            <div
              key={`${m.timestamp}-${idx}`}
              style={{
                marginBottom: 10,
                display: "flex",
                justifyContent:
                  m.role === "patient" ? "flex-start" : m.role === "doctor" ? "flex-end" : "center",
              }}
            >
              <div
                style={{
                  maxWidth: "78%",
                  padding: 10,
                  borderRadius: 10,
                  background:
                    m.role === "patient" ? "#ffffff" : m.role === "doctor" ? "#dff2ff" : "#eef5e9",
                  border: "1px solid #ddd",
                }}
              >
                <div style={{ fontSize: 11, color: "#666", marginBottom: 4 }}>
                  {m.role} • {new Date(m.timestamp).toLocaleTimeString()}
                </div>
                <div style={{ whiteSpace: "pre-wrap" }}>{m.text}</div>
              </div>
            </div>
          ))}
        </div>

        <div style={{ padding: 12, borderTop: "1px solid #eee" }}>
          <textarea
            value={doctorText}
            onChange={(e) => setDoctorText(e.target.value)}

  )
