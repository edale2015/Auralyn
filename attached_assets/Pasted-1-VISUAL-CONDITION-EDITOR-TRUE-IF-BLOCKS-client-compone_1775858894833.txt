import { useState } from "react";

export default function IfBlockEditor({ node, update }: any) {

  const [field, setField] = useState(node.field || "");
  const [value, setValue] = useState(node.value || "");

  function save() {
    update({
      ...node,
      if: {
        field,
        equals: value
      }
    });
  }

  return (
    <div className="bg-yellow-800 p-3 rounded">
      <h4>IF Condition</h4>

      <input
        placeholder="field"
        value={field}
        onChange={e => setField(e.target.value)}
      />

      <input
        placeholder="equals"
        value={value}
        onChange={e => setValue(e.target.value)}
      />

      <button onClick={save}>Save</button>
    </div>
  );
}

  return roles[role]?.includes(action);
}
🔧 Middleware
export function auth(roleRequired: string) {
  return (req: any, res: any, next: any) => {
    const role = req.headers["x-role"];
    if (!can(role, roleRequired)) {
      return res.status(403).send("Forbidden");
    }
    next();
  };
}
