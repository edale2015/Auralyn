1️⃣ Clinical Contradiction Engine

This engine detects impossible or conflicting symptom combinations.

Examples:

male + pregnancy
no fever + high fever
no cough + productive cough

These errors appear constantly in AI triage systems and cause bad reasoning.

Create
server/core/contradictionEngine.ts
type ContradictionRule = {
  a: string
  b: string
  message: string
}

const contradictionRules: ContradictionRule[] = [

  { a: "male", b: "pregnancy", message: "Pregnancy reported in male patient" },

  { a: "no_fever", b: "fever", message: "Fever contradiction" },

  { a: "no_cough", b: "productive_cough", message: "Cough contradiction" },

  { a: "testicular_pain", b: "female", message: "Testicular pain reported in female patient" }

]

export function contradictionEngine(symptoms: string[]) {

  const conflicts = []

  for (const rule of contradictionRules) {

    if (symptoms.includes(rule.a) && symptoms.includes(rule.b)) {

      conflicts.push(rule.message)
    }
  }

  return conflicts
}

2️⃣ Clinical Evidence Aggregator Engine

Right now you have three reasoning engines:

• Bayesian
• Case similarity
• Knowledge graph

They must be merged intelligently.

Create
server/core/evidenceAggregatorEngine.ts
type DxScore = {
  diagnosis: string
  score: number
}

export function evidenceAggregatorEngine(

  bayesian: DxScore[],
  similarity: DxScore[],
  graph: DxScore[]

) {

  const scores: Record<string, number> = {}

  function add(list: DxScore[], weight: number) {

    for (const dx of list) {

      scores[dx.diagnosis] =
        (scores[dx.diagnosis] || 0) + dx.score * weight
    }
  }

  add(bayesian, 0.5)
  add(similarity, 0.3)
  add(graph, 0.2)

  return Object.entries(scores)
    .map(([diagnosis, score]) => ({ diagnosis, score }))
    .sort((a, b) => b.score - a.score)
}
Why these weights?
Bayesian → strongest signal
Similarity → real cases
Graph → background knowledge

Later your RL module can learn these weights automatically.

1. Graph Data Model

Each medical concept becomes a node, and relationships become edges.

Node types typically include:

symptom

diagnosis

complaint

red_flag

question

medication

disposition

Edge types describe relationships:

symptom → diagnosis (supports)

diagnosis → red_flag (danger)

complaint → symptom (typical)

diagnosis → question (confirm)

diagnosis → medication (treat)

2. Graph Engine

Create:

server/core/clinicalGraphEngine.ts
export type GraphNode = {
  id: string
  type: string
}

export type GraphEdge = {
  from: string
  to: string
  relation: string
  weight?: number
}

export class ClinicalGraph {

  nodes: Map<string, GraphNode> = new Map()
  edges: GraphEdge[] = []

  addNode(id: string, type: string) {
    if (!this.nodes.has(id)) {
      this.nodes.set(id, { id, type })
    }
  }

  addEdge(from: string, to: string, relation: string, weight = 1) {

    this.addNode(from, "unknown")
    this.addNode(to, "unknown")

    this.edges.push({
      from,
      to,
      relation,
      weight
    })
  }

  getNeighbors(nodeId: string, relation?: string) {

    return this.edges
      .filter(e => e.from === nodeId && (!relation || e.relation === relation))
      .map(e => e.to)
  }

  scoreDiagnoses(symptoms: string[]) {

    const scores: Record<string, number> = {}

    for (const s of symptoms) {

      const edges = this.edges.filter(
        e => e.from === s && e.relation === "supports"
      )

      for (const edge of edges) {

        scores[edge.to] =
          (scores[edge.to] || 0) + (edge.weight || 1)
      }
    }

    return Object.entries(scores)
      .map(([dx, score]) => ({ diagnosis: dx, score }))
      .sort((a, b) => b.score - a.score)
  }
}
3. Graph Builder (from your Google-Sheet CSVs)

Create:

scripts/buildClinicalReasoningGraph.ts
import fs from "fs"
import csv from "csv-parser"
import { ClinicalGraph } from "../server/core/clinicalGraphEngine"

const graph = new ClinicalGraph()

function loadClusterRules() {

  return new Promise<void>((resolve) => {

    fs.createReadStream("server/data/csv/CLUSTER_SCORING_RULES.csv")

      .pipe(csv())

      .on("data", (row) => {

        const symptom = row.symptom
        const diagnosis = row.diagnosis
        const weight = Number(row.weight || 1)

        graph.addEdge(symptom, diagnosis, "supports", weight)

      })

      .on("end", resolve)
  })
}

function loadRedFlags() {

  return new Promise<void>((resolve) => {

    fs.createReadStream("server/data/csv/RED_FLAGS.csv")

      .pipe(csv())

      .on("data", (row) => {

        const diagnosis = row.diagnosis
        const flag = row.red_flag

        graph.addEdge(diagnosis, flag, "danger")

      })

      .on("end", resolve)
  })
}

function loadComplaintRegistry() {

  return new Promise<void>((resolve) => {

    fs.createReadStream("server/data/csv/COMPLAINT_REGISTRY.csv")

      .pipe(csv())

      .on("data", (row) => {

        const complaint = row.complaint
        const symptom = row.symptom

        graph.addEdge(complaint, symptom, "typical")

      })

      .on("end", resolve)
  })
}

async function buildGraph() {

  await loadClusterRules()
  await loadRedFlags()
  await loadComplaintRegistry()

  fs.writeFileSync(
    "server/data/clinical_reasoning_graph.json",
    JSON.stringify(graph, null, 2)
  )

  console.log("Clinical reasoning graph created")
}
