# Auralyn Production Upgrade Patch

Below is a paste-ready implementation bundle to address the three requested areas:

1. **Full KB migration**
2. **Complete golden-case expansion**
3. **Productionized BullMQ rollout for every queue**

This patch assumes a TypeScript/Express/Postgres/Drizzle codebase with Redis available in production and in-memory fallback only for local development. Adjust import paths if your repo differs.

### `server/kb/kbRepository.ts`

```ts
import { and, desc, eq } from "drizzle-orm";
import { db } from "../db";
import { kbEntities, kbEntityVersions, kbSources } from "../../shared/schema";
import type { UpsertKBEntityInput } from "./kbTypes";

async function resolveSource(sourceKey?: string, sourceName?: string, sourceType?: string) {
  if (!sourceKey) return null;

  const existing = await db.query.kbSources.findFirst({
    where: eq(kbSources.sourceKey, sourceKey),
  });
  if (existing) return existing;

  const [created] = await db.insert(kbSources).values({
    sourceKey,
    name: sourceName || sourceKey,
    sourceType: sourceType || "migration",
    isAuthoritative: sourceType === "db",
  }).returning();

  return created;
}

export async function upsertKBEntity(input: UpsertKBEntityInput) {
  const source = await resolveSource(input.sourceKey, input.sourceName, input.sourceType);
  const existing = await db.query.kbEntities.findFirst({
    where: and(
      eq(kbEntities.entityType, input.entityType),
      eq(kbEntities.entityKey, input.entityKey),
    ),
  });

  if (!existing) {
    const [created] = await db.insert(kbEntities).values({
      entityType: input.entityType,
      entityKey: input.entityKey,
      title: input.title,
      status: input.status || "active",
      version: 1,
      tags: input.tags || [],
      metadata: input.metadata || {},
      currentContent: input.content,
      sourceId: source?.id,
      createdBy: input.changedBy || "system:migration",
      updatedBy: input.changedBy || "system:migration",
    }).returning();

    await db.insert(kbEntityVersions).values({
      entityId: created.id,
      version: 1,
      title: input.title,
      content: input.content,
      changeSummary: input.changeSummary || "Initial migration",
      changedBy: input.changedBy || "system:migration",
    });

    return created;
  }

  const nextVersion = existing.version + 1;
  const [updated] = await db.update(kbEntities)
    .set({
      title: input.title,
      status: input.status || existing.status,
      version: nextVersion,
      tags: input.tags || existing.tags,
      metadata: input.metadata || existing.metadata,
      currentContent: input.content,
      sourceId: source?.id || existing.sourceId,
      updatedBy: input.changedBy || "system:migration",
      updatedAt: new Date(),
    })
    .where(eq(kbEntities.id, existing.id))
    .returning();

  await db.insert(kbEntityVersions).values({
    entityId: existing.id,
    version: nextVersion,
    title: input.title,
    content: input.content,
    changeSummary: input.changeSummary || "Updated during migration",
    changedBy: input.changedBy || "system:migration",
  });

  return updated;
}

export async function getActiveKBEntity(entityType: string, entityKey: string) {
  return db.query.kbEntities.findFirst({
    where: and(
      eq(kbEntities.entityType, entityType as any),
      eq(kbEntities.entityKey, entityKey),
      eq(kbEntities.status, "active"),
    ),
  });
}

export async function listKBEntities(entityType?: string) {
  return db.query.kbEntities.findMany({
    where: entityType ? eq(kbEntities.entityType, entityType as any) : undefined,
    orderBy: [desc(kbEntities.updatedAt)],
  });
}

export async function getKBEntityHistory(entityId: string) {
  return db.query.kbEntityVersions.findMany({
    where: eq(kbEntityVersions.entityId, entityId),
    orderBy: [desc(kbEntityVersions.version)],
  });
}
```

```ts
interface LegacyLoader<T> {
  file: string;
  entityType: Parameters<typeof upsertKBEntity>[0]["entityType"];
  toRecord: (raw: T) => {
    entityKey: string;
    title: string;
    content: Record<string, any>;

    metadata?: Record<string, any>;

}

async function readJsonArray<T>(filePath: string): Promise<T[]> {
  const content = await fs.readFile(filePath, "utf8");
  const parsed = JSON.parse(content);
  if (!Array.isArray(parsed)) throw new Error(`Expected array in ${filePath}`);
  return parsed as T[];
}

export async function runFullKBMigration(projectRoot = process.cwd()) {
  const results: Array<{ entityType: string; file: string; count: number }> = [];

  for (const loader of loaders) {
    const absolute = path.join(projectRoot, loader.file);
    try {
      const rows = await readJsonArray<any>(absolute);
      let migrated = 0;
      for (const row of rows) {
        const mapped = loader.toRecord(row);
        await upsertKBEntity({
          entityType: loader.entityType,
          entityKey: mapped.entityKey,
          title: mapped.title,
          content: mapped.content,
          tags: mapped.tags,
          metadata: mapped.metadata,
          sourceKey: `legacy:${loader.file}`,
          sourceName: loader.file,
          sourceType: "migration",
          changedBy: "system:full-kb-migration",
          changeSummary: `Migrated from ${loader.file}`,
          status: "active",
        });
        migrated += 1;
      }
      results.push({ entityType: loader.entityType, file: loader.file, count: migrated });
    } catch (error) {
      if ((error as NodeJS.ErrnoException).code === "ENOENT") continue;
      throw error;
    }
  }

  return {
    ok: true,
    results,
    migratedAt: new Date().toISOString(),
  };
}
```

```ts
import { getActiveKBEntity } from "./kbRepository";

function ensureContent<T = any>(row: any, type: string, key: string): T {
  if (!row?.currentContent) {
    throw new Error(`Missing KB content for ${type}:${key}`);
  }
  return row.currentContent as T;
}

export async function getComplaintDefinition(complaint: string) {
  const row = await getActiveKBEntity("complaint", complaint);
  return ensureContent(row, "complaint", complaint);
}

export async function getCoreQuestionsForComplaint(complaint: string) {
  // For speed, store complaint-indexed arrays in complaint.currentContent.coreQuestions during migration,
  // or add a query by tags in a second pass.
  const complaintRow = await getActiveKBEntity("complaint", complaint);
  const content = ensureContent<any>(complaintRow, "complaint", complaint);
  return content.coreQuestions || [];
}

export async function getRedFlagsForComplaint(complaint: string) {
  const complaintRow = await getActiveKBEntity("complaint", complaint);
  const content = ensureContent<any>(complaintRow, "complaint", complaint);
  return content.redFlags || [];
}

export async function getDispositionRulesForComplaint(complaint: string) {
  const complaintRow = await getActiveKBEntity("complaint", complaint);
  const content = ensureContent<any>(complaintRow, "complaint", complaint);
  return content.dispositionRules || [];
}
```

```ts
      caseKey: input.caseKey,

      ageYears: input.ageYears,
      sex: input.sex,

      tags: input.tags || [],
      presentation: input.presentation,
      expected: input.expected,
      sourceId: source.id,
      status: "approved",
      authoredBy: "system:golden-expansion",
      approvedBy: "system:golden-expansion",
    }).returning();
    return created;
  }

  const [updated] = await db.update(goldenCases)
    .set({
      title: input.title,
      complaint: input.complaint,
      ageYears: input.ageYears,
      sex: input.sex,
      difficulty: input.difficulty || existing.difficulty,
      safetyClass: input.safetyClass || existing.safetyClass,

      presentation: input.presentation,
      expected: input.expected,
      updatedAt: new Date(),
    })
    .where(eq(goldenCases.id, existing.id))
    .returning();

  return updated;
}

export async function listApprovedGoldenCases(complaint?: string) {

export async function recordGoldenRun(goldenCaseId: string, payload: {

    engineVersion: payload.engineVersion,

    score: payload.score,
    passed: payload.passed,
  }).returning();
  return row;
}

export async function incrementCoverage(complaint: string, riskBand: string, ageBand: string) {
  const existing = await db.query.goldenCaseCoverage.findFirst({
    where: and(
      eq(goldenCaseCoverage.complaint, complaint),
      eq(goldenCaseCoverage.riskBand, riskBand),
      eq(goldenCaseCoverage.ageBand, ageBand),
    ),
  });

  if (!existing) {
    await db.insert(goldenCaseCoverage).values({
      complaint,
      riskBand,
      ageBand,
      count: 1,
      targetCount: 10,
      metadata: {},
    });
    return;
  }

  await db.update(goldenCaseCoverage)
    .set({ count: existing.count + 1, updatedAt: new Date() })
    .where(eq(goldenCaseCoverage.id, existing.id));
}

export async function getCoverageGaps() {
  return db.execute(sql`
    select complaint, risk_band, age_band, count, target_count,
           greatest(target_count - count, 0) as missing
    from golden_case_coverage
    where count < target_count
    order by complaint, risk_band, age_band
  `);
}
```

```ts
    riskBands: [
      { key: "moderate", painScale: 5, redFlags: [] },
      { key: "high", painScale: 8, redFlags: ["diaphoresis", "sob", "exertional"] },
      { key: "critical", painScale: 9, redFlags: ["hypotension", "syncope", "crushing_pain"] },
    ],
    sexes: ["female", "male"],
  },
];

function buildGoldenCase(template: ComplaintTemplate, ageBand: { key: string; ageYears: number }, riskBand: any, sex: string, variant: number) {
  const caseKey = `${template.complaint}:${ageBand.key}:${riskBand.key}:${sex}:v${variant}`;
  return {
    caseKey,
    title: `${template.complaint} ${ageBand.key} ${riskBand.key} ${sex} variant ${variant}`,
    complaint: template.complaint,
    ageYears: ageBand.ageYears,
    sex,
    difficulty: riskBand.key === "critical" ? 5 : riskBand.key === "high" ? 4 : riskBand.key === "moderate" ? 3 : 2,
    safetyClass: riskBand.key === "critical" ? "critical" : riskBand.key === "high" ? "high" : "standard",
    tags: [template.complaint, ageBand.key, riskBand.key, sex],
    presentation: {
      complaint: template.complaint,
      age: ageBand.ageYears,
      sex,
      variant,
      symptoms: {
        fever: riskBand.fever || false,
        sob: riskBand.sob || false,
        painScale: riskBand.painScale || 0,
        redFlags: riskBand.redFlags || [],
      },
      modifiers: {
        onsetDays: variant + 1,
        pmh: variant % 2 === 0 ? ["asthma"] : [],
        meds: [],
        allergies: [],
      },
    },
    expected: {
      triage: riskBand.key === "critical" ? "emergent" : riskBand.key === "high" ? "urgent" : "routine",
      disposition: riskBand.key === "critical" ? "ER_NOW" : riskBand.key === "high" ? "SAME_DAY_EVAL" : "HOME_OR_OUTPATIENT",
      likelyDiagnoses:
        template.complaint === "cough"
          ? riskBand.key === "high" ? ["pneumonia", "pulmonary_embolism"] : ["viral_uri", "bronchitis"]
          : template.complaint === "sore_throat"
          ? riskBand.key === "high" ? ["peritonsillar_abscess", "epiglottitis"] : ["viral_pharyngitis", "strep_pharyngitis"]
          : ["acs", "musculoskeletal_chest_pain"],
      mustAsk: ["duration", "severity", "red_flags", "pmh", "meds"],
      mustNotMiss: riskBand.redFlags || [],
      requiredWorkup:
        template.complaint === "chest_pain" && riskBand.key !== "moderate" ? ["ekg", "troponin"] : [],
      requiredRedFlags: riskBand.redFlags || [],
    },
  };
}

export async function expandGoldenCases() {
  const created: string[] = [];

  for (const template of templates) {
    for (const ageBand of template.ageBands) {
      for (const riskBand of template.riskBands) {
        for (const sex of template.sexes || ["unknown"]) {
          for (let variant = 1; variant <= 5; variant++) {
            const golden = buildGoldenCase(template, ageBand, riskBand, sex, variant);
            await upsertGoldenCase(golden);
            await incrementCoverage(template.complaint, riskBand.key, ageBand.key);
            created.push(golden.caseKey);
          }
        }
      }
    }
  }

  return {
    ok: true,
    createdCount: created.length,
    sample: created.slice(0, 20),
  };
}
```

### `server/golden/goldenCaseRunner.ts`

```ts
import { listApprovedGoldenCases, recordGoldenRun } from "./goldenCaseRepository";
import { runSystem } from "../engine/runSystem";

function scoreGoldenCase(expected: any, actual: any) {
  let score = 0;
  if (expected.disposition && actual?.disposition === expected.disposition) score += 0.4;
  if (expected.triage && actual?.triage === expected.triage) score += 0.2;

  const actualDx = new Set(actual?.diagnoses || []);
  const expectedDx = new Set(expected?.likelyDiagnoses || []);
  const dxHits = [...expectedDx].filter((d) => actualDx.has(d)).length;
  score += Math.min(0.2, expectedDx.size ? (dxHits / expectedDx.size) * 0.2 : 0);

  const asked = new Set(actual?.questionsAsked || []);
  const mustAskHits = (expected.mustAsk || []).filter((q: string) => asked.has(q)).length;
  score += Math.min(0.1, expected.mustAsk?.length ? (mustAskHits / expected.mustAsk.length) * 0.1 : 0);

  const workup = new Set(actual?.workup || []);
  const workupHits = (expected.requiredWorkup || []).filter((w: string) => workup.has(w)).length;
  score += Math.min(0.1, expected.requiredWorkup?.length ? (workupHits / expected.requiredWorkup.length) * 0.1 : 0);

  return Number(score.toFixed(4));
}

export async function runGoldenCases(complaint?: string) {
  const cases = await listApprovedGoldenCases(complaint);
  const results = [];

  for (const gc of cases) {
    const actual = await runSystem(gc.presentation);
    const score = scoreGoldenCase(gc.expected, actual);
    const passed = score >= 0.8;

    await recordGoldenRun(gc.id, {
      systemVersion: process.env.APP_VERSION || "dev",
      engineVersion: process.env.ENGINE_VERSION || "dev",
      result: actual,
      score,
      passed,
    });

    results.push({
      id: gc.id,
      caseKey: gc.caseKey,
      complaint: gc.complaint,
      score,
      passed,
    });
  }

  return {
    total: results.length,
    passed: results.filter((r) => r.passed).length,
    failed: results.filter((r) => !r.passed).length,
    results,
  };
}
```

```ts
import { db } from "../../db";
import { queueJobs } from "../../../shared/schema";
import { and, eq } from "drizzle-orm";

export async function trackQueuedJob(queueName: string, jobId: string, jobName: string, payload: Record<string, any>) {
  const existing = await db.query.queueJobs.findFirst({
    where: and(eq(queueJobs.queueName, queueName), eq(queueJobs.jobId, jobId)),
  });

  if (existing) return existing;

  const [created] = await db.insert(queueJobs).values({
    queueName,
    jobId,
    jobName,
    status: "queued",
    payload,
  }).returning();

  return created;
}

export async function trackJobStatus(queueName: string, jobId: string, patch: {
  status: "queued" | "active" | "completed" | "failed" | "delayed";
  attemptsMade?: number;
  result?: Record<string, any>;
  error?: string;
}) {
  await db.update(queueJobs)
    .set({
      status: patch.status,
      attemptsMade: patch.attemptsMade ?? 0,
      result: patch.result,
      error: patch.error,
      updatedAt: new Date(),
    })
    .where(and(eq(queueJobs.queueName, queueName), eq(queueJobs.jobId, jobId)));
}
```

### `server/queues/bullmq/baseWorker.ts`

```ts
import { Job, Worker, QueueEvents } from "bullmq";
import { getBullConnection } from "./connection";
import { trackJobStatus } from "./jobTracker";
import type { QueueName } from "./queueNames";

export function createTrackedWorker<T = any>(
  queueName: QueueName,
  handler: (job: Job<T>) => Promise<any>,
  concurrency = 5,
) {
  const connection = getBullConnection();

  const worker = new Worker<T>(
    queueName,
    async (job) => {
      await trackJobStatus(queueName, String(job.id), {
        status: "active",
        attemptsMade: job.attemptsMade,
      });

      try {
        const result = await handler(job);
        await trackJobStatus(queueName, String(job.id), {
          status: "completed",
          attemptsMade: job.attemptsMade,
          result,
        });
        return result;
      } catch (error) {
        await trackJobStatus(queueName, String(job.id), {
          status: "failed",
          attemptsMade: job.attemptsMade,
          error: error instanceof Error ? error.stack || error.message : String(error),
        });
        throw error;
      }
    },
    {
      connection,
      concurrency,
      limiter: { max: 200, duration: 1000 },
    },
  );

  const events = new QueueEvents(queueName, { connection });
  events.on("delayed", async ({ jobId }) => {
    await trackJobStatus(queueName, String(jobId), { status: "delayed" });
  });

  worker.on("error", (err) => {
    console.error(`[BullMQ:${queueName}] worker error`, err);
  });

  return { worker, events };
}
```

## 12) Acceptance checklist

### Full KB migration is complete when:

* Runtime reads complaint packs and rules from `kb_entities.current_content`
* Legacy JSON or Sheets are no longer in the hot path
* Every KB entity type has rows in `kb_entities`
* Version history exists in `kb_entity_versions`
* `/api/kb/migrate/full` runs cleanly

### Golden-case expansion is complete when:

* Coverage matrix exists for every target complaint/risk/age cell
* Expansion endpoint seeds 5+ variants per cell
* Runner records outcomes to `golden_case_runs`
* Coverage gaps are visible at `/api/golden/coverage-gaps`

### BullMQ rollout is complete when:

* Every asynchronous side effect is published to a named BullMQ queue
* Every queue has a dedicated worker
* Queue job states are persisted in `queue_jobs`
* Health endpoint verifies Redis/BullMQ
