// server/monitoring/systemMonitor.ts

import { db } from "../db";

export async function logEngineStatus(engine, status, latency, error = null) {
  await db.insert("engine_logs").values({
    engine,
    status,
    latency,
    error
  });
}

export async function getSystemHealth() {
  const logs = await db.select().from("engine_logs").limit(50);

  const summary = {};

  for (const log of logs) {
    if (!summary[log.engine]) {
      summary[log.engine] = {
        healthy: 0,
        error: 0
      };
    }

    if (log.status === "healthy") summary[log.engine].healthy++;
    else summary[log.engine].error++;
  }

  return summary;
}

export function SystemHealthPanel() {
  const [health, setHealth] = useState({});

  useEffect(() => {
    const interval = setInterval(() => {
      fetch("/api/monitoring/health")
        .then(r => r.json())
        .then(setHealth);
    }, 2000);

    return () => clearInterval(interval);
  }, []);

  return (
    <div>
      {Object.entries(health).map(([engine, stats]) => (
        <div key={engine}>
          <strong>{engine}</strong>
          <span style={{
            color: stats.error > 0 ? "red" : "green"
          }}>
            ●
          </span>
        </div>
      ))}
    </div>
  );
}

export async function runBatchSimulation(inputs) {
  const results = [];

  for (const input of inputs) {
    const r = await runSimulation(input);
    results.push(r);
  }

  return results;
}

// server/monitoring/predictiveEngine.ts

import { db } from "../db";

export async function predictFailures() {
  const logs = await db.select().from("engine_logs").limit(100);

  const errorRate = logs.filter(l => l.status === "error").length / logs.length;

  if (errorRate > 0.1) {
    console.log("🚨 Predicted system instability");
  }

  return { errorRate };
}
