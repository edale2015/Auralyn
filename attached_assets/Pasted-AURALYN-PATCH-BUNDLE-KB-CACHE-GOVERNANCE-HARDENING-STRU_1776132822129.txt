🔒 Auth Fix (CRITICAL)

📁 server/kb/middleware/kbAuthMiddleware.ts

import type { Request, Response, NextFunction } from "express";

function resolveRole(req: Request): string | undefined {
  return (req as any).user?.role || (req as any).actor?.role;
}

function deny(res: Response, code: string, message: string) {
  return res.status(403).json({ ok: false, code, message });
}

export function requireKbAdmin(req: Request, res: Response, next: NextFunction) {
  const role = resolveRole(req);

  if (!role || !["kb_admin", "system_admin", "super_admin"].includes(role)) {
    return deny(res, "KB_ADMIN_REQUIRED", "KB admin required");
  }

  next();
}

export function requireKbWrite(req: Request, res: Response, next: NextFunction) {
  const role = resolveRole(req);

  if (!role || !["kb_admin", "system_admin", "super_admin", "physician", "clinician"].includes(role)) {
    return deny(res, "KB_WRITE_REQUIRED", "KB write role required");
  }

  next();
}

🧠 KB Version Export (FIXES DRIFT MONITOR)

📁 server/kb/kbRuntime.ts

import crypto from "crypto";

let _kbVersion = "init";

function refreshVersion() {
  const hash = crypto.createHash("sha256");
  hash.update(JSON.stringify({
    priors: _priors?.data ?? [],
    redFlags: _redFlags?.data ?? [],
    treatments: _treatments?.data ?? []
  }));
  _kbVersion = hash.digest("hex");
}

export function getKbVersion() {
  return _kbVersion;
}
