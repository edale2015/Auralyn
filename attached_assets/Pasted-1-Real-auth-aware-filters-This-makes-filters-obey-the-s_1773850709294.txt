1) Real auth-aware filters

This makes filters obey the signed-in user’s clinic and role instead of trusting the browser.

/server/middleware/authContext.ts
import { Request, Response, NextFunction } from "express";
import jwt from "jsonwebtoken";

const JWT_SECRET = process.env.JWT_SECRET || "dev-secret";

export type AuthUser = {
  id: string;
  email: string;
  role: "admin" | "physician" | "reviewer" | "executive";
  clinicId: string;
};

export function authContext(req: Request, res: Response, next: NextFunction) {
  const header = req.headers.authorization;

  if (!header?.startsWith("Bearer ")) {
    return res.status(401).json({ error: "Missing bearer token" });
  }

  const token = header.replace("Bearer ", "");

  try {
    const decoded = jwt.verify(token, JWT_SECRET) as AuthUser;
    res.locals.user = decoded;
    next();
  } catch {
    return res.status(401).json({ error: "Invalid token" });
  }
}
/server/services/authAwareFilters.ts
export type DashboardFilterInput = {
  clinicId?: string;
  startDate?: string;
  endDate?: string;
  complaint?: string;
  physicianId?: string;
};

export type DashboardAuthUser = {
  id: string;
  role: "admin" | "physician" | "reviewer" | "executive";
  clinicId: string;
};

export function resolveAuthAwareFilters(
  requested: DashboardFilterInput,
  user: DashboardAuthUser
): DashboardFilterInput {
  const resolved: DashboardFilterInput = {
    ...requested
  };

  if (user.role !== "executive") {
    resolved.clinicId = user.clinicId;
  } else if (!resolved.clinicId) {
    resolved.clinicId = user.clinicId;
  }

  if (user.role === "physician") {
    resolved.physicianId = user.id;
  }

  return resolved;
}

export async function saveDashboardView(payload: {
  userId: string;
  clinicId: string;
  name: string;
  viewType: string;
  filters: unknown;
}) {
  const [row] = await db.insert(dashboardViews).values(payload).returning();
  return row;
}

export async function listDashboardViews(userId: string, clinicId: string) {
  return db.query.dashboardViews.findMany({
    where: (t, { and, eq }) => and(eq(t.userId, userId), eq(t.clinicId, clinicId)),
    orderBy: (t, { desc }) => [desc(t.createdAt)]
  });
}

export async function deleteDashboardView(id: number, userId: string) {
  return db.delete(dashboardViews).where((t, { and, eq }) => and(eq(t.id, id), eq(t.userId, userId)));
}

3) Role-based board packet export

This prevents the wrong people from exporting sensitive board packets.

/server/middleware/roleGate.ts
import { Request, Response, NextFunction } from "express";

export function roleGate(roles: string[]) {
  return (_req: Request, res: Response, next: NextFunction) => {
    const user = res.locals.user;
    if (!user || !roles.includes(user.role)) {
      return res.status(403).json({ error: "Forbidden" });
    }
    next();
  };
}

import { db } from "../db";
import { alerts } from "../db/alertsSchema";

export async function createDbAlert(payload: {
  type: string;
  entityId: string;
  severity: string;
  message: string;
}) {
  const [row] = await db.insert(alerts).values(payload).returning();
  return row;
}

export async function listDbAlerts(includeAcknowledged = true) {
  if (includeAcknowledged) {
    return db.query.alerts.findMany({
      orderBy: (t, { desc }) => [desc(t.createdAt)]
    });
  }

  return db.query.alerts.findMany({
    where: (t, { eq }) => eq(t.acknowledged, false),
    orderBy: (t, { desc }) => [desc(t.createdAt)]
  });
}

export async function acknowledgeDbAlert(id: number, userId: string) {
  const [row] = await db
    .update(alerts)
    .set({
      acknowledged: true,
      acknowledgedBy: userId,
      acknowledgedAt: new Date()
    })
    .where((t, { eq }) => eq(t.id, id))
    .returning();

  return row;
}

7) Benchmarks by complaint and clinic type

This lets you compare apples to apples instead of apples to helicopters.

/server/services/benchmarks.ts
export type BenchmarkRow = {
  clinicType: string;
  complaint: string;
  accuracy: number;
  overrideRate: number;
  escalationRate: number;
  avgSatisfaction: number;
};

export function buildBenchmarks(rows: BenchmarkRow[]) {
  return rows.map(row => {
    let band: "top" | "middle" | "needs_attention" = "middle";

    if (row.accuracy >= 0.92 && row.overrideRate <= 0.06 && row.escalationRate <= 0.08) {
      band = "top";
    } else if (row.accuracy < 0.8 || row.overrideRate > 0.15 || row.escalationRate > 0.18) {
      band = "needs_attention";
    }

    return {
      ...row,
      accuracyPct: Number((row.accuracy * 100).toFixed(1)),
      overrideRatePct: Number((row.overrideRate * 100).toFixed(1)),
      escalationRatePct: Number((row.escalationRate * 100).toFixed(1)),
      band
    };
  });
}
