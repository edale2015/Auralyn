Trace store
# learning/trace_store.py
import json, os

class TraceStore:

    DIR = "traces"

    def save(self, trace):

        os.makedirs(self.DIR, exist_ok=True)

        path = f"{self.DIR}/{trace['case_id']}.json"

        with open(path, "w") as f:
            json.dump(trace, f, indent=2)

    def load_all(self):

        traces = []

        for f in os.listdir(self.DIR):
            with open(f"{self.DIR}/{f}") as file:
                traces.append(json.load(file))

        return traces

Evaluation agent
# agents/evaluation_agent.py
class EvaluationAgent:

    def __init__(self, gold_store):
        self.gold = gold_store

    def evaluate(self, trace):

        gold = self.gold.get(trace["case_id"])

        if not gold:
            return {"passed": True}

        dispo_match = trace["disposition"] == gold["expected_disposition"]

        dx_match = gold["expected_dx"] in trace["differential"]

        return {
            "passed": dispo_match and dx_match,
            "disposition_match": dispo_match,
            "diagnosis_match": dx_match
        }

Failure classifier
# learning/failure_classifier.py
class FailureClassifier:

    def classify(self, trace, evaluation):

        if not evaluation["disposition_match"]:
            return "undertriage"

        if not evaluation["diagnosis_match"]:
            return "differential_missing"

        return "unknown"

            return {
                "type": "add_question",
                "target": "CORE_QUESTIONS"
            }

        if failure == "undertriage":
            return {
                "type": "strengthen_disp_rule",
                "target": "DISP_RULES"
            }

        return None
Level 5 — Regression Safety Testing

Goal: Prevent dangerous changes.

Regression runner
# learning/regression_runner.py
class RegressionRunner:

    def __init__(self, controller):
        self.controller = controller

    def test(self, proposal):

        cases = load_golden_cases()

        failures = 0

        for case in cases:

            result = self.controller.simulate(case, proposal)

            if not result["safe"]:
                failures += 1

        return failures == 0

Level 7 — Reinforcement Learning

Goal: Learn which decision paths work best.

# optimization/reinforcement_engine.py
class ReinforcementEngine:

    def __init__(self):
        self.q = {}

    def update(self, state, action, reward):

        key = (state, action)

        old = self.q.get(key, 0)

        self.q[key] = old + 0.1*(reward-old)

    def best(self, state, actions):

        values = [self.q.get((state,a),0) for a in actions]

        return actions[values.index(max(values))]
Level 8 — Clinical Knowledge Graph

Goal: Learn symptom → diagnosis relationships.

# optimization/reasoning_graph.py
class ReasoningGraph:

    def __init__(self):
        self.graph = {}

    def add(self, symptom, diagnosis):

        if symptom not in self.graph:
            self.graph[symptom] = {}

        self.graph[symptom][diagnosis] = \
            self.graph[symptom].get(diagnosis,0)+1

    def rank(self, symptoms):

        scores = {}

        for s in symptoms:
            if s not in self.graph:
                continue

            for dx,w in self.graph[s].items():
                scores[dx] = scores.get(dx,0)+w

        return sorted(scores,key=scores.get,reverse=True)
Level 9 — Predictive Risk Modeling

Goal: Predict dangerous outcomes early.

# prediction/risk_model.py
class RiskModel:

    def __init__(self):
        self.weights={}

    def train(self,features,outcome):

        for f in features:
            self.weights[f]=self.weights.get(f,0)+outcome

    def score(self,features):

        return sum(self.weights.get(f,0) for f in features)
Level 10 — Autonomous Improvement Orchestrator

Goal: Run the entire learning loop automatically.

# orchestration/improvement_scheduler.py
class ImprovementScheduler:

    def __init__(self, controller):
        self.controller = controller

    def run_cycle(self):

        traces = load_recent_traces()

        for trace in traces:

            evaluation = self.controller.evaluate(trace)

            if not evaluation["passed"]:

                failure = self.controller.classify(trace,evaluation)

                proposal = self.controller.propose(failure)

                if self.controller.regression_test(proposal):

                    self.controller.promote(proposal)
