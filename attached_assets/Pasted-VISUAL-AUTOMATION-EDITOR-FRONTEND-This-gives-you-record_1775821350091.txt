import { useState } from "react";

export default function AutomationStudio() {
  const [steps, setSteps] = useState<any[]>([]);

  function addStep(type: string) {
    setSteps([...steps, { type, selector: "", value: "" }]);
  }

  function updateStep(i: number, key: string, value: any) {
    const copy = [...steps];
    copy[i][key] = value;
    setSteps(copy);
  }

  async function saveTemplate() {
    await fetch("/api/templates/save", {
      method: "POST",
      body: JSON.stringify({ steps })
    });
    alert("Saved");
  }

  return (
    <div className="p-6 text-white bg-black">
      <h1 className="text-xl font-bold">⚙️ Automation Studio</h1>

      <button onClick={() => addStep("click")}>+ Click</button>
      <button onClick={() => addStep("fill")}>+ Fill</button>

      {steps.map((s, i) => (
        <div key={i} className="mt-2">
          <input
            placeholder="selector"
            onChange={e => updateStep(i, "selector", e.target.value)}
          />
          {s.type === "fill" && (
            <input
              placeholder="value"
              onChange={e => updateStep(i, "value", e.target.value)}
            />
          )}
        </div>
      ))}

      <button onClick={saveTemplate}>💾 Save Template</button>
    </div>
  );
}

import OpenAI from "openai";

const openai = new OpenAI();

export async function generateTemplateFromPrompt(prompt: string) {

  const res = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      {
        role: "system",
        content: "Generate Playwright automation steps in JSON"
      },
      { role: "user", content: prompt }
    ]
  });

  return JSON.parse(res.choices[0].message.content!);
}
B. Auto-Repair Agent
export async function repairTemplate(template: any, failure: any) {

  // Example: selector drift
  if (failure.selector) {
    template.steps = template.steps.map((s: any) => {
      if (s.selector === failure.selector) {
        return {
          ...s,
          selector: `[placeholder*="${failure.selector}" i]`
        };
      }
      return s;
    });
  }

  return template;
}
