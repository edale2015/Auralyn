// server/integrations/ecwAuth.ts

import axios from "axios";

let cachedToken: any = null;

export async function getEcwAccessToken() {
  if (cachedToken && cachedToken.expires_at > Date.now()) {
    return cachedToken.access_token;
  }

  const res = await axios.post(
    process.env.ECW_TOKEN_URL!,
    new URLSearchParams({
      grant_type: "client_credentials",
      client_id: process.env.ECW_CLIENT_ID!,
      client_secret: process.env.ECW_CLIENT_SECRET!,
      scope: "patient/*.write patient/*.read"
    })
  );

  cachedToken = {
    access_token: res.data.access_token,
    expires_at: Date.now() + res.data.expires_in * 1000
  };

  return cachedToken.access_token;
}
🧾 1B. Create Encounter + Note (FHIR)
// server/integrations/ecwFhirService.ts

import axios from "axios";
import { getEcwAccessToken } from "./ecwAuth";

export async function createEcwEncounter(patientId: string, result: any) {
  const token = await getEcwAccessToken();

  const encounter = {
    resourceType: "Encounter",
    status: "finished",
    subject: { reference: `Patient/${patientId}` },
    reasonCode: [{ text: result.diagnosis }],
    note: [
      {
        text: `
HPI: ${result.trace.complaint}
Assessment: ${result.diagnosis}
Triage: ${result.triage}
Plan: ${result.plan.summary}
        `
      }
    ]
  };

  const res = await axios.post(
    `${process.env.ECW_FHIR_BASE}/Encounter`,
    encounter,
    {
      headers: {
        Authorization: `Bearer ${token}`,
        "Content-Type": "application/fhir+json"
      }
    }
  );

  return res.data;
}

🔥 2B. Reinforcement Update
// server/learning/rlhfEngine.ts

import { updateWeight } from "./weightStore";

export function reinforceOutcome(predicted: any, actual: any) {
  if (predicted.diagnosis === actual.diagnosis) {
    updateWeight(predicted.diagnosis, +0.1);
  } else {
    updateWeight(predicted.diagnosis, -0.2);
    updateWeight(actual.diagnosis, +0.2);
  }

  if (actual.triage === "ER" && predicted.triage !== "ER") {
    updateWeight("ER_bias", +0.5);
  }
}
