Interactive Map UI

client/components/intelligence-map/IntelligenceMapCanvas.tsx

Uses React Flow for interactive graph navigation.

import React,{useEffect,useState} from "react"
import ReactFlow,{MiniMap,Controls} from "reactflow"

export default function IntelligenceMapCanvas({onSelectNode}){

 const [graph,setGraph]=useState(null)

 useEffect(()=>{

  fetch("/api/intelligence-map")
  .then(r=>r.json())
  .then(setGraph)

 },[])

 if(!graph) return null

 const nodes = graph.nodes.map(n=>({

  id:n.id,
  data:{label:n.label},
  position:{x:Math.random()*500,y:Math.random()*500}

 }))

 const edges = graph.edges.map((e,i)=>({

  id:`e${i}`,
  source:e.from,
  target:e.to

 }))

 return(

 <div style={{height:600}}>

 <ReactFlow
  nodes={nodes}
  edges={edges}
  onNodeClick={(e,node)=>onSelectNode(node.id)}
 >

 <MiniMap/>
 <Controls/>

 </ReactFlow>

 </div>

 )

}

Debugger UI

client/pages/ReasoningDebugger.tsx

import React,{useState} from "react"

export default function ReasoningDebugger(){

 const [trace,setTrace]=useState(null)

 async function run(){

  const r = await fetch("/api/reasoning-debug",{
   method:"POST",
   headers:{"Content-Type":"application/json"},
   body:JSON.stringify({complaint:"cough"})
  })

  setTrace(await r.json())

 }

 return(

 <div>

 <h2>Clinical Reasoning Debugger</h2>

 <button onClick={run}>
 Run Trace
 </button>

 <pre>{JSON.stringify(trace,null,2)}</pre>

 </div>

 )

}
