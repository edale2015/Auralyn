✅ FIX
Limit buffer
const MAX_EVENTS = 1000;
let events = [];

export function emitEvent(e) {
  events.push(e);

  if (events.length > MAX_EVENTS) {
    events.shift();
  }

  listeners.forEach(l => l(e));
}
