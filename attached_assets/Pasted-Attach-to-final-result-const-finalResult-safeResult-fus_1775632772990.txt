import { AgentOpinion } from "../agents/agentTypes";

export function mapTelemedToAgents(result: any): AgentOpinion[] {
  const agents: AgentOpinion[] = [];

  // 🧪 Differential → Diagnostic Agent
  if (result.differential?.length) {
    agents.push({
      agentId: "diagnostic_engine",
      domain: "diagnosis",
      conclusion: result.differential[0].diagnosis,
      confidence: result.differential[0].confidence ?? 0.5,
      reasoning: "Top differential from probability engine",
      evidence: result.differential,
      priority: 2,
    });
  }

  // 🚑 Triage Agent
  agents.push({
    agentId: "triage_engine",
    domain: "triage",
    conclusion: result.triage.level,
    confidence: result.triage.urgencyScore / 100,
    reasoning: result.triage.reason,
    priority: 3,
  });

  // 🛑 Safety Agent (highest priority)
  if (result.safetyAlerts?.length) {
    agents.push({
      agentId: "safety_engine",
      domain: "safety",
      conclusion: "emergency",
      confidence: 1,
      reasoning: result.safetyAlerts.join(", "),
      priority: 10,
    });
  }

  // 💊 Treatment Agent
  if (result.medicationSuggestions?.length) {
    agents.push({
      agentId: "treatment_engine",
      domain: "treatment",
      conclusion: result.medicationSuggestions[0],
      confidence: 0.6,
      reasoning: "Suggested treatment pathway",
      priority: 1,
    });
  }

  return agents;
}
