import { v4 as uuidv4 } from "uuid";

export function generateInterventions(patient: any) {
  const interventions = [];

  const { sepsis, vitals } = patient;

  // 🚨 SEPSIS PROTOCOL
  if (sepsis.level === "HIGH") {
    interventions.push({
      id: uuidv4(),
      type: "ORDER",
      action: "lactate",
      priority: "STAT"
    });

    interventions.push({
      id: uuidv4(),
      type: "ORDER",
      action: "blood_cultures",
      priority: "STAT"
    });

    interventions.push({
      id: uuidv4(),
      type: "MED",
      action: "broad_spectrum_antibiotics",
      priority: "STAT"
    });

    interventions.push({
      id: uuidv4(),
      type: "MED",
      action: "IV_fluids_30ml_per_kg",
      priority: "STAT"
    });

    if (vitals.sbp < 90) {
      interventions.push({
        id: uuidv4(),
        type: "MED",
        action: "norepinephrine",
        priority: "STAT"
      });
    }
  }

  return interventions;
}
🧠 2. PHYSICIAN SAFETY GATE (CRITICAL)

You explicitly wanted non-autonomous prescribing without oversight.

📁 server/clinical/interventionGate.ts
export function applySafetyGate(interventions: any[], patient: any) {
  return interventions.map(i => {
    if (i.type === "MED") {
      return {
        ...i,
        requiresApproval: true,
        status: "PENDING_PHYSICIAN"
      };
    }

    return {
      ...i,
      requiresApproval: false,
      status: "AUTO_APPROVED"
    };
  });
}

import axios from "axios";

const FHIR_BASE = process.env.FHIR_BASE_URL;
const TOKEN = process.env.FHIR_TOKEN;

export async function createServiceRequest(patientId: string, order: any) {
  return axios.post(
    `${FHIR_BASE}/ServiceRequest`,
    {
      resourceType: "ServiceRequest",
      status: "active",
      intent: "order",
      subject: { reference: `Patient/${patientId}` },
      code: {
        text: order.action
      }
    },
    {
      headers: {
        Authorization: `Bearer ${TOKEN}`
      }
    }
  );
}

export async function createMedicationRequest(patientId: string, med: any) {
  return axios.post(
    `${FHIR_BASE}/MedicationRequest`,
    {
      resourceType: "MedicationRequest",
      status: "active",
      intent: "order",
      subject: { reference: `Patient/${patientId}` },
      medicationCodeableConcept: {
        text: med.action
      }
    },
    {
      headers: {
        Authorization: `Bearer ${TOKEN}`
      }
    }
  );
}
🛏️ 4. ICU BED ALLOCATOR (PREDICTIVE)
📁 server/hospital/icuAllocator.ts
interface Bed {
  id: string;
  occupied: boolean;
  hospital: string;
}

let beds: Bed[] = [
  { id: "ICU-1", occupied: false, hospital: "NYC-1" },
  { id: "ICU-2", occupied: false, hospital: "NYC-1" },
  { id: "ICU-3", occupied: true, hospital: "NYC-2" }
];

export function allocateICUBed(patient: any) {
  if (patient.sepsis.level !== "HIGH") return null;

  const available = beds.find(b => !b.occupied);

  if (!available) {
    return {
      status: "NO_BED_AVAILABLE",
      action: "TRANSFER_REQUIRED"
    };
  }

  available.occupied = true;

  return {
    status: "ALLOCATED",
    bedId: available.id,
    hospital: available.hospital
  };
}
🔁 5. FULL PIPELINE INTEGRATION
📁 server/agents/fullClinicalPipeline.ts
import { processPatient } from "./patientAgentRunner";
import { generateInterventions } from "../clinical/interventionEngine";
import { applySafetyGate } from "../clinical/interventionGate";
import { allocateICUBed } from "../hospital/icuAllocator";
import { createServiceRequest, createMedicationRequest } from "../ehr/fhirClient";

export async function runFullPipeline(patient: any) {
  const enriched = await processPatient(patient);

  // 🧠 Generate interventions
  let interventions = generateInterventions(enriched);

  // 🔐 Apply safety gate
  interventions = applySafetyGate(interventions, enriched);

  // 🛏️ ICU allocation
  const icu = allocateICUBed(enriched);

  // 🔗 EHR write (only auto-approved)
  for (const i of interventions) {
    if (!i.requiresApproval) {
      if (i.type === "ORDER") {
        await createServiceRequest(patient.id, i);
      }
    }
  }

  return {
    ...enriched,
    interventions,
    icu
  };
}
