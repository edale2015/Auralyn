// server/patient/sessionStore.ts

const sessions = new Map();

export function createSession(id, data) {
  sessions.set(id, {
    ...data,
    status: "pending",
    createdAt: new Date()
  });
}

export function updateSession(id, update) {
  const existing = sessions.get(id);
  sessions.set(id, { ...existing, ...update });
}

export function getAllSessions() {
  return Array.from(sessions.entries()).map(([id, s]) => ({
    id,
    ...s
  }));
}

// client/components/PatientDashboard.tsx

export function PatientDashboard() {
  const [patients, setPatients] = useState([]);

  useEffect(() => {
    const interval = setInterval(() => {
      fetch("/api/patients/queue")
        .then(r => r.json())
        .then(setPatients);
    }, 2000);

    return () => clearInterval(interval);
  }, []);

  return (
    <div>
      <h2>Live Patients</h2>
      {patients.map(p => (
        <div key={p.id}>
          <strong>{p.id}</strong>
          <span>{p.status}</span>
        </div>
      ))}
    </div>
  );
}

// middleware

const rateLimit = {};

export function limiter(req, res, next) {
  const ip = req.ip;
  rateLimit[ip] = (rateLimit[ip] || 0) + 1;

  if (rateLimit[ip] > 100) {
    return res.status(429).send("Too many requests");
  }

  next();
}
RETRY + FALLBACK
export async function withRetry(fn, retries = 3) {
  try {
    return await fn();
  } catch (e) {
    if (retries === 0) throw e;
    return withRetry(fn, retries - 1);
  }
}
