import { db } from "../../db";
import { auditStep } from "../../audit/auditLogger";
import { reloadKbCache } from "../kbRuntime";
import { publishKbInvalidation } from "../../events/kbInvalidation";
// import your real table once added to shared/schema
import { canonicalPathways } from "../../../shared/schema";

export async function createCanonicalPathway(
  data: {
    pathwayId: string;
    sourceType: string;
    complaintId: string;
    syndromeId: string;
    label: string;
    requiredFeatures: string[];
    positiveWeights: Record<string, number>;
    negativeWeights: Record<string, number>;
    exclusions: string[];
    treatmentClass: string;
    medicationKey?: string;
    canonicalDisposition: string;
    rationale: string[];
    active?: boolean;
  },
  actorId: string,
  traceId: string
) {
  await db.insert(canonicalPathways).values({
    ...data,
    active: true,
    createdBy: actorId,
    updatedBy: actorId,
  });

  await auditStep({
    traceId,
    step: "kb_canonical_pathway_created",
    input: { actorId, ...data },
    output: { pathwayId: data.pathwayId },
    metadata: { actorId },
  });

  await reloadKbCache(traceId);
  await publishKbInvalidation({
    table: "canonicalPathways",
    ruleId: data.pathwayId,
    action: "create",
    traceId,
  });

  return { ok: true, pathwayId: data.pathwayId };
}

export async function retireCanonicalPathway(
  pathwayId: string,
  actorId: string,
  traceId: string,
  reason: string
) {
  await db
    .update(canonicalPathways)
    .set({
      active: false,
      retiredAt: new Date(),
      retiredBy: actorId,
      retirementReason: reason,
      updatedBy: actorId,
    })
    .where((canonicalPathways as any).pathwayId.eq(pathwayId));

  await auditStep({
    traceId,
    step: "kb_canonical_pathway_retired",
    input: { pathwayId, actorId, reason },
    output: { pathwayId, active: false },
    metadata: { actorId },
  });

  await reloadKbCache(traceId);
  await publishKbInvalidation({
    table: "canonicalPathways",
    ruleId: pathwayId,
    action: "retire",
    traceId,
  });

  return { ok: true, pathwayId };
}

export async function previewCanonicalPromotionHandler(req: Request, res: Response) {
  const parsed = CanonicalPathwayPromotionSchema.safeParse(req.body);
  if (!parsed.success) {
    return res.status(400).json({ ok: false, error: parsed.error.flatten() });
  }

  const body = parsed.data;

  return res.json({
    ok: true,
    preview: {
      complaintId: body.complaintId,
      syndromeId: body.syndromeId,
      label: body.label,
      canonicalDisposition: body.canonicalDisposition,
      treatmentClass: body.treatmentClass,
      requiredFeaturesCount: body.requiredFeatures.length,
      positiveFeatureCount: Object.keys(body.positiveWeights).length,
      negativeFeatureCount: Object.keys(body.negativeWeights).length,
      wouldCreate: true,
    },
  });
}

export async function promoteCanonicalPathwayHandler(req: Request, res: Response) {
  const parsed = CanonicalPathwayPromotionSchema.safeParse(req.body);
  if (!parsed.success) {
    return res.status(400).json({ ok: false, error: parsed.error.flatten() });
  }

  const body = parsed.data;

  const result = await createCanonicalPathway(
    {
      pathwayId: uuidv4(),
      sourceType: body.sourceType,
      complaintId: body.complaintId,
      syndromeId: body.syndromeId,
      label: body.label,
      requiredFeatures: body.requiredFeatures,
      positiveWeights: body.positiveWeights,
      negativeWeights: body.negativeWeights,
      exclusions: body.exclusions,
      treatmentClass: body.treatmentClass,
      medicationKey: body.medicationKey,
      canonicalDisposition: body.canonicalDisposition,
      rationale: body.rationale,
    },
    body.actorId,
    body.traceId
  );

  return res.json({ ok: true, result });
}

export async function retireCanonicalPathwayHandler(req: Request, res: Response) {
  const parsed = CanonicalPathwayRetireSchema.safeParse(req.body);
  if (!parsed.success) {
    return res.status(400).json({ ok: false, error: parsed.error.flatten() });
  }

  const { pathwayId, actorId, traceId, reason } = parsed.data;
  const result = await retireCanonicalPathway(pathwayId, actorId, traceId, reason);
  return res.json({ ok: true, result });

export async function generateCanonicalDraftFromCaseHandler(req: Request, res: Response) {
  const parsed = CanonicalDraftFromCaseSchema.safeParse(req.body);
  if (!parsed.success) {
    return res.status(400).json({ ok: false, error: parsed.error.flatten() });
  }

    draft: {
      sourceType: "golden_case",
      complaintId: body.complaint,

      label: canonical.winningSyndrome?.label ?? "Undifferentiated phenotype",
      requiredFeatures: Object.entries(body.features)
        .filter(([, v]) => v === true)
        .map(([k]) => k),
      positiveWeights: {},
      negativeWeights: {},
      exclusions: [],
      treatmentClass: canonical.treatment.class,
      medicationKey: canonical.treatment.medicationKey,
      canonicalDisposition: canonical.disposition.disposition,
      rationale: canonical.notesForClinician,

  });

7) server/services/goldenCaseConsistencyIntegration.ts

Your golden-case layer should call this after each run. The design goal is exactly what your repo notes recommend: every golden case should run in CI and production, and a changed disposition should be visible as a mismatch.

Use this service
import { Request, Response } from "express";
import { runClinicalConsistencyEngine } from "./clinicalConsistencyEngine";

function scoreGoldenMatch(args: {
  expectedSyndromeId?: string;
  expectedMedicationKey?: string;
  expectedDisposition?: string;
  actual: ReturnType<typeof runClinicalConsistencyEngine>;
}) {
  const reasons: string[] = [];
  let matched = 0;
  let total = 0;

  if (args.expectedSyndromeId) {
    total += 1;
    if (args.actual.winningSyndrome?.syndromeId === args.expectedSyndromeId) matched += 1;
    else reasons.push(`Expected syndrome ${args.expectedSyndromeId}, got ${args.actual.winningSyndrome?.syndromeId ?? "none"}`);
  }

  if (args.expectedMedicationKey) {
    total += 1;
    if (args.actual.treatment.medicationKey === args.expectedMedicationKey) matched += 1;
    else reasons.push(`Expected medication ${args.expectedMedicationKey}, got ${args.actual.treatment.medicationKey ?? "none"}`);
  }

  if (args.expectedDisposition) {
    total += 1;
    if (args.actual.disposition.disposition === args.expectedDisposition) matched += 1;
    else reasons.push(`Expected disposition ${args.expectedDisposition}, got ${args.actual.disposition.disposition}`);
  }

  return {
    totalChecks: total,
    matchedChecks: matched,
    score: total === 0 ? 1 : matched / total,
    reasons,
  };
}

export async function evaluateGoldenCaseAgainstCanonicalHandler(req: Request, res: Response) {
  const {
    caseId,
    complaint,
    features,
    expectedSyndromeId,
    expectedMedicationKey,
    expectedDisposition,
  } = req.body;

  const actual = runClinicalConsistencyEngine(complaint, features || {});
  const comparison = scoreGoldenMatch({
    expectedSyndromeId,
    expectedMedicationKey,
    expectedDisposition,
    actual,
  });

  return res.json({
    ok: true,
    caseId,
    canonical: actual,
    comparison,
  });
}

export async function evaluateGoldenCaseBatchAgainstCanonicalHandler(req: Request, res: Response) {
  const cases = Array.isArray(req.body?.cases) ? req.body.cases : [];

  const results = cases.map((c: any) => {
    const actual = runClinicalConsistencyEngine(c.complaint, c.features || {});
    const comparison = scoreGoldenMatch({
      expectedSyndromeId: c.expectedSyndromeId,
      expectedMedicationKey: c.expectedMedicationKey,
      expectedDisposition: c.expectedDisposition,
      actual,
    });

    return {
      caseId: c.caseId,
      complaint: c.complaint,
      canonical: actual,
      comparison,
    };
  });

  const avgScore =
    results.length === 0
      ? 0
      : results.reduce((sum, r) => sum + r.comparison.score, 0) / results.length;

  return res.json({
    ok: true,
    batchCount: results.length,
    avgScore,
    results,
  });
}

import { Request, Response } from "express";
import { runClinicalConsistencyEngine } from "./clinicalConsistencyEngine";

const VALID_CATEGORIES = new Set([
  "diagnosis_incorrect",
  "diagnosis_incomplete",
  "disposition_too_aggressive",
  "disposition_insufficient",
  "medication_inappropriate",
  "documentation_error",
  "patient_preference",
  "clinical_context_not_captured",
  "other",
]);

const inMemoryOverrides: any[] = [];

function reviewNeeded(category: string, reasonText: string) {
  const autoSafe = new Set([
    "patient_preference",
    "clinical_context_not_captured",
  ]);

  if (autoSafe.has(category)) return false;
  if (/allergy|contraindication|pregnan/i.test(reasonText)) return false;
  return true;
}

export async function createPhysicianOverrideHandler(req: Request, res: Response) {
  const {
    encounterId,
    physicianId,
    complaint,
    features,
    clinicianSelectedDisposition,
    clinicianSelectedMedicationKey,
    category,
    reasonText,
  } = req.body;

  if (!VALID_CATEGORIES.has(category)) {
    return res.status(400).json({ ok: false, error: "Invalid override category" });
  }

  if (!reasonText || String(reasonText).trim().length < 5) {
    return res.status(400).json({ ok: false, error: "Override rationale is required" });
  }

  const canonical = runClinicalConsistencyEngine(complaint, features || {});
  const needsReview = reviewNeeded(category, reasonText);

  const overrideRecord = {
    id: `ovr_${Date.now()}`,
    encounterId,
    physicianId,
    category,
    reasonText,
    complaint,
    phenotypeHash: canonical.phenotypeHash,
    canonicalDisposition: canonical.disposition.disposition,
    selectedDisposition: clinicianSelectedDisposition,
    canonicalMedicationKey: canonical.treatment.medicationKey,
    selectedMedicationKey: clinicianSelectedMedicationKey,
    reviewNeeded: needsReview,
    createdAt: new Date().toISOString(),
  };

  inMemoryOverrides.push(overrideRecord);

  return res.json({
    ok: true,
    override: overrideRecord,
    canonical,
  });
}

export async function listPhysicianOverridesHandler(_req: Request, res: Response) {
  return res.json({
    ok: true,
    overrides: inMemoryOverrides.slice().reverse(),
  });
}

Middleware
import { Request, Response, NextFunction } from "express";

export function requireDispositionFirst(req: Request, res: Response, next: NextFunction) {
  const {
    finalDisposition,
    finalMedicationOrders,
    dispositionReviewedAt,
  } = req.body || {};

  if (finalMedicationOrders?.length && !finalDisposition) {
    return res.status(400).json({
      ok: false,
      error: "Disposition must be selected before final medication orders.",
    });
  }

  if (finalMedicationOrders?.length && !dispositionReviewedAt) {
    return res.status(400).json({
      ok: false,
      error: "Disposition review timestamp required before medication finalization.",
    });
  }

  next();
}

              <div><strong>Selected disposition:</strong> {o.selectedDisposition || "unchanged"}</div>
              <div><strong>Review needed:</strong> {o.reviewNeeded ? "Yes" : "No"}</div>
            </div>
            <div className="mt-2 text-sm">
              <strong>Reason:</strong> {o.reasonText}
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
