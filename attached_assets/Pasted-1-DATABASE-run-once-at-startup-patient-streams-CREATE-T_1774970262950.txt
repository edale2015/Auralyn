import { WebSocketServer } from 'ws'

export const wss = new WebSocketServer({ port: 8081 })

export function broadcastUpdate(payload: any) {
  const msg = JSON.stringify(payload)
  wss.clients.forEach(c => {
    if (c.readyState === 1) c.send(msg)
  })
}
⚙️ 3) BACKEND — DETERIORATION ENGINE
// server/engine/deteriorationEngine.ts
export async function detectDeterioration(db: any, patientId: string) {
  const series = await db.patient_live_stream.findRecent(patientId)
  const rules = await db.kb_deterioration_rules.findAll({ is_active: true })

  let risk = 0
  const alerts: any[] = []

  for (const r of rules) {
    const vals = series.filter((s: any) => s.feature_key === r.feature_key).map((s: any) => s.value)
    if (vals.length < 2) continue

    const delta = vals[vals.length - 1] - vals[0]

    if (r.trend === 'rising' && delta > r.threshold) {
      risk += r.risk_weight
      alerts.push({ ...r, delta })
    }
  }

  return { risk, alerts }
}

// server/routes/controlRoutes.ts
import express from 'express'
import { broadcastUpdate } from '../ws/patientStream'
import { detectDeterioration } from '../engine/deteriorationEngine'

const router = express.Router()

// LIVE STREAM (push sample or from pipeline)
router.post('/stream', async (req, res) => {
  const { patient_id, feature_key, value } = req.body

  await req.db.patient_live_stream.insert({ patient_id, feature_key, value })

  broadcastUpdate({ patient_id, feature_key, value })

  res.json({ ok: true })
})

// ALERTS
router.get('/alerts/:patientId', async (req, res) => {
  const out = await detectDeterioration(req.db, req.params.patientId)
  res.json(out)
})

// client/src/pages/ControlTowerV2.tsx
import { useEffect, useState } from 'react'
import LivePatients from '../components/LivePatients'
import AlertsPanel from '../components/AlertsPanel'
import VoicePanel from '../components/VoicePanel'
import DecisionTree from '../components/DecisionTree'
import ScoringPanel from '../components/ScoringPanel'
import QuestionPanel from '../components/QuestionPanel'

export default function ControlTowerV2() {
  const [stream, setStream] = useState<any[]>([])
  const [alerts, setAlerts] = useState<any[]>([])

  useEffect(() => {
    const ws = new WebSocket('ws://localhost:8081')

    ws.onmessage = e => {
      const data = JSON.parse(e.data)
      setStream(prev => [data, ...prev.slice(0, 20)])
    }
  }, [])

  return (
    <div className="grid grid-cols-4 gap-4 p-4">
      <LivePatients data={stream} />
      <DecisionTree />
      <ScoringPanel />
      <QuestionPanel />
      <CounterfactualPanel />
      <VoicePanel />
      <AlertsPanel alerts={alerts} />
    </div>
  )
}
