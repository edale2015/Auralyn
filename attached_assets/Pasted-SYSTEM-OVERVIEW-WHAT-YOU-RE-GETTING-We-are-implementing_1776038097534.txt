🧪 1. MULTI-PATIENT SIMULATION ENGINE
server/simulation/patientGenerator.ts
import { v4 as uuidv4 } from "uuid";

export function generatePatient() {
  const age = Math.floor(Math.random() * 90) + 1;

  return {
    id: uuidv4(),
    age,
    vitals: {
      hr: rand(60, 140),
      rr: rand(12, 35),
      temp: rand(36, 40),
      sbp: rand(80, 180),
      spo2: rand(85, 100),
    },
    symptoms: pickSymptoms(),
    labs: {
      lactate: rand(0.5, 6),
      wbc: rand(3, 20),
    },
  };
}

function rand(min: number, max: number) {
  return Math.round((Math.random() * (max - min) + min) * 10) / 10;
}

function pickSymptoms() {
  const all = ["fever", "cough", "sob", "confusion", "chest pain"];
  return all.filter(() => Math.random() > 0.5);
}
server/simulation/multiPatientSimulator.ts
import { generatePatient } from "./patientGenerator";
import { runValidation } from "../evals/validationHarness";

export async function simulatePatients(n: number = 1000) {
  const patients = Array.from({ length: n }, generatePatient);

  const results = [];

  for (const p of patients) {
    const result = await runValidation(p);
    results.push(result);
  }

  return results;
}
🚨 2. SEPSIS + ICU ENGINE
server/clinical/sepsisEngine.ts
export function calculateNEWS2(vitals: any) {
  let score = 0;

  if (vitals.rr > 25) score += 3;
  if (vitals.spo2 < 92) score += 3;
  if (vitals.temp > 38.5) score += 2;
  if (vitals.sbp < 90) score += 3;
  if (vitals.hr > 130) score += 3;

  return score;
}

export function calculateQSOFA(vitals: any, mentalStatus: string) {
  let score = 0;

  if (vitals.rr >= 22) score += 1;
  if (vitals.sbp <= 100) score += 1;
  if (mentalStatus === "altered") score += 1;

  return score;
}

export function detectSepsis(patient: any) {
  const news = calculateNEWS2(patient.vitals);
  const qsofa = calculateQSOFA(patient.vitals, "normal");

  const lactateHigh = patient.labs.lactate > 2;

  return {
    sepsisRisk: news > 5 || qsofa >= 2 || lactateHigh,
    news,
    qsofa,
  };
}
server/clinical/icuPredictor.ts
export function predictICUNeed(patient: any, sepsis: any) {
  let risk = 0;

  if (sepsis.sepsisRisk) risk += 0.5;
  if (patient.vitals.sbp < 90) risk += 0.3;
  if (patient.labs.lactate > 4) risk += 0.4;

  return {
    icuProbability: Math.min(risk, 1),
    needsICU: risk > 0.6,
  };
}
🧪 3. VALIDATION HARNESS
server/evals/validationHarness.ts
import { detectSepsis } from "../clinical/sepsisEngine";
import { predictICUNeed } from "../clinical/icuPredictor";

export async function runValidation(patient: any) {
  const sepsis = detectSepsis(patient);
  const icu = predictICUNeed(patient, sepsis);

  const expected = deriveExpected(patient);

  const correct =
    expected.sepsis === sepsis.sepsisRisk &&
    expected.icu === icu.needsICU;

  return {
    patientId: patient.id,
    sepsis,
    icu,
    expected,
    correct,
  };
}

function deriveExpected(patient: any) {
  return {
    sepsis: patient.labs.lactate > 2,
    icu: patient.vitals.sbp < 90,
  };
}
🤖 4. RLHF AUTO-WEIGHT TUNING
server/rlhf/weightUpdater.ts
import { db } from "../db";

export async function updateWeights(results: any[]) {
  const failures = results.filter((r) => !r.correct);

  if (failures.length < 50) return;

  const delta = 0.02;

  for (const f of failures) {
    await db.query(`
      UPDATE clinical_weights
      SET weight = weight + $1
      WHERE feature = 'lactate'
    `, [delta]);
  }

  return {
    updated: true,
    failures: failures.length,
  };
}

🧬 5. GOLDEN CASE GENERATOR
server/golden/goldenGenerator.ts
import { db } from "../db";

export async function generateGoldenCases(results: any[]) {
  const highRisk = results.filter(
    (r) => r.sepsis.sepsisRisk || r.icu.needsICU
  );

  for (const r of highRisk) {
    await db.query(`
      INSERT INTO golden_cases (patient_id, data, expected)
      VALUES ($1, $2, $3)
    `, [
      r.patientId,
      JSON.stringify(r),
      JSON.stringify(r.expected),
    ]);
  }

  return { count: highRisk.length };
}
📦 6. ZIP EXPORT SYSTEM
server/export/zipExporter.ts
import archiver from "archiver";
import fs from "fs";

export async function exportZip() {
  const output = fs.createWriteStream("hospital_system.zip");
  const archive = archiver("zip");

  archive.pipe(output);

  archive.directory("server/", "server");
  archive.directory("frontend/", "frontend");

  await archive.finalize();

  return "hospital_system.zip";
}
🌐 7. COMMAND CENTER API
server/api/commandCenter.routes.ts
import express from "express";
import { simulatePatients } from "../simulation/multiPatientSimulator";
import { updateWeights } from "../rlhf/weightUpdater";
import { generateGoldenCases } from "../golden/goldenGenerator";
import { exportZip } from "../export/zipExporter";

const router = express.Router();

router.post("/simulate", async (req, res) => {
  const results = await simulatePatients(1000);
  res.json(results);
});

router.post("/rlhf", async (req, res) => {
  const { results } = req.body;
  const update = await updateWeights(results);
  res.json(update);
});

router.post("/golden", async (req, res) => {
  const { results } = req.body;
  const g = await generateGoldenCases(results);
  res.json(g);
});

router.get("/export", async (req, res) => {
  const zip = await exportZip();
  res.download(zip);
});

export default router;
