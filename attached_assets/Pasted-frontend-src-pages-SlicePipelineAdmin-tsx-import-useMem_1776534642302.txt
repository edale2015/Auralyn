Make sure the export response includes ok: true and the slices array so the UI can render properly:

router.post('/export-claude-slices', async (_req, res) => {
  try {
    const result = await exportClaudeSlices();
    res.json({
      ok: true,
      ...result,
    });
  } catch (error: any) {
    res.status(500).json({
      ok: false,
      error: error?.message ?? 'Unknown export error',
    });
  }
});
