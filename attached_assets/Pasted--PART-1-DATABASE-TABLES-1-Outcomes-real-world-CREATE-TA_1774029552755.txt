🔗 PART 4 — OUTCOME MATCHING ENGINE
📁 server/rwe/matchEngine.ts
import { db } from "../db";

export async function matchOutcomes() {
  const rows = await db.execute(`
    SELECT * FROM real_world_outcomes
    WHERE matched = false
      AND actual_outcome IS NOT NULL
  `);

  for (const r of rows.rows) {
    const match = r.predicted_diagnosis === r.actual_outcome;

    await db.execute(`
      UPDATE real_world_outcomes
      SET matched = true
      WHERE id = ${r.id}
    `);
  }

  return rows.rows.length;
}
📊 PART 5 — LIVE METRICS ENGINE (RWE)
📁 server/rwe/metricsEngine.ts
import { db } from "../db";

export async function computeRealWorldMetrics() {
  const rows = await db.execute(`
    SELECT * FROM real_world_outcomes
    WHERE matched = true
  `);

  let correct = 0;

  rows.rows.forEach(r => {
    if (r.predicted_diagnosis === r.actual_outcome) {
      correct++;
    }
  });

  const accuracy = correct / (rows.rows.length || 1);

  return {
    accuracy,
    sampleSize: rows.rows.length
  };
}
🔁 PART 6 — CONTINUOUS RWE LOOP
📁 server/system/rweLoop.ts
import { matchOutcomes } from "../rwe/matchEngine";
import { computeRealWorldMetrics } from "../rwe/metricsEngine";
import { emitEvent } from "../controlTower/eventBus";

export function startRweLoop() {
  setInterval(async () => {
    const matched = await matchOutcomes();
    const metrics = await computeRealWorldMetrics();

    emitEvent({
      type: "RWE_UPDATE",
      payload: { matched, metrics },
      timestamp: Date.now()
    });

  }, 60000);
}
