import { getSystemState } from "../control/systemState";
import { computeScale } from "../infra/autoscale";
import { runSimulation } from "../simulation/simulationEngine";

export async function runAutopilot() {

  const state = getSystemState();
  const actions: string[] = [];

  // 🔥 Scaling decision
  if (state.infrastructure.regions.length > 0) {
    const scale = computeScale(50);
    actions.push(`Scale to ${scale} workers`);
  }

  // 🔥 ML drift response
  if (state.ml.drift) {
    actions.push("Trigger retraining");
  }

  // 🔥 Simulation refresh
  if (Date.now() - state.simulation.lastRun > 60000) {
    await runSimulation(1000);
    actions.push("Refreshed simulation");
  }

  return {
    actions,
    mode: "SAFE_AUTOPILOT"
  };
}
