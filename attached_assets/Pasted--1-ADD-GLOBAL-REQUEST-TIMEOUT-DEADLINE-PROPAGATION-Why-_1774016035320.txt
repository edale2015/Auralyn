export function withDeadline(ms = 3000) {
  return (req, res, next) => {
    const timer = setTimeout(() => {
      res.status(504).json({ error: "Request timeout" });
    }, ms);

    res.on("finish", () => clearTimeout(timer));
    next();
  };
}
🔥 Apply globally
app.use("/api/clinical", withDeadline(3000));

export async function detectDrift(recent, baseline) {
  const diff = Math.abs(recent.avgAge - baseline.avgAge);

  if (diff > 10) {
    return {
      drift: true,
      message: "Patient age distribution shifted significantly"
    };
  }

  return { drift: false };
}

export function guardrailCheck({ diagnosis, meds }) {
  if (diagnosis === "pneumonia" && meds.includes("cough_suppressant")) {
    return {
      override: true,
      reason: "Cough suppression contraindicated in pneumonia"
    };
  }

  return { override: false };
}
