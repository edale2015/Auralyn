let failureCount = 0;

export function circuitBreaker(fn) {
  return async (...args) => {
    if (failureCount > 5) {
      throw new Error("Circuit open");
    }

    try {
      const res = await fn(...args);
      failureCount = 0;
      return res;
    } catch (e) {
      failureCount++;
      throw e;
    }
  };
}
