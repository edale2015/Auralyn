1. Telemedicine Assistant Service

Central coordinator that uses the Clinical Brain and UCSM state.

server/assistant/telemedicineAssistantService.ts
import { rebuildClinicalState }
from "../core/state/clinicalStateProjector"

import { runClinicalBrain }
from "../brain/clinicalBrainEngine"

import { suggestQuestions }
from "../assistant/dynamicQuestionService"

export async function runTelemedicineAssistant(caseId:string){

  const state =
    await rebuildClinicalState(caseId)

  const reasoning =
    await runClinicalBrain(state)

  const questions =
    suggestQuestions({
      complaint:reasoning.differential?.[0]?.diagnosis,
      symptoms:state.symptoms
    })

  return {

    complaint:state.complaint,

    triage:reasoning.triage,

    differential:reasoning.differential,

    contradictions:reasoning.contradictions,

    resources:reasoning.resources,

    pathway:reasoning.pathway,

    nextQuestions:questions

  }

}

2. Telemedicine Assistant API
server/routes/telemedicineAssistantRoutes.ts
import express from "express"

import { publishEvent }
from "../core/events/eventPublisher"

import { runTelemedicineAssistant }
from "../assistant/telemedicineAssistantService"

const router = express.Router()

router.post("/api/telemed/assistant", async(req,res)=>{

  const { caseId, message } = req.body

  await publishEvent(
    caseId,
    "PATIENT_MESSAGE",
    { message }
  )

  const result =
    await runTelemedicineAssistant(caseId)

  res.json({
    ok:true,
    result
  })

})

export default router

3. Telegram / WhatsApp Adapter

This connects messaging platforms to the assistant.

server/integrations/telegramAdapter.ts
import fetch from "node-fetch"

const TELEGRAM_API =
  "https://api.telegram.org/bot"

export async function sendTelegramMessage(
  botToken:string,
  chatId:string,
  text:string
){

  await fetch(
    `${TELEGRAM_API}${botToken}/sendMessage`,
    {
      method:"POST",
      headers:{
        "Content-Type":"application/json"
      },
      body:JSON.stringify({
        chat_id:chatId,
        text
      })
    }
  )

}

server/routes/telegramWebhook.ts
import express from "express"

import { runTelemedicineAssistant }
from "../assistant/telemedicineAssistantService"

import { publishEvent }
from "../core/events/eventPublisher"

const router = express.Router()

router.post("/telegram/webhook", async(req,res)=>{

  const message =
    req.body.message?.text

  const chatId =
    req.body.message?.chat?.id

  const caseId =
    `tg_${chatId}`

  await publishEvent(
    caseId,
    "PATIENT_MESSAGE",
    { message }
  )

  const result =
    await runTelemedicineAssistant(caseId)

  const reply =
    formatAssistantReply(result)

  await sendTelegramMessage(
    process.env.TELEGRAM_BOT_TOKEN,
    chatId,
    reply
  )

  res.sendStatus(200)

})

function formatAssistantReply(result:any){

  const top =
    result.differential?.[0]

  return `
Possible diagnosis:
${top?.diagnosis}

Urgency: ${result.triage.level}

Suggested questions:
${result.nextQuestions.join("\n")}

Recommended actions:
${result.resources?.recommendedActions
  ?.map((a:any)=>a.diagnosis)
  .join("\n")}
`
}

export default router

client/src/pages/RapidTelemedicineConsole.tsx

import React, { useState } from "react"

type AssistantResult = {
  complaint?: string
  triage?: { level: string; reason?: string }
  differential?: { diagnosis: string; score: number }[]
  nextQuestions?: string[]
  resources?: { recommendedActions?: any[] }
}

export default function RapidTelemedicineConsole(){

  const [caseId] = useState(`visit_${Date.now()}`)
  const [message,setMessage] = useState("")
  const [result,setResult] = useState<AssistantResult|null>(null)

  async function analyze(){

    if(!message.trim()) return

    const res = await fetch("/api/telemed/assistant",{
      method:"POST",
      headers:{ "Content-Type":"application/json" },
      body:JSON.stringify({
        caseId,
        message
      })
    })

    const data = await res.json()

    setResult(data.result)
  }

  function appendQuestion(q:string){
    setMessage(m => m + " " + q)
  }

  function copyText(text:string){
    navigator.clipboard.writeText(text)
  }

  const note = `
Chief Complaint:
${result?.complaint ?? ""}

History:
${message}

Assessment:
${result?.differential?.[0]?.diagnosis ?? ""}

Plan:
${result?.resources?.recommendedActions
  ?.map((a:any)=>a.diagnosis)
  .join("\n")}

Disposition:
${result?.triage?.level}
`

  const discharge = `
Based on our visit your symptoms are most consistent with:

${result?.differential?.[0]?.diagnosis ?? ""}

Recommended care:
${result?.resources?.recommendedActions
  ?.map((a:any)=>a.diagnosis)
  .join("\n")}

Return immediately if symptoms worsen.
`

  return (

  <div style={{
    display:"grid",
    gridTemplateColumns:"2fr 1fr 1fr",
    gap:"20px",
    padding:"20px",
    fontFamily:"Arial"
  }}>

    {/* LEFT COLUMN – VISIT INPUT */}

    <div>

      <h2>Patient Message</h2>

      <textarea
        style={{width:"100%",height:"120px"}}
        value={message}
        onChange={e=>setMessage(e.target.value)}
        placeholder="Paste patient message from Telegram / WhatsApp"
      />

      <div style={{marginTop:"10px"}}>

        {[
          "cough",
          "sore throat",
          "UTI symptoms",
          "rash",
          "ear pain",
          "sinus pressure",
          "fever",
          "chest pain",
          "abdominal pain"
        ].map(c => (
          <button
            key={c}
            onClick={()=>setMessage(m=>m+" "+c)}
            style={{margin:"3px"}}
          >
            {c}
          </button>
        ))}

      </div>

      <div style={{marginTop:"10px"}}>

        <button
          onClick={analyze}
          style={{
            background:"#2c7be5",
            color:"white",
            padding:"10px"
          }}
        >
          Analyze Visit
        </button>

      </div>

      <h3 style={{marginTop:"20px"}}>Suggested Questions</h3>

      {result?.nextQuestions?.map(q=>(
        <div key={q}>
          <button
            onClick={()=>appendQuestion(q)}
          >
            {q}
          </button>
        </div>
      ))}

    </div>


    {/* MIDDLE COLUMN – REASONING */}

    <div>

      <h2>Differential</h2>

      {result?.differential?.map(d => (

        <div key={d.diagnosis}>

          {d.diagnosis}
          {" "}
          ({Math.round(d.score*100)}%)

        </div>

      ))}

      <h2 style={{marginTop:"20px"}}>Triage</h2>

      <div>

        {result?.triage?.level}

      </div>

      <h2 style={{marginTop:"20px"}}>Recommended Actions</h2>

      {result?.resources?.recommendedActions?.map((a:any)=>(
        <div key={a.diagnosis}>
          {a.diagnosis}
        </div>
      ))}

    </div>


    {/* RIGHT COLUMN – NOTE + DISCHARGE */}

    <div>

      <h2>Chart Note</h2>

      <pre style={{background:"#f5f5f5",padding:"10px"}}>
        {note}
      </pre>

      <button onClick={()=>copyText(note)}>
        Copy Chart Note
      </button>

      <h2 style={{marginTop:"20px"}}>Discharge</h2>

      <pre style={{background:"#f5f5f5",padding:"10px"}}>
        {discharge}
      </pre>

      <button onClick={()=>copyText(discharge)}>
        Copy Discharge Message
      </button>

    </div>

  </div>

  )

}
