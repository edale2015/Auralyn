// server/integrations/fhirClient.ts

import fetch from "node-fetch";

const FHIR_BASE = process.env.FHIR_BASE_URL;
const TOKEN = process.env.FHIR_TOKEN;

export async function fhirRequest(endpoint: string, method = "GET", body?: any) {
  const res = await fetch(`${FHIR_BASE}/${endpoint}`, {
    method,
    headers: {
      "Authorization": `Bearer ${TOKEN}`,
      "Content-Type": "application/fhir+json"
    },
    body: body ? JSON.stringify(body) : undefined
  });

  return res.json();
}
🧬 Pull Patient Data
// server/integrations/getPatientContext.ts

export async function getPatientContext(patientId: string) {
  const [patient, meds, allergies] = await Promise.all([
    fhirRequest(`Patient/${patientId}`),
    fhirRequest(`MedicationRequest?patient=${patientId}`),
    fhirRequest(`AllergyIntolerance?patient=${patientId}`)
  ]);

  return { patient, meds, allergies };
}
🧾 Push Encounter + Diagnosis
// server/integrations/writeEncounter.ts

export async function writeEncounter(patientId, result) {
  return fhirRequest("Encounter", "POST", {
    resourceType: "Encounter",
    status: "finished",
    subject: { reference: `Patient/${patientId}` },
    diagnosis: [
      {
        condition: {
          display: result.diagnosis
        }
      }
    ]
  });
}

// server/billing/codingEngine.ts

export function generateCodes(result: any) {
  const icdMap = {
    "viral_uri": "J06.9",
    "chest_pain": "R07.9"
  };

  const cptMap = {
    "HOME": "99213",
    "URGENT_CARE": "99214",
    "ED": "99285"
  };

  return {
    icd10: icdMap[result.diagnosis] || "R69",
    cpt: cptMap[result.disposition]
  };
}
💰 Payer Optimization Layer
// server/billing/payerEngine.ts

export function optimizeBilling(codes, payerRules) {
  if (payerRules.requireHighComplexity && codes.cpt === "99213") {
    return { ...codes, cpt: "99214" };
  }

  return codes;
}

export async function computeMetrics() {
  const res = await pool.query(`
    SELECT
      COUNT(*) as total,
      SUM(CASE WHEN disposition = 'ED' THEN 1 ELSE 0 END) as escalations
    FROM cases
  `);

  return res.rows[0];
}

// server/fda/goldenValidator.ts

export async function validateGoldenCases(cases) {
  let correct = 0;

  for (const c of cases) {
    const result = await runCognitiveBrain(c.input);

    if (result.disposition === c.expectedDisposition) {
      correct++;
    }
  }

  return {
    accuracy: correct / cases.length
  };
}
🧾 Audit Hash Chain (FDA CRITICAL)
// server/audit/hashChain.ts

import crypto from "crypto";

let lastHash = "";

export function generateAuditHash(data: any) {
  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(data) + lastHash)
    .digest("hex");

  lastHash = hash;
  return hash;
}
