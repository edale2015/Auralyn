// server/middleware/tenantContext.ts

export function enforceTenant(req: any, res: any, next: any) {
  const clinicId = req.physician?.clinicId;

  if (!clinicId) {
    return res.status(403).json({ error: "No clinic context" });
  }

  req.clinicId = clinicId;
  next();
}
✅ Multi-Clinic Aggregation API
// server/routes/networkRoutes.ts

router.get("/network/summary", requireRole(["admin"]), async (_req, res) => {
  const data = await db.query(`
    SELECT clinic_id,
           SUM(patients_seen) as patients,
           SUM(revenue_generated) as revenue,
           AVG(denial_rate) as denial_rate
    FROM clinic_metrics
    GROUP BY clinic_id
  `);

  res.json(data.rows);
});

✅ ROI CALCULATOR
// server/payer/roiEngine.ts

export function calculateROI(metrics: any) {
  const erSavings = metrics.erAvoided * 1500; // avg ER cost
  const revenue = metrics.visits * metrics.ratePerVisit;

  const total = revenue + erSavings;

  return {
    revenue,
    erSavings,
    totalValue: total,
  };
}

export function scoreClinicOpportunity(clinic: any) {
  let score = 0;

  if (clinic.erRate > 0.3) score += 5;
  if (clinic.denialRate > 0.2) score += 5;
  if (clinic.volume > 1000) score += 3;

  return score;
}
