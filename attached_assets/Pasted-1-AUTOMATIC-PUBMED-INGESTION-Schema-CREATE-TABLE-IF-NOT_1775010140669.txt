// server/guidelines/pubmed.ts
import fetch from 'node-fetch'

export async function fetchPubMed(query: string) {
  const search = await fetch(
    `https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi?db=pubmed&retmode=json&term=${encodeURIComponent(query)}`
  ).then(r => r.json())

  const ids = search.esearchresult.idlist.slice(0, 5)

  const details = await fetch(
    `https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esummary.fcgi?db=pubmed&retmode=json&id=${ids.join(',')}`
  ).then(r => r.json())

  return Object.values(details.result).filter((r: any) => r.uid)
}
