🏥 2. MULTI-CLINIC ISOLATION
📁 server/middleware/clinicScope.ts
export function clinicScope(req: any, res: any, next: any) {
  const clinicId = req.headers["x-clinic-id"];

  if (!clinicId) {
    return res.status(400).json({ error: "Missing clinic context" });
  }

  req.clinicId = clinicId;
  next();
}

🔐 3. AUTH + ROLE SYSTEM
📁 server/auth/authMiddleware.ts
import jwt from "jsonwebtoken";

export function requireAuth(req: any, res: any, next: any) {
  const token = req.headers.authorization?.split(" ")[1];

  if (!token) return res.status(401).json({ error: "Unauthorized" });

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET!);
    req.user = decoded;
    next();
  } catch {
    return res.status(401).json({ error: "Invalid token" });
  }
}
📁 server/auth/roles.ts
export function requireRole(role: string) {
  return (req: any, res: any, next: any) => {
    if (req.user.role !== role) {
      return res.status(403).json({ error: "Forbidden" });
    }
    next();
  };
}
