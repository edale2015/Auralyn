2. Event stream (persistent)
// server/core/events/eventStream.ts

import fs from "fs/promises"
import path from "path"
import { ClinicalEvent } from "./eventTypes"

const FILE = path.resolve(
  process.cwd(),
  "server/data/runtime/clinical_events.ndjson"
)

export async function appendEvent(event: ClinicalEvent) {

  await fs.appendFile(
    FILE,
    JSON.stringify(event) + "\n"
  )
}

export async function readEvents(caseId?: string) {

  const raw = await fs.readFile(FILE,"utf8")

  const rows = raw.split("\n")
    .filter(Boolean)
    .map(JSON.parse)

  if(!caseId) return rows

  return rows.filter(r => r.caseId === caseId)
}

This becomes your source of truth.

4. State projector (rebuild state)

Instead of storing state permanently, rebuild it from events.

// server/core/state/clinicalStateProjector.ts

import { readEvents } from "../events/eventStream"
import { ClinicalState } from "./clinicalStateTypes"

export async function rebuildClinicalState(caseId:string){

  const events = await readEvents(caseId)

  const state: ClinicalState = {
    caseId,
    events:[]
  }

  for(const event of events){

    switch(event.type){

      case "PATIENT_MESSAGE":

        state.symptoms =
          (state.symptoms ?? "") +
          " " +
          event.payload.message

      break

      case "COMPLAINT_IDENTIFIED":
        state.complaint = event.payload.complaint
      break

      case "DIFFERENTIAL_UPDATED":
        state.differential = event.payload.differential
      break

      case "ALERTS_UPDATED":
        state.alerts = event.payload.alerts
      break

      case "MEDICATION_PLAN":
        state.medicationPlan = event.payload.medication
      break

      case "DISPOSITION_SET":
        state.disposition = event.payload.disposition
      break

      case "NOTE_READY":
        state.chartNote = event.payload.note
      break

    }

  }

  return state
}

This means any service can reconstruct state.

5. Reasoning worker

Workers process events asynchronously.

// server/core/workers/reasoningWorker.ts

import { rebuildClinicalState } from "../state/clinicalStateProjector"
import { publishEvent } from "../events/eventPublisher"
import { generateDifferential } from "../../assistant/telemedicineDifferentialService"
import { checkSafetyAlerts } from "../../assistant/telemedicineSafetyService"

export async function runReasoningWorker(caseId:string){

  const state = await rebuildClinicalState(caseId)

  if(!state.symptoms) return

  const complaint =
    state.symptoms.includes("cough")
      ? "cough"
      : state.symptoms.includes("throat")
      ? "sore_throat"
      : "unknown"

  await publishEvent(
    caseId,
    "COMPLAINT_IDENTIFIED",
    { complaint }
  )

  const diff = generateDifferential({
    complaint,
    symptoms:state.symptoms
  })

  await publishEvent(
    caseId,
    "DIFFERENTIAL_UPDATED",
    { differential: diff }
  )

  const alerts = checkSafetyAlerts(state.symptoms)

  await publishEvent(
    caseId,
    "ALERTS_UPDATED",
    { alerts }
  )

}

Workers can run on separate machines.
