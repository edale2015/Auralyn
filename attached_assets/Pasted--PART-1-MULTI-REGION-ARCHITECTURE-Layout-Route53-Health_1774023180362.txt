let primary = new pg.Pool({
  connectionString: process.env.DATABASE_URL_PRIMARY
});

let replica = new pg.Pool({
  connectionString: process.env.DATABASE_URL_REPLICA
});

export function getDb(mode = "write") {
  if (mode === "read") return replica;
  return primary;
}

🔁 PART 7 — REQUEST RETRY + REGION FAILOVER
📁 server/utils/multiRegionFetch.ts
export async function fetchWithFailover(urls, options) {
  for (const url of urls) {
    try {
      const res = await fetch(url, options);
      if (res.ok) return res;
    } catch (e) {
      console.warn("Region failed:", url);
    }
  }

  throw new Error("All regions failed");
}
