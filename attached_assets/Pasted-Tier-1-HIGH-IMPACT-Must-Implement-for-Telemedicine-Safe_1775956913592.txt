// server/governance/permissionEngine.ts
export function checkClinicalPermission(action: string, context: any) {
  if (action === "prescribe_antibiotic") {
    if (!context.confirmed_bacterial_features) {
      return { allowed: false, reason: "No bacterial criteria met" };
    }
  }

  if (action === "discharge_patient") {
    if (context.red_flags_present) {
      return { allowed: false, reason: "Red flags require escalation" };
    }
  }

  return { allowed: true };
}

👉 This directly solves:

“Z-pack for viral illness” problem
Over-treatment variability
Liability exposure
