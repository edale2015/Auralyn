1) Types

server/knowledge/knowledgeGraphTypes.ts

export type KnowledgeNodeType =
  | "complaint"
  | "symptom"
  | "question"
  | "skill"
  | "engine"
  | "diagnosis"
  | "protocol"
  | "disposition";

export interface KnowledgeNode {
  id: string;
  type: KnowledgeNodeType;
  label: string;
  metadata?: Record<string, any>;
}

export interface KnowledgeEdge {
  id: string;
  from: string;
  to: string;
  relation:
    | "requires"
    | "asks"
    | "handled_by"
    | "suggests"
    | "governed_by"
    | "can_lead_to"
    | "supports"
    | "escalates_to";
  weight?: number;
  metadata?: Record<string, any>;
}

export interface ClinicalKnowledgeGraph {
  nodes: KnowledgeNode[];
  edges: KnowledgeEdge[];
}

3) Store

server/knowledge/knowledgeGraphStore.ts

import { clinicalKnowledgeGraph } from "./clinicalKnowledgeGraph";
import { ClinicalKnowledgeGraph, KnowledgeEdge, KnowledgeNode } from "./knowledgeGraphTypes";

const graph: ClinicalKnowledgeGraph = {
  nodes: [...clinicalKnowledgeGraph.nodes],
  edges: [...clinicalKnowledgeGraph.edges],
};

export function getKnowledgeGraph(): ClinicalKnowledgeGraph {
  return graph;
}

export function addKnowledgeNode(node: KnowledgeNode) {
  const exists = graph.nodes.some(n => n.id === node.id);
  if (!exists) graph.nodes.push(node);
}

export function addKnowledgeEdge(edge: KnowledgeEdge) {
  const exists = graph.edges.some(e => e.id === edge.id);
  if (!exists) graph.edges.push(edge);
}

export function getNodeById(id: string) {
  return graph.nodes.find(n => n.id === id) || null;
}

export function getEdgesForNode(id: string) {
  return graph.edges.filter(e => e.from === id || e.to === id);
}
4) Query engine

server/knowledge/knowledgeGraphQueryEngine.ts

import { getKnowledgeGraph, getNodeById } from "./knowledgeGraphStore";

export function getNeighborhood(nodeId: string) {
  const graph = getKnowledgeGraph();
  const center = getNodeById(nodeId);
  if (!center) return null;

  const directEdges = graph.edges.filter(e => e.from === nodeId || e.to === nodeId);
  const neighborIds = new Set<string>();
  directEdges.forEach(edge => {
    if (edge.from !== nodeId) neighborIds.add(edge.from);
    if (edge.to !== nodeId) neighborIds.add(edge.to);
  });

  const neighbors = graph.nodes.filter(n => neighborIds.has(n.id));

  return {
    center,
    neighbors,
    edges: directEdges,
  };
}

export function findComplaintPathway(complaintId: string) {
  const graph = getKnowledgeGraph();

  const complaint = graph.nodes.find(n => n.id === complaintId && n.type === "complaint");
  if (!complaint) return null;

  const skills = graph.edges
    .filter(e => e.from === complaintId && e.relation === "requires")
    .map(e => graph.nodes.find(n => n.id === e.to))
    .filter(Boolean);

  const questions = graph.edges
    .filter(e => e.from === complaintId && e.relation === "asks")
    .map(e => graph.nodes.find(n => n.id === e.to))
    .filter(Boolean);

  const protocols = graph.edges
    .filter(e => e.from === complaintId && e.relation === "governed_by")
    .map(e => graph.nodes.find(n => n.id === e.to))
    .filter(Boolean);

  const diagnoses = graph.edges
    .filter(e => e.from === complaintId && e.relation === "suggests")
    .map(e => graph.nodes.find(n => n.id === e.to))
    .filter(Boolean);

  const engineIds = new Set<string>();
  skills.forEach((skill: any) => {
    graph.edges
      .filter(e => e.from === skill.id && e.relation === "handled_by")
      .forEach(e => engineIds.add(e.to));
  });
  protocols.forEach((protocol: any) => {
    graph.edges
      .filter(e => e.from === protocol.id && e.relation === "handled_by")
      .forEach(e => engineIds.add(e.to));
  });

  const engines = graph.nodes.filter(n => engineIds.has(n.id));

  return {
    complaint,
    skills,
    questions,
    protocols,
    engines,
    diagnoses,
  };
}

export function searchKnowledgeGraph(term: string) {
  const graph = getKnowledgeGraph();
  const q = term.trim().toLowerCase();

  return graph.nodes.filter(node => {
    if (node.label.toLowerCase().includes(q)) return true;
    return node.id.toLowerCase().includes(q);
  });
}

import { addKnowledgeEdge, addKnowledgeNode } from "./knowledgeGraphStore";

export function syncComplaintSkillsToGraph(
  complaint: string,
  skills: Array<{ id: string; label: string; priority?: string; engineIds?: string[] }>
) {
  const complaintId = `complaint:${complaint}`;

  addKnowledgeNode({
    id: complaintId,
    type: "complaint",
    label: complaint.replaceAll("_", " "),
  });

  skills.forEach((skill, idx) => {
    const skillId = skill.id.startsWith("skill:") ? skill.id : `skill:${skill.id}`;
    addKnowledgeNode({
      id: skillId,
      type: "skill",
      label: skill.label,
      metadata: { priority: skill.priority || "medium" },
    });

    addKnowledgeEdge({
      id: `sync_req_${complaint}_${idx}`,
      from: complaintId,
      to: skillId,
      relation: "requires",
      weight: 1,
    });

    (skill.engineIds || []).forEach((engineId, engineIdx) => {
      const fullEngineId = engineId.startsWith("engine:") ? engineId : `engine:${engineId}`;
      addKnowledgeNode({
        id: fullEngineId,
        type: "engine",
        label: engineId.replace(/^engine:/, ""),
      });

      addKnowledgeEdge({
        id: `sync_handled_${complaint}_${idx}_${engineIdx}`,
        from: skillId,
        to: fullEngineId,
        relation: "handled_by",
        weight: 0.9,
      });
    });
  });
}
6) API routes

server/routes/knowledgeGraphRoutes.ts

import express from "express";
import { getKnowledgeGraph } from "../knowledge/knowledgeGraphStore";
import {
  findComplaintPathway,
  getNeighborhood,
  searchKnowledgeGraph,
} from "../knowledge/knowledgeGraphQueryEngine";

const router = express.Router();

router.get("/knowledge-graph", (_req, res) => {
  res.json(getKnowledgeGraph());
});

router.get("/knowledge-graph/search", (req, res) => {
  const q = String(req.query.q || "");
  res.json(searchKnowledgeGraph(q));
});

router.get("/knowledge-graph/node/:nodeId", (req, res) => {
  const data = getNeighborhood(req.params.nodeId);
  if (!data) return res.status(404).json({ error: "node_not_found" });
  res.json(data);
});

router.get("/knowledge-graph/pathway/:complaintId", (req, res) => {
  const data = findComplaintPathway(req.params.complaintId);
  if (!data) return res.status(404).json({ error: "complaint_not_found" });
  res.json(data);
});

export default router;

7) Explorer UI

client/components/knowledge/KnowledgeGraphExplorer.tsx

import React, { useEffect, useState } from "react";

export default function KnowledgeGraphExplorer({
  onSelectNode,
}: {
  onSelectNode: (nodeId: string) => void;
}) {
  const [graph, setGraph] = useState<any>(null);
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<any[]>([]);

  useEffect(() => {
    fetch("/api/knowledge-graph")
      .then(r => r.json())
      .then(setGraph);
  }, []);

  async function search() {
    const res = await fetch(`/api/knowledge-graph/search?q=${encodeURIComponent(query)}`);
    setResults(await res.json());
  }

  return (
    <div>
      <h3>Knowledge Graph Explorer</h3>

      <div style={{ display: "flex", gap: 8, marginBottom: 12 }}>
        <input
          value={query}
          onChange={e => setQuery(e.target.value)}
          placeholder="Search complaint, skill, engine, diagnosis..."
        />
        <button onClick={search}>Search</button>
      </div>

      {results.length > 0 && (
        <div style={{ marginBottom: 16 }}>
          <h4>Search Results</h4>
          {results.map((node) => (
            <div
              key={node.id}
              style={{ border: "1px solid #ccc", padding: 8, marginBottom: 8, cursor: "pointer" }}
              onClick={() => onSelectNode(node.id)}
            >
              <b>{node.label}</b> <span>({node.type})</span>
            </div>
          ))}
        </div>
      )}

      {graph && (
        <div>
          <h4>Graph Summary</h4>
          <p>Nodes: {graph.nodes.length}</p>
          <p>Edges: {graph.edges.length}</p>
        </div>
      )}
    </div>
  );
}
8) Node panel

client/components/knowledge/KnowledgeNodePanel.tsx

import React, { useEffect, useState } from "react";

export default function KnowledgeNodePanel({ nodeId }: { nodeId: string }) {
  const [data, setData] = useState<any>(null);

  useEffect(() => {
    if (!nodeId) return;
    fetch(`/api/knowledge-graph/node/${encodeURIComponent(nodeId)}`)
      .then(r => r.json())
      .then(setData);
  }, [nodeId]);

  if (!nodeId) return <div>Select a node.</div>;
  if (!data) return <div>Loading node...</div>;

  return (
    <div>
      <h3>{data.center.label}</h3>
      <p>Type: {data.center.type}</p>

      <h4>Neighbors</h4>
      <ul>
        {data.neighbors.map((n: any) => (
          <li key={n.id}>{n.label} ({n.type})</li>
        ))}
      </ul>

      <h4>Edges</h4>
      <pre>{JSON.stringify(data.edges, null, 2)}</pre>
    </div>
  );
}
