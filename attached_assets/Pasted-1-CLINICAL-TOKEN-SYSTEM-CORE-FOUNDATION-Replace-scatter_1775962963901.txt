export interface ClinicalTokenSet {
  complaint: string;

  // Core reasoning
  posterior: Record<string, number>;
  redFlags: string[];
  modifiers: Record<string, any>;

  // Safety
  riskLevel: "low" | "moderate" | "high" | "critical";
  requiresPhysicianReview: boolean;

  // Output control
  allowedDiagnoses: string[];
  blockedDiagnoses: string[];

  // Audit
  traceId: string;
}

export function createClinicalTokenSet(input: Partial<ClinicalTokenSet>): ClinicalTokenSet {
  return {
    complaint: input.complaint || "unknown",
    posterior: input.posterior || {},
    redFlags: input.redFlags || [],
    modifiers: input.modifiers || {},
    riskLevel: input.riskLevel || "low",
    requiresPhysicianReview: input.requiresPhysicianReview ?? false,
    allowedDiagnoses: [],
    blockedDiagnoses: [],
    traceId: generateTraceId(),
  };
}

function generateTraceId(): string {
  return "TRACE_" + Date.now() + "_" + Math.random().toString(36).slice(2);
}

import { ClinicalTokenSet } from "../core/clinicalTokens";

export function applyDecisionTemperature(tokens: ClinicalTokenSet) {
  const maxProb = Math.max(...Object.values(tokens.posterior));

  if (tokens.redFlags.length > 0) {
    tokens.riskLevel = "critical";
  } else if (maxProb > 0.8) {
    tokens.riskLevel = "high";
  } else if (maxProb > 0.5) {
    tokens.riskLevel = "moderate";
  } else {
    tokens.riskLevel = "low";
  }

  // 🔥 Temperature rules
  switch (tokens.riskLevel) {
    case "critical":
      tokens.requiresPhysicianReview = true;
      tokens.allowedDiagnoses = topDiagnoses(tokens, 1);
      break;

    case "high":
      tokens.allowedDiagnoses = topDiagnoses(tokens, 2);
      break;

    case "moderate":
      tokens.allowedDiagnoses = topDiagnoses(tokens, 3);
      break;

    case "low":
      tokens.allowedDiagnoses = Object.keys(tokens.posterior);
      break;
  }

  return tokens;
}

function topDiagnoses(tokens: ClinicalTokenSet, n: number): string[] {
  return Object.entries(tokens.posterior)
    .sort((a, b) => b[1] - a[1])
    .slice(0, n)
    .map(([dx]) => dx);
}

import { ClinicalTokenSet } from "../core/clinicalTokens";

export function applyShadowSafety(tokens: ClinicalTokenSet) {
  // 🔥 Sepsis example
  if (
    tokens.posterior["sepsis"] > 0.2 ||
    tokens.modifiers["fever"] &&
    tokens.modifiers["tachycardia"]
  ) {
    tokens.redFlags.push("possible_sepsis");
    tokens.riskLevel = "critical";
    tokens.requiresPhysicianReview = true;
  }

  // 🔥 PE override
  if (tokens.posterior["pulmonary_embolism"] > 0.15) {
    tokens.allowedDiagnoses = ["pulmonary_embolism"];
  }

  return tokens;
}

import crypto from "crypto";
import { ClinicalTokenSet } from "../core/clinicalTokens";

export function buildTrace(tokens: ClinicalTokenSet) {
  const trace = {
    id: tokens.traceId,
    timestamp: new Date().toISOString(),
    complaint: tokens.complaint,
    posterior: tokens.posterior,
    redFlags: tokens.redFlags,
    riskLevel: tokens.riskLevel,
    allowedDiagnoses: tokens.allowedDiagnoses,
  };

  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(trace))
    .digest("hex");

  return {
    ...trace,
    hash,
  };
}

import { ClinicalTokenSet } from "../core/clinicalTokens";

export function generateClinicalOutput(tokens: ClinicalTokenSet) {
  if (tokens.requiresPhysicianReview) {
    return {
      disposition: "physician_review_required",
      diagnoses: tokens.allowedDiagnoses,
      message: "Case flagged for physician review due to risk profile.",
    };
  }

  return {
    disposition: determineDisposition(tokens),
    diagnoses: tokens.allowedDiagnoses,
    message: generateMessage(tokens),
  };
}

function determineDisposition(tokens: ClinicalTokenSet): string {
  if (tokens.riskLevel === "critical") return "ER";
  if (tokens.riskLevel === "high") return "urgent_care";
  if (tokens.riskLevel === "moderate") return "follow_up";
  return "home_care";
}

function generateMessage(tokens: ClinicalTokenSet): string {
  return `Most likely diagnosis: ${tokens.allowedDiagnoses[0]}`;
}

  let tokens = createClinicalTokenSet(input);

  tokens = applyShadowSafety(tokens);

  const output = generateClinicalOutput(tokens);

  return {
    trace,
    output,
  };
}
