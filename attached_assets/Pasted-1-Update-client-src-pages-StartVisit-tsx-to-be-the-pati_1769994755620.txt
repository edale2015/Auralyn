import React, { useMemo, useState } from "react";
import { useNavigate } from "react-router-dom";

function extractToken(input: string): string {
  const s = (input || "").trim();
  if (!s) return "";

  // If they pasted a full URL, try to parse it.
  try {
    const url = new URL(s);
    const path = url.pathname || "";
    const m =
      path.match(/\/simple\/([^/]+)/) ||
      path.match(/\/intake\/([^/]+)/);

    if (m?.[1]) return decodeURIComponent(m[1]);
  } catch {
    // Not a URL, continue
  }

  // If they pasted a path like /simple/TOKEN
  const m2 =
    s.match(/\/simple\/([^/]+)/) ||
    s.match(/\/intake\/([^/]+)/);
  if (m2?.[1]) return decodeURIComponent(m2[1]);

  // Otherwise treat the whole thing as a token
  return s;
}

export default function StartVisit() {
  const nav = useNavigate();
  const [raw, setRaw] = useState("");

  const token = useMemo(() => extractToken(raw), [raw]);

  return (
    <div style={styles.page}>
      <div style={styles.card}>
        <h1 style={styles.h1}>Secure Check-in</h1>
        <p style={styles.muted}>
          Paste your secure link token (or paste the full WhatsApp link).
        </p>

        <input
          style={styles.input}
          value={raw}
          onChange={(e) => setRaw(e.target.value)}
          placeholder="Paste token or link"
          autoCapitalize="none"
          autoCorrect="off"
        />

        <button
          style={{ ...styles.button, opacity: token ? 1 : 0.5 }}
          disabled={!token}
          onClick={() => nav(`/simple/${encodeURIComponent(token)}`)}
        >
          Continue
        </button>

        <div style={styles.hr} />

        <p style={styles.smallMuted}>
          If your link expired, reply <b>LINK</b> on WhatsApp to get a new one.
        </p>

        <p style={styles.smallMuted}>
          Clinician? <a href="/" style={styles.link}>Log in</a>
        </p>
      </div>
    </div>
  );
}

function formatTimeLeft(totalSec: number) {
  const s = Math.max(0, totalSec);
  const m = Math.floor(s / 60);
  const r = s % 60;
  return `${m}:${String(r).padStart(2, "0")}`;
}

Countdown effect snippet

Inside SimpleIntake.tsx, after verification succeeds and you set sessionExpiresAtMs:

const [sessionExpiresAtMs, setSessionExpiresAtMs] = useState<number | null>(null);
const [timeLeftSec, setTimeLeftSec] = useState<number>(0);

useEffect(() => {
  if (!sessionExpiresAtMs) return;

  const tick = () => {
    const left = Math.floor((sessionExpiresAtMs - Date.now()) / 1000);
    setTimeLeftSec(left);
  };

  tick();
  const id = window.setInterval(tick, 1000);
  return () => window.clearInterval(id);
}, [sessionExpiresAtMs]);
