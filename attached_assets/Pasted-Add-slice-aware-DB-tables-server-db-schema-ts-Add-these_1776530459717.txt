import path from 'path';

function ensureDir(dir: string) {
  fs.mkdirSync(dir, { recursive: true });
}

function readIfExists(filePath: string): string {
  const abs = path.join(ROOT, filePath);
  if (!fs.existsSync(abs)) return `// FILE NOT FOUND: ${filePath}\n`;
  return fs.readFileSync(abs, 'utf8');
}

function writeSliceFile(exportDir: string, slice: SliceDef): string {
  const outPath = path.join(exportDir, `${slice.id}.md`);
  const parts: string[] = [];

  parts.push(`# ${slice.title}`);
  parts.push('');
  parts.push('## Review Prompt');
  parts.push('');
  parts.push(slice.prompt);
  parts.push('');
  parts.push('## Files');
  parts.push('');

  for (const file of slice.files) {
    const ext = path.extname(file).replace('.', '') || 'txt';
    const content = readIfExists(file);
    parts.push(`### ${file}`);
    parts.push('');
    parts.push('```' + ext);
    parts.push(content);
    parts.push('```');
    parts.push('');
  }

  fs.writeFileSync(outPath, parts.join('\n'));
  return outPath;
}

async function zipDirectory(sourceDir: string, zipPath: string): Promise<void> {
  await new Promise<void>((resolve, reject) => {
    const output = fs.createWriteStream(zipPath);
    const archive = archiver('zip', { zlib: { level: 9 } });
    output.on('close', () => resolve());
    archive.on('error', reject);
    archive.pipe(output);
    archive.directory(sourceDir, false);
    archive.finalize();
  });
}

export async function exportClaudeSlices() {
  const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
  const exportDir = path.join(EXPORT_ROOT, timestamp);
  ensureDir(exportDir);

  const exportedSlices: ExportedSlice[] = [];

  for (const slice of SLICE_DEFS) {
    const outputFile = writeSliceFile(exportDir, slice);

    await db.insert(reviewSlices).values({
      sliceId: slice.id,
      title: slice.title,
      prompt: slice.prompt,
      files: slice.files,
      exportPath: outputFile,
    });

    exportedSlices.push({
      sliceId: slice.id,
      title: slice.title,
      prompt: slice.prompt,
      files: slice.files,
      outputFile,
    });
  }

  const zipPath = path.join(exportDir, 'claude-review-slices.zip');
  await zipDirectory(exportDir, zipPath);

  return {
    exportDir,
    zipPath,
    slices: exportedSlices,
  };
}
4) Store Claude findings per slice
server/routes/claudeSliceRoutes.ts

This lets you upload Claude’s result for one slice at a time.

import express from 'express';
import { db } from '../db/client';
import { reviewSlices, claudeSliceReviews } from '../db/schema';
import { eq } from 'drizzle-orm';

const router = express.Router();

router.post('/submit-findings', async (req, res) => {
  try {
    const { sliceId, claudeFindings } = req.body;

    if (!sliceId || !claudeFindings) {
      return res.status(400).json({ ok: false, error: 'sliceId and claudeFindings are required' });
    }

    const rows = await db.select().from(reviewSlices).where(eq(reviewSlices.sliceId, sliceId));
    const slice = rows[0];

    if (!slice) {
      return res.status(404).json({ ok: false, error: 'Slice not found' });
    }

    const inserted = await db.insert(claudeSliceReviews).values({
      reviewSliceId: slice.id,
      claudeFindings,
      status: 'completed',
    }).returning();

    res.json({ ok: true, review: inserted[0] });
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});

export default router;

5) Send each slice to OpenAI separately
server/research/openaiSliceReview.ts
import fs from 'fs';
import { db } from '../db/client';
import { reviewSlices, claudeSliceReviews, openaiSliceReviews } from '../db/schema';
import { eq } from 'drizzle-orm';
import { runOpenAIReviewStage } from './openaiReviewClient';

export async function reviewClaudeSliceWithOpenAI(sliceId: string) {
  const sliceRows = await db.select().from(reviewSlices).where(eq(reviewSlices.sliceId, sliceId));
  const slice = sliceRows[0];
  if (!slice) throw new Error('Slice not found');

  const claudeRows = await db.select().from(claudeSliceReviews).where(eq(claudeSliceReviews.reviewSliceId, slice.id));
  const latestClaude = claudeRows.at(-1);
  if (!latestClaude) throw new Error('No Claude review found for this slice');

  const relevantCode: Record<string, string> = {};
  for (const file of (slice.files as string[])) {
    try {
      relevantCode[file] = fs.readFileSync(file, 'utf8');
    } catch {
      relevantCode[file] = `// FILE NOT FOUND: ${file}`;
    }
  }

  const review = await runOpenAIReviewStage({
    claudeRecommendations: latestClaude.claudeFindings,
    relevantCode,
    articleSummary: `Slice review for ${slice.title}`,
    systemContext:
      `You are reviewing the Auralyn slice "${slice.title}". Stay within this slice unless a cross-slice dependency is clearly required.`,
  });

  const inserted = await db.insert(openaiSliceReviews).values({
    reviewSliceId: slice.id,
    claudeSliceReviewId: latestClaude.id,
    summaryForUser: review.summaryForUser,
    reviewJson: review,
    overallVerdict: review.overallVerdict,
    status: 'completed',
  }).returning();

  return inserted[0];
}

import { db } from '../db/client';
import { reviewSlices, openaiSliceReviews, sliceProposals } from '../db/schema';
import { eq } from 'drizzle-orm';

export async function buildSliceProposals(sliceId: string) {
  const sliceRows = await db.select().from(reviewSlices).where(eq(reviewSlices.sliceId, sliceId));
  const slice = sliceRows[0];
  if (!slice) throw new Error('Slice not found');

  const reviewRows = await db.select().from(openaiSliceReviews).where(eq(openaiSliceReviews.reviewSliceId, slice.id));
  const latestReview = reviewRows.at(-1);
  if (!latestReview) throw new Error('No OpenAI slice review found');

  const review = latestReview.reviewJson as any;
  const created = [];

  for (const item of review.recommendedUpgrades ?? []) {
    const inserted = await db.insert(sliceProposals).values({
      reviewSliceId: slice.id,
      openaiSliceReviewId: latestReview.id,
      title: item.title,
      rationale: item.rationale,
      affectedFiles: item.affectedFiles,
      patchBundle: {},
      validationPlan: item.validationPlan,
      validationStatus: 'pending',
      approved: false,
      replitStatus: 'pending',
    }).returning();

    created.push(inserted[0]);
  }

  return created;
}
7) Validate one slice at a time
server/research/sliceValidation.ts

This is important: validation should be slice-targeted too.

import { db } from '../db/client';
import { sliceProposals } from '../db/schema';
import { eq } from 'drizzle-orm';
import { runFullValidation } from '../validation/validationRunner';
import { generateFullCaseSet } from '../validation/fullCaseGenerator';

function filterCasesForSlice(title: string, allCases: any[]) {
  const t = title.toLowerCase();

  if (t.includes('diagnosis')) {
    return allCases.filter(c =>
      ['pe', 'acs', 'sepsis', 'stroke'].some(x => (c.id || '').includes(x))
    );
  }

  if (t.includes('disposition') || t.includes('safety')) {
    return allCases.filter(c => c.minimumSafeDisposition);
  }

  return allCases.slice(0, 200);
}

export async function validateSliceProposal(proposalId: number) {
  const rows = await db.select().from(sliceProposals).where(eq(sliceProposals.id, proposalId));
  const proposal = rows[0];
  if (!proposal) throw new Error('Slice proposal not found');

  const allCases = generateFullCaseSet();
  const cases = filterCasesForSlice(proposal.title, allCases);

  const result = await runFullValidation(cases);

  const status =
    result.unsafeUndercalls > 0 ? 'failed' :
    result.passRate < 0.85 ? 'failed' :
    'passed';

  await db.update(sliceProposals)
    .set({ validationStatus: status })
    .where(eq(sliceProposals.id, proposalId));

  return { proposalId, status, result };
}

export async function approveSliceProposal(proposalId: number, approvedBy: string) {
  const rows = await db.select().from(sliceProposals).where(eq(sliceProposals.id, proposalId));
  const proposal = rows[0];
  if (!proposal) throw new Error('Slice proposal not found');

  if (proposal.validationStatus !== 'passed') {
    throw new Error('Cannot approve until slice validation passes');
  }

  await db.update(sliceProposals)
    .set({
      approved: true,
      approvedBy,
    })
    .where(eq(sliceProposals.id, proposalId));

  return { ok: true };
}
9) Export one slice to GitHub, then Replit
server/research/sliceGitHubReplitExport.ts
import { db } from '../db/client';
import { sliceProposals } from '../db/schema';
import { eq } from 'drizzle-orm';
import { Octokit } from '@octokit/rest';

function getOctokit() {
  return new Octokit({ auth: process.env.GITHUB_TOKEN! });
}

function buildReplitSlicePacket(proposal: any) {
  return {
    'research/replit/SLICE_REVIEW_PACKET.md': `# Slice Review Packet

Proposal ID: ${proposal.id}
Title: ${proposal.title}

Rationale:
${proposal.rationale}

Affected Files:
${(proposal.affectedFiles as string[]).map((f: string) => `- ${f}`).join('\n')}

Validation Plan:
${(proposal.validationPlan as string[]).map((x: string) => `- ${x}`).join('\n')}

Constraints:
- Stay within this slice unless a dependency is truly required
- Do not bypass physician review
- Do not weaken hallucination controls
- Keep diff small and testable
`,
    'research/replit/SLICE_IMPLEMENTATION_TASK.md': `Implement this slice only.

1. Independently review the slice
2. Suggest any safer changes
3. Write code only for this slice
4. List risks introduced
5. List what to re-validate
`,
  };
}

export async function exportSliceProposalToGitHubAndReplit(proposalId: number) {
  const rows = await db.select().from(sliceProposals).where(eq(sliceProposals.id, proposalId));
  const proposal = rows[0];
  if (!proposal) throw new Error('Slice proposal not found');
  if (!proposal.approved) throw new Error('Slice proposal must be approved first');

  const octokit = getOctokit();
  const owner = process.env.GITHUB_OWNER!;
  const repo = process.env.GITHUB_REPO!;
  const base = process.env.GITHUB_BASE_BRANCH || 'main';
  const branchName = `slice/${proposal.id}-${Date.now()}`;

  const baseRef = await octokit.git.getRef({
    owner,
    repo,
    ref: `heads/${base}`,
  });

  await octokit.git.createRef({
    owner,
    repo,
    ref: `refs/heads/${branchName}`,
    sha: baseRef.data.object.sha,
  });

  const packet = buildReplitSlicePacket(proposal);

  for (const [filePath, content] of Object.entries(packet)) {
    await octokit.repos.createOrUpdateFileContents({
      owner,
      repo,
      path: filePath,
      message: `Slice ${proposal.id}: add ${filePath}`,
      content: Buffer.from(content, 'utf8').toString('base64'),
      branch: branchName,
    });
  }

  const pr = await octokit.pulls.create({
    owner,
    repo,
    title: `[Slice Upgrade] ${proposal.title}`,
    head: branchName,
    base,
    body: `Slice-based proposal export for proposal ${proposal.id}. Review and implement only this slice.`,
  });

  await db.update(sliceProposals)
    .set({
      githubBranch: branchName,
      githubPrUrl: pr.data.html_url,
      replitStatus: 'ready_for_review',
    })
    .where(eq(sliceProposals.id, proposalId));

  return {
    proposalId,
    branchName,
    prUrl: pr.data.html_url,
    replitInstructions:
      `Open branch ${branchName} in Replit, review research/replit/SLICE_REVIEW_PACKET.md, then ask Replit Agent to independently review and implement only this slice.`,
  };
}

13) Best workflow

Use this order:

Click Export Claude Review Slices
Upload one slice to Claude
Paste Claude findings back into /api/claude-slices/submit-findings
Run /api/slice-pipeline/openai-review/:sliceId
Run /api/slice-pipeline/build-proposals/:sliceId
Run /api/slice-pipeline/validate-proposal/:proposalId
Human approve
Export to GitHub + Replit for that slice
Replit independently reviews and implements that slice
Re-run validation for that slice and then system-wide

That’s tidy, safe, and much easier to audit.
