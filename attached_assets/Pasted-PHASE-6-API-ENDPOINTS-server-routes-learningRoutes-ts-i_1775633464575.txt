export interface Counterfactual {
  variable: string;
  current: any;
  alternative: any;
  effect: string;
  impactScore: number;
}

export interface CounterfactualResult {
  keyFactors: Counterfactual[];
  summary: string;
}

export function computeCounterfactuals(params: {
  result: any;
}): CounterfactualResult {
  const { result } = params;

  const cf: Counterfactual[] = [];

  const triage = result.triage?.level;

  // 🚑 Example: chest pain risk modifiers
  if (result.complaint?.toLowerCase().includes("chest")) {
    cf.push({
      variable: "diaphoresis",
      current: "absent",
      alternative: "present",
      effect: "Would increase likelihood of ACS and elevate triage",
      impactScore: 0.9,
    });

    cf.push({
      variable: "exertional pain",
      current: "unknown",
      alternative: "present",
      effect: "Would increase urgency to high/emergency",
      impactScore: 0.85,
    });
  }

  // 🧪 Infection modifier
  if (!result.safetyAlerts?.length) {
    cf.push({
      variable: "fever",
      current: "unknown",
      alternative: "present",
      effect: "Would increase infectious differential weight",
      impactScore: 0.6,
    });
  }

  // 🫁 Respiratory
  if (!result.triage?.level || result.triage.level !== "emergency") {
    cf.push({
      variable: "shortness_of_breath",
      current: "absent",
      alternative: "present",
      effect: "Would escalate triage to emergency",
      impactScore: 0.95,
    });
  }

  const sorted = cf.sort((a, b) => b.impactScore - a.impactScore);

  return {
    keyFactors: sorted.slice(0, 5),
    summary: sorted.length
      ? `Top factor: ${sorted[0].variable} could significantly change outcome`
      : "No strong counterfactual factors identified",
  };
}
