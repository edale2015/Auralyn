// server/engines/scoring/perc.ts

export function calculatePERC(a: any) {
  const fail =
    a.age > 50 ||
    a.hr > 100 ||
    a.o2 < 95 ||
    a.hemoptysis === "yes" ||
    a.estrogen === "yes" ||
    a.prior_dvt === "yes";

  return {
    negative: !fail,
    recommendation: fail ? "Cannot rule out PE" : "PE unlikely",
  };
}
📦 2. HEART Score (Chest Pain)
export function calculateHEART(a: any) {
  let score = 0;

  if (a.history === "high") score += 2;
  if (a.age > 65) score += 2;
  if (a.risk_factors === "yes") score += 1;

  return {
    score,
    risk: score >= 7 ? "High risk ACS" : "Low/Moderate",
  };
}
📦 3. CHA₂DS₂-VASc (Stroke Risk AFib)
export function calculateCHA2DS2(a: any) {
  let score = 0;
  if (a.age > 75) score += 2;
  if (a.diabetes === "yes") score += 1;
  if (a.htn === "yes") score += 1;

  return {
    score,
    recommendation: score >= 2 ? "Anticoagulate" : "Low risk",
  };
}

export function calculateOttawaAnkle(a: any) {

📦 5. Pediatric Fever Risk
export function calculatePedsFever(a: any) {
  if (a.age < 2 && a.fever === "yes") {
    return { risk: "High", recommendation: "Evaluate urgently" };
  }
  return { risk: "Low" };
}
📦 6. Alvarado Score (Appendicitis)
export function calculateAlvarado(a: any) {
  let score = 0;

  if (a.rlq_pain === "yes") score += 2;
  if (a.fever === "yes") score += 1;
  if (a.nausea === "yes") score += 1;

  return {
    score,
    risk: score >= 7 ? "Likely appendicitis" : "Low",
  };
}
📦 7. TIMI Score (ACS)
export function calculateTIMI(a: any) {
  let score = 0;
  if (a.age > 65) score++;
  if (a.risk_factors === "yes") score++;

  return {
    score,
    risk: score >= 3 ? "High" : "Low",
  };
}
📦 8. GCS (Neuro)
export function calculateGCS(a: any) {
  const score = (a.eye || 4) + (a.verbal || 5) + (a.motor || 6);

  return {
    score,
    severity: score < 8 ? "Severe" : "Mild",
  };
}
📦 9. NEWS2 (Early Warning Score)
export function calculateNEWS2(a: any) {
  let score = 0;

  if (a.rr > 25) score += 2;
  if (a.o2 < 92) score += 2;

  return {
    score,
    risk: score >= 5 ? "High risk deterioration" : "Low",
  };
}
📦 10. CIWA (Alcohol Withdrawal)
export function calculateCIWA(a: any) {
  let score = a.tremor || 0 + a.sweating || 0;

  return {
    score,
    severity: score > 15 ? "Severe withdrawal" : "Mild",
  };
}

📦 Outcome engine
// server/engines/outcomeLearningEngine.ts

const memory: any[] = [];

export function logOutcome(input: any) {
  memory.push(input);
}

export function learn() {
  const insights: any = {};

  for (const m of memory) {
    const key = m.packId;

    insights[key] = insights[key] || {
      correct: 0,
      incorrect: 0,
    };

    if (m.correct) insights[key].correct++;
    else insights[key].incorrect++;
  }

  return insights;
}
📦 Route
// server/routes/outcomeRoutes.ts

import express from "express";
import { logOutcome, learn } from "../engines/outcomeLearningEngine";

const router = express.Router();

router.post("/log", (req, res) => {
  logOutcome(req.body);
  res.json({ ok: true });
});

router.get("/learn", (_req, res) => {
  res.json(learn());
});

export default router;
