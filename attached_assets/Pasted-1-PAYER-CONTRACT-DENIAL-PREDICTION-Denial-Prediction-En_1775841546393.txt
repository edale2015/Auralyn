export function predictDenial(claim: any) {

  let score = 0;

  if (!claim.insurance) score += 3;
  if (!claim.cpt) score += 2;
  if (claim.cpt === "99285" && claim.disposition !== "ER_NOW") score += 2;

  return {
    risk: score > 3 ? "high" : "low",
    probability: score / 10
  };
}

import OpenAI from "openai";

const openai = new OpenAI();

export async function patientChat(input: string) {

  const res = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "You are a medical triage assistant." },
      { role: "user", content: input }
    ]
  });

  return res.choices[0].message.content;
}

export default function PatientApp() {

  const [msg, setMsg] = useState("");
  const [resp, setResp] = useState("");

  async function send() {
    const r = await fetch("/api/patient/chat", {
      method: "POST",
      body: JSON.stringify({ msg })
    });
    setResp(await r.text());
  }

  return (
    <div>
      <input onChange={e=>setMsg(e.target.value)} />
      <button onClick={send}>Send</button>
      <p>{resp}</p>
    </div>
  );
}
