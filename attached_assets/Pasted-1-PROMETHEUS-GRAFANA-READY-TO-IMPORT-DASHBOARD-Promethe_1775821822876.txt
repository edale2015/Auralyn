🔧 OAuth exchange + token
export async function exchangeCodeForToken(code: string) {
  const res = await fetch(`${process.env.EPIC_ISSUER}/token`, {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: new URLSearchParams({
      grant_type: "authorization_code",
      code,
      redirect_uri: process.env.SMART_REDIRECT!,
      client_id: process.env.SMART_CLIENT_ID!
    })
  });

  if (!res.ok) throw new Error("SMART token exchange failed");
  return res.json();
}

export async function getPatientFHIR(id: string, token: string) {
  const res = await fetch(`${process.env.FHIR_BASE}/Patient/${id}`, {
    headers: { Authorization: `Bearer ${token}` }
  });
  if (!res.ok) throw new Error("FHIR patient read failed");

🔧 Write Observation (vitals)
export async function postObservation(patientId: string, token: string, obs: any) {
  const body = {
    resourceType: "Observation",
    status: "final",
    subject: { reference: `Patient/${patientId}` },
    code: { text: obs.code },
    valueQuantity: { value: obs.value, unit: obs.unit }
  };

  const res = await fetch(`${process.env.FHIR_BASE}/Observation`, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${token}`,
      "Content-Type": "application/fhir+json"
    },
    body: JSON.stringify(body)
  });

  if (!res.ok) throw new Error("FHIR observation failed");
  return res.json();
}

from fastapi import FastAPI
import lightgbm as lgb
import numpy as np
import joblib

app = FastAPI()
model = joblib.load("model.pkl")

@app.post("/predict")
def predict(features: dict):
    x = np.array([[features["age"], features["sbp"], features["spo2"], features["hr"], features["rr"], features["chestPain"]]])
    p = float(model.predict(x)[0])
    return {"probability": p}

(3) 🔄 Retry with jitter (network resilience)
export async function retry<T>(fn: () => Promise<T>, tries = 3) {
  let err: any;
  for (let i = 0; i < tries; i++) {
    try { return await fn(); }
    catch (e) {
      err = e;
      await new Promise(r => setTimeout(r, 200 * Math.pow(2, i) + Math.random()*100));
    }
  }
  throw err;
}

(5) 🧠 Explainability (SHAP-like proxy for linear model)
export function explainLinear(weights: any, features: any) {
  const contrib: Record<string, number> = {};
  for (const k of Object.keys(weights)) {
    if (k === "bias") continue;
    contrib[k] = (weights[k] ?? 0) * (features[k] ?? 0);
  }
  return Object.entries(contrib).sort((a,b)=>Math.abs(b[1]) - Math.abs(a[1]));
}
