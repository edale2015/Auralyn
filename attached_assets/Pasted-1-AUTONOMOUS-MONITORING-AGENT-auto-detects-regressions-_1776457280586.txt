import { detectRegressions } from "../validation/regressionTracker";
import { buildRegressionAlerts } from "../validation/regressionAlerts";

let lastSnapshot: any = null;

export async function monitoringAgent(currentSnapshot: any) {
  if (!lastSnapshot) {
    lastSnapshot = currentSnapshot;
    return;
  }

  const regression = detectRegressions(lastSnapshot, currentSnapshot);
  const alerts = buildRegressionAlerts(regression);

  if (alerts.length > 0) {
    console.log("🚨 REGRESSION DETECTED:", alerts);

    // HARD SAFETY BLOCK
    if (regression.unsafeIncrease) {
      global.SYSTEM_MODE = "SAFE_MODE";
      console.log("🛑 SYSTEM SWITCHED TO SAFE MODE");
    }
  }

  lastSnapshot = currentSnapshot;
}

import { broadcastPatientUpdate } from "../ws/patientStream";

function generateVitals() {
  return {
    hr: 60 + Math.random() * 80,
    bp: 90 + Math.random() * 60,
    spo2: 90 + Math.random() * 10,
    temp: 36 + Math.random() * 3,
  };
}

export function startICUSimulator() {
  setInterval(() => {
    for (let i = 0; i < 50; i++) {
      const patient = {
        id: `ICU_${i}`,
        vitals: generateVitals(),
        timestamp: Date.now(),
      };

      broadcastPatientUpdate(patient);
    }
  }, 1000);
}

export async function fhirWrite(resource: any) {
  return fetch(process.env.FHIR_ENDPOINT!, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.FHIR_TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify(resource)
  });
}
