export function runGlobalOrchestration(input: any) {

  const continents = groupBy(input.regions, r => r.continent);

  const globalSignals = [];

  for (const [continent, regions] of Object.entries(continents)) {
    const volume = regions.reduce((s: number, r: any) => s + r.summary.totalPatients, 0);

    globalSignals.push({
      continent,
      volume,
      trend: computeTrend(regions)
    });
  }

  return {
    globalSignals,
    recommendedRedistribution: findUnderloadedRegions(input.regions)
  };
}

function groupBy(arr: any[], fn: any) {
  return arr.reduce((acc, item) => {
    const key = fn(item);
    if (!acc[key]) acc[key] = [];
    acc[key].push(item);
    return acc;
  }, {});
}

function computeTrend(regions: any[]) {
  return regions.some(r => r.surgeState.status === "critical")
    ? "spiking"
    : "stable";
}

function findUnderloadedRegions(regions: any[]) {
  return regions
    .filter(r => r.capacityState.systemState === "stable")
    .map(r => r.regionName);
}

export function detectPandemicSignals(input: any) {

  const symptomCounts: Record<string, number> = {};

  for (const p of input.patients) {
    for (const s of p.symptoms) {
      symptomCounts[s] = (symptomCounts[s] || 0) + 1;
    }
  }

  const clusters = Object.entries(symptomCounts)
    .filter(([_, count]) => count > 100);

  const respiratoryCluster =
    symptomCounts["cough"] > 200 &&
    symptomCounts["fever"] > 200;

  const giCluster =
    symptomCounts["vomiting"] > 150 &&
    symptomCounts["diarrhea"] > 150;

  return {
    clusters,
    respiratoryCluster,
    giCluster,
    alert: respiratoryCluster || giCluster
  };
}

export function simulateSpread(input: any) {

  const R0 = input.R0 ?? 1.5;
  const population = input.population ?? 100000;
  const infected = input.initialInfected ?? 10;

  const nextDay =
    infected * R0;

  return {
    current: infected,
    nextDay,
    nextWeek: nextDay * 7,
    riskLevel:
      nextDay > 500 ? "high" :
      nextDay > 100 ? "medium" : "low"
  };
}

export function earlyWarningSystem(input: any) {

  if (input.respiratoryCluster && input.trend === "spiking") {
    return {
      alert: "Respiratory outbreak emerging",
      action: "Increase telemed + reduce ER overload"
    };
  }

  if (input.giCluster) {
    return {
      alert: "GI outbreak detected",
      action: "Public health notification"
    };
  }

  return { alert: null };
}
