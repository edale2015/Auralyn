Create File
server/core/clinicalMemoryEngine.ts
import fs from "fs";
import path from "path";

const MEMORY_PATH = path.join(process.cwd(), "clinical_memory.ndjson");

export interface ClinicalCase {

  complaint: string

  answers: Record<string, any>

  predictedDifferentials: any[]

  predictedDisposition: string

  finalDiagnosis?: string

  finalDisposition?: string

  outcomeKnown?: boolean

  timestamp: string
}

export function storeClinicalCase(caseData: ClinicalCase) {

  const line = JSON.stringify(caseData)

  fs.appendFileSync(MEMORY_PATH, line + "\n")
}

export function loadClinicalMemory(): ClinicalCase[] {

  if (!fs.existsSync(MEMORY_PATH)) return []

  const lines = fs.readFileSync(MEMORY_PATH, "utf8").split("\n")

  return lines
    .filter(Boolean)
    .map(l => JSON.parse(l))
}

export function findSimilarMemoryCases(
  complaint: string,
  answers: Record<string, any>,
  limit = 5
) {

  const cases = loadClinicalMemory()

  const scored = cases.map(c => {

    let score = 0

    if (c.complaint === complaint) score += 2

    for (const k of Object.keys(answers)) {

      if (answers[k] === c.answers[k]) score += 1
    }

    return { case: c, score }
  })

  return scored
    .sort((a,b)=>b.score-a.score)
    .slice(0,limit)
}

import { loadClinicalMemory } from "./clinicalMemoryEngine"

export function generateLearningReport() {

  const cases = loadClinicalMemory()

  let correct = 0

  let total = 0

  for (const c of cases) {

    if (!c.outcomeKnown) continue

    total++

    if (c.finalDisposition === c.predictedDisposition) {
      correct++
    }
  }

  return {

    totalCases: total,

    accuracy: total ? correct / total : 0
  }
}
