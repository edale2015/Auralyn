import { calculateCentorScore, centorDecision } from "./centorEngine";
import { calculateStrepProbability } from "./bayesianStrepEngine";
import { calculateConfidence } from "./confidenceEngine";

export interface ClinicalInput {
  fever: boolean;
  tonsillarExudate: boolean;
  tenderAnteriorCervicalNodes: boolean;
  absenceOfCough: boolean;
  age: number;
}

export interface ClinicalOutput {
  centorScore: number;
  centorRecommendation: string;
  probability: number;
  finalDecision: string;
  confidence: string;
  reasoning: string[];
}

export function runClinicalDecision(input: ClinicalInput): ClinicalOutput {
  const reasoning: string[] = [];

  // 🧮 Centor
  const centorScore = calculateCentorScore(input);
  const centorRecommendation = centorDecision(centorScore);

  reasoning.push(`Centor score: ${centorScore}`);

  // 📊 Bayesian
  const probability = calculateStrepProbability({
    fever: input.fever,
    exudate: input.tonsillarExudate,
    nodes: input.tenderAnteriorCervicalNodes,
    cough: !input.absenceOfCough
  });

  reasoning.push(`Estimated bacterial probability: ${probability.toFixed(2)}`);

  // 🧠 Final decision logic (override layer)
  let finalDecision = "NO_ANTIBIOTIC";

  if (centorScore >= 4 || probability > 0.65) {
    finalDecision = "ANTIBIOTIC";
    reasoning.push("High clinical likelihood → treat");
  } else if (centorScore >= 2 || probability > 0.4) {
    finalDecision = "TEST_OR_DELAYED";
    reasoning.push("Intermediate risk → test or delayed Rx");
  } else {
    reasoning.push("Low likelihood → no antibiotic");
  }

  const confidence = calculateConfidence(probability);

  return {
    centorScore,
    centorRecommendation,
    probability,
    finalDecision,
    confidence,
    reasoning
  };
}

import twilio from "twilio";

const client = twilio(
  process.env.TWILIO_SID,
  process.env.TWILIO_AUTH
);

// 🔒 Basic PHI scrub (extend as needed)
function scrubText(text: string): string {
  return text.replace(/\b\d{3}-\d{2}-\d{4}\b/g, "[REDACTED]");
}

export async function speakToPatient({
  phone,
  message
}: {
  phone: string;
  message: string;
}) {
  const safeMessage = scrubText(message);

  return client.calls.create({
    to: phone,
    from: process.env.TWILIO_NUMBER,
    twiml: `<Response><Say voice="Polly.Joanna">${safeMessage}</Say></Response>`
  });
}

// 🎯 Helper: generate clinical explanation voice
export function generateVoiceMessage(decision: string): string {
  if (decision === "ANTIBIOTIC") {
    return "Your symptoms suggest a bacterial infection. Antibiotics are appropriate.";
  }

  if (decision === "TEST_OR_DELAYED") {
    return "Your symptoms are in a middle range. Testing or delayed treatment is appropriate.";
  }

  return "Your symptoms suggest a viral process. Antibiotics are not likely to help.";
}

export interface DebateInput {
  centorScore: number;
  probability: number;
}

export interface DebateOutput {
  decision: string;
  confidence: string;
  proArguments: string[];
  conArguments: string[];
  summary: string;
}

export function runAntibioticDebate(input: DebateInput): DebateOutput {
  const pro: string[] = [];
  const con: string[] = [];

  let proWeight = 0;
  let conWeight = 0;

  // 🟢 Pro-antibiotic agent
  if (input.centorScore >= 4) {
    pro.push("High Centor score");
    proWeight += 2;
  }

  if (input.probability > 0.65) {
    pro.push("High bacterial probability");
    proWeight += 2;
  }

  // 🔴 Conservative agent
  if (input.centorScore <= 2) {
    con.push("Low Centor score");
    conWeight += 2;
  }

  if (input.probability < 0.3) {
    con.push("Low bacterial probability");
    conWeight += 2;
  }

  // ⚖️ Decision logic
  let decision = "NO_ANTIBIOTIC";
  let confidence = "LOW";

  if (proWeight > conWeight) {
    decision = "ANTIBIOTIC";
  } else if (proWeight === conWeight) {
    decision = "TEST_OR_DELAYED";
  }

  // 🎯 Confidence
  const delta = Math.abs(proWeight - conWeight);

  if (delta >= 2) confidence = "HIGH";
  else if (delta === 1) confidence = "MEDIUM";

  return {
    decision,
    confidence,
    proArguments: pro,
    conArguments: con,
    summary: `Pro: ${pro.join(", ")} | Con: ${con.join(", ")}`
  };
}
🌐 4) UNIFIED ROUTE (DASHBOARD-COMPATIBLE)

👉 This is the single endpoint your dashboards call

/server/routes/fullClinicalPipeline.ts
import express from "express";
import { runClinicalDecision } from "../services/clinical/clinicalDecisionEngine";
import { runAntibioticDebate } from "../services/communication/debateEngine";
import { speakToPatient, generateVoiceMessage } from "../services/communication/voiceService";

const router = express.Router();

router.post("/full-pipeline", async (req, res) => {
  try {
    const { symptoms, phone } = req.body;

    // 🧠 Core clinical decision
    const clinical = runClinicalDecision(symptoms);

    // 🧠 Debate layer
    const debate = runAntibioticDebate({
      centorScore: clinical.centorScore,
      probability: clinical.probability
    });

    // 🎤 Voice output
    if (phone) {
      const message = generateVoiceMessage(debate.decision);
      await speakToPatient({ phone, message });
    }

    res.json({
      clinical,
      debate,
      finalDecision: debate.decision
    });

  } catch (err) {
    res.status(500).json({ error: "Pipeline failed" });
  }
});

export default router;
