import { db } from '../db/client';


export async function getValidationDashboard() {
  const rows = await db.execute(`SELECT * FROM validation_runs ORDER BY created_at DESC LIMIT 50`);


  const summaries = rows.rows.map((r: any) => r.summary);


  const latest = summaries[0] || {};


  const unsafeTrend = summaries.map((s: any) => s.unsafeUndercalls);


  return {
    latest,
    unsafeTrend,
    passRates: summaries.map((s: any) => s.passRate),
    timestamps: rows.rows.map((r: any) => r.created_at),
  };
}

// server/controlTower/calibrationService.ts


export function calibrationByComplaint(results: any[]) {
  const map: Record<string, any[]> = {};


  for (const r of results) {
    if (!map[r.complaint]) map[r.complaint] = [];


    map[r.complaint].push({
      confidence: r.confidence,
      correct: r.correct,
    });
  }


  const output: any = {};


  for (const complaint in map) {
    const rows = map[complaint];


    const avgConf = rows.reduce((a, b) => a + b.confidence, 0) / rows.length;
    const accuracy = rows.filter(r => r.correct).length / rows.length;


    output[complaint] = {
      avgConfidence: avgConf,
      accuracy,
      gap: avgConf - accuracy,
    };
  }


  return output;
}

4) RLHF SAFE TUNING LOOP
Bounded Update Engine
// server/rlhf/rlhfEngine.ts


export function boundedUpdate(oldWeight: number, delta: number) {
  const MAX_DELTA = 0.02; // 2% max change


  const clipped = Math.max(-MAX_DELTA, Math.min(MAX_DELTA, delta));


  return oldWeight + clipped;
}
RLHF Trainer
// server/rlhf/trainer.ts


import { boundedUpdate } from './rlhfEngine';


export function trainFromOutcomes(model: any, outcomes: any[]) {
  const updates: any = {};


  for (const o of outcomes) {
    if (!updates[o.feature]) updates[o.feature] = 0;


    updates[o.feature] += o.correct ? 0.01 : -0.01;
  }


  const newModel: any = { ...model };


  for (const f in updates) {
    newModel[f] = boundedUpdate(model[f] || 0, updates[f]);
  }


  return newModel;
}
Physician Approval Gate
// server/rlhf/approval.ts


export function requireApproval(proposedModel: any, currentModel: any) {
  const changes = [];


  for (const k in proposedModel) {
    const diff = Math.abs((proposedModel[k] || 0) - (currentModel[k] || 0));
    if (diff > 0.01) {
      changes.push({ feature: k, diff });
    }
  }


  return {
    requiresApproval: changes.length > 0,
    changes,
  };
}
