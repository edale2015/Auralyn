client/src/utils/routeValidator.ts
import { ROUTES } from "../routes/routeRegistry";

export function validateRoutes() {
  const paths = Object.values(ROUTES);
  const unique = new Set(paths);

  if (unique.size !== paths.length) {
    throw new Error("❌ Duplicate routes detected");
  }
}

Call in App.tsx:

validateRoutes();

type Props = {
  status: "active" | "degraded" | "stub";
};

export default function EngineBadge({ status }: Props) {
  const styles = {
    active: "bg-green-100 text-green-700",
    degraded: "bg-yellow-100 text-yellow-700",
    stub: "bg-gray-100 text-gray-500"
  };

  return (
    <span className={`px-2 py-1 rounded text-xs ${styles[status]}`}>
      {status}
    </span>
  );
}
