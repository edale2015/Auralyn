1️⃣ Decision Replay Engine

This reconstructs exactly how a case decision happened.

server/engines/decisionReplayEngine.ts
export interface ReplayStep {
  engine: string
  input: any
  output: any
  timestamp: number
}

export interface CaseReplay {
  caseId: string
  complaint: string
  steps: ReplayStep[]
}

export class DecisionReplayEngine {

  buildReplay(caseRecord: any): CaseReplay {

    const steps: ReplayStep[] = []

    if (caseRecord.redFlagResult) {
      steps.push({
        engine: "Red Flag Safety Engine",
        input: caseRecord.answers,
        output: caseRecord.redFlagResult,
        timestamp: Date.now()
      })
    }

    if (caseRecord.similarityMatches) {
      steps.push({
        engine: "Case Similarity Engine",
        input: caseRecord.features,
        output: caseRecord.similarityMatches,
        timestamp: Date.now()
      })
    }

    if (caseRecord.bayesScores) {
      steps.push({
        engine: "Bayesian Differential Engine",
        input: caseRecord.features,
        output: caseRecord.bayesScores,
        timestamp: Date.now()
      })
    }

    if (caseRecord.finalDisposition) {
      steps.push({
        engine: "Disposition Ensemble Engine",
        input: caseRecord.bayesScores,
        output: caseRecord.finalDisposition,
        timestamp: Date.now()
      })
    }

    return {
      caseId: caseRecord.id,
      complaint: caseRecord.complaint,
      steps
    }
  }

}
2️⃣ Compact Question Composer Engine

This is extremely important for your Telegram / WhatsApp workflows.

It converts one complaint question set into multiple messaging formats automatically.

server/engines/compactQuestionComposer.ts
export interface Question {
  id: string
  text: string
  type: "yesno" | "number" | "text" | "multiple"
  options?: string[]
}

export interface QuestionBundle {
  complaint: string
  questions: Question[]
}

export class CompactQuestionComposer {

  toTelegram(bundle: QuestionBundle) {
    return bundle.questions.map(q => ({
      text: q.text,
      reply_markup: {
        keyboard: q.options || [["Yes", "No"]],
        resize_keyboard: true
      }
    }))
  }

  toTelegramMiniApp(bundle: QuestionBundle) {
    return bundle.questions.map(q => ({
      id: q.id,
      label: q.text,
      type: q.type,
      options: q.options
    }))
  }

  toWhatsApp(bundle: QuestionBundle) {
    return bundle.questions.map(q => ({
      type: "interactive",
      body: {
        text: q.text
      },
      action: {
        buttons: q.options?.map(o => ({ title: o }))
      }
    }))
  }

  toWhatsAppFlow(bundle: QuestionBundle) {
    return bundle.questions.map(q => ({
      name: q.id,
      question: q.text,
      answers: q.options || []
    }))
  }

}

6️⃣ Decision Replay API Route
server/routes/decisionReplayRoutes.ts
import express from "express"
import { DecisionReplayEngine } from "../engines/decisionReplayEngine"
import { getCase } from "../services/caseStore"

const router = express.Router()
const engine = new DecisionReplayEngine()

router.get("/replay/:caseId", (req, res) => {

  const caseRecord = getCase(req.params.caseId)

  if (!caseRecord) {
    return res.status(404).send("Case not found")
  }

  const replay = engine.buildReplay(caseRecord)

  res.json(replay)

})

export default router

8️⃣ Decision Replay Tab
client/components/DecisionReplayTab.tsx
import React, { useState } from "react"
import ReplayTimeline from "./ReplayTimeline"

export default function DecisionReplayTab() {

  const [caseId,setCaseId] = useState("")
  const [replay,setReplay] = useState<any>(null)

  async function loadReplay(){

    const res = await fetch(`/api/replay/${caseId}`)
    const data = await res.json()

    setReplay(data)
  }

  return (

    <div>

      <h2>Decision Replay</h2>

      <input
        value={caseId}
        onChange={e => setCaseId(e.target.value)}
        placeholder="Case ID"
      />

      <button onClick={loadReplay}>
        Load Replay
      </button>

      {replay && <ReplayTimeline replay={replay} />}

    </div>

  )
}
