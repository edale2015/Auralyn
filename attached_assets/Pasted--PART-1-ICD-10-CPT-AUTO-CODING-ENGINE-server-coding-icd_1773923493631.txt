🔥 PART 1 — ICD-10 + CPT AUTO-CODING ENGINE
📁 server/coding/icdCptEngine.ts
type DiagnosisCluster = {
  id: string;
  name: string;
  likelihood: number;
};

type ICDCode = {
  code: string;
  description: string;
};

type CPTCode = {
  code: string;
  description: string;
};

type CodingResult = {
  primaryICD: ICDCode;
  secondaryICDs: ICDCode[];
  cptCodes: CPTCode[];
};

export class ICDCPTAutoCoder {
  private icdMap: Record<string, ICDCode[]> = {
    "pneumonia": [
      { code: "J18.9", description: "Pneumonia, unspecified organism" }
    ],
    "asthma_exacerbation": [
      { code: "J45.901", description: "Unspecified asthma with exacerbation" }
    ],
    "uri": [
      { code: "J06.9", description: "Acute upper respiratory infection" }
    ]
  };

  private cptMap: Record<string, CPTCode[]> = {
    "telemed_basic": [
      { code: "99213", description: "Established patient visit" }
    ],
    "telemed_complex": [
      { code: "99214", description: "Moderate complexity visit" }
    ],
    "critical": [
      { code: "99285", description: "Emergency high severity visit" }
    ]
  };

  mapDiagnosisToICD(cluster: DiagnosisCluster): ICDCode[] {
    return this.icdMap[cluster.name] || [
      { code: "R69", description: "Unknown illness" }
    ];
  }

  determineCPT(disposition: string, complexityScore: number): CPTCode[] {
    if (disposition === "er_now") return this.cptMap["critical"];
    if (complexityScore > 0.7) return this.cptMap["telemed_complex"];
    return this.cptMap["telemed_basic"];
  }

  generateCoding(
    clusters: DiagnosisCluster[],
    disposition: string,
    complexityScore: number
  ): CodingResult {
    const sorted = clusters.sort((a, b) => b.likelihood - a.likelihood);

    const primary = this.mapDiagnosisToICD(sorted[0])[0];

    const secondary = sorted.slice(1, 3).flatMap(c =>
      this.mapDiagnosisToICD(c)
    );

    const cpt = this.determineCPT(disposition, complexityScore);

    return {
      primaryICD: primary,
      secondaryICDs: secondary,
      cptCodes: cpt
    };
  }
}

🔥 PART 3 — UNIFIED ENCOUNTER BUNDLE (EHR + BILLING + AUDIT)

This is the money engine + compliance engine.

📁 server/encounter/encounterBuilder.ts
type Patient = {
  id: string;
  age: number;
  sex: string;
};

type ClinicalOutput = {
  clusters: any[];
  disposition: string;
  reasoning: string;
};

type AuditLog = {
  steps: string[];
  timestamp: string;
};

export class EncounterBuilder {
  buildEncounter(
    patient: Patient,
    clinical: ClinicalOutput,
    coding: any,
    audit: AuditLog
  ) {
    return {
      encounter_id: `enc_${Date.now()}`,
      patient,

      clinical: {
        diagnosis_clusters: clinical.clusters,
        disposition: clinical.disposition,
        reasoning: clinical.reasoning
      },

      coding: {
        icd10: [
          coding.primaryICD,
          ...coding.secondaryICDs
        ],
        cpt: coding.cptCodes
      },

      billing: {
        status: "ready_for_submission",
        estimated_revenue: this.estimateRevenue(coding.cptCodes)
      },

      audit,

      fhir_export: this.toFHIR(patient, clinical, coding)
    };
  }

  estimateRevenue(cptCodes: any[]) {
    const pricing: Record<string, number> = {
      "99213": 75,
      "99214": 110,
      "99285": 400
    };

    return cptCodes.reduce((sum, cpt) => {
      return sum + (pricing[cpt.code] || 50);
    }, 0);
  }

  toFHIR(patient: Patient, clinical: any, coding: any) {
    return {
      resourceType: "Bundle",
      type: "collection",
      entry: [
        {
          resource: {
            resourceType: "Patient",
            id: patient.id,
            gender: patient.sex,
            birthDate: `${new Date().getFullYear() - patient.age}`
          }
        },
        {
          resource: {
            resourceType: "Condition",
            code: {
              coding: coding.icd10.map((c: any) => ({
                code: c.code,
                display: c.description
              }))
            }
          }
        }
      ]
    };
  }
}
🔗 PART 4 — CONNECT EVERYTHING (MAIN PIPELINE)
📁 server/pipeline/fullEncounterPipeline.ts
import { ICDCPTAutoCoder } from "../coding/icdCptEngine";
import { EncounterBuilder } from "../encounter/encounterBuilder";
import { FDANarrativeEngine } from "../fda/fdaNarrativeEngine";

export async function runFullEncounterPipeline(input: any) {
  const coder = new ICDCPTAutoCoder();
  const builder = new EncounterBuilder();
  const fda = new FDANarrativeEngine();

  const coding = coder.generateCoding(
    input.clusters,
    input.disposition,
    input.complexityScore
  );

  const encounter = builder.buildEncounter(
    input.patient,
    input.clinical,
    coding,
    input.audit
  );

  const fdaNarrative = fda.generateNarrative(input.metrics);

  return {
    encounter,
    fdaNarrative
  };
}
