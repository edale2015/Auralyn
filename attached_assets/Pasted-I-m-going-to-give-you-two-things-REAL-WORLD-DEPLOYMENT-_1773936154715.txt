📊 2. PHYSICIAN DISAGREEMENT TRACKER (CRITICAL)

This is how you improve safely.

/server/learning/disagreement.ts
let disagreements: any[] = [];

export function logDisagreement(ai: any, physician: any) {
  disagreements.push({
    ai,
    physician,
    timestamp: new Date().toISOString(),
  });
}

export function analyzeDisagreements() {
  const summary: Record<string, number> = {};

  disagreements.forEach(d => {
    const key = `${d.ai.dx}→${d.physician.dx}`;
    summary[key] = (summary[key] || 0) + 1;
  });

  return summary;
}

let trustScores: Record<string, number> = {};

export function updateTrust(complaint: string, success: boolean) {
  if (!trustScores[complaint]) trustScores[complaint] = 0.5;

  trustScores[complaint] += success ? 0.05 : -0.1;
}

export function canAutoHandle(complaint: string) {
  return trustScores[complaint] > 0.85;
}
