// server/engines/consensusEngine.ts

type AgentOpinion = {
  agent: string;
  diagnosis: string;
  confidence: number;
  reasoning?: string;
};

const AGENT_WEIGHTS = {
  infectious: 1.2,
  pulmonary: 1.1,
  cardiology: 1.3,
  general: 1.0,
};

export function runConsensus(opinions: AgentOpinion[]) {
  const scores: Record<string, number> = {};

  for (const op of opinions) {
    const weight = AGENT_WEIGHTS[op.agent] || 1.0;
    const score = op.confidence * weight;

    if (!scores[op.diagnosis]) scores[op.diagnosis] = 0;
    scores[op.diagnosis] += score;
  }

  const ranked = Object.entries(scores)
    .sort((a, b) => b[1] - a[1])
    .map(([diagnosis, score]) => ({ diagnosis, score }));

  return {
    topDiagnosis: ranked[0]?.diagnosis,
    ranked,
    rawOpinions: opinions,
  };
}

// server/agent/clinicalConsensusOrchestrator.ts

import { runSubagent } from "./subagent";
import { runConsensus } from "../engines/consensusEngine";

export async function runClinicalConsensus(patientData: any) {
  const prompts = [
    { agent: "infectious", prompt: `Evaluate infectious causes: ${JSON.stringify(patientData)}` },
    { agent: "pulmonary", prompt: `Evaluate pulmonary causes: ${JSON.stringify(patientData)}` },
    { agent: "cardiology", prompt: `Evaluate cardiac causes: ${JSON.stringify(patientData)}` },
  ];

  const results = await Promise.all(
    prompts.map(async (p) => {
      const res = await runSubagent(p.prompt);

      return {
        agent: p.agent,
        diagnosis: res.diagnosis,
        confidence: res.confidence || 0.7,
      };
    })
  );

  return runConsensus(results);
}

// server/engines/nextBestQuestion.ts

type Question = {
  question: string;
  affects: string[]; // diagnoses
  weight: number;
};

export function getNextBestQuestion(
  differential: { diagnosis: string; probability: number }[],
  questions: Question[]
) {
  let bestQ = null;
  let bestScore = -Infinity;

  for (const q of questions) {
    let infoGain = 0;

    for (const dx of differential) {
      if (q.affects.includes(dx.diagnosis)) {
        infoGain += dx.probability * q.weight;
      }
    }

    if (infoGain > bestScore) {
      bestScore = infoGain;
      bestQ = q;
    }
  }

  return {
    question: bestQ?.question,
    expectedImpact: bestScore,
  };
}

// server/engines/dispositionGuardrail.ts

export function applyDispositionGuardrail(input: {
  diagnosis: string;
  riskScore: number;
  redFlags: string[];
  llmDisposition: string;
}) {
  // 🚨 HARD OVERRIDES
  if (input.redFlags.length > 0) {
    return {
      finalDisposition: "ER",
      reason: "Red flags present",
      override: true,
    };
  }

  if (input.riskScore > 0.8) {
    return {
      finalDisposition: "ER",
      reason: "High risk score",
      override: true,
    };
  }

  if (input.riskScore > 0.5) {
    return {
      finalDisposition: "Urgent Care",
      reason: "Moderate risk",
      override: true,
    };
  }

  return {
    finalDisposition: input.llmDisposition,
    override: false,
  };
}
