type State = {
  riskScore: number;
  sepsisProb: number;
  vitals: any;
};

type Action =
  | "observe"
  | "order_labs"
  | "give_fluids"
  | "escalate_ICU"
  | "transfer_hospital";

type Outcome = {
  icu: boolean;
  mortality: boolean;
  losHours: number;
};

type Experience = {
  state: State;
  action: Action;
  reward: number;
};

const Q: Record<string, number> = {};

function getKey(state: State, action: Action) {
  return JSON.stringify(state) + "|" + action;
}

export function computeReward(outcome: Outcome) {
  let reward = 0;

  if (!outcome.mortality) reward += 100;
  if (!outcome.icu) reward += 20;
  reward -= outcome.losHours * 0.5;

  return reward;
}

export function updateQ(state: State, action: Action, reward: number) {
  const key = getKey(state, action);
  const alpha = 0.1;

  Q[key] = (Q[key] || 0) + alpha * (reward - (Q[key] || 0));
}

export function chooseBestAction(state: State, possible: Action[]) {
  return possible.sort(
    (a, b) => (Q[getKey(state, b)] || 0) - (Q[getKey(state, a)] || 0)
  )[0];
}

export function optimizeHospitalFlow(patients: any[], beds: any[]) {
  const load = patients.length;
  const availableBeds = beds.filter(b => b.available).length;

  let strategy = "NORMAL";

  if (load > 50 && availableBeds < 5) {
    strategy = "DIVERT";
  }

  if (load > 70) {
    strategy = "CRITICAL_OVERLOAD";
  }

  return {
    load,
    availableBeds,
    strategy,
    actions: getActions(strategy)
  };
}

function getActions(strategy: string) {
  switch (strategy) {
    case "DIVERT":
      return ["redirect_low_risk", "accelerate_discharges"];

    case "CRITICAL_OVERLOAD":
      return ["block_new_admissions", "activate_overflow_units"];

    default:
      return ["standard_flow"];
  }
}

import { optimizeHospitalFlow } from "../ops/hospitalOptimizer";

export async function runGlobalBrain(patients: any[], beds: any[], hospitals: any[], emsCalls: any[]) {
  const system = await runSystemCycle(patients, beds, hospitals);

  const ops = optimizeHospitalFlow(patients, beds);

  const emsRouting = emsCalls.map(call =>
    routeEMS(call, hospitals)
  );

  const snapshot = {
    system,
    ops,
    emsRouting
  };

  broadcast("GLOBAL_BRAIN_UPDATE", snapshot);

  return snapshot;
}

import { useEffect, useState } from "react";

export default function GlobalCommand() {
  const [data, setData] = useState<any>(null);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:3000/ws");

    ws.onmessage = (msg) => {
      const parsed = JSON.parse(msg.data);
      if (parsed.type === "GLOBAL_BRAIN_UPDATE") {
        setData(parsed.payload);
      }
    };
  }, []);

  if (!data) return <div>Loading...</div>;

  return (
    <div className="p-4 grid grid-cols-3 gap-4">
      <div>
        <h2>⚙️ Hospital Ops</h2>
        <div>Strategy: {data.ops.strategy}</div>
      </div>

      <div>
        <h2>🚑 EMS Routing</h2>
        {data.emsRouting.map((r: any, i: number) => (
          <div key={i}>
            EMS → Hospital {r.assignedHospital}
          </div>
        ))}
      </div>

      <div>
        <h2>🧠 System Status</h2>
        <div>Patients: {data.system.twins.length}</div>
      </div>
    </div>
  );
}
