A) Learning signal extraction

Convert failures into structured signals:

// server/learning/learningSignals.ts

export interface LearningSignal {
  caseId: string;
  failureType:
    | "wrong_disposition"
    | "missed_diagnosis"
    | "unsafe_pass"
    | "over_escalation"
    | "uncertainty_misclassification";

  expected: any;
  actual: any;

  context: {
    complaint: string;
    symptoms: string[];
    posterior?: any;
    decision?: any;
  };
}

export function extractLearningSignals(
  test: SystemTestCase,
  result: SystemRunResult,
  validation: ValidationResult
): LearningSignal[] {

  const signals: LearningSignal[] = [];

  for (const failure of validation.failures) {

    if (failure.includes("disposition")) {
      signals.push({
        caseId: test.id,
        failureType: "wrong_disposition",
        expected: test.expected.disposition,
        actual: result.decision?.finalDisposition,
        context: {
          complaint: result.resolvedComplaint!,
          symptoms: result.parsed?.secondary ?? [],
          posterior: result.posterior,
          decision: result.decision,
        },
      });
    }

    if (failure.includes("diagnosis")) {
      signals.push({
        caseId: test.id,
        failureType: "missed_diagnosis",
        expected: test.expected.primaryDiagnosis,
        actual: result.decision?.posterior.topDiagnosis,
        context: {
          complaint: result.resolvedComplaint!,
          symptoms: result.parsed?.secondary ?? [],
        },
      });
    }
  }

  return signals;
}

export type FixType =
  | "adjust_prior"
  | "add_feature_likelihood"
  | "adjust_threshold"

export interface ProposedFix {
  id: string;

  type: FixType;

  target: {
    complaint: string;
    diagnosis?: string;
    parameter?: string;
  };

  change: {
    from?: any;
    to: any;
  };

  reason: string;

  sourceSignalId: string;

  category: "clinical" | "performance";

  autoApprove: boolean;
}
Example fix generator
// server/learning/fixGenerator.ts

export function generateFixes(signals: LearningSignal[]): ProposedFix[] {

  const fixes: ProposedFix[] = [];

  for (const s of signals) {

    // ── Missed diagnosis → adjust prior ─────────────
    if (s.failureType === "missed_diagnosis") {

      fixes.push({
        id: crypto.randomUUID(),
        type: "adjust_prior",
        target: {
          complaint: s.context.complaint,
          diagnosis: s.expected,
        },
        change: {
          to: "increase",
        },
        reason: `Missed expected diagnosis ${s.expected}`,
        sourceSignalId: s.caseId,
        category: "clinical",
        autoApprove: false, // NEVER auto
      });
    }

    // ── Over-escalation → adjust threshold ──────────
    if (s.failureType === "over_escalation") {

      fixes.push({
        id: crypto.randomUUID(),
        type: "adjust_threshold",
        target: {
          parameter: "riskThreshold",
        },
        change: {
          to: "increase",
        },
        reason: "Too many ER escalations",
        sourceSignalId: s.caseId,
        category: "clinical",
        autoApprove: false,
      });
    }
  }

  return fixes;
}

// server/learning/learningOrchestrator.ts

export async function runLearningCycle() {

  const testResults = await runTestSuite(TEST_CASES);

  const allSignals: LearningSignal[] = [];

  for (const r of testResults) {
    if (!r.passed) {
      const signals = extractLearningSignals(
        TEST_CASES.find(t => t.id === r.id)!,
        r.trace,
        { passed: r.passed, failures: r.failures }
      );
      allSignals.push(...signals);
    }
  }

  const fixes = generateFixes(allSignals);

  // Send into your existing governance system
  await submitImprovementActions(fixes);
}

E) Improvement verification
function compareResults(before, after) {

  let improved = 0;
  let worsened = 0;

  for (let i = 0; i < before.length; i++) {
    if (!before[i].passed && after[i].passed) improved++;
    if (before[i].passed && !after[i].passed) worsened++;
  }

  return {
    improved,
    worsened,
    net: improved - worsened,
  };
}
