type Result<T=any> = { ok: true; data: T } | { ok: false; error: string };

async function safeFetch(url: string, init: RequestInit): Promise<Result> {
  try {
    const res = await fetch(url, init);
    const txt = await res.text();
    if (!res.ok) return { ok: false, error: `HTTP ${res.status}: ${txt}` };
    try { return { ok: true, data: JSON.parse(txt) }; }
    catch { return { ok: true, data: txt }; }
  } catch (e: any) {
    return { ok: false, error: e?.message || "network_error" };
  }
}

// 🏥 Hospital (API)
export async function connectHospital(patient: any): Promise<Result> {
  return safeFetch(process.env.HOSPITAL_API!, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.HOSPITAL_TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify(patient)
  });
}

  if (!r.ok) onFail(r.error);

}

type RegionState = {
  name: string;
  load: number;        // 0..1
  latencyMs: number;
  healthy: boolean;
};

export function pickBestRegion(regions: RegionState[]) {
  return regions
    .filter(r => r.healthy)
    .sort((a, b) => (a.load + a.latencyMs/1000) - (b.load + b.latencyMs/1000))[0];
}

export function rebalance(regions: RegionState[]) {
  const hot = regions.filter(r => r.load > 0.8);
  const cold = regions.filter(r => r.load < 0.5);

  return hot.map(h => ({
    from: h.name,
    to: cold[0]?.name,
    action: "shift_traffic"
  }));
}

type Provider = {
  id: string;
  specialty: string;
  distanceKm: number;
  load: number;      // 0..1
  slaMs: number;     // expected response time
};

export function matchProvider(patient: any, providers: Provider[]) {
  const candidates = providers.filter(p => p.specialty === patient.complaint);

  return candidates.sort((a, b) => {
    const scoreA = a.distanceKm * 0.4 + a.load * 0.4 + (a.slaMs/1000) * 0.2;
    const scoreB = b.distanceKm * 0.4 + b.load * 0.4 + (b.slaMs/1000) * 0.2;
    return scoreA - scoreB;
  })[0] || null;
}

type Visit = { cost: number; revenue: number; latencyMs: number };

export function optimizeWorkflow(visits: Visit[]) {
  const totalRevenue = visits.reduce((s,v)=>s+v.revenue,0);
  const totalCost = visits.reduce((s,v)=>s+v.cost,0);
  const avgLatency = visits.reduce((s,v)=>s+v.latencyMs,0) / Math.max(1, visits.length);

  return {
    profit: totalRevenue - totalCost,
    margin: totalRevenue ? (totalRevenue - totalCost)/totalRevenue : 0,
    avgLatency
  };
}
🔧 Decision loop (apply changes)
export function applyOptimization(metrics: any) {
  const actions: string[] = [];

  if (metrics.margin < 0.2) actions.push("reduce_cost_path");
  if (metrics.avgLatency > 1500) actions.push("enable_fast_path");

  return actions;
}

export function nextBestQuestion(dx: {name:string, p:number}[], qs: {id:string, weight:number}[]) {
  let best = null, bestScore = -Infinity;
  for (const q of qs) {
    const score = dx.reduce((s,d)=>s + d.p * q.weight, 0);
    if (score > bestScore) { bestScore = score; best = q.id; }
  }
  return best;
}

(3) 🔁 Robust retry with jitter (for all externals)
export async function retry<T>(fn: ()=>Promise<T>, tries=3) {
  let err:any;
  for (let i=0;i<tries;i++){
    try { return await fn(); }
    catch(e){ err=e; await new Promise(r=>setTimeout(r, 200*(2**i)+Math.random()*100)); }
  }
  throw err;
}
(4) 📊 Real-time anomaly (Z-score)
export function zAnomaly(series: number[]) {
  const n = series.length || 1;
  const mean = series.reduce((a,b)=>a+b,0)/n;
  const varr = series.reduce((a,b)=>a+(b-mean)**2,0)/n;
  const sd = Math.sqrt(varr || 1);
  const last = series[n-1] ?? 0;
  return Math.abs((last - mean)/sd) > 3;
}
