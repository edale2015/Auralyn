2) Engine: server/engines/crossComplaintBoostEngine.ts
import fs from "node:fs";

export type CrossBoostRule = {
  Rule_ID: string;
  Applies_To_Complaint: string;
  When_AnyAnswers_Logic: string;
  Target_Dx_ID: string;
  Points: number;
  Max_Applications: number;
  Severity: "LOW" | "MODERATE" | "HIGH";
  Message: string;
};

export type CrossAdjustment = {
  ruleId: string;
  targetDxId: string;
  points: number;
  severity: "LOW" | "MODERATE" | "HIGH";
  message: string;
};

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQ = false;
  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"' && line[i + 1] === '"') { cur += '"'; i++; continue; }
    if (ch === '"') { inQ = !inQ; continue; }
    if (ch === "," && !inQ) { out.push(cur); cur = ""; continue; }
    cur += ch;
  }
  out.push(cur);
  return out.map(s => s.trim().replace(/^"|"$/g, ""));
}

export function loadCrossComplaintBoosts(csvPath: string): CrossBoostRule[] {
  if (!fs.existsSync(csvPath)) return [];
  const text = fs.readFileSync(csvPath, "utf8").trim();
  if (!text) return [];
  const lines = text.split("\n").map(l => l.trim()).filter(Boolean);
  const header = splitCsvLine(lines[0]);
  const idx = (k: string) => header.indexOf(k);

  return lines.slice(1).map(line => {
    const c = splitCsvLine(line);
    return {
      Rule_ID: c[idx("Rule_ID")] ?? "",
      Applies_To_Complaint: c[idx("Applies_To_Complaint")] ?? "",
      When_AnyAnswers_Logic: c[idx("When_AnyAnswers_Logic")] ?? "",
      Target_Dx_ID: c[idx("Target_Dx_ID")] ?? "",
      Points: Number(c[idx("Points")] ?? "0"),
      Max_Applications: Number(c[idx("Max_Applications")] ?? "1"),
      Severity: (c[idx("Severity")] ?? "LOW") as any,
      Message: c[idx("Message")] ?? "",
    };
  }).filter(r => r.Rule_ID && r.Applies_To_Complaint && r.When_AnyAnswers_Logic && r.Target_Dx_ID);
}

/**
 * Simple evaluator for "any.<field>=true/false" with AND/OR.
 * Matches your consistency engine style.
 */
export function evalAnyLogic(expr: string, anyAnswers: Record<string, unknown>): boolean {
  const norm = expr.replace(/\s+/g, " ").trim();
  const orParts = norm.split(" OR ").map(s => s.trim());
  for (const orPart of orParts) {
    const andParts = orPart.split(" AND ").map(s => s.trim());
    let ok = true;
    for (const p of andParts) {
      if (!p.startsWith("any.")) { ok = false; break; }
      const [lhs, rhs] = p.split("=");
      const key = lhs.replace("any.", "").trim();
      const want = rhs.trim();
      const val = anyAnswers[key];

      const isTrue = (v: any) => v === true || v === "yes" || v === "true";
      const isFalse = (v: any) => v === false || v === "no" || v === "false";

      if (want === "true") ok = ok && isTrue(val);
      else if (want === "false") ok = ok && isFalse(val);
      else { ok = false; break; }
    }
    if (ok) return true;
  }
  return false;
}

export function applyCrossComplaintBoosts(params: {
  complaintSlug: string;
  anyAnswers: Record<string, unknown>;
  rules: CrossBoostRule[];
  scores: Record<string, number>; // dxId -> score
}): { scores: Record<string, number>; adjustments: CrossAdjustment[] } {
  const outScores = { ...params.scores };
  const adjustments: CrossAdjustment[] = [];

  for (const r of params.rules) {
    if (r.Applies_To_Complaint !== params.complaintSlug) continue;
    if (!evalAnyLogic(r.When_AnyAnswers_Logic, params.anyAnswers)) continue;

    const current = outScores[r.Target_Dx_ID] ?? 0;
    outScores[r.Target_Dx_ID] = current + r.Points;

    adjustments.push({
      ruleId: r.Rule_ID,
      targetDxId: r.Target_Dx_ID,
      points: r.Points,
      severity: r.Severity,
      message: r.Message,
    });
  }

  return { scores: outScores, adjustments };
}

scripts/test-cross-complaint.ts
import fs from "node:fs";
import { loadCrossComplaintBoosts, applyCrossComplaintBoosts } from "../server/engines/crossComplaintBoostEngine";

const RULES_PATH = "server/data/csv/CROSS_COMPLAINT_BOOSTS.csv";
const GOLDENS_PATH = "server/data/csv/CROSS_COMPLAINT_GOLDENS.jsonl";

function readJsonl(p: string): any[] {
  const lines = fs.readFileSync(p, "utf8").split("\n").map(l => l.trim()).filter(Boolean);
  return lines.map(l => JSON.parse(l));
}

function main() {
  const rules = loadCrossComplaintBoosts(RULES_PATH);
  const goldens = readJsonl(GOLDENS_PATH);

  const fails: string[] = [];

  for (const g of goldens) {
    const { scores: outScores, adjustments } = applyCrossComplaintBoosts({
      complaintSlug: g.complaintSlug,
      anyAnswers: g.anyAnswers ?? {},
      rules,
      scores: g.baseScores ?? {},
    });

    const gotRuleIds = adjustments.map((a) => a.ruleId).sort();
    const wantRuleIds = (g.expect.adjustRuleIds ?? []).slice().sort();

    const okRules = JSON.stringify(gotRuleIds) === JSON.stringify(wantRuleIds);
    const target = g.expect.targetDx;
    const base = Number((g.baseScores ?? {})[target] ?? 0);
    const after = Number(outScores[target] ?? 0);
    const minAdded = Number(g.expect.minAddedPoints ?? 0);

    const okDelta = (after - base) >= minAdded;

    if (!okRules || !okDelta) {
      fails.push(
        `${g.id} FAIL\n` +
        `  expected rules=${JSON.stringify(wantRuleIds)} minAdded=${minAdded} target=${target}\n` +
        `  got      rules=${JSON.stringify(gotRuleIds)} delta=${after-base} outScore=${after}`
      );
    }
  }

  if (fails.length) {
    console.error(`CROSS-COMPLAINT GOLDENS FAIL (${fails.length}/${goldens.length})`);
    for (const f of fails.slice(0, 50)) console.error(f);
    process.exit(1);
  }

  console.log(`CROSS-COMPLAINT GOLDENS PASS (${goldens.length}/${goldens.length}) ✅`);
  process.exit(0);
}

main();
