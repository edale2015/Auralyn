// server/engines/selfImprovementEngine.ts

import { detectErrors } from "./errorDetectionEngine";
import { generateFixes } from "./autoFixEngine";

export async function runSelfImprovementCycle() {
  const feedback = await loadFeedbackLogs();

  const errors = detectErrors(feedback);

  const fixes = generateFixes(errors);

  return {
    totalCases: feedback.length,
    errors,
    fixes
  };
}

function groupErrors(errors) {
  const map = {};

  for (const e of errors) {
    const key = e.complaint;

    if (!map[key]) map[key] = [];
    map[key].push(e);
  }

  return map;
}

// server/engines/explainabilityGraphEngine.ts

export function buildExplainabilityGraph(trace) {
  const nodes = [];
  const edges = [];

  nodes.push({ id: "complaint", label: trace.complaint });

  trace.questions.forEach((q, i) => {
    const id = `q_${i}`;
    nodes.push({ id, label: q.question + ": " + q.answer });
    edges.push({ from: "complaint", to: id });
  });

  trace.modifiers.forEach((m, i) => {
    const id = `m_${i}`;
    nodes.push({ id, label: `Modifier: ${m.name}` });
    edges.push({ from: "complaint", to: id });
  });

  trace.rules.forEach((r, i) => {
    const id = `r_${i}`;
    nodes.push({ id, label: `Rule: ${r.name}` });
    edges.push({ from: "complaint", to: id });
  });

  trace.clusters.forEach((c, i) => {
    const id = `c_${i}`;
    nodes.push({ id, label: `${c.name} (${c.score})` });
    edges.push({ from: "complaint", to: id });
  });

  nodes.push({
    id: "final",
    label: `Final: ${trace.diagnosis} (${trace.triage})`
  });

  edges.push({ from: "complaint", to: "final" });

  return { nodes, edges };
}

function SelfImprovePanel() {
  const [data, setData] = useState(null);

  useEffect(() => {
    fetch("/api/self-improve/run")
      .then(res => res.json())
      .then(setData);
  }, []);

  return (
    <div>
      {data?.fixes.map((fix, i) => (
        <div key={i}>
          <p>{fix.suggestion}</p>
          <button>Approve</button>
          <button>Reject</button>
        </div>
      ))}
    </div>
  );
}
🔥 Simulation Panel
function SimulationPanel() {
  const [result, setResult] = useState(null);

  const run = async () => {
    const res = await fetch("/api/simulate-all", { method: "POST" });
    setResult(await res.json());
  };

  return (
    <div>
      <button onClick={run}>Run Simulation</button>

      {result && (
        <div>
          <p>Accuracy: {result.accuracy}%</p>
          <p>Under-triage: {result.underTriage}%</p>
        </div>
      )}
    </div>
  );
}
