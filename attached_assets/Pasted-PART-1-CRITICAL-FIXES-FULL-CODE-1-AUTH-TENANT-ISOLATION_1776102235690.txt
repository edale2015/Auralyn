// server/services/sessionStore.ts
import { randomUUID } from "crypto";

interface Session {
  id: string;
  clinicId: string;
  state: any;
  createdAt: number;
}

const sessions = new Map<string, Session>();

export function createSession(clinicId: string, state: any) {
  const id = randomUUID();

  sessions.set(id, {
    id,
    clinicId,
    state,
    createdAt: Date.now(),
  });

  return id;
}

export function getSession(id: string, clinicId: string) {
  const session = sessions.get(id);

  if (!session || session.clinicId !== clinicId) {
    return null;
  }

  return session.state;
}

export function updateSession(id: string, clinicId: string, state: any) {
  const existing = sessions.get(id);

  if (!existing || existing.clinicId !== clinicId) {
    throw new Error("Session access violation");
  }

  existing.state = state;
}

export async function unifiedRedFlagCheck(input: {
  caseState: any;
  vitals?: any;
  complaint: string;
  features: string[];
}) {
  const flags = new Set<string>();

  // 1️⃣ Q-based
  detectRedFlags(input.caseState).forEach(f => flags.add(f));

  // 2️⃣ vitals agent
  const agent = new RedFlagAgent();

    get: (k: string) => (k === "vitals" ? input.vitals : {}),
    tryGet: () => ({}),
    clone: function () { return this; },
    set: function () {},
  } as any);

  (ctx as any).redFlags?.forEach((f: string) => flags.add(f));

  // 3️⃣ safetyLayer
  const safety = runSafetyCheck(input.complaint, input.features);
  safety.triggered_flags.forEach(f => flags.add(f));

  return {
    flags: Array.from(flags),
    emergent: flags.size > 0,
  };
}
