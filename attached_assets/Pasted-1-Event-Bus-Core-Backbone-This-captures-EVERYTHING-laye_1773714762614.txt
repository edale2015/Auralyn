import { EventEmitter } from "events"

export interface SystemEvent {
  type: "reasoning" | "health" | "error"
  source: string
  payload: any
  timestamp: number
}

class EventBus extends EventEmitter {

  emitEvent(event: SystemEvent) {
    this.emit("event", event)
  }

  subscribe(listener: (e: SystemEvent) => void) {
    this.on("event", listener)
  }
}

export const eventBus = new EventBus()

import { eventBus } from "../../realtime/eventBus"

function track(name: string, fn: () => any) {

  const start = Date.now()

  try {

    const result = fn()

    eventBus.emitEvent({
      type: "reasoning",
      source: name,
      payload: {
        duration: Date.now() - start,
        output: result
      },
      timestamp: Date.now()
    })

    return result

  } catch (err: any) {

    eventBus.emitEvent({
      type: "error",
      source: name,
      payload: err.message,
      timestamp: Date.now()
    })

    throw err
  }
}

3️⃣ System Health Monitor

Tracks engines + services.

server/realtime/systemHealthMonitor.ts
interface ServiceStatus {
  name: string
  status: "healthy" | "warning" | "down"
  latency: number
}

const services: Record<string, ServiceStatus> = {}

export function updateService(
  name: string,
  latency: number,
  error?: boolean
) {

  services[name] = {
    name,
    latency,
    status: error
      ? "down"
      : latency > 1000
      ? "warning"
      : "healthy"
  }
}

export function getSystemHealth() {
  return Object.values(services)
}
4️⃣ WebSocket Server

Pushes real-time updates.

server/realtime/websocketServer.ts
import { Server } from "ws"
import { eventBus } from "./eventBus"
import { getSystemHealth } from "./systemHealthMonitor"

export function startWebSocket(server: any) {

  const wss = new Server({ server })

  wss.on("connection", ws => {

    const listener = (event: any) => {
      ws.send(JSON.stringify({
        type: "event",
        data: event
      }))
    }

    eventBus.subscribe(listener)

    const interval = setInterval(() => {

      ws.send(JSON.stringify({
        type: "health",
        data: getSystemHealth()
      }))

    }, 2000)

    ws.on("close", () => {
      clearInterval(interval)
    })
  })
}
5️⃣ React Real-Time Dashboard
client/src/pages/RealtimeClinicalMonitor.tsx
import { useEffect, useState } from "react"

export default function RealtimeClinicalMonitor() {

  const [events, setEvents] = useState<any[]>([])
  const [health, setHealth] = useState<any[]>([])

  useEffect(() => {

    const ws = new WebSocket("ws://localhost:3000")

    ws.onmessage = (msg) => {

      const data = JSON.parse(msg.data)

      if (data.type === "event") {

        setEvents(prev => [data.data, ...prev.slice(0, 50)])
      }

      if (data.type === "health") {

        setHealth(data.data)
      }
    }

    return () => ws.close()

  }, [])

  return (

    <div style={{ padding: 20 }}>

      <h1>⚡ Real-Time Clinical Monitor</h1>

      <div style={{ display: "flex", gap: 20 }}>

        {/* Reasoning Feed */}
        <div style={{ flex: 1 }}>

          <h2>Reasoning Flow</h2>

          {events.map((e, i) => (

            <div key={i}>

              <b>{e.source}</b> ({e.type})  
              <br />
              {JSON.stringify(e.payload)}
            </div>

          ))}

        </div>

        {/* Health Panel */}
        <div style={{ flex: 1 }}>

          <h2>System Health</h2>

          {health.map((s, i) => (

            <div key={i}>

              <b>{s.name}</b>  
              Status: {s.status}  
              Latency: {s.latency}ms

            </div>

          ))}

        </div>

      </div>

    </div>
  )
}
