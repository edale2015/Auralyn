🧠 1) REASONING TRACE (Control Tower Integration)
/server/services/communication/reasoningTrace.ts
export interface CommunicationTrace {
  patientId: string;
  complaint: string;
  visitCount: number;
  demandDetected: boolean;
  bacterialCriteria: boolean;
  tone: string;
  scriptVariant: string;
  decision: "NO_ANTIBIOTIC" | "DELAYED_RX" | "ANTIBIOTIC_GIVEN";
  reasoning: string[];
  timestamp: Date;
}

export function buildCommunicationTrace(input: {
  patientId: string;
  complaint: string;
  visitCount: number;
  demandDetected: boolean;
  bacterialCriteria: boolean;
  tone: string;
  scriptVariant: string;
}): CommunicationTrace {

  const reasoning: string[] = [];

  if (input.demandDetected) reasoning.push("Patient requested antibiotics");
  if (!input.bacterialCriteria) reasoning.push("No bacterial criteria met");
  if (input.visitCount >= 3) reasoning.push("Repeat visit pattern");

  const decision = input.bacterialCriteria
    ? "ANTIBIOTIC_GIVEN"
    : input.demandDetected
    ? "DELAYED_RX"
    : "NO_ANTIBIOTIC";

  return {
    ...input,
    decision,
    reasoning,
    timestamp: new Date()
  };
}
🎲 2) A/B TESTING ENGINE
/server/services/communication/abTestingEngine.ts
type ScriptGroup = "A" | "B";

export function assignABGroup(patientId: string): ScriptGroup {
  // Deterministic hashing → same patient always same group
  const hash = patientId.split("").reduce((acc, c) => acc + c.charCodeAt(0), 0);
  return hash % 2 === 0 ? "A" : "B";
}

export function getABScript(group: ScriptGroup, baseScript: string): string {
  if (group === "A") {
    return baseScript;
  }

  // Variant B = slightly more assertive clarity
  return `
I understand your concern, and I want to make sure we’re using the most effective treatment for what’s actually happening.

${baseScript}

The key is timing treatment so it works—not just starting it early.
`;
}
🔁 3) SELF-LEARNING ENGINE (RLHF-lite)
/server/services/communication/learningEngine.ts
interface OutcomeData {
  scriptVariant: string;
  antibioticsGiven: boolean;
  returnVisit: boolean;
  patientSatisfaction?: number;
}

const weights: Record<string, number> = {
  neutral_variant: 1,
  frustrated_variant: 1,
  demanding_variant: 1,
  anxious_variant: 1
};

export function updateWeights(outcome: OutcomeData) {
  let delta = 0;

  // Reward good outcomes
  if (!outcome.antibioticsGiven) delta += 0.1;
  if (!outcome.returnVisit) delta += 0.1;
  if ((outcome.patientSatisfaction || 0) > 4) delta += 0.2;

  // Penalize bad outcomes
  if (outcome.returnVisit) delta -= 0.2;

  weights[outcome.scriptVariant] =
    (weights[outcome.scriptVariant] || 1) + delta;

  // Clamp weights
  weights[outcome.scriptVariant] = Math.max(
    0.5,
    Math.min(2, weights[outcome.scriptVariant])
  );
}

export function getBestVariant(): string {
  return Object.entries(weights).sort((a, b) => b[1] - a[1])[0][0];
}
🌐 4) ADVANCED API ROUTE
/server/routes/communicationAdvancedRoutes.ts
import express from "express";
import { generateCommunicationScript } from "../services/communication/scriptEngine";
import { assignABGroup, getABScript } from "../services/communication/abTestingEngine";
import { buildCommunicationTrace } from "../services/communication/reasoningTrace";

const router = express.Router();

router.post("/full-communication", async (req, res) => {
  try {
    const { patientId } = req.body;

    // 1. Generate base script
    const base = generateCommunicationScript(req.body);

    // 2. A/B assignment
    const group = assignABGroup(patientId);
    const script = getABScript(group, base.script);

    // 3. Build reasoning trace
    const trace = buildCommunicationTrace({
      patientId,
      complaint: req.body.complaint,
      visitCount: req.body.visitCount,
      demandDetected: req.body.patientText?.includes("zpack") || false,
      bacterialCriteria: req.body.hasBacterialCriteria,
      tone: base.tone,
      scriptVariant: base.variant
    });

    res.json({
      script,
      abGroup: group,
      trace
    });

  } catch (err) {
    res.status(500).json({ error: "Advanced communication failed" });
  }
});

export default router;

import { useQuery } from "@tanstack/react-query";

export default function CommunicationControlTower() {
  const { data } = useQuery({
    queryKey: ["communicationTrace"],
    queryFn: async () => {
      const res = await fetch("/api/communication/traces");
      return res.json();
    }
  });

  return (
    <div className="p-6">
      <h1 className="text-xl font-bold mb-4">
        Communication Control Tower
      </h1>

      {data?.map((trace: any, i: number) => (
        <div key={i} className="border p-4 mb-3 rounded-xl">
          <p><b>Complaint:</b> {trace.complaint}</p>
          <p><b>Decision:</b> {trace.decision}</p>
          <p><b>Tone:</b> {trace.tone}</p>
          <p><b>Variant:</b> {trace.scriptVariant}</p>

          <div className="mt-2">
            <b>Reasoning:</b>
            <ul className="list-disc ml-5">
              {trace.reasoning.map((r: string, j: number) => (
                <li key={j}>{r}</li>
              ))}
            </ul>
          </div>
        </div>
      ))}
    </div>
  );
}
