self_improve_agent.py
import json
import subprocess
from dataclasses import dataclass
from typing import Any, Dict, List


@dataclass
class CaseResult:
    case_id: str
    complaint: str
    predicted_disposition: str
    expected_disposition: str
    predicted_diagnoses: List[str]
    expected_diagnoses: List[str]
    trace_path: str
    passed: bool


class ImprovementAgent:
    def __init__(self, llm, repo_root: str):
        self.llm = llm
        self.repo_root = repo_root

    def load_failed_cases(self) -> List[CaseResult]:
        with open("data/failed_cases.json", "r") as f:
            raw = json.load(f)
        return [CaseResult(**x) for x in raw]

    def load_trace(self, trace_path: str) -> Dict[str, Any]:
        with open(trace_path, "r") as f:
            return json.load(f)

    def summarize_failure(self, case: CaseResult, trace: Dict[str, Any]) -> Dict[str, Any]:
        return {
            "case_id": case.case_id,
            "complaint": case.complaint,
            "predicted_disposition": case.predicted_disposition,
            "expected_disposition": case.expected_disposition,
            "predicted_diagnoses": case.predicted_diagnoses,
            "expected_diagnoses": case.expected_diagnoses,
            "questions_asked": trace.get("questions_asked", []),
            "rules_triggered": trace.get("rules_triggered", []),
            "modifier_data": trace.get("modifier_data", {}),
            "missing_fields": trace.get("missing_fields", []),
        }

    def propose_change(self, failure_summary: Dict[str, Any]) -> Dict[str, Any]:
        prompt = {
            "role": "clinical_config_improver",
            "instruction": (
                "Propose a minimal safe structured configuration change. "
                "Do not rewrite code. Prefer edits to complaint questions, "
                "disposition rules, cluster scoring, or output templates."
            ),
            "failure_summary": failure_summary,
            "allowed_targets": [
                "CORE_QUESTIONS",
                "DISP_RULES",
                "CLUSTER_SCORING_RULES",
                "OUTPUT_TEMPLATES",
                "GLOBAL_SECONDARY",
                "MED_TO_CONDITION_TRIGGERS"
            ]
        }
        response = self.llm.generate_json(prompt)
        return response

    def apply_change(self, proposal: Dict[str, Any]) -> bool:
        target = proposal["target_table"]
        action = proposal["action"]

        if target == "CORE_QUESTIONS":
            return self._apply_to_json_file("config/core_questions.json", proposal)
        if target == "DISP_RULES":
            return self._apply_to_json_file("config/disp_rules.json", proposal)
        if target == "CLUSTER_SCORING_RULES":
            return self._apply_to_json_file("config/cluster_scoring_rules.json", proposal)
        if target == "OUTPUT_TEMPLATES":
            return self._apply_to_json_file("config/output_templates.json", proposal)
        if target == "GLOBAL_SECONDARY":
            return self._apply_to_json_file("config/global_secondary.json", proposal)
        if target == "MED_TO_CONDITION_TRIGGERS":
            return self._apply_to_json_file("config/med_triggers.json", proposal)

        return False

    def _apply_to_json_file(self, path: str, proposal: Dict[str, Any]) -> bool:
        with open(path, "r") as f:
            data = json.load(f)

        # Simplified example mutation logic
        section = proposal.get("section")
        entry = proposal.get("entry")
        if section not in data:
            data[section] = []
        data[section].append(entry)

        with open(path, "w") as f:
            json.dump(data, f, indent=2)
        return True

    def run_regression(self) -> bool:
        result = subprocess.run(
            ["python", "tests/run_regression.py"],
            cwd=self.repo_root,
            capture_output=True,
            text=True
        )
        print(result.stdout)
        print(result.stderr)
        return result.returncode == 0

    def commit_candidate(self, message: str) -> None:
        subprocess.run(["git", "add", "."], cwd=self.repo_root)
        subprocess.run(["git", "commit", "-m", message], cwd=self.repo_root)

    def run_once(self) -> None:
        failed_cases = self.load_failed_cases()
        if not failed_cases:
            print("No failed cases found.")
            return

        for case in failed_cases[:5]:
            trace = self.load_trace(case.trace_path)
            failure_summary = self.summarize_failure(case, trace)
            proposal = self.propose_change(failure_summary)

            print("Proposal:", json.dumps(proposal, indent=2))

            ok = self.apply_change(proposal)
            if not ok:
                print(f"Failed to apply proposal for {case.case_id}")
                continue

            passed = self.run_regression()
            if passed:
                self.commit_candidate(
                    f"Auto-improve {case.complaint} from failed case {case.case_id}"
                )
            else:
                subprocess.run(["git", "checkout", "--", "."], cwd=self.repo_root)
                print(f"Rejected proposal for {case.case_id}")

# agents/controller_agent.py

from agents.triage_agent import TriageAgent
from agents.evaluation_agent import EvaluationAgent
from agents.failure_agent import FailureAgent
from agents.proposal_agent import ProposalAgent
from agents.regression_agent import RegressionAgent
from agents.promotion_agent import PromotionAgent

class ControllerAgent:

    def __init__(self):
        self.triage = TriageAgent()
        self.evaluator = EvaluationAgent()
        self.failure = FailureAgent()
        self.proposal = ProposalAgent()
        self.regression = RegressionAgent()
        self.promoter = PromotionAgent()

    def run_case(self, case):

        trace = self.triage.run(case)

        result = self.evaluator.evaluate(trace)

        if not result["passed"]:

            failure = self.failure.classify(trace, result)

            proposal = self.proposal.generate(failure)

            regression_ok = self.regression.test(proposal)

            if regression_ok:
                self.promoter.queue(proposal)

        return result

class EvaluationAgent:

    def __init__(self):
        self.gold = GoldenCaseStore()

    def evaluate(self, trace):

        gold = self.gold.get(trace["case_id"])

        disposition_match = trace["disposition"] == gold["expected_disposition"]

        diagnosis_match = (
            gold["expected_diagnosis"]
            in trace["differentials"]

        passed = disposition_match and diagnosis_match

        return {
            "passed": passed,
            "disposition_match": disposition_match,
            "diagnosis_match": diagnosis_match
        }

LEVEL 3 — Reinforcement Learning Engine

optimization/reinforcement_engine.py
import numpy as np

class ReinforcementEngine:

    def __init__(self):
        self.learning_rate = 0.1
        self.q_table = {}

    def update(self, state, action, reward):

        key = (state, action)

        old_value = self.q_table.get(key, 0)

        new_value = old_value + self.learning_rate * (reward - old_value)

        self.q_table[key] = new_value

    def best_action(self, state, actions):

        values = []

        for action in actions:
            values.append(self.q_table.get((state, action), 0))

        if not values:
            return actions[0]

        return actions[np.argmax(values)]

LEVEL 3.2 Learning Memory

Stores patterns across cases.

memory/learning_memory.py
class LearningMemory:

    def __init__(self):
        self.patterns = {}

    def record_case(self, complaint, features, outcome):

        key = (complaint, tuple(sorted(features)))

        if key not in self.patterns:
            self.patterns[key] = {
                "count": 0,
                "outcomes": {}
            }

        self.patterns[key]["count"] += 1

        self.patterns[key]["outcomes"][outcome] = \
            self.patterns[key]["outcomes"].get(outcome, 0) + 1

    def probable_outcome(self, complaint, features):

        key = (complaint, tuple(sorted(features)))

        if key not in self.patterns:
            return None

        outcomes = self.patterns[key]["outcomes"]

        return max(outcomes, key=outcomes.get)

This creates a case memory database.

LEVEL 4 — Question Optimization

The system learns which questions are most valuable.

optimization/question_optimizer.py
class QuestionOptimizer:

    def __init__(self):
        self.question_scores = {}

    def record_value(self, question_id, diagnostic_gain):

        if question_id not in self.question_scores:
            self.question_scores[question_id] = []

        self.question_scores[question_id].append(diagnostic_gain)

    def rank_questions(self, questions):

        ranked = []

        for q in questions:

            score = sum(self.question_scores.get(q, [0]))

            ranked.append((q, score))

        ranked.sort(key=lambda x: x[1], reverse=True)

        return [q[0] for q in ranked]

LEVEL 4.2 Clinical Reasoning Graph
optimization/reasoning_graph.py
class ReasoningGraph:

    def __init__(self):
        self.graph = {}

    def add_edge(self, symptom, diagnosis):

        if symptom not in self.graph:
            self.graph[symptom] = {}

        self.graph[symptom][diagnosis] = \
            self.graph[symptom].get(diagnosis, 0) + 1

    def rank_diagnoses(self, symptoms):

        scores = {}

        for s in symptoms:

            if s not in self.graph:
                continue

            for d, w in self.graph[s].items():
                scores[d] = scores.get(d, 0) + w

        return sorted(scores, key=scores.get, reverse=True)

This builds a symptom → diagnosis knowledge graph from real cases.

import numpy as np

class RiskModel:

    def __init__(self):
        self.weights = {}

    def train(self, features, outcome):

        for f in features:

            self.weights[f] = self.weights.get(f, 0) + outcome

    def risk_score(self, features):

        score = 0

        for f in features:
            score += self.weights.get(f, 0)

        return score
