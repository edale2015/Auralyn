// server/qa/suggestionEngine.ts
export async function generateSuggestions(complaint, context) {
  const suggestions = []

  if (complaint === 'headache' && context.femaleAge < 50) {
    suggestions.push("Consider pregnancy test")
  }

  if (complaint === 'asthma') {
    suggestions.push("Check albuterol frequency")
  }

  if (complaint === 'headache') {
    suggestions.push("Check blood pressure")
  }

  return suggestions
}
