The fastest fix: add a provider-only “create test token” endpoint

This avoids terminal commands entirely and makes testing instant.

Add endpoint

POST /api/provider/intake/test-token

Requires X-Provider-Key

Body (optional): { token?: string, code?: string, expiryMinutes?: number }

Returns: { token, code, expiresAtMs }

Then you can click a button in your provider UI: “Create Test Token” and jump straight to /simple/:token.

Minimal Express handler (drop-in)

Put this in your existing server/intake/routes.summary.ts (or wherever provider routes live):

import crypto from "crypto";
import { getIntakeStore } from "../intakeStorage"; // adjust path to your store selector

const store = getIntakeStore();

function requireProvider(req: any, res: any, next: any) {
  const key = req.headers["x-provider-key"];
  if (!process.env.PROVIDER_API_KEY) return res.status(500).json({ ok: false, error: "Missing PROVIDER_API_KEY" });
  if (key !== process.env.PROVIDER_API_KEY) return res.status(401).json({ ok: false, error: "Unauthorized" });
  next();
}

function randToken(prefix = "test") {
  return `${prefix}_${crypto.randomBytes(6).toString("hex")}`;
}

function randCode() {
  return String(Math.floor(100000 + Math.random() * 900000));
}

router.post("/api/provider/intake/test-token", requireProvider, async (req, res) => {
  try {
    const token = (req.body?.token || randToken("ehr")).toString();
    const code = (req.body?.code || randCode()).toString();
    const expiryMinutes = Number(req.body?.expiryMinutes || 60);
    const expiresAtMs = Date.now() + expiryMinutes * 60 * 1000;

    await store.createSession(token, code, expiresAtMs);

    return res.json({ ok: true, token, code, expiresAtMs });
  } catch (e: any) {
    return res.status(400).json({ ok: false, error: e?.message || "Failed to create token" });
  }
});


Now you can create unlimited fresh tokens without touching the terminal.
