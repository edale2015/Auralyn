🔹 learning_engine.py
import json

MEMORY_FILE = "memory.json"

def log_step(step, result):
    try:
        data = json.load(open(MEMORY_FILE))
    except:
        data = []

    data.append({"step": step, "result": result})

    json.dump(data, open(MEMORY_FILE, "w"))

def get_best_action(step):
    try:
        data = json.load(open(MEMORY_FILE))
    except:
        return None

    matches = [d for d in data if d["step"] == step]

    if matches:
        return matches[-1]["result"]

    return None

🔹 recorder.py
import pyautogui
import time

def record_session(duration=30):
    actions = []

    start = time.time()

    while time.time() - start < duration:
        x, y = pyautogui.position()
        actions.append({
            "time": time.time(),
            "x": x,
            "y": y
        })
        time.sleep(0.5)

    return actions
🔹 replay.py
import pyautogui
import time

def replay(actions):
    for a in actions:
        pyautogui.click(a["x"], a["y"])
        time.sleep(0.5)

from concurrent.futures import ThreadPoolExecutor

def run_batch(jobs):
    results = []

    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(run_job, j) for j in jobs]

        for f in futures:
            results.append(f.result())

    return results
