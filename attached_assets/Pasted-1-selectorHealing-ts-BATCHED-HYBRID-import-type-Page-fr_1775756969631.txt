import type { Page } from "playwright";

function escAttr(v: string) {
  return v.replace(/\\/g, "\\\\").replace(/"/g, '\\"');
}

function escText(v: string) {
  return v.replace(/\\/g, "\\\\").replace(/"/g, '\\"');
}

export async function healSelector(page: Page, selector: string): Promise<string | null> {
  try {
    if (await page.locator(selector).count() > 0) return selector;
  } catch {}

  if (!selector.startsWith("#")) return null;

  const id = selector.slice(1).trim();
  const cssCandidates = [
    `[name="${escAttr(id)}"]`,
    `[aria-label="${escAttr(id)}"]`,
    `[placeholder*="${escAttr(id)}" i]`,
  ];

  // 🔥 SINGLE round-trip batch check
  const found = await page.evaluate((candidates) => {
    for (const c of candidates) {
      try {
        if (document.querySelector(c)) return c;
      } catch {}
    }
    return null;
  }, cssCandidates);

  if (found) return found;

  // Playwright-specific fallback
  const text = escText(id);
  const pwCandidates = [
    `label:has-text("${text}") + input`,
    `label:has-text("${text}") + select`,
  ];

  for (const c of pwCandidates) {
    try {
      if (await page.locator(c).count() > 0) return c;
    } catch {}
  }

  return null;
}
🧠 2. templateRecorder.ts (DYNAMIC-SAFE)
type Step =
  | { type: "click"; selector: string }
  | { type: "fill"; selector: string; value: string }
  | { type: "waitForVisible"; selector: string }
  | { type: "waitForNetworkIdle" };

export class TemplateRecorder {
  private steps: Step[] = [];
  private keyMap = new Map<string, string>();

  normalizeKey(label: string) {
    const base = label.toLowerCase().replace(/[^a-z0-9]+/g, " ").trim()
      .split(" ")
      .map((w, i) => i === 0 ? w : w[0].toUpperCase() + w.slice(1))
      .join("");

    if (!this.keyMap.has(base)) {
      this.keyMap.set(base, label);
      return base;
    }

    // 🔥 collision-safe
    let i = 2;
    while (this.keyMap.has(`${base}__${i}`)) i++;
    const newKey = `${base}__${i}`;
    this.keyMap.set(newKey, label);
    return newKey;
  }

  recordClick(selector: string) {
    this.steps.push({ type: "click", selector });
    this.steps.push({ type: "waitForVisible", selector }); // dynamic-safe
  }

  recordFill(selector: string, value: string) {
    this.steps.push({ type: "fill", selector, value });
  }

  recordNetworkWait() {
    this.steps.push({ type: "waitForNetworkIdle" });
  }

  getTemplate() {
    return this.steps;
  }
}
🧠 3. templateStore.ts (VERSIONED + ROLLBACK)
import { db } from "../db";

export async function saveTemplate(template: any, actorId: string) {

  const existing = await db.query.templates.findFirst({
    where: { id: template.id }
  });

  if (existing) {
    // 🔥 version history
    await db.insert("template_versions").values({
      template_id: template.id,
      version: existing.version,
      data: JSON.stringify(existing),
      created_at: new Date().toISOString()
    });

    await db.update("templates")
      .set({
        data: JSON.stringify(template),
        version: existing.version + 1,
        updated_by: actorId
      })
      .where({ id: template.id });

  } else {
    await db.insert("templates").values({
      id: template.id,
      data: JSON.stringify(template),
      version: 1,
      created_by: actorId
    });
  }
}

🧠 5. templateValidator.ts (HEALTH CHECK ENGINE)
import { Browser } from "playwright";

export async function validateTemplate(browser: Browser, template: any) {

  const page = await browser.newPage();
  const issues = [];

  try {
    await page.goto(template.url);

    for (let i = 0; i < template.steps.length; i++) {
      const step = template.steps[i];
      if (!step.selector) continue;

      const count = await page.locator(step.selector).count();

      if (count === 0) {
        issues.push({ step: i, error: "missing selector" });
      }
    }

    return {
      ok: issues.length === 0,
      issues
    };

  } finally {
    await page.close();
  }
}
