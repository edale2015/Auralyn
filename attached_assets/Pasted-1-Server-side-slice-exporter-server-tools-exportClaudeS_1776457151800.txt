function ensureDir(dir: string) {
  fs.mkdirSync(dir, { recursive: true });
}

function readIfExists(filePath: string): string {
  const abs = path.join(ROOT, filePath);
  if (!fs.existsSync(abs)) {
    return `// FILE NOT FOUND: ${filePath}\n`;
  }
  return fs.readFileSync(abs, 'utf8');
}

function writeSliceFile(sliceDir: string, slice: SliceDef) {
  const outPath = path.join(sliceDir, `${slice.id}.md`);
  const parts: string[] = [];

  parts.push(`# ${slice.title}`);
  parts.push('');
  parts.push('## Review Prompt');
  parts.push('');
  parts.push(slice.prompt);
  parts.push('');
  parts.push('## Files');
  parts.push('');

  for (const file of slice.files) {
    const ext = path.extname(file).replace('.', '') || 'txt';
    const content = readIfExists(file);
    parts.push(`### ${file}`);
    parts.push('');
    parts.push('```' + ext);
    parts.push(content);
    parts.push('```');
    parts.push('');
  }

  fs.writeFileSync(outPath, parts.join('\n'));
}

function writeManifest(exportDir: string, slices: SliceDef[]) {
  const manifest = {
    generatedAt: new Date().toISOString(),
    root: ROOT,
    slices: slices.map((s) => ({
      id: s.id,
      title: s.title,
      prompt: s.prompt,
      files: s.files,
      outputFile: `${s.id}.md`,
    })),
  };

  const manifestPath = path.join(exportDir, 'manifest.json');
  fs.writeFileSync(manifestPath, JSON.stringify(manifest, null, 2));
  return manifestPath;
}

function writeReviewPrompts(exportDir: string, slices: SliceDef[]) {
  const lines: string[] = [];
  lines.push('# Claude Review Prompts');
  lines.push('');

  for (const s of slices) {
    lines.push(`## ${s.title}`);
    lines.push('');
    lines.push(`Use file: \`${s.id}.md\``);
    lines.push('');
    lines.push(s.prompt);
    lines.push('');
    lines.push(
      'Then ask: "List the TOP 5 MOST DANGEROUS FAILURE MODES in this section. Be specific. Do not give generic advice."'
    );
    lines.push('');
  }

  lines.push('## Final Meta Prompt');
  lines.push('');
  lines.push(
    'You have reviewed all modules of a medical triage system. Now answer:'
  );
  lines.push('1. Where can unsafe under-triage still occur?');
  lines.push('2. What is the single most dangerous failure path?');
  lines.push('3. Which module gives a false sense of safety?');
  lines.push('4. What should be fixed first before clinical deployment?');
  lines.push('');

  const promptsPath = path.join(exportDir, 'REVIEW_PROMPTS.md');
  fs.writeFileSync(promptsPath, lines.join('\n'));
  return promptsPath;
}

async function zipDirectory(sourceDir: string, zipPath: string): Promise<void> {
  ensureDir(path.dirname(zipPath));

  await new Promise<void>((resolve, reject) => {
    const output = fs.createWriteStream(zipPath);
    const archive = archiver('zip', { zlib: { level: 9 } });

    output.on('close', () => resolve());
    archive.on('error', (err) => reject(err));

    archive.pipe(output);
    archive.directory(sourceDir, false);
    archive.finalize();
  });
}

export async function exportClaudeSlices(): Promise<ExportResult> {
  const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
  const exportDir = path.join(EXPORT_ROOT, timestamp);
  ensureDir(exportDir);

  for (const slice of SLICE_DEFS) {
    writeSliceFile(exportDir, slice);
  }

  const manifestPath = writeManifest(exportDir, SLICE_DEFS);
  const promptsPath = writeReviewPrompts(exportDir, SLICE_DEFS);

  const zipPath = path.join(exportDir, 'claude-review-slices.zip');
  await zipDirectory(exportDir, zipPath);

  return {
    exportDir,
    zipPath,
    manifestPath,
    promptsPath,
    sliceCount: SLICE_DEFS.length,
  };
}

3) Admin API route for in-app export
server/routes/adminClaudeExportRoutes.ts
import express from 'express';
import path from 'path';
import fs from 'fs';
import { exportClaudeSlices } from '../tools/exportClaudeSlices';

const router = express.Router();

router.post('/export-claude-slices', async (_req, res) => {
  try {
    const result = await exportClaudeSlices();
    res.json({
      ok: true,
      ...result,
      downloadUrl: `/api/admin/claude-export/download?path=${encodeURIComponent(result.zipPath)}`,
    });
  } catch (error: any) {
    res.status(500).json({
      ok: false,
      error: error?.message ?? 'Unknown export error',
    });
  }
});

router.get('/claude-export/download', async (req, res) => {
  try {
    const rawPath = String(req.query.path || '');
    if (!rawPath) {
      return res.status(400).json({ ok: false, error: 'path is required' });
    }

    const normalized = path.normalize(rawPath);
    if (!fs.existsSync(normalized)) {
      return res.status(404).json({ ok: false, error: 'file not found' });
    }

    res.download(normalized, 'claude-review-slices.zip');
  } catch (error: any) {
    res.status(500).json({
      ok: false,
      error: error?.message ?? 'Download failed',
    });
  }
});

export default router;

5) Simple React admin page with button

import { useState } from "react";

type ExportResponse = {
  ok: boolean;
  exportDir?: string;
  zipPath?: string;
  manifestPath?: string;
  promptsPath?: string;
  sliceCount?: number;
  downloadUrl?: string;
  error?: string;
};

export default function AdminClaudeExport() {
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<ExportResponse | null>(null);

  const handleExport = async () => {
    setLoading(true);
    setResult(null);

    try {
      const res = await fetch("/api/admin/export-claude-slices", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
      });

      const data = await res.json();
      setResult(data);
    } catch (err: any) {
      setResult({
        ok: false,
        error: err?.message ?? "Export failed",
      });
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-neutral-950 text-white p-6">
      <div className="max-w-3xl mx-auto space-y-6">
        <div>
          <h1 className="text-3xl font-bold">Claude Review Export</h1>
          <p className="text-neutral-400 mt-2">
            Export repo slices, prompts, and manifest for a structured Claude review.
          </p>
        </div>

        <div className="rounded-2xl border border-neutral-800 bg-neutral-900 p-6 shadow-lg">
          <button
            onClick={handleExport}
            disabled={loading}
            className="rounded-xl bg-blue-600 px-4 py-2 font-medium hover:bg-blue-500 disabled:opacity-50"
          >
            {loading ? "Exporting..." : "Export Claude Review Slices"}
          </button>

          {result && (
            <div className="mt-6 rounded-xl bg-neutral-950 p-4 border border-neutral-800">
              {result.ok ? (
                <div className="space-y-2">

8) My recommendation on whether the in-app button is worth it

Yes. Very worth it.

Because otherwise review prep becomes:

manual
inconsistent
easy to postpone
easy to do badly

With the button, your workflow becomes:

click Export Claude Review Slices
download zip
upload to Claude
review by slice
fix
export again
run second review pass

That’s clean, repeatable, and much more “FDA brain” than “vibes and hope.”

9) My advice for using it safely

Do not export:

.env
secrets
raw PHI
production logs with identifiers

So if you want the next version, I’d add:

file allowlist only
automatic secret-pattern scrubber
optional PHI scrubber before export
