🔧 Claim scrubber (denial prevention)
export function scrubClaim(claim: any) {
  const issues: string[] = [];

  if (!claim.insurance) issues.push("missing_insurance");
  if (!claim.cpt) issues.push("missing_cpt");

  if (claim.cpt === "99285" && claim.disposition !== "ER_NOW") {
    issues.push("overcoding");
    claim.cpt = "99284";
  }

  return { claim, issues, ok: issues.length === 0 };
}
🔧 Revenue dashboard API
export function revenueKPIs(claims: any[]) {
  const total = claims.length;
  const denied = claims.filter(c => c.denied).length;
  const est = claims.reduce((s,c)=>s+(c.amount||0),0);

  return {
    total,
    denialRate: total ? denied/total : 0,
    estimatedRevenue: est
  };
}

    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "Collect symptoms, keep it concise, safe." },
      { role: "user", content: text }
    ]
  });

  const parsed = {
    complaint: text,
    answers: {},
    rawText: text
  };

  const triage = await runFinalPipeline({
    patientId: "mobile",
    ...parsed,
    sessionId: "mobile",
    actorId: "patient"
  }, "mobile-chat");

  return {
    reply: llm.choices[0].message.content,
    disposition: triage.safetyResult.finalDisposition
  };
}
🔧 React Native app (full screen)
import { useState } from "react";
import { View, TextInput, Button, Text, ScrollView } from "react-native";

export default function PatientApp() {
  const [input, setInput] = useState("");
  const [messages, setMessages] = useState<any[]>([]);

  async function send() {
    const res = await fetch("https://api/patient/chat", {
      method: "POST",
      body: JSON.stringify({ text: input }),
      headers: { "Content-Type": "application/json" }
    });
    const data = await res.json();

    setMessages([...messages,
      { role: "user", text: input },
      { role: "ai", text: data.reply, disposition: data.disposition }
    ]);
    setInput("");
  }

  return (
    <View style={{ flex: 1, padding: 20 }}>
      <ScrollView>
        {messages.map((m, i) => (
          <Text key={i}>
            {m.role === "user" ? "🧑 " : "🤖 "} {m.text}
            {m.disposition && ` (${m.disposition})`}
          </Text>
        ))}
      </ScrollView>

      <TextInput value={input} onChangeText={setInput} />
      <Button title="Send" onPress={send} />
    </View>
  );
}
