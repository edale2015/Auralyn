Scenario Randomizer

server/simulation/scenarioRandomizer.ts

export function randomValue(list:any[]){

 return list[Math.floor(Math.random()*list.length)]

}

export function generateScenarioVariables(){

 return{

  age:Math.floor(Math.random()*80)+18,

  sex:randomValue(["male","female"]),

  duration:randomValue(["2","3","5","7"]),

  symptom1:randomValue(["fever","shortness of breath","nausea","fatigue"]),

  symptom2:randomValue(["vomiting","dizziness","sweating"]),

  history:randomValue(["hypertension","smoking","diabetes"]),

  location:randomValue(["left arm","jaw","back"])

 }

}
Clinical Scenario Generator Engine

server/simulation/clinicalScenarioGenerator.ts

import {scenarioTemplates} from "./scenarioTemplates"
import {generateScenarioVariables} from "./scenarioRandomizer"

export function generateClinicalScenario(complaint:string){

 const templates = scenarioTemplates[complaint]

 if(!templates) return null

 const template = templates[Math.floor(Math.random()*templates.length)]

 const vars = generateScenarioVariables()

 let narrative = template

 Object.entries(vars).forEach(([k,v])=>{

  narrative = narrative.replace(`{${k}}`,String(v))

 })

 return{

  complaint,
  narrative,
  variables:vars

 }

}
Scenario API

server/routes/scenarioGeneratorRoutes.ts

import express from "express"
import {generateClinicalScenario} from "../simulation/clinicalScenarioGenerator"

const router=express.Router()

router.get("/scenario/:complaint",(req,res)=>{

 const scenario = generateClinicalScenario(req.params.complaint)

 res.json(scenario)

})

export default router
