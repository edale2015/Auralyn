import OpenAI from "openai";
import { Skill } from "../skills/skillRegistry";

const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY! });

export async function generateLLMSkill(): Promise<Skill> {
  const prompt = `
You are a clinical reasoning assistant.
Generate a diagnostic rule in JSON:
{
  "condition": "...",
  "logic": "if/then rule",
  "output": "diagnosis"
}
Safe, conservative, no risky assumptions.
`;

  const response = await client.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{ role: "user", content: prompt }],
  });

  const text = response.choices[0].message.content || "{}";

  let parsed: any = {};
  try {
    parsed = JSON.parse(text);
  } catch {
    parsed = { condition: "fallback", output: "unknown" };
  }

  const logic = (input: any) => {
    if (input.symptom?.includes(parsed.condition)) {
      return { diagnosis: parsed.output, confidence: 0.7 };
    }
    return { diagnosis: "unknown", confidence: 0.5 };
  };

  return {
    id: `llm_${Date.now()}`,
    name: "llm_generated_skill",
    logic,
    score: 0,
    status: "candidate",
  };
}

📁 server/reasoning/debateEngine.ts
interface AgentOpinion {
  agent: string;
  diagnosis: string;
  confidence: number;
}

export function runDebate(opinions: AgentOpinion[]) {
  const votes: Record<string, number> = {};

  for (const op of opinions) {
    votes[op.diagnosis] = (votes[op.diagnosis] || 0) + op.confidence;
  }

  const sorted = Object.entries(votes).sort((a, b) => b[1] - a[1]);

  return {
    finalDiagnosis: sorted[0][0],
    score: sorted[0][1],
    allVotes: votes,
  };
}

import fs from "fs";
import path from "path";

export interface Guideline {
  id: string;
  condition: string;
  recommendations: string[];
  version: number;
}

const guidelinePath = path.join(__dirname, "../../data/guidelines.json");

export function generateGuideline(condition: string): Guideline {
  const guideline: Guideline = {
    id: `g_${Date.now()}`,
    condition,
    recommendations: [
      "Assess severity",
      "Check red flags",
      "Provide supportive care",
    ],
    version: 1,
  };

  saveGuideline(guideline);

  return guideline;
}

export function saveGuideline(g: Guideline) {
  let existing: Guideline[] = [];

  if (fs.existsSync(guidelinePath)) {
    existing = JSON.parse(fs.readFileSync(guidelinePath, "utf-8"));
  }

  existing.push(g);

  fs.writeFileSync(guidelinePath, JSON.stringify(existing, null, 2));
}
🧪 4. CONTINUOUS FDA VALIDATION LOOP
📁 server/fda/continuousValidation.ts
import { runClinicalDebate } from "../reasoning/orchestrator";

interface TestCase {
  input: any;
  expected: string;
}

const testCases: TestCase[] = [
  { input: { symptom: "cough" }, expected: "viral_uri" },
  { input: { redFlag: true }, expected: "emergency" },
];

export function runContinuousValidation() {
  let correct = 0;

  for (const test of testCases) {
    const result = runClinicalDebate(test.input);

    if (result.finalDiagnosis === test.expected) {
      correct++;
    }
  }

  const accuracy = correct / testCases.length;

  return {
    accuracy,
    passed: accuracy > 0.8,
  };
}
