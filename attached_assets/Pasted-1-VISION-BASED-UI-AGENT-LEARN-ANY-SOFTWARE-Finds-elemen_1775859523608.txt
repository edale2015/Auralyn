import OpenAI from "openai";
const openai = new OpenAI();

export async function findByVision(screenshotBase64: string, goal: string) {
  const res = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{
      role: "user",
      content: [
        { type: "input_text", text: `Locate UI element for: ${goal}. Return x,y.` },
        { type: "input_image", image_base64: screenshotBase64 }
      ]
    }]
  });

  // Expect: { x: number, y: number }
  return JSON.parse(res.choices[0].message.content!);
}

export async function smartClick(page: any, label: string) {
  const sel = await findElement(page, label);
  if (sel) return page.click(sel);

  const img = await page.screenshot({ encoding: "base64" });
  const { x, y } = await findByVision(img, label);
  return clickAt(page, x, y);
}
🎥 2) SCREEN RECORDING + REPLAY DEBUGGER
🔧 Record session
export async function recordSession(page: any) {
  const events: any[] = [];

  page.on("click", (e: any) => events.push({ type: "click", x: e.x, y: e.y }));
  page.on("request", (r: any) => events.push({ type: "request", url: r.url() }));

  return events;
}
🔁 Replay
export async function replay(page: any, events: any[]) {
  for (const e of events) {
    if (e.type === "click") {
      await page.mouse.click(e.x, e.y);
    }
  }
}
