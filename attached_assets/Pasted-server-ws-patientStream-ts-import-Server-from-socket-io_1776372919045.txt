import { Server } from "socket.io";

let io: Server;

export function initWebSocket(server: any) {
  io = new Server(server, {
    cors: { origin: "*" },
  });

  io.on("connection", (socket) => {
    console.log("🟢 Client connected:", socket.id);

    socket.on("disconnect", () => {
      console.log("🔴 Client disconnected:", socket.id);
    });
  });
}

export function broadcastPatientUpdate(data: any) {
  if (io) {
    io.emit("patient_update", data);
  }
}

📁 Frontend WebSocket Hook
import { io } from "socket.io-client";
import { useEffect, useState } from "react";

export function usePatientStream() {
  const [patients, setPatients] = useState<any[]>([]);

  useEffect(() => {
    const socket = io("http://localhost:3000");

    socket.on("patient_update", (data) => {
      setPatients(prev => {
        const updated = prev.filter(p => p.patientId !== data.patientId);
        return [data, ...updated];
      });
    });

    return () => socket.disconnect();
  }, []);

  return patients;
}

📁 server/simulation/digitalTwin.ts
import { runDiagnosis } from "../clinical/diagnosisEngine";
import { runDispositionSafetyPipeline } from "../clinical/dispositionEngine";
import { generateFullCaseSet } from "../validation/fullCaseGenerator";
import { pushToControlTower } from "../wsEmitter";

export async function runDigitalTwin() {
  const cases = generateFullCaseSet();

  for (const c of cases) {
    const diagnosis = await runDiagnosis(c, global.kb);

    const result = runDispositionSafetyPipeline({
      ...c,
      posterior: diagnosis.posterior,
    });

    pushToControlTower({
      id: c.id,
      risk: Math.random() * 5,
      topDiagnosis: result.topDiagnosis,
      flags: result.reasons,
    });
  }
}

import { useEffect, useState } from "react";

export default function FDADashboard() {
  const [data, setData] = useState<any>(null);

  useEffect(() => {
    fetch("/api/control-tower/validation-dashboard")
      .then(res => res.json())
      .then(setData);
  }, []);

  if (!data) return <div>Loading...</div>;

  return (
    <div className="p-6 bg-black text-white">
      <h1 className="text-2xl mb-4">📊 FDA Validation Dashboard</h1>

      <div className="mb-4">
        <div>Pass Rate: {data.latest.passRate}</div>
        <div>Unsafe Undercalls: {data.latest.unsafeUndercalls}</div>
      </div>

      {/* Graph */}
      <div className="bg-gray-800 p-4 rounded">
        <h2>Pass Rate Trend</h2>
        {data.passRates.map((p: number, i: number) => (
          <div key={i} style={{ width: `${p * 100}%` }} className="bg-green-500 h-2 mb-1" />
        ))}
      </div>

      <button
        className="mt-4 bg-blue-600 p-2 rounded"
        onClick={() => window.open("/fda_report.json")}
      >
        Export FDA Report
      </button>
    </div>
  );
}
