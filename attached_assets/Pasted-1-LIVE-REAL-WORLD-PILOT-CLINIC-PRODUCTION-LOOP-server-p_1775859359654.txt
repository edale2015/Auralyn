export async function findElement(page: any, label: string) {

  const strategies = [
    `text=${label}`,
    `[placeholder*="${label}" i]`,
    `[aria-label*="${label}" i]`,
    `label:has-text("${label}") + input`
  ];

  for (const s of strategies) {
    try {
      if (await page.locator(s).count() > 0) return s;
    } catch {}
  }

  return null;
}
🔧 Execute UI workflow
export async function runUIAutomation(template: any) {

  const browser = await chromium.launch();
  const page = await browser.newPage();

  await page.goto(template.url);

  for (const step of template.steps) {

    const selector = await findElement(page, step.label);

    if (!selector) throw new Error("Element not found");

    if (step.type === "click") await page.click(selector);
    if (step.type === "fill") await page.fill(selector, step.value);
  }

  await browser.close();
}
