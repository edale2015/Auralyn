CREATE TABLE IF NOT EXISTS audit_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  type TEXT,               -- decision | change | learning | override
  entity TEXT,             -- skill, rule, pathway
  entity_id TEXT,
  before JSONB,
  after JSONB,
  user_id TEXT,
  source TEXT,             -- system | clinician | learning_engine
  created_at TIMESTAMP DEFAULT now()
);

CREATE TABLE IF NOT EXISTS audit_reports (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  report JSONB,
  created_at TIMESTAMP DEFAULT now()
);

// server/audit/generateAuditReport.ts
export async function generateAuditReport(db: any) {
  const events = await db.audit_events.findAll()

  const summary = {
    total_changes: events.length,
    system_changes: events.filter(e=>e.source==='system').length,
    clinician_changes: events.filter(e=>e.source==='clinician').length,
    learning_changes: events.filter(e=>e.source==='learning_engine').length
  }

  const report = {
    summary,
    timestamp: new Date()
  }

  await db.audit_reports.insert({ report })

  return report
}

// server/policy/optimizePolicy.ts
export async function optimizePolicy(db: any, metrics: any) {
  const current = await db.policy_state.findOne({ policy_name: 'triage' })

  let updated = { ...current.parameters }

  if (metrics.false_reassurance_rate > 0.03) {
    updated.threshold += 0.05
  }

  if (metrics.cost > 200) {
    updated.workup_budget -= 50
  }

  const change = {
    before: current.parameters,
    after: updated
  }

  await db.policy_updates.insert({
    policy_name: 'triage',
    change,
    impact: metrics
  })

  return change
}
