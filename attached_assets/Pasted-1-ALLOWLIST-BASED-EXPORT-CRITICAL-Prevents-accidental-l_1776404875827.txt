import { ALLOWLIST } from "./allowlist";

function isAllowed(filePath: string) {
  return ALLOWLIST.some(prefix => filePath.startsWith(prefix));
}

function readIfExists(filePath: string): string {
  if (!isAllowed(filePath)) {
    return `// BLOCKED BY ALLOWLIST: ${filePath}\n`;
  }

  const abs = path.join(ROOT, filePath);

  if (!fs.existsSync(abs)) {
    return `// FILE NOT FOUND: ${filePath}\n`;
  }

  return fs.readFileSync(abs, "utf8");
}
🔐 2. SECRET SCRUBBER

👉 Prevents leaking:

API keys
JWT secrets
DB credentials
📁 server/tools/secretScrubber.ts
const SECRET_PATTERNS = [
  /process\.env\.[A-Z0-9_]+/g,
  /api[_-]?key\s*[:=]\s*['"][^'"]+['"]/gi,
  /secret\s*[:=]\s*['"][^'"]+['"]/gi,
  /password\s*[:=]\s*['"][^'"]+['"]/gi,
  /postgres:\/\/[^\\s]+/gi,
];

export function scrubSecrets(content: string): string {
  let result = content;

  for (const pattern of SECRET_PATTERNS) {
    result = result.replace(pattern, "[REDACTED_SECRET]");
  }

  return result;
}
🔧 Apply in exporter
import { scrubSecrets } from "./secretScrubber";

const raw = fs.readFileSync(abs, "utf8");
return scrubSecrets(raw);

🔁 4. DIFF-BASED EXPORT (MOST POWERFUL)

👉 Only send Claude what changed
👉 avoids re-reviewing 10,000 lines

📁 server/tools/diffTracker.ts
import fs from "fs";
import crypto from "crypto";
import path from "path";

const HASH_FILE = path.join(process.cwd(), "exports/claude-review/fileHashes.json");

type HashMap = Record<string, string>;

function hash(content: string) {
  return crypto.createHash("sha256").update(content).digest("hex");
}

export function loadPreviousHashes(): HashMap {
  if (!fs.existsSync(HASH_FILE)) return {};
  return JSON.parse(fs.readFileSync(HASH_FILE, "utf8"));
}

export function saveHashes(map: HashMap) {
  fs.mkdirSync(path.dirname(HASH_FILE), { recursive: true });
  fs.writeFileSync(HASH_FILE, JSON.stringify(map, null, 2));
}

export function hasChanged(filePath: string, content: string, prev: HashMap) {
  const h = hash(content);
  return prev[filePath] !== h;
}

export function preExportCheck() {
  const forbidden = [".env", "node_modules"];

  for (const f of forbidden) {
    if (fs.existsSync(f)) {
      console.warn(`⚠ WARNING: ${f} exists — ensure not exported`);
    }
  }
}
