export function loadPipeline(path: string) {
  const file = fs.readFileSync(path, "utf8")
  return yaml.load(file)
}

🧠 YAML Executor
// server/yaml/executor.ts

import { FlowContext } from "../core/FlowContext"
import { registry } from "../registry"

export async function runYamlPipeline(config: any, input: any) {
  const ctx = new FlowContext(input)

  for (const step of config.flow) {
    if (step.parallel) {
      await Promise.all(
        step.parallel.map((name: string) =>
          registry[name]().run(ctx)
        )
      )
    }

    if (step.sequential) {
      for (const name of step.sequential) {
        await registry[name]().run(ctx)
      }
    }
  }

  return ctx.dump()
}

// server/diagnosis/graphBayesianEngine.ts

import { getRelatedDiseases } from "../graph/queries"
import { BayesianEngine } from "../bayesian/engine"

export async function computeDiagnosis(ctx: any) {
  const symptoms = ctx.symptoms

  let graphScores: Record<string, number> = {}

  for (const symptom of symptoms) {
    const diseases = await getRelatedDiseases(symptom)

    diseases.forEach(d => {
      graphScores[d] = (graphScores[d] || 0) + 1
    })
  }

  const bayes = BayesianEngine.compute(ctx)

  // 🔥 Fusion logic
  const finalScores: Record<string, number> = {}

  for (const dx in bayes) {
    const graphBoost = graphScores[dx] || 0
    finalScores[dx] = bayes[dx] * (1 + 0.2 * graphBoost)
  }

  return finalScores
}

🧠 Debate Engine
// server/debate/debateEngine.ts

export async function runDebate(agents: any[], ctx: any) {
  const opinions = await Promise.all(
    agents.map(a => a.evaluate(ctx))
  )

  // 🔥 Debate scoring
  const scores: Record<string, number> = {}

  opinions.forEach(o => {
    scores[o.diagnosis] =
      (scores[o.diagnosis] || 0) + o.confidence
  })

  return {
    opinions,
    consensus: Object.entries(scores).sort(
      (a, b) => b[1] - a[1]
    )[0]
  }
}
