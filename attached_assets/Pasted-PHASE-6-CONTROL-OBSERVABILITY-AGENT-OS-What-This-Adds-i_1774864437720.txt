type AgentStatus = "idle" | "running" | "error";

interface Agent {
  id: string;
  name: string;
  status: AgentStatus;
  lastRun?: number;
  error?: string;
}

const agents = new Map<string, Agent>();

export function registerAgent(agent: Agent) {
  agents.set(agent.id, agent);
}

export function updateAgentStatus(id: string, status: AgentStatus, error?: string) {
  const agent = agents.get(id);
  if (!agent) return;

  agent.status = status;
  agent.lastRun = Date.now();
  agent.error = error;
}

export function getAgents() {
  return Array.from(agents.values());
}

  };
}

function generateDiff(a: any, b: any) {
  return {
    changedDiagnosis: a.diagnosis !== b.diagnosis,
    confidenceDelta: (b.confidence || 0) - (a.confidence || 0),
  };

export async function runAutonomousFlow(input: any) {
  const context = { input, startedAt: Date.now() };

  // 1) PLAN
  const plan = await planTasks(input);
  emitEvent({ id: crypto.randomUUID(), type: "PLAN_CREATED", timestamp: Date.now(), payload: plan });

  // 2) POLICY CHECK (budget, safety, allowed actions)
  const policy = enforcePolicy(plan, context);
  if (!policy.allowed) {
    return { blocked: true, reason: policy.reason };
  }

  // 3) EXECUTE GRAPH
  try {
    const result = await executeGraph(plan);
    emitEvent({ id: crypto.randomUUID(), type: "FLOW_SUCCESS", timestamp: Date.now(), payload: result });
    return result;
  } catch (err: any) {
    // 4) SELF-HEAL
    const recovered = await selfHeal(plan, err);
    emitEvent({ id: crypto.randomUUID(), type: "FLOW_RECOVERED", timestamp: Date.now(), payload: recovered });
    return recovered;
  }
}

import { updateAgentStatus } from "../../phase6/agents/agentRegistry";
import { emitEvent } from "../../phase6/monitoring/eventBus";

export async function executeGraph(tasks: any[]) {
  const results: Record<string, any> = {};

  for (const t of tasks) {
    try {
      updateAgentStatus(t.agent, "running");

      // TODO: map agent → function
      const res = await runAgent(t.agent, results);

      results[t.id] = res;

      updateAgentStatus(t.agent, "idle");
      emitEvent({ id: crypto.randomUUID(), type: "TASK_OK", timestamp: Date.now(), payload: { task: t.id } });
    } catch (e: any) {
      updateAgentStatus(t.agent, "error", e.message);
      emitEvent({ id: crypto.randomUUID(), type: "TASK_FAIL", timestamp: Date.now(), payload: { task: t.id, error: e.message } });
      throw e;
    }
  }

  return results;
}

  }
}

type AgentOpinion = {
  agent: string;
  diagnosis: string;
  confidence: number;
  reasoning: string;
};

export async function runDebate(input: any) {
  const agents = [
    "diagnosis_primary",
    "diagnosis_secondary",
    "safety_agent",
  ];

  const opinions: AgentOpinion[] = [];

  for (const agent of agents) {
    const result = await simulateAgentOpinion(agent, input);
    opinions.push(result);
  }

  const consensus = resolveConsensus(opinions);

  return {
    opinions,
    consensus,
  };
}

async function simulateAgentOpinion(agent: string, input: any): Promise<AgentOpinion> {
  // Hook into real engines later
  if (agent === "diagnosis_primary") {
    return {
      agent,
      diagnosis: "viral_pharyngitis",
      confidence: 0.82,
      reasoning: "No red flags, short duration",
    };
  }

  if (agent === "diagnosis_secondary") {
    return {
      agent,
      diagnosis: "strep_pharyngitis",
      confidence: 0.65,
      reasoning: "Possible bacterial etiology",
    };
  }

  return {
    agent,
    diagnosis: "safe",
    confidence: 0.95,
    reasoning: "No emergency features",
  };
}

function resolveConsensus(opinions: AgentOpinion[]) {
  const sorted = opinions.sort((a, b) => b.confidence - a.confidence);
  return sorted[0];
}

import axios from "axios";

async function runLoadTest() {
  const total = 1000;
  const concurrency = 50;

  let completed = 0;

  async function worker() {
    while (completed < total) {
      completed++;

      try {
        await axios.post("http://localhost:3000/api/autonomy/run", {
          complaint: "sore_throat",
          fever: false,
        });

        console.log("Success", completed);
      } catch (e) {
        console.error("Error", completed);
      }
    }
  }

  await Promise.all(
    Array.from({ length: concurrency }).map(worker)
  );

  console.log("Load test complete");
}
