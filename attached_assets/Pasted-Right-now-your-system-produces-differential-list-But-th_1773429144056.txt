import { DIAGNOSTIC_MODELS } from "./diagnosticConfidenceModel"

export function computeDiagnosisProbability(
  diagnosis:string,
  evidence:any
){

  const model = DIAGNOSTIC_MODELS[diagnosis]

  if(!model)
    return 0.05

  let probability = model.priors

  for(const key of Object.keys(model.likelihoods)){

    if(evidence[key])
      probability *= model.likelihoods[key]
  }

  return probability
}

import { extractDiagnosticEvidence }
from "./diagnosticEvidenceService"

import { computeDiagnosisProbability }
from "./diagnosticProbabilityAggregator"

export function computeDiagnosticConfidence(state:any){

  const evidence =
    extractDiagnosticEvidence(state)

  const differential =
    state.differential ?? []

  const results:any[]=[]

  for(const d of differential){

    const diagnosis =
      typeof d === "string"
      ? d
      : d.diagnosis

    const ruleScore =
      typeof d === "string"
      ? 0.3
      : d.score ?? 0.3

    const probability =
      computeDiagnosisProbability(
        diagnosis,
        evidence
      )

    const similarityScore =
      d.similarityScore ?? 0

    const combined =
      ruleScore * 0.4 +
      similarityScore * 0.3 +
      probability * 0.3

    results.push({

      diagnosis,

      probability,

      ruleScore,

      similarityScore,

      confidence:combined

    })

  }

  return results
    .sort((a,b)=>b.confidence-a.confidence)

}

3. Question Policy Trainer

This updates weights for questions.

server/learning/questionWeightTrainer.ts
import { loadQuestionPolicy, saveQuestionPolicy }
from "./questionPolicyStore"

import { computeQuestionImpact }
from "./questionImpactAnalyzer"

export async function trainQuestionPolicy(
  question:string,
  complaint:string,
  entropyBefore:number,
  entropyAfter:number
){

  const policy = await loadQuestionPolicy()

  const impact =
    computeQuestionImpact({
      entropyBefore,
      entropyAfter
    })

  let row =
    policy.find(
      p =>
        p.question === question &&
        p.complaint === complaint
    )

  if(!row){

    row = {
      question,
      complaint,
      weight:1,
      timesAsked:0,
      impactScore:0
    }

    policy.push(row)

  }

  row.timesAsked += 1

  row.impactScore =
    (row.impactScore * (row.timesAsked - 1) + impact)
    / row.timesAsked

  row.weight =
    0.7 * row.weight + 0.3 * row.impactScore

  await saveQuestionPolicy(policy)

}

Similarity Outcome Scorer

server/similarity/similarityOutcomeScorer.ts
export function outcomeQualityWeight(row:any){

  if(!row.outcome) return 0.5

  let weight = 1

  if(row.outcome.topPredictionMatch)
    weight += 0.5

  if(row.outcome.dispositionMatch)
    weight += 0.5

  if(row.outcome.safetyMiss)
    weight -= 1

  if(row.outcome.actualDiagnosis)
    weight += 0.2

  return Math.max(0,weight)

}

3. Differential Probability Calibration Engine

This adjusts diagnostic probabilities using outcomes.

server/diagnostics/differentialCalibrationEngine.ts
export function calibrateDifferential(
  differential:any[],
  similarCases:any[]
){

  const weights:Record<string,number> = {}

  for(const c of similarCases){

    const dx =
      c.outcome?.actualDiagnosis ||
      c.differential?.[0]

    if(!dx) continue

    weights[dx] =
      (weights[dx] ?? 0) +
      c.similarityScore

  }

  return differential.map(d=>({

    ...d,

    calibratedScore:
      (d.score ?? 0) +
      (weights[d.diagnosis] ?? 0)

  }))
  .sort((a,b)=>b.calibratedScore-a.calibratedScore)

}

export function analyzeTemporalProgression(events:any[]){

  const timeline:any[] = []

    if(e.type === "PATIENT_MESSAGE"){

      timeline.push({
        timestamp:e.timestamp,
        text:e.payload.message
      })

  return progression

}

export function computeDiagnosticConsensus(params:any){

  const votes:Record<string,number> = {}

  for(const source of params.sources){

    for(const d of source){

      votes[d.diagnosis] =
        (votes[d.diagnosis] ?? 0) + d.score

    }

  }

  return Object.entries(votes)
    .map(([diagnosis,score])=>({
      diagnosis,
      score
    }))
    .sort((a,b)=>b.score-a.score)

}

export async function trainOutcomeModel(){

  const rows =
    await loadOutcomeMemory()

  const stats:any = {}

  for(const r of rows){

    const dx =
      r.actualDiagnosis

    stats[dx] ??= {
      count:0
    }

    stats[dx].count++

  }

  return stats

}

7. Confidence Calibration Engine

Improves diagnostic probability accuracy over time.

export function calibrateConfidence(differential:any[]){

  const total =
    differential.reduce(
      (s,d)=>s+d.score,
      0
    )

  return differential.map(d=>({

    diagnosis:d.diagnosis,
    probability:
      d.score / total

  }))

}


This converts raw scores into probabilities.
