// server/engines/coverageEngine.ts

export async function computeSystemCoverage({
  complaints,
  modifiers,
  questions,
  rules,
  plans
}) {
  const systems = {};

  for (const c of complaints) {
    const sys = c.system;

    if (!systems[sys]) {
      systems[sys] = {
        complaints: 0,
        modifiers: false,
        questions: false,
        rules: false,
        plans: false
      };
    }

    systems[sys].complaints++;

    systems[sys].modifiers = modifiers.some(m => m.system === sys);
    systems[sys].questions = questions.some(q => q.system === sys);
    systems[sys].rules = rules.some(r => r.system === sys);
    systems[sys].plans = plans.some(p => p.system === sys);
  }

  return Object.entries(systems).map(([system, data]) => ({
    system,
    ...data,
    status:
      data.modifiers &&
      data.questions &&
      data.rules &&
      data.plans
        ? "COMPLETE"
        : "INCOMPLETE"
  }));
}

// server/engines/autoPackGenerator.ts

export function generatePack(complaint, systemData) {
  return {
    id: complaint.packId,
    complaint: complaint.name,
    system: complaint.system,

    questions: systemData.questions.filter(q => q.complaint === complaint.name),

    modifiers: systemData.modifiers.filter(m => m.system === complaint.system),

    rules: systemData.rules.filter(r => r.complaint === complaint.name),

    plan: systemData.plans[complaint.name]
  };
}
