📁 server/stress/loadGenerator.ts
import axios from "axios";

const API = "http://localhost:3000/api/clinical/run";

function randomPatient(i: number) {
  const cases = [
    { age: 70, cough: true },
    { age: 65, chestPain: true },
    { age: 25, fever: true },
    { age: 3, fever: true },
    { age: 30, pregnant: true, medications: ["ibuprofen"] },
    { age: 50, sob: true },
  ];

  return cases[i % cases.length];
}

export async function runLoadTest(total = 1000, concurrency = 25) {
  let completed = 0;
  let failed = 0;
  let latencies: number[] = [];

  const runBatch = async (batchSize: number) => {
    const promises = [];

    for (let i = 0; i < batchSize; i++) {
      const patient = randomPatient(completed + i);

      const start = Date.now();

      promises.push(
        axios.post(API, patient)
          .then(() => {
            latencies.push(Date.now() - start);
            completed++;
          })
          .catch(() => {
            failed++;
          })
      );
    }

    await Promise.all(promises);
  };

  while (completed + failed < total) {
    await runBatch(concurrency);
    console.log(`Progress: ${completed + failed}/${total}`);
  }

  const avgLatency =
    latencies.reduce((a, b) => a + b, 0) / latencies.length;

  return {
    total,
    completed,
    failed,
    avgLatency,
    maxLatency: Math.max(...latencies)
  };
}

// server/stress/metricsAnalyzer.ts

import { db } from "../db";

export async function analyzeSystem() {
  const logs = await db.select().from("engine_logs");

  const errorRate =
    logs.filter(l => l.status === "error").length / logs.length;

  const avgLatency =
    logs.reduce((a, l) => a + l.latency, 0) / logs.length;

  return {
    errorRate,
    avgLatency,
    totalLogs: logs.length
  };
}

import { chromium } from "playwright";

export async function runUIAutomation(task) {
  const browser = await chromium.launch();
  const page = await browser.newPage();

  await page.goto(task.url);

  for (const step of task.steps) {
    if (step.type === "click") {
      await page.click(step.selector);
    }

    if (step.type === "type") {
      await page.fill(step.selector, step.value);
    }
  }

  await browser.close();
}

import { google } from "googleapis";

export async function readSheet() {
  const sheets = google.sheets("v4");

  const res = await sheets.spreadsheets.values.get({
    spreadsheetId: process.env.SHEET_ID,
    range: "Sheet1!A1:Z100"
  });

  return res.data.values;
}
