import { Request, Response, NextFunction } from "express";

export function requireClinicAccess(req: Request, res: Response, next: NextFunction) {
  const physician = (req as any).physician;

  if (!physician) {
    return res.status(401).json({ error: "Unauthorized" });
  }

  if (!physician.clinicId) {
    return res.status(403).json({ error: "Missing clinicId" });
  }

  next();
}

import axios from "axios";

export async function probeExternalServices() {
  const servicesToCheck = [
    { name: "OpenAI API", url: "https://api.openai.com/v1/models" },
    { name: "PubMed API", url: "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi" },
  ];

  for (const svc of servicesToCheck) {
    const start = Date.now();
    try {
      await axios.get(svc.url, { timeout: 2000 });
      updateService(svc.name, "external", Date.now() - start);
    } catch {
      updateService(svc.name, "external", 9999, true);
    }
  }
}
