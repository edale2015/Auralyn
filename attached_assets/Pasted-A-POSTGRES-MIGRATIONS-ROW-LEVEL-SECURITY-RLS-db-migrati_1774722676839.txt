import axios from "axios";
import { env } from "../../config/env";


let tokenCache: any = null;


export async function getAccessToken() {
  if (tokenCache && tokenCache.expires > Date.now()) {
    return tokenCache.token;
  }


  const res = await axios.post(env.FHIR_BASE_URL + "/oauth/token", {
    grant_type: "client_credentials",
    client_id: env.FHIR_CLIENT_ID,
    client_secret: env.FHIR_CLIENT_SECRET,
    audience: env.FHIR_AUDIENCE,
  });


  tokenCache = {
    token: res.data.access_token,
    expires: Date.now() + res.data.expires_in * 1000,
  };


  return tokenCache.token;
}

5. BAYESIAN DIFFERENTIAL ENGINE
clinical/bayesianEngine.ts
export function bayesianUpdate(priors: any, evidence: string[]) {
  const scores: any = {};


  for (const dx in priors) {
    let prob = priors[dx];


    evidence.forEach((e) => {
      if (dx.includes(e)) prob *= 1.5;
      else prob *= 0.9;
    });


    scores[dx] = prob;
  }


  return Object.entries(scores).sort((a: any, b: any) => b[1] - a[1]);
}

export function resolveConflict(input: {
  deterministic: any;
  probabilistic: any;
}) {
  const { deterministic, probabilistic } = input;


  // HARD SAFETY OVERRIDE
  if (deterministic.disposition === "ER_NOW") {
    return {
      final: deterministic,
      override: "DETERMINISTIC_SAFETY",
      reason: "Hard safety rule triggered",
    };
  }


  // CONFIDENCE WEIGHTED MERGE
  if (probabilistic.confidence > 0.85) {
    return {
      final: probabilistic,
      override: "PROBABILISTIC_HIGH_CONFIDENCE",
    };
  }


  // DEFAULT SAFE FALLBACK
  return {
    final: deterministic,
    override: "SAFE_DEFAULT",
  };
}

export function qSOFA(input: any) {

import { detectSepsis } from "./sepsis";
import { PEWS } from "./pediatric";
import { obstetricCheck } from "./obstetric";
import { suicideRisk } from "./mentalHealth";
import { resolveConflict } from "./conflictResolver";


export function safetyPipeline(input: any, deterministic: any, probabilistic: any) {
  const sepsis = detectSepsis(input);
  if (sepsis.highRisk) return sepsis;


  const peds = PEWS(input);
  if (peds.critical) return { disposition: "ER_NOW", reason: "PEWS" };


  const ob = obstetricCheck(input);
  if (ob?.emergency) return ob;


  const psych = suicideRisk(input);
  if (psych.highRisk) return psych;


  return resolveConflict({ deterministic, probabilistic });
}
