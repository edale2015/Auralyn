@app.route("/submit", methods=["POST"])
def submit():
    data = request.json

    programs = determine_programs(data)

    results = []
    for program in programs:
        result = run_application_flow(program, data)
        results.append(result)

    return jsonify({
        "programs": programs,
        "results": results
    })

if __name__ == "__main__":
    app.run(debug=True)

🔹 agent.py (CORE RPA ENGINE)
from playwright.sync_api import sync_playwright
from automation.perception import find_field
from automation.actions import fill_field, click_button

def run_application_flow(program, user_data):
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=False)
        page = browser.new_page()

        if program == "SNAP":
            page.goto("https://example-snap-form.com")

        # Perception loop
        fields = find_field(page)

        for field in fields:
            label = field["label"]

            if label in user_data:
                fill_field(page, field, user_data[label])

        click_button(page, "Submit")

        return {"program": program, "status": "submitted"}
🔹 perception.py
def find_field(page):
    inputs = page.query_selector_all("input")

    fields = []
    for i in inputs:
        label = i.get_attribute("name") or "unknown"
        fields.append({
            "element": i,
            "label": label
        })

    return fields
🔹 actions.py
def fill_field(page, field, value):
    field["element"].fill(str(value))

def click_button(page, text):
    buttons = page.query_selector_all("button")

    for b in buttons:
        if text.lower() in b.inner_text().lower():
            b.click()
            break

import json

MEMORY_FILE = "memory.json"

def log_step(step, result):
    try:
        data = json.load(open(MEMORY_FILE))
    except:
        data = []

    data.append({"step": step, "result": result})

    json.dump(data, open(MEMORY_FILE, "w"))

def get_best_action(step):
    try:
        data = json.load(open(MEMORY_FILE))
    except:
        return None

    matches = [d for d in data if d["step"] == step]

    if matches:
        return matches[-1]["result"]

    return None

🔹 batch_runner.py
from concurrent.futures import ThreadPoolExecutor

def run_batch(jobs):
    results = []

    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(run_job, j) for j in jobs]

        for f in futures:
            results.append(f.result())

    return results

import base64
from openai import OpenAI

client = OpenAI()

def analyze_screen(image_path, task_context):
    with open(image_path, "rb") as f:
        b64 = base64.b64encode(f.read()).decode()

    response = client.chat.completions.create(
        model="gpt-5.3",
        messages=[
            {
                "role": "system",
                "content": "You are a UI automation agent. Identify actionable elements."
            },
            {
                "role": "user",
                "content": [
                    {"type": "text", "text": f"Task: {task_context}"},
                    {
                        "type": "image_url",
                        "image_url": f"data:image/png;base64,{b64}"
                    }
                ]
            }
        ]
    )

    return response.choices[0].message.content

conn = sqlite3.connect("memory.db")

def log_step(step, decision):
    conn.execute(
        "INSERT INTO logs (step, decision) VALUES (?, ?)",
