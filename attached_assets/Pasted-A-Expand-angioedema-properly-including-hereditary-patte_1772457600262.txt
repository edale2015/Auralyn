E) Phase 2B: 500-case cross-system stress simulation

Here’s a starter script (Node/TS) that creates mixed cases and logs disposition distribution + top dx frequency. It assumes you can call your harness runner programmatically or via CLI.

scripts/simulate-stress.ts
import { execSync } from "node:child_process";
import fs from "node:fs";

const SYSTEMS = ["cardio","pulm","gi","neuro","gu","derm","endo","psych","id","tox","ophtho","ortho_trauma","general","msk","renal","hemeonc","rheum","environmental"];

const N = 500;

type SimResult = { disposition: string; top_dx: string; complaint: string };

const results: SimResult[] = [];

function randBool(p=0.3){ return Math.random() < p; }

for (let i=0;i<N;i++){
  // pick random complaint slug from your registry in real version
  // placeholder: run an existing harness command that supports randomization
  // Example: npm run harness:random -- --n 1 --seed <i>
  const out = execSync(`npm run harness:random -- --n 1 --seed ${i}`, { encoding: "utf8" });

  // Expect your harness prints JSON on one line; adapt to your output format.
  // Example line: {"complaint":"pulm_cough","disposition":"URGENT","top_dx":"pneumonia"}
  const line = out.trim().split("\n").find(l => l.startsWith("{"));
  if (!line) continue;
  const obj = JSON.parse(line);
  results.push({ complaint: obj.complaint, disposition: obj.disposition, top_dx: obj.top_dx });
}

// Summaries
const dispCounts = results.reduce((m,r)=>{ m[r.disposition]=(m[r.disposition]??0)+1; return m; }, {} as Record<string,number>);
const topDxCounts = results.reduce((m,r)=>{ const k=`${r.complaint}:${r.top_dx}`; m[k]=(m[k]??0)+1; return m; }, {} as Record<string,number>);

fs.writeFileSync("stress_results.json", JSON.stringify({ dispCounts, topDxCounts, results }, null, 2));
console.log("Wrote stress_results.json", dispCounts);
