1️⃣ CLINICAL VALIDATION ENGINE
📁 server/validation/clinicalValidationEngine.ts
import { db } from "../db";
import { kbGoldenCases } from "../../shared/schema";

export interface ValidationResult {
  caseId: string;
  passed: boolean;
  diagnosisMatch: boolean;
  dispositionMatch: boolean;
  score: number;
  errors: string[];
}

export async function runValidation(caseInput: any, expected: any): Promise<ValidationResult> {
  const result = await runTriage(caseInput); // YOUR existing engine

  const diagnosisMatch = result.diagnosis === expected.diagnosis;
  const dispositionMatch = result.disposition === expected.disposition;

  const score =
    (diagnosisMatch ? 0.5 : 0) +
    (dispositionMatch ? 0.5 : 0);

  return {
    caseId: expected.caseId,
    passed: score === 1,
    diagnosisMatch,
    dispositionMatch,
    score,
    errors: [
      !diagnosisMatch ? "Diagnosis mismatch" : "",
      !dispositionMatch ? "Disposition mismatch" : "",
    ].filter(Boolean),
  };
}
2️⃣ GOLDEN CASE AUTO-TEST ENGINE
📁 server/validation/goldenCaseRunner.ts
import { db } from "../db";
import { kbGoldenCases, goldenCaseRuns } from "../../shared/schema";
import { runValidation } from "./clinicalValidationEngine";

export async function runAllGoldenCases() {
  const cases = await db.select().from(kbGoldenCases);

  const results = [];

  for (const c of cases) {
    const result = await runValidation(
      c.structuredInputs,
      {
        caseId: c.caseId,
        diagnosis: c.expectedDiagnosis,
        disposition: c.expectedDisposition,
      }
    );

    results.push(result);

    await db.insert(goldenCaseRuns).values({
      goldenCaseId: c.id,
      result,
      score: result.score,
      passed: result.passed,
      runBatch: new Date().toISOString(),
    });
  }

  return results;
}

3️⃣ FDA VALIDATION DASHBOARD BACKEND
📁 server/validation/fdaMetrics.ts
import { db } from "../db";
import { goldenCaseRuns } from "../../shared/schema";

export async function computeFDAMetrics() {
  const runs = await db.select().from(goldenCaseRuns);

  const total = runs.length;
  const passed = runs.filter(r => r.passed).length;

  const accuracy = total ? passed / total : 0;

  return {
    totalCases: total,
    passed,
    accuracy,
    f1Score: accuracy, // placeholder (can expand later)
    status: accuracy >= 0.80 ? "PASS" : "FAIL",
  };
}

5️⃣ FRONTEND — FDA DASHBOARD UI
📁 frontend/src/pages/FDADashboard.tsx
import { useEffect, useState } from "react";

export default function FDADashboard() {
  const [metrics, setMetrics] = useState<any>(null);

  useEffect(() => {
    fetch("/api/fda/metrics")
      .then(res => res.json())
      .then(setMetrics);
  }, []);

  if (!metrics) return <div>Loading...</div>;

  return (
    <div className="p-6">
      <h1 className="text-xl font-bold mb-4">FDA Validation Dashboard</h1>

      <div>Total Cases: {metrics.totalCases}</div>
      <div>Passed: {metrics.passed}</div>
      <div>Accuracy: {(metrics.accuracy * 100).toFixed(2)}%</div>

      <div className={`mt-4 p-2 ${metrics.status === "PASS" ? "bg-green-500" : "bg-red-500"}`}>
        {metrics.status}
      </div>
    </div>
  );
}
