type Outcome = {
  caseId: string;
  predictedDx: string;
  actualDx: string;
  correct: boolean;
  timestamp: number;
};

const outcomes: Outcome[] = [];

export function logOutcome(o: Outcome) {
  outcomes.push(o);
}

export function getRecentOutcomes(limit = 100) {
  return outcomes.slice(-limit);
}
📁 server/learning/weightStore.ts
const dxWeights = new Map<string, number>();

export function getWeight(dx: string) {
  return dxWeights.get(dx) || 1;
}

export function adjustWeight(dx: string, delta: number) {
  const current = dxWeights.get(dx) || 1;
  dxWeights.set(dx, Math.max(0.1, current + delta));
}

export function getAllWeights() {
  return Object.fromEntries(dxWeights);
}
📁 server/learning/selfLearningEngine.ts
import { getRecentOutcomes } from "./outcomeStore";
import { adjustWeight } from "./weightStore";

export function runSelfLearning() {
  const outcomes = getRecentOutcomes(200);

  for (const o of outcomes) {
    if (o.correct) {
      adjustWeight(o.predictedDx, +0.05);
    } else {
      adjustWeight(o.predictedDx, -0.1);
      adjustWeight(o.actualDx, +0.1);
    }
  }

  console.log("🧠 Self-learning cycle complete");
}
🔁 Run loop
setInterval(() => {
  runSelfLearning();
}, 60000);

📁 server/golden/goldenRunner.ts
import { GOLDEN_CASES } from "./goldenCases";

export async function runGoldenCases(runFlow: any) {
  const results = [];

  for (const gc of GOLDEN_CASES) {
    const res = await runFlow(gc.id, gc.input);

    const predicted = res?.diagnosis?.name;
    const correct = predicted === gc.expectedDx;

    results.push({
      caseId: gc.id,
      predicted,
      expected: gc.expectedDx,
      correct
    });
  }

  return results;
}
📁 server/golden/goldenMonitor.ts
import { runGoldenCases } from "./goldenRunner";

export async function monitorGoldenCases(runFlow: any) {
  const results = await runGoldenCases(runFlow);

  const failures = results.filter(r => !r.correct);

  if (failures.length > 0) {
    console.log("🚨 GOLDEN CASE FAILURES:", failures);
  } else {
    console.log("✅ Golden cases all passing");
  }
}

📁 server/safety/safetyGuard.ts
type SafetyResult = {
  allowed: boolean;
  reason?: string;
  severity?: "low" | "medium" | "high" | "critical";
};

export function runSafetyGuard(output: any): SafetyResult {
  if (!output) {
    return { allowed: false, reason: "No output", severity: "high" };
  }

  if (!output.diagnosis) {
    return { allowed: false, reason: "Missing diagnosis", severity: "high" };
  }

  if (output.redFlags?.length > 0) {
    return {
      allowed: false,
      reason: "Red flag detected",
      severity: "critical"
    };
  }

  if (output.confidence < 0.3) {
    return {
      allowed: false,
      reason: "Low confidence",
      severity: "medium"
    };
  }

  return { allowed: true };
}

export async function runFullClinicalFlow(caseId: string, input: any) {

  const safety = runSafetyGuard(result);

  if (!safety.allowed) {

      severity: safety.severity

  }

  return result;
}
