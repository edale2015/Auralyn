export function extractDiagnosticEvidence(state:any){

  const symptoms = (state.symptoms ?? "").toLowerCase()

  const evidence:any = {}

  evidence.fever =
    symptoms.includes("fever")

  evidence.cough =
    symptoms.includes("cough")

  evidence.shortnessBreath =
    symptoms.includes("shortness of breath")

  evidence.chestPain =

  evidence.neckStiffness =
    symptoms.includes("neck stiffness")

  evidence.radiation =
    symptoms.includes("radiation")

  evidence.diaphoresis =
    symptoms.includes("diaphoresis")

  return evidence
}

3. Probability Aggregator

Compute posterior probability.

server/diagnostics/diagnosticProbabilityAggregator.ts
import { DIAGNOSTIC_MODELS } from "./diagnosticConfidenceModel"

export function computeDiagnosisProbability(
  diagnosis:string,
  evidence:any
){

  const model = DIAGNOSTIC_MODELS[diagnosis]

  if(!model)
    return 0.05

  let probability = model.priors

  for(const key of Object.keys(model.likelihoods)){

    if(evidence[key])
      probability *= model.likelihoods[key]
  }

  return probability
}
4. Confidence Aggregation

Merge three signals:

rule engine
similarity engine
probability model
server/diagnostics/diagnosticConfidenceService.ts
import { extractDiagnosticEvidence }
from "./diagnosticEvidenceService"

import { computeDiagnosisProbability }
from "./diagnosticProbabilityAggregator"

export function computeDiagnosticConfidence(state:any){

  const evidence =
    extractDiagnosticEvidence(state)

  const differential =
    state.differential ?? []

  const results:any[]=[]

  for(const d of differential){

    const diagnosis =
      typeof d === "string"
      ? d
      : d.diagnosis

    const ruleScore =
      typeof d === "string"
      ? 0.3
      : d.score ?? 0.3

    const probability =
      computeDiagnosisProbability(
        diagnosis,
        evidence
      )

    const similarityScore =
      d.similarityScore ?? 0

    const combined =
      ruleScore * 0.4 +
      similarityScore * 0.3 +
      probability * 0.3

    results.push({

      diagnosis,

      probability,

      ruleScore,

      similarityScore,

      confidence:combined

    })

  }

  return results
    .sort((a,b)=>b.confidence-a.confidence)

}
