import { runScoring } from "../engines/scoringEngine";
import { runDiagnosis } from "../engines/diagnosisEngine";
import { runBilling } from "../engines/billingEngine";

import { validateInput } from "../core/validation";

export async function runFullClinicalFlow(input: any) {
  const start = Date.now();

  try {
    // 1. Validate
    const validated = validateInput(input);

    // 2. Scoring
    const scores = await runScoring(validated);

    // 3. Diagnosis
    const diagnosis = await runDiagnosis(validated, scores);

    // 4. Billing
    const billing = await runBilling(diagnosis);

    // 5. Log outcome
    await logOutcome({
      input,
      scores,
      diagnosis,
      billing
    });

    // 6. Trigger learning async
    runLearningCycle().catch(console.error);

    // 7. Audit
    await auditLog({
      type: "FULL_FLOW",
      latency: Date.now() - start
    });

    return {
      success: true,
      scores,
      diagnosis,
      billing
    };

  } catch (err) {
    return {
      success: false,
      error: err.message
    };
  }
}

📁 server/engines/learningEngine.ts
import { db } from "../db";

export async function runLearningCycle() {
  const outcomes = await db.select().from("outcomes").limit(100);

  for (const o of outcomes) {
    if (o.predicted !== o.actual) {
      await db.update("weights")
        .set({
          value: db.raw("value - 0.01")
        })
        .where({ diagnosis: o.predicted });
    } else {
      await db.update("weights")
        .set({
          value: db.raw("value + 0.01")
        })
        .where({ diagnosis: o.predicted });
    }
  }

  console.log("Learning cycle complete");
}
🤖 PART 3 — MULTI-AGENT COORDINATOR
📁 server/agents/agentCoordinator.ts
const agents = new Map();

export function registerAgent(name, fn) {
  agents.set(name, {
    fn,
    status: "idle",
    lastRun: null
  });
}

export async function runAgent(name) {
  const agent = agents.get(name);
  if (!agent) throw new Error("Agent not found");

  agent.status = "running";
  agent.lastRun = new Date();

  try {
    await agent.fn();
    agent.status = "healthy";
  } catch (e) {
    agent.status = "error";
  }
}

export function getAgentStatus() {
  return Array.from(agents.entries());
}
