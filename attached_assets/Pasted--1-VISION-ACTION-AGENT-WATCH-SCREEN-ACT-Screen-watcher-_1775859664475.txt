import { findByVision, clickAt } from "./visionAgent";

export async function runVisionAgent(page: any, goal: string) {

  for (let i = 0; i < 5; i++) {

    const screenshot = await page.screenshot({ encoding: "base64" });
    const pos = await findByVision(screenshot, goal);

    if (pos?.x) {
      await clickAt(page, pos.x, pos.y);
      return { success: true };
    }

    await page.waitForTimeout(500);
  }

  return { success: false };
}

export function enterpriseOptimize(claim: any) {

  const strategies = [
    c => c.insurance === "Private" ? "99285" : c.cpt,
    c => c.complexity === "medium" ? "99284" : c.cpt
  ];

  for (const s of strategies) {
    claim.cpt = s(claim);
  }

  return claim;
}
🔧 Denial feedback learning
export function learnFromDenials(claims: any[]) {

  const patterns: any = {};

  claims.forEach(c => {
    if (c.denied) {
      patterns[c.cpt] = (patterns[c.cpt] || 0) + 1;
    }
  });

  return patterns;
}
