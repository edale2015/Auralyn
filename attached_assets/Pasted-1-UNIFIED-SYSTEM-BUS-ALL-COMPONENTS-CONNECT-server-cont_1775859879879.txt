📡 5) LIVE STREAM (REAL-TIME UPDATES)
server/control/liveStream.ts
import { WebSocketServer } from "ws";
import { systemBus } from "./systemBus";

export function startStream(server: any) {

  const wss = new WebSocketServer({ server });

  wss.on("connection", ws => {

    const handler = (data: any) => {
      ws.send(JSON.stringify(data));
    };

    systemBus.on("update", handler);

    ws.on("close", () => systemBus.off("update", handler));
  });
}

import { useEffect, useState } from "react";

export default function MasterControl() {

  const [state, setState] = useState<any>(null);

  useEffect(() => {
    const load = async () => {
      const res = await fetch("/api/control/state");
      setState(await res.json());
    };

    load();
    const i = setInterval(load, 2000);
    return () => clearInterval(i);
  }, []);

  async function trigger(action: string) {
    await fetch("/api/control/action", {
      method: "POST",
      body: JSON.stringify({ action }),
      headers: { "Content-Type": "application/json" }
    });
  }

  if (!state) return <div>Loading...</div>;

  return (
    <div className="bg-black text-white p-6">

      <h1>🧠 Master Control Tower</h1>

      {/* CLINICAL */}
      <div>
        <h2>Clinical</h2>
        <p>Cases: {state.clinical.activeCases}</p>
        <p>Safety: {state.clinical.safetyMismatch}</p>
      </div>

      {/* AUTOMATION */}
      <div>
        <h2>Automation</h2>
        <p>Templates: {state.automation.templates}</p>
      </div>

      {/* REVENUE */}
      <div>
        <h2>Revenue</h2>
        <p>${state.revenue.dailyRevenue}</p>
      </div>

      {/* INTEGRATIONS */}
      <div>
        <h2>Integrations</h2>
        <p>Epic: {state.integrations.epic}</p>
        <p>ECW: {state.integrations.ecw}</p>
      </div>

      {/* ACTIONS */}
      <div className="mt-4 space-x-2">
        <button onClick={()=>trigger("runSimulation")}>🧪 Sim</button>
        <button onClick={()=>trigger("stressTest")}>🔥 Stress</button>
