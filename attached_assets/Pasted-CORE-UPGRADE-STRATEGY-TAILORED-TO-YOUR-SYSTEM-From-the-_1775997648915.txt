// server/agents/contextIsolatedRunner.ts

import { v4 as uuidv4 } from "uuid";

interface ClinicalTask {
  name: string;
  input: any;
  execute: (ctx: any) => Promise<any>;
}

export async function runClinicalWave(tasks: ClinicalTask[]) {
  const results: Record<string, any> = {};

  await Promise.all(
    tasks.map(async (task) => {
      const contextId = uuidv4();

      // 🧠 Fresh isolated context per task
      const freshContext = {
        contextId,
        timestamp: Date.now(),
        memory: {}, // no carryover
      };

      console.log(`Running task ${task.name} in isolated context ${contextId}`);

      const result = await task.execute(freshContext);

      results[task.name] = result;
    })
  );

  return results;
}

// server/safety/clinicalGates.ts

export function enforceClinicalGates({
  diagnosis,
  scores,
  disposition,
}: {
  diagnosis: any;
  scores: any;
  disposition: string;
}) {
  // 🚨 Gate 1: must have scoring
  if (!scores || Object.keys(scores).length === 0) {
    throw new Error("BLOCKED: No clinical scoring present");
  }

  // 🚨 Gate 2: high-risk must escalate
  if (scores.NEWS2 >= 5 && disposition !== "ED") {
    throw new Error("BLOCKED: High NEWS2 must go to ED");
  }

  // 🚨 Gate 3: antibiotics misuse prevention
  if (
    diagnosis.primary === "viral URI" &&
    disposition.includes("antibiotic")
  ) {
    throw new Error("BLOCKED: Antibiotics not indicated");
  }

  return true;
}

export function buildClinicalTrace({
  symptoms,
  questions,
  scores,
  diagnosis,
  disposition,
}) {
  return {
    traceId: Date.now(),
    steps: [
      { stage: "input", data: symptoms },
      { stage: "questions", data: questions },
      { stage: "scoring", data: scores },
      { stage: "diagnosis", data: diagnosis },
      { stage: "disposition", data: disposition },
    ],
    timestamp: new Date().toISOString(),
  };
}

// server/kb/specEngine.ts

export async function loadDispositionRules(db: any) {
  const rules = await db.query("SELECT * FROM disposition_rules");

  return rules.map((rule: any) => ({
    condition: new Function("input", `return ${rule.condition}`),
    action: rule.action,
  }));
}

export function applyRules(input: any, rules: any[]) {
  for (const rule of rules) {
    if (rule.condition(input)) {
      return rule.action;
    }
  }
  return "uncertain";
}
