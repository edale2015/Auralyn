type CPTResult = {
  codes: string[];
  justification: string;
};

export function generateCPT(tokens: any): CPTResult {
  const codes: string[] = [];

  // 🔥 E/M coding based on complexity
  if (tokens.riskLevel === "low") codes.push("99213");
  if (tokens.riskLevel === "moderate") codes.push("99214");
  if (tokens.riskLevel === "high") codes.push("99215");

  // 🔥 Add-on procedures
  if (tokens.allowedDiagnoses.includes("strep")) {
    codes.push("87880"); // rapid strep
  }

  if (tokens.allowedDiagnoses.includes("covid")) {
    codes.push("87635");
  }

  return {
    codes,
    justification: `Risk: ${tokens.riskLevel}, Diagnoses: ${tokens.allowedDiagnoses.join(",")}`
  };
}
/server/billing/revenueOptimizer.ts
export function optimizeRevenue(codes: string[]) {
  const payerMultipliers: any = {
    "99213": 1.0,
    "99214": 1.3,
    "99215": 1.6,
  };

  return codes.map(code => ({
    code,
    adjustedValue: payerMultipliers[code] || 1
  }));
}

  const correct = result.disposition === expected.disposition;

    correct,

  };

/server/fda/fdaDashboard.ts
let stats = {
  total: 0,
  correct: 0,
};

export function recordValidation(result: any) {
  stats.total++;

  if (result.correct) stats.correct++;
}

export function getFDAMetrics() {
  return {
    accuracy: stats.correct / stats.total,
    totalCases: stats.total,
  };
}

let baseline: any = {};

export function updateBaseline(data: any) {
  baseline = data;
}

export function detectDrift(current: any) {
  const driftScore = Math.abs(
    JSON.stringify(current).length - JSON.stringify(baseline).length
  );

  return {
    driftDetected: driftScore > 50,
    driftScore,
  };
}

import { Server } from "ws";

const wss = new Server({ port: 8080 });

export function startPatientStream() {
  wss.on("connection", (ws) => {
    ws.on("message", (data) => {
      const parsed = JSON.parse(data.toString());

      if (parsed.vitals.hr > 120) {
        ws.send(JSON.stringify({
          alert: "tachycardia",
          severity: "high"
        }));
      }
    });
  });
}
