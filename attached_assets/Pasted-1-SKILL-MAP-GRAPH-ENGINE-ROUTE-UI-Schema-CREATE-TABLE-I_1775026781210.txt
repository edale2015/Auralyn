⚙️ Backend
// server/qa/skillGraph.ts
export async function buildSkillGraph(db: any) {
  const skills = await db.kb_skills.findAll()
  const complaints = await db.kb_complaints.findAll()
  const modifiers = await db.kb_modifiers.findAll()

  const nodes = [
    ...skills.map((s: any) => ({ id: s.id, label: s.name, type: 'skill' })),
    ...complaints.map((c: any) => ({ id: c.id, label: c.name, type: 'complaint' })),
    ...modifiers.map((m: any) => ({ id: m.id, label: m.name, type: 'modifier' }))
  ]

  const edges: any[] = []

  for (const s of skills) {
    for (const c of complaints) {
      if (s.complaints?.includes(c.id)) {
        edges.push({
          id: `${s.id}-${c.id}`,
          source: s.id,
          target: c.id,
          label: 'handles'
        })
      }
    }

    for (const m of modifiers) {
      if (s.modifiers?.includes(m.id)) {
        edges.push({
          id: `${s.id}-${m.id}`,
          source: s.id,
          target: m.id,
          label: 'modifies'
        })
      }
    }
  }

  return { nodes, edges }
}
🔌 Route
// server/routes/skillRoutes.ts
router.get('/skill-graph', async (req, res) => {
  const graph = await buildSkillGraph(req.db)
  res.json(graph)

// server/qa/skillCoverage.ts
export async function evaluateSkillCoverage(db: any) {
  const complaints = await db.kb_complaints.findAll()
  const skills = await db.kb_skills.findAll()

  const issues: any[] = []

  for (const c of complaints) {
    const relatedSkills = skills.filter((s: any) =>
      s.complaints?.includes(c.id)
    )

    if (relatedSkills.length === 0) {
      issues.push({
        type: 'CRITICAL',
        complaint: c.name,
        issue: 'No skill coverage'
      })
    }

    const hasModifiers = relatedSkills.some((s: any) => s.modifiers?.length)
    if (!hasModifiers) {
      issues.push({
        type: 'WARNING',
        complaint: c.name,
        issue: 'Missing modifier logic'
      })
    }
  }

  return issues
}

// server/qa/skillOptimizer.ts
export async function suggestSkills(db: any) {
  const complaints = await db.kb_complaints.findAll()
  const skills = await db.kb_skills.findAll()

  const suggestions: any[] = []

  for (const c of complaints) {
    const relatedSkills = skills.filter((s: any) =>
      s.complaints?.includes(c.id)
    )

    if (relatedSkills.length < 3) {
      suggestions.push({
        complaint: c.name,
        suggestion: 'Add diagnostic + safety skill',
        reason: 'Low skill density'
      })
    }

    if (!relatedSkills.some((s: any) => s.name.includes('red_flag'))) {
      suggestions.push({
        complaint: c.name,
        suggestion: 'Add red flag skill',
        reason: 'Missing emergency detection'
      })
    }
  }

  return suggestions
}
