1. MASTER CLINICAL AGENT LOOP
// server/engine/clinicalAgentLoop.ts

import { dispatchTools } from "./toolDispatch";
import { checkPermission } from "../governance/permissionEngine";
import { compressContext } from "../context/compression";
import { bus } from "../events/eventBus";

export async function clinicalAgentLoop(session) {
  bus.emit("session_start", { sessionId: session.id });

  while (true) {
    // 1. Compress context if needed
    session.messages = compressContext(session.messages);

    // 2. Call model
    const response = await callLLM({
      messages: session.messages,
      tools: TOOL_REGISTRY
    });

    session.messages.push(response);

    if (response.stop_reason !== "tool_use") break;

    const results = [];

    for (const toolCall of response.tool_calls) {
      bus.emit("pre_tool_use", toolCall);

      const permission = checkPermission(toolCall);

      if (!permission.allowed) {
        results.push({
          tool_use_id: toolCall.id,
          content: `Blocked: ${permission.reason}`
        });
        continue;
      }

      const output = await dispatchTools(toolCall);

      bus.emit("post_tool_use", { toolCall, output });

      results.push({
        tool_use_id: toolCall.id,
        content: output
      });
    }

    session.messages.push({ role: "tool", content: results });
  }

  bus.emit("session_end", { sessionId: session.id });

  return session;
}

🛑 3. PERMISSION GOVERNANCE ENGINE
// server/governance/permissionEngine.ts

export function checkPermission(toolCall) {
  const { name, input } = toolCall;

  if (name === "generate_disposition") {
    if (input.red_flags_present) {
      return {
        allowed: false,
        reason: "Cannot discharge with red flags"
      };
    }
  }

  if (name === "prescribe_antibiotic") {
    if (!input.bacterial_criteria_met) {
      return {
        allowed: false,
        reason: "No bacterial indication"
      };
    }
  }

  return { allowed: true };
}

// server/context/compression.ts

export function compressContext(messages) {
  if (messages.length < 20) return messages;

  const summary = {
    chief_complaint: extract(messages, "chief_complaint"),
    positives: extract(messages, "positives"),
    negatives: extract(messages, "negatives"),
    red_flags: extract(messages, "red_flags"),
    timeline: extract(messages, "timeline")
  };

  return [
    {
      role: "system",
      content: `CLINICAL SUMMARY:\n${JSON.stringify(summary)}`
    },
    ...messages.slice(-6)
  ];
}
📡 5. EVENT BUS (HOOKABLE)
// server/events/eventBus.ts

class EventBus {
  constructor() {
    this.handlers = {};
  }

  on(event, handler) {
    if (!this.handlers[event]) this.handlers[event] = [];
    this.handlers[event].push(handler);
  }

  emit(event, payload) {
    (this.handlers[event] || []).forEach(h => h(payload));
  }
}

export const bus = new EventBus();
