import Database from "better-sqlite3";
import path from "path";

const DB_PATH = process.env.DB_PATH || path.join(process.cwd(), "data.sqlite");
export const db = new Database(DB_PATH);

export function initDb() {
  db.exec(`
    PRAGMA journal_mode = WAL;

    CREATE TABLE IF NOT EXISTS intake_sessions (
      token TEXT PRIMARY KEY,
      code_hash TEXT NOT NULL,
      expires_at INTEGER NOT NULL,
      used_at INTEGER,
      created_at INTEGER NOT NULL
    );

    CREATE TABLE IF NOT EXISTS cases (
      case_id TEXT PRIMARY KEY,
      token TEXT NOT NULL,
      status TEXT NOT NULL,
      created_at INTEGER NOT NULL,
      updated_at INTEGER NOT NULL,
      current_step INTEGER DEFAULT 0,
      draft_json TEXT DEFAULT '{}',
      intake_json TEXT DEFAULT '{}',
      assistant_json TEXT DEFAULT '{}',
      summary_html TEXT,
      summary_pdf_path TEXT
    );

    CREATE TABLE IF NOT EXISTS files (
      file_id TEXT PRIMARY KEY,
      token TEXT NOT NULL,
      original_name TEXT NOT NULL,
      mime_type TEXT NOT NULL,
      storage_path TEXT NOT NULL,
      created_at INTEGER NOT NULL
    );

    CREATE INDEX IF NOT EXISTS idx_cases_token ON cases(token);
    CREATE INDEX IF NOT EXISTS idx_files_token ON files(token);
  `);
}

function escapeHtml(s: string) {
  return String(s)
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;")
    .replaceAll('"', "&quot;")
    .replaceAll("'", "&#039;");
}

export function saveSummaryHtml(caseId: string, html: string) {
  const dir = process.env.SUMMARY_DIR || path.join(process.cwd(), "summaries");
  if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
  const p = path.join(dir, `${caseId}.html`);
  fs.writeFileSync(p, html, "utf8");
  return p;
}

import { Router } from "express";
import crypto from "crypto";
import { db } from "./db";
import type { DraftPayload, SubmitPayload, VerifyPayload, CaseStatus } from "./types";
import { renderSummaryHtml, saveSummaryHtml } from "./pdf";

export const intakeRouter = Router();

function nowMs() { return Date.now(); }

function sha256(s: string) {
  return crypto.createHash("sha256").update(s).digest("hex");
}

function newId(prefix: string) {
  return `${prefix}_${new Date().toISOString().slice(0,10).replaceAll("-","")}_${crypto.randomBytes(6).toString("hex")}`;
}

function getOrCreateCase(token: string) {
  const row = db.prepare(`SELECT * FROM cases WHERE token = ? ORDER BY created_at DESC LIMIT 1`).get(token);
  if (row) return row;

  const caseId = newId("CASE");
  const ts = nowMs();
  db.prepare(`
    INSERT INTO cases (case_id, token, status, created_at, updated_at, current_step, draft_json, intake_json, assistant_json)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
  `).run(caseId, token, "draft", ts, ts, 0, "{}", "{}", "{}");

  return db.prepare(`SELECT * FROM cases WHERE case_id = ?`).get(caseId);
}

// 1) VERIFY
intakeRouter.post("/api/intake/:token/verify", (req, res) => {
  const token = req.params.token;
  const body = req.body as VerifyPayload;

  if (!body?.code || String(body.code).trim().length < 4) {
    return res.status(400).json({ ok: false, error: "Missing code." });
  }

  const session = db.prepare(`SELECT * FROM intake_sessions WHERE token = ?`).get(token);
  if (!session) return res.status(404).json({ ok: false, error: "Invalid link." });

  if (session.used_at) return res.status(410).json({ ok: false, error: "This link has already been used." });
  if (Number(session.expires_at) < nowMs()) return res.status(410).json({ ok: false, error: "This link has expired." });

  const codeHash = sha256(String(body.code).trim());
  if (codeHash !== session.code_hash) return res.status(401).json({ ok: false, error: "Incorrect code." });

  // Create a case if none exists
  const c = getOrCreateCase(token);

  return res.json({
    ok: true,
    caseId: c.case_id,
    status: c.status,
    currentStep: c.current_step,
    // flowId can be added here when you plug in your Google Sheets flow system
    flowId: "DEFAULT_FLOW"
  });
});

// 2) SAVE DRAFT
intakeRouter.post("/api/intake/:token/save_draft", (req, res) => {
  const token = req.params.token;
  const body = req.body as DraftPayload;

  const c = getOrCreateCase(token);
  const ts = nowMs();

  const mergedDraft = safeMergeJson(c.draft_json, body?.draft || {});
  const step = Number.isFinite(body?.currentStep) ? body.currentStep : c.current_step;

  db.prepare(`
    UPDATE cases
    SET draft_json = ?, current_step = ?, updated_at = ?, status = ?
    WHERE case_id = ?
  `).run(JSON.stringify(mergedDraft), step, ts, "draft", c.case_id);

  return res.json({ ok: true });
});

function safeMergeJson(existingJson: string, incoming: Record<string, any>) {
  let existing: any = {};
  try { existing = JSON.parse(existingJson || "{}"); } catch {}
  return { ...existing, ...incoming };
}

function statusText(s: CaseStatus) {
  if (s === "draft") return "Continue your intake.";
  if (s === "submitted") return "Submitted. Provider review pending.";
  if (s === "in_review") return "In review.";
  if (s === "signed") return "Complete. Summary available.";
  if (s === "closed") return "Closed.";
  return "Unknown.";
}

function computeRedFlags(body: SubmitPayload): string[] {
  const flags: string[] = [];
  const s = body.symptoms || {};
  if (s["chest_pain"] === "yes") flags.push("Chest pain");
  if (s["shortness_of_breath"] === "yes") flags.push("Shortness of breath");
  if (s["confusion"] === "yes") flags.push("Confusion");
  return flags;
}

function computeTriage(body: SubmitPayload): string {
  const flags = computeRedFlags(body);
  if (flags.length) return "Urgent screen: needs clinician review now";
  return "Routine";
}

server/routes.files.ts
import { Router } from "express";
import multer from "multer";
import path from "path";
import crypto from "crypto";
import { db } from "./db";
import { UPLOAD_DIR } from "./storage";
import fs from "fs";

export const filesRouter = Router();

const upload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 8 * 1024 * 1024 } // 8MB
});

function newFileId() {
  return `FILE_${crypto.randomBytes(10).toString("hex")}`;
}

filesRouter.post("/api/intake/:token/upload", upload.single("file"), (req, res) => {
  const token = req.params.token;
  const file = req.file;
  if (!file) return res.status(400).json({ ok: false, error: "No file." });

  const fileId = newFileId();
  const ext = safeExt(file.mimetype, file.originalname);
  const storagePath = path.join(UPLOAD_DIR, `${fileId}${ext}`);

  fs.writeFileSync(storagePath, file.buffer);

  db.prepare(`
    INSERT INTO files (file_id, token, original_name, mime_type, storage_path, created_at)
    VALUES (?, ?, ?, ?, ?, ?)
  `).run(fileId, token, file.originalname, file.mimetype, storagePath, Date.now());

  return res.json({ ok: true, fileId, mimeType: file.mimetype, name: file.originalname });
});

filesRouter.get("/api/file/:fileId", (req, res) => {
  const fileId = req.params.fileId;
  const row = db.prepare(`SELECT * FROM files WHERE file_id = ?`).get(fileId);
  if (!row) return res.status(404).send("Not found");

  res.setHeader("Content-Type", row.mime_type);
  return res.sendFile(row.storage_path);
});

function safeExt(mime: string, name: string) {
  const byName = path.extname(name || "").slice(0, 10);
  if (byName) return byName;
  if (mime === "image/jpeg") return ".jpg";
  if (mime === "image/png") return ".png";
  if (mime === "application/pdf") return ".pdf";
  return "";
}

const API_BASE = import.meta.env.VITE_API_BASE || "http://localhost:3001";

export async function postJson<T>(path: string, body: any): Promise<T> {
  const res = await fetch(`${API_BASE}${path}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body)
  });
  const data = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(data?.error || `Request failed (${res.status})`);
  return data as T;
}

export async function getJson<T>(path: string): Promise<T> {
  const res = await fetch(`${API_BASE}${path}`);
  const data = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(data?.error || `Request failed (${res.status})`);
  return data as T;
}

export async function uploadFile(token: string, file: File) {
  const fd = new FormData();
  fd.append("file", file);
  const res = await fetch(`${API_BASE}/api/intake/${token}/upload`, { method: "POST", body: fd });
  const data = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(data?.error || `Upload failed (${res.status})`);
  return data as { ok: true; fileId: string; name: string; mimeType: string };
}

import { useState } from "react";
import { postJson } from "../api";

export default function VerifyCard({ token, onVerified }: { token: string; onVerified: (caseId: string) => void }) {
  const [code, setCode] = useState("");
  const [dob, setDob] = useState("");
  const [err, setErr] = useState<string | null>(null);
  const [busy, setBusy] = useState(false);

  async function verify() {
    setErr(null);
    setBusy(true);
    try {
      const data = await postJson<{ ok: true; caseId: string }>(`/api/intake/${token}/verify`, { code, dob: dob || undefined });
      onVerified(data.caseId);
    } catch (e: any) {
      setErr(e?.message || "Verification failed");
    } finally {
      setBusy(false);
    }
  }

  return (
    <div className="card">
      <h2>Verify your secure link</h2>
      <p className="muted">Enter the 6-digit code you received.</p>

      <div className="row">
        <input value={code} onChange={(e) => setCode(e.target.value)} placeholder="6-digit code" />
        <button disabled={busy || code.length < 4} onClick={verify}>{busy ? "Checking..." : "Verify"}</button>
      </div>

      <div className="row">
        <input value={dob} onChange={(e) => setDob(e.target.value)} placeholder="DOB (optional) e.g. 1980-01-31" />
      </div>

      {err && <div className="error">{err}</div>}
    </div>
  );
}

web/src/components/UploadPanel.tsx
import { useState } from "react";
import { uploadFile } from "../api";

export default function UploadPanel({
  token,
  attachments,
  setAttachments
}: {
  token: string;
  attachments: string[];
  setAttachments: (ids: string[]) => void;
}) {
  const [busy, setBusy] = useState(false);
  const [err, setErr] = useState<string | null>(null);

  async function onPick(file: File | null) {
    if (!file) return;
    setErr(null);
    setBusy(true);
    try {
      const r = await uploadFile(token, file);
      setAttachments([...attachments, r.fileId]);
    } catch (e: any) {
      setErr(e?.message || "Upload failed");
    } finally {
      setBusy(false);
    }
  }

  return (
    <div className="card">
      <h2>Upload photos or documents</h2>
      <p className="muted">Examples: rash photo, throat photo, test result.</p>
      <input
        type="file"
        accept="image/*,application/pdf"
        disabled={busy}
        onChange={(e) => onPick(e.target.files?.[0] || null)}
      />
      {busy && <div className="muted">Uploading...</div>}
      {err && <div className="error">{err}</div>}

      {!!attachments.length && (
        <div className="muted small">Attached: {attachments.length} file(s)</div>
      )}
    </div>
  );
}

web/src/pages/Intake.tsx
import { useEffect, useMemo, useRef, useState } from "react";
import { useNavigate, useParams } from "react-router-dom";
import VerifyCard from "../components/VerifyCard";
import SymptomGrid, { Tri } from "../components/SymptomGrid";
import UploadPanel from "../components/UploadPanel";
import ConsentPanel from "../components/ConsentPanel";
import ReviewSubmit from "../components/ReviewSubmit";
import { postJson } from "../api";

type Step = 0 | 1 | 2 | 3 | 4;

export default function Intake() {
  const { token = "" } = useParams();
  const nav = useNavigate();

  const [verifiedCaseId, setVerifiedCaseId] = useState<string | null>(null);
  const [step, setStep] = useState<Step>(0);

  const [chiefComplaint, setChiefComplaint] = useState("");
  const [freeText, setFreeText] = useState("");
  const [symptoms, setSymptoms] = useState<Record<string, Tri>>({});
  const [attachments, setAttachments] = useState<string[]>([]);
  const [consent, setConsent] = useState<any>({
    telehealth: false,
    privacy: false,
    signatureName: "",
    signedAt: ""
  });

  const [submitBusy, setSubmitBusy] = useState(false);
  const [submitErr, setSubmitErr] = useState<string | null>(null);

  // Autosave draft
  const autosaveTimer = useRef<number | null>(null);
  const draftPayload = useMemo(() => ({
    currentStep: step,
    draft: { chiefComplaint, freeText, symptoms, attachments, consent }
  }), [step, chiefComplaint, freeText, symptoms, attachments, consent]);

  useEffect(() => {
    if (!verifiedCaseId) return;
    if (autosaveTimer.current) window.clearTimeout(autosaveTimer.current);
    autosaveTimer.current = window.setTimeout(async () => {
      try {
        await postJson(`/api/intake/${token}/save_draft`, draftPayload);
      } catch {
        // Silent draft save failure (patient can still submit)
      }
    }, 900);
  }, [draftPayload, verifiedCaseId, token]);

  async function submit() {
    setSubmitErr(null);
    setSubmitBusy(true);
    try {
      const payload = {
        chiefComplaint,
        freeText,
        symptoms,
        attachments,
        consent: { ...consent, signedAt: new Date().toISOString() }
      };
      await postJson(`/api/intake/${token}/submit`, payload);
      nav(`/intake/${token}/status`);
    } catch (e: any) {
      setSubmitErr(e?.message || "Submit failed");
    } finally {
      setSubmitBusy(false);
    }
  }

  if (!token) return <div className="container"><div className="card">Missing token.</div></div>;

  return (
    <div className="container">
      {!verifiedCaseId ? (
        <VerifyCard token={token} onVerified={(caseId) => { setVerifiedCaseId(caseId); setStep(1); }} />
      ) : (
        <>
          <div className="card">
            <h1>Visit Intake</h1>
            <div className="muted small">Step {step} of 4</div>
            <div className="row">
              <button disabled={step <= 1} onClick={() => setStep((step - 1) as Step)}>Back</button>
              <button disabled={step >= 4} onClick={() => setStep((step + 1) as Step)}>Next</button>
            </div>
          </div>

          {step === 1 && (
            <div className="card">
              <h2>Reason for visit</h2>
              <div className="row">
                <input value={chiefComplaint} onChange={(e) => setChiefComplaint(e.target.value)} placeholder="Chief complaint (e.g., sore throat)" />
              </div>
              <div className="row">
                <textarea value={freeText} onChange={(e) => setFreeText(e.target.value)} placeholder="Anything else you'd like us to know?" />
              </div>
            </div>
          )}

          {step === 2 && (
            <SymptomGrid value={symptoms} onChange={setSymptoms} />
          )}

          {step === 3 && (
            <UploadPanel token={token} attachments={attachments} setAttachments={setAttachments} />
          )}

          {step === 4 && (
            <>
              <ConsentPanel consent={consent} setConsent={setConsent} />
              <ReviewSubmit
                chiefComplaint={chiefComplaint}
                busy={submitBusy}
                error={submitErr}
                canSubmit={!!chiefComplaint && !!consent.telehealth && !!consent.privacy && !!consent.signatureName}
                onSubmit={submit}
              />
            </>
          )}
        </>
      )}
    </div>
  );
}

            </div>

            {data.status === "signed" && (

                <Link to={`/intake/${token}/summary`}>View summary</Link>

web/src/pages/Summary.tsx
import { useEffect, useState } from "react";
import { useParams, Link } from "react-router-dom";

const API_BASE = import.meta.env.VITE_API_BASE || "http://localhost:3001";

export default function Summary() {
  const { token = "" } = useParams();
  const [html, setHtml] = useState<string>("");
  const [err, setErr] = useState<string | null>(null);

  useEffect(() => {
    async function load() {
      setErr(null);
      try {
        const res = await fetch(`${API_BASE}/api/intake/${token}/summary`);
        if (!res.ok) throw new Error(`Not available (${res.status})`);
        setHtml(await res.text());
      } catch (e: any) {
        setErr(e?.message || "Failed to load summary");
      }
    }
    load();
  }, [token]);

  return (
    <div className="container">
      <div className="card">
        <h1>Visit Summary</h1>
        <div className="row">
          <Link to={`/intake/${token}/status`}>Back to status</Link>
        </div>
        {err && <div className="error">{err}</div>}
      </div>

      {!err && (
        <div className="card">
          <div dangerouslySetInnerHTML={{ __html: html }} />
        </div>
      )}
    </div>
  );
}
