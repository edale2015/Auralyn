// server/compliance/fdaFullSubmission.ts

import fs from "fs";
import path from "path";
import { pdfmetrics } from "reportlab/pdfbase/pdfmetrics";
import { UnicodeCIDFont } from "reportlab/pdfbase/cidfonts";
import {
  SimpleDocTemplate, Paragraph, Spacer
} from "reportlab/platypus";
import { getPerformanceStats } from "./performanceRegistry";

const BASE = "./fda_submission";

function write(name: string, content: string) {
  fs.mkdirSync(BASE, { recursive: true });
  fs.writeFileSync(path.join(BASE, name), content);
}

export async function generateFDASubmission() {
  const stats = getPerformanceStats();

  write("01_device_description.md", `
# Device Description
AI clinical triage system (assistive mode) combining rules, scoring systems, RLHF, and case memory.
Human-in-the-loop for high-risk outputs.
`);

  write("02_intended_use.md", `
# Intended Use
Assist clinicians in triage, differential diagnosis, and care planning. Not for autonomous discharge decisions.
`);

  write("03_system_architecture.md", `
# System Architecture
Unified pipeline: routing → questions → modifiers → rules/clusters → RLHF → memory → triage/safety → plan → trace.
`);

  write("04_risk_analysis.md", `
# Risk Analysis
Primary hazard: under-triage.
Mitigations:
- hard safety overrides
- physician review for ER/high risk
- continuous simulation + outcome learning
`);

  write("05_clinical_evaluation.md", `
# Clinical Evaluation
Validation via large-scale simulation + retrospective outcome comparison.
`);

  write("06_software_documentation.md", `
# Software Documentation
Versioned modules, audit logs, explainability graph, rule builder, and pack validation.
`);

  write("07_verification_validation.md", `
# Verification & Validation
Accuracy: ${((stats.accuracy || 0) * 100).toFixed(2)}%
Under-triage: ${(stats.underTriage || 0).toFixed(4)}
Simulations: ${stats.total || 0}
`);

  write("08_cybersecurity.md", `
# Cybersecurity
AES-256 encryption, TLS 1.2+, RBAC, PHI redaction, audit logging, rate limiting.
`);

  write("09_post_market_plan.md", `
# Post-Market Plan
Continuous learning loop, incident logging, periodic re-validation, physician override review.
`);

  // Build PDF
  pdfmetrics.registerFont(new UnicodeCIDFont("STSong-Light")); // Unicode-safe

  const doc = new SimpleDocTemplate(path.join(BASE, "fda_submission.pdf"));
  const story: any[] = [];

  const files = fs.readdirSync(BASE).filter(f => f.endsWith(".md")).sort();

  for (const f of files) {
    const text = fs.readFileSync(path.join(BASE, f), "utf-8");
    story.push(new Paragraph(text.replace(/\n/g, "<br/>")));
    story.push(new Spacer(1, 16));
  }

  doc.build(story);
  return { path: path.join(BASE, "fda_submission.pdf") };
}
🔌 Route
router.get("/compliance/fda/generate", requireRole(["admin"]), async (req, res) => {
  const out = await generateFDASubmission();
  res.download(out.path);
});

// server/validation/validate10k.ts

import pLimit from "p-limit";

export async function run10kValidation(repo) {
  const N = 10000;
  const limit = pLimit(20);
  const results: any[] = [];

  await Promise.all(
    Array.from({ length: N }).map((_, i) =>
      limit(async () => {
        const input = generateSyntheticCase();
        const result = await runClinicalPipeline(input, repo);
        const truth = generateGroundTruth(input);

        results.push({
          correct: result.diagnosis === truth.diagnosis,
          underTriage:
            truth.triage === "ER" && result.triage !== "ER"
        });
      })
    )
  );

  const total = results.length;
  const correct = results.filter(r => r.correct).length;
  const under = results.filter(r => r.underTriage).length;

  return {
    total,
    accuracy: correct / total,
    underTriage: under / total
  };
}

router.post("/validation/10k", requireRole(["admin"]), async (req, res) => {
