Clinical Brain Engine
server/brain/clinicalBrain.ts
export interface BrainContext {
  complaint: string
  patientData: any
  caseId: string
}

export class ClinicalBrain {

  engines: any[] = []

  registerEngine(engine: any) {
    this.engines.push(engine)
  }

  async run(context: BrainContext) {

    let state = context

    for (const engine of this.engines) {

      const start = Date.now()

      state = await engine.run(state)

      const duration = Date.now() - start

      console.log("Engine:", engine.name, "Duration:", duration)

    }

    return state
  }

}
