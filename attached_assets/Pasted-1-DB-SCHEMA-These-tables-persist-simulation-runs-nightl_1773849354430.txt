1. DB SCHEMA

These tables persist simulation runs, nightly control runs, and executive snapshots.

/server/db/executiveSchema.ts
import {
  pgTable,
  serial,
  text,
  timestamp,
  jsonb,
  numeric
} from "drizzle-orm/pg-core";

export const simulationRuns = pgTable("simulation_runs", {
  id: serial("id").primaryKey(),
  packId: text("pack_id").notNull(),
  complaint: text("complaint").notNull(),
  createdAt: timestamp("created_at").defaultNow().notNull(),
  strategyResults: jsonb("strategy_results").notNull()
});

export const controlRuns = pgTable("control_runs", {
  id: serial("id").primaryKey(),
  clinicId: text("clinic_id").notNull(),
  runAt: timestamp("run_at").defaultNow().notNull(),
  input: jsonb("input").notNull(),
  output: jsonb("output").notNull()
});

export const executiveSnapshots = pgTable("executive_snapshots", {
  id: serial("id").primaryKey(),
  clinicId: text("clinic_id").notNull(),
  snapshotDate: timestamp("snapshot_date").defaultNow().notNull(),
  totalCases: numeric("total_cases").notNull(),
  reviewedCases: numeric("reviewed_cases").notNull(),
  escalatedCases: numeric("escalated_cases").notNull(),
  overrideRate: numeric("override_rate").notNull(),
  avgSatisfaction: numeric("avg_satisfaction").notNull(),
  avgCostPerCase: numeric("avg_cost_per_case").notNull(),
  avgRevenuePerCase: numeric("avg_revenue_per_case").notNull(),
  complaintBreakdown: jsonb("complaint_breakdown").notNull(),
  physicianBreakdown: jsonb("physician_breakdown").notNull()
});
2. DB PERSISTENCE SERVICES
/server/services/executiveStore.ts
import { db } from "../db";
import {
  simulationRuns,
  controlRuns,
  executiveSnapshots
} from "../db/executiveSchema";

export async function saveSimulationRun(
  packId: string,
  complaint: string,
  strategyResults: unknown[]
) {
  const [row] = await db.insert(simulationRuns).values({
    packId,
    complaint,
    strategyResults
  }).returning();

  return row;
}

export async function listSimulationRuns(limit = 20) {
  return db.query.simulationRuns.findMany({
    orderBy: (t, { desc }) => [desc(t.createdAt)],
    limit
  });
}

export async function saveControlRun(
  clinicId: string,
  input: unknown,
  output: unknown
) {
  const [row] = await db.insert(controlRuns).values({
    clinicId,
    input,
    output
  }).returning();

  return row;
}

export async function listControlRuns(clinicId?: string, limit = 30) {
  if (clinicId) {
    return db.query.controlRuns.findMany({
      where: (t, { eq }) => eq(t.clinicId, clinicId),
      orderBy: (t, { desc }) => [desc(t.runAt)],
      limit
    });
  }

  return db.query.controlRuns.findMany({
    orderBy: (t, { desc }) => [desc(t.runAt)],
    limit
  });
}

export async function saveExecutiveSnapshot(payload: {
  clinicId: string;
  totalCases: number;
  reviewedCases: number;
  escalatedCases: number;
  overrideRate: number;
  avgSatisfaction: number;
  avgCostPerCase: number;
  avgRevenuePerCase: number;
  complaintBreakdown: unknown;
  physicianBreakdown: unknown;
}) {
  const [row] = await db.insert(executiveSnapshots).values({
    clinicId: payload.clinicId,
    totalCases: String(payload.totalCases),
    reviewedCases: String(payload.reviewedCases),
    escalatedCases: String(payload.escalatedCases),
    overrideRate: String(payload.overrideRate),
    avgSatisfaction: String(payload.avgSatisfaction),
    avgCostPerCase: String(payload.avgCostPerCase),
    avgRevenuePerCase: String(payload.avgRevenuePerCase),
    complaintBreakdown: payload.complaintBreakdown,
    physicianBreakdown: payload.physicianBreakdown
  }).returning();

  return row;
}

export async function listExecutiveSnapshots(clinicId: string, limit = 30) {
  return db.query.executiveSnapshots.findMany({
    where: (t, { eq }) => eq(t.clinicId, clinicId),
    orderBy: (t, { desc }) => [desc(t.snapshotDate)],
    limit
  });
}
3. DRILLDOWN ENGINES

These give complaint and physician drilldowns instead of just top-level numbers.

/server/services/executiveDrilldown.ts
export type ComplaintDrilldownRow = {
  complaint: string;
  totalCases: number;
  escalationRate: number;
  overrideRate: number;
  avgConfidence: number;
  avgSatisfaction: number;
};

export type PhysicianDrilldownRow = {
  physicianId: string;
  totalCases: number;
  avgReviewSeconds: number;
  overrideRate: number;
  avgSatisfaction: number;
  highRiskCases: number;
};

export function buildComplaintDrilldown(rows: ComplaintDrilldownRow[]) {
  return rows
    .map(r => {
      let status: "good" | "watch" | "critical" = "good";

      if (r.overrideRate > 0.18 || r.escalationRate > 0.2) status = "critical";
      else if (r.overrideRate > 0.1 || r.escalationRate > 0.12) status = "watch";

      return {
        ...r,
        status
      };
    })
    .sort((a, b) => b.totalCases - a.totalCases);
}

export function buildPhysicianDrilldown(rows: PhysicianDrilldownRow[]) {
  return rows
    .map(r => {
      let status: "good" | "watch" | "critical" = "good";

      if (r.overrideRate > 0.18 || r.avgSatisfaction < 4.0) status = "critical";
      else if (r.overrideRate > 0.1 || r.avgReviewSeconds > 30) status = "watch";

      return {
        ...r,
        status
      };
    })
    .sort((a, b) => b.totalCases - a.totalCases);
}
4. CHART-SHAPING ENGINE

This converts snapshots into chart-ready series.

/server/services/executiveCharts.ts
export function buildExecutiveChartSeries(
  snapshots: Array<{
    snapshotDate: string | Date;
    totalCases: string | number;
    overrideRate: string | number;
    avgSatisfaction: string | number;
    avgCostPerCase: string | number;
    avgRevenuePerCase: string | number;
  }>
) {
  const ordered = [...snapshots].sort(
    (a, b) => new Date(a.snapshotDate).getTime() - new Date(b.snapshotDate).getTime()
  );

  return ordered.map(s => {
    const revenue = Number(s.avgRevenuePerCase);
    const cost = Number(s.avgCostPerCase);
    const marginPct = revenue ? ((revenue - cost) / revenue) * 100 : 0;

    return {
      date: new Date(s.snapshotDate).toISOString().slice(0, 10),
      totalCases: Number(s.totalCases),
      overrideRatePct: Number((Number(s.overrideRate) * 100).toFixed(2)),
      avgSatisfaction: Number(Number(s.avgSatisfaction).toFixed(2)),
      avgCostPerCase: Number(Number(s.avgCostPerCase).toFixed(2)),
      avgRevenuePerCase: Number(Number(s.avgRevenuePerCase).toFixed(2)),
      marginPct: Number(marginPct.toFixed(2))
    };
  });
}
5. UPDATED ROUTES WITH DB + DRILLDOWNS
/server/routes/executiveDb.ts
import express from "express";
import {
  saveSimulationRun,
  listSimulationRuns,
  saveControlRun,
  listControlRuns,
  saveExecutiveSnapshot,
  listExecutiveSnapshots
} from "../services/executiveStore";
import { buildComplaintDrilldown, buildPhysicianDrilldown } from "../services/executiveDrilldown";
import { buildExecutiveChartSeries } from "../services/executiveCharts";
import { getSimulationPackById } from "../services/simulationPacks";
import { runSimulationLab } from "../services/simulationLab";
import { runAdaptiveControlLoop } from "../services/adaptiveControlLoop";

const router = express.Router();

router.post("/simulation-runs", async (req, res) => {
  const { packId, strategies = [] } = req.body;
  const pack = getSimulationPackById(packId);

  if (!pack) {
    return res.status(404).json({ error: "Pack not found" });
  }

  const results = runSimulationLab(pack.cases, strategies);
  const saved = await saveSimulationRun(pack.id, pack.complaint, results);

  res.json({ saved, results });
});

router.get("/simulation-runs", async (req, res) => {
  const limit = Number(req.query.limit || 20);
  const rows = await listSimulationRuns(limit);
  res.json(rows);
});

router.post("/control-runs", async (req, res) => {
  const output = runAdaptiveControlLoop(req.body);
  const saved = await saveControlRun(req.body.clinicId, req.body, output);
  res.json(saved);
});

router.get("/control-runs", async (req, res) => {
  const clinicId = typeof req.query.clinicId === "string" ? req.query.clinicId : undefined;
  const limit = Number(req.query.limit || 30);
  const rows = await listControlRuns(clinicId, limit);
  res.json(rows);
});

router.post("/snapshots", async (req, res) => {
  const saved = await saveExecutiveSnapshot(req.body);
  res.json(saved);
});

router.get("/snapshots/:clinicId", async (req, res) => {
  const rows = await listExecutiveSnapshots(req.params.clinicId, Number(req.query.limit || 30));
  res.json(rows);
});

router.post("/drilldown/complaints", async (req, res) => {
  res.json(buildComplaintDrilldown(req.body.rows || []));
});

router.post("/drilldown/physicians", async (req, res) => {
  res.json(buildPhysicianDrilldown(req.body.rows || []));
});

router.get("/charts/:clinicId", async (req, res) => {
  const rows = await listExecutiveSnapshots(req.params.clinicId, Number(req.query.limit || 30));
  res.json(buildExecutiveChartSeries(rows as any));
});

export default router;
