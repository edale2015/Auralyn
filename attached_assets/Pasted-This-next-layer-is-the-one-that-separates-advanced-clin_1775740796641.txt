import { runFinalPipeline } from "../clinical/finalPipeline";
import { db } from "../db";
import { auditStep } from "../audit/auditLogger";

export interface GoldenCase {
  caseId: string;
  input: any;
  expectedDisposition: string;
  expectedDiagnosis?: string;
}

export interface SimulationResult {
  totalCases: number;
  correctDisposition: number;
  incorrectDisposition: number;
  safetyMismatches: number;
  details: Array<{
    caseId: string;
    expected: string;
    actual: string;
    match: boolean;
    safetyMismatch: boolean;
  }>;
}

export async function runGoldenCaseSimulation(
  cases: GoldenCase[],
  traceId: string
): Promise<SimulationResult> {

  let correct = 0;
  let incorrect = 0;
  let safetyMismatches = 0;

  const details = [];

  for (const c of cases) {
    try {
      const result = await runFinalPipeline(c.input, traceId);

      const actual = result.safetyResult.finalDisposition;
      const expected = c.expectedDisposition;

      const match = actual === expected;

      if (match) correct++;
      else incorrect++;

      const safetyMismatch =
        expected === "ER_NOW" && actual !== "ER_NOW";

      if (safetyMismatch) safetyMismatches++;

      details.push({
        caseId: c.caseId,
        expected,
        actual,
        match,
        safetyMismatch,
      });

    } catch (err) {
      incorrect++;
      details.push({
        caseId: c.caseId,
        expected: c.expectedDisposition,
        actual: "ERROR",
        match: false,
        safetyMismatch: true,
      });
    }
  }

  const result: SimulationResult = {
    totalCases: cases.length,
    correctDisposition: correct,
    incorrectDisposition: incorrect,
    safetyMismatches,
    details,
  };

  await auditStep({
    traceId,
    step: "golden_case_simulation",
    input: { totalCases: cases.length },
    output: result,
    metadata: {},
  });

  return result;
}

export async function validateChangeWithGoldenCases(
  proposal: any,

  traceId: string
) {

  // HARD SAFETY BLOCK

    throw new Error(
      `[GOLDEN CASE BLOCK] Safety regression detected: ${result.safetyMismatches} cases lost ER_NOW detection`
    );
  }

  // QUALITY THRESHOLD
  const accuracy =
    result.correctDisposition / result.totalCases;

  if (accuracy < 0.95) {
    throw new Error(
      `[GOLDEN CASE BLOCK] Accuracy dropped to ${accuracy}`
    );
  }

  return {
    approved: true,
    metrics: result,
  };
}

import { getKbCache } from "./kbRuntime";
import Redis from "ioredis";

const redis = new Redis(process.env.REDIS_URL!);

const INSTANCE_ID = process.env.INSTANCE_ID || "instance-" + Math.random();

export async function broadcastKbVersion() {
  const version = getKbCache().version;

  await redis.publish("kb_version_channel", JSON.stringify({
    instanceId: INSTANCE_ID,
    version,
    ts: Date.now()
  }));
}

export function startKbVersionListener() {
  const sub = new Redis(process.env.REDIS_URL!);

  sub.subscribe("kb_version_channel");

  sub.on("message", (_, msg) => {
    const data = JSON.parse(msg);

    if (data.version !== getKbCache().version) {
      console.error(
        `[KB DRIFT DETECTED] Instance ${data.instanceId} has version ${data.version}, local=${getKbCache().version}`
      );
    }
  });
}

import { getKbCache } from "../kb/kbRuntime";

export function computeNextBestQuestion(
  differential: { diagnosis: string; probability: number }[],
  askedQuestions: string[]
) {

  const kb = getKbCache();

  const candidateQuestions = kb.priors
    .map(p => p.complaintId)
    .filter(q => !askedQuestions.includes(q));

  let bestQuestion = null;
  let bestScore = -Infinity;

  for (const q of candidateQuestions) {

    let infoGain = 0;

    for (const dx of differential) {
      infoGain += dx.probability * Math.random(); // placeholder
    }

    if (infoGain > bestScore) {
      bestScore = infoGain;
      bestQuestion = q;
    }
  }

  return {
    question: bestQuestion,
    expectedInformationGain: bestScore
  };
}

export async function runClinicalDebate(
  reasoningResult: any,
  safetyResult: any
) {

  const agents = [
    { name: "safety_agent", weight: 0.5 },
    { name: "diagnostic_agent", weight: 0.3 },
    { name: "risk_agent", weight: 0.2 },
  ];

  let score = 0;

  for (const agent of agents) {
    const vote = simulateAgentVote(agent.name, reasoningResult, safetyResult);
    score += vote * agent.weight;
  }

  const disagreement = Math.abs(score - 0.5);

  return {
    consensusScore: score,
    disagreement,
    escalate: disagreement > 0.3
  };
}

function simulateAgentVote(name: string, reasoning: any, safety: any) {
  if (name === "safety_agent") {
    return safety.finalDisposition === "ER_NOW" ? 1 : 0.7;
  }
  return Math.random();
}

export function computeCognitiveBudget(input: any) {

  let budget = 1;

  if (input.vitalSigns?.systolicBp < 100) budget += 2;
  if (input.ageYears > 65) budget += 1;
  if (input.symptoms.includes("chest_pain")) budget += 2;

  return {
    budgetLevel: Math.min(budget, 5),
    enableAdvancedReasoning: budget >= 3,
    enableDebate: budget >= 4
  };
}
