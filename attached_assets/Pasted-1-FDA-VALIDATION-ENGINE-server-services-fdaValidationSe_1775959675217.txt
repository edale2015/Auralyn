import { GoldenCaseRunResult } from "../types/clinical";

class FDAValidationService {
  generateReport(runs: GoldenCaseRunResult[]) {
    const total = runs.length;
    const passed = runs.filter(r => r.passed).length;

    const accuracy = total === 0 ? 0 : passed / total;

    const highRiskFailures = runs.filter(r =>
      !r.passed && r.mismatches.some(m => m.includes("ED now"))
    );

    return {
      totalCases: total,
      passed,
      failed: total - passed,
      accuracy,
      highRiskFailures: highRiskFailures.length,
      fdaReady: accuracy >= 0.8 && highRiskFailures.length === 0,
      generatedAt: new Date().toISOString(),
    };
  }
}

export const fdaValidationService = new FDAValidationService();

import crypto from "crypto";

interface HashRecord {
  id: number;
  timestamp: string;
  data: any;
  prevHash: string;
  hash: string;
}

class HashChain {
  private chain: HashRecord[] = [];

  private computeHash(record: Omit<HashRecord, "hash">) {
    return crypto
      .createHash("sha256")
      .update(JSON.stringify(record))
      .digest("hex");
  }

  add(data: any) {
    const prev = this.chain[this.chain.length - 1];

    const record: Omit<HashRecord, "hash"> = {
      id: this.chain.length + 1,
      timestamp: new Date().toISOString(),
      data,
      prevHash: prev?.hash || "GENESIS",
    };

    const hash = this.computeHash(record);

    const full: HashRecord = { ...record, hash };
    this.chain.push(full);

    return full;
  }

  verify() {
    for (let i = 1; i < this.chain.length; i++) {
      const prev = this.chain[i - 1];
      const current = this.chain[i];

      if (current.prevHash !== prev.hash) return false;

      const recalculated = this.computeHash({
        id: current.id,
        timestamp: current.timestamp,
        data: current.data,
        prevHash: current.prevHash,
      });

      if (recalculated !== current.hash) return false;
    }

    return true;
  }

  getChain() {
    return this.chain;
  }
}

export const auditHashChain = new HashChain();

📉 3. DRIFT DETECTION ENGINE

This detects when your system’s behavior changes over time.

📁 /server/services/driftDetectionService.ts
interface DriftMetric {
  complaint: string;
  avgConfidence: number;
  avgRisk: number;
}

class DriftDetectionService {
  private history: DriftMetric[] = [];

  record(metric: DriftMetric) {
    this.history.push(metric);
  }

  detect() {
    if (this.history.length < 10) return { drift: false };

    const recent = this.history.slice(-5);
    const older = this.history.slice(-10, -5);

    const avg = (arr: DriftMetric[]) =>
      arr.reduce((sum, m) => sum + m.avgConfidence, 0) / arr.length;

    const recentAvg = avg(recent);
    const olderAvg = avg(older);

    const diff = Math.abs(recentAvg - olderAvg);

    return {
      drift: diff > 0.1,
      difference: diff,
      recentAvg,
      olderAvg,
    };
  }
}

export const driftService = new DriftDetectionService();

import { useEffect, useState } from "react";

export default function SystemValidationDashboard() {
  const [fda, setFda] = useState<any>(null);
  const [drift, setDrift] = useState<any>(null);

  useEffect(() => {
    fetch("/api/fda/report")
      .then(res => res.json())
      .then(setFda);
  }, []);

  const checkDrift = async () => {
    const res = await fetch("/api/drift");
    setDrift(await res.json());
  };

  return (
    <div style={{ padding: 20 }}>
      <h1>⚖️ FDA Validation Dashboard</h1>

      <h2>📊 Validation Report</h2>
      <pre>{JSON.stringify(fda, null, 2)}</pre>

      <button onClick={checkDrift}>Check Drift</button>

      <h2>📉 Drift Detection</h2>
      <pre>{JSON.stringify(drift, null, 2)}</pre>
    </div>
  );
}
