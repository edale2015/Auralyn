// server/governance/governanceQueue.ts

import { db } from "../db";
import { governanceItems } from "@shared/schema";
import { eq, desc } from "drizzle-orm";

export async function addGovernanceItem(item: {
  id: string;
  sheet: string;
  change: any;
  risk: string;
  reason?: string;
}) {
  await db.insert(governanceItems).values({
    ...item,
    status: "pending",
  });
}

export async function listGovernanceQueue(filter?: { status?: string }) {
  return db
    .select()
    .from(governanceItems)
    .where(filter?.status ? eq(governanceItems.status, filter.status) : undefined)
    .orderBy(desc(governanceItems.createdAt));
}

export async function updateGovernanceStatus(
  id: string,
  status: "approved" | "rejected",
  reviewedBy: string
) {
  await db
    .update(governanceItems)
    .set({
      status,
      reviewedBy,
      reviewedAt: new Date(),
    })
    .where(eq(governanceItems.id, id));

  return true;
}

import { applyModelChange } from "./applyEngine";

export async function applyApprovedUpdate(itemId: string, reviewedBy: string) {
  const queue = await listGovernanceQueue({ status: "pending" });
  const item = queue.find((i) => i.id === itemId);

  if (!item) throw new Error("Item not found");

  await applyModelChange(item.change);

  await updateGovernanceStatus(itemId, "approved", reviewedBy);
