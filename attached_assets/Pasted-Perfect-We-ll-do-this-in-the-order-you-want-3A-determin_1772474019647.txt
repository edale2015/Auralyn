src/engine/dxPriority.ts
import fs from "node:fs";

export type DxPriorityMap = Map<string, Map<string, number>>; 
// complaint_slug -> (dx_id -> priority)

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQ = false;
  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"' && line[i + 1] === '"') { cur += '"'; i++; continue; }
    if (ch === '"') { inQ = !inQ; continue; }
    if (ch === "," && !inQ) { out.push(cur); cur = ""; continue; }
    cur += ch;
  }
  out.push(cur);
  return out.map(s => s.trim().replace(/^"|"$/g, ""));
}

export function loadDxPriority(csvPath: string): DxPriorityMap {
  if (!fs.existsSync(csvPath)) return new Map();
  const text = fs.readFileSync(csvPath, "utf8").trim();
  if (!text) return new Map();

  const lines = text.split("\n").map(l => l.trim()).filter(Boolean);
  const header = splitCsvLine(lines[0]);
  const idxComplaint = header.indexOf("Complaint_Slug");
  const idxDx = header.indexOf("Dx_ID");
  const idxPri = header.indexOf("Priority");

  const map: DxPriorityMap = new Map();

  for (const line of lines.slice(1)) {
    const cols = splitCsvLine(line);
    const complaint = cols[idxComplaint];
    const dx = cols[idxDx];
    const pri = Number(cols[idxPri] ?? "0");
    if (!complaint || !dx || Number.isNaN(pri)) continue;

    if (!map.has(complaint)) map.set(complaint, new Map());
    map.get(complaint)!.set(dx, pri);
  }
  return map;
}

export function getDxPriority(
  priMap: DxPriorityMap,
  complaintSlug: string,
  dxId: string
): number {
  return priMap.get(complaintSlug)?.get(dxId) ?? 0;
}
Winner selection helper

Drop this into your scoring module (where you pick top dx):

import { DxPriorityMap, getDxPriority } from "./dxPriority";

export type DxScore = { dxId: string; score: number };

export function pickWinnerDeterministic(
  complaintSlug: string,
  scores: DxScore[],
  priMap: DxPriorityMap
): DxScore | null {
  if (!scores.length) return null;

  // 1) max score
  let maxScore = -Infinity;
  for (const s of scores) maxScore = Math.max(maxScore, s.score);
  const tied = scores.filter(s => s.score === maxScore);

  if (tied.length === 1) return tied[0];

  // 2) highest dx priority
  let maxPri = -Infinity;
  for (const t of tied) maxPri = Math.max(maxPri, getDxPriority(priMap, complaintSlug, t.dxId));
  const priTied = tied.filter(t => getDxPriority(priMap, complaintSlug, t.dxId) === maxPri);

  if (priTied.length === 1) return priTied[0];

  // 3) stable alphabetical tie-break
  priTied.sort((a, b) => a.dxId.localeCompare(b.dxId));
  return priTied[0];
}

scripts/replay.ts
import fs from "node:fs";
import process from "node:process";

// import your engine entry point
import { runComplaint } from "../src/engine/runComplaint"; // adjust to your real path

const CASE_STORE = "data/case_store.jsonl";

function loadCase(caseId: string): any | null {
  if (!fs.existsSync(CASE_STORE)) return null;
  const lines = fs.readFileSync(CASE_STORE, "utf8").split("\n").filter(Boolean);
  for (const line of lines) {
    try {
      const obj = JSON.parse(line);
      if (obj.case_id === caseId) return obj;
    } catch {}
  }
  return null;
}

async function main() {
  const caseId = process.argv[2];
  if (!caseId) {
    console.error("Usage: tsx scripts/replay.ts <CASE_ID>");
    process.exit(2);
  }

  const c = loadCase(caseId);
  if (!c) {
    console.error(`Case not found: ${caseId}`);
    process.exit(1);
  }

  const res = await runComplaint({
    complaintSlug: c.complaint_slug,
    answers: c.answers
  });

  console.log(JSON.stringify({
    case_id: c.case_id,
    complaint_slug: c.complaint_slug,
    disposition: res.disposition,
    top_dx: res.top_dx,
    confidence: res.confidence,       // comes from 2D below
    explanation: res.explanation      // comes from 2D below
  }, null, 2));
}

main().catch(e => {
  console.error(e);
  process.exit(1);
});

LOW: margin ≤ 1 or many suppressors

top 5 suppressor hits
