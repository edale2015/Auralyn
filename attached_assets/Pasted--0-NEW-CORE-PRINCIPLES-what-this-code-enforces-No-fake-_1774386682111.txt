export function publish(channel: string, data: any) {
  redis.publish(channel, JSON.stringify(data));
}

export function subscribe(channel: string, handler: any) {
  const sub = redis.duplicate();
  sub.subscribe(channel);
  sub.on("message", (_, msg) => handler(JSON.parse(msg)));
}

  const result = await executeStep(step);

  const verified = await verifyStep(step);

  if (!verified) {
    throw new Error("Execution failed verification");
  }

}
📁 Retry logic
export async function retryExecution(step: any, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      return await executeWithVerification(step);
    } catch {
      if (i === retries - 1) throw;
    }
  }
}

export async function searchVectors(queryVec: number[]) {
  const keys = await redis.keys("vec:*");
  const results = [];

  for (const k of keys) {
    const v = JSON.parse(await redis.get(k)!);
    const score = cosineSimilarity(queryVec, v);
    results.push({ id: k, score });
  }

  return results.sort((a, b) => b.score - a.score).slice(0, 5);
}
