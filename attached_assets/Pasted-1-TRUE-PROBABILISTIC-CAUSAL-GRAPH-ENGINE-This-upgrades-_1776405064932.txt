type Node = {
  id: string;
  parents: string[];
  cpt: Record<string, number>; // conditional probability table
};

type Network = Record<string, Node>;

export function computePosterior(
  network: Network,
  evidence: Record<string, boolean>
) {
  const posterior: Record<string, number> = {};

  for (const nodeId in network) {
    const node = network[nodeId];

    let prob = 0.5;

    if (node.parents.length === 0) {
      prob = node.cpt["prior"] || 0.5;
    } else {
      const key = node.parents
        .map(p => (evidence[p] ? "1" : "0"))
        .join("");

      prob = node.cpt[key] || 0.5;
    }

    posterior[nodeId] = evidence[nodeId] !== undefined
      ? (evidence[nodeId] ? 1 : 0)
      : prob;
  }

  return posterior;
}

🏥 2) MULTI-SITE DEPLOYMENT MANAGER
📁 server/deployment/siteManager.ts
type Site = {
  id: string;
  name: string;
  activePatients: number;
  capacity: number;
};

const sites: Site[] = [
  { id: "NYC_UWS", name: "Upper West Side", activePatients: 5, capacity: 20 },
  { id: "NYC_HARLEM", name: "Harlem", activePatients: 10, capacity: 25 },
];

export function assignSite() {
  return sites
    .sort((a, b) => (a.activePatients / a.capacity) - (b.activePatients / b.capacity))[0];
}
📁 Deployment API
export function routeToSite(patient: any) {
  const site = assignSite();

  site.activePatients++;

  return {
    ...patient,
    assignedSite: site.id,
  };
}

export function simulateLegalCase(caseData: any) {
  const risks = [];

  if (caseData.unsafeUndercall) {
    risks.push("Failure to escalate high-risk patient");
  }

  if (caseData.missingData) {
    risks.push("Decision made with incomplete data");
  }

  if (caseData.noAuditTrail) {
    risks.push("No traceability of decision-making");
  }

  return {
    riskLevel: risks.length > 2 ? "HIGH" : "MEDIUM",
    issues: risks,
  };
}
📁 Legal report generator
export function generateLegalReport(caseData: any) {
  const sim = simulateLegalCase(caseData);

  return {
    summary: "Medical AI Legal Risk Assessment",
    riskLevel: sim.riskLevel,
    issues: sim.issues,
    recommendation:
      sim.riskLevel === "HIGH"
        ? "Requires physician oversight"
        : "Within acceptable bounds",
  };
}
