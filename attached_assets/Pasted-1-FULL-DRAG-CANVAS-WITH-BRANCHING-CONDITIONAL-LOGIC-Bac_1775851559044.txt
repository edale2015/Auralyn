import { getStep } from "./registry";

type Node = {
  id: string;
  name: string;
  next?: string;
  if?: { field: string; equals: any; then: string; else?: string };
};

export async function runBranchWorkflow(nodes: Node[], startId: string, input: any) {
  const map = Object.fromEntries(nodes.map(n => [n.id, n]));
  let current = map[startId];
  let ctx = { ...input };

  while (current) {
    const fn = getStep(current.name);
    if (!fn) throw new Error(`Missing step: ${current.name}`);
    ctx = await fn(ctx);

    if (current.if) {
      const val = ctx[current.if.field];
      const nextId = val === current.if.equals ? current.if.then : current.if.else;
      current = nextId ? map[nextId] : undefined;
    } else {
      current = current.next ? map[current.next] : undefined;
    }
  }
  return ctx;
}

const ACTIONS = {
  scale_workers: () => console.log("Scaling"),
  retrain: () => console.log("Retraining"),
  validate_templates: () => console.log("Validating")
};

export async function runHighAutonomy(state: any) {
  const level = autonomyLevel(state);
  const plan = [];

  if (state.ml.drift) plan.push("retrain");
  if (state.infrastructure.queueDepth > 50) plan.push("scale_workers");

  for (const a of plan) {
    if (level === "manual") continue;
    if (level === "assist" && a !== "validate_templates") continue;
    await ACTIONS[a]?.();
  }

  return { level, plan };
}
