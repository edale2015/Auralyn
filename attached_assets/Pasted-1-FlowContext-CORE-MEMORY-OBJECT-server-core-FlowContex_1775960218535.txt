// server/core/FlowContext.ts

export class FlowContext {
  private data: Record<string, any> = {}

  constructor(initial: Record<string, any> = {}) {
    this.data = { ...initial }
  }

  get<T>(key: string): T {
    if (!(key in this.data)) {
      throw new Error(`Missing key in context: ${key}`)
    }
    return this.data[key]
  }

  set(key: string, value: any) {
    this.data[key] = value
  }

  merge(other: FlowContext) {
    this.data = { ...this.data, ...other.data }
  }

  dump() {
    return this.data
  }
}

🧩 3. Example Agent (Red Flag Detector)
// server/agents/redFlagAgent.ts

import { MedicalAgent } from "../core/MedicalAgent"
import { FlowContext } from "../core/FlowContext"

export class RedFlagAgent extends MedicalAgent {
  constructor() {
    super({
      name: "redFlagAgent",
      consumes: ["symptoms", "vitals"],
      provides: ["redFlags"]
    })
  }

  async run(ctx: FlowContext): Promise<FlowContext> {
    const symptoms = ctx.get<string[]>("symptoms")
    const vitals = ctx.get<any>("vitals")

    const redFlags = []

    if (symptoms.includes("chest pain") && vitals.hr > 120) {
      redFlags.push("possible PE")
    }

    ctx.set("redFlags", redFlags)
    return ctx
  }
}
🧩 4. DAG Executor (CRITICAL UPGRADE)
// server/core/DAGExecutor.ts

import { MedicalAgent } from "./MedicalAgent"
import { FlowContext } from "./FlowContext"

export class DAGExecutor {
  constructor(private agents: MedicalAgent[]) {}

  validate() {
    const allProvides = new Set<string>()

    this.agents.forEach(agent => {
      agent.meta.provides.forEach(p => allProvides.add(p))
    })

    this.agents.forEach(agent => {
      agent.meta.consumes.forEach(c => {
        if (!allProvides.has(c)) {
          throw new Error(
            `Agent ${agent.meta.name} requires missing key: ${c}`
          )
        }
      })
    })
  }

  async run(ctx: FlowContext): Promise<FlowContext> {
    this.validate()

    for (const agent of this.agents) {
      const result = await agent.run(ctx)
      ctx.merge(result)
    }

    return ctx
  }
}

async runParallel(
  layers: MedicalAgent[][],
  ctx: FlowContext
): Promise<FlowContext> {
  for (const layer of layers) {
    const results = await Promise.all(
      layer.map(agent => agent.run(ctx))
    )

    results.forEach(r => ctx.merge(r))
  }

  return ctx
}
