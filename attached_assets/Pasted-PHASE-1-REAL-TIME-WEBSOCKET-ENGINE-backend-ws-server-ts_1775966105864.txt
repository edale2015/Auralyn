import { WebSocketServer } from "ws";

const wss = new WebSocketServer({ port: 8080 });

let patients: any[] = [];

function generateVitals() {
  return {
    hr: Math.floor(Math.random() * 40) + 60,
    spo2: Math.floor(Math.random() * 5) + 95,
    temp: (Math.random() * 2 + 97).toFixed(1),
    bp: `${110 + Math.floor(Math.random() * 20)}/${70 + Math.floor(Math.random() * 10)}`
  };
}

setInterval(() => {
  patients = Array.from({ length: 5 }).map((_, i) => ({
    id: i + 1,
    name: `Patient ${i + 1}`,
    vitals: generateVitals(),
    status:
      Math.random() > 0.85
        ? "critical"
        : Math.random() > 0.7
        ? "warning"
        : "stable"
  }));

  const payload = JSON.stringify({ type: "PATIENT_UPDATE", data: patients });

  wss.clients.forEach((client: any) => {
    if (client.readyState === 1) client.send(payload);
  });
}, 2000);

console.log("🟢 WebSocket server running on ws://localhost:8080");
🚀 PHASE 2 — FRONTEND WEBSOCKET HOOK
📁 /frontend/src/hooks/usePatientStream.ts
import { useEffect, useState } from "react";

export function usePatientStream() {
  const [patients, setPatients] = useState<any[]>([]);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:8080");

    ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      if (message.type === "PATIENT_UPDATE") {
        setPatients(message.data);
      }
    };

    return () => ws.close();
  }, []);

  return patients;
}

import { useEffect, useState } from "react";
import { api } from "../api/client";

export function useLiveSystem() {
  const [state, setState] = useState<any>({
    patients: [],
    decision: null,
    alerts: []
  });

  useEffect(() => {
    // REST pull
    api.get("/system/summary").then(res => {
      setState(prev => ({ ...prev, ...res.data }));
    });

    // WebSocket push
    const ws = new WebSocket("ws://localhost:8080");

    ws.onmessage = (event) => {
      const msg = JSON.parse(event.data);

      if (msg.type === "PATIENT_UPDATE") {
        setState(prev => ({ ...prev, patients: msg.data }));
      }

      if (msg.type === "ALERT") {
        setState(prev => ({
          ...prev,
          alerts: [msg.data, ...prev.alerts]
        }));
      }
    };

    return () => ws.close();
  }, []);

  return state;
}

import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

export async function generateInsights(patient: any) {
  const prompt = `
You are a clinical AI assistant.

Patient:
HR: ${patient.vitals.hr}
SpO2: ${patient.vitals.spo2}
Temp: ${patient.vitals.temp}

Tasks:
1. Identify risk level
2. Suggest next action
3. Highlight urgency

Return JSON:
{
  "risk": "...",
  "action": "...",
  "priority": "low | medium | high"
}
`;

  const res = await client.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{ role: "user", content: prompt }]
  });

  return JSON.parse(res.choices[0].message.content);
}

import { useEffect, useState } from "react";
import { api } from "../api/client";

export function AIInsightPanel({ patient }: any) {
  const [insight, setInsight] = useState<any>(null);

  useEffect(() => {
    if (!patient) return;

    api.post("/insights", patient).then(res => {
      setInsight(res.data);
    });
  }, [patient]);

  if (!insight) return null;

  return (
    <div className="card" style={{ marginTop: "12px" }}>
      <h4>🧠 AI Insight</h4>

      <p><b>Risk:</b> {insight.risk}</p>
      <p><b>Action:</b> {insight.action}</p>

      <p style={{
        color:
          insight.priority === "high"
            ? "red"
            : insight.priority === "medium"
            ? "orange"
            : "green"
      }}>
        Priority: {insight.priority.toUpperCase()}
      </p>
    </div>
  );
}
