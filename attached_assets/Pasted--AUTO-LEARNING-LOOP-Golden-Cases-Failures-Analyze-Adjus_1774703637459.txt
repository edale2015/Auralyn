🔥 PART 1 — WEIGHT STORE (PERSISTENT MEMORY)
File: server/learning/weightStore.ts
import fs from "fs";

const FILE = "./data/weights.json";

let weights: Record<string, number> = {};

export function loadWeights() {
  if (fs.existsSync(FILE)) {
    weights = JSON.parse(fs.readFileSync(FILE, "utf-8"));
  }
}

export function getWeight(key: string) {
  return weights[key] ?? 0;
}

export function updateWeight(key: string, delta: number) {
  weights[key] = (weights[key] ?? 0) + delta;
}

export function saveWeights() {
  fs.writeFileSync(FILE, JSON.stringify(weights, null, 2));
}
🔥 PART 2 — RLHF ENGINE
File: server/learning/rlhfEngine.ts
import { updateWeight, saveWeights } from "./weightStore";

export function applyRLHF(result: any) {
  const { expected, actual, trace } = result;

  if (!trace) return;

  const correct = expected?.diagnosis === actual?.diagnosis;

  trace.forEach((node: any) => {
    const key = node.name;

    if (correct) {
      updateWeight(key, +0.2); // reward
    } else {
      updateWeight(key, -0.3); // penalize
    }
  });

  saveWeights();
}
🔥 PART 3 — AUTO LEARNING LOOP
File: server/learning/autoLearningLoop.ts
import { applyRLHF } from "./rlhfEngine";

export async function runLearningCycle() {
  console.log("🧠 Learning cycle started");

  const cases = await getGoldenCases();
  let results = [];

  for (const c of cases) {
    const result = await runFullClinicalFlow(c.input);

    const enriched = {
      expected: c.expected,
      actual: result,
      trace: result.trace,
    };

    applyRLHF(enriched);

    results.push(enriched);
  }

  console.log("✅ Learning cycle complete");
  return results;
}
