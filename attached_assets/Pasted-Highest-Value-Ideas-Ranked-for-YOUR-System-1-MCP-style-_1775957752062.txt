// server/mcp/medicalMCP.ts

export interface MCPTool {
  name: string;
  description: string;
  execute: (input: any, context: MCPContext) => Promise<any>;
}

export interface MCPContext {
  patientId?: string;
  complaint?: string;
  authUser?: string;
  auditId?: string;
}

class MedicalMCPRegistry {
  private tools: Map<string, MCPTool> = new Map();

  register(tool: MCPTool) {
    this.tools.set(tool.name, tool);
  }

  async execute(toolName: string, input: any, context: MCPContext) {
    const tool = this.tools.get(toolName);
    if (!tool) throw new Error(`Tool ${toolName} not found`);
    return tool.execute(input, context);
  }
}

export const medicalMCP = new MedicalMCPRegistry();

// server/workflows/clinicalWorkflowEngine.ts

export async function runClinicalWorkflow(input) {
  const context = { complaint: input.complaint };

  const steps = [
    "intake.collect",
    "questions.nextBest",
    "diagnosis.run",
    "risk.assess",
    "disposition.determine",
    "ehr.document"
  ];

  let state = input;

  for (const step of steps) {
    state = await medicalMCP.execute(step, state, context);
  }

  return state;
}
