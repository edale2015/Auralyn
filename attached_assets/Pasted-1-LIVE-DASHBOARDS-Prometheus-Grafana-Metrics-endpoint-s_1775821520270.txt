const register = new client.Registry();

export const httpLatency = new client.Histogram({
  name: "http_latency_ms",

});

export const erRateGauge = new client.Gauge({
  name: "er_rate",
  help: "ER rate from simulation",
});

register.registerMetric(httpLatency);
register.registerMetric(erRateGauge);

export async function metricsHandler(_: any, res: any) {
  res.set("Content-Type", register.contentType);
  res.end(await register.metrics());
}

🔧 Simple admission risk model (upgradeable later)

export function predictAdmissionML(input: any) {
  let x = 0;

  x += input.ageYears > 65 ? 1.2 : 0;
  x += input.vitals?.systolicBp < 100 ? 1.5 : 0;
  x += input.vitals?.oxygenSaturation < 92 ? 2.0 : 0;
  x += input.complaint === "chest_pain" ? 1.3 : 0;

  const prob = 1 / (1 + Math.exp(-x));

  return {
    probability: prob,
    risk: prob > 0.7 ? "high" : prob > 0.4 ? "medium" : "low"
  };
}
