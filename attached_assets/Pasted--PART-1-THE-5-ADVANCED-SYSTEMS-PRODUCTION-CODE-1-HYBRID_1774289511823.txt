// server/memory/hybridMemory.ts

import { Pool } from "pg";

const pool = new Pool({ connectionString: process.env.DB_URL });

export async function storeMemory(node: any, embedding: number[]) {
  await pool.query(
    `INSERT INTO memory_nodes (id, type, data, embedding)
     VALUES ($1,$2,$3,$4)`,
    [node.id, node.type, node.data, embedding]
  );
}

export async function vectorSearch(embedding: number[]) {
  const res = await pool.query(
    `SELECT *, embedding <-> $1 AS distance
     FROM memory_nodes
     ORDER BY distance ASC LIMIT 5`,
    [embedding]
  );
  return res.rows;
}
2️⃣ 🎯 CLINICAL SCORING ENGINE (Centor, CURB-65)
// server/clinical/scoringEngine.ts

export function centorScore(input: any) {
  let score = 0;
  if (input.fever) score++;
  if (input.tonsillarExudate) score++;
  if (input.tenderNodes) score++;
  if (!input.cough) score++;

  return score;
}

export function curb65(input: any) {
  let score = 0;
  if (input.confusion) score++;
  if (input.urea > 7) score++;
  if (input.respRate >= 30) score++;
  if (input.bp < 90) score++;
  if (input.age >= 65) score++;

  return score;
}
Store scoring into memory
import { storeMemory } from "../memory/hybridMemory";

export async function logClinicalDecision(input: any) {
  const centor = centorScore(input);
  const curb = curb65(input);

  await storeMemory({
    id: `decision-${Date.now()}`,
    type: "clinical_decision",
    data: { input, centor, curb }
  }, input.embedding);
}

import { vectorSearch } from "../memory/hybridMemory";
import { centorScore } from "../clinical/scoringEngine";

export async function clinicalReasoning(input: any) {
  const similar = await vectorSearch(input.embedding);

  const score = centorScore(input);

  return {
    score,
    similarCases: similar,
    recommendation:
      score >= 3 ? "test_and_treat" : "supportive_care"
  };
}
