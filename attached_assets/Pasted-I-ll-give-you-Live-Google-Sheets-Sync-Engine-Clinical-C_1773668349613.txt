import {google} from "googleapis"

const sheets = google.sheets("v4")

export async function fetchSheet(spreadsheetId:string,range:string,auth:any){

 const response = await sheets.spreadsheets.values.get({

  spreadsheetId,
  range,
  auth

 })

 return response.data.values

}
Sheet Sync Engine

This fetches sheets and runs the ingestion pipeline.

server/sheets/sheetSyncEngine.ts

import {fetchSheet} from "./googleSheetsClient"
import {runSheetGraphPipeline} from "../ingestion/sheetToGraphPipeline"

export async function runLiveSheetSync(config:any){

 const results:any=[]

 for(const sheet of config.sheets){

   const data = await fetchSheet(
     config.spreadsheetId,
     sheet.range,
     config.auth
   )

   const result = runSheetGraphPipeline(data)

   results.push({
    sheet:sheet.name,
    result
   })

 }

 return results

}
Sheet Diff Engine

Detects what changed.

server/sheets/sheetDiffEngine.ts

export function diffSheets(oldRows:any[],newRows:any[]){

 const changes:any=[]

 const oldMap=new Map()

 oldRows.forEach(r=>{
  oldMap.set(JSON.stringify(r),true)
 })

 newRows.forEach(r=>{

  const key=JSON.stringify(r)

  if(!oldMap.has(key)){

   changes.push({
    type:"added",
    row:r
   })

  }

 })

 return changes

}

Change Impact Analyzer

Determines how serious a change is.

server/audit/changeImpactAnalyzer.ts

export function analyzeChangeImpact(change:any){

 if(change.sheet==="DISPOSITION_RULES"){

  return{
   severity:"critical",
   impact:"triage logic modified"
  }

 }

 if(change.sheet==="CORE_QUESTIONS"){

  return{
   severity:"medium",
   impact:"question flow changed"
  }

 }

 return{
  severity:"low",
  impact:"data update"
 }

}
