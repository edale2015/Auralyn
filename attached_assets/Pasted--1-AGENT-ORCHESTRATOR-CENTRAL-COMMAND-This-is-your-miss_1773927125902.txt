This is your mission control tower.

📁 /server/agents/orchestrator.ts
type Agent = {
  name: string;
  run: (context: any) => Promise<any>;
};

let agents: Agent[] = [];

export function registerAgent(agent: Agent) {
  agents.push(agent);
}

export async function runAgents(context: any) {
  let results: any = {};

  for (const agent of agents) {
    const output = await agent.run(context);
    results[agent.name] = output;
  }

  return results;
}

🔄 3. EVENT BUS (AGENTS TALK HERE)

Agents don’t just run… they react to events.

📁 /server/agents/eventBus.ts
type EventHandler = (data: any) => void;

let subscribers: Record<string, EventHandler[]> = {};

export function subscribe(event: string, handler: EventHandler) {
  if (!subscribers[event]) subscribers[event] = [];
  subscribers[event].push(handler);
}

export function publish(event: string, data: any) {
  if (subscribers[event]) {
    subscribers[event].forEach(fn => fn(data));
  }
}

export function decideNextActions(agentResults: any) {
  let actions: string[] = [];

  if (agentResults.safety?.alert === "ER_NOW") {
    actions.push("send_emergency_alert");
    return actions;
  }

  if (agentResults.diagnosis?.confidence < 0.6) {
    actions.push("request_physician_review");
  }

  actions.push("send_patient_response");

  return actions;
}
