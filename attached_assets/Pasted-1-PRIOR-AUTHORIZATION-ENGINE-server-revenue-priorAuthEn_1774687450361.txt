📊 4. POPULATION HEALTH DASHBOARD
📁 populationHealth.ts
const cases: any[] = [];

export function logPopulationCase(c: any) {
  cases.push(c);
}

export function getHeatmap() {
  const map: Record<string, number> = {};

  for (const c of cases) {
    const zip = c.zip || "unknown";
    map[zip] = (map[zip] || 0) + 1;
  }

  return map;
}
📁 API
router.get("/population/heatmap", (_, res) => {
  res.json(getHeatmap());
});

export function assignVariant(caseId: string) {
  return caseId.charCodeAt(0) % 2 === 0 ? "A" : "B";
}
🔌 Use in flow

const result =
  variant === "A"

    : await runPipelineV2(input);
