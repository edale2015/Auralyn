📁 server/revenue/reimbursementOptimizer.ts
import { getAllPayers } from "./payerRegistry";

export function optimizeBilling(caseData: {
  diagnosis: string;
  cptOptions: string[];
}) {
  const payers = getAllPayers();

  let best: any = null;

  for (const payer of payers) {
    for (const cpt of caseData.cptOptions) {
      const rate = payer.reimbursementRates[cpt] || 0;

      const adjusted =
        rate *
        (1 - payer.denialRate) *
        (1 / payer.avgDaysToPay);

      if (!best || adjusted > best.score) {
        best = {
          payer: payer.name,
          cpt,
          expectedRevenue: rate,
          score: adjusted
        };
      }
    }
  }

  return best;
}
📁 server/revenue/claimTracker.ts
type Claim = {
  claimId: string;
  payer: string;
  status: "submitted" | "paid" | "denied";
  amount: number;
};

const claims: Claim[] = [];

export function trackClaim(c: Claim) {
  claims.push(c);
}

export function getPayerPerformance() {
  const summary: any = {};

  for (const c of claims) {
    if (!summary[c.payer]) {
      summary[c.payer] = { paid: 0, denied: 0 };
    }

    if (c.status === "paid") summary[c.payer].paid++;
    if (c.status === "denied") summary[c.payer].denied++;
  }

  return summary;
}

import { getClinics, registerClinic } from "./clinicRegistry";

export function evaluateScaling() {
  const clinics = getClinics();

  for (const c of clinics) {
    const load = c.activeCases / (c.physicians * 10);

    if (load > 0.8) {
      const newClinic = {
        clinicId: "clinic_" + Date.now(),
        location: c.location,
        physicians: 2,
        activeCases: 0
      };

      registerClinic(newClinic);

      console.log("🚀 Spun up new clinic:", newClinic);
    }
  }
}

📁 Multi-Clinic Router
export function routeToClinic(caseData: any) {
  const clinics = getClinics();

  clinics.sort((a, b) =>
    (a.activeCases / a.physicians) -
    (b.activeCases / b.physicians)
  );

  return clinics[0];
}
