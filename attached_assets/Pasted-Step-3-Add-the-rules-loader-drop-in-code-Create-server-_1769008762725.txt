import { google } from "googleapis";

type CacheEntry = { expiresAt: number; value: Record<string, any> };
const CACHE: CacheEntry = { expiresAt: 0, value: {} };
const TTL_MS = 5 * 60 * 1000;

function envOrThrow(name: string): string {
  const v = process.env[name];
  if (!v) throw new Error(`Missing env var: ${name}`);
  return v;
}

function parseValue(valueType: string, raw: any): any {
  const t = String(valueType || "").trim().toLowerCase();
  const s = String(raw ?? "").trim();
  if (t === "number") {
    const n = Number(s);
    return Number.isFinite(n) ? n : null;
  }
  if (t === "boolean") return s.toLowerCase() === "true";
  // text/json fallback
  if (t === "json") {
    try { return JSON.parse(s); } catch { return null; }
  }
  return s;
}

export async function getEntFluRulesFromSheet(): Promise<Record<string, any>> {
  const now = Date.now();
  if (CACHE.expiresAt > now) return CACHE.value;

  const spreadsheetId = envOrThrow("SHEETS_SPREADSHEET_ID");
  const credsPath = process.env.GOOGLE_APPLICATION_CREDENTIALS;

  const auth = new google.auth.GoogleAuth({
    scopes: ["https://www.googleapis.com/auth/spreadsheets.readonly"],
    keyFile: credsPath || undefined,
  });

  const sheets = google.sheets({ version: "v4", auth });
  const range = `ENT_FLU_RULES!A1:E500`;

  const resp = await sheets.spreadsheets.values.get({
    spreadsheetId,
    range,
    valueRenderOption: "UNFORMATTED_VALUE",
  });

  const values = resp.data.values || [];
  if (values.length < 2) throw new Error("ENT_FLU_RULES is empty.");

  const header = values[0].map((h) => String(h ?? "").trim());
  const idx = (name: string) => header.indexOf(name);

  const iKey = idx("rule_key");
  const iType = idx("value_type");
  const iVal = idx("value");
  const iActive = idx("active");

  if (iKey < 0 || iType < 0 || iVal < 0 || iActive < 0) {
    throw new Error("ENT_FLU_RULES missing required headers.");
  }

  const rules: Record<string, any> = {};
  for (const row of values.slice(1)) {
    const active = String(row[iActive] ?? "").trim().toUpperCase() === "Y";
    if (!active) continue;

    const key = String(row[iKey] ?? "").trim();
    if (!key) continue;

    const type = String(row[iType] ?? "").trim();
    const val = parseValue(type, row[iVal]);

    rules[key] = val;
  }

  CACHE.value = rules;
  CACHE.expiresAt = now + TTL_MS;
  return rules;
}

export function invalidateEntFluRulesCache() {
  CACHE.expiresAt = 0;
  CACHE.value = {};
}

export async function computeEntFluProposal(answers: any) {

  }

  const requireFever = rules.TAMIFLU_REQUIRE_FEVER ?? defaults.TAMIFLU_REQUIRE_FEVER;

  const redFlagDisposition = rules.RED_FLAG_DISPOSITION ?? defaults.RED_FLAG_DISPOSITION;
  const nonRedFlagDisposition = rules.NON_RED_FLAG_DISPOSITION ?? defaults.NON_RED_FLAG_DISPOSITION;

  const proposeCovidTest = rules.PROPOSE_COVID_TEST ?? defaults.PROPOSE_COVID_TEST;
  const proposeFluTestIfTamiflu = rules.PROPOSE_FLU_TEST_IF_TAMIFLU ?? defaults.PROPOSE_FLU_TEST_IF_TAMIFLU;

  const redFlag =
    !!answers.RF_SOB || !!answers.RF_CP || !!answers.RF_NEURO || !!answers.RF_DEHY;

  const onsetDays = typeof answers.ONSET_DAYS === "number" ? answers.ONSET_DAYS : null;

  const feverOk = requireFever ? !!answers.FEVER : true;
  const achesOk = requireAches ? !!answers.ACHES : true;

  const tamifluEligible =
    !redFlag &&
    onsetDays !== null &&
    onsetDays <= tamifluMaxDays &&
    feverOk &&
    achesOk;

  const paxlovidFlag = answers.COVID_POS === "yes";

  const meds: string[] = ["acetaminophen", "saline nasal spray", "guaifenesin"];
  const avoid: string[] = [];

  if (answers.SSRI) avoid.push("dextromethorphan");
  if (answers.HTN || answers.ANXIETY) avoid.push("pseudoephedrine / phenylephrine");

    redFlag,
    tamifluEligible,
    paxlovidFlag,
    disposition,
