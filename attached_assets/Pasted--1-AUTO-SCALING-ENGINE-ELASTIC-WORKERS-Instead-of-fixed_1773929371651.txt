📈 1. AUTO-SCALING ENGINE (ELASTIC WORKERS)

Instead of fixed workers, we dynamically scale based on load.

🧠 Scaling Logic
Queue Size → Worker Count

0–100     → 1 worker
100–500   → 3 workers
500–2000  → 10 workers
2000+     → 25+ workers
📁 /server/scaling/autoScaler.ts
import { exec } from "child_process";
import { intakeQueue } from "../queue/queue";

export async function autoScale() {
  const counts = await intakeQueue.getJobCounts("waiting");
  const waiting = counts.waiting || 0;

  let targetWorkers = 1;

  if (waiting > 2000) targetWorkers = 25;
  else if (waiting > 500) targetWorkers = 10;
  else if (waiting > 100) targetWorkers = 3;

  exec(`bash scale_workers.sh ${targetWorkers}`);
}

export function updateClinicModel(clinicId: string, outcome: any) {
  const profile = clinicProfiles[clinicId];

  if (!profile) return;

  const { diagnosis, correct } = outcome;

  if (!profile.weights[diagnosis]) {
    profile.weights[diagnosis] = 0;
  }

  profile.weights[diagnosis] += correct ? 1 : -1;
}
🧠 APPLY CLINIC TUNING
export function adjustDiagnosisForClinic(clinicId: string, dxList: any[]) {
  const profile = clinicProfiles[clinicId];

  if (!profile) return dxList;

  return dxList.sort(
    (a, b) =>
      (profile.weights[b.name] || 0) -
      (profile.weights[a.name] || 0)
  );
}

let activeCalls = 0;
const MAX_CALLS = 500;

export function routeCall() {
  if (activeCalls >= MAX_CALLS) {
    return { status: "queue" };
  }

  activeCalls++;
  return { status: "connected" };
}

export function endCall() {
  activeCalls--;
}
📁 /server/voice/conversationEngine.ts
export async function handleConversation(input: any) {
  const { text, history } = input;

  const next = generateNextStep(text, history);

  return {
    response: next,
    continue: true,
  };
}

function generateNextStep(text: string, history: any[]) {
  if (text.includes("pain")) {
    return "Can you rate your pain from 1 to 10?";
  }

  return "Tell me more about your symptoms.";
}
