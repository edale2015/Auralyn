/**
 * scripts/apply-profile-pack.ts
 *
 * Step 3: Apply Profile Pack
 * - Activates a targeted subset of inert CSR rows for a complaint
 * - Sets CLUSTER_ID tier, WHEN_EXPR, POINTS
 * - Ensures tier priorities exist in DX_PRIORITY
 *
 * Usage:
 *   npx tsx scripts/apply-profile-pack.ts <cc_id> <PROFILE_ID>
 *   npx tsx scripts/apply-profile-pack.ts sinus_pressure ENT_SINUS_V1 --dry-run
 */

import fs from "fs";
import path from "path";

type Args = { ccId: string; profileId: string; dryRun: boolean };

function parseArgs(argv: string[]): Args {
  const ccId = argv[0];
  const profileId = argv[1];
  if (!ccId || !profileId) {
    console.error("Usage: npx tsx scripts/apply-profile-pack.ts <cc_id> <PROFILE_ID> [--dry-run]");
    process.exit(2);
  }
  return { ccId, profileId, dryRun: argv.includes("--dry-run") };
}

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQuotes = false;

  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"') {
      if (inQuotes && line[i + 1] === '"') {
        cur += '"';
        i++;
      } else {
        inQuotes = !inQuotes;
      }
      continue;
    }
    if (ch === "," && !inQuotes) {
      out.push(cur);
      cur = "";
      continue;
    }
    cur += ch;
  }
  out.push(cur);
  return out.map((s) => s.trim());
}

function readCsv(filePath: string) {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return { headers: [] as string[], rows: [] as Record<string, string>[] };

  const headers = splitCsvLine(lines[0]);
  const rows: Record<string, string>[] = [];

  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    if (cols.every((c) => c === "")) continue;
    const r: Record<string, string> = {};
    for (let j = 0; j < headers.length; j++) r[headers[j]] = cols[j] ?? "";
    rows.push(r);
  }
  return { headers, rows };
}

function writeCsv(filePath: string, headers: string[], rows: Record<string, string>[], dryRun: boolean) {
  const lines: string[] = [];
  lines.push(headers.join(","));
  for (const r of rows) {
    const cols = headers.map((h) => {
      const v = r[h] ?? "";
      if (v.includes(",") || v.includes('"') || v.includes("\n")) return `"${v.replace(/"/g, '""')}"`;
      return v;
    });
    lines.push(cols.join(","));
  }
  if (dryRun) {
    console.log(`[DRY] Would write ${rows.length} rows -> ${filePath}`);
    return;
  }
  fs.writeFileSync(filePath, lines.join("\n") + "\n", "utf8");
}

type Profile = {
  cluster_prefix: string;
  tiers: Record<string, { points: number; priority: number }>;
  activate: Array<{ evidence: string; tier: "PRIMARY" | "SECONDARY" | "BENIGN"; when: string; points?: number }>;
};

function mustHave(headers: string[], needed: string[], fileLabel: string) {
  for (const h of needed) {
    if (!headers.includes(h)) throw new Error(`${fileLabel} missing header: ${h}`);
  }
}

function main() {
  const { ccId, profileId, dryRun } = parseArgs(process.argv.slice(2));
  const root = process.cwd();

  const PROFILE_PATH = path.join(root, "data", "complaints", "profile_packs.json");
  const CSR_PATH = path.join(root, "server", "data", "csv", "CLUSTER_SCORING_RULES.csv");
  const DXP_PATH = path.join(root, "server", "data", "csv", "DX_PRIORITY.csv");

  if (!fs.existsSync(PROFILE_PATH)) throw new Error(`Missing profile file: ${PROFILE_PATH}`);
  if (!fs.existsSync(CSR_PATH)) throw new Error(`Missing: ${CSR_PATH}`);
  if (!fs.existsSync(DXP_PATH)) throw new Error(`Missing: ${DXP_PATH}`);

  const profiles = JSON.parse(fs.readFileSync(PROFILE_PATH, "utf8")) as Record<string, Profile>;
  const profile = profiles[profileId];
  if (!profile) throw new Error(`Profile not found: ${profileId}`);

  // ---- CSR updates ----
  const csr = readCsv(CSR_PATH);
  mustHave(csr.headers, ["CC_ID", "CLUSTER_ID", "RULE_ID", "POINTS", "WHEN_EXPR", "EVIDENCE_LABEL"], "CLUSTER_SCORING_RULES.csv");

  const tierClusterId = (tier: string) => `CL_${profile.cluster_prefix}_${tier}`;

  let csrTouched = 0;
  let csrMissing = 0;
  const changes: string[] = [];

  for (const item of profile.activate) {
    const target = csr.rows.find((r) => r.CC_ID === ccId && r.EVIDENCE_LABEL === item.evidence);
    if (!target) {
      csrMissing++;
      changes.push(`CSR MISSING: CC_ID=${ccId} evidence=${item.evidence} (no row found to update)`);
      continue;
    }

    const newClusterId = tierClusterId(item.tier);
    const newWhen = item.when;
    const newPoints = String(item.points ?? profile.tiers[item.tier]?.points ?? 0);

    const before = { CLUSTER_ID: target.CLUSTER_ID, WHEN_EXPR: target.WHEN_EXPR, POINTS: target.POINTS };

    let touched = false;
    if (target.CLUSTER_ID !== newClusterId) { target.CLUSTER_ID = newClusterId; touched = true; }
    if (target.WHEN_EXPR !== newWhen) { target.WHEN_EXPR = newWhen; touched = true; }
    if (target.POINTS !== newPoints) { target.POINTS = newPoints; touched = true; }

    if (touched) {
      csrTouched++;
      changes.push(
        `CSR UPDATE: ${ccId} evidence=${item.evidence} tier=${item.tier} ` +
        `CLUSTER_ID ${before.CLUSTER_ID} -> ${newClusterId}; ` +
        `POINTS ${before.POINTS} -> ${newPoints}; ` +
        `WHEN_EXPR ${before.WHEN_EXPR} -> ${newWhen}`
      );
    }
  }

  // ---- DX_PRIORITY ensure tier rows ----
  const dxp = readCsv(DXP_PATH);
  mustHave(dxp.headers, ["CC_ID", "CLUSTER_ID", "PRIORITY"], "DX_PRIORITY.csv");

  const tierRows = (["PRIMARY", "SECONDARY", "BENIGN"] as const).map((tier) => {
    return {
      CC_ID: ccId,
      CLUSTER_ID: tierClusterId(tier),
      PRIORITY: String(profile.tiers[tier]?.priority ?? (tier === "PRIMARY" ? 100 : tier === "SECONDARY" ? 70 : 50))
    };
  });

  let dxpAdded = 0;
  let dxpUpdated = 0;

  for (const tr of tierRows) {
    const existing = dxp.rows.find((r) => r.CC_ID === ccId && r.CLUSTER_ID === tr.CLUSTER_ID);
    if (!existing) {
      dxp.rows.push(tr);
      dxpAdded++;
      changes.push(`DXP ADD: ${ccId} ${tr.CLUSTER_ID} PRIORITY=${tr.PRIORITY}`);
    } else if (existing.PRIORITY !== tr.PRIORITY) {
      const before = existing.PRIORITY;
      existing.PRIORITY = tr.PRIORITY;
      dxpUpdated++;
      changes.push(`DXP UPDATE: ${ccId} ${tr.CLUSTER_ID} PRIORITY ${before} -> ${tr.PRIORITY}`);
    }
  }

  // write changes
  writeCsv(CSR_PATH, csr.headers, csr.rows, dryRun);
  writeCsv(DXP_PATH, dxp.headers, dxp.rows, dryRun);

  // summary
  console.log("\n=== Profile Pack Applied ===");
  console.log(`CC_ID: ${ccId}`);
  console.log(`PROFILE: ${profileId}`);
  console.log(`CSR updated: ${csrTouched}`);
  console.log(`CSR missing targets: ${csrMissing}`);
  console.log(`DX_PRIORITY added: ${dxpAdded}`);
  console.log(`DX_PRIORITY updated: ${dxpUpdated}`);
  if (changes.length) {
    console.log("\nChanges:");
    for (const c of changes) console.log(`- ${c}`);
  } else {
    console.log("\nNo changes (already applied).");
  }
}

main();

    maxN++;

    csr.rows.push({

      RULE_ID: ruleId,
      POINTS: "0",
      WHEN_EXPR: "false",
      EVIDENCE_LABEL: ev

    added++;
    changes.push(`ADD CSR: ${ccId} evidence=${ev} RULE_ID=${ruleId} CLUSTER_ID=${stubClusterId}`);

  writeCsv(CSR_PATH, csr.headers, csr.rows, dryRun);

  } else {
    console.log("\nNo changes (all targets already exist).");
  }
}

function readSeed(filePath: string): Array<{ CC_ID: string; PROFILE_ID: string }> {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);

import fs from "fs";
import path from "path";
import { spawnSync } from "child_process";

type Args = {

  continueOnFail: boolean;

function parseArgs(argv: string[]): Args {

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQuotes = false;

  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"') {
      if (inQuotes && line[i + 1] === '"') {
        cur += '"';
        i++;
      } else {
        inQuotes = !inQuotes;
      }
      continue;
    }
    if (ch === "," && !inQuotes) {
      out.push(cur);
      cur = "";
      continue;
    }
    cur += ch;
  }
  out.push(cur);
  return out.map((s) => s.trim());
}

function readSeed(filePath: string): Array<{ CC_ID: string; PROFILE_ID: string }> {

  if (!lines.length) return [];

  const idxCc = headers.indexOf("CC_ID");
  const idxPf = headers.indexOf("PROFILE_ID");

  if (idxCc < 0 || idxPf < 0) {
    throw new Error(`Seed CSV must have headers CC_ID,PROFILE_ID (got: ${headers.join(",")})`);
  }

  const rows: Array<{ CC_ID: string; PROFILE_ID: string }> = [];
  for (let i = 1; i < lines.length; i++) {

    const pf = (cols[idxPf] ?? "").trim();
    if (!cc || !pf) continue;
    rows.push({ CC_ID: cc, PROFILE_ID: pf });
  }

}

type StepResult = {

  profileId: string;

function run(cmd: string, args: string[], dryRun: boolean): StepResult {
  const stepName: StepResult["step"] =
    args.includes("scripts/ensure-profile-rows.ts") ? "ensure-profile-rows" : "apply-profile-pack";

function applyFilters(
  rows: Array<{ CC_ID: string; PROFILE_ID: string }>,
  args: Args

function readCsv(filePath: string) {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return { headers: [] as string[], rows: [] as Record<string, string>[] };

  const headers = splitCsvLine(lines[0]);
  const rows: Record<string, string>[] = [];
  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    if (cols.every((c) => c === "")) continue;
    const r: Record<string, string> = {};
    for (let j = 0; j < headers.length; j++) r[headers[j]] = cols[j] ?? "";
    rows.push(r);
  }
  return { headers, rows };
}

function writeCsv(filePath: string, headers: string[], rows: Record<string, string>[], dryRun: boolean) {
  const lines: string[] = [];
  lines.push(headers.join(","));
  for (const r of rows) {
    const cols = headers.map((h) => {
      const v = r[h] ?? "";
      if (v.includes(",") || v.includes('"') || v.includes("\n")) return `"${v.replace(/"/g, '""')}"`;
      return v;
    });
    lines.push(cols.join(","));
  }
  if (dryRun) {
    console.log(`[DRY] Would write ${rows.length} rows -> ${filePath}`);
    return;
  }
  fs.writeFileSync(filePath, lines.join("\n") + "\n", "utf8");
}

function makeLimiter(n: number) {
  let active = 0;
  const queue: Array<() => void> = [];
  const next = () => {
    if (active >= n) return;
    const job = queue.shift();
    if (!job) return;
    active++;
    job();
  };
  return function limit<T>(fn: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      queue.push(() => {
        fn()
          .then(resolve)
          .catch(reject)
          .finally(() => {
            active--;
            next();
          });
      });
      next();
    });
  };
}

  for (const r of csr.rows) {
    const ccId = r.CC_ID;
    if (!ccId) continue;
    if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
    if (r.RULE_ID) csrRuleIdsByCc.get(ccId)!.add(r.RULE_ID);

    if (r.EVIDENCE_LABEL) {
      csrByCcEvidence.set(`${ccId}||${r.EVIDENCE_LABEL}`, r);
    }
  }

  // Index DXP
  const dxpByCcCluster = new Map<string, Record<string, string>>(); // key = `${ccId}||${clusterId}`
  for (const r of dxp.rows) {
    if (r.CC_ID && r.CLUSTER_ID) dxpByCcCluster.set(`${r.CC_ID}||${r.CLUSTER_ID}`, r);
  }

  const limiter = makeLimiter(args.parallel);

  console.log(`\n=== ULTRA Bulk Profile Apply (${seed.length} rows) ===`);
  console.log(`parallel planning: ${args.parallel}`);
  if (args.dryRun) console.log("Mode: DRY RUN");
  if (args.continueOnFail) console.log("Mode: CONTINUE ON FAIL");
  if (args.onlyProfile) console.log(`Filter: only-profile=${args.onlyProfile}`);
  if (args.ccs.length) console.log(`Filter: cc in [${args.ccs.join(", ")}]`);
  if (args.summaryJsonPath) console.log(`Will write: ${args.summaryJsonPath}`);
  console.log("");

  const plans: ItemPlan[] = [];

  const runOne = async (ccId: string, profileId: string): Promise<ItemPlan> => {
    const p: ItemPlan = {
      ccId,
      profileId,
      ok: true,
      errors: [],
      csrAdded: 0,
      csrUpdated: 0,
      dxpAdded: 0,
      dxpUpdated: 0,
      notes: [],
    };

    const profile = profiles[profileId];
    if (!profile) {
      p.ok = false;
      p.errors.push(`Profile not found: ${profileId}`);
      return p;
    }

    const prefix = profile.cluster_prefix;

    const tierClusterId = (tier: "PRIMARY" | "SECONDARY" | "BENIGN") => `CL_${prefix}_${tier}`;

    // ---- ENSURE CSR rows exist for activation targets ----
    // RULE_ID format: CSR_<prefix>_<NN> (unique within CC_ID + prefix)
    const ruleIdRe = new RegExp(`^CSR_${prefix}_(\\d+)$`);
    let maxN = 0;
    const existingRuleIds = csrRuleIdsByCc.get(ccId) ?? new Set<string>();
    for (const rid of existingRuleIds) {
      const m = rid.match(ruleIdRe);
      if (m) {
        const n = parseInt(m[1], 10);
        if (!Number.isNaN(n)) maxN = Math.max(maxN, n);
      }
    }

    for (const item of profile.activate) {
      const key = `${ccId}||${item.evidence}`;
      const existing = csrByCcEvidence.get(key);

      if (!existing) {
        maxN++;
        const ruleId = `CSR_${prefix}_${String(maxN).padStart(2, "0")}`;

        const newRow: Record<string, string> = {
          CC_ID: ccId,
          CLUSTER_ID: tierClusterId("BENIGN"), // placeholder until apply
          RULE_ID: ruleId,
          POINTS: "0",
          WHEN_EXPR: "false",
          EVIDENCE_LABEL: item.evidence,
        };

        csr.rows.push(newRow);
        csrByCcEvidence.set(key, newRow);

        if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
        csrRuleIdsByCc.get(ccId)!.add(ruleId);

        p.csrAdded++;
        p.notes.push(`CSR add evidence=${item.evidence} RULE_ID=${ruleId}`);
      }
    }

    // ---- APPLY targeted activations ----
    for (const item of profile.activate) {
      const row = csrByCcEvidence.get(`${ccId}||${item.evidence}`);
      if (!row) {
        // should not happen after ensure
        p.ok = false;
        p.errors.push(`CSR missing after ensure for evidence=${item.evidence}`);
        if (!args.continueOnFail) return p;
        continue;
      }

      const wantCluster = tierClusterId(item.tier);
      const wantWhen = item.when;
      const wantPoints = String(item.points ?? profile.tiers[item.tier].points);

      let changed = false;
      if (row.CLUSTER_ID !== wantCluster) { row.CLUSTER_ID = wantCluster; changed = true; }
      if (row.WHEN_EXPR !== wantWhen) { row.WHEN_EXPR = wantWhen; changed = true; }
      if (row.POINTS !== wantPoints) { row.POINTS = wantPoints; changed = true; }

      if (changed) p.csrUpdated++;
    }

    // ---- ENSURE DXP tiers exist ----
    for (const tier of ["PRIMARY", "SECONDARY", "BENIGN"] as const) {
      const clusterId = tierClusterId(tier);
      const wantPriority = String(profile.tiers[tier].priority);
      const key = `${ccId}||${clusterId}`;

      const existing = dxpByCcCluster.get(key);
      if (!existing) {
        const newRow: Record<string, string> = {
          CC_ID: ccId,
          CLUSTER_ID: clusterId,
          PRIORITY: wantPriority,
        };
        dxp.rows.push(newRow);
        dxpByCcCluster.set(key, newRow);
        p.dxpAdded++;
      } else if (existing.PRIORITY !== wantPriority) {
        existing.PRIORITY = wantPriority;
        p.dxpUpdated++;
      }
    }

    return p;
  };

  const tasks = seed.map((r) =>
    limiter(async () => runOne(r.CC_ID, r.PROFILE_ID))
      .then((p) => {
        plans.push(p);
        const status = p.ok ? "✅" : "❌";
        console.log(`${status} ${p.ccId} -> ${p.profileId} | CSR +${p.csrAdded} upd ${p.csrUpdated} | DXP +${p.dxpAdded} upd ${p.dxpUpdated}`);
        if (!p.ok && p.errors.length) {
          for (const e of p.errors) console.log(`   - ${e}`);
        }
        if (!p.ok && !args.continueOnFail) {
          throw new Error(`Stopping on failure (use --continue-on-fail): ${p.ccId}`);
        }
      })
  );

  Promise.allSettled(tasks).then((settled) => {
    const hadStop = settled.some((s) => s.status === "rejected");

    const okCount = plans.filter((p) => p.ok).length;
    const failCount = plans.length - okCount;

    console.log("\n=== ULTRA Summary ===");
    console.log(`attempted: ${plans.length}`);
    console.log(`ok: ${okCount}`);
    console.log(`fail: ${failCount}`);
    console.log(`CSR added: ${plans.reduce((a, p) => a + p.csrAdded, 0)}`);
    console.log(`CSR updated: ${plans.reduce((a, p) => a + p.csrUpdated, 0)}`);
    console.log(`DXP added: ${plans.reduce((a, p) => a + p.dxpAdded, 0)}`);
    console.log(`DXP updated: ${plans.reduce((a, p) => a + p.dxpUpdated, 0)}`);

    // Write summary JSON if requested
    const summary = {
      seedPath: args.seedPath,
      dryRun: args.dryRun,
      continueOnFail: args.continueOnFail,
      onlyProfile: args.onlyProfile ?? null,
      ccs: args.ccs,
      parallel: args.parallel,
      attempted: plans.length,
      ok: okCount,
      fail: failCount,
      totals: {
        csrAdded: plans.reduce((a, p) => a + p.csrAdded, 0),
        csrUpdated: plans.reduce((a, p) => a + p.csrUpdated, 0),
        dxpAdded: plans.reduce((a, p) => a + p.dxpAdded, 0),
        dxpUpdated: plans.reduce((a, p) => a + p.dxpUpdated, 0),
      },
      results: plans,
      generated_at: new Date().toISOString(),
    };

    if (args.summaryJsonPath) {
      if (args.dryRun) {
        console.log(`[DRY] Would write summary JSON -> ${args.summaryJsonPath}`);
      } else {
        fs.mkdirSync(path.dirname(args.summaryJsonPath), { recursive: true });
        fs.writeFileSync(args.summaryJsonPath, JSON.stringify(summary, null, 2) + "\n", "utf8");
        console.log(`Wrote summary JSON: ${args.summaryJsonPath}`);
      }
    }

    // Commit write once (unless dry-run)
    if (!args.dryRun) {
      writeCsv(CSR_PATH, csr.headers, csr.rows, false);
      writeCsv(DXP_PATH, dxp.headers, dxp.rows, false);
      console.log("\nWrote CSR + DXP once (safe commit).");
    } else {
      console.log("\n[DRY] No files written.");
    }

    // exit code
    if (hadStop) process.exit(1);
    process.exit(failCount > 0 ? 1 : 0);
  }).catch((e) => {
    console.error("Fatal:", e?.stack ?? e);
    process.exit(1);
  });
}

 *   --backup-dir <path>     default: server/data/csv/_tx_backups

import fs from "fs";
import path from "path";

type Args = {
  seedPath?: string;              // optional if --rollback is used
  dryRun: boolean;
  continueOnFail: boolean;
  onlyProfile?: string;
  ccs: string[];
  listOnly: boolean;
  summaryJsonPath?: string;
  parallel: number;

  // tx
  backupDir: string;
  runId?: string;
  noBackup: boolean;
  rollbackRunId?: string;
};

function parseArgs(argv: string[]): Args {
  const args: Args = {
    seedPath: argv[0] && !argv[0].startsWith("--") ? argv[0] : undefined,
    dryRun: argv.includes("--dry-run"),
    continueOnFail: argv.includes("--continue-on-fail"),
    ccs: [],
    listOnly: argv.includes("--list"),
    parallel: 4,

    backupDir: "server/data/csv/_tx_backups",
    noBackup: argv.includes("--no-backup"),
  };

  for (let i = 0; i < argv.length; i++) {
    const a = argv[i];
    if (a === "--only-profile") args.onlyProfile = argv[++i];
    else if (a === "--cc") args.ccs.push(argv[++i]);
    else if (a === "--summary-json") args.summaryJsonPath = argv[++i];
    else if (a === "--parallel") args.parallel = Math.max(1, parseInt(argv[++i] ?? "4", 10) || 4);

    else if (a === "--backup-dir") args.backupDir = argv[++i];
    else if (a === "--run-id") args.runId = argv[++i];
    else if (a === "--rollback") args.rollbackRunId = argv[++i];
  }

  return args;
}

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQuotes = false;

  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"') {
      if (inQuotes && line[i + 1] === '"') {
        cur += '"';
        i++;
      } else {
        inQuotes = !inQuotes;
      }
      continue;
    }
    if (ch === "," && !inQuotes) {
      out.push(cur);
      cur = "";
      continue;
    }
    cur += ch;
  }
  out.push(cur);
  return out.map((s) => s.trim());
}

function readCsv(filePath: string) {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return { headers: [] as string[], rows: [] as Record<string, string>[] };

  const headers = splitCsvLine(lines[0]);
  const rows: Record<string, string>[] = [];
  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    if (cols.every((c) => c === "")) continue;
    const r: Record<string, string> = {};
    for (let j = 0; j < headers.length; j++) r[headers[j]] = cols[j] ?? "";
    rows.push(r);
  }
  return { headers, rows };
}

function writeCsv(filePath: string, headers: string[], rows: Record<string, string>[], dryRun: boolean) {
  const lines: string[] = [];
  lines.push(headers.join(","));
  for (const r of rows) {
    const cols = headers.map((h) => {
      const v = r[h] ?? "";
      if (v.includes(",") || v.includes('"') || v.includes("\n")) return `"${v.replace(/"/g, '""')}"`;
      return v;
    });
    lines.push(cols.join(","));
  }
  if (dryRun) {
    console.log(`[DRY] Would write ${rows.length} rows -> ${filePath}`);
    return;
  }
  fs.writeFileSync(filePath, lines.join("\n") + "\n", "utf8");
}

function mustHave(headers: string[], needed: string[], fileLabel: string) {
  for (const h of needed) {
    if (!headers.includes(h)) throw new Error(`${fileLabel} missing header: ${h}`);
  }
}

type Profile = {
  cluster_prefix: string;
  tiers: Record<"PRIMARY" | "SECONDARY" | "BENIGN", { points: number; priority: number }>;
  activate: Array<{ evidence: string; tier: "PRIMARY" | "SECONDARY" | "BENIGN"; when: string; points?: number }>;
};

function readSeed(filePath: string): Array<{ CC_ID: string; PROFILE_ID: string }> {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return [];

  const headers = splitCsvLine(lines[0]);
  const idxCc = headers.indexOf("CC_ID");
  const idxPf = headers.indexOf("PROFILE_ID");
  if (idxCc < 0 || idxPf < 0) {
    throw new Error(`Seed CSV must have headers CC_ID,PROFILE_ID (got: ${headers.join(",")})`);
  }

  const rows: Array<{ CC_ID: string; PROFILE_ID: string }> = [];
  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    const cc = (cols[idxCc] ?? "").trim();
    const pf = (cols[idxPf] ?? "").trim();
    if (!cc || !pf) continue;
    rows.push({ CC_ID: cc, PROFILE_ID: pf });
  }
  return rows;
}

function applyFilters(rows: Array<{ CC_ID: string; PROFILE_ID: string }>, args: Args) {
  let out = rows;
  if (args.onlyProfile) out = out.filter((r) => r.PROFILE_ID === args.onlyProfile);
  if (args.ccs.length) {
    const set = new Set(args.ccs.map((x) => x.trim()).filter(Boolean));
    out = out.filter((r) => set.has(r.CC_ID));
  }
  return out;
}

type ItemPlan = {
  ccId: string;
  profileId: string;
  ok: boolean;
  errors: string[];
  csrAdded: number;
  csrUpdated: number;
  dxpAdded: number;
  dxpUpdated: number;
  notes: string[];
};

function makeLimiter(n: number) {
  let active = 0;
  const queue: Array<() => void> = [];
  const next = () => {
    if (active >= n) return;
    const job = queue.shift();
    if (!job) return;
    active++;
    job();
  };
  return function limit<T>(fn: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      queue.push(() => {
        fn()
          .then(resolve)
          .catch(reject)
          .finally(() => {
            active--;
            next();
          });
      });
      next();
    });
  };
}

function makeRunId() {
  const d = new Date();
  const pad = (n: number) => String(n).padStart(2, "0");
  // YYYYMMDD_HHMMSS
  return `${d.getFullYear()}${pad(d.getMonth() + 1)}${pad(d.getDate())}_${pad(d.getHours())}${pad(d.getMinutes())}${pad(d.getSeconds())}`;
}

function copyFile(src: string, dst: string, dryRun: boolean) {
  if (dryRun) {
    console.log(`[DRY] Would copy ${src} -> ${dst}`);
    return;
  }
  fs.mkdirSync(path.dirname(dst), { recursive: true });
  fs.copyFileSync(src, dst);
}

function rollbackFromBackup(params: {
  root: string;
  backupDir: string;
  runId: string;
  csrPath: string;
  dxpPath: string;
}) {
  const { root, backupDir, runId, csrPath, dxpPath } = params;
  const base = path.isAbsolute(backupDir) ? backupDir : path.join(root, backupDir);
  const runDir = path.join(base, runId);

  const csrBak = path.join(runDir, "CLUSTER_SCORING_RULES.csv.bak");
  const dxpBak = path.join(runDir, "DX_PRIORITY.csv.bak");
  const manifest = path.join(runDir, "manifest.json");

  if (!fs.existsSync(runDir)) throw new Error(`Backup run dir not found: ${runDir}`);
  if (!fs.existsSync(csrBak)) throw new Error(`Missing CSR backup: ${csrBak}`);
  if (!fs.existsSync(dxpBak)) throw new Error(`Missing DXP backup: ${dxpBak}`);

  // Restore
  fs.copyFileSync(csrBak, csrPath);
  fs.copyFileSync(dxpBak, dxpPath);

  console.log("✅ Rollback complete.");
  console.log(`Restored:\n- ${csrPath}\n- ${dxpPath}`);
  if (fs.existsSync(manifest)) {
    console.log(`Manifest: ${manifest}`);
  }
}

function main() {
  const args = parseArgs(process.argv.slice(2));
  const root = process.cwd();

  const PROFILE_PATH = path.join(root, "data", "complaints", "profile_packs.json");
  const CSR_PATH = path.join(root, "server", "data", "csv", "CLUSTER_SCORING_RULES.csv");
  const DXP_PATH = path.join(root, "server", "data", "csv", "DX_PRIORITY.csv");

  if (!fs.existsSync(PROFILE_PATH)) throw new Error(`Missing profile file: ${PROFILE_PATH}`);
  if (!fs.existsSync(CSR_PATH)) throw new Error(`Missing: ${CSR_PATH}`);
  if (!fs.existsSync(DXP_PATH)) throw new Error(`Missing: ${DXP_PATH}`);

  // --- ROLLBACK MODE ---
  if (args.rollbackRunId) {
    rollbackFromBackup({
      root,
      backupDir: args.backupDir,
      runId: args.rollbackRunId,
      csrPath: CSR_PATH,
      dxpPath: DXP_PATH,
    });
    process.exit(0);
  }

  if (!args.seedPath) {
    throw new Error("Missing seed.csv path. Usage: npx tsx scripts/apply-profile-pack-bulk.ts <seed.csv> [flags]");
  }

  const seedAbs = path.isAbsolute(args.seedPath) ? args.seedPath : path.join(root, args.seedPath);
  if (!fs.existsSync(seedAbs)) throw new Error(`Seed not found: ${seedAbs}`);

  const seedAll = readSeed(seedAbs);
  const seed = applyFilters(seedAll, args);

  if (args.listOnly) {
    console.log("\n=== Bulk Profile Apply Plan ===");
    console.log(`Seed rows: ${seedAll.length}`);
    console.log(`After filters: ${seed.length}`);
    console.log("");
    for (const r of seed) console.log(`- ${r.CC_ID} -> ${r.PROFILE_ID}`);
    process.exit(0);
  }

  const profiles = JSON.parse(fs.readFileSync(PROFILE_PATH, "utf8")) as Record<string, Profile>;

  const csr = readCsv(CSR_PATH);
  const dxp = readCsv(DXP_PATH);

  mustHave(csr.headers, ["CC_ID", "CLUSTER_ID", "RULE_ID", "POINTS", "WHEN_EXPR", "EVIDENCE_LABEL"], "CLUSTER_SCORING_RULES.csv");
  mustHave(dxp.headers, ["CC_ID", "CLUSTER_ID", "PRIORITY"], "DX_PRIORITY.csv");

  // Index CSR for quick lookup
  const csrByCcEvidence = new Map<string, Record<string, string>>(); // key = `${ccId}||${evidence}`
  const csrRuleIdsByCc = new Map<string, Set<string>>();

  for (const r of csr.rows) {
    const ccId = r.CC_ID;
    if (!ccId) continue;
    if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
    if (r.RULE_ID) csrRuleIdsByCc.get(ccId)!.add(r.RULE_ID);

    if (r.EVIDENCE_LABEL) {
      csrByCcEvidence.set(`${ccId}||${r.EVIDENCE_LABEL}`, r);
    }
  }

  // Index DXP
  const dxpByCcCluster = new Map<string, Record<string, string>>(); // key = `${ccId}||${clusterId}`
  for (const r of dxp.rows) {
    if (r.CC_ID && r.CLUSTER_ID) dxpByCcCluster.set(`${r.CC_ID}||${r.CLUSTER_ID}`, r);
  }

  const limiter = makeLimiter(args.parallel);

  console.log(`\n=== ULTRA Bulk Profile Apply (${seed.length} rows) ===`);
  console.log(`parallel planning: ${args.parallel}`);
  if (args.dryRun) console.log("Mode: DRY RUN");
  if (args.continueOnFail) console.log("Mode: CONTINUE ON FAIL");
  if (args.onlyProfile) console.log(`Filter: only-profile=${args.onlyProfile}`);
  if (args.ccs.length) console.log(`Filter: cc in [${args.ccs.join(", ")}]`);
  if (args.summaryJsonPath) console.log(`Will write: ${args.summaryJsonPath}`);
  if (!args.noBackup) console.log(`Backup dir: ${args.backupDir}`);
  console.log("");

  const plans: ItemPlan[] = [];

  const runOne = async (ccId: string, profileId: string): Promise<ItemPlan> => {
    const p: ItemPlan = {
      ccId,
      profileId,
      ok: true,
      errors: [],
      csrAdded: 0,
      csrUpdated: 0,
      dxpAdded: 0,
      dxpUpdated: 0,
      notes: [],
    };

    const profile = profiles[profileId];
    if (!profile) {
      p.ok = false;
      p.errors.push(`Profile not found: ${profileId}`);
      return p;
    }

    const prefix = profile.cluster_prefix;
    const tierClusterId = (tier: "PRIMARY" | "SECONDARY" | "BENIGN") => `CL_${prefix}_${tier}`;

    // ENSURE CSR rows exist
    const ruleIdRe = new RegExp(`^CSR_${prefix}_(\\d+)$`);
    let maxN = 0;
    const existingRuleIds = csrRuleIdsByCc.get(ccId) ?? new Set<string>();
    for (const rid of existingRuleIds) {
      const m = rid.match(ruleIdRe);
      if (m) {
        const n = parseInt(m[1], 10);
        if (!Number.isNaN(n)) maxN = Math.max(maxN, n);
      }
    }

    for (const item of profile.activate) {
      const key = `${ccId}||${item.evidence}`;
      const existing = csrByCcEvidence.get(key);
      if (!existing) {
        maxN++;
        const ruleId = `CSR_${prefix}_${String(maxN).padStart(2, "0")}`;
        const newRow: Record<string, string> = {
          CC_ID: ccId,
          CLUSTER_ID: tierClusterId("BENIGN"), // placeholder until apply
          RULE_ID: ruleId,
          POINTS: "0",
          WHEN_EXPR: "false",
          EVIDENCE_LABEL: item.evidence,
        };
        csr.rows.push(newRow);
        csrByCcEvidence.set(key, newRow);
        if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
        csrRuleIdsByCc.get(ccId)!.add(ruleId);
        p.csrAdded++;
        p.notes.push(`CSR add evidence=${item.evidence} RULE_ID=${ruleId}`);
      }
    }

    // APPLY targeted activations
    for (const item of profile.activate) {
      const row = csrByCcEvidence.get(`${ccId}||${item.evidence}`);
      if (!row) {
        p.ok = false;
        p.errors.push(`CSR missing after ensure for evidence=${item.evidence}`);
        if (!args.continueOnFail) return p;
        continue;
      }

      const wantCluster = tierClusterId(item.tier);
      const wantWhen = item.when;
      const wantPoints = String(item.points ?? profile.tiers[item.tier].points);

      let changed = false;
      if (row.CLUSTER_ID !== wantCluster) { row.CLUSTER_ID = wantCluster; changed = true; }
      if (row.WHEN_EXPR !== wantWhen) { row.WHEN_EXPR = wantWhen; changed = true; }
      if (row.POINTS !== wantPoints) { row.POINTS = wantPoints; changed = true; }

      if (changed) p.csrUpdated++;
    }

    // ENSURE DXP tiers exist
    for (const tier of ["PRIMARY", "SECONDARY", "BENIGN"] as const) {
      const clusterId = tierClusterId(tier);
      const wantPriority = String(profile.tiers[tier].priority);
      const key = `${ccId}||${clusterId}`;

      const existing = dxpByCcCluster.get(key);
      if (!existing) {
        const newRow: Record<string, string> = {
          CC_ID: ccId,
          CLUSTER_ID: clusterId,
          PRIORITY: wantPriority,
        };
        dxp.rows.push(newRow);
        dxpByCcCluster.set(key, newRow);
        p.dxpAdded++;
      } else if (existing.PRIORITY !== wantPriority) {
        existing.PRIORITY = wantPriority;
        p.dxpUpdated++;
      }
    }

    return p;
  };

  const tasks = seed.map((r) =>
    limiter(async () => runOne(r.CC_ID, r.PROFILE_ID))
      .then((p) => {
        plans.push(p);
        const status = p.ok ? "✅" : "❌";
        console.log(`${status} ${p.ccId} -> ${p.profileId} | CSR +${p.csrAdded} upd ${p.csrUpdated} | DXP +${p.dxpAdded} upd ${p.dxpUpdated}`);
        if (!p.ok && p.errors.length) for (const e of p.errors) console.log(`   - ${e}`);
        if (!p.ok && !args.continueOnFail) throw new Error(`Stopping on failure (use --continue-on-fail): ${p.ccId}`);
      })
  );

  Promise.allSettled(tasks).then((settled) => {
    const hadStop = settled.some((s) => s.status === "rejected");
    const okCount = plans.filter((p) => p.ok).length;
    const failCount = plans.length - okCount;

    console.log("\n=== ULTRA Summary ===");
    console.log(`attempted: ${plans.length}`);
    console.log(`ok: ${okCount}`);
    console.log(`fail: ${failCount}`);
    console.log(`CSR added: ${plans.reduce((a, p) => a + p.csrAdded, 0)}`);
    console.log(`CSR updated: ${plans.reduce((a, p) => a + p.csrUpdated, 0)}`);
    console.log(`DXP added: ${plans.reduce((a, p) => a + p.dxpAdded, 0)}`);
    console.log(`DXP updated: ${plans.reduce((a, p) => a + p.dxpUpdated, 0)}`);

    const summary = {
      seedPath: args.seedPath,
      dryRun: args.dryRun,
      continueOnFail: args.continueOnFail,
      onlyProfile: args.onlyProfile ?? null,
      ccs: args.ccs,
      parallel: args.parallel,
      attempted: plans.length,
      ok: okCount,
      fail: failCount,
      totals: {
        csrAdded: plans.reduce((a, p) => a + p.csrAdded, 0),
        csrUpdated: plans.reduce((a, p) => a + p.csrUpdated, 0),
        dxpAdded: plans.reduce((a, p) => a + p.dxpAdded, 0),
        dxpUpdated: plans.reduce((a, p) => a + p.dxpUpdated, 0),
      },
      results: plans,
      generated_at: new Date().toISOString(),
    };

    // Summary JSON
    if (args.summaryJsonPath) {
      if (args.dryRun) {
        console.log(`[DRY] Would write summary JSON -> ${args.summaryJsonPath}`);
      } else {
        fs.mkdirSync(path.dirname(args.summaryJsonPath), { recursive: true });
        fs.writeFileSync(args.summaryJsonPath, JSON.stringify(summary, null, 2) + "\n", "utf8");
        console.log(`Wrote summary JSON: ${args.summaryJsonPath}`);
      }
    }

    // Commit write once (unless dry-run)
    if (args.dryRun) {
      console.log("\n[DRY] No files written.");
      process.exit(hadStop || failCount > 0 ? 1 : 0);
      return;
    }

    // Transaction backup
    const runId = args.runId ?? makeRunId();
    const backupBase = path.isAbsolute(args.backupDir) ? args.backupDir : path.join(root, args.backupDir);
    const runDir = path.join(backupBase, runId);

    if (!args.noBackup) {
      fs.mkdirSync(runDir, { recursive: true });

      const csrBak = path.join(runDir, "CLUSTER_SCORING_RULES.csv.bak");
      const dxpBak = path.join(runDir, "DX_PRIORITY.csv.bak");

      copyFile(CSR_PATH, csrBak, false);
      copyFile(DXP_PATH, dxpBak, false);

      const manifest = {
        run_id: runId,
        created_at: new Date().toISOString(),
        seed_path: args.seedPath,
        backup_dir: runDir,
        files: [
          { src: CSR_PATH, backup: csrBak },
          { src: DXP_PATH, backup: dxpBak }
        ],
        summary
      };

      fs.writeFileSync(path.join(runDir, "manifest.json"), JSON.stringify(manifest, null, 2) + "\n", "utf8");
      console.log(`\n🧾 Transaction backup created: ${runDir}`);
      console.log(`Rollback with: npx tsx scripts/apply-profile-pack-bulk.ts --rollback ${runId}`);
    }

    // Now commit the write
    writeCsv(CSR_PATH, csr.headers, csr.rows, false);
    writeCsv(DXP_PATH, dxp.headers, dxp.rows, false);
    console.log("\nWrote CSR + DXP once (safe commit).");

    process.exit(hadStop || failCount > 0 ? 1 : 0);
  }).catch((e) => {
    console.error("Fatal:", e?.stack ?? e);
    process.exit(1);
  });
}

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQuotes = false;

  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"') {
      if (inQuotes && line[i + 1] === '"') {
        cur += '"';
        i++;
      } else {
        inQuotes = !inQuotes;
      }
      continue;
    }
    if (ch === "," && !inQuotes) {
      out.push(cur);
      cur = "";
      continue;
    }
    cur += ch;
  }
  out.push(cur);
  return out.map((s) => s.trim());
}

function readCsv(filePath: string) {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return { headers: [] as string[], rows: [] as Record<string, string>[] };

  const headers = splitCsvLine(lines[0]);
  const rows: Record<string, string>[] = [];
  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    if (cols.every((c) => c === "")) continue;
    const r: Record<string, string> = {};
    for (let j = 0; j < headers.length; j++) r[headers[j]] = cols[j] ?? "";
    rows.push(r);
  }
  return { headers, rows };
}

function writeCsv(filePath: string, headers: string[], rows: Record<string, string>[], dryRun: boolean) {
  const lines: string[] = [];
  lines.push(headers.join(","));
  for (const r of rows) {
    const cols = headers.map((h) => {
      const v = r[h] ?? "";
      if (v.includes(",") || v.includes('"') || v.includes("\n")) return `"${v.replace(/"/g, '""')}"`;
      return v;
    });
    lines.push(cols.join(","));
  }
  if (dryRun) {
    console.log(`[DRY] Would write ${rows.length} rows -> ${filePath}`);
    return;
  }
  fs.writeFileSync(filePath, lines.join("\n") + "\n", "utf8");
}

function mustHave(headers: string[], needed: string[], fileLabel: string) {
  for (const h of needed) {
    if (!headers.includes(h)) throw new Error(`${fileLabel} missing header: ${h}`);
  }
}

function readSeed(filePath: string): Array<{ CC_ID: string; PROFILE_ID: string }> {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return [];

  const headers = splitCsvLine(lines[0]);
  const idxCc = headers.indexOf("CC_ID");
  const idxPf = headers.indexOf("PROFILE_ID");
  if (idxCc < 0 || idxPf < 0) {
    throw new Error(`Seed CSV must have headers CC_ID,PROFILE_ID (got: ${headers.join(",")})`);
  }

  const rows: Array<{ CC_ID: string; PROFILE_ID: string }> = [];
  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    const cc = (cols[idxCc] ?? "").trim();
    const pf = (cols[idxPf] ?? "").trim();
    if (!cc || !pf) continue;
    rows.push({ CC_ID: cc, PROFILE_ID: pf });
  }
  return rows;
}

function applyFilters(rows: Array<{ CC_ID: string; PROFILE_ID: string }>, args: Args) {

  if (args.ccs.length) {
    const set = new Set(args.ccs.map((x) => x.trim()).filter(Boolean));
    out = out.filter((r) => set.has(r.CC_ID));
  }
  return out;
}

function makeLimiter(n: number) {
  let active = 0;
  const queue: Array<() => void> = [];
  const next = () => {
    if (active >= n) return;
    const job = queue.shift();
    if (!job) return;
    active++;
    job();
  };
  return function limit<T>(fn: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      queue.push(() => {
        fn()
          .then(resolve)
          .catch(reject)
          .finally(() => {
            active--;
            next();
          });
      });
      next();
    });
  };
}

  if (args.rollbackRunId) {
    rollbackFromBackup({
      root,
      backupDir: args.backupDir,
      runId: args.rollbackRunId,
      csrPath: CSR_PATH,
      dxpPath: DXP_PATH,
    });
    process.exit(0);
  }

  if (!args.seedPath) {
    throw new Error("Missing seed.csv path. Usage: npx tsx scripts/apply-profile-pack-bulk.ts <seed.csv> [flags]");
  }

  const seedAbs = path.isAbsolute(args.seedPath) ? args.seedPath : path.join(root, args.seedPath);
  if (!fs.existsSync(seedAbs)) throw new Error(`Seed not found: ${seedAbs}`);

  const seedAll = readSeed(seedAbs);
  const seed = applyFilters(seedAll, args);

  if (args.listOnly) {
    console.log("\n=== Bulk Profile Apply Plan ===");
    console.log(`Seed rows: ${seedAll.length}`);
    console.log(`After filters: ${seed.length}`);
    console.log("");
    for (const r of seed) console.log(`- ${r.CC_ID} -> ${r.PROFILE_ID}`);
    process.exit(0);
  }

  const profiles = JSON.parse(fs.readFileSync(PROFILE_PATH, "utf8")) as Record<string, Profile>;

  const csr = readCsv(CSR_PATH);
  const dxp = readCsv(DXP_PATH);

  mustHave(csr.headers, ["CC_ID", "CLUSTER_ID", "RULE_ID", "POINTS", "WHEN_EXPR", "EVIDENCE_LABEL"], "CLUSTER_SCORING_RULES.csv");
  mustHave(dxp.headers, ["CC_ID", "CLUSTER_ID", "PRIORITY"], "DX_PRIORITY.csv");

  // Index CSR
  const csrByCcEvidence = new Map<string, Record<string, string>>(); // `${ccId}||${evidence}`
  const csrRuleIdsByCc = new Map<string, Set<string>>();

  for (const r of csr.rows) {
    const ccId = r.CC_ID;
    if (!ccId) continue;
    if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
    if (r.RULE_ID) csrRuleIdsByCc.get(ccId)!.add(r.RULE_ID);

    if (r.EVIDENCE_LABEL) csrByCcEvidence.set(`${ccId}||${r.EVIDENCE_LABEL}`, r);
  }

  // Index DXP
  const dxpByCcCluster = new Map<string, Record<string, string>>(); // `${ccId}||${clusterId}`
  for (const r of dxp.rows) {
    if (r.CC_ID && r.CLUSTER_ID) dxpByCcCluster.set(`${r.CC_ID}||${r.CLUSTER_ID}`, r);
  }

  const limiter = makeLimiter(args.parallel);

  console.log(`\n=== ULTRA Bulk Profile Apply (${seed.length} rows) ===`);
  console.log(`parallel planning: ${args.parallel}`);
  if (args.dryRun) console.log("Mode: DRY RUN");
  if (args.continueOnFail) console.log("Mode: CONTINUE ON FAIL");
  if (args.onlyProfile) console.log(`Filter: only-profile=${args.onlyProfile}`);
  if (args.ccs.length) console.log(`Filter: cc in [${args.ccs.join(", ")}]`);
  if (args.summaryJsonPath) console.log(`Will write: ${args.summaryJsonPath}`);
  if (!args.noBackup) console.log(`Backup dir: ${args.backupDir}`);
  if (!args.noValidate) console.log(`Validate cmd: ${args.validateCmd}`);
  console.log("");

  const plans: ItemPlan[] = [];

  const runOne = async (ccId: string, profileId: string): Promise<ItemPlan> => {
    const p: ItemPlan = {
      ccId,
      profileId,
      ok: true,
      errors: [],
      csrAdded: 0,
      csrUpdated: 0,
      dxpAdded: 0,
      dxpUpdated: 0,
      notes: [],
    };

    const profile = profiles[profileId];
    if (!profile) {
      p.ok = false;
      p.errors.push(`Profile not found: ${profileId}`);
      return p;
    }

    const prefix = profile.cluster_prefix;
    const tierClusterId = (tier: "PRIMARY" | "SECONDARY" | "BENIGN") => `CL_${prefix}_${tier}`;

    // ENSURE CSR rows exist
    const ruleIdRe = new RegExp(`^CSR_${prefix}_(\\d+)$`);
    let maxN = 0;
    const existingRuleIds = csrRuleIdsByCc.get(ccId) ?? new Set<string>();
    for (const rid of existingRuleIds) {
      const m = rid.match(ruleIdRe);
      if (m) {
        const n = parseInt(m[1], 10);
        if (!Number.isNaN(n)) maxN = Math.max(maxN, n);
      }
    }

    for (const item of profile.activate) {
      const key = `${ccId}||${item.evidence}`;
      const existing = csrByCcEvidence.get(key);
      if (!existing) {
        maxN++;
        const ruleId = `CSR_${prefix}_${String(maxN).padStart(2, "0")}`;
        const newRow: Record<string, string> = {
          CC_ID: ccId,
          CLUSTER_ID: tierClusterId("BENIGN"),
          RULE_ID: ruleId,
          POINTS: "0",
          WHEN_EXPR: "false",
          EVIDENCE_LABEL: item.evidence,
        };
        csr.rows.push(newRow);
        csrByCcEvidence.set(key, newRow);

        if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
        csrRuleIdsByCc.get(ccId)!.add(ruleId);

        p.csrAdded++;
        p.notes.push(`CSR add evidence=${item.evidence} RULE_ID=${ruleId}`);
      }
    }

    // APPLY activations
    for (const item of profile.activate) {
      const row = csrByCcEvidence.get(`${ccId}||${item.evidence}`);
      if (!row) {
        p.ok = false;
        p.errors.push(`CSR missing after ensure for evidence=${item.evidence}`);
        if (!args.continueOnFail) return p;
        continue;
      }

      const wantCluster = tierClusterId(item.tier);
      const wantWhen = item.when;
      const wantPoints = String(item.points ?? profile.tiers[item.tier].points);

      let changed = false;
      if (row.CLUSTER_ID !== wantCluster) { row.CLUSTER_ID = wantCluster; changed = true; }
      if (row.WHEN_EXPR !== wantWhen) { row.WHEN_EXPR = wantWhen; changed = true; }
      if (row.POINTS !== wantPoints) { row.POINTS = wantPoints; changed = true; }
      if (changed) p.csrUpdated++;
    }

    // ENSURE DXP tiers exist
    for (const tier of ["PRIMARY", "SECONDARY", "BENIGN"] as const) {
      const clusterId = tierClusterId(tier);
      const wantPriority = String(profile.tiers[tier].priority);
      const key = `${ccId}||${clusterId}`;

      const existing = dxpByCcCluster.get(key);
      if (!existing) {
        const newRow: Record<string, string> = { CC_ID: ccId, CLUSTER_ID: clusterId, PRIORITY: wantPriority };
        dxp.rows.push(newRow);
        dxpByCcCluster.set(key, newRow);
        p.dxpAdded++;
      } else if (existing.PRIORITY !== wantPriority) {
        existing.PRIORITY = wantPriority;
        p.dxpUpdated++;
      }
    }

    return p;
  };

  const tasks = seed.map((r) =>
    limiter(async () => runOne(r.CC_ID, r.PROFILE_ID))
      .then((p) => {
        plans.push(p);
        const status = p.ok ? "✅" : "❌";
        console.log(
          `${status} ${p.ccId} -> ${p.profileId} | CSR +${p.csrAdded} upd ${p.csrUpdated} | DXP +${p.dxpAdded} upd ${p.dxpUpdated}`
        );
        if (!p.ok && p.errors.length) for (const e of p.errors) console.log(`   - ${e}`);
        if (!p.ok && !args.continueOnFail) throw new Error(`Stopping on failure (use --continue-on-fail): ${p.ccId}`);
      })
  );

  Promise.allSettled(tasks)
    .then((settled) => {
      const hadStop = settled.some((s) => s.status === "rejected");
      const okCount = plans.filter((p) => p.ok).length;
      const failCount = plans.length - okCount;

      console.log("\n=== ULTRA Summary ===");
      console.log(`attempted: ${plans.length}`);
      console.log(`ok: ${okCount}`);
      console.log(`fail: ${failCount}`);
      console.log(`CSR added: ${plans.reduce((a, p) => a + p.csrAdded, 0)}`);
      console.log(`CSR updated: ${plans.reduce((a, p) => a + p.csrUpdated, 0)}`);
      console.log(`DXP added: ${plans.reduce((a, p) => a + p.dxpAdded, 0)}`);
      console.log(`DXP updated: ${plans.reduce((a, p) => a + p.dxpUpdated, 0)}`);

      const summary = {
        seedPath: args.seedPath,
        dryRun: args.dryRun,
        continueOnFail: args.continueOnFail,
        onlyProfile: args.onlyProfile ?? null,
        ccs: args.ccs,
        parallel: args.parallel,
        attempted: plans.length,
        ok: okCount,
        fail: failCount,
        totals: {
          csrAdded: plans.reduce((a, p) => a + p.csrAdded, 0),
          csrUpdated: plans.reduce((a, p) => a + p.csrUpdated, 0),
          dxpAdded: plans.reduce((a, p) => a + p.dxpAdded, 0),
          dxpUpdated: plans.reduce((a, p) => a + p.dxpUpdated, 0),
        },
        results: plans,
        generated_at: new Date().toISOString(),
      };

      // Summary JSON
      if (args.summaryJsonPath) {
        if (args.dryRun) {
          console.log(`[DRY] Would write summary JSON -> ${args.summaryJsonPath}`);
        } else {
          fs.mkdirSync(path.dirname(args.summaryJsonPath), { recursive: true });
          fs.writeFileSync(args.summaryJsonPath, JSON.stringify(summary, null, 2) + "\n", "utf8");
          console.log(`Wrote summary JSON: ${args.summaryJsonPath}`);
        }
      }

      // DRY RUN: no writes
      if (args.dryRun) {
        console.log("\n[DRY] No files written.");
        process.exit(hadStop || failCount > 0 ? 1 : 0);
        return;
      }

      // Transaction backup
      const runId = args.runId ?? makeRunId();
      const backupBase = path.isAbsolute(args.backupDir) ? args.backupDir : path.join(root, args.backupDir);
      const runDir = path.join(backupBase, runId);

      let csrBak = "";
      let dxpBak = "";

      if (!args.noBackup) {
        fs.mkdirSync(runDir, { recursive: true });

        csrBak = path.join(runDir, "CLUSTER_SCORING_RULES.csv.bak");
        dxpBak = path.join(runDir, "DX_PRIORITY.csv.bak");

        copyFile(CSR_PATH, csrBak, false);
        copyFile(DXP_PATH, dxpBak, false);

        const manifest = {
          run_id: runId,
          created_at: new Date().toISOString(),
          seed_path: args.seedPath,
          backup_dir: runDir,
          files: [
            { src: CSR_PATH, backup: csrBak },
            { src: DXP_PATH, backup: dxpBak },
          ],
          summary,
        };

        fs.writeFileSync(path.join(runDir, "manifest.json"), JSON.stringify(manifest, null, 2) + "\n", "utf8");
        console.log(`\n🧾 Transaction backup created: ${runDir}`);
        console.log(`Rollback with: npx tsx scripts/apply-profile-pack-bulk.ts --rollback ${runId}`);
      }

      // Commit write once
      writeCsv(CSR_PATH, csr.headers, csr.rows, false);
      writeCsv(DXP_PATH, dxp.headers, dxp.rows, false);
      console.log("\nWrote CSR + DXP once (safe commit).");

      // Validation (post-write) with auto-rollback
      if (!args.noValidate) {
        console.log(`\n🔎 Validation: ${args.validateCmd}`);
        const v = runShell(args.validateCmd);

        if (!v.ok) {
          console.log(`\n❌ Validation failed (exit ${v.code}).`);

          if (args.noBackup) {
            console.log("⚠️ No backup was created (--no-backup), cannot auto-rollback safely.");
          } else if (args.noAutoRollback) {
            console.log("⚠️ Auto-rollback disabled (--no-auto-rollback).");
            console.log(`Backup run id: ${runId}`);
          } else {
            console.log("↩️  Auto-rollback to transaction backup...");
            try {
              if (!csrBak || !dxpBak) {
                // defensive: reconstruct from runDir
                csrBak = path.join(runDir, "CLUSTER_SCORING_RULES.csv.bak");
                dxpBak = path.join(runDir, "DX_PRIORITY.csv.bak");
              }
              rollbackNow(csrBak, dxpBak, CSR_PATH, DXP_PATH);
              console.log("✅ Rollback complete.");
              console.log(`Backup run id: ${runId}`);
            } catch (e: any) {
              console.log("⚠️ Rollback failed:", e?.message ?? String(e));
              console.log(`Backup run id: ${runId}`);
            }
          }

          process.exit(1);
          return;
        }

        console.log("✅ Validation passed.");
      }

      process.exit(hadStop || failCount > 0 ? 1 : 0);
    })
    .catch((e) => {
      console.error("Fatal:", e?.stack ?? e);
      process.exit(1);
    });
}

/**
 * scripts/apply-profile-pack-bulk.ts
 * ULTRA DELUXE: bulk ensure+apply profile packs with:
 * - Safe parallel planning (no concurrent file writers)
 * - Single-commit write of CSR + DXP
 * - Transaction backups (CSR + DXP) + manifest.json
 * - Rollback mode: --rollback <RUN_ID>
 * - Post-write validation with auto-rollback on failure
 * - Belt & suspenders file commit:
 *    - write to temp files
 *    - fsync
 *    - rename over originals
 *    - optionally keep .prev copies
 *
 * CSVs touched:
 *   server/data/csv/CLUSTER_SCORING_RULES.csv
 *   server/data/csv/DX_PRIORITY.csv
 *
 * Profile packs:
 *   data/complaints/profile_packs.json
 *
 * Seed CSV:
 *   data/complaints/profile_apply_seed.csv (CC_ID,PROFILE_ID)
 *
 * Usage:
 *   npx tsx scripts/apply-profile-pack-bulk.ts data/complaints/profile_apply_seed.csv
 *
 * Flags (common):
 *   --dry-run
 *   --continue-on-fail
 *   --only-profile <PROFILE_ID>
 *   --cc <cc_id>                 (repeatable)
 *   --list
 *   --summary-json <path>
 *   --parallel <N>
 *
 * Transaction:
 *   --backup-dir <path>          default: server/data/csv/_tx_backups
 *   --run-id <id>                default: autogenerated YYYYMMDD_HHMMSS
 *   --no-backup
 *   --rollback <RUN_ID>
 *
 * Validation:
 *   --no-validate
 *   --validate-cmd "<cmd>"       default: npm run abi:complaints
 *   --no-auto-rollback
 *
 * Belt & suspenders:
 *   --keep-prev                  keep <file>.prev.<runId> copies of originals
 *   --no-atomic                  fall back to direct write (not recommended)
 */

import fs from "fs";
import path from "path";
import { spawnSync } from "child_process";

type Args = {
  seedPath?: string;
  dryRun: boolean;
  continueOnFail: boolean;
  onlyProfile?: string;
  ccs: string[];
  listOnly: boolean;
  summaryJsonPath?: string;
  parallel: number;

  backupDir: string;
  runId?: string;
  noBackup: boolean;
  rollbackRunId?: string;

  noValidate: boolean;
  validateCmd: string;
  noAutoRollback: boolean;

  keepPrev: boolean;
  noAtomic: boolean;
};

type Profile = {
  cluster_prefix: string;
  tiers: Record<"PRIMARY" | "SECONDARY" | "BENIGN", { points: number; priority: number }>;
  activate: Array<{ evidence: string; tier: "PRIMARY" | "SECONDARY" | "BENIGN"; when: string; points?: number }>;
};

type ItemPlan = {
  ccId: string;
  profileId: string;
  ok: boolean;
  errors: string[];
  csrAdded: number;
  csrUpdated: number;
  dxpAdded: number;
  dxpUpdated: number;
  notes: string[];
};

function parseArgs(argv: string[]): Args {
  const args: Args = {
    seedPath: argv[0] && !argv[0].startsWith("--") ? argv[0] : undefined,
    dryRun: argv.includes("--dry-run"),
    continueOnFail: argv.includes("--continue-on-fail"),
    ccs: [],
    listOnly: argv.includes("--list"),
    parallel: 4,

    backupDir: "server/data/csv/_tx_backups",
    noBackup: argv.includes("--no-backup"),

    noValidate: argv.includes("--no-validate"),
    validateCmd: "npm run abi:complaints",
    noAutoRollback: argv.includes("--no-auto-rollback"),

    keepPrev: argv.includes("--keep-prev"),
    noAtomic: argv.includes("--no-atomic"),
  };

  for (let i = 0; i < argv.length; i++) {
    const a = argv[i];
    if (a === "--only-profile") args.onlyProfile = argv[++i];
    else if (a === "--cc") args.ccs.push(argv[++i]);
    else if (a === "--summary-json") args.summaryJsonPath = argv[++i];
    else if (a === "--parallel") args.parallel = Math.max(1, parseInt(argv[++i] ?? "4", 10) || 4);

    else if (a === "--backup-dir") args.backupDir = argv[++i];
    else if (a === "--run-id") args.runId = argv[++i];
    else if (a === "--rollback") args.rollbackRunId = argv[++i];

    else if (a === "--validate-cmd") args.validateCmd = argv[++i];
  }

  return args;
}

function splitCsvLine(line: string): string[] {
  const out: string[] = [];
  let cur = "";
  let inQuotes = false;

  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"') {
      if (inQuotes && line[i + 1] === '"') {
        cur += '"';
        i++;
      } else {
        inQuotes = !inQuotes;
      }
      continue;
    }
    if (ch === "," && !inQuotes) {
      out.push(cur);
      cur = "";
      continue;
    }
    cur += ch;
  }
  out.push(cur);
  return out.map((s) => s.trim());
}

function readCsv(filePath: string) {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return { headers: [] as string[], rows: [] as Record<string, string>[] };

  const headers = splitCsvLine(lines[0]);
  const rows: Record<string, string>[] = [];
  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    if (cols.every((c) => c === "")) continue;
    const r: Record<string, string> = {};
    for (let j = 0; j < headers.length; j++) r[headers[j]] = cols[j] ?? "";
    rows.push(r);
  }
  return { headers, rows };
}

function encodeCsv(headers: string[], rows: Record<string, string>[]) {
  const lines: string[] = [];
  lines.push(headers.join(","));
  for (const r of rows) {
    const cols = headers.map((h) => {
      const v = r[h] ?? "";
      if (v.includes(",") || v.includes('"') || v.includes("\n")) return `"${v.replace(/"/g, '""')}"`;
      return v;
    });
    lines.push(cols.join(","));
  }
  return lines.join("\n") + "\n";
}

function mustHave(headers: string[], needed: string[], fileLabel: string) {
  for (const h of needed) {
    if (!headers.includes(h)) throw new Error(`${fileLabel} missing header: ${h}`);
  }
}

function readSeed(filePath: string): Array<{ CC_ID: string; PROFILE_ID: string }> {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (!lines.length) return [];

  const headers = splitCsvLine(lines[0]);
  const idxCc = headers.indexOf("CC_ID");
  const idxPf = headers.indexOf("PROFILE_ID");
  if (idxCc < 0 || idxPf < 0) {
    throw new Error(`Seed CSV must have headers CC_ID,PROFILE_ID (got: ${headers.join(",")})`);
  }

  const rows: Array<{ CC_ID: string; PROFILE_ID: string }> = [];
  for (let i = 1; i < lines.length; i++) {
    const cols = splitCsvLine(lines[i]);
    const cc = (cols[idxCc] ?? "").trim();
    const pf = (cols[idxPf] ?? "").trim();
    if (!cc || !pf) continue;
    rows.push({ CC_ID: cc, PROFILE_ID: pf });
  }
  return rows;
}

function applyFilters(rows: Array<{ CC_ID: string; PROFILE_ID: string }>, args: Args) {
  let out = rows;
  if (args.onlyProfile) out = out.filter((r) => r.PROFILE_ID === args.onlyProfile);
  if (args.ccs.length) {
    const set = new Set(args.ccs.map((x) => x.trim()).filter(Boolean));
    out = out.filter((r) => set.has(r.CC_ID));
  }
  return out;
}

function makeLimiter(n: number) {
  let active = 0;
  const queue: Array<() => void> = [];
  const next = () => {
    if (active >= n) return;
    const job = queue.shift();
    if (!job) return;
    active++;
    job();
  };
  return function limit<T>(fn: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      queue.push(() => {
        fn()
          .then(resolve)
          .catch(reject)
          .finally(() => {
            active--;
            next();
          });
      });
      next();
    });
  };
}

function makeRunId() {
  const d = new Date();
  const pad = (n: number) => String(n).padStart(2, "0");
  return `${d.getFullYear()}${pad(d.getMonth() + 1)}${pad(d.getDate())}_${pad(d.getHours())}${pad(d.getMinutes())}${pad(d.getSeconds())}`;
}

function copyFile(src: string, dst: string) {
  fs.mkdirSync(path.dirname(dst), { recursive: true });
  fs.copyFileSync(src, dst);
}

function runShell(cmdLine: string) {
  const res = spawnSync("sh", ["-lc", cmdLine], { stdio: "inherit", env: process.env });
  return { ok: (res.status ?? 1) === 0, code: res.status ?? 1 };
}

function rollbackFromBackup(params: {
  root: string;
  backupDir: string;
  runId: string;
  csrPath: string;
  dxpPath: string;
}) {
  const { root, backupDir, runId, csrPath, dxpPath } = params;
  const base = path.isAbsolute(backupDir) ? backupDir : path.join(root, backupDir);
  const runDir = path.join(base, runId);

  const csrBak = path.join(runDir, "CLUSTER_SCORING_RULES.csv.bak");
  const dxpBak = path.join(runDir, "DX_PRIORITY.csv.bak");
  const manifest = path.join(runDir, "manifest.json");

  if (!fs.existsSync(runDir)) throw new Error(`Backup run dir not found: ${runDir}`);
  if (!fs.existsSync(csrBak)) throw new Error(`Missing CSR backup: ${csrBak}`);
  if (!fs.existsSync(dxpBak)) throw new Error(`Missing DXP backup: ${dxpBak}`);

  fs.copyFileSync(csrBak, csrPath);
  fs.copyFileSync(dxpBak, dxpPath);

  console.log("✅ Rollback complete.");
  console.log(`Restored:\n- ${csrPath}\n- ${dxpPath}`);
  if (fs.existsSync(manifest)) console.log(`Manifest: ${manifest}`);
}

function rollbackNow(csrBak: string, dxpBak: string, csrPath: string, dxpPath: string) {
  fs.copyFileSync(csrBak, csrPath);
  fs.copyFileSync(dxpBak, dxpPath);
}

function atomicWriteFile(targetPath: string, content: string, runId: string, keepPrev: boolean) {
  const dir = path.dirname(targetPath);
  const base = path.basename(targetPath);
  const tmp = path.join(dir, `.${base}.tmp.${process.pid}.${Date.now()}`);
  const prev = path.join(dir, `${base}.prev.${runId}`);

  // Optionally keep a local "prev" copy (extra belt)
  if (keepPrev) {
    fs.copyFileSync(targetPath, prev);
  }

  // Write tmp then fsync
  const fd = fs.openSync(tmp, "w");
  try {
    fs.writeFileSync(fd, content, "utf8");
    fs.fsyncSync(fd);
  } finally {
    fs.closeSync(fd);
  }

  // Rename temp over target (atomic on POSIX in same filesystem)
  fs.renameSync(tmp, targetPath);

  // Fsync directory entry (best effort)
  try {
    const dfd = fs.openSync(dir, "r");
    try { fs.fsyncSync(dfd); } finally { fs.closeSync(dfd); }
  } catch {
    // ignore
  }
}

function main() {
  const args = parseArgs(process.argv.slice(2));
  const root = process.cwd();

  const PROFILE_PATH = path.join(root, "data", "complaints", "profile_packs.json");
  const CSR_PATH = path.join(root, "server", "data", "csv", "CLUSTER_SCORING_RULES.csv");
  const DXP_PATH = path.join(root, "server", "data", "csv", "DX_PRIORITY.csv");

  if (!fs.existsSync(PROFILE_PATH)) throw new Error(`Missing profile file: ${PROFILE_PATH}`);
  if (!fs.existsSync(CSR_PATH)) throw new Error(`Missing: ${CSR_PATH}`);
  if (!fs.existsSync(DXP_PATH)) throw new Error(`Missing: ${DXP_PATH}`);

  // Rollback mode
  if (args.rollbackRunId) {
    rollbackFromBackup({
      root,
      backupDir: args.backupDir,
      runId: args.rollbackRunId,
      csrPath: CSR_PATH,
      dxpPath: DXP_PATH,
    });
    process.exit(0);
  }

  if (!args.seedPath) {
    throw new Error("Missing seed.csv path. Usage: npx tsx scripts/apply-profile-pack-bulk.ts <seed.csv> [flags]");
  }

  const seedAbs = path.isAbsolute(args.seedPath) ? args.seedPath : path.join(root, args.seedPath);
  if (!fs.existsSync(seedAbs)) throw new Error(`Seed not found: ${seedAbs}`);

  const seedAll = readSeed(seedAbs);
  const seed = applyFilters(seedAll, args);

  if (args.listOnly) {
    console.log("\n=== Bulk Profile Apply Plan ===");
    console.log(`Seed rows: ${seedAll.length}`);
    console.log(`After filters: ${seed.length}`);
    console.log("");
    for (const r of seed) console.log(`- ${r.CC_ID} -> ${r.PROFILE_ID}`);
    process.exit(0);
  }

  const profiles = JSON.parse(fs.readFileSync(PROFILE_PATH, "utf8")) as Record<string, Profile>;

  const csr = readCsv(CSR_PATH);
  const dxp = readCsv(DXP_PATH);

  mustHave(csr.headers, ["CC_ID", "CLUSTER_ID", "RULE_ID", "POINTS", "WHEN_EXPR", "EVIDENCE_LABEL"], "CLUSTER_SCORING_RULES.csv");
  mustHave(dxp.headers, ["CC_ID", "CLUSTER_ID", "PRIORITY"], "DX_PRIORITY.csv");

  // Index CSR
  const csrByCcEvidence = new Map<string, Record<string, string>>();
  const csrRuleIdsByCc = new Map<string, Set<string>>();
  for (const r of csr.rows) {
    const ccId = r.CC_ID;
    if (!ccId) continue;
    if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
    if (r.RULE_ID) csrRuleIdsByCc.get(ccId)!.add(r.RULE_ID);
    if (r.EVIDENCE_LABEL) csrByCcEvidence.set(`${ccId}||${r.EVIDENCE_LABEL}`, r);
  }

  // Index DXP
  const dxpByCcCluster = new Map<string, Record<string, string>>();
  for (const r of dxp.rows) {
    if (r.CC_ID && r.CLUSTER_ID) dxpByCcCluster.set(`${r.CC_ID}||${r.CLUSTER_ID}`, r);
  }

  const limiter = makeLimiter(args.parallel);

  console.log(`\n=== ULTRA Bulk Profile Apply (${seed.length} rows) ===`);
  console.log(`parallel planning: ${args.parallel}`);
  if (args.dryRun) console.log("Mode: DRY RUN");
  if (args.continueOnFail) console.log("Mode: CONTINUE ON FAIL");
  if (args.onlyProfile) console.log(`Filter: only-profile=${args.onlyProfile}`);
  if (args.ccs.length) console.log(`Filter: cc in [${args.ccs.join(", ")}]`);
  if (args.summaryJsonPath) console.log(`Will write: ${args.summaryJsonPath}`);
  if (!args.noBackup) console.log(`Backup dir: ${args.backupDir}`);
  if (!args.noValidate) console.log(`Validate cmd: ${args.validateCmd}`);
  if (!args.noAtomic) console.log("Commit mode: atomic temp-write + rename");
  else console.log("Commit mode: direct write (--no-atomic)");
  if (args.keepPrev) console.log("Keep prev: enabled");
  console.log("");

  const plans: ItemPlan[] = [];

  const runOne = async (ccId: string, profileId: string): Promise<ItemPlan> => {
    const p: ItemPlan = {
      ccId,
      profileId,
      ok: true,
      errors: [],
      csrAdded: 0,
      csrUpdated: 0,
      dxpAdded: 0,
      dxpUpdated: 0,
      notes: [],
    };

    const profile = profiles[profileId];
    if (!profile) {
      p.ok = false;
      p.errors.push(`Profile not found: ${profileId}`);
      return p;
    }

    const prefix = profile.cluster_prefix;
    const tierClusterId = (tier: "PRIMARY" | "SECONDARY" | "BENIGN") => `CL_${prefix}_${tier}`;

    // ENSURE CSR rows exist
    const ruleIdRe = new RegExp(`^CSR_${prefix}_(\\d+)$`);
    let maxN = 0;
    const existingRuleIds = csrRuleIdsByCc.get(ccId) ?? new Set<string>();
    for (const rid of existingRuleIds) {
      const m = rid.match(ruleIdRe);
      if (m) {
        const n = parseInt(m[1], 10);
        if (!Number.isNaN(n)) maxN = Math.max(maxN, n);
      }
    }

    for (const item of profile.activate) {
      const key = `${ccId}||${item.evidence}`;
      const existing = csrByCcEvidence.get(key);
      if (!existing) {
        maxN++;
        const ruleId = `CSR_${prefix}_${String(maxN).padStart(2, "0")}`;
        const newRow: Record<string, string> = {
          CC_ID: ccId,
          CLUSTER_ID: tierClusterId("BENIGN"),
          RULE_ID: ruleId,
          POINTS: "0",
          WHEN_EXPR: "false",
          EVIDENCE_LABEL: item.evidence,
        };
        csr.rows.push(newRow);
        csrByCcEvidence.set(key, newRow);
        if (!csrRuleIdsByCc.has(ccId)) csrRuleIdsByCc.set(ccId, new Set());
        csrRuleIdsByCc.get(ccId)!.add(ruleId);
        p.csrAdded++;
      }
    }

    // APPLY activations
    for (const item of profile.activate) {
      const row = csrByCcEvidence.get(`${ccId}||${item.evidence}`);
      if (!row) {
        p.ok = false;
        p.errors.push(`CSR missing after ensure for evidence=${item.evidence}`);
        if (!args.continueOnFail) return p;
        continue;
      }

      const wantCluster = tierClusterId(item.tier);
      const wantWhen = item.when;
      const wantPoints = String(item.points ?? profile.tiers[item.tier].points);

      let changed = false;
      if (row.CLUSTER_ID !== wantCluster) { row.CLUSTER_ID = wantCluster; changed = true; }
      if (row.WHEN_EXPR !== wantWhen) { row.WHEN_EXPR = wantWhen; changed = true; }
      if (row.POINTS !== wantPoints) { row.POINTS = wantPoints; changed = true; }
      if (changed) p.csrUpdated++;
    }

    // ENSURE DXP tiers exist
    for (const tier of ["PRIMARY", "SECONDARY", "BENIGN"] as const) {
      const clusterId = tierClusterId(tier);
      const wantPriority = String(profile.tiers[tier].priority);
      const key = `${ccId}||${clusterId}`;

      const existing = dxpByCcCluster.get(key);
      if (!existing) {
        const newRow: Record<string, string> = { CC_ID: ccId, CLUSTER_ID: clusterId, PRIORITY: wantPriority };
        dxp.rows.push(newRow);
        dxpByCcCluster.set(key, newRow);
        p.dxpAdded++;
      } else if (existing.PRIORITY !== wantPriority) {
        existing.PRIORITY = wantPriority;
        p.dxpUpdated++;
      }
    }

    return p;
  };

  const tasks = seed.map((r) =>
    limiter(async () => runOne(r.CC_ID, r.PROFILE_ID)).then((p) => {
      plans.push(p);
      const status = p.ok ? "✅" : "❌";
      console.log(
        `${status} ${p.ccId} -> ${p.profileId} | CSR +${p.csrAdded} upd ${p.csrUpdated} | DXP +${p.dxpAdded} upd ${p.dxpUpdated}`
      );
      if (!p.ok && p.errors.length) for (const e of p.errors) console.log(`   - ${e}`);
      if (!p.ok && !args.continueOnFail) throw new Error(`Stopping on failure (use --continue-on-fail): ${p.ccId}`);
    })
  );

  Promise.allSettled(tasks)
    .then((settled) => {
      const hadStop = settled.some((s) => s.status === "rejected");
      const okCount = plans.filter((p) => p.ok).length;
      const failCount = plans.length - okCount;

      const summary = {
        seedPath: args.seedPath,
        dryRun: args.dryRun,
        continueOnFail: args.continueOnFail,
        onlyProfile: args.onlyProfile ?? null,
        ccs: args.ccs,
        parallel: args.parallel,
        attempted: plans.length,
        ok: okCount,
        fail: failCount,
        totals: {
          csrAdded: plans.reduce((a, p) => a + p.csrAdded, 0),
          csrUpdated: plans.reduce((a, p) => a + p.csrUpdated, 0),
          dxpAdded: plans.reduce((a, p) => a + p.dxpAdded, 0),
          dxpUpdated: plans.reduce((a, p) => a + p.dxpUpdated, 0),
        },
        results: plans,
        generated_at: new Date().toISOString(),
      };

      if (args.summaryJsonPath) {
        if (args.dryRun) console.log(`[DRY] Would write summary JSON -> ${args.summaryJsonPath}`);
        else {
          fs.mkdirSync(path.dirname(args.summaryJsonPath), { recursive: true });
          fs.writeFileSync(args.summaryJsonPath, JSON.stringify(summary, null, 2) + "\n", "utf8");
          console.log(`Wrote summary JSON: ${args.summaryJsonPath}`);
        }
      }

      if (args.dryRun) {
        console.log("\n[DRY] No files written.");
        process.exit(hadStop || failCount > 0 ? 1 : 0);
        return;
      }

      // Transaction backup
      const runId = args.runId ?? makeRunId();
      const backupBase = path.isAbsolute(args.backupDir) ? args.backupDir : path.join(root, args.backupDir);
      const runDir = path.join(backupBase, runId);

      let csrBak = "";
      let dxpBak = "";

      if (!args.noBackup) {
        fs.mkdirSync(runDir, { recursive: true });
        csrBak = path.join(runDir, "CLUSTER_SCORING_RULES.csv.bak");
        dxpBak = path.join(runDir, "DX_PRIORITY.csv.bak");

        copyFile(CSR_PATH, csrBak);
        copyFile(DXP_PATH, dxpBak);

        fs.writeFileSync(
          path.join(runDir, "manifest.json"),
          JSON.stringify(
            {
              run_id: runId,
              created_at: new Date().toISOString(),
              seed_path: args.seedPath,
              backup_dir: runDir,
              files: [
                { src: CSR_PATH, backup: csrBak },
                { src: DXP_PATH, backup: dxpBak },
              ],
              summary,
            },
            null,
            2
          ) + "\n",
          "utf8"
        );

        console.log(`\n🧾 Transaction backup created: ${runDir}`);
        console.log(`Rollback with: npx tsx scripts/apply-profile-pack-bulk.ts --rollback ${runId}`);
      }

      // BELT & SUSPENDERS COMMIT
      const csrContent = encodeCsv(csr.headers, csr.rows);
      const dxpContent = encodeCsv(dxp.headers, dxp.rows);

      if (args.noAtomic) {
        fs.writeFileSync(CSR_PATH, csrContent, "utf8");
        fs.writeFileSync(DXP_PATH, dxpContent, "utf8");
      } else {
        atomicWriteFile(CSR_PATH, csrContent, runId, args.keepPrev);
        atomicWriteFile(DXP_PATH, dxpContent, runId, args.keepPrev);
      }

      console.log("\nCommitted CSR + DXP (atomic).");

      // Validation with auto-rollback
      if (!args.noValidate) {
        console.log(`\n🔎 Validation: ${args.validateCmd}`);
        const v = runShell(args.validateCmd);

        if (!v.ok) {
          console.log(`\n❌ Validation failed (exit ${v.code}).`);

          if (args.noBackup) {
            console.log("⚠️ No backup created (--no-backup). Cannot auto-rollback safely.");
          } else if (args.noAutoRollback) {
            console.log("⚠️ Auto-rollback disabled (--no-auto-rollback).");
            console.log(`Backup run id: ${runId}`);
          } else {
            console.log("↩️  Auto-rollback to transaction backup...");
            try {
              if (!csrBak || !dxpBak) {
                csrBak = path.join(runDir, "CLUSTER_SCORING_RULES.csv.bak");
                dxpBak = path.join(runDir, "DX_PRIORITY.csv.bak");
              }
              rollbackNow(csrBak, dxpBak, CSR_PATH, DXP_PATH);
              console.log("✅ Rollback complete.");
              console.log(`Backup run id: ${runId}`);
            } catch (e: any) {
              console.log("⚠️ Rollback failed:", e?.message ?? String(e));
              console.log(`Backup run id: ${runId}`);
            }
          }

          process.exit(1);
          return;
        }

        console.log("✅ Validation passed.");
      }

      process.exit(hadStop || failCount > 0 ? 1 : 0);
    })
    .catch((e) => {
      console.error("Fatal:", e?.stack ?? e);
      process.exit(1);
    });
}

main();
