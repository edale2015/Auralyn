CREATE TABLE kb_feature_models (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  rule_id TEXT NOT NULL,

  feature_key TEXT NOT NULL,              -- "painful_arc"
  feature_type TEXT NOT NULL,             -- boolean | categorical | numeric | range

  -- POSITIVE EVIDENCE
  p_present FLOAT,                        -- P(feature present | Dx)

  -- NEGATIVE EVIDENCE
  p_absent FLOAT,                         -- P(feature absent | Dx)

  -- CATEGORICAL VALUES (JSON)
  categorical_map JSONB,                  -- { "mild":0.3, "moderate":0.6, "severe":0.9 }

  -- NUMERIC MODEL
  mean FLOAT,
  std_dev FLOAT,

  -- RANGE MODEL
  min_value FLOAT,
  max_value FLOAT,

  weight FLOAT DEFAULT 1.0,
  is_required BOOLEAN DEFAULT FALSE,      -- if true, absence penalizes heavily

  source TEXT DEFAULT 'manual',
  created_at TIMESTAMP DEFAULT now()
);

🧠 NEW SCORING ENGINE (DROP-IN)

function gaussian(x, mean, std) {
  return Math.exp(-((x - mean) ** 2) / (2 * std ** 2))
}

export async function runAdvancedDiagnosis(input: any) {
  const diagnoses = await db.kb_diagnosis_rules.findAll({ is_active: true })

  const results = []

  for (const dx of diagnoses) {
    const features = await db.kb_feature_models.findAll({
      rule_id: dx.rule_id
    })

    let score = Math.log(dx.base_probability)

    for (const f of features) {
      const val = input[f.feature_key]

      // BOOLEAN
      if (f.feature_type === 'boolean') {
        if (val === true) score += Math.log(f.p_present || 0.5)
        else score += Math.log(f.p_absent || 0.5)
      }

      // CATEGORICAL
      else if (f.feature_type === 'categorical') {
        const likelihood = f.categorical_map?.[val] || 0.01
        score += Math.log(likelihood)
      }

      // NUMERIC
      else if (f.feature_type === 'numeric') {
        const likelihood = gaussian(val, f.mean, f.std_dev)
        score += Math.log(likelihood)
      }

      // RANGE
      else if (f.feature_type === 'range') {
        if (val >= f.min_value && val <= f.max_value) {
          score += Math.log(f.p_present || 0.8)
        } else {
          score += Math.log(f.p_absent || 0.2)
        }
      }
    }

    results.push({
      diagnosis: dx.diagnosis_name,
      score,
      source: 'KB_DB'
    })
  }

  return results.sort((a, b) => b.score - a.score)
}
