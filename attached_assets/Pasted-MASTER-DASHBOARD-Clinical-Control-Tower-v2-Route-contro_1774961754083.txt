const wss = new WebSocket.Server({ port: 8080 })

export function broadcastUpdate(data) {
  wss.clients.forEach(client => {
    if (client.readyState === WebSocket.OPEN) {
      client.send(JSON.stringify(data))
    }
  })
}

// client/components/LivePatientPanel.tsx
export default function LivePatientPanel() {
  const [patients, setPatients] = useState([])

  useEffect(() => {
    const ws = new WebSocket('ws://localhost:8080')
    ws.onmessage = (e) => {
      const data = JSON.parse(e.data)
      setPatients(prev => [...prev, data])
    }
  }, [])

  return (
    <div>
      <h3>Live Patients</h3>
      {patients.map((p, i) => (
        <div key={i}>
          {p.patient_id} — HR: {p.heart_rate} — Risk: {p.risk}
        </div>
      ))}
    </div>
  )
}

🧠 Entity extraction
async function extractClinicalEntities(text: string) {
  return await openai.chat.completions.create({
    model: 'gpt-5',
    messages: [{
      role: 'system',
      content: 'Extract symptoms, duration, severity, vitals'
    }, {
      role: 'user',
      content: text
    }]
  })
}

// server/engine/deteriorationEngine.ts
export async function detectDeterioration(patientId) {
  const series = await db.patient_live_stream.findRecent(patientId)
  const rules = await db.kb_deterioration_rules.findAll()

  let risk = 0
  const alerts = []

  for (const r of rules) {
    const values = series
      .filter(s => s.feature_key === r.feature_key)
      .map(s => s.value)

    if (values.length < 2) continue

    const trend = values[values.length - 1] - values[0]

    if (r.trend === 'rising' && trend > r.threshold) {
      risk += r.risk_weight
      alerts.push(r)
    }
  }

  return { risk, alerts }
}
