import { AgentOpinion } from "../agents/agentTypes";

export interface ReasoningNode {
  id: string;
  type: string;
  weight: number;
  data: any;
}

export function buildReasoningGraphFromTelemed(opinions: AgentOpinion[]) {
  const nodes: ReasoningNode[] = [];

  for (const op of opinions) {
    nodes.push({
      id: op.agentId,
      type: op.domain,
      weight: op.confidence * (op.priority ?? 1),
      data: op,
    });
  }

  return nodes;
}

📁 server/ws/telemedCognitiveStream.ts
import { Server } from "ws";

interface CognitivePayload {
  caseId: string;
  agents: any[];
  council: any;
  finalDecision: any;
  reasoningGraph: any[];
  timestamp: number;
}

const clients = new Set<any>();

export function initTelemedCognitiveStream(wss: Server) {
  wss.on("connection", (ws) => {
    clients.add(ws);

    ws.on("close", () => {
      clients.delete(ws);
    });
  });
}

export function broadcastCognitiveUpdate(payload: CognitivePayload) {
  const message = JSON.stringify({
    type: "telemed_cognition",
    data: payload,
  });

  for (const client of clients) {
    if (client.readyState === 1) {
      client.send(message);
    }
  }
}
