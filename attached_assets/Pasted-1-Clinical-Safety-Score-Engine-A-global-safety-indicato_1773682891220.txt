1️⃣ Clinical Safety Score Engine

A global safety indicator that continuously evaluates the system.

It calculates a 0–100 safety score from multiple subscores:

Metric	Weight
Red Flag Detection Accuracy	35%
Diagnostic Accuracy	30%
Protocol Compliance	20%
Question Coverage	10%
Disposition Agreement	5%
Backend
server/safety/clinicalSafetyScoreEngine.ts
import crypto from "crypto"

export interface SafetyMetrics {
  redFlagAccuracy: number
  diagnosticAccuracy: number
  protocolCompliance: number
  questionCoverage: number
  dispositionAgreement: number
}

export interface SafetyScore {
  score: number
  metrics: SafetyMetrics
  timestamp: number
  hash: string
}

export class ClinicalSafetyScoreEngine {

  calculate(metrics: SafetyMetrics): SafetyScore {

    const score =
      metrics.redFlagAccuracy * 0.35 +
      metrics.diagnosticAccuracy * 0.30 +
      metrics.protocolCompliance * 0.20 +
      metrics.questionCoverage * 0.10 +
      metrics.dispositionAgreement * 0.05

    const payload = JSON.stringify(metrics)

    const hash = crypto
      .createHash("sha256")
      .update(payload)
      .digest("hex")

    return {
      score: Number(score.toFixed(2)),
      metrics,
      timestamp: Date.now(),
      hash
    }
  }
}

export const safetyScoreEngine = new ClinicalSafetyScoreEngine()

interface EngineStats {

  engineName: string
  calls: number
  totalLatency: number
  errors: number
  cost: number
}

const engineStats: Record<string, EngineStats> = {}

export function recordEngineCall(
  engineName: string,
  latency: number,
  cost: number,
  error: boolean
) {

  if (!engineStats[engineName]) {

    engineStats[engineName] = {
      engineName,
      calls: 0,
      totalLatency: 0,
      errors: 0,
      cost: 0
    }
  }

  const stat = engineStats[engineName]

  stat.calls++
  stat.totalLatency += latency
  stat.cost += cost

  if (error) stat.errors++
}

export function getEngineStats() {
  return Object.values(engineStats)
}
4️⃣ Knowledge Graph Gap Visualizer

This finds holes in your medical knowledge graph.

Detects:

• missing complaint questions
• missing protocols
• missing diagnoses
• missing triage rules

server/graph/gapAnalyzer.ts
export interface GraphGap {

  type: string
  nodeId: string
  description: string
}

export class KnowledgeGraphGapAnalyzer {

  analyze(graph: any): GraphGap[] {

    const gaps: GraphGap[] = []

    graph.complaints.forEach((c: any) => {

      if (!c.questions || c.questions.length === 0) {
        gaps.push({
          type: "missing_questions",
          nodeId: c.id,
          description: "Complaint has no questions"
        })
      }

      if (!c.protocol) {
        gaps.push({
          type: "missing_protocol",
          nodeId: c.id,
          description: "No protocol attached"
        })
      }
    })

    return gaps
  }
}

export const gapAnalyzer = new KnowledgeGraphGapAnalyzer()

interface EngineWeights {

  bayesian: number
  similarity: number
  protocol: number
}

export class AutonomousTuningAgent {

  weights: EngineWeights = {
    bayesian: 0.5,
    similarity: 0.3,
    protocol: 0.2
  }

  adjust(outcomes: any[]) {

    const failures = outcomes.filter(o => o.correct === false)

    if (failures.length > outcomes.length * 0.2) {

      this.weights.protocol += 0.05
      this.weights.bayesian -= 0.05
    }

    return this.weights
  }
}

1️⃣ Backend — Control Center Service
server/controlCenter/controlCenterService.ts
import { safetyScoreEngine } from "../safety/clinicalSafetyScoreEngine"
import { getEngineStats } from "../performance/engineProfiler"
import { gapAnalyzer } from "../graph/gapAnalyzer"

export interface ControlCenterSnapshot {

  safetyScore: any
  engineStats: any[]
  graphGaps: any[]
  systemHealth: string
  timestamp: number
}

export class ControlCenterService {

  generateSnapshot(graph: any, metrics: any): ControlCenterSnapshot {

    const safety = safetyScoreEngine.calculate(metrics)

    const engines = getEngineStats()

    const gaps = gapAnalyzer.analyze(graph)

    let health = "healthy"

    if (safety.score < 70) health = "warning"

    if (safety.score < 50) health = "critical"

    return {
      safetyScore: safety,
      engineStats: engines,
      graphGaps: gaps,
      systemHealth: health,
      timestamp: Date.now()
    }
  }
}

export const controlCenterService = new ControlCenterService()

3️⃣ Backend — Alert Engine

Detects critical problems.

server/controlCenter/alertEngine.ts
export interface Alert {

  id: string
  level: "info" | "warning" | "critical"
  message: string
  timestamp: number
}

export class AlertEngine {

  detect(snapshot: any): Alert[] {

    const alerts: Alert[] = []

    if (snapshot.safetyScore.score < 60) {

      alerts.push({
        id: "safety_low",
        level: "critical",
        message: "Clinical safety score below threshold",
        timestamp: Date.now()
      })
    }

    if (snapshot.graphGaps.length > 10) {

      alerts.push({
        id: "graph_gaps",
        level: "warning",
        message: "Knowledge graph contains missing protocol or question nodes",
        timestamp: Date.now()
      })
    }

    return alerts
  }
}

export const alertEngine = new AlertEngine()
