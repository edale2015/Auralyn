// server/testing/systemRunner.ts

export interface SystemRunResult {
  caseId: string;

  parsed?: ParsedComplaint;
  resolvedComplaint?: string;

  nodeTrace: NodeExecutionResult[];

  posterior?: PosteriorAnalysis;

  decision?: DecisionContext;

  patientResponse?: string;

  errors: string[];
}

export async function runSystemTestCase(
  test: SystemTestCase
): Promise<SystemRunResult> {

  const errors: string[] = [];

  const caseId = `test-${test.id}-${Date.now()}`;

  let state: CaseState = {
    caseId,
    symptoms: [],
    ...test.input.patientContext,
  };

  let parsed: ParsedComplaint | undefined;
  let config: ComplaintConfig | undefined;
  let posterior: PosteriorAnalysis | undefined;
  let decision: DecisionContext | undefined;
  let nodeTrace: NodeExecutionResult[] = [];

  try {
    // ── Parse ─────────────────────────────
    parsed = parseComplaint(test.input.message);

    // ── Resolve complaint ─────────────────
    const resolved = await resolveComplaint(parsed);
    config = await loadComplaintConfig(resolved.ccId);

    // ── Run nodes ─────────────────────────
    const nodeResult = await runComplaintNodes(config, state);
    nodeTrace = nodeResult.nodeResults;

    if (!nodeResult.success) {
      errors.push(`Node failure at ${nodeResult.failedNode}`);
      return { caseId, parsed, nodeTrace, errors };
    }

    state = nodeResult.finalState;

    // ── Load priors ───────────────────────
    const priorsBundle = await loadComplaintPriors(config.ccId);

    // ── Bayesian differential ─────────────
    const differential = runDifferential(
      state.symptoms,
      priorsBundle.priors
    );

    posterior = analyzePosterior(differential);

    // ── Final decision ────────────────────
    decision = await computeFinalDecision({
      state,
      posterior,
      erProbability: state.scores?.erRisk ?? 0,
      store: getTestEscalationStore(),
    });

    // ── Patient output ────────────────────
    const response = buildPatientResponse(decision);

    return {
      caseId,
      parsed,
      resolvedComplaint: config.ccId,
      nodeTrace,
      posterior,
      decision,
      patientResponse: response,
      errors,
    };

  } catch (err) {
    errors.push(err instanceof Error ? err.message : String(err));

    return {
      caseId,
      parsed,
      resolvedComplaint: config?.ccId,
      nodeTrace,
      posterior,
      decision,
      errors,
    };
  }
}
C) Validator (this is where truth is enforced)
// server/testing/validator.ts

export interface ValidationResult {
  passed: boolean;
  failures: string[];
}

export function validateTestCase(
  test: SystemTestCase,
  result: SystemRunResult
): ValidationResult {

  const failures: string[] = [];

  const decision = result.decision;

  if (!decision) {
    failures.push("No decision produced");
    return { passed: false, failures };
  }

  // ── Disposition ───────────────────────
  if (test.expected.disposition) {
    if (decision.finalDisposition !== test.expected.disposition) {
      failures.push(
        `Expected disposition ${test.expected.disposition}, got ${decision.finalDisposition}`
      );
    }
  }

  // ── Diagnosis ─────────────────────────
  if (test.expected.primaryDiagnosis) {
    if (decision.posterior.topDiagnosis !== test.expected.primaryDiagnosis) {
      failures.push(
        `Expected diagnosis ${test.expected.primaryDiagnosis}, got ${decision.posterior.topDiagnosis}`
      );
    }
  }

  // ── Must include differential ─────────
  for (const d of test.expected.mustIncludeDifferential ?? []) {
    if (!decision.posterior.differential.some(x => x.diagnosis === d)) {
      failures.push(`Missing expected differential: ${d}`);
    }
  }

  // ── Must not include ──────────────────
  for (const d of test.expected.mustNotIncludeDifferential ?? []) {
    if (decision.posterior.differential.some(x => x.diagnosis === d)) {
      failures.push(`Unexpected differential present: ${d}`);
    }
  }

  // ── Safety gate ───────────────────────
  if (test.expected.mustTriggerSafetyGate) {
    if (decision.safety.allowed) {
      failures.push("Expected safety gate to block, but it passed");
    }
  }

  // ── Physician review ──────────────────
  if (test.expected.requiresPhysicianReview) {
    if (decision.finalDisposition !== "AWAITING_PHYSICIAN") {
      failures.push("Expected physician review but did not occur");
    }
  }

  return {
    passed: failures.length === 0,
    failures,
  };
}
D) Runner for multiple cases
// server/testing/testSuiteRunner.ts

export async function runTestSuite(
  tests: SystemTestCase[]
) {

  const results = [];

  for (const test of tests) {
    const run = await runSystemTestCase(test);
    const validation = validateTestCase(test, run);

    results.push({
      id: test.id,
      passed: validation.passed,
      failures: validation.failures,
      trace: run,
    });
  }

  return results;
}
