Clinical Skill Engine

Tracks required knowledge for a complaint.

server/engines/clinicalSkillEngine.ts

export class ClinicalSkillEngine {

  skillMap: any = {

    cough: [
      "red_flag_detection",
      "pneumonia_risk",
      "asthma_history",
      "chronic_cough"
    ],

    chest_pain: [
      "mi_risk",
      "pe_risk",
      "aortic_dissection",
      "gerd"
    ]

  }

  run(context:any){

    const skills = this.skillMap[context.complaint] || []

    return {
      ...context,
      requiredSkills: skills
    }

  }

}
Protocol Selection Engine

Chooses guideline source.

server/engines/protocolSelectionEngine.ts

export class ProtocolSelectionEngine {

 run(context:any){

   let protocol = "general_triage"

   if(context.complaint === "cough")
     protocol = "cdc_respiratory"

   if(context.complaint === "chest_pain")
     protocol = "emergency_cardiac"

   return {
     ...context,
     protocol
   }

 }

}
Confidence Calibration Engine

Detects over-confidence or under-confidence.

server/engines/confidenceCalibrationEngine.ts

export class ConfidenceCalibrationEngine {

 run(context:any){

   const confidence = context.confidence || 0

   let calibrated = confidence

   if(confidence > 0.9)
     calibrated = confidence * 0.85

   if(confidence < 0.4)
     calibrated = confidence + 0.1

   return {
     ...context,
     calibratedConfidence: calibrated
   }

 }

}
Clinical Simulation Engine

Creates synthetic cases for testing.

server/engines/clinicalSimulationEngine.ts

export class ClinicalSimulationEngine {

 generateCase(complaint:string){

   const syntheticCase = {

     complaint,
     age: Math.floor(Math.random()*70),
     fever: Math.random() > 0.5,
     duration: Math.floor(Math.random()*10),
     sob: Math.random() > 0.7

   }

   return syntheticCase

 }

}
