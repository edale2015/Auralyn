export function rankGuidelines(g) {
  const evidence =
    g.type === 'meta_analysis' ? 5 :
    g.type === 'RCT' ? 4 :
    g.type === 'cohort' ? 3 : 1

  const recency = (new Date().getFullYear() - g.year < 5) ? 2 : 0
  const consensus = g.citedBy > 50 ? 2 : 1

  const final = evidence + recency + consensus

  return { ...g, final_score: final }
}

export function computeCalibration(data) {
  const bins = Array(10).fill(0).map(()=>({pred:0, actual:0, n:0}))

  data.forEach(d=>{
    const i = Math.floor(d.predicted_prob * 10)
    bins[i].pred += d.predicted_prob
    bins[i].actual += d.actual_outcome
    bins[i].n++
  })

  return bins.map(b=>({
    predicted: b.pred / (b.n||1),
    actual: b.actual / (b.n||1)
  }))
}

// server/causal/ate.ts
type Row = { treatment: number; outcome: number; pscore: number; yhat?: number }

export function computeATE_IPW(rows: Row[]) {
  let treated = 0, control = 0

  rows.forEach(r => {
    if (r.treatment === 1) treated += r.outcome / Math.max(1e-3, r.pscore)
    else control += r.outcome / Math.max(1e-3, (1 - r.pscore))
  })

  return (treated / rows.length) - (control / rows.length)
}

// Doubly-robust (uses outcome model yhat if available)
export function computeATE_DR(rows: Row[]) {
  let sum = 0
  rows.forEach(r => {
    const mu = r.yhat ?? 0
    const w = r.treatment / Math.max(1e-3, r.pscore) - (1 - r.treatment) / Math.max(1e-3, 1 - r.pscore)
    sum += mu + w * (r.outcome - mu)
  })
  return sum / rows.length
}

// server/calibration/recalibrate.ts
export function plattScale(p: number, a: number, b: number) {
  const z = a * p + b
  return 1 / (1 + Math.exp(-z))
}

// isotonic (piecewise via breakpoints)
export function isotonicScale(p: number, breakpoints: {x:number;y:number}[]) {
  // assume sorted
  for (let i=0;i<breakpoints.length-1;i++){
    const a = breakpoints[i], b = breakpoints[i+1]
    if (p >= a.x && p <= b.x) {
      const t = (p - a.x) / Math.max(1e-6, (b.x - a.x))
      return a.y + t * (b.y - a.y)
    }
  }
  return p
}

export function applyCalibration(p: number, model: any) {
  if (model.type === 'platt') {
    const { a, b } = model.params
    return plattScale(p, a, b)
  }
  if (model.type === 'isotonic') {
    return isotonicScale(p, model.params.breakpoints)
  }
  return p
}
