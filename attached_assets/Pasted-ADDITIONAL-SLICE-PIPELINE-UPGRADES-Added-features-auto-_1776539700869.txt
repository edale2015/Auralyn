import { db } from '../db/client';
import { sliceAuditLog } from '../db/schema';


export async function logSliceAudit(args: {
  reviewSliceId?: number | null;
  proposalId?: number | null;
  eventType: string;
  payload?: any;
}) {
  const inserted = await db.insert(sliceAuditLog).values({
    reviewSliceId: args.reviewSliceId ?? null,
    proposalId: args.proposalId ?? null,
    eventType: args.eventType,
    payload: args.payload ?? {},
  }).returning();


  return inserted[0];
}

import express from 'express';
import { db } from '../db/client';
import { reviewSlices, claudeSliceDrafts } from '../db/schema';
import { eq } from 'drizzle-orm';
import { logSliceAudit } from '../services/sliceAuditService';


const router = express.Router();


router.get('/draft/:sliceId', async (req, res) => {
  try {
    const slices = await db.select().from(reviewSlices).where(eq(reviewSlices.sliceId, req.params.sliceId));
    const slice = slices[0];
    if (!slice) return res.status(404).json({ ok: false, error: 'Slice not found' });


    const drafts = await db.select().from(claudeSliceDrafts).where(eq(claudeSliceDrafts.reviewSliceId, slice.id));
    const draft = drafts.at(-1) ?? null;
    res.json({ ok: true, draft });
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});


router.post('/draft/:sliceId', async (req, res) => {
  try {
    const slices = await db.select().from(reviewSlices).where(eq(reviewSlices.sliceId, req.params.sliceId));
    const slice = slices[0];
    if (!slice) return res.status(404).json({ ok: false, error: 'Slice not found' });


    const draftText = String(req.body.draftText || '');


    const inserted = await db.insert(claudeSliceDrafts).values({
      reviewSliceId: slice.id,
      draftText,
      updatedAt: new Date(),
    }).returning();


    await logSliceAudit({
      reviewSliceId: slice.id,
      eventType: 'claude_draft_saved',
      payload: { length: draftText.length },
    });


    res.json({ ok: true, draft: inserted[0] });
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});


export default router;

function makeSimpleLineDiff(oldText: string, newText: string) {
  const oldLines = oldText.split('
');
  const newLines = newText.split('
');
  const max = Math.max(oldLines.length, newLines.length);
  const rows: Array<{ type: 'same' | 'add' | 'remove' | 'change'; oldLine?: string; newLine?: string }> = [];


  for (let i = 0; i < max; i++) {
    const oldLine = oldLines[i];
    const newLine = newLines[i];
    if (oldLine === newLine) rows.push({ type: 'same', oldLine, newLine });
    else if (oldLine === undefined) rows.push({ type: 'add', newLine });
    else if (newLine === undefined) rows.push({ type: 'remove', oldLine });
    else rows.push({ type: 'change', oldLine, newLine });
  }


  return rows;
}


router.get('/proposal/:proposalId', async (req, res) => {
  try {
    const rows = await db.select().from(sliceProposals).where(eq(sliceProposals.id, Number(req.params.proposalId)));
    const proposal = rows[0];
    if (!proposal) return res.status(404).json({ ok: false, error: 'Proposal not found' });


    const patchBundle = (proposal.patchBundle || {}) as Record<string, string>;
    const diffs: Record<string, any[]> = {};


    for (const [filePath, proposed] of Object.entries(patchBundle)) {
      let current = '';
      try {
        current = fs.readFileSync(filePath, 'utf8');
      } catch {
        current = '';
      }
      diffs[filePath] = makeSimpleLineDiff(current, proposed);
    }


    res.json({ ok: true, diffs });
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});


export default router;

server/routes/sliceBulkRoutes.ts
import express from 'express';
import { db } from '../db/client';
import { reviewSlices, sliceProposals } from '../db/schema';
import { logSliceAudit } from '../services/sliceAuditService';
import { reviewClaudeSliceWithOpenAI } from '../research/openaiSliceReview';
import { buildSliceProposals } from '../research/sliceProposalBuilder';
import { validateSliceProposal } from '../research/sliceValidation';


const router = express.Router();


router.post('/run-all', async (_req, res) => {
  try {
    const slices = await db.select().from(reviewSlices);
    const progress: any[] = [];


    for (const slice of slices) {
      const item: any = { sliceId: slice.sliceId, openai: 'pending', proposals: 0, validation: 'pending' };
      try {
        await reviewClaudeSliceWithOpenAI(slice.sliceId);
        item.openai = 'completed';


        const proposals = await buildSliceProposals(slice.sliceId);
        item.proposals = proposals.length;


        for (const proposal of proposals) {
          const result = await validateSliceProposal(proposal.id);
          item.validation = result.status;
          await logSliceAudit({
            reviewSliceId: slice.id,
            proposalId: proposal.id,
            eventType: 'bulk_validation_completed',
            payload: result,
          });
        }
      } catch (error: any) {
        item.error = error?.message ?? 'Unknown error';
        item.validation = 'failed';
      }
      progress.push(item);
    }


    res.json({ ok: true, progress });
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});


router.get('/progress', async (_req, res) => {
  try {
    const proposals = await db.select().from(sliceProposals);
    const summary = {
      total: proposals.length,
      passed: proposals.filter((p) => p.validationStatus === 'passed').length,
      failed: proposals.filter((p) => p.validationStatus === 'failed').length,
      pending: proposals.filter((p) => p.validationStatus === 'pending').length,
      approved: proposals.filter((p) => p.approved).length,
    };
    res.json({ ok: true, summary, proposals });
  } catch (error: any) {
    res.status(500).json({ ok: false, error: error?.message ?? 'Unknown error' });
  }
});


export default router;

async function runBulk() {
  setLoading('bulk');
  setMessage('');
  try {
    const res = await fetch('/api/slice-bulk/run-all', { method: 'POST' });
    const data = await res.json();
    if (!res.ok || data.ok === false) throw new Error(data.error || 'Bulk run failed');
    setBulkSummary({ progress: data.progress });
    setMessage('Bulk slice run completed.');
  } catch (err: any) {
    setMessage(err?.message ?? 'Bulk run failed');
  } finally {
    setLoading(null);
  }
}


async function loadBulkProgress() {
  try {
    const res = await fetch('/api/slice-bulk/progress');

    // ignore
  }
}
