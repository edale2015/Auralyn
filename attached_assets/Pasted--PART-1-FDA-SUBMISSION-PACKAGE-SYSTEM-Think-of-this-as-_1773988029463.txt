🧪 1. VALIDATION RUNNER (SIMULATED CLINICAL TRIAL)
import { runFullClinicalFlow } from "../orchestrator/clinicalOrchestrator";

export async function runValidation(dataset: any[]) {
  const results = [];

  for (const caseData of dataset) {
    const res = await runFullClinicalFlow(caseData.input);

    results.push({
      input: caseData.input,
      predicted: res?.diagnosis?.primary || null,
      actual: caseData.actual
    });
  }

  return results;
}
📊 2. METRICS ENGINE (FDA-STYLE)
export function computeMetrics(results) {
  let tp = 0, fp = 0, fn = 0;

  for (const r of results) {
    if (r.predicted === r.actual) tp++;
    else {
      fp++;
      fn++;
    }
  }

  const sensitivity = tp / (tp + fn || 1);
  const precision = tp / (tp + fp || 1);

  return {
    total: results.length,
    sensitivity,
    precision,
    accuracy: tp / results.length
  };
}
📄 3. REPORT GENERATOR
import fs from "fs";

export function generateFDAReport(metrics, results) {
  const report = {
    summary: "Clinical AI Validation Report",
    metrics,
    sampleResults: results.slice(0, 20),
    timestamp: new Date()
  };

  fs.writeFileSync("fda_report.json", JSON.stringify(report, null, 2));

  return report;
}
📦 4. EXPORT BUNDLE
import archiver from "archiver";
import fs from "fs";

export function exportFDABundle() {
  const output = fs.createWriteStream("fda_bundle.zip");
  const archive = archiver("zip");

  archive.pipe(output);
  archive.file("fda_report.json", { name: "report.json" });
  archive.finalize();
}
