🤖 2. MULTI-AGENT CLINICAL COUNCIL
📁 server/agents/clinicalCouncil.ts
type Opinion = {
  agent: string;
  diagnosis: string;
  confidence: number;
};

export function clinicalCouncil(opinions: Opinion[]) {
  const scores: any = {};

  for (const o of opinions) {
    if (!scores[o.diagnosis]) scores[o.diagnosis] = 0;
    scores[o.diagnosis] += o.confidence;
  }

  const best = Object.entries(scores)
    .sort((a: any, b: any) => b[1] - a[1])[0];

  return {
    consensusDiagnosis: best[0],
    score: best[1],
    disagreement: opinions.length > 2,
  };
}

📉 3. TIME-SERIES BAYESIAN MODEL (TREND AWARE)
📁 server/prediction/timeSeriesEngine.ts
type History = {
  hr: number[];
  bp: number[];
  spo2: number[];
};

export function computeTrendRisk(history: History) {
  const slope = (arr: number[]) =>
    arr[arr.length - 1] - arr[0];

  const risk = {
    hrTrend: slope(history.hr),
    bpTrend: slope(history.bp),
    spo2Trend: slope(history.spo2),
  };

  let score = 0;

  if (risk.hrTrend > 20) score += 0.3;
  if (risk.bpTrend < -20) score += 0.4;
  if (risk.spo2Trend < -5) score += 0.4;

  return Math.min(score, 1);
}
🧪 4. FDA CLINICAL TRIAL SIMULATOR
📁 server/fda/clinicalTrialSim.ts
import { runDiagnosis } from "../clinical/diagnosisEngine";

export async function runClinicalTrial(cases: any[]) {
  let correct = 0;
  let total = cases.length;

  for (const c of cases) {
    const result = await runDiagnosis(c, global.kb);

    const top = Object.entries(result.posterior)
      .sort((a: any, b: any) => b[1] - a[1])[0][0];

    if (top === c.expectedDiagnosis) {
      correct++;
    }
  }

  return {
    accuracy: correct / total,
    total,
  };
}
📁 Expanded metrics
export function buildTrialMetrics(results: any[]) {
  return {
    sensitivity: results.filter(r => r.truePositive).length / results.length,
    specificity: results.filter(r => r.trueNegative).length / results.length,
    errorRate: results.filter(r => r.error).length / results.length,
  };
}

import { useEffect, useState } from "react";
import { io } from "socket.io-client";

export default function ICUDashboard() {
  const [patients, setPatients] = useState<any[]>([]);

  useEffect(() => {
    const socket = io("http://localhost:3000");

    socket.on("vitals_update", (data) => {
      setPatients(prev => [data, ...prev.slice(0, 50)]);
    });

    return () => socket.disconnect();
  }, []);

  return (
    <div className="p-6 bg-black text-white">
      <h1>🏥 ICU Live Feed</h1>

      {patients.map((p, i) => (
        <div key={i} className="border-b p-2">
          {p.patientId} | HR: {p.vitals.hr.toFixed(0)} | BP: {p.vitals.bp.toFixed(0)} | O2: {p.vitals.spo2.toFixed(0)}
        </div>
      ))}
    </div>
  );
}
