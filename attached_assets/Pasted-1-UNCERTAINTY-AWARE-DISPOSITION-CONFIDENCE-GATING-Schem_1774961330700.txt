⚙️ Engine
// server/engine/confidenceDisposition.ts
type Dx = { diagnosis: string; posterior: number }

function softmaxLogScores(items: { score: number }[]) {
  const max = Math.max(...items.map(i => i.score))
  const exps = items.map(i => Math.exp(i.score - max))
  const Z = exps.reduce((a, b) => a + b, 0)
  return exps.map(e => e / Z)
}

export async function computeDispositionWithUncertainty(
  complaintId: string,
  logScored: { diagnosis: string; score: number }[],
  baseDisposition: string
) {
  const posteriors = softmaxLogScores(logScored)
  const ranked = logScored
    .map((d, i) => ({ ...d, posterior: posteriors[i] }))
    .sort((a, b) => b.posterior - a.posterior)

  const top = ranked[0]
  const second = ranked[1]
  const uncertainty = 1 - (top?.posterior || 0)
  const margin = (top?.posterior || 0) - (second?.posterior || 0)

  const rules = await db.kb_confidence_rules.findAll({
    complaint_id: complaintId
  })

  let escalated = baseDisposition

  for (const r of rules.sort((a, b) => a.priority - b.priority)) {
    if (uncertainty >= r.min_confidence) {
      escalated = r.action
    }
  }

  // Diagnosis-specific floor (cannot go lower than X)
  const risk = await db.kb_diagnosis_risk.findOne({
    diagnosis: top?.diagnosis
  })
  if (risk && risk.min_disposition) {
    const order = ['MONITOR', 'URGENT', 'ER_NOW', 'CALL_911']
    if (order.indexOf(escalated) < order.indexOf(risk.min_disposition)) {
      escalated = risk.min_disposition
    }
  }

  return {
    disposition: escalated,
    uncertainty,
    margin,
    top_dx: top?.diagnosis,
    trace: {
      uncertainty,
      margin,
      rule_hits: rules.map(r => r.id)
    }
  }
}

⚙️ Engine (greedy value/cost selector)
// server/engine/workupOptimizer.ts
type CandidateTest = {
  test_name: string
  utility: number
  cost: number
  risk: number
}

export async function optimizeWorkup(
  currentDx: { diagnosis: string; posterior: number }[],
  budget = 1000
) {
  const tests = await db.kb_workup_costs.findAll({ is_active: true })
  const utils = await db.kb_test_utility.findAll({ is_active: true })

  // Compute utility weighted by current differential
  const scored: CandidateTest[] = []

  for (const t of tests) {
    let u = 0
    for (const d of currentDx) {
      const tu = utils.find(x => x.test_name === t.test_name && x.diagnosis === d.diagnosis)
      if (tu) u += d.posterior * tu.info_gain
    }
    const value = u / Math.max(1, t.cost) - (t.risk_score || 0)
    scored.push({
      test_name: t.test_name,
      utility: value,
      cost: t.cost,
      risk: t.risk_score || 0
    })
  }

  // Greedy selection under budget
  const selected: CandidateTest[] = []
  let remaining = budget

  for (const t of scored.sort((a, b) => b.utility - a.utility)) {
    if (t.cost <= remaining && t.utility > 0) {
      selected.push(t)
      remaining -= t.cost
    }
  }

  return {
    recommended: selected.map(t => t.test_name),
    total_cost: budget - remaining,
    trace: selected
  }
}

// server/engine/counterfactualExplainer.ts
export async function generateCounterfactuals(input: any, baseResults: any[]) {
  const top = baseResults[0]
  const features = await db.kb_feature_models.findAll({
    rule_id: `DX_BAY_${top.diagnosis.toUpperCase()}`
  })

  const suggestions = []

  for (const f of features) {
    const current = input[f.feature_key]

    // Try flipping boolean
    if (f.feature_type === 'boolean') {
      const flipped = !current
      const newInput = { ...input, [f.feature_key]: flipped }

      const newRes = await runAdvancedDiagnosis(newInput)
      const newTop = newRes[0]

      if (newTop.diagnosis !== top.diagnosis) {
        suggestions.push({
          feature: f.feature_key,
          change: flipped,
          new_top: newTop.diagnosis,
          impact: 'diagnosis_flip'
        })
      }
    }

    // Numeric perturbation
    if (f.feature_type === 'numeric') {
      const delta = f.std_dev || 10
      const newInput = { ...input, [f.feature_key]: current + delta }
      const newRes = await runAdvancedDiagnosis(newInput)
      if (newRes[0].diagnosis !== top.diagnosis) {
        suggestions.push({
          feature: f.feature_key,
          change: `+${delta}`,
          new_top: newRes[0].diagnosis,
          impact: 'diagnosis_flip'
        })
      }
    }
  }

  return suggestions.slice(0, 5) // top 5 most impactful
}
