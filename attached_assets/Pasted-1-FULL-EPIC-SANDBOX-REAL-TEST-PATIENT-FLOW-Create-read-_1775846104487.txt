export async function epicTestPatientFlow(token: string) {

  // 🔹 Create patient (sandbox supports this in some modes)

    method: "POST",
    headers: {
      Authorization: `Bearer ${token}`,
      "Content-Type": "application/fhir+json"
    },
    body: JSON.stringify({
      resourceType: "Patient",
      name: [{ given: ["Test"], family: "Patient" }]
    })
  }).then(r => r.json());

  // 🔹 Run triage
  const triage = await runFinalPipeline({
    patientId: patient.id,
    rawText: "chest pain",
    answers: {},
    sessionId: "epic-test",
    actorId: "system"
  }, "epic-test");

  // 🔹 Write result
  await postObservation(patient.id, token, {
    code: "Triage Result",
    value: triage.safetyResult.finalDisposition,
    unit: ""
  });

  return triage;
}
💰 2) PAYER CONTRACT SIMULATION (REAL CURVES)
🔧 Contract model
export function simulatePayerContract(claim: any) {

  const base = {
    "99285": 500,
    "99284": 300,
    "99213": 120
  };

  let reimbursement = base[claim.cpt] || 0;

  // modifiers
  if (claim.time > 60) reimbursement *= 1.1;
  if (claim.complexity === "high") reimbursement *= 1.2;

  // denial adjustment
  if (claim.denialRisk > 0.5) reimbursement *= 0.6;

  return reimbursement;
}

import { useState } from "react";
import { View, TextInput, Button, Text } from "react-native";

export default function App() {

  const [text, setText] = useState("");
  const [resp, setResp] = useState("");

  async function send() {
    const res = await fetch("https://api/patient/chat", {
      method: "POST",
      body: JSON.stringify({ text }),
      headers: { "Content-Type": "application/json" }
    });

    const data = await res.json();
    setResp(data.reply + " → " + data.disposition);
  }

  return (
    <View style={{ padding: 30 }}>
      <Text>🧠 Auralyn Patient</Text>
      <TextInput value={text} onChangeText={setText} />
      <Button title="Send" onPress={send} />
      <Text>{resp}</Text>
    </View>
  );
}

export function nextSecondaryQuestion(context: any) {

  if (!context.age) return "How old are you?";
  if (!context.symptoms.includes("fever")) return "Do you have a fever?";
  if (!context.duration) return "How long has this been going on?";

  return null;
}
