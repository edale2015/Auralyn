// server/engine/smartIntake.ts
export async function getSmartQuestions(db, dxDist, asked, complaintId) {
  const questions = await db.kb_question_logic.findAll({ complaint_id: complaintId })

  const scored = questions
    .filter(q => !asked.includes(q.question_key))
    .map(q => {
      let gain = 0
      for (const dx of dxDist) {
        gain += dx.posterior * q.info_gain
      }
      return {
        key: q.question_key,
        score: gain + q.red_flag_weight
      }
    })

  return scored.sort((a, b) => b.score - a.score).slice(0, 5)
}

⚙️ Backend
// server/engine/confidenceHeatmap.ts
export function buildHeatmap(features, diagnoses) {
  return diagnoses.map(dx => ({
    diagnosis: dx.name,
    contributions: features.map(f => ({
      feature: f.key,
      value: dx.featureMap?.[f.key] || 0
    }))
  }))
}
🎛️ UI
