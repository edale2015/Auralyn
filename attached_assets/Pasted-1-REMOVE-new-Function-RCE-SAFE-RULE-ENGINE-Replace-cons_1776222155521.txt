// server/monitoring/safeRuleEngine.ts

export type Operator = ">" | "<" | ">=" | "<=" | "==" | "!=";

export interface SafeRule {
  metric: string;
  operator: Operator;
  value: number;
}

export function evaluateSafeRule(rule: SafeRule, metrics: Record<string, number>): boolean {
  const metricValue = metrics[rule.metric];

  if (metricValue === undefined) return false;

  switch (rule.operator) {
    case ">": return metricValue > rule.value;
    case "<": return metricValue < rule.value;
    case ">=": return metricValue >= rule.value;
    case "<=": return metricValue <= rule.value;
    case "==": return metricValue === rule.value;
    case "!=": return metricValue !== rule.value;
    default: return false;
  }
}

// server/agents/safeWeightAdjuster.ts

export async function safeAdjustWeight(agent: string, score: number) {
  if (score < 0) {
    throw new Error(`Circuit breaker: ${agent} unsafe`);
  }

  const bounds = { min: 0.5, max: 1.5 };

  return db.transaction(async (tx) => {
    const current = await tx.select()
      .from("agent_weights")
      .where({ agent })
      .limit(1);

    let newWeight = current[0].weight * (1 + score * 0.01);

    newWeight = Math.max(bounds.min, Math.min(bounds.max, newWeight));

    await tx.update("agent_weights")
      .set({ weight: newWeight })
      .where({ agent });

    await auditStep({
      traceId: `weight-${agent}`,
      step: "weight_adjustment",
      input: { score },
      output: { newWeight }
    });

    return newWeight;
  });
}

    const parsed = parser.parse(expr);

  } catch (e) {
    console.error("[SafeEval] Failed:", expr);
