// server/clinical/complaintNodeRunner.ts

import { auditStep } from "../audit/auditLogger";

export type NodeId =
  | "RED_FLAGS"
  | "CORE_QUESTIONS"
  | "SCORING"
  | "DISPOSITION"
  | "TEMPLATE"
  | "EXAM_OVERRIDE"
  | "DIFF_CONFIDENCE"
  | "RED_FLAG_MASTER_JOIN"
  | "SPOT_INTERVENTIONS"
  | "SPECIALIST_COUNCIL";

export interface NodeExecutionResult {
  nodeId: NodeId;
  success: boolean;
  error?: string;
}

export interface RunnerResult {
  success: boolean;
  finalState: CaseState;
  nodeResults: NodeExecutionResult[];
  failedNode?: NodeId;
}

const REQUIRED_NODES: NodeId[] = [
  "RED_FLAGS",
  "CORE_QUESTIONS",
  "SCORING",
  "DISPOSITION",
];

export async function runComplaintNodes(
  config: ComplaintConfig,
  initialState: CaseState
): Promise<RunnerResult> {

  const state = { ...initialState };
  const nodeResults: NodeExecutionResult[] = [];

  // ── Validate required nodes exist ───────────────────────
  for (const required of REQUIRED_NODES) {
    if (!config.nodes.includes(required)) {
      throw new Error(
        `[NodeRunner] Missing required node: ${required} for complaint ${config.ccId}`
      );
    }
  }

  for (const nodeId of config.nodes) {
    try {
      const engine = getNodeEngine(nodeId);

      if (!engine) {
        throw new Error(`No engine registered for node ${nodeId}`);
      }

      const before = structuredClone(state);

      const updated = await engine(state, config);

      Object.assign(state, updated);

      nodeResults.push({ nodeId, success: true });

      await auditStep({
        traceId: state.caseId,
        step: `node_${nodeId}`,
        input: before,
        output: updated,
      });

    } catch (err) {
      const errorMsg =
        err instanceof Error ? err.message : String(err);

      nodeResults.push({
        nodeId,
        success: false,
        error: errorMsg,
      });

      // 🚨 Fail closed for critical nodes
      if (REQUIRED_NODES.includes(nodeId)) {
        return {
          success: false,
          finalState: state,
          nodeResults,
          failedNode: nodeId,
        };
      }

      // Non-critical node → continue
      console.warn(`[NodeRunner] Non-critical node failed: ${nodeId}`, errorMsg);
    }
  }

  return {
    success: true,
    finalState: state,
    nodeResults,
  };
}

// server/clinical/complaintConfigLoader.ts

const configCache = new Map<string, ComplaintConfig>();
const configMeta = new Map<string, { version: number; loadedAt: number }>();

const CACHE_TTL_MS = 5 * 60 * 1000; // 5 minutes

export async function loadComplaintConfig(
  ccId: string
): Promise<ComplaintConfig> {

  const cached = configCache.get(ccId);
  const meta = configMeta.get(ccId);

  const isFresh =
    cached &&
    meta &&
    Date.now() - meta.loadedAt < CACHE_TTL_MS;

  if (isFresh) {
    return cached;
  }

  try {
    const raw = await loadFromRegistry(ccId); // CSV / Sheets

    validateConfig(raw); // corruption guard

    configCache.set(ccId, raw);
    configMeta.set(ccId, {
      version: raw.version,
      loadedAt: Date.now(),
    });

    return raw;

  } catch (err) {
    console.error(`[ConfigLoader] Failed to load ${ccId}`, err);

    // 🧠 Fallback to last known good config
    if (cached) {
      console.warn(`[ConfigLoader] Using stale config for ${ccId}`);
      return cached;
    }

    throw err; // no fallback available
  }
}

export function scoringEngine(
  state: CaseState,
  config: ComplaintConfig
): Partial<CaseState> {

  const moduleId = config.scoringModule as ScoringModuleId;

  const scorer = SCORER_MAP[moduleId];

  if (!scorer) {
    throw new Error(
      `[Scoring] No scorer found for module "${config.scoringModule}" in ${config.ccId}`

  const result = scorer(state);

  if (!result || typeof result !== "object") {
    throw new Error(`[Scoring] Invalid scorer output for ${moduleId}`);

  return {
    scores: result.scores,
    differential: result.differential,
  };
}

function mergeState(

  update: NodeUpdate,
  nodeId: NodeId

  const next = { ...current };

// server/clinical/complaintResolver.ts

export interface ComplaintResolution {
  ccId: string;
  source: "primary" | "alias" | "fallback";
  confidence: "high" | "low";
}

export async function resolveComplaint(
  parsed: ParsedComplaint | undefined
): Promise<ComplaintResolution> {

  if (!parsed) {
    return fallbackComplaint("no_parse");
  }

  const registry = await getComplaintRegistry(); // DB or cached

  // ── Exact match ─────────────────────────────
  const direct = registry.find(r => r.ccId === parsed.primary);

  if (direct && direct.enabled) {
    return {
      ccId: direct.ccId,
      source: "primary",
      confidence: parsed.confidence,
    };
  }

  // ── Alias match ─────────────────────────────
  for (const entry of registry) {
    if (entry.aliases.includes(parsed.primary) && entry.enabled) {
      return {
        ccId: entry.ccId,
        source: "alias",
        confidence: "low",
      };
    }
  }

  // ── Fallback ────────────────────────────────
  return fallbackComplaint(parsed.primary);
}

function fallbackComplaint(reason: string): ComplaintResolution {
  return {
    ccId: "general_symptom",
    source: "fallback",
    confidence: "low",
  };
}

A) Where priors usually go wrong
1. Priors are treated like “just config”

A DiagnosisPrior object looks innocent:

  diagnosis: "flu",

    fever: 0.9,
    chills: 0.8

}

But these numbers are not ordinary settings. They are probabilistic assumptions. If they are malformed, stale, contradictory, or copied carelessly, the engine still outputs crisp posteriors with fake confidence.

2. baseProbability is often unvalidated

Typical failure modes:

0
negative
> 1
NaN
all diagnoses with tiny arbitrary values
wildly inconsistent values across complaints

If a prior is 0, log(0) detonates.
If all priors are invalid, your posterior layer becomes a probability graveyard.

👉 Priors must be treated like validated numeric model data, not freeform JSON.

3. featureLikelihoods often mix semantics

This is one of the biggest hidden problems.

What does:

featureLikelihoods["fever"] = 0.8

actually mean?

Possible interpretations:

probability of fever given diagnosis
frequency estimate from chart data
expert confidence
normalized weight
pseudo-likelihood made by hand

If the source semantics are mixed, then the engine is multiplying apples, oranges, and an occasional stapler.

👉 Every likelihood table needs a declared meaning.

4. Missing-feature handling can hide data quality problems

From the earlier Bayesian review, the engine uses a floor likelihood for missing features. That is okay as a runtime fallback, but if a diagnosis is missing 80 percent of its feature table, the engine may still “work” while actually being mostly driven by smoothing.

That means:

garbage-in is quietly softened
incomplete diagnosis models remain in production
missing data masquerades as low evidence
5. No complaint-specific validation boundary

A diagnosis prior for cough may accidentally leak into sore_throat if loading is loose or keyed incorrectly. This can happen through:

bad CSV row joins
wrong ccId
reused differential tables
stale cache after complaint edits

That kind of cross-complaint contamination is a stealth bug. The numbers are valid, just valid for the wrong complaint.

which version loaded
when it loaded
where it came from
who approved it
whether it was partially stale

function validateDiagnosisPrior(prior: DiagnosisPrior, ccId: string): void {
  if (!prior.diagnosis?.trim()) {
    throw new Error(`[Priors] ${ccId}: missing diagnosis name`);
  }

  if (!isFiniteProbability(prior.baseProbability)) {
    throw new Error(
      `[Priors] ${ccId}:${prior.diagnosis} invalid baseProbability=${prior.baseProbability}`
    );
  }

  if (!prior.featureLikelihoods || typeof prior.featureLikelihoods !== "object") {
    throw new Error(`[Priors] ${ccId}:${prior.diagnosis} missing featureLikelihoods`);
  }

  const featureNames = Object.keys(prior.featureLikelihoods);

  if (featureNames.length === 0) {
    throw new Error(`[Priors] ${ccId}:${prior.diagnosis} has no modeled features`);
  }

  for (const feature of featureNames) {
    const value = prior.featureLikelihoods[feature];

    if (!normalizeFeatureName(feature)) {
      throw new Error(`[Priors] ${ccId}:${prior.diagnosis} contains blank feature name`);

function validatePriorBundle(bundle: PriorBundle): void {
  if (!bundle.ccId?.trim()) {
    throw new Error("[Priors] Missing ccId");
  }

  if (!Number.isFinite(bundle.version) || bundle.version <= 0) {
    throw new Error(`[Priors] ${bundle.ccId}: invalid version ${bundle.version}`);
  }

  if (!bundle.loadedAt?.trim()) {
    throw new Error(`[Priors] ${bundle.ccId}: missing loadedAt`);
  }

  if (!Array.isArray(bundle.priors) || bundle.priors.length === 0) {
    throw new Error(`[Priors] ${bundle.ccId}: no priors loaded`);
  }

  const seenDiagnoses = new Set<string>();

  for (const prior of bundle.priors) {
    validateDiagnosisPrior(prior, bundle.ccId);

    const key = prior.diagnosis.trim().toLowerCase();
    if (seenDiagnoses.has(key)) {
      throw new Error(`[Priors] ${bundle.ccId}: duplicate diagnosis "${prior.diagnosis}"`);
    }
    seenDiagnoses.add(key);
  }

  // Optional coverage warning, not hard failure
  const allFeatures = new Set<string>();
  for (const prior of bundle.priors) {
    Object.keys(prior.featureLikelihoods).forEach(f => allFeatures.add(normalizeFeatureName(f)));
  }

  if (allFeatures.size < 3) {
    console.warn(`[Priors] ${bundle.ccId}: very low feature coverage (${allFeatures.size} total features)`);
  }
}

function assemblePriorBundle(ccId: string, rows: RawPriorRow[]): PriorBundle {
  const wrongComplaint = rows.find(r => r.ccId !== ccId);
  if (wrongComplaint) {
    throw new Error(
      `[Priors] Complaint contamination: expected ${ccId}, found row for ${wrongComplaint.ccId}`
    );
  }

  const byDiagnosis = new Map<string, DiagnosisPrior>();

  for (const row of rows) {
    if (!byDiagnosis.has(row.diagnosis)) {
      byDiagnosis.set(row.diagnosis, {
        diagnosis: row.diagnosis,
        baseProbability: row.baseProbability,
        featureLikelihoods: {},
        version: row.version,
        tableName: "diagnosis_priors",
      });
    }

    byDiagnosis.get(row.diagnosis)!.featureLikelihoods[row.feature] = row.likelihood;
  }

  const bundle: PriorBundle = {
    ccId,
    version: rows[0]?.version ?? 1,
    loadedAt: new Date().toISOString(),
    source: "KB_DB",
    priors: Array.from(byDiagnosis.values()),
  };

  validatePriorBundle(bundle);
  return bundle;
}

2. Compute entropy + margin
function computeEntropy(differential: DifferentialResult[]): number {
  return -differential.reduce((sum, d) => {
    return sum + (d.posterior > 0 ? d.posterior * Math.log(d.posterior) : 0);
  }, 0);
}

function analyzePosterior(differential: DifferentialResult[]): PosteriorAnalysis {
  if (!differential.length) {
    throw new Error("Empty differential");
  }

  const sorted = [...differential].sort((a, b) => b.posterior - a.posterior);

  const top = sorted[0];
  const second = sorted[1];

  const entropy = computeEntropy(sorted);
  const margin = second ? top.posterior - second.posterior : 1;

  const isUncertain =
    top.posterior < 0.6 ||
    margin < 0.15 ||
    entropy > 1.2;

  return {
    topDiagnosis: top.diagnosis,
    topPosterior: top.posterior,
    differential: sorted,
    entropy,
    margin,
    isUncertain,
  };
}

// server/clinical/finalDecisionEngine.ts

export async function computeFinalDecision(
  input: {
    state: CaseState;
    posterior: PosteriorAnalysis;
    erProbability: number;
    store: EscalationStore;
  }
): Promise<DecisionContext> {

  const ctx: DecisionContext = {
    caseId: input.state.caseId,
    patientId: input.state.patientId,
    posterior: input.posterior,
    initialDisposition: "",
    riskOverrideApplied: false,
    safety: { allowed: false, code: "", reason: "" },
    trace: [],
  };

  // ── Step 1: Initial disposition ───────────────────────
  ctx.initialDisposition = deriveDisposition(input.posterior, input.state);
  ctx.trace.push(`Initial disposition: ${ctx.initialDisposition}`);

  // ── Step 2: Risk override ─────────────────────────────
  if (applyRiskOverride(input.posterior)) {
    ctx.initialDisposition = "ER_NOW";
    ctx.riskOverrideApplied = true;
    ctx.trace.push("Risk override applied → ER_NOW");
  }

  // ── Step 3: Safety gate ───────────────────────────────
  const safety = await clinicalSafetyGate({
    riskScore: input.erProbability,
    patientId: input.state.patientId,
    action: "disposition",
  });

  ctx.safety = {
    allowed: safety.allowed,
    code: safety.code,
    reason: safety.reason,
  };

  ctx.trace.push(`Safety gate: ${safety.code}`);

  if (!safety.allowed) {
    ctx.finalDisposition = "BLOCKED";
    return ctx; // FAIL CLOSED
  }

  // ── Step 4: Physician checkpoint ──────────────────────
  if (DISPOSITIONS_REQUIRING_APPROVAL.includes(ctx.initialDisposition as any)) {

    const checkpoint = await createOrCheckCheckpoint(ctx);

    ctx.checkpoint = checkpoint;
    ctx.trace.push(`Checkpoint: ${checkpoint.status}`);

    if (checkpoint.status === "pending") {
      ctx.finalDisposition = "AWAITING_PHYSICIAN";
      return ctx;
    }

    if (checkpoint.status === "rejected") {
      ctx.finalDisposition = "REJECTED_BY_PHYSICIAN";
      return ctx;
    }

    if (checkpoint.status === "expired") {
      ctx.finalDisposition = "ESCALATED_NO_RESPONSE";
      return ctx;
    }
  }

  // ── Step 5: Escalation adjustment ─────────────────────
  const advisory = await escalationControl(input.store);

  let finalProb = input.erProbability;

  if (advisory.adjust) {
    finalProb = Math.max(0, Math.min(1, finalProb + advisory.probabilityDelta));
    ctx.trace.push(`Escalation adjustment applied (${advisory.reason})`);
  }

  // ── Step 6: Final disposition ─────────────────────────
  ctx.finalDisposition =
    finalProb >= 0.5 ? "ER_NOW" : ctx.initialDisposition;

  ctx.trace.push(`Final disposition: ${ctx.finalDisposition}`);

  // ── Step 7: Record outcome ────────────────────────────
  await recordDisposition(ctx.finalDisposition, input.store);

  return ctx;
}

export interface SystemRunResult {

  parsed?: ParsedComplaint;
  resolvedComplaint?: string;

  nodeTrace: NodeExecutionResult[];

  posterior?: PosteriorAnalysis;

  decision?: DecisionContext;

  patientResponse?: string;

export async function runSystemTestCase(
  test: SystemTestCase
): Promise<SystemRunResult> {

  const errors: string[] = [];

  const caseId = `test-${test.id}-${Date.now()}`;

  let state: CaseState = {
    caseId,
    symptoms: [],
    ...test.input.patientContext,
  };

  let parsed: ParsedComplaint | undefined;
  let config: ComplaintConfig | undefined;
  let posterior: PosteriorAnalysis | undefined;
  let decision: DecisionContext | undefined;
  let nodeTrace: NodeExecutionResult[] = [];

  try {
    // ── Parse ─────────────────────────────
    parsed = parseComplaint(test.input.message);

    // ── Resolve complaint ─────────────────
    const resolved = await resolveComplaint(parsed);
    config = await loadComplaintConfig(resolved.ccId);

    // ── Run nodes ─────────────────────────
    const nodeResult = await runComplaintNodes(config, state);
    nodeTrace = nodeResult.nodeResults;

    if (!nodeResult.success) {
      errors.push(`Node failure at ${nodeResult.failedNode}`);
      return { caseId, parsed, nodeTrace, errors };
    }

    state = nodeResult.finalState;

    // ── Load priors ───────────────────────
    const priorsBundle = await loadComplaintPriors(config.ccId);

    // ── Bayesian differential ─────────────
    const differential = runDifferential(
      state.symptoms,
      priorsBundle.priors
    );

    posterior = analyzePosterior(differential);

    // ── Final decision ────────────────────
    decision = await computeFinalDecision({
      state,
      posterior,
      erProbability: state.scores?.erRisk ?? 0,
      store: getTestEscalationStore(),
    });

    // ── Patient output ────────────────────
    const response = buildPatientResponse(decision);

    return {
      caseId,
      parsed,
      resolvedComplaint: config.ccId,
      nodeTrace,
      posterior,
      decision,
      patientResponse: response,
      errors,
    };

  } catch (err) {
    errors.push(err instanceof Error ? err.message : String(err));

    return {
      caseId,
      parsed,
      resolvedComplaint: config?.ccId,
      nodeTrace,
      posterior,

C) Validator (this is where truth is enforced)
// server/testing/validator.ts

export interface ValidationResult {
  passed: boolean;
  failures: string[];
}

export function validateTestCase(
  test: SystemTestCase,
  result: SystemRunResult
): ValidationResult {

  const failures: string[] = [];

  const decision = result.decision;

  if (!decision) {
    failures.push("No decision produced");
    return { passed: false, failures };
  }

  // ── Disposition ───────────────────────
  if (test.expected.disposition) {
    if (decision.finalDisposition !== test.expected.disposition) {
      failures.push(
        `Expected disposition ${test.expected.disposition}, got ${decision.finalDisposition}`
      );
    }
  }

  // ── Diagnosis ─────────────────────────
  if (test.expected.primaryDiagnosis) {
    if (decision.posterior.topDiagnosis !== test.expected.primaryDiagnosis) {
      failures.push(
        `Expected diagnosis ${test.expected.primaryDiagnosis}, got ${decision.posterior.topDiagnosis}`
      );
    }
  }

  // ── Must include differential ─────────
  for (const d of test.expected.mustIncludeDifferential ?? []) {
    if (!decision.posterior.differential.some(x => x.diagnosis === d)) {
      failures.push(`Missing expected differential: ${d}`);
    }
  }

  // ── Must not include ──────────────────
  for (const d of test.expected.mustNotIncludeDifferential ?? []) {
    if (decision.posterior.differential.some(x => x.diagnosis === d)) {
      failures.push(`Unexpected differential present: ${d}`);
    }
  }

  // ── Safety gate ───────────────────────
  if (test.expected.mustTriggerSafetyGate) {
    if (decision.safety.allowed) {
      failures.push("Expected safety gate to block, but it passed");
    }
  }

  // ── Physician review ──────────────────
  if (test.expected.requiresPhysicianReview) {
    if (decision.finalDisposition !== "AWAITING_PHYSICIAN") {
      failures.push("Expected physician review but did not occur");
    }
  }

  return {
    passed: failures.length === 0,
    failures,
  };
}

// server/learning/learningSignals.ts

export interface LearningSignal {
  caseId: string;
  failureType:
    | "wrong_disposition"
    | "missed_diagnosis"
    | "unsafe_pass"
    | "over_escalation"
    | "uncertainty_misclassification";

  expected: any;
  actual: any;

  context: {
    complaint: string;
    symptoms: string[];
    posterior?: any;
    decision?: any;
  };
}

export function extractLearningSignals(
  test: SystemTestCase,
  result: SystemRunResult,
  validation: ValidationResult
): LearningSignal[] {

  const signals: LearningSignal[] = [];

  for (const failure of validation.failures) {

    if (failure.includes("disposition")) {
      signals.push({
        caseId: test.id,
        failureType: "wrong_disposition",
        expected: test.expected.disposition,
        actual: result.decision?.finalDisposition,
        context: {
          complaint: result.resolvedComplaint!,
          symptoms: result.parsed?.secondary ?? [],
          posterior: result.posterior,
          decision: result.decision,
        },
      });
    }

    if (failure.includes("diagnosis")) {
      signals.push({
        caseId: test.id,
        failureType: "missed_diagnosis",
        expected: test.expected.primaryDiagnosis,
        actual: result.decision?.posterior.topDiagnosis,
        context: {
          complaint: result.resolvedComplaint!,
          symptoms: result.parsed?.secondary ?? [],
        },
      });
    }
  }

  return signals;
}

        },
        reason: "Too many ER escalations",

function compareResults(before, after) {

  let improved = 0;
  let worsened = 0;

  for (let i = 0; i < before.length; i++) {
    if (!before[i].passed && after[i].passed) improved++;
    if (before[i].passed && !after[i].passed) worsened++;
  }

  return {
    improved,
    worsened,
    net: improved - worsened,
  };
}

export async function createGoldenCase(params: {
  title: string;
  complaintId: string;
  input: GoldenCaseInput;
  expected: GoldenCaseExpected;
  metadata?: GoldenCaseMetadata;
  createdBy: string;
}): Promise<GoldenCaseRecord> {
  const id = crypto.randomUUID();
  const now = new Date().toISOString();

  await db.insert(goldenCases).values({
    id,
    title: params.title,
    complaintId: params.complaintId,
    inputJson: params.input,
    expectedJson: params.expected,
    metadataJson: params.metadata ?? null,
    version: 1,
    status: "active",
    createdBy: params.createdBy,
    createdAt: now,
    updatedBy: params.createdBy,
    updatedAt: now,
    lastApprovedExpectedAt: now,
    lastApprovedExpectedBy: params.createdBy,
  });

  await auditStep({
    traceId: `golden-case-${id}`,
    step: "golden_case_created",
    input: null,
    output: { id, title: params.title, complaintId: params.complaintId },
    metadata: { createdBy: params.createdBy },

  const rows = await db.select().from(goldenCases).where(eq(goldenCases.id, id)).limit(1);
  return mapGoldenCaseRow(rows[0]);
}

function buildDiff(expected: GoldenCaseExpected, result: any, failures: string[]): GoldenCaseDiff {
  return {
    failures,
    expected,
    actual: {
      disposition: result?.decision?.finalDisposition,
      finalDisposition: result?.decision?.finalDisposition,
      primaryDiagnosis: result?.decision?.posterior?.topDiagnosis,
      differential: result?.decision?.posterior?.differential?.map((d: any) => d.diagnosis) ?? [],

export async function runGoldenCase(
  goldenCaseId: string,
  suiteRunId: string
): Promise<GoldenCaseRunResult> {
  const goldenCase = await getGoldenCase(goldenCaseId);
  if (!goldenCase) {
    throw new Error(`Golden case not found: ${goldenCaseId}`);
  }

  const testCase = {
    id: goldenCase.id,
    input: goldenCase.input,
    expected: goldenCase.expected,
    metadata: goldenCase.metadata,
  };

  const systemRun = await runSystemTestCase(testCase);
  const validation = validateTestCase(testCase as any, systemRun as any);

  const status = systemRun.errors.length > 0
    ? "error"
    : validation.passed
    ? "passed"
    : "failed";

  const diff = buildDiff(goldenCase.expected, systemRun, validation.failures);

  const runRow = await db
    .insert(goldenCaseRuns)
    .values({
      goldenCaseId,
      suiteRunId,
      status,
      actualJson: diff.actual,
      diffJson: diff,
      traceJson: systemRun,
      modelVersion: "current",
      configVersion: "current",
      kbVersion: "current",
      runAt: new Date().toISOString(),
    })
    .onConflictDoNothing({
      target: [goldenCaseRuns.goldenCaseId, goldenCaseRuns.suiteRunId],
    });

  await auditStep({
    traceId: `golden-case-${goldenCaseId}-run-${suiteRunId}`,
    step: "golden_case_run_completed",
    input: { goldenCaseId, suiteRunId },
    output: { status },
    metadata: { failures: validation.failures },
  });

  return {
    goldenCaseId,
    suiteRunId,
    status,
    diff,
    trace: systemRun,
  };
}

export async function runAllGoldenCases(): Promise<GoldenCaseRunResult[]> {

  return results;
}

This is where golden cases become governed instead of slippery.

Use this when the current system behavior is actually better than the old expected result, and a physician decides to bless that as the new truth.

// server/goldenCases/goldenCaseReviewService.ts
import { eq } from "drizzle-orm";
import { db } from "../db";
import { auditStep } from "../audit/auditLogger";
import {
  goldenCases,
  goldenCaseRuns,
  goldenCaseReviews,
} from "../../shared/schema";

export async function acceptCurrentRunAsExpected(params: {
  goldenCaseId: string;
  goldenCaseRunId: number;
  reviewerId: string;
  note?: string;
}) {
  const caseRows = await db
    .select()
    .from(goldenCases)
    .where(eq(goldenCases.id, params.goldenCaseId))
    .limit(1);

  if (!caseRows.length) {
    throw new Error(`Golden case not found: ${params.goldenCaseId}`);
  }

  const runRows = await db
    .select()
    .from(goldenCaseRuns)
    .where(eq(goldenCaseRuns.id, params.goldenCaseRunId))
    .limit(1);

  if (!runRows.length) {
    throw new Error(`Golden case run not found: ${params.goldenCaseRunId}`);
  }

  const goldenCase = caseRows[0];
  const run = runRows[0];
  const previousExpected = goldenCase.expectedJson;
  const diff = run.diffJson as any;

  const newExpected = {
    ...previousExpected,
    disposition: diff.actual?.finalDisposition ?? previousExpected.disposition,
    primaryDiagnosis: diff.actual?.primaryDiagnosis ?? previousExpected.primaryDiagnosis,
    mustIncludeDifferential: diff.actual?.differential ?? previousExpected.mustIncludeDifferential,
    mustTriggerSafetyGate:
      diff.actual?.safetyAllowed === false
        ? true
        : previousExpected.mustTriggerSafetyGate,
  };

  await db.transaction(async (tx) => {
    await tx.insert(goldenCaseReviews).values({
      goldenCaseId: params.goldenCaseId,
      goldenCaseRunId: params.goldenCaseRunId,
      reviewerId: params.reviewerId,
      decision: "accept_current_as_expected",
      note: params.note ?? null,
      previousExpectedJson: previousExpected,
      newExpectedJson: newExpected,
      reviewedAt: new Date().toISOString(),
    });

    await tx
      .update(goldenCases)
      .set({
        expectedJson: newExpected,
        version: goldenCase.version + 1,
        updatedBy: params.reviewerId,
        updatedAt: new Date().toISOString(),
        lastApprovedExpectedAt: new Date().toISOString(),
        lastApprovedExpectedBy: params.reviewerId,
      })
      .where(eq(goldenCases.id, params.goldenCaseId));
  });

  await auditStep({
    traceId: `golden-case-review-${params.goldenCaseId}`,
    step: "golden_case_expected_updated",
    input: { goldenCaseId: params.goldenCaseId, goldenCaseRunId: params.goldenCaseRunId },
    output: { reviewerId: params.reviewerId },
    metadata: { note: params.note ?? null },
  });

  return { ok: true };
}

export async function rejectCurrentRunKeepExpected(params: {
  goldenCaseId: string;

B) Rate limiter (protect against abuse + runaway loops)
// server/guardrails/rateLimiter.ts

const userRequestCounts = new Map<string, { count: number; ts: number }>();

const WINDOW_MS = 60_000;
const MAX_REQUESTS = 30;

export function rateLimitGuard() {
  return (req, res, next) => {

    const user = req.user?.sub || "anon";
    const now = Date.now();

    const entry = userRequestCounts.get(user) ?? { count: 0, ts: now };

    if (now - entry.ts > WINDOW_MS) {
      entry.count = 0;
      entry.ts = now;
    }

    entry.count++;

    userRequestCounts.set(user, entry);

    if (entry.count > MAX_REQUESTS) {
      return res.status(429).json({
        error: "Too many requests",
      });
    }

    next();
  };
}

// server/guardrails/circuitBreaker.ts

interface CircuitState {
  failures: number;
  openUntil?: number;
}

const circuits = new Map<string, CircuitState>();

export async function withCircuitBreaker<T>(
  name: string,
  fn: () => Promise<T>
): Promise<T> {

  const state = circuits.get(name) ?? { failures: 0 };

  if (state.openUntil && Date.now() < state.openUntil) {
    throw new Error(`[CircuitOpen] ${name}`);
  }

  try {
    const result = await fn();
    state.failures = 0;
    circuits.set(name, state);
    return result;

  } catch (err) {
    state.failures++;

    if (state.failures >= 5) {
      state.openUntil = Date.now() + 60_000; // 1 min cooldown
    }

    circuits.set(name, state);
    throw err;
  }
}

// server/routes/traceRoutes.ts
import { Router } from "express";
import { requireAuth, requirePermission } from "../auth/authzMiddleware";
import { getTraceByCaseId, getTraceByGoldenRunId, replayTrace } from "../trace/traceService";

const router = Router();

router.get(
  "/case/:caseId",
  requireAuth,
  requirePermission("audit:view"),
  async (req, res) => {
    const trace = await getTraceByCaseId(req.params.caseId);
    if (!trace) return res.status(404).json({ error: "Trace not found" });
    res.json(trace);
  }
);

router.get(
  "/run/:runId",
  requireAuth,
  requirePermission("audit:view"),
  async (req, res) => {
    const trace = await getTraceByGoldenRunId(Number(req.params.runId));
    if (!trace) return res.status(404).json({ error: "Trace not found" });
    res.json(trace);
  }
);

router.post(
  "/replay",
  requireAuth,
  requirePermission("audit:view"),
  async (req, res) => {
    const result = await replayTrace({
      caseId: req.body?.caseId,
      goldenCaseRunId: req.body?.goldenCaseRunId,
      actorId: req.user?.sub,
    });
    res.json(result);
  }
);

export default router;

export async function getTraceByGoldenRunId(runId: number) {
  const rows = await db
    .select()
    .from(goldenCaseRuns)
    .where(eq(goldenCaseRuns.id, runId))
    .limit(1);

export async function replayTrace(params: {
  caseId?: string;
  goldenCaseRunId?: number;
  actorId?: string;
}) {
  if (params.goldenCaseRunId) {
    const rows = await db
      .select()
      .from(goldenCaseRuns)
      .where(eq(goldenCaseRuns.id, params.goldenCaseRunId))
      .limit(1);

    if (!rows.length) {
      throw new Error(`Golden run not found: ${params.goldenCaseRunId}`);
    }

    const trace = rows[0].traceJson as any;
    const input = trace?.parsed?.raw
      ? { message: trace.parsed.raw, patientContext: {} }
      : trace?.input;

    if (!input?.message) {
      throw new Error("Replay unavailable: missing original input message");
    }

    const rerun = await runSystemTestCase({
      id: crypto.randomUUID(),
      input,
      expected: {},
      metadata: { category: "regression", description: "Replay" },
    });

    return {
      ok: true,
      replayed: true,
      actorId: params.actorId,
      trace: rerun,
    };
  }

  if (params.caseId) {
    const live = await loadLiveCaseTrace(params.caseId);
    if (!live?.parsed?.raw) {
      throw new Error("Replay unavailable: missing original live-case message");
    }

    const rerun = await runSystemTestCase({
      id: crypto.randomUUID(),
      input: {
        message: live.parsed.raw,
        patientContext: {},
      },
      expected: {},
      metadata: { category: "regression", description: "Replay" },
    });

    return {
      ok: true,
      replayed: true,
      actorId: params.actorId,
      trace: rerun,
    };
  }

  throw new Error("Replay requires caseId or goldenCaseRunId");
}

// server/monitoring/alertService.ts
import { and, desc, eq, isNull, or, lt } from "drizzle-orm";
import { db } from "../db";
import { auditStep } from "../audit/auditLogger";
import { operationalAlerts, AlertLevel, AlertStatus } from "../../shared/schema";

function nowIso() {
  return new Date().toISOString();
}

export async function createAlert(params: {
  source: string;
  code: string;
  level: AlertLevel;
  message: string;
  details?: Record<string, unknown>;
}) {
  const rows = await db
    .insert(operationalAlerts)
    .values({
      source: params.source,
      code: params.code,
      level: params.level,
      status: "active",
      message: params.message,
      detailsJson: params.details ?? null,
      createdAt: nowIso(),
    })
    .returning();

  const created = rows[0];

  await auditStep({
    traceId: `alert-${created.id}`,
    step: "alert_created",
    input: null,
    output: {
      id: created.id,
      source: created.source,
      code: created.code,
      level: created.level,
    },
    metadata: {
      message: created.message,
    },
  });

  return created;
}

export async function getVisibleAlerts(limit = 100) {
  const now = new Date();

  return db
    .select()
    .from(operationalAlerts)
    .where(
      or(
        eq(operationalAlerts.status, "active"),
        eq(operationalAlerts.status, "acknowledged"),
        and(
          eq(operationalAlerts.status, "muted"),
          lt(operationalAlerts.mutedUntil, now)
        )
      )
    )
    .orderBy(desc(operationalAlerts.createdAt))
    .limit(limit);
}

export async function getAllAlerts(limit = 200) {
  return db
    .select()
    .from(operationalAlerts)
    .orderBy(desc(operationalAlerts.createdAt))
    .limit(limit);
}

export async function acknowledgeAlert(params: {
  alertId: number;
  actorId: string;
  note?: string;
}) {
  const rows = await db
    .update(operationalAlerts)
    .set({
      status: "acknowledged",
      acknowledgedAt: nowIso(),
      acknowledgedBy: params.actorId,
      acknowledgmentNote: params.note ?? null,
    })
    .where(eq(operationalAlerts.id, params.alertId))
    .returning();

  if (!rows.length) {
    throw new Error(`Alert not found: ${params.alertId}`);
  }

  await auditStep({
    traceId: `alert-${params.alertId}`,
    step: "alert_acknowledged",
    input: { alertId: params.alertId },
    output: { actorId: params.actorId },
    metadata: { note: params.note ?? null },
  });

  return rows[0];
}

export async function muteAlert(params: {
  alertId: number;
  actorId: string;
  muteMinutes: number;
  note?: string;
}) {
  const mutedUntil = new Date(Date.now() + params.muteMinutes * 60_000).toISOString();

  const rows = await db
    .update(operationalAlerts)
    .set({
      status: "muted",
      mutedUntil,
      mutedBy: params.actorId,
      muteNote: params.note ?? null,
    })
    .where(eq(operationalAlerts.id, params.alertId))
    .returning();

  if (!rows.length) {
    throw new Error(`Alert not found: ${params.alertId}`);
  }

  await auditStep({
    traceId: `alert-${params.alertId}`,
    step: "alert_muted",
    input: { alertId: params.alertId },
    output: { actorId: params.actorId, mutedUntil },
    metadata: {
      muteMinutes: params.muteMinutes,
      note: params.note ?? null,
    },
  });

  return rows[0];
}

export async function resolveAlert(params: {
  alertId: number;
  actorId: string;
  note?: string;
}) {
  const rows = await db
    .update(operationalAlerts)
    .set({
      status: "resolved",
      resolvedAt: nowIso(),
      resolvedBy: params.actorId,
      resolutionNote: params.note ?? null,
    })
    .where(eq(operationalAlerts.id, params.alertId))
    .returning();

  if (!rows.length) {
    throw new Error(`Alert not found: ${params.alertId}`);

    step: "alert_resolved",
    input: { alertId: params.alertId },
    output: { actorId: params.actorId },
    metadata: { note: params.note ?? null },
  });

  return rows[0];

// server/trace/traceDiff.ts
type DiffEntry = {
  field: string;
  left: unknown;
  right: unknown;
};

function pushIfDifferent(diffs: DiffEntry[], field: string, left: unknown, right: unknown) {
  const l = JSON.stringify(left ?? null);
  const r = JSON.stringify(right ?? null);
  if (l !== r) {
    diffs.push({ field, left, right });
  }
}

export function diffTraces(left: any, right: any) {
  const diffs: DiffEntry[] = [];

  pushIfDifferent(diffs, "parsed.primary", left?.parsed?.primary, right?.parsed?.primary);
  pushIfDifferent(diffs, "resolvedComplaint", left?.resolvedComplaint, right?.resolvedComplaint);

  pushIfDifferent(diffs, "posterior.topDiagnosis", left?.posterior?.topDiagnosis, right?.posterior?.topDiagnosis);
  pushIfDifferent(diffs, "posterior.topPosterior", left?.posterior?.topPosterior, right?.posterior?.topPosterior);
  pushIfDifferent(diffs, "posterior.entropy", left?.posterior?.entropy, right?.posterior?.entropy);
  pushIfDifferent(diffs, "posterior.margin", left?.posterior?.margin, right?.posterior?.margin);

  pushIfDifferent(diffs, "decision.initialDisposition", left?.decision?.initialDisposition, right?.decision?.initialDisposition);
  pushIfDifferent(diffs, "decision.finalDisposition", left?.decision?.finalDisposition, right?.decision?.finalDisposition);
  pushIfDifferent(diffs, "decision.safety.code", left?.decision?.safety?.code, right?.decision?.safety?.code);
  pushIfDifferent(diffs, "decision.safety.allowed", left?.decision?.safety?.allowed, right?.decision?.safety?.allowed);

  pushIfDifferent(
    diffs,
    "posterior.differential",
    left?.posterior?.differential?.map((d: any) => [d.diagnosis, d.posterior]),
    right?.posterior?.differential?.map((d: any) => [d.diagnosis, d.posterior])
  );

  pushIfDifferent(
    diffs,
    "nodeTrace",
    left?.nodeTrace?.map((n: any) => [n.nodeId, n.success, n.error ?? null]),
    right?.nodeTrace?.map((n: any) => [n.nodeId, n.success, n.error ?? null])
  );

  pushIfDifferent(diffs, "patientResponse", left?.patientResponse, right?.patientResponse);
  pushIfDifferent(diffs, "errors", left?.errors ?? [], right?.errors ?? []);

  return {
    diffCount: diffs.length,
    diffs,
  };
}

// server/monitoring/alertService.ts
import { and, desc, eq, gt, isNull, or } from "drizzle-orm";
import { db } from "../db";
import { auditStep } from "../audit/auditLogger";
import {
  operationalAlerts,
  operationalAlertEvents,
  AlertLevel,
} from "../../shared/schema";

function now() {
  return new Date();
}

function nowIso() {
  return now().toISOString();
}

export function buildAlertDedupKey(params: {
  source: string;
  code: string;
  complaintId?: string;
  subsystem?: string;
}) {
  return [
    params.source,
    params.code,
    params.complaintId ?? "all-complaints",
    params.subsystem ?? "all-subsystems",
  ].join(":");
}

export async function createOrBumpAlert(params: {
  source: string;
  code: string;
  level: AlertLevel;
  message: string;
  details?: Record<string, unknown>;
  dedupKey?: string;
}) {
  const dedupKey =
    params.dedupKey ??
    buildAlertDedupKey({
      source: params.source,
      code: params.code,
    });

  const existing = await db
    .select()
    .from(operationalAlerts)
    .where(
      and(
        eq(operationalAlerts.dedupKey, dedupKey),
        or(
          eq(operationalAlerts.status, "active"),
          eq(operationalAlerts.status, "acknowledged"),
          eq(operationalAlerts.status, "muted")
        )
      )
    )
    .limit(1);

  if (existing.length) {
    const current = existing[0];
    const updated = await db
      .update(operationalAlerts)
      .set({
        lastSeenAt: nowIso(),
        count: current.count + 1,
        message: params.message,
        detailsJson: params.details ?? current.detailsJson ?? null,
        // If muted and mute expired, bring it back to active
        status:
          current.status === "muted" &&
          current.mutedUntil &&
          new Date(current.mutedUntil).getTime() < Date.now()
            ? "active"
            : current.status,
      })
      .where(eq(operationalAlerts.id, current.id))
      .returning();

    await db.insert(operationalAlertEvents).values({
      alertId: current.id,
      eventType: "dedup_incremented",
      actorId: null,
      note: null,
      payloadJson: { count: current.count + 1 },
      createdAt: nowIso(),
    });

    return updated[0];
  }

  const created = await db
    .insert(operationalAlerts)
    .values({
      source: params.source,
      code: params.code,
      dedupKey,
      level: params.level,
      status: "active",
      message: params.message,
      detailsJson: params.details ?? null,
      count: 1,
      firstSeenAt: nowIso(),
      lastSeenAt: nowIso(),
      createdAt: nowIso(),
    })
    .returning();

  const alert = created[0];

  await db.insert(operationalAlertEvents).values({
    alertId: alert.id,
    eventType: "created",
    actorId: null,
    note: null,
    payloadJson: params.details ?? {},
    createdAt: nowIso(),
  });

  await auditStep({
    traceId: `alert-${alert.id}`,
    step: "alert_created",
    input: null,
    output: {
      id: alert.id,
      source: alert.source,
      code: alert.code,
      level: alert.level,
    },
    metadata: {
      message: alert.message,
      dedupKey,
    },
  });

  return alert;
}

export async function getVisibleAlerts(limit = 100) {
  return db
    .select()
    .from(operationalAlerts)
    .where(
      or(
        eq(operationalAlerts.status, "active"),
        eq(operationalAlerts.status, "acknowledged"),
        and(
          eq(operationalAlerts.status, "muted"),
          gt(operationalAlerts.mutedUntil, now())
        )
      )
    )
    .orderBy(desc(operationalAlerts.lastSeenAt))
    .limit(limit);
}

export async function acknowledgeAlert(params: {
  alertId: number;
  actorId: string;
  note?: string;
}) {
  const rows = await db
    .update(operationalAlerts)
    .set({
      status: "acknowledged",
      acknowledgedAt: nowIso(),
      acknowledgedBy: params.actorId,
      acknowledgmentNote: params.note ?? null,
    })
    .where(eq(operationalAlerts.id, params.alertId))
    .returning();

  if (!rows.length) throw new Error(`Alert not found: ${params.alertId}`);

  await db.insert(operationalAlertEvents).values({
    alertId: params.alertId,
    eventType: "acknowledged",
    actorId: params.actorId,
    note: params.note ?? null,
    payloadJson: {},
    createdAt: nowIso(),
  });

  await auditStep({
    traceId: `alert-${params.alertId}`,
    step: "alert_acknowledged",
    input: { alertId: params.alertId },
    output: { actorId: params.actorId },
    metadata: { note: params.note ?? null },
  });

  return rows[0];
}

export async function muteAlert(params: {
  alertId: number;
  actorId: string;
  muteMinutes: number;
  note?: string;
}) {
  const mutedUntil = new Date(Date.now() + params.muteMinutes * 60_000).toISOString();

  const rows = await db
    .update(operationalAlerts)
    .set({
      status: "muted",
      mutedUntil,
      mutedBy: params.actorId,
      muteNote: params.note ?? null,
    })
    .where(eq(operationalAlerts.id, params.alertId))
    .returning();

  if (!rows.length) throw new Error(`Alert not found: ${params.alertId}`);

  await db.insert(operationalAlertEvents).values({
    alertId: params.alertId,
    eventType: "muted",
    actorId: params.actorId,
    note: params.note ?? null,
    payloadJson: { muteMinutes: params.muteMinutes, mutedUntil },

    traceId: `alert-${params.alertId}`,
    step: "alert_muted",

      kbVersion: params.kbVersion ?? null,
      modelVersion: params.modelVersion ?? null,
      priorsVersion: params.priorsVersion ?? null,
      safetyConfigVersion: params.safetyConfigVersion ?? null,
      createdAt: new Date().toISOString(),
    })
    .onConflictDoUpdate({
      target: traceSnapshots.traceId,
      set: {
        inputJson: params.input ?? null,
        parsedJson: params.parsed ?? null,

        kbVersion: params.kbVersion ?? null,
        modelVersion: params.modelVersion ?? null,
        priorsVersion: params.priorsVersion ?? null,
        safetyConfigVersion: params.safetyConfigVersion ?? null,
      },
    })
    .returning();

  return rows[0];
}

  caseId?: string;
  goldenCaseRunId?: number;
  actorId?: string;
  mode?: "current" | "frozen";
}) {
  const mode = params.mode ?? "current";

  let snapshot: any | undefined;
  if (params.caseId) {
    snapshot = await getTraceByCaseId(params.caseId);
  } else if (params.goldenCaseRunId) {
    snapshot = await getTraceByGoldenRunId(params.goldenCaseRunId);
  }

    throw new Error("Replay unavailable: missing original input");
  }

  // current mode reruns with current config
  // frozen mode is scaffolded here for future version-pinned loader support
  const rerun = await runSystemTestCase({
    id: crypto.randomUUID(),
    input: {
      message: snapshot.parsed.raw,
      patientContext: {},
    },
    expected: {},
    metadata: {
      category: "regression",
      description: mode === "frozen" ? "Frozen replay" : "Replay",
    },
  });

    replayed: true,

// server/trace/traceDiff.ts
type DiffEntry = {
  field: string;
  left: unknown;
  right: unknown;
};

function pushIfDifferent(diffs: DiffEntry[], field: string, left: unknown, right: unknown) {
  if (JSON.stringify(left ?? null) !== JSON.stringify(right ?? null)) {
    diffs.push({ field, left, right });
  }
}

export function diffTraces(left: any, right: any) {
  const diffs: DiffEntry[] = [];

  pushIfDifferent(diffs, "parsed.primary", left?.parsed?.primary, right?.parsed?.primary);
  pushIfDifferent(diffs, "resolvedComplaint", left?.resolvedComplaint, right?.resolvedComplaint);

  pushIfDifferent(diffs, "posterior.topDiagnosis", left?.posterior?.topDiagnosis, right?.posterior?.topDiagnosis);
  pushIfDifferent(diffs, "posterior.topPosterior", left?.posterior?.topPosterior, right?.posterior?.topPosterior);
  pushIfDifferent(diffs, "posterior.entropy", left?.posterior?.entropy, right?.posterior?.entropy);
  pushIfDifferent(diffs, "posterior.margin", left?.posterior?.margin, right?.posterior?.margin);

  pushIfDifferent(diffs, "decision.initialDisposition", left?.decision?.initialDisposition, right?.decision?.initialDisposition);
  pushIfDifferent(diffs, "decision.finalDisposition", left?.decision?.finalDisposition, right?.decision?.finalDisposition);
  pushIfDifferent(diffs, "decision.safety.code", left?.decision?.safety?.code, right?.decision?.safety?.code);
  pushIfDifferent(diffs, "decision.safety.allowed", left?.decision?.safety?.allowed, right?.decision?.safety?.allowed);

  pushIfDifferent(
    diffs,
    "posterior.differential",
    left?.posterior?.differential?.map((d: any) => [d.diagnosis, d.posterior]),
    right?.posterior?.differential?.map((d: any) => [d.diagnosis, d.posterior])
  );

  pushIfDifferent(
    diffs,
    "nodeTrace",
    left?.nodeTrace?.map((n: any) => [n.nodeId, n.success, n.error ?? null]),
    right?.nodeTrace?.map((n: any) => [n.nodeId, n.success, n.error ?? null])
  );

  pushIfDifferent(diffs, "patientResponse", left?.patientResponse, right?.patientResponse);
  pushIfDifferent(diffs, "errors", left?.errors ?? [], right?.errors ?? []);

  return {
    diffCount: diffs.length,
    diffs,
  };
}

// server/testing/runSystemTestCaseFrozen.ts

import crypto from "crypto";
import type { SystemTestCase } from "./types";
import type { SystemRunResult } from "./systemRunner";
import { parseComplaint } from "../chat/parseComplaint";
import { resolveComplaint } from "../clinical/complaintResolver";
import { runComplaintNodes } from "../clinical/complaintNodeRunner";
import { runDifferential } from "../clinical/bayesianDifferential";
import { analyzePosterior } from "../clinical/posteriorAnalysis";
import { buildPatientResponse } from "../clinical/outputBuilder";
import { getTestEscalationStore } from "./testStores";
import { computeFinalDecisionWithInjectedSafetyConfig } from "../clinical/finalDecisionEngineInjected";
import {
  loadComplaintConfigVersion,
  loadComplaintPriorsVersion,
  loadSafetyConfigVersion,
} from "../versioning/versionedLoaders";

export interface FrozenReplayVersions {
  configVersion: string;
  kbVersion?: string;
  modelVersion?: string;
  priorsVersion: string;
  safetyConfigVersion: string;
}

export async function runSystemTestCaseFrozen(
  test: SystemTestCase,
  versions: FrozenReplayVersions
): Promise<SystemRunResult> {
  const errors: string[] = [];
  const caseId = `frozen-${crypto.randomUUID()}`;

  let state: any = {
    caseId,
    symptoms: [],
    ...test.input.patientContext,
  };

  let parsed: any;
  let config: any;
  let posterior: any;
  let decision: any;
  let nodeTrace: any[] = [];

  try {
    parsed = parseComplaint(test.input.message);

    const resolved = await resolveComplaint(parsed);

    // 🔒 Load historical versions, not current
    config = await loadComplaintConfigVersion(resolved.ccId, versions.configVersion);

    const nodeResult = await runComplaintNodes(config, state);
    nodeTrace = nodeResult.nodeResults;

    if (!nodeResult.success) {
      errors.push(`Node failure at ${nodeResult.failedNode}`);
      return {
        caseId,
        parsed,
        resolvedComplaint: resolved.ccId,
        nodeTrace,
        errors,
      };
    }

    state = nodeResult.finalState;

    const priorBundle = await loadComplaintPriorsVersion(
      resolved.ccId,
      versions.priorsVersion
    );

    const differential = runDifferential(state.symptoms, priorBundle.priors);
    posterior = analyzePosterior(differential);

    const safetyConfig = await loadSafetyConfigVersion(
      versions.safetyConfigVersion
    );

    decision = await computeFinalDecisionWithInjectedSafetyConfig({
      state,
      posterior,
      erProbability: state.scores?.erRisk ?? 0,
      store: getTestEscalationStore(),
      safetyConfig,
    });

    const response = buildPatientResponse(decision);

    return {
      caseId,
      parsed,
      resolvedComplaint: resolved.ccId,
      nodeTrace,
      posterior,
      decision,
      patientResponse: response,
      errors,
    };
  } catch (err) {
    errors.push(err instanceof Error ? err.message : String(err));

    return {
      caseId,
      parsed,
      resolvedComplaint: config?.ccId,
      nodeTrace,
      posterior,
      decision,
      errors,
    };
  }
}
C. Final decision engine with injected safety config

This keeps frozen replay from accidentally calling the current active safety config.

// server/clinical/finalDecisionEngineInjected.ts

import { clinicalSafetyGate } from "./safetyGate";
import { escalationControl, recordDisposition } from "./escalationGuard";
import type { PersistedSafetyGateConfig } from "./safetyConfigService";
import { deriveDisposition } from "./dispositionDerivation";
import { applyRiskOverride } from "./posteriorRiskOverride";
import { DISPOSITIONS_REQUIRING_APPROVAL, createOrCheckCheckpoint } from "../compliance/physicianCheckpoint";

export interface DecisionContext {
  caseId: string;
  patientId?: string;
  posterior: any;
  initialDisposition: string;
  riskOverrideApplied: boolean;
  safety: {
    allowed: boolean;
    code: string;
    reason: string;
  };
  checkpoint?: {
    required: boolean;
    status: "pending" | "approved" | "rejected" | "expired";
  };
  finalDisposition?: string;
  trace: string[];
}

function clamp01(value: number) {
  return Math.max(0, Math.min(1, value));
}

export async function computeFinalDecisionWithInjectedSafetyConfig(input: {
  state: any;
  posterior: any;
  erProbability: number;
  store: any;
  safetyConfig: PersistedSafetyGateConfig;
}): Promise<DecisionContext> {
  const ctx: DecisionContext = {
    caseId: input.state.caseId,
    patientId: input.state.patientId,
    posterior: input.posterior,
    initialDisposition: "",
    riskOverrideApplied: false,
    safety: { allowed: false, code: "", reason: "" },
    trace: [],
  };

  ctx.initialDisposition = deriveDisposition(input.posterior, input.state);
  ctx.trace.push(`Initial disposition: ${ctx.initialDisposition}`);

  if (applyRiskOverride(input.posterior)) {
    ctx.initialDisposition = "ER_NOW";
    ctx.riskOverrideApplied = true;
    ctx.trace.push("Risk override applied → ER_NOW");
  }

  const safety = await clinicalSafetyGate(
    {
      riskScore: input.erProbability,
      patientId: input.state.patientId,
      action: "disposition",
    },
    input.safetyConfig
  );

  ctx.safety = {
    allowed: safety.allowed,
    code: safety.code,
    reason: safety.reason,
  };

  ctx.trace.push(`Safety gate: ${safety.code}`);

  if (!safety.allowed) {
    ctx.finalDisposition = "BLOCKED";
    return ctx;
  }

  if (DISPOSITIONS_REQUIRING_APPROVAL.includes(ctx.initialDisposition as any)) {
    const checkpoint = await createOrCheckCheckpoint(ctx);
    ctx.checkpoint = checkpoint;
    ctx.trace.push(`Checkpoint: ${checkpoint.status}`);

    if (checkpoint.status === "pending") {
      ctx.finalDisposition = "AWAITING_PHYSICIAN";
      return ctx;
    }
    if (checkpoint.status === "rejected") {
      ctx.finalDisposition = "REJECTED_BY_PHYSICIAN";
      return ctx;
    }
    if (checkpoint.status === "expired") {
      ctx.finalDisposition = "ESCALATED_NO_RESPONSE";
      return ctx;
    }
  }

  const advisory = await escalationControl(input.store);

  let finalProb = input.erProbability;
  if (advisory.adjust) {
    finalProb = clamp01(finalProb + advisory.probabilityDelta);
    ctx.trace.push(`Escalation adjustment applied (${advisory.reason})`);
  }

  ctx.finalDisposition = finalProb >= 0.5 ? "ER_NOW" : ctx.initialDisposition;
  ctx.trace.push(`Final disposition: ${ctx.finalDisposition}`);

  await recordDisposition(ctx.finalDisposition, input.store);

  return ctx;
}

// server/trace/traceService.ts

import crypto from "crypto";
import { runSystemTestCase } from "../testing/systemRunner";
import { runSystemTestCaseFrozen } from "../testing/runSystemTestCaseFrozen";
import {
  getTraceSnapshotByCaseId,
  getTraceSnapshotByRunId,
  persistTraceSnapshot,
} from "./traceSnapshotService";

export async function replayTrace(params: {
  caseId?: string;
  goldenCaseRunId?: number;
  actorId?: string;
  mode?: "current" | "frozen";
}) {
  const mode = params.mode ?? "current";

  let snapshot: any | undefined;
  if (params.caseId) {
    snapshot = await getTraceSnapshotByCaseId(params.caseId);
  } else if (params.goldenCaseRunId) {
    snapshot = await getTraceSnapshotByRunId(params.goldenCaseRunId);
  }

  if (!snapshot?.parsedJson?.raw) {
    throw new Error("Replay unavailable: missing original input");
  }

  const testCase = {
    id: crypto.randomUUID(),
    input: {
      message: snapshot.parsedJson.raw,
      patientContext: snapshot.inputJson?.patientContext ?? {},
    },
    expected: {},
    metadata: {
      category: "regression" as const,
      description: mode === "frozen" ? "Frozen replay" : "Replay",
    },
  };

  const rerun =
    mode === "frozen"
      ? await runSystemTestCaseFrozen(testCase, {
          configVersion: snapshot.configVersion,
          kbVersion: snapshot.kbVersion ?? undefined,
          modelVersion: snapshot.modelVersion ?? undefined,
          priorsVersion: snapshot.priorsVersion,
          safetyConfigVersion: snapshot.safetyConfigVersion,
        })
      : await runSystemTestCase(testCase);

  const replayTraceId = `replay-${crypto.randomUUID()}`;

  await persistTraceSnapshot({
    traceId: replayTraceId,
    source: "replay",
    caseId: params.caseId,
    goldenCaseRunId: params.goldenCaseRunId,
    input: testCase.input,
    parsed: rerun.parsed,
    nodeTrace: rerun.nodeTrace,
    posterior: rerun.posterior,
    decision: rerun.decision,
    patientResponse: rerun.patientResponse,
    errors: rerun.errors,
    configVersion: snapshot.configVersion,
    kbVersion: snapshot.kbVersion ?? undefined,
    modelVersion: snapshot.modelVersion ?? undefined,
    priorsVersion: snapshot.priorsVersion,
    safetyConfigVersion: snapshot.safetyConfigVersion,
  });

  return {
    ok: true,
    replayed: true,
    mode,
    actorId: params.actorId,
    traceId: replayTraceId,
    trace: rerun,
  };
}

// server/guardrails/circuitBreaker.ts

export interface CircuitState {
  failures: number;
  openUntil?: number;
  lastError?: string;
  lastFailureAt?: number;
}

const circuits = new Map<string, CircuitState>();

export function getCircuitSnapshot() {
  return Array.from(circuits.entries()).map(([name, state]) => ({
    name,
    failures: state.failures,
    open: Boolean(state.openUntil && Date.now() < state.openUntil),
    openUntil: state.openUntil ? new Date(state.openUntil).toISOString() : null,
    lastError: state.lastError ?? null,
    lastFailureAt: state.lastFailureAt
      ? new Date(state.lastFailureAt).toISOString()
      : null,
  }));
}

export function resetCircuit(name: string) {
  circuits.set(name, { failures: 0 });
}

export async function withCircuitBreaker<T>(
  name: string,
  fn: () => Promise<T>
): Promise<T> {
  const state = circuits.get(name) ?? { failures: 0 };

  if (state.openUntil && Date.now() < state.openUntil) {
    throw new Error(`[CircuitOpen] ${name}`);
  }

  try {
    const result = await fn();
    state.failures = 0;
    state.openUntil = undefined;
    state.lastError = undefined;
    state.lastFailureAt = undefined;
    circuits.set(name, state);
    return result;
  } catch (err) {
    state.failures++;
    state.lastError = err instanceof Error ? err.message : String(err);
    state.lastFailureAt = Date.now();

    if (state.failures >= 5) {
      state.openUntil = Date.now() + 60_000;
    }

    circuits.set(name, state);
    throw err;
  }
}

  const queryClient = useQueryClient();

  const { data } = useQuery<CircuitItem[]>({
    queryKey: ["/api/circuits"],
    queryFn: () => apiRequest("GET", "/api/circuits"),
    refetchInterval: 3000,
  });

    mutationFn: (name: string) => apiRequest("POST", `/api/circuits/${encodeURIComponent(name)}/reset`, {}),

      toast({ title: "Circuit reset" });

      toast({ title: "Failed to reset circuit", description: err.message, variant: "destructive" });
    },
  });

    <div className="space-y-2 max-h-72 overflow-y-auto">
      {(data ?? []).map((c) => (
        <div key={c.name} className="rounded border p-3 text-sm space-y-2">
          <div className="flex items-center justify-between">
            <span className="font-medium">{c.name}</span>
            <Badge variant={c.open ? "destructive" : "outline"}>
              {c.open ? "OPEN" : "CLOSED"}
            </Badge>
          </div>
          <div className="text-xs text-muted-foreground">
            Failures: {c.failures}
          </div>
          {c.openUntil && (
            <div className="text-xs text-muted-foreground">
              Open until: {c.openUntil}
            </div>

              {c.lastError}

        </div>
      ))}
