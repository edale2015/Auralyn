🧠 Graph Engine (Neo4j-Compatible)
// server/graph/graphStore.ts

import neo4j from "neo4j-driver"

const driver = neo4j.driver(
  process.env.NEO4J_URI!,
  neo4j.auth.basic(
    process.env.NEO4J_USER!,
    process.env.NEO4J_PASSWORD!
  )
)

export async function createNode(type: string, name: string) {
  const session = driver.session()
  await session.run(
    `MERGE (n:${type} {name: $name}) RETURN n`,
    { name }
  )
  await session.close()
}

export async function createRelation(
  from: string,
  to: string,
  relation: string
) {
  const session = driver.session()
  await session.run(
    `
    MATCH (a {name: $from}), (b {name: $to})
    MERGE (a)-[:${relation}]->(b)
    `,
    { from, to }
  )
  await session.close()
}
🔍 Query Example (Clinical Reasoning)
// server/graph/queries.ts

export async function getRelatedDiseases(symptom: string) {
  const session = driver.session()

  const result = await session.run(
    `
    MATCH (s:symptom {name: $symptom})-[:INDICATES]->(d:disease)
    RETURN d.name as disease
    `,
    { symptom }
  )

  return result.records.map(r => r.get("disease"))
}

// server/api/dag.ts

export function getDAG() {
  return {
    nodes: agents.map(a => ({
      id: a.meta.name
    })),
    edges: agents.flatMap(a =>
      a.meta.consumes.map(c => ({
        from: c,
        to: a.meta.name
      }))
    )
  }
}
🎨 Frontend (React Flow)
// client/components/DAGViewer.tsx

import ReactFlow from "reactflow"

export function DAGViewer({ dag }) {
  const nodes = dag.nodes.map(n => ({
    id: n.id,
    data: { label: n.id },
    position: { x: Math.random()*400, y: Math.random()*400 }
  }))

  const edges = dag.edges.map((e, i) => ({
    id: String(i),
    source: e.from,
    target: e.to
  }))

  return <ReactFlow nodes={nodes} edges={edges} />
}
