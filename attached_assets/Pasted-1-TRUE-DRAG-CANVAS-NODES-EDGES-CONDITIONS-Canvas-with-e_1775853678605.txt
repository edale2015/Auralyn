import ReactFlow, {
  addEdge, Background, Controls,
  useNodesState, useEdgesState
} from "reactflow";
import "reactflow/dist/style.css";

const initialNodes = [
  { id: "start", position: { x: 50, y: 50 }, data: { label: "Start" } },
];

export default function WorkflowCanvasFull() {
  const [nodes, setNodes, onNodesChange] = useNodesState(initialNodes);
  const [edges, setEdges, onEdgesChange] = useEdgesState([]);

  function onConnect(params: any) {
    setEdges((eds) => addEdge(params, eds));
  }

  async function save() {
    await fetch("/api/workflows/save", {
      method: "POST",
      headers: {"Content-Type":"application/json"},
      body: JSON.stringify({ nodes, edges })
    });
    alert("Saved");
  }

  return (
    <div style={{ height: 600 }}>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onConnect={onConnect}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
      >
        <Background />
        <Controls />
      </ReactFlow>
      <button onClick={save}>💾 Save</button>
    </div>
  );
}
🧠 Backend: convert edges → runnable graph
export function edgesToGraph(nodes: any[], edges: any[]) {
  const map: any = {};
  nodes.forEach(n => map[n.id] = { ...n, next: [] });
  edges.forEach(e => map[e.source].next.push(e.target));
  return map;
}

import { useState } from "react";

export default function AlertRules() {
  const [expr, setExpr] = useState("latency > 2000");
  const [target, setTarget] = useState("slack");

  async function save() {
    await fetch("/api/alerts/rules", {
      method: "POST",
      headers: {"Content-Type":"application/json"},
      body: JSON.stringify({ expr, target })
    });
    alert("Saved rule");
  }

  return (
    <div>
      <h2>🚨 Alert Rules</h2>
      <input value={expr} onChange={e=>setExpr(e.target.value)} />
      <select onChange={e=>setTarget(e.target.value)}>
        <option value="slack">Slack</option>
        <option value="whatsapp">WhatsApp</option>
      </select>
      <button onClick={save}>Save</button>
    </div>
  );
}

const rules: any[] = [];

export function addRule(rule: any) { rules.push(rule); }

export function evalRules(metrics: any) {
  rules.forEach(r => {
    if (eval(r.expr)) {
      if (r.target === "slack") sendSlackAlert("Rule fired");
      if (r.target === "whatsapp") sendWhatsAppAlert("Rule fired");

  });
}

export async function runGolden(cases: any[]) {
  return Promise.all(cases.map(async c => {
    const r = await runFinalPipeline(c.input, "golden");
    return {
      expected: c.expected,
      actual: r.safetyResult.finalDisposition,
      match: c.expected === r.safetyResult.finalDisposition
    };
  }));
}
