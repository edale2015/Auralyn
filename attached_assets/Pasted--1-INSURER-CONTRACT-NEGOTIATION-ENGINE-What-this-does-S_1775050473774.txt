import { db } from "../db";
import { contracts, claims, outcomes } from "../db/schema";

export type NegotiationStrategy =
  | "anchor_high"
  | "value_based"
  | "bundled_rate"
  | "risk_share"
  | "standard";

export interface ContractScore {
  payerId: string;
  avgReimbursement: number;
  denialRate: number;
  outcomeScore: number;
  score: number;
  recommendedStrategy: NegotiationStrategy;
}

export async function scoreContract(payerId: string): Promise<ContractScore> {
  const claimData = await db.select().from(claims).where({ payerId });
  const outcomeData = await db.select().from(outcomes).where({ payerId });

  const avgReimbursement =
    claimData.reduce((sum, c) => sum + c.amountPaid, 0) / (claimData.length || 1);

  const denialRate =
    claimData.filter(c => c.status === "denied").length / (claimData.length || 1);

  const outcomeScore =
    outcomeData.reduce((sum, o) => sum + o.qualityScore, 0) /
    (outcomeData.length || 1);

  const score =
    avgReimbursement * 0.4 +
    (1 - denialRate) * 0.3 +
    outcomeScore * 0.3;

  let strategy: NegotiationStrategy = "standard";

  if (score > 0.8) strategy = "anchor_high";
  else if (outcomeScore > 0.85) strategy = "value_based";
  else if (denialRate > 0.2) strategy = "bundled_rate";
  else if (score < 0.5) strategy = "risk_share";

  return {
    payerId,
    avgReimbursement,
    denialRate,
    outcomeScore,
    score,
    recommendedStrategy: strategy,
  };
}

export interface QualityMetric {
  name: string;
  numerator: number;
  denominator: number;
  rate: number;
}

export async function computeHEDISMetrics() {
  const data = await db.select().from(encounters);

  const total = data.length;

  const followUpCompleted = data.filter(
    e => e.followUpCompleted === true
  ).length;

  const antibioticAppropriate = data.filter(
    e => e.antibioticAppropriate === true
  ).length;

  const metrics: QualityMetric[] = [
    {
      name: "Follow-up Completion Rate",
      numerator: followUpCompleted,
      denominator: total,
      rate: followUpCompleted / (total || 1),
    },
    {
      name: "Appropriate Antibiotic Use",
      numerator: antibioticAppropriate,
      denominator: total,
      rate: antibioticAppropriate / (total || 1),
    },
  ];

  return metrics;
}

export interface ClinicianPerformance {
  clinicianId: string;
  totalCases: number;
  avgDecisionTime: number;
  escalationRate: number;
  accuracyScore: number;
  riskScore: number;
}

export async function computeClinicianPerformance(clinicianId: string) {
  const data = await db
    .select()
    .from(encounters)
    .where({ clinicianId });

  const totalCases = data.length;

  const avgDecisionTime =
    data.reduce((sum, e) => sum + e.decisionTimeMs, 0) /
    (totalCases || 1);

  const escalationRate =
    data.filter(e => e.escalated === true).length /
    (totalCases || 1);

  const accuracyScore =
    data.reduce((sum, e) => sum + e.correctDiagnosis, 0) /
    (totalCases || 1);

  const riskScore =
    escalationRate * 0.4 + (1 - accuracyScore) * 0.6;

  return {
    clinicianId,
    totalCases,
    avgDecisionTime,
    escalationRate,
    accuracyScore,
    riskScore,
  };
}

import { db } from "../db";
import { claims } from "../db/schema";

interface Prediction {
  probability: number;
  riskLevel: "low" | "medium" | "high";
  reasons: string[];
}

export async function predictDenial(payerId: string, cptCode: string): Promise<Prediction> {
  const history = await db.select().from(claims).where({ payerId, cptCode });

  const denialRate =
    history.filter(c => c.status === "denied").length / (history.length || 1);

  const reasons: string[] = [];

  if (denialRate > 0.3) reasons.push("High historical denial rate");
  if (cptCode.startsWith("9")) reasons.push("High-risk CPT family");
  if (history.length < 5) reasons.push("Low data confidence");

  let riskLevel: Prediction["riskLevel"] = "low";
  if (denialRate > 0.3) riskLevel = "high";
  else if (denialRate > 0.15) riskLevel = "medium";

  return {
    probability: denialRate,
    riskLevel,
    reasons,
  };
}

import { db } from "../db";
import { contracts } from "../db/schema";
import { predictDenial } from "../insurer/denialPredictor";

export async function optimizeReimbursement(payerId: string, cptOptions: string[]) {
  let bestOption = null;
  let bestScore = -Infinity;

  for (const cpt of cptOptions) {
    const contract = await db.select().from(contracts).where({ payerId, cptCode: cpt }).limit(1);

    if (!contract.length) continue;

    const rate = Number(contract[0].rate);
    const denial = await predictDenial(payerId, cpt);

    const expectedValue = rate * (1 - denial.probability);

    if (expectedValue > bestScore) {
      bestScore = expectedValue;
      bestOption = {
        cpt,
        expectedValue,
        denialRisk: denial,
      };
    }
  }

  return bestOption;
}

export async function computeOutcomeWeightedRevenue() {
  const data = await db.select().from(encounters);

  let totalRevenue = 0;
  let weightedRevenue = 0;

  for (const e of data) {
    const revenue = Number(e.reimbursement || 0);

    totalRevenue += revenue;
    weightedRevenue += revenue * outcome;
  }

  const avgOutcome = data.length
    ? data.reduce((sum, e) => sum + Number(e.outcomeScore || 0), 0) / data.length
    : 0;

  return {
    totalRevenue,
    weightedRevenue,
    avgOutcome,
    efficiencyScore: weightedRevenue / (totalRevenue || 1),
  };
}

📁 server/learning/rlhfEngine.ts
import { db } from "../db";
import { encounters } from "../db/schema";

interface ModelWeights {
  diagnosisWeight: number;
  escalationPenalty: number;
  outcomeWeight: number;
}

let weights: ModelWeights = {
  diagnosisWeight: 1,
  escalationPenalty: 1,
  outcomeWeight: 1,
};

export async function runRLHFUpdate() {
  const data = await db.select().from(encounters);

  let correct = 0;
  let escalations = 0;
  let outcomeSum = 0;

  for (const e of data) {
    correct += Number(e.correctDiagnosis || 0);
    escalations += e.escalated ? 1 : 0;
    outcomeSum += Number(e.outcomeScore || 0);
  }

  const total = data.length || 1;

  const accuracy = correct / total;
  const escalationRate = escalations / total;
  const avgOutcome = outcomeSum / total;

  // RLHF-style adjustment
  weights.diagnosisWeight *= 1 + (accuracy - 0.8);
  weights.escalationPenalty *= 1 + (escalationRate - 0.2);
  weights.outcomeWeight *= 1 + (avgOutcome - 0.75);

  return {
    updatedWeights: weights,
    metrics: { accuracy, escalationRate, avgOutcome },
  };
}

import { db } from "../db";
import { encounters } from "../db/schema";

export async function predictFailureRisk(agent: string): Promise<number> {
  const data = await db.select().from(encounters);

  const total = data.length || 1;

  const errorRate =
    data.filter(e => Number(e.correctDiagnosis) < 0.7).length / total;

  const escalationRate =
    data.filter(e => e.escalated).length / total;

  const latency =
    data.reduce((sum, e) => sum + Number(e.decisionTimeMs || 0), 0) / total;

  // Normalize latency (simple heuristic)
  const latencyScore = Math.min(latency / 5000, 1);

  const risk =
    errorRate * 0.4 +
    escalationRate * 0.3 +
    latencyScore * 0.3;

  return Math.min(risk, 1);
}

interface AgentOpinion {
  agent: string;
  diagnosis: string;
  confidence: number;
}

export function runDebate(opinions: AgentOpinion[]) {
  const votes: Record<string, number> = {};

  for (const op of opinions) {
    votes[op.diagnosis] = (votes[op.diagnosis] || 0) + op.confidence;
  }

  const sorted = Object.entries(votes).sort((a, b) => b[1] - a[1]);

  return {
    finalDiagnosis: sorted[0][0],
    score: sorted[0][1],
    allVotes: votes,
  };
}

import { EvolutionCandidate } from "./types";
import { simulateCandidate } from "./simulator";

import { governCandidate } from "./governor";
import { deployCandidate } from "./deployer";

export async function runEvolutionPipeline(candidate: EvolutionCandidate) {
  const simResult = await simulateCandidate(candidate);

  const score = await scoreCandidate(candidate, simResult);
  candidate.score = score;

  const decision = await governCandidate(candidate);

  if (decision === "approve") {
    await deployCandidate(candidate);
    candidate.status = "approved";
  } else {
    candidate.status = "rejected";
  }

  return candidate;
}

  if (candidate.type === "table") {

    if (config.risk === "high" && candidate.score! < 0.9) {

  const data = await db.select().from(encounters);

  const total = data.length || 1;

  const goodOutcomes =
    data.filter(e => Number(e.outcomeScore) > 0.8).length / total;

  const poorOutcomes =
    data.filter(e => Number(e.outcomeScore) < 0.5).length / total;

  const readmissions =
    data.filter(e => e.escalated === true).length / total;

  return {
    goodOutcomes,
    poorOutcomes,
    readmissions,
  };
}

export async function getFinancialSignals() {
  const data = await db.select().from(claims);

  const total = data.length || 1;

  const denialRate =
    data.filter(c => c.status === "denied").length / total;

  const avgReimbursement =
    data.reduce((sum, c) => sum + Number(c.amountPaid), 0) / total;

  return {
    denialRate,
    avgReimbursement,
  };
}

export async function fetchFHIR(resource: string, id: string) {
  const res = await fetch(`${FHIR_BASE}/${resource}/${id}`, {

export function normalizeEncounter(fhir: any) {
  return {
    patientId: fhir.subject?.reference,
    diagnosis: fhir.reasonCode?.[0]?.text,
    timestamp: fhir.period?.start,
  };
}

export function normalizeObservation(obs: any) {
  return {
    type: obs.code?.text,
    value: obs.valueQuantity?.value,
    unit: obs.valueQuantity?.unit,
    timestamp: obs.effectiveDateTime,
  };
}

import { db } from "../db";

export async function buildPatientState(patientId: string) {
  const encounters = await db.select().from("encounters").where({ patientId });
  const devices = await db.select().from("device_data").where({ patientId });

  const latestVitals = devices.slice(-5);

  const avgSpO2 =
    latestVitals
      .filter(d => d.type === "spo2")
      .reduce((sum, d) => sum + d.value, 0) /
    (latestVitals.length || 1);

  const escalationHistory =
    encounters.filter(e => e.escalated).length / (encounters.length || 1);

  return {
    patientId,
    avgSpO2,
    escalationHistory,
    recentDiagnosis: encounters.slice(-1)[0]?.diagnosis,
  };
}
🔮 2. OUTCOME PREDICTOR (LIGHTWEIGHT MODEL)
📁 server/prediction/outcomePredictor.ts
export function predictOutcome(state: any, treatment: string) {
  let base = 0.7;

  if (state.avgSpO2 < 94) base -= 0.2;
  if (state.escalationHistory > 0.3) base -= 0.2;

  if (treatment === "antibiotic") base += 0.1;
  if (treatment === "supportive") base += 0.05;

  return Math.max(0, Math.min(1, base));
}

📁 server/treatment/treatmentOptimizer.ts
import { predictOutcome } from "../prediction/outcomePredictor";

export function optimizeTreatment(state: any, options: string[]) {
  let best = null;
  let bestScore = -Infinity;

  for (const option of options) {
    const outcome = predictOutcome(state, option);

    const safetyPenalty = state.avgSpO2 < 92 && option !== "admit" ? -0.3 : 0;

    const score = outcome + safetyPenalty;

    if (score > bestScore) {
      bestScore = score;
      best = {
        treatment: option,
        score,
        predictedOutcome: outcome,
      };
    }
  }

  return best;
}

      recommendation: "Seek immediate care",

export async function withTimeout<T>(
  fn: () => Promise<T>,
  ms: number,
  fallback: T
): Promise<T> {
  return new Promise((resolve) => {
    const t = setTimeout(() => resolve(fallback), ms);

    fn()
      .then(r => {
        clearTimeout(t);
        resolve(r);
      })
      .catch(() => {
        clearTimeout(t);
        resolve(fallback);
      });
  });
}

import jwt from "jsonwebtoken";

export function requireAuth(req: any, res: any, next: any) {
  const token = req.headers.authorization?.split(" ")[1];

  if (!token) return res.status(401).json({ error: "Unauthorized" });

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET!);
    req.user = decoded;
    next();
  } catch {
    return res.status(401).json({ error: "Invalid token" });
  }
}

import { adapterRegistry } from "./adapterRegistry";

export async function runAdapterCycle(type: string, context: any) {
  const adapter = adapterRegistry[type as keyof typeof adapterRegistry];

  if (!adapter) throw new Error("Unknown adapter");

  const candidate = await adapter.generate(context);

  const sim = await adapter.simulate(candidate, context);

  const result = await adapter.score(sim);

  if (result.safe && result.score > 0.7) {
    await adapter.deploy(candidate, context);
    return { status: "deployed", candidate, result };
  }

  return { status: "rejected", candidate, result };
}

import fs from "fs";
import path from "path";

const DIR = path.join(__dirname, "../../data/artifacts");
if (!fs.existsSync(DIR)) fs.mkdirSync(DIR, { recursive: true });

export interface Artifact {
  id: string;
  type: string;          // skill | table | medication | policy
  target: string;        // CORE_QUESTIONS, etc.
  version: number;
  payload: any;
  createdAt: string;
  parentVersion?: number;
}

function file(type: string, target: string) {
  return path.join(DIR, `${type}_${target}.json`);
}

export function listArtifacts(type: string, target: string): Artifact[] {
  const f = file(type, target);
  if (!fs.existsSync(f)) return [];
  return JSON.parse(fs.readFileSync(f, "utf-8"));
}

export function saveArtifact(a: Omit<Artifact, "version" | "createdAt">) {
  const existing = listArtifacts(a.type, a.target);
  const version = (existing[existing.length - 1]?.version || 0) + 1;

  const artifact: Artifact = {
    ...a,
    version,
    createdAt: new Date().toISOString(),
  };

  existing.push(artifact);
  fs.writeFileSync(file(a.type, a.target), JSON.stringify(existing, null, 2));
  return artifact;
}

export function rollbackArtifact(type: string, target: string, version: number) {
  const all = listArtifacts(type, target);
  const found = all.find(v => v.version === version);
  if (!found) throw new Error("Version not found");

  // append a rollback copy as latest
  return saveArtifact({
    id: `${type}_${target}_${Date.now()}`,
    type,
    target,
    payload: found.payload,
    parentVersion: found.version,
  });
}

export interface EvalCase {
  input: any;
  expected: string; // or "blocked"
}

export async function evaluatePair(
  cases: EvalCase[],
  applyCandidate: (on: boolean) => void
) {
  // baseline
  applyCandidate(false);
  const base = await runSuite(cases);

  // candidate
  applyCandidate(true);
  const cand = await runSuite(cases);

  return {
    base,
    cand,
    delta: {
      accuracy: cand.accuracy - base.accuracy,
      safety: cand.safety - base.safety,
    },
  };
}

async function runSuite(cases: EvalCase[]) {
  let correct = 0, safe = 0;

  for (const c of cases) {
    const r = await runFullClinicalFlow({
      patientId: "eval",
      complaint: c.input.complaint,
      data: c.input.data || {},
    });

    const ok =
      (c.expected === "blocked" && r.status === "blocked") ||
      r.diagnosis === c.expected;

    if (ok) correct++;
    if (r.status !== "blocked") safe++; // crude; replace with real safety flags
  }

  return {
    accuracy: correct / cases.length,
    safety: safe / cases.length,
  };
}

const API_BASE = "/api/control";

async function readJson<T>(res: Response): Promise<T> {
  if (!res.ok) {
    const text = await res.text();
    throw new Error(text || `Request failed: ${res.status}`);
  }
  return res.json() as Promise<T>;
}

export async function fetchArtifacts(
  type: string,
  target: string
): Promise<ArtifactVersion[]> {
  const res = await fetch(`${API_BASE}/artifacts/${encodeURIComponent(type)}/${encodeURIComponent(target)}`);
  return readJson<ArtifactVersion[]>(res);
}

export async function rollbackArtifact(params: {
  type: string;
  target: string;

  const res = await fetch(`${API_BASE}/rollback`, {

import { useEffect, useMemo, useState } from "react";
import { fetchArtifacts, rollbackArtifact } from "../../api/controlPlane";
import type { ArtifactVersion } from "../../types/control";
import SectionCard from "./SectionCard";

const TYPES = ["skill", "table", "medication", "policy"];
const DEFAULT_TARGETS: Record<string, string[]> = {
  skill: ["diagnosis"],
  table: ["CORE_QUESTIONS", "DX_CANDIDATES", "MODIFIER_EFFECTS"],
  medication: ["MEDICATION_RULES"],
  policy: ["GLOBAL_POLICY"],
};

function safeStringify(value: unknown): string {
  try {
    return JSON.stringify(value, null, 2);
  } catch {
    return String(value);
  }
}

export default function ArtifactBrowser() {
  const [type, setType] = useState("table");
  const [target, setTarget] = useState("CORE_QUESTIONS");
  const [versions, setVersions] = useState<ArtifactVersion[]>([]);
  const [selectedVersion, setSelectedVersion] = useState<number | null>(null);
  const [loading, setLoading] = useState(false);
  const [working, setWorking] = useState(false);
  const [error, setError] = useState<string>("");

  async function load() {
    setLoading(true);
    setError("");
    try {
      const data = await fetchArtifacts(type, target);
      setVersions(data);
      setSelectedVersion(data[data.length - 1]?.version ?? null);
    } catch (e: any) {
      setError(e.message || "Failed to load artifacts");
      setVersions([]);
      setSelectedVersion(null);
    } finally {
      setLoading(false);
    }
  }

  useEffect(() => {
    load();
  }, [type, target]);

  const selected = useMemo(
    () => versions.find(v => v.version === selectedVersion) ?? null,
    [versions, selectedVersion]
  );

  const previous = useMemo(() => {
    if (!selected) return null;
    return versions.find(v => v.version === selected.version - 1) ?? null;
  }, [versions, selected]);

  async function onRollback() {
    if (!selected) return;
    const confirmText = `Rollback ${type}:${target} to version ${selected.version}?`;
    if (!window.confirm(confirmText)) return;

    setWorking(true);
    setError("");
    try {
      await rollbackArtifact({
        type,
        target,
        version: selected.version,
      });
      await load();
    } catch (e: any) {
      setError(e.message || "Rollback failed");
    } finally {
      setWorking(false);
    }
  }

  return (
    <SectionCard
      title="Artifacts & Versioning"
      description="Browse version history, inspect payloads, and rollback safely."
      rightSlot={
        <button
          onClick={load}
          className="rounded-xl border px-3 py-2 text-sm font-medium hover:bg-slate-50"
        >
          Refresh
        </button>
      }
    >
      <div className="grid gap-4 lg:grid-cols-[220px_220px_1fr]">
        <div>
          <label className="mb-1 block text-sm font-medium text-slate-700">Type</label>
          <select
            value={type}
            onChange={e => {
              const nextType = e.target.value;
              setType(nextType);
              setTarget(DEFAULT_TARGETS[nextType][0]);
            }}
            className="w-full rounded-xl border px-3 py-2"
          >
            {TYPES.map(t => (
              <option key={t} value={t}>{t}</option>
            ))}
          </select>
        </div>

        <div>
          <label className="mb-1 block text-sm font-medium text-slate-700">Target</label>
          <select
            value={target}
            onChange={e => setTarget(e.target.value)}
            className="w-full rounded-xl border px-3 py-2"
          >
            {DEFAULT_TARGETS[type].map(t => (
              <option key={t} value={t}>{t}</option>
            ))}
          </select>
        </div>

        <div className="flex items-end justify-end">
          <button
            onClick={onRollback}
            disabled={!selected || working}
            className="rounded-xl bg-slate-900 px-4 py-2 text-sm font-medium text-white disabled:opacity-50"
          >
            {working ? "Rolling back..." : "Rollback to Selected Version"}
          </button>
        </div>
      </div>

      {error ? (
        <div className="mt-4 rounded-xl border border-red-200 bg-red-50 px-3 py-2 text-sm text-red-700">
          {error}
        </div>
      ) : null}

      <div className="mt-5 grid gap-4 lg:grid-cols-[280px_1fr_1fr]">
        <div className="rounded-xl border">
          <div className="border-b px-4 py-3 text-sm font-semibold">Versions</div>
          <div className="max-h-[420px] overflow-auto">
            {loading ? (
              <div className="px-4 py-3 text-sm text-slate-500">Loading...</div>
            ) : versions.length === 0 ? (
              <div className="px-4 py-3 text-sm text-slate-500">No versions found.</div>
            ) : (
              versions
                .slice()

                .map(v => (

                ))

          </div>
        </div>

import { useState } from "react";
import { deployBundle } from "../../api/controlPlane";
import SectionCard from "./SectionCard";

const EXAMPLE = {
  name: "cough-improvement-bundle",
  changes: [
    {
      type: "table",
      target: "CORE_QUESTIONS",
      payload: { question: "Do you have shortness of breath?" },
    },
    {
      type: "table",
      target: "DX_CANDIDATES",
      payload: { diagnosis: "viral_uri", weight: 0.72 },
    },
  ],
};

export default function BundleManager() {
  const [jsonText, setJsonText] = useState(JSON.stringify(EXAMPLE, null, 2));
  const [working, setWorking] = useState(false);
  const [result, setResult] = useState<any>(null);
  const [error, setError] = useState("");

  async function onDeploy() {
    setWorking(true);
    setError("");
    setResult(null);

    try {
      const payload = JSON.parse(jsonText);
      const res = await deployBundle(payload);
      setResult(res);
    } catch (e: any) {
      setError(e.message || "Bundle deploy failed");
    } finally {
      setWorking(false);
    }
  }

  return (
    <SectionCard
      title="Change Bundles"
      description="Deploy multi-entity changes atomically. Useful when a complaint update touches questions, diagnoses, and policies together."
    >
      <textarea
        value={jsonText}
        onChange={e => setJsonText(e.target.value)}
        className="min-h-[220px] w-full rounded-xl border p-3 font-mono text-xs"
      />

      <div className="mt-4 flex items-center gap-3">
        <button
          onClick={onDeploy}
          disabled={working}
          className="rounded-xl bg-slate-900 px-4 py-2 text-sm font-medium text-white disabled:opacity-50"
        >
          {working ? "Deploying..." : "Deploy Bundle"}
        </button>
      </div>

      {error ? (
        <div className="mt-4 rounded-xl border border-red-200 bg-red-50 px-3 py-2 text-sm text-red-700">
          {error}
        </div>
      ) : null}

      {result ? (
        <pre className="mt-4 overflow-auto rounded-xl border bg-slate-50 p-3 text-xs">
          {JSON.stringify(result, null, 2)}

      ) : null}
    </SectionCard>
  );
}

📁 server/ws/realtime.ts
import { Server } from "ws";

let clients: any[] = [];

export function initWebSocket(server: any) {
  const wss = new Server({ server });

  wss.on("connection", (ws) => {
    clients.push(ws);

    ws.on("close", () => {
      clients = clients.filter(c => c !== ws);
    });
  });
}

export function broadcast(event: string, data: any) {
  clients.forEach(ws => {
    ws.send(JSON.stringify({ event, data }));
  });
}
Frontend hook
// client/src/hooks/useRealtime.ts
import { useEffect } from "react";

export function useRealtime(onMessage: any) {
  useEffect(() => {
    const ws = new WebSocket("ws://localhost:3000");

    ws.onmessage = (e) => {
      const msg = JSON.parse(e.data);
      onMessage(msg);
    };

    return () => ws.close();
  }, []);
}

export function predictSystemRisk(metrics: any) {
  let risk = 0;

  if (metrics.accuracy < 0.8) risk += 0.3;
  if (metrics.safety < 0.85) risk += 0.4;
  if (metrics.denialRate > 0.2) risk += 0.2;

  return Math.min(risk, 1);
}

  const delta = Math.abs(current.accuracy - previous.accuracy);

  if (delta > 0.15) {
    return {
      anomaly: true,
      reason: "Accuracy drop",

  }

  return { anomaly: false };
}

import { db } from "../db";

export async function aggregateAcrossClinics() {
  const data = await db.select().from("encounters");

  const byClinic: any = {};

  data.forEach((d: any) => {
    if (!byClinic[d.clinicId]) byClinic[d.clinicId] = [];
    byClinic[d.clinicId].push(d);
  });

  return byClinic;
}

1. FAILURE MODES (MOST CRITICAL)
Silent failures — highest danger:
	∙	Confident wrong answers. Multi-agent debate/aggregation will converge on plausible-sounding consensus even when all agents share the same flawed training signal. A genuinely rare but dangerous presentation (atypical MI in a 40-year-old woman, epiglottitis presenting as sore throat) that doesn’t pattern-match to known high-risk triggers passes through clean. The system doesn’t know what it doesn’t know.

Athena’s FHIR layer, for example, has known field-level inconsistencies in how it surfaces medication lists, allergy codes, and problem lists. If your ingestion layer doesn’t validate completeness on every pull, the agent reasons on incomplete data with no flag to the clinician.
	∙	Temporal data corruption. The unified patient timeline assumes clean timestamps. In real EHRs, lab results are timestamped on resulted time, not collection time. Vitals from device integrations often have clock drift. A system reasoning about “trend over last 6 hours” could be reasoning on out-of-order data and silently produce the wrong conclusion.
	∙	Outcome attribution failure. If a patient sees your system at urgent care, then deteriorates and is admitted via the ED, your outcome signal will either miss that event entirely (no data sharing) or attribute it ambiguously. The system may score that encounter as a success because it sees no negative outcome — the patient just never came back.

2. SELF-IMPROVEMENT RISKS
	∙	The shared scoring function is the single most dangerous component. Combining accuracy + safety + outcomes + financial impact into one score creates a gradient the optimizer will exploit. If denial rates or reimbursement signals have any weight, the system will — over time and subtly — drift toward documentation patterns and diagnosis coding that maximizes reimbursement. This is not hypothetical; it is the direction any gradient will travel if financial signals are present. You will not see this in any individual decision. You will see it in aggregate coding patterns over 18 months.
	∙	Feedback loop on clinician confirmation bias. Clinicians confirm outputs they agree with and override outputs they disagree with. But the cases where the system is most dangerously wrong are exactly the cases where a tired clinician at 6pm scans the recommendation and clicks confirm. The learning signal will be: “confirmed = correct.” High-confidence wrong outputs on low-scrutiny cases will be reinforced.
	∙	Skill evolution without clinical semantics. When “skills” (reasoning logic) evolve, the system has no way to know whether a changed reasoning path is better medicine or just better at passing your internal scoring metrics. A skill that learns to always recommend a CBC on chest pain complaints will score well on safety metrics and may improve reimbursement — but it’s not better clinical reasoning.
	∙	Golden case rot. Your continuous validation depends on golden cases remaining representative. If the system trains toward those cases, they cease to be a neutral test. More subtly: medicine changes. A golden case set built in 2024 will be subtly wrong by 2027 as guidelines shift. Without a process for actively retiring and replacing golden cases, validation becomes a formality.
	∙	Table evolution without semantic locks. If medication tables and diagnosis tables evolve, you need explicit constraints preventing clinically nonsensical associations from being reinforced. 	∙	Automation bias is your real safety threat, not autonomous prescribing. 	∙	Highest-risk patient populations your system is most likely to fail on:
	∙	Pediatric patients — weight-based dosing, age-specific normal ranges, and presentations that are categorically different from adults. If your training data skews adult urgent care, pediatric outputs will be overconfident and wrong in ways that are hard to detect.
	∙	Patients with limited English proficiency — complaint capture via intake may lose clinical nuance. “My stomach hurts” entered by a translator may be the patient’s description of chest tightness.
	∙	Patients with multiple comorbidities — your complaint/diagnosis/modifier tables likely encode single-condition logic. Multi-system patients will have outputs that are locally reasonable per complaint but globally incoherent.

export function computeDataCompleteness(input: any) {
  let score = 1;

  if (!input.history) score -= 0.2;
  if (!input.medications) score -= 0.2;
  if (!input.allergies) score -= 0.2;
  if (!input.vitals) score -= 0.2;
  if (!input.duration) score -= 0.2;

  return Math.max(0, score);
}

export async function runAdapterCycle(type: string, context: any) {
  const adapter = adapterRegistry[type];

  const candidate = await adapter.generate(context);
  const sim = await adapter.simulate(candidate, context);

  const domainScores = scoreByDomain(sim, context.real || {});

  const domainsUsed = Object.keys(domainScores);

  const guard = validateDomainUsage(context.target, domainsUsed);

  if (!guard.allowed) {
    return {
      status: "blocked",
      reason: "domain_violation",
      violations: guard.violations,
    };
  }

  const totalScore =
    domainScores.clinical_safety * 1.0 +
    domainScores.clinical_quality * 0.9;

  if (totalScore > 0.7) {
    await adapter.deploy(candidate);
    return { status: "deployed", score: totalScore };
  }

  return { status: "rejected", score: totalScore };
}

export function evaluateDataIntegrity(input: any) {
  const issues: string[] = [];

  let completeness = 1.0;
  let reliability = 1.0;

  // --- COMPLETENESS CHECKS ---
  if (!input.history) {
    completeness -= 0.2;
    issues.push("missing_history");
  }

  if (!input.medications) {
    completeness -= 0.2;
    issues.push("missing_medications");
  }

  if (!input.allergies) {
    completeness -= 0.2;
    issues.push("missing_allergies");
  }

  if (!input.vitals) {
    completeness -= 0.2;
    issues.push("missing_vitals");
  }

  if (!input.duration) {
    completeness -= 0.2;
    issues.push("missing_duration");
  }

  // --- RELIABILITY CHECKS ---
  if (input.source === "fhir" && !input.fhirComplete) {
    reliability -= 0.2;
    issues.push("incomplete_fhir");
  }

  if (input.deviceData?.unstable) {
    reliability -= 0.2;
    issues.push("device_unreliable");
  }

  if (input.timeline?.outOfOrder) {
    reliability -= 0.2;
    issues.push("timeline_inconsistent");
  }

  return {
    completeness: Math.max(0, completeness),
    reliability: Math.max(0, reliability),
    issues,
  };
}

export function assessDeviceReliability(data: any[]) {
  let unstable = false;

  const spo2 = data.filter(d => d.type === "spo2");

  if (spo2.some(d => d.value < 80 || d.value > 100)) {
    unstable = true;
  }

  if (spo2.length > 1) {
    const variance =
      Math.max(...spo2.map(d => d.value)) -
      Math.min(...spo2.map(d => d.value));

    if (variance > 10) unstable = true;
  }

  return {
    unstable,
  };
}

export function computeUncertainty(input: any, result: any) {
  let uncertainty = 0;

  // Low confidence from model
  if (result.score < 0.6) uncertainty += 0.3;

  // Data completeness penalty
  if (input.dataQuality?.completeness < 0.7) uncertainty += 0.3;

  // Reliability penalty
  if (input.dataQuality?.reliability < 0.7) uncertainty += 0.2;

  // Sparse symptom input
  if (!input.symptoms || input.symptoms.length < 2) {
    uncertainty += 0.2;
  }

  return Math.min(1, uncertainty);
}

export function detectOOD(input: any) {

  return {
    ood: flags.length > 0,

  };
}

3. APPROVAL REQUIREMENTS ENGINE
📁 server/governance/promotionPolicy.ts
import { DOMAIN_PERMISSIONS } from "./domainPermissions";
import { resolveDomain } from "./domainResolver";

export interface PromotionDecisionInput {
  target: string;
  score: number;
  safe: boolean;
  hasValidationPass: boolean;
  hasOODRisk?: boolean;
  hasDataIntegrityConcern?: boolean;
}

export interface PromotionDecision {
  allowed: boolean;
  action: "auto_promote" | "queue_for_review" | "reject";
  reason: string;
  domain: string;
  requiredReviewerRole?: string;
}

export function evaluatePromotionPolicy(
  input: PromotionDecisionInput
): PromotionDecision {
  const domain = resolveDomain(input.target);
  const permission = DOMAIN_PERMISSIONS[domain];

  if (!input.safe) {
    return {
      allowed: false,
      action: "reject",
      reason: "Candidate failed safety checks.",
      domain,
    };
  }

  if (!input.hasValidationPass) {
    return {
      allowed: false,
      action: "reject",
      reason: "Golden validation did not pass.",
      domain,
    };
  }

  if (input.hasOODRisk || input.hasDataIntegrityConcern) {
    return {
      allowed: false,
      action: "queue_for_review",
      reason: "Additional review required due to uncertainty or data-quality concern.",
      domain,
      requiredReviewerRole: permission.requiredReviewerRole,
    };
  }

  if (input.score < permission.minScore) {
    return {
      allowed: false,
      action: "reject",
      reason: `Score ${input.score.toFixed(3)} below domain threshold ${permission.minScore}.`,
      domain,
    };
  }

  if (permission.promotionMode === "auto") {
    return {
      allowed: true,
      action: "auto_promote",
      reason: "Domain allows auto-promotion and all gates passed.",
      domain,
      requiredReviewerRole: permission.requiredReviewerRole,
    };
  }

  if (permission.promotionMode === "review_required") {
    return {
      allowed: false,
      action: "queue_for_review",
      reason: "Domain requires human review before promotion.",
      domain,
      requiredReviewerRole: permission.requiredReviewerRole,
    };
  }

  return {
    allowed: false,
    action: "reject",
    reason: "Domain is blocked from automated promotion.",
    domain,
  };
}

function readAll(): ReviewQueueItem[] {
  if (!fs.existsSync(FILE)) return [];
  return JSON.parse(fs.readFileSync(FILE, "utf-8"));
}

function writeAll(items: ReviewQueueItem[]) {
  fs.writeFileSync(FILE, JSON.stringify(items, null, 2));
}

export function addToReviewQueue(
  item: Omit<ReviewQueueItem, "status" | "createdAt">
) {
  const items = readAll();
  const full: ReviewQueueItem = {
    ...item,
    status: "pending",
    createdAt: new Date().toISOString(),
  };
  items.push(full);
  writeAll(items);
  return full;
}

export function listReviewQueue(status?: ReviewQueueItem["status"]) {
  const items = readAll();
  return status ? items.filter(i => i.status === status) : items;
}

export function reviewQueueItem(params: {
  id: string;
  reviewer: string;
  decision: "approved" | "rejected";
  note?: string;
}) {
  const items = readAll();
  const idx = items.findIndex(i => i.id === params.id);
  if (idx === -1) {
    throw new Error("Review queue item not found");
  }

  items[idx] = {
    ...items[idx],
    status: params.decision,
    reviewedAt: new Date().toISOString(),
    reviewedBy: params.reviewer,
    reviewNote: params.note || "",
  };

  writeAll(items);
  return items[idx];
}

import { adapterRegistry } from "./adapterRegistry";
import { scoreByDomain } from "../../meta/domainScorer";
import { validateDomainUsage } from "../../meta/domainGuard";
import { evaluatePromotionPolicy } from "../../governance/promotionPolicy";
import { addToReviewQueue } from "../../governance/reviewQueueStore";
import { logDomainPermissionEvent } from "../../governance/domainPermissionAudit";

export async function runAdapterCycle(type: string, context: any) {
  const adapter = adapterRegistry[type as keyof typeof adapterRegistry];
  if (!adapter) {
    return {
      status: "error",
      reason: `Unknown adapter type: ${type}`,
    };
  }

  const candidate = await adapter.generate(context);
  const sim = await adapter.simulate(candidate, context);
  const scored = await adapter.score(sim);

  const domainScores = scoreByDomain(sim, context.real || {});
  const domainsUsed = Object.keys(domainScores);

  const domainGuard = validateDomainUsage(context.target, domainsUsed as any);
  if (!domainGuard.allowed) {
    const blocked = {
      status: "blocked",
      reason: "domain_violation",
      violations: domainGuard.violations,
      candidate,
    };

    logDomainPermissionEvent({
      event: "DOMAIN_VIOLATION_BLOCKED",
      target: context.target,
      adapterType: type,
      violations: domainGuard.violations,
      candidateId: candidate?.id || null,
    });

    return blocked;
  }

  const totalScore =
    (domainScores.clinical_safety || 0) * 1.0 +
    (domainScores.clinical_quality || 0) * 0.9 +
    (domainScores.workflow || 0) * 0.2;

  const decision = evaluatePromotionPolicy({
    target: context.target,
    score: totalScore,
    safe: !!scored.safe,
    hasValidationPass: context.hasValidationPass !== false,
    hasOODRisk: !!context.hasOODRisk,
    hasDataIntegrityConcern: !!context.hasDataIntegrityConcern,
  });

  logDomainPermissionEvent({
    event: "PROMOTION_POLICY_EVALUATED",
    target: context.target,
    adapterType: type,
    totalScore,
    decision,
    candidateId: candidate?.id || null,
  });

  if (decision.action === "auto_promote") {
    await adapter.deploy(candidate, context);
    return {
      status: "deployed",
      candidate,
      result: scored,
      score: totalScore,
      decision,
    };
  }

  if (decision.action === "queue_for_review") {
    const queued = addToReviewQueue({
      id: candidate?.id || `${type}_${Date.now()}`,
      target: context.target,
      domain: decision.domain,
      candidate,
      score: totalScore,
      safe: !!scored.safe,
      requiredReviewerRole: decision.requiredReviewerRole || "clinician",
      reason: decision.reason,
    });

    return {
      status: "queued_for_review",
      candidate,
      result: scored,
      score: totalScore,
      decision,
      queueItem: queued,
    };
  }

  return {
    status: "rejected",
    candidate,
    result: scored,
    score: totalScore,
    decision,
  };
}

export function assessCaseRisk(caseData: any) {
  let risk = 0;

  if (caseData.override) risk += 0.3;
  if (caseData.escalated) risk += 0.4;
  if (caseData.outcomeScore < 0.5) risk += 0.5;

      (now - new Date(c.createdAt).getTime()) / (1000 * 60 * 60 * 24);

    return ageDays > 180; // 6 months
  });
}

export async function runGoldenValidation() {
  const cases = selectValidationCases();

  let correct = 0;

  for (const c of cases) {
    const res = await runFullClinicalFlow({
      patientId: "test",
      complaint: c.complaint,
      data: c.input,
    });

    if (res.diagnosis === c.expected) {
      correct++;
    }

    c.usageCount++;
    c.lastValidated = new Date().toISOString();
  }

  return {
    accuracy: correct / cases.length,
    total: cases.length,
  };
}

export function blockAutonomousActions(output: any) {
  if (output.prescription) {
    return {
      blocked: true,
      reason: "autonomous_prescribing_not_allowed",
    };
  }

  if (output.disposition === "admit" && !output.requiresClinicianApproval) {
    return {
      blocked: true,
      reason: "autonomous_disposition_not_allowed",
    };
  }

  return { blocked: false };
}
