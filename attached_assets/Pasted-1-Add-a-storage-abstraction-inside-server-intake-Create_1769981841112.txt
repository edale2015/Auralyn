export type CaseStatus = "draft" | "submitted" | "in_review" | "signed" | "closed";

export type DraftPayload = {
  currentStep: number;
  draft: Record<string, any>;
};

export type VerifyPayload = {
  code: string;
  dob?: string;

export type SubmitPayload = {
  chiefComplaint: string;
  freeText?: string;
  symptoms: Record<string, "yes" | "no" | "ns">;
  modifiers?: Record<string, any>;
  meds?: any[];
  allergies?: any[];
  pmh?: any;
  pharmacy?: any;
  attachments?: string[];
  consent: {
    telehealth: boolean;
    privacy: boolean;
    signatureName: string;
    signedAt: string;
  };
};

export type FileMeta = {
  fileId: string;
  token: string;
  originalName: string;
  mimeType: string;
  storagePath: string;
  createdAt: number;
};

export interface IntakeStore {
  // intake session token + 6 digit code
  createSession(token: string, code: string, expiresAtMs: number): Promise<void>;
  verifySession(token: string, code: string): Promise<void>; // throws on failure

  // case lifecycle
  getOrCreateCaseForToken(token: string): Promise<{
    caseId: string;
    status: CaseStatus;
    currentStep: number;
  }>;

  saveDraft(token: string, payload: DraftPayload): Promise<void>;
  submit(token: string, intake: SubmitPayload, assistant: any): Promise<{ caseId: string }>;

  getStatus(token: string): Promise<{
    ok: true;
    caseId: string;
    status: CaseStatus;
    updatedAt: number;
    nextActionText: string;
  }>;

  getSummaryHtml(token: string): Promise<string>; // throws if not signed

  // provider
  getCase(caseId: string): Promise<any>;
  signCase(caseId: string): Promise<void>;

  // files
  addFileMeta(meta: FileMeta): Promise<void>;
  getFileMeta(fileId: string): Promise<FileMeta | null>;
}

server/intake/storage/crypto.ts
import crypto from "crypto";

export function sha256(s: string) {
  return crypto.createHash("sha256").update(s).digest("hex");
}

export function newId(prefix: string) {
  return `${prefix}_${crypto.randomBytes(10).toString("hex")}`;
}

export function nowMs() {
  return Date.now();
}

export function statusText(s: string) {
  if (s === "draft") return "Continue your intake.";
  if (s === "submitted") return "Submitted. Provider review pending.";
  if (s === "in_review") return "In review.";
  if (s === "signed") return "Complete. Summary available.";
  if (s === "closed") return "Closed.";
  return "Unknown.";
}

server/intake/storage/index.ts
import type { IntakeStore } from "./driver";
import { makeSqliteStore } from "./sqliteDriver";
import { makeFirestoreStore } from "./firestoreDriver";

let singleton: IntakeStore | null = null;

export function getIntakeStore(): IntakeStore {
  if (singleton) return singleton;
  const driver = (process.env.STORAGE_DRIVER || "sqlite").toLowerCase();
  singleton = driver === "firestore" ? makeFirestoreStore() : makeSqliteStore();
  return singleton;
}

2) Convert your existing SQLite layer into sqliteDriver.ts

This is the “direct translation”: your current db.ts logic should move behind the interface.

server/intake/storage/sqliteDriver.ts
import type { IntakeStore, DraftPayload, SubmitPayload, FileMeta, CaseStatus } from "./driver";
import { sha256, nowMs, statusText } from "./crypto";
import { db } from "../db"; // <— reuse your existing better-sqlite3 db instance
import { renderSummaryHtml } from "../pdf"; // or wherever your summary generator lives

export function makeSqliteStore(): IntakeStore {
  return {
    async createSession(token, code, expiresAtMs) {
      const hash = sha256(code.trim());
      db.prepare(`
        INSERT OR REPLACE INTO intake_sessions (token, code_hash, expires_at, used_at, created_at)
        VALUES (?, ?, ?, NULL, ?)
      `).run(token, hash, expiresAtMs, nowMs());
    },

    async verifySession(token, code) {
      const s = db.prepare(`SELECT * FROM intake_sessions WHERE token=?`).get(token);
      if (!s) throw new Error("Invalid link.");
      if (s.used_at) throw new Error("This link has already been used.");
      if (Number(s.expires_at) < nowMs()) throw new Error("This link has expired.");
      if (sha256(code.trim()) !== s.code_hash) throw new Error("Incorrect code.");
    },

    async getOrCreateCaseForToken(token) {
      const row = db.prepare(`SELECT * FROM cases WHERE token=? ORDER BY created_at DESC LIMIT 1`).get(token);
      if (row) {
        return { caseId: row.case_id, status: row.status as CaseStatus, currentStep: row.current_step ?? 0 };
      }

      const caseId = `CASE_${nowMs()}_${Math.random().toString(16).slice(2)}`;
      const ts = nowMs();
      db.prepare(`
        INSERT INTO cases (case_id, token, status, created_at, updated_at, current_step, draft_json, intake_json, assistant_json)
        VALUES (?, ?, 'draft', ?, ?, 0, '{}', '{}', '{}')
      `).run(caseId, token, ts, ts);

      return { caseId, status: "draft", currentStep: 0 };
    },

    async saveDraft(token, payload: DraftPayload) {
      const c = await this.getOrCreateCaseForToken(token);
      const row = db.prepare(`SELECT * FROM cases WHERE case_id=?`).get(c.caseId);

      let existing: any = {};
      try { existing = JSON.parse(row?.draft_json || "{}"); } catch {}
      const merged = { ...existing, ...payload.draft };

      db.prepare(`
        UPDATE cases
        SET draft_json=?, current_step=?, updated_at=?, status='draft'
        WHERE case_id=?
      `).run(JSON.stringify(merged), payload.currentStep, nowMs(), c.caseId);
    },

    async submit(token, intake: SubmitPayload, assistant: any) {
      const c = await this.getOrCreateCaseForToken(token);

      db.prepare(`
        UPDATE cases
        SET intake_json=?, assistant_json=?, status='submitted', updated_at=?
        WHERE case_id=?
      `).run(JSON.stringify(intake), JSON.stringify(assistant), nowMs(), c.caseId);

      db.prepare(`UPDATE intake_sessions SET used_at=? WHERE token=?`).run(nowMs(), token);

      return { caseId: c.caseId };
    },

    async getStatus(token) {
      const row = db.prepare(`SELECT * FROM cases WHERE token=? ORDER BY created_at DESC LIMIT 1`).get(token);
      if (!row) throw new Error("Not found.");
      return {
        ok: true as const,
        caseId: row.case_id,
        status: row.status as CaseStatus,
        updatedAt: row.updated_at,
        nextActionText: statusText(row.status)
      };
    },

    async getSummaryHtml(token) {
      const row = db.prepare(`SELECT * FROM cases WHERE token=? ORDER BY created_at DESC LIMIT 1`).get(token);
      if (!row) throw new Error("Not found.");
      if (row.status !== "signed") throw new Error("Summary not available yet.");
      return row.summary_html || "<html><body>No summary.</body></html>";
    },

    async getCase(caseId: string) {
      const row = db.prepare(`SELECT * FROM cases WHERE case_id=?`).get(caseId);
      if (!row) throw new Error("Not found.");
      return {
        caseId: row.case_id,
        status: row.status,
        intake: JSON.parse(row.intake_json || "{}"),
        assistant: JSON.parse(row.assistant_json || "{}"),
        updatedAt: row.updated_at
      };
    },

    async signCase(caseId: string) {
      const row = db.prepare(`SELECT * FROM cases WHERE case_id=?`).get(caseId);
      if (!row) throw new Error("Not found.");
      const intake = JSON.parse(row.intake_json || "{}");
      const assistant = JSON.parse(row.assistant_json || "{}");
      const html = renderSummaryHtml(caseId, intake, assistant);

      db.prepare(`
        UPDATE cases
        SET status='signed', summary_html=?, updated_at=?
        WHERE case_id=?
      `).run(html, nowMs(), caseId);
    },

    async addFileMeta(meta: FileMeta) {
      db.prepare(`
        INSERT INTO files (file_id, token, original_name, mime_type, storage_path, created_at)
        VALUES (?, ?, ?, ?, ?, ?)
      `).run(meta.fileId, meta.token, meta.originalName, meta.mimeType, meta.storagePath, meta.createdAt);
    },

    async getFileMeta(fileId: string) {
      const row = db.prepare(`SELECT * FROM files WHERE file_id=?`).get(fileId);
      if (!row) return null;
      return {
        fileId: row.file_id,
        token: row.token,
        originalName: row.original_name,
        mimeType: row.mime_type,
        storagePath: row.storage_path,
        createdAt: row.created_at
      };
    }
  };
}


Key point: this lets SQLite remain fully supported with zero behavior change.

3) Add Firestore driver (new)

Install in your server package:

npm i firebase-admin

server/intake/storage/firestoreDriver.ts
import admin from "firebase-admin";
import type { IntakeStore, DraftPayload, SubmitPayload, FileMeta, CaseStatus } from "./driver";
import { sha256, nowMs, statusText } from "./crypto";
import { renderSummaryHtml } from "../pdf";

function initFirebase() {
  if (admin.apps.length) return;
  admin.initializeApp({
    projectId: process.env.FIREBASE_PROJECT_ID
    // In dev: GOOGLE_APPLICATION_CREDENTIALS points to service account JSON.
    // In prod on GCP: default creds typically work.
  });
}

export function makeFirestoreStore(): IntakeStore {
  initFirebase();
  const db = admin.firestore();

  const sessions = db.collection("intake_sessions");
  const cases = db.collection("cases");
  const files = db.collection("files");

  async function latestCaseByToken(token: string) {
    const q = await cases.where("token", "==", token).orderBy("created_at", "desc").limit(1).get();
    return q.empty ? null : q.docs[0];
  }

  return {
    async createSession(token, code, expiresAtMs) {
      await sessions.doc(token).set({
        token,
        code_hash: sha256(code.trim()),
        expires_at: expiresAtMs,
        used_at: null,
        created_at: nowMs()
      }, { merge: true });
    },

    async verifySession(token, code) {
      const snap = await sessions.doc(token).get();
      if (!snap.exists) throw new Error("Invalid link.");
      const s: any = snap.data();
      if (s.used_at) throw new Error("This link has already been used.");
      if (Number(s.expires_at) < nowMs()) throw new Error("This link has expired.");
      if (sha256(code.trim()) !== s.code_hash) throw new Error("Incorrect code.");
    },

    async getOrCreateCaseForToken(token) {
      const doc = await latestCaseByToken(token);
      if (doc) {
        const d: any = doc.data();
        return { caseId: d.case_id, status: d.status as CaseStatus, currentStep: d.current_step ?? 0 };
      }

      const caseId = `CASE_${nowMs()}_${Math.random().toString(16).slice(2)}`;
      const ts = nowMs();
      await cases.doc(caseId).set({
        case_id: caseId,
        token,
        status: "draft",
        created_at: ts,
        updated_at: ts,
        current_step: 0,
        draft_json: {},
        intake_json: {},
        assistant_json: {},
        summary_html: null
      });

      return { caseId, status: "draft", currentStep: 0 };
    },

    async saveDraft(token, payload: DraftPayload) {
      const { caseId } = await this.getOrCreateCaseForToken(token);
      const ref = cases.doc(caseId);

      await db.runTransaction(async (tx) => {
        const snap = await tx.get(ref);
        const d: any = snap.exists ? snap.data() : {};
        const existingDraft = d?.draft_json || {};
        const merged = { ...existingDraft, ...payload.draft };

        tx.set(ref, {
          draft_json: merged,
          current_step: payload.currentStep,
          status: "draft",
          updated_at: nowMs()
        }, { merge: true });
      });
    },

    async submit(token, intake: SubmitPayload, assistant: any) {
      const { caseId } = await this.getOrCreateCaseForToken(token);

      await cases.doc(caseId).set({
        intake_json: intake,
        assistant_json: assistant,
        status: "submitted",
        updated_at: nowMs()
      }, { merge: true });

      await sessions.doc(token).set({ used_at: nowMs() }, { merge: true });

      return { caseId };
    },

    async getStatus(token) {
      const doc = await latestCaseByToken(token);
      if (!doc) throw new Error("Not found.");
      const d: any = doc.data();
      return {
        ok: true as const,
        caseId: d.case_id,
        status: d.status as CaseStatus,
        updatedAt: d.updated_at,
        nextActionText: statusText(d.status)
      };
    },

    async getSummaryHtml(token) {
      const doc = await latestCaseByToken(token);
      if (!doc) throw new Error("Not found.");
      const d: any = doc.data();
      if (d.status !== "signed") throw new Error("Summary not available yet.");
      return d.summary_html || "<html><body>No summary.</body></html>";
    },

    async getCase(caseId: string) {
      const snap = await cases.doc(caseId).get();
      if (!snap.exists) throw new Error("Not found.");
      const d: any = snap.data();
      return {
        caseId: d.case_id,
        status: d.status,
        intake: d.intake_json || {},
        assistant: d.assistant_json || {},
        updatedAt: d.updated_at
      };
    },

    async signCase(caseId: string) {
      const ref = cases.doc(caseId);
      const snap = await ref.get();
      if (!snap.exists) throw new Error("Not found.");
      const d: any = snap.data();

      const html = renderSummaryHtml(caseId, d.intake_json || {}, d.assistant_json || {});
      await ref.set({
        status: "signed",
        summary_html: html,
        updated_at: nowMs()
      }, { merge: true });
    },

    async addFileMeta(meta: FileMeta) {
      await files.doc(meta.fileId).set(meta, { merge: true });
    },

    async getFileMeta(fileId: string) {
      const snap = await files.doc(fileId).get();
      if (!snap.exists) return null;
      return snap.data() as FileMeta;
    }
  };
}

4) Update your existing intake routes to use the store

You don’t need to change route registration or the server entry at all.

In each of these files, do the same pattern:

server/intake/routes.intake.ts (minimal edits)

At the top:

import { getIntakeStore } from "./storage";
const store = getIntakeStore();


Then swap DB calls:

Verify endpoint:

await store.verifySession(token, code);
const c = await store.getOrCreateCaseForToken(token);
return res.json({ ok: true, caseId: c.caseId, status: c.status, currentStep: c.currentStep, flowId: "DEFAULT_FLOW" });


Save draft:

await store.saveDraft(token, req.body);
return res.json({ ok: true });


Submit:

const assistant = computeAssistant(intake); // your existing logic
const { caseId } = await store.submit(token, intake, assistant);
return res.json({ ok: true, caseId });


Status:

return res.json(await store.getStatus(token));


Summary:

const html = await store.getSummaryHtml(token);
res.setHeader("Content-Type", "text/html; charset=utf-8");
return res.send(html);

server/intake/routes.files.ts

At the end of upload handler:

await store.addFileMeta({ fileId, token, originalName, mimeType, storagePath, createdAt: Date.now() });


For download:

Fetch meta via store, and (important) validate token ownership the way you already do:

const meta = await store.getFileMeta(fileId);
if (!meta) return res.status(404).send("Not found");

server/intake/routes.summary.ts (provider case view + sign)

Replace reading case by SQLite with:

const c = await store.getCase(caseId);


and sign with:

await store.signCase(caseId);
