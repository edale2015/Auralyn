// server/qa/mergeSkills.ts
import cosine from 'compute-cosine-similarity'

export async function findMergeCandidates(db: any) {
  const skills = await db.kb_skills.findAll()
  const pairs: any[] = []

  for (let i = 0; i < skills.length; i++) {
    for (let j = i + 1; j < skills.length; j++) {
      const sim = cosine(skills[i].embedding, skills[j].embedding)
      if (sim > 0.85) {
        pairs.push({
          skill_a: skills[i].id,
          skill_b: skills[j].id,
          similarity: sim
        })
      }
    }
  }

  return pairs
}

export function mergeSkillLogic(a: any, b: any) {
  return {
    name: `${a.name}_${b.name}_merged`,
    complaints: [...new Set([...(a.complaints||[]), ...(b.complaints||[])])],
    modifiers: [...new Set([...(a.modifiers||[]), ...(b.modifiers||[])])],
    rules: [...new Set([...(a.rules||[]), ...(b.rules||[])])]
  }
}
🔌 Route
router.get('/merge-candidates', async (req, res) => {
  const pairs = await findMergeCandidates(req.db)
  res.json(pairs)
})

router.post('/merge-apply', async (req, res) => {
  const { skillA, skillB } = req.body

  const merged = mergeSkillLogic(skillA, skillB)

  await req.db.merged_skills.insert({
    new_skill_id: merged.name,
    original_skills: [skillA.id, skillB.id],
    merged_logic: merged
  })

  res.json(merged)
})

// server/qa/dependencyOptimizer.ts
export async function optimizeDependencies(db: any) {
  const skills = await db.kb_skills.findAll()

  const suggestions: any[] = []

  for (const s of skills) {
    if (!s.modifiers?.length) {
      suggestions.push({
        skill: s.name,
        suggestion: "Add modifier dependency"
      })
    }
  }

  return suggestions
}

// server/qa/minSkillSet.ts
export async function computeMVSS(db: any) {
  const complaints = await db.kb_complaints.findAll()
  const skills = await db.kb_skills.findAll()

  const required: Set<string> = new Set()

  for (const c of complaints) {
    const related = skills.filter((s: any) =>
      s.complaints?.includes(c.id)
    )

    if (related.length) {
      required.add(related[0].id)
    }
  }

  return Array.from(required)
}
🎛️ UI

    </div>
  )
}
