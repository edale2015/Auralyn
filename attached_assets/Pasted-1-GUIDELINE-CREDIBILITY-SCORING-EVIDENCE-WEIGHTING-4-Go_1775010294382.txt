⚙️ Scoring Engine
export function computeEvidenceStrength(e) {
  let score = 0

  if (e.source_type === 'meta_analysis') score += 5
  if (e.source_type === 'RCT') score += 4
  if (e.source_type === 'cohort') score += 3
  if (e.source_type === 'expert_opinion') score += 1

  score += Math.log10(e.sample_size || 1)
  score += (new Date().getFullYear() - e.year < 5 ? 2 : 0)
  score += e.journal_impact || 0

  return score
}

export async function generateReviewCycle(db) {
  const specialties = ['ENT','PULM','CARDIO','NEURO']

  for (const s of specialties) {
    await db.specialty_review_cycles.insert({
      specialty: s,
      cycle_date: new Date(),
      status: 'pending'
    })
  }
}

export function analyzeOutcomes(rows) {
  const mismatches = []

  rows.forEach(r => {
    if (r.predicted_dx !== r.actual_dx) {
      mismatches.push(r)
    }
  })

  return {
    mismatch_rate: mismatches.length / rows.length,
    clusters: groupByDiagnosis(mismatches)
  }
}
