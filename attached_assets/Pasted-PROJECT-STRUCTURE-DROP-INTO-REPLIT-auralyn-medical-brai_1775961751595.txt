from fastapi import WebSocket

class ConnectionManager:
    def __init__(self):
        self.connections = []

    async def connect(self, ws: WebSocket):
        await ws.accept()
        self.connections.append(ws)

    async def broadcast(self, message: str):
        for conn in self.connections:
            await conn.send_text(message)

manager = ConnectionManager()

import requests

FHIR_BASE = "https://api.epic.com/interconnect-fhir-oauth/api/FHIR/R4"

def get_patient(patient_id, token):
    headers = {"Authorization": f"Bearer {token}"}
    r = requests.get(f"{FHIR_BASE}/Patient/{patient_id}", headers=headers)
    return r.json()

def create_observation(data, token):
    headers = {
        "Authorization": f"Bearer {token}",
        "Content-Type": "application/json"
    }
    return requests.post(f"{FHIR_BASE}/Observation", json=data, headers=headers).json()

🧭 7. CONTROL TOWER (NODE DASHBOARD API)
/control_tower/server.ts
import express from "express";

const app = express();
app.use(express.json());

let systemState = {
  plugins: [
    { name: "clinical", status: "healthy" },
    { name: "fhir", status: "healthy" },
    { name: "billing", status: "healthy" }
  ]
};

app.get("/status", (_, res) => {
  res.json(systemState);
});

app.post("/toggle", (req, res) => {
  const { name } = req.body;

  systemState.plugins = systemState.plugins.map(p =>
    p.name === name ? { ...p, status: "disabled" } : p
  );

  res.json({ ok: true });
});

app.listen(3000, () => {
  console.log("Control Tower running on port 3000");
});
