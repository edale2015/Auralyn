🧾 Table: kb_feature_likelihoods (CRITICAL TABLE)
CREATE TABLE kb_feature_likelihoods (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  rule_id TEXT NOT NULL,                     -- FK to kb_diagnosis_rules.rule_id
  feature_key TEXT NOT NULL,                 -- e.g. "painful_arc"
  feature_value TEXT,                        -- "yes", "no", "severe"

  created_at TIMESTAMP DEFAULT now()
);

⚙️ 3. BACKEND MIGRATION ENGINE
File: server/migrations/migrateCsvToKb.ts
import fs from 'fs'
import { db } from '../db'

export async function migrateCsvPriors(csvData: any[]) {
  for (const row of csvData) {
    const ruleId = `DX_BAY_${row.diagnosis.toUpperCase()}`

    await db.kb_diagnosis_rules.upsert({
      rule_id: ruleId,
      diagnosis_name: row.diagnosis,
      complaint_id: row.complaint || 'bayesian_global',
      base_probability: row.baseProbability || 0.01
    })

    if (row.featureLikelihoods) {
      for (const [feature, likelihood] of Object.entries(row.featureLikelihoods)) {
        await db.kb_feature_likelihoods.insert({
          rule_id: ruleId,
          feature_key: feature,
          feature_value: 'yes',
          likelihood: likelihood as number
        })
      }
    }
  }
}
🧠 4. NEW DIAGNOSIS ENGINE (KB-ONLY)
File: server/engine/kbDiagnosisEngine.ts
export async function runDiagnosis(input: any) {
  const diagnoses = await db.kb_diagnosis_rules.findAll({ is_active: true })

  const results = []

  for (const dx of diagnoses) {
    const features = await db.kb_feature_likelihoods.findAll({
      rule_id: dx.rule_id
    })

    let score = Math.log(dx.base_probability)

    for (const f of features) {
      if (input[f.feature_key] === f.feature_value) {
        score += Math.log(f.likelihood * (f.weight || 1))
      }
    }

    results.push({
      diagnosis: dx.diagnosis_name,
      score,
      source: 'KB_DB'
    })
  }

  return results.sort((a, b) => b.score - a.score)
}
