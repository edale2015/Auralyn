import { db } from "../db";

export async function recordOutcome({
  input,
  predicted,
  actual
}) {
  await db.insert("outcomes").values({
    input,
    predicted,
    actual
  });
}

export async function runLearningCycle() {
  const outcomes = await db
    .select()
    .from("outcomes")
    .orderBy("created_at DESC")
    .limit(200);

  for (const o of outcomes) {
    const delta = o.predicted === o.actual ? 0.02 : -0.05;

    await db.execute(`
      UPDATE weights
      SET value = value + ${delta}
      WHERE diagnosis = '${o.predicted}'
    `);
  }

  console.log("🧠 Learning cycle complete");
}

💊 Drug Interaction Engine
// server/engines/safety/drugInteraction.ts

const interactions = {
  warfarin: ["ibuprofen", "aspirin"],
  lisinopril: ["potassium"]
};

export function checkDrugInteractions(meds) {
  const issues = [];

  meds.forEach(m => {
    if (interactions[m]) {
      issues.push(...interactions[m]);
    }
  });

  return issues;
}
