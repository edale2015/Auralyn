// server/guidelines/ingest.ts
export async function ingestGuideline(text) {
  const parsed = await extractClinicalLogic(text)

  await db.guideline_documents.insert({
    source: 'manual',
    content: text,
    parsed
  })

  return parsed
}
⚙️ LLM extraction (THIS is what you wanted)
async function extractClinicalLogic(text) {
  return await openai.chat.completions.create({
    model: "gpt-5",
    messages: [
      {
        role: "system",
        content: `
Extract clinical decision rules:
- conditions
- actions
- modifiers
- red flags
Return structured JSON
        `
      },
      { role: "user", content: text }
    ]
  })
}
🔄 Convert to suggestions
export function generateGuidelineSuggestions(parsed) {
  return parsed.rules.map(r => ({
    complaint: r.complaint,
    recommendation: r.action,
    rationale: r.reason,
    confidence: r.confidence
  }))
}

// server/qa/compareGuideline.ts
export async function compareToGuideline(complaint) {
  const systemRules = await db.kb_rules.findAll({ complaint })
  const guidelineRules = await db.guideline_recommendations.findAll({ complaint })

  const gaps = []

  guidelineRules.forEach(g => {
    if (!systemRules.some(s => s.rule === g.recommendation)) {
      gaps.push({
        missing: g.recommendation,
        rationale: g.rationale
      })
    }
  })

  return gaps
}
