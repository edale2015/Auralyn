🔐 AUTH SYSTEM (JWT + SAFETY)
/server/auth/authMiddleware.ts
import jwt from "jsonwebtoken";

export function authMiddleware(req: any, res: any, next: any) {
  const token = req.headers.authorization?.split(" ")[1];

  if (!token) return res.status(401).send("Unauthorized");

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET!);
    req.user = decoded;
    next();
  } catch {
    res.status(403).send("Invalid token");
  }
}

🧠 FULL PIPELINE (FINAL FORM)
/server/pipeline/fullPipeline.ts
import { createClinicalTokenSet } from "../core/clinicalTokens";
import { applyDecisionTemperature } from "../engine/decisionTemperature";
import { applyShadowSafety } from "../safety/shadowEngine";
import { buildTrace } from "../audit/traceEngine";
import { generateClinicalOutput } from "../output/clinicalOutput";
import { runSpecialistCouncil } from "../agents/specialistCouncil";

export async function runFullPipeline(input: any) {
  let tokens = createClinicalTokenSet(input);

  tokens = applyDecisionTemperature(tokens);
  tokens = applyShadowSafety(tokens);

  const trace = buildTrace(tokens);
  const specialistConsensus = await runSpecialistCouncil(tokens);
  const output = generateClinicalOutput(tokens);

  return {
    trace,
    specialistConsensus,
    output,
  };
}

🧠 GOLDEN CASE TESTING
/server/golden/goldenHarness.ts
import { runFullPipeline } from "../pipeline/fullPipeline";

export async function runGoldenCases(cases: any[]) {
  return Promise.all(
    cases.map(async (c) => {
      const result = await runFullPipeline(c.input);

      return {
        expected: c.expected,
        actual: result.output,
        pass: c.expected.disposition === result.output.disposition,
      };
    })
  );
}
🧠 FHIR INTEGRATION
/server/fhir/fhirService.ts
import axios from "axios";

export async function pushFHIR(patientId: string, data: any) {
  return axios.post(
    `${process.env.FHIR_URL}/Observation`,
    {
      resourceType: "Observation",
      subject: { reference: `Patient/${patientId}` },
      valueString: JSON.stringify(data),
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.FHIR_TOKEN}`,
      },
    }
  );
}

      .then(res => res.json())
      .then(setData);
