🧮 1) CENTOR ENGINE
/server/services/clinical/centorEngine.ts
export interface CentorInput {
  fever: boolean;
  tonsillarExudate: boolean;
  tenderAnteriorCervicalNodes: boolean;
  absenceOfCough: boolean;
  age: number;
}

export function calculateCentorScore(input: CentorInput): number {
  let score = 0;

  if (input.fever) score++;
  if (input.tonsillarExudate) score++;
  if (input.tenderAnteriorCervicalNodes) score++;
  if (input.absenceOfCough) score++;

  if (input.age < 15) score++;
  if (input.age > 44) score--;

  return score;
}

export function centorDecision(score: number): string {
  if (score <= 1) return "NO_ANTIBIOTIC";
  if (score === 2 || score === 3) return "TEST_OR_DELAYED_RX";
  if (score >= 4) return "EMPIRIC_ANTIBIOTIC";
  return "UNKNOWN";
}
📊 2) BAYESIAN STREP ENGINE
/server/services/clinical/bayesianStrepEngine.ts
export function calculateStrepProbability(features: {
  fever: boolean;
  exudate: boolean;
  nodes: boolean;
  cough: boolean;
}): number {
  let probability = 0.1; // baseline

  if (features.fever) probability += 0.2;
  if (features.exudate) probability += 0.25;
  if (features.nodes) probability += 0.2;
  if (!features.cough) probability += 0.15;

  return Math.min(probability, 0.95);
}
🧠 3) MULTI-AGENT DEBATE ENGINE
/server/services/communication/debateEngine.ts
export interface DebateResult {
  decision: string;
  reasoning: string[];
}

export function runAntibioticDebate(input: {
  centorScore: number;
  strepProbability: number;
}): DebateResult {

  const pro: string[] = [];
  const con: string[] = [];

  if (input.centorScore >= 4) {
    pro.push("High Centor score supports antibiotics");
  }

  if (input.strepProbability > 0.6) {
    pro.push("High probability of bacterial infection");
  }

  if (input.centorScore <= 2) {
    con.push("Low Centor score argues against antibiotics");
  }

  if (input.strepProbability < 0.3) {
    con.push("Low bacterial probability");
  }

  // 🧠 Consensus logic
  if (pro.length > con.length) {
    return {
      decision: "ANTIBIOTIC_GIVEN",
      reasoning: pro
    };
  }

  return {
    decision: "NO_ANTIBIOTIC_OR_DELAYED",
    reasoning: con
  };
}

import twilio from "twilio";

const client = twilio(process.env.TWILIO_SID, process.env.TWILIO_AUTH);

export async function speakToPatient(phone: string, message: string) {
  return client.calls.create({
    to: phone,
    from: process.env.TWILIO_NUMBER,
    twiml: `<Response><Say>${message}</Say></Response>`
  });
}
🌐 5) FULL CLINICAL DECISION ROUTE
/server/routes/clinicalDecisionRoutes.ts
import express from "express";
import { calculateCentorScore, centorDecision } from "../services/clinical/centorEngine";
import { calculateStrepProbability } from "../services/clinical/bayesianStrepEngine";
import { runAntibioticDebate } from "../services/communication/debateEngine";
import { speakToPatient } from "../services/communication/voiceService";

const router = express.Router();

router.post("/full-decision", async (req, res) => {
  try {
    const { patient, symptoms, phone } = req.body;

    // 1. Centor
    const centorScore = calculateCentorScore(symptoms);
    const centorResult = centorDecision(centorScore);

    // 2. Bayesian
    const probability = calculateStrepProbability({
      fever: symptoms.fever,
      exudate: symptoms.tonsillarExudate,
      nodes: symptoms.tenderAnteriorCervicalNodes,
      cough: symptoms.cough
    });

    // 3. Debate
    const debate = runAntibioticDebate({
      centorScore,
      strepProbability: probability
    });

    // 4. Voice output
    const message = `
Based on your symptoms, your risk of a bacterial infection is low.
Antibiotics are not likely to help at this stage.
We will treat your symptoms and monitor for changes.
`;

    if (phone) {
      await speakToPatient(phone, message);
    }

    res.json({
      centorScore,
      centorResult,
      probability,
      decision: debate.decision,
      reasoning: debate.reasoning
    });

  } catch (err) {
    res.status(500).json({ error: "Decision pipeline failed" });
  }
});

export default router;
