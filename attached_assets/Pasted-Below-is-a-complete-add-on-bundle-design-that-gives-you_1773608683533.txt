import fetch from "node-fetch"

export async function literatureScraperEngine(query: string) {

  const url =
    `https://api.ncbi.nlm.nih.gov/lit/ctxp/v1/pubmed/?format=citation&term=${encodeURIComponent(query)}`

  const res = await fetch(url)

  const text = await res.text()

  return {
    source: "pubmed",
    content: text
  }
}

Engine 3: Evidence Weighting Engine

server/research/evidenceWeightingEngine.ts

export function evidenceWeightingEngine(sourceType: string) {

  const weights: Record<string, number> = {

    guideline: 1.0,
    review: 0.9,
    journal: 0.8,
    sheet: 0.7,
    journalism: 0.4,
    forum: 0.2,
    commentary: 0.2
  }

  return weights[sourceType] ?? 0.1
}

This prevents unreliable sources from influencing reasoning.

Engine 4: Graph Deduplication Engine

server/research/graphDeduplicationEngine.ts

export function graphDeduplicationEngine(edges:any[]) {

  const seen = new Set()

  const deduped = []

  for(const e of edges){

    const key = `${e.from}-${e.relation}-${e.to}`

    if(!seen.has(key)){

      seen.add(key)

      deduped.push(e)
    }
  }

  return deduped
}
Engine 5: Knowledge Gap Detector

server/research/knowledgeGapEngine.ts

export function knowledgeGapEngine(graph:any, complaints:string[]) {

  const gaps:any[] = []

  for(const c of complaints){

    const connected =
      graph.edges.filter((e:any)=>e.from===c)

    if(connected.length < 5){

      gaps.push({
        complaint:c,
        issue:"low knowledge coverage"
      })
    }
  }

  return gaps
}

This identifies complaints that need more differential data.

import React, { useEffect, useRef } from "react";
import mermaid from "mermaid";

mermaid.initialize({ startOnLoad: false, securityLevel: "loose" });

type Props = {
  chart: string;
};

export default function MermaidDiagram({ chart }: Props) {
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    let mounted = true;

    async function renderChart() {
      if (!ref.current) return;
      const id = `mermaid-${Date.now()}`;
      try {
        const { svg } = await mermaid.render(id, chart);
        if (mounted && ref.current) {
          ref.current.innerHTML = svg;
        }
      } catch (err) {
        if (ref.current) {
          ref.current.innerHTML = `<pre style="color:red;">${String(err)}</pre>`;
        }
      }
    }

    renderChart();

    return () => {
      mounted = false;
    };
  }, [chart]);

  return <div ref={ref} />;
}

That gives you in-app visual maps, which is much better than diagram exile.

A. Conversation Audit Engine

Scores the interaction.

server/core/conversationAuditEngine.ts

export type ConversationAudit = {
  empathyScore: number;
  completenessScore: number;
  clarityScore: number;
  safetyScore: number;
  deescalationScore: number;
  missedQuestions: string[];
  repeatedQuestions: string[];
  unsafePhrases: string[];
  suggestions: string[];
};

export function conversationAuditEngine(input: {
  messages: { role: string; content: string }[];
  complaint?: string;
  expectedQuestions?: string[];
}): ConversationAudit {
  const text = input.messages.map(m => m.content.toLowerCase()).join("\n");

  const missedQuestions =
    (input.expectedQuestions || []).filter(q => !text.includes(q.toLowerCase()));

  const unsafePhrases: string[] = [];
  if (text.includes("nothing to worry about")) unsafePhrases.push("nothing to worry about");
  if (text.includes("definitely benign")) unsafePhrases.push("definitely benign");

  const repeatedQuestions: string[] = [];
  const seen = new Map<string, number>();
  for (const msg of input.messages) {
    if (msg.role !== "assistant") continue;
    const c = msg.content.trim().toLowerCase();
    seen.set(c, (seen.get(c) || 0) + 1);
  }
  for (const [k, v] of seen.entries()) {
    if (v > 1 && k.length > 10) repeatedQuestions.push(k);
  }

  const empathyHits = [
    "i'm sorry",
    "that sounds difficult",
    "i want to help",
    "you are safe",
    "thank you for explaining"
  ].filter(p => text.includes(p)).length;

  const deescalationHits = [
    "take a breath",
    "we can go step by step",
    "you are not alone",
    "let's focus on the most important part first"
  ].filter(p => text.includes(p)).length;

  return {
    empathyScore: Math.min(1, empathyHits / 3),
    completenessScore: Math.max(0, 1 - missedQuestions.length / Math.max(1, (input.expectedQuestions || []).length)),
    clarityScore: repeatedQuestions.length ? 0.65 : 0.9,
    safetyScore: unsafePhrases.length ? 0.45 : 0.95,
    deescalationScore: Math.min(1, deescalationHits / 2),
    missedQuestions,
    repeatedQuestions,
    unsafePhrases,
    suggestions: [
      ...(missedQuestions.length ? ["Ask the highest-yield missed complaint questions first."] : []),
      ...(unsafePhrases.length ? ["Remove over-reassuring language and replace with safety-net wording."] : []),
      ...(repeatedQuestions.length ? ["Avoid repeating the same question unless confirming a contradiction."] : []),
    ],
  };
}
B. Prompt Improvement Engine

Rewrites the next question or response.

server/core/promptImprovementEngine.ts

export type PromptImprovementInput = {
  currentPrompt: string;
  goal: "empathy" | "clarity" | "deescalation" | "clinical_precision";
};

export function promptImprovementEngine(input: PromptImprovementInput): string {
  const p = input.currentPrompt.trim();

  switch (input.goal) {
    case "empathy":
      return `Please respond with warmth, validate the patient's experience briefly, and then ask the next clinically useful question: ${p}`;
    case "clarity":
      return `Rewrite this in simple language at about a 6th-grade reading level, keeping the clinical meaning exact: ${p}`;
    case "deescalation":
      return `Rewrite this to reduce anxiety, avoid alarmist phrasing, and guide the patient step by step while preserving safety: ${p}`;
    case "clinical_precision":
      return `Rewrite this to ask one high-yield, non-redundant clinical question with precise wording: ${p}`;
    default:
      return p;
  }
}
C. Tone Strategy Engine

Chooses the best interaction style for the situation.

server/core/toneStrategyEngine.ts

export type ToneStrategy =
  | "calm_reassuring"
  | "urgent_directive"
  | "empathetic_gathering"
  | "structured_triage"
  | "deescalation_support";

export function toneStrategyEngine(input: {
  anxietyLevel?: number;
  severityLevel?: "low" | "moderate" | "high" | "critical";
  contradiction?: boolean;
  aggression?: boolean;
}): ToneStrategy {
  if (input.severityLevel === "critical") return "urgent_directive";
  if (input.aggression) return "deescalation_support";
  if ((input.anxietyLevel || 0) > 0.7) return "calm_reassuring";
  if (input.contradiction) return "structured_triage";
  return "empathetic_gathering";
}
D. De-escalation Protocol Engine

Applies a structured response when the patient is upset, panicked, hostile, or overwhelmed.

server/core/deEscalationProtocolEngine.ts

export type DeEscalationPlan = {
  opening: string;
  steps: string[];
  avoid: string[];
};

export function deEscalationProtocolEngine(input: {
  scenario: "panic" | "anger" | "frustration" | "overwhelm";
}): DeEscalationPlan {
  switch (input.scenario) {
    case "panic":
      return {
        opening: "I can help you step through this one piece at a time.",
        steps: [
          "Acknowledge distress briefly.",
          "Ask one immediate safety question.",
          "Give one simple instruction.",
          "Avoid multiple questions at once."
        ],
        avoid: ["Don't overload with long explanations.", "Don't minimize fear."]
      };
    case "anger":
      return {
        opening: "I hear that this has been frustrating.",
        steps: [
          "Reflect emotion without arguing.",
          "State the next concrete step.",
          "Keep wording concise and neutral."
        ],
        avoid: ["Don't sound defensive.", "Don't debate the patient's feelings."]
      };
    case "frustration":
      return {
        opening: "Let's simplify this and focus on the most important part first.",
        steps: [
          "Summarize the issue in one sentence.",
          "Ask the highest-yield unanswered question.",
          "Show forward movement."
        ],
        avoid: ["Don't repeat prior questions unnecessarily."]
      };
    default:
      return {
        opening: "We can take this step by step.",
        steps: [
          "Slow the pace.",
          "Use short sentences.",
          "Ask one question at a time."
        ],
        avoid: ["Don't crowd the patient with choices."]
      };
  }
}
