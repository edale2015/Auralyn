// server/fda/validationEngine.ts

interface TestCase {
  input: any;
  expectedDisposition: string;
}

export function runValidationSuite(cases: TestCase[]) {
  let correct = 0;

  const results = cases.map((c) => {
    const output = runTriage(c.input);

    const pass = output.disposition === c.expectedDisposition;
    if (pass) correct++;

    return { ...c, output, pass };
  });

  return {
    accuracy: correct / cases.length,
    results,
  };
}

3. 🔐 IMMUTABLE AUDIT TRAIL (CRITICAL)
// server/audit/auditTrail.ts

import crypto from "crypto";

let chain = [];

export function logAudit(entry) {
  const prevHash = chain.length ? chain[chain.length - 1].hash : "";

  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(entry) + prevHash)
    .digest("hex");

  const record = { ...entry, hash, prevHash };

  chain.push(record);

  return record;
}
