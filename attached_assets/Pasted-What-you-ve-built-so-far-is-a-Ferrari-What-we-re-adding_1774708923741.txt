File: server/safety/safetyGuard.ts
export function safetyGuard(result: any) {
  const issues = [];

  // 1. High-risk complaints must not be low acuity
  if (
    result.input?.complaint === "chest_pain" &&
    result.disposition === "home"
  ) {
    issues.push("Chest pain cannot be discharged home");
  }

  // 2. Low confidence block
  if (result.confidence && result.confidence < 0.4) {
    issues.push("Low confidence decision");
  }

  // 3. Missing red flag check
  if (!result.trace?.find((t: any) => t.name === "RED_FLAG_GATE")) {
    issues.push("Red flag gate missing");
  }

  return {
    allowed: issues.length === 0,
    issues,
  };
}

🔥 PART 2 — DRIFT DETECTION ENGINE
File: server/monitoring/driftDetector.ts
let history: any[] = [];

export function recordOutcome(result: any) {
  history.push({
    timestamp: Date.now(),
    correct: result.correct,
    confidence: result.confidence,
  });

  if (history.length > 500) history.shift();
}

export function detectDrift() {
  if (history.length < 50) return { drift: false };

  const recent = history.slice(-50);
  const older = history.slice(0, 50);

  const recentAcc =
    recent.filter((x) => x.correct).length / recent.length;
  const oldAcc =
    older.filter((x) => x.correct).length / older.length;

  return {
    drift: recentAcc < oldAcc - 0.1,
    recentAcc,
    oldAcc,
  };
}

🔥 PART 1 — FDA VALIDATION ENGINE
File: server/fda/validationEngine.ts
export function computeMetrics(results: any[]) {
  let TP = 0, FP = 0, FN = 0, TN = 0;

  results.forEach(r => {
    if (r.correct) TP++;
    else FP++;
  });

  const total = results.length;

  const accuracy = TP / total;
  const precision = TP / (TP + FP || 1);
  const recall = TP / (TP + FN || 1);
  const f1 = 2 * (precision * recall) / (precision + recall || 1);

  return {
    total,
    accuracy,
    precision,
    recall,
    f1
  };
}

export function stratify(results: any[]) {
  const groups = {
    pediatric: [],
    adult: [],
    highRisk: [],
    lowRisk: []
  };

  results.forEach(r => {
    if (r.input.age < 18) groups.pediatric.push(r);
    else groups.adult.push(r);

    if (r.safety === "HIGH") groups.highRisk.push(r);
    else groups.lowRisk.push(r);
  });

  return groups;
}
🔥 PART 3 — VALIDATION RUNNER
File: server/fda/validationRunner.ts
import { computeMetrics } from "./validationEngine";
import { stratify } from "./stratified";

export async function runValidation(cases: any[]) {
  const results = [];

  for (const c of cases) {
    const output = await runFullClinicalFlow(c.input);

    results.push({
      input: c.input,
      expected: c.expected,
      actual: output,
      correct: c.expected.diagnosis === output.diagnosis,
      safety: output.safetyLevel || "UNKNOWN"
    });
  }

  const overall = computeMetrics(results);
  const groups = stratify(results);

  const groupMetrics = {};
  for (const key in groups) {
    groupMetrics[key] = computeMetrics(groups[key]);
  }

  return {
    overall,
    groupMetrics,
    totalCases: results.length
  };
}
