📁 Case Persistence (Postgres + Redis cache)
import { pg } from "./postgres";
import { redis } from "../state/redisClient";

export async function saveCase(caseData: any) {
  await pg.query(
    `INSERT INTO cases (case_id, complaint, diagnosis, risk_score)
     VALUES ($1,$2,$3,$4)
     ON CONFLICT (case_id) DO UPDATE SET diagnosis=$3`,
    [
      caseData.caseId,
      caseData.complaint,
      caseData.diagnosis,
      caseData.riskScore
    ]
  );

  await redis.set(`case:${caseData.caseId}`, JSON.stringify(caseData));
}
📁 Read-through Cache
export async function getCase(caseId: string) {
  const cached = await redis.get(`case:${caseId}`);
  if (cached) return JSON.parse(cached);

  const res = await pg.query(
    `SELECT * FROM cases WHERE case_id=$1`,
    [caseId]
  );

  if (!res.rows.length) return null;

  await redis.set(`case:${caseId}`, JSON.stringify(res.rows[0]));

  return res.rows[0];
}

📊 PART 3 — LIVE MONITORING + ANOMALY DETECTION

This is your system nervous system

📁 server/monitoring/metricsCollector.ts
const metrics = {
  latency: [] as number[],
  errors: 0,
  requests: 0
};

export function recordRequest(latency: number, error = false) {
  metrics.requests++;
  metrics.latency.push(latency);

  if (error) metrics.errors++;
}

export function getMetrics() {
  const avgLatency =
    metrics.latency.reduce((a, b) => a + b, 0) /
    (metrics.latency.length || 1);

  return {
    avgLatency,
    errorRate: metrics.errors / (metrics.requests || 1),
    total: metrics.requests
  };
}
📁 server/monitoring/anomalyDetector.ts
export function detectAnomaly(metrics: any) {
  const anomalies = [];

  if (metrics.avgLatency > 2000) {
    anomalies.push("HIGH_LATENCY");
  }

  if (metrics.errorRate > 0.05) {
    anomalies.push("HIGH_ERROR_RATE");
  }

  return anomalies;
}
📁 Alert Engine
import { sendPhysicianAlert } from "../alerts/physicianAlertService";

export async function handleAnomalies() {
  const metrics = getMetrics();
  const anomalies = detectAnomaly(metrics);

  if (anomalies.length) {
    await sendPhysicianAlert({
      caseId: "system",
      priority: "HIGH",
      reason: anomalies.join(", ")
    });
  }
}
