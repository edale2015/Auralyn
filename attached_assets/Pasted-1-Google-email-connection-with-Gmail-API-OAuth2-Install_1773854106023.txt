import { db } from "../db";
import { googleEmailConnections } from "../db/googleEmailSchema";

export async function upsertGoogleEmailConnection(payload: {
  clinicId: string;
  userId: string;
  emailAddress: string;
  refreshToken: string;
}) {
  const existing = await db.query.googleEmailConnections.findFirst({
    where: (t, { and, eq }) =>
      and(eq(t.clinicId, payload.clinicId), eq(t.userId, payload.userId))
  });

  if (existing) {
    const [updated] = await db
      .update(googleEmailConnections)
      .set({
        emailAddress: payload.emailAddress,
        refreshToken: payload.refreshToken
      })
      .where((t, { eq }) => eq(t.id, existing.id))
      .returning();

    return updated;
  }

  const [created] = await db.insert(googleEmailConnections).values(payload).returning();
  return created;
}

export async function getGoogleEmailConnection(clinicId: string, userId: string) {
  return db.query.googleEmailConnections.findFirst({
    where: (t, { and, eq }) =>
      and(eq(t.clinicId, clinicId), eq(t.userId, userId))
  });
}
/server/services/googleEmail.ts
import { google } from "googleapis";
import { getGoogleEmailConnection } from "./googleEmailStore";

const CLIENT_ID = process.env.GOOGLE_CLIENT_ID!;
const CLIENT_SECRET = process.env.GOOGLE_CLIENT_SECRET!;
const REDIRECT_URI = process.env.GOOGLE_REDIRECT_URI!;
const SCOPES = (process.env.GOOGLE_EMAIL_SCOPES || "https://www.googleapis.com/auth/gmail.send").split(",");

export function createGoogleOAuthClient() {
  return new google.auth.OAuth2(CLIENT_ID, CLIENT_SECRET, REDIRECT_URI);
}

export function buildGoogleEmailAuthUrl(state: string) {
  const client = createGoogleOAuthClient();
  return client.generateAuthUrl({
    access_type: "offline",
    prompt: "consent",
    scope: SCOPES,
    state
  });
}

export async function exchangeGoogleEmailCode(code: string) {
  const client = createGoogleOAuthClient();
  const { tokens } = await client.getToken(code);
  return tokens;
}

function encodeMessage(to: string, subject: string, body: string, from: string) {
  const message = [
    `From: ${from}`,
    `To: ${to}`,
    "Content-Type: text/plain; charset=utf-8",
    "MIME-Version: 1.0",
    `Subject: ${subject}`,
    "",
    body
  ].join("\n");

  return Buffer.from(message)
    .toString("base64")
    .replace(/\+/g, "-")
    .replace(/\//g, "_")
    .replace(/=+$/g, "");
}

export async function sendGoogleEmail(payload: {
  clinicId: string;
  userId: string;
  to: string;
  subject: string;
  body: string;
}) {
  const connection = await getGoogleEmailConnection(payload.clinicId, payload.userId);

  if (!connection) {
    throw new Error("No Google email connection found");
  }

  const client = createGoogleOAuthClient();
  client.setCredentials({
    refresh_token: connection.refreshToken
  });

  const gmail = google.gmail({ version: "v1", auth: client });

  const raw = encodeMessage(
    payload.to,
    payload.subject,
    payload.body,
    connection.emailAddress
  );

  const result = await gmail.users.messages.send({
    userId: "me",
    requestBody: { raw }
  });

  return result.data;
}
/server/routes/googleEmail.ts
import express from "express";
import crypto from "crypto";
import { authContext } from "../middleware/authContext";
import { buildGoogleEmailAuthUrl, exchangeGoogleEmailCode, sendGoogleEmail } from "../services/googleEmail";
import { upsertGoogleEmailConnection } from "../services/googleEmailStore";

const router = express.Router();

function signState(input: { clinicId: string; userId: string }) {
  const payload = JSON.stringify(input);
  const sig = crypto
    .createHmac("sha256", process.env.GOOGLE_EMAIL_STATE_SECRET || "dev-state")
    .update(payload)
    .digest("hex");

  return Buffer.from(JSON.stringify({ payload, sig })).toString("base64url");
}

function verifyState(state: string) {
  const decoded = JSON.parse(Buffer.from(state, "base64url").toString("utf8"));
  const expected = crypto
    .createHmac("sha256", process.env.GOOGLE_EMAIL_STATE_SECRET || "dev-state")
    .update(decoded.payload)
    .digest("hex");

  if (expected !== decoded.sig) {
    throw new Error("Invalid state signature");
  }

  return JSON.parse(decoded.payload) as { clinicId: string; userId: string };
}

router.get("/connect", authContext, async (_req, res) => {
  const user = res.locals.user;
  const state = signState({ clinicId: user.clinicId, userId: user.id });
  const url = buildGoogleEmailAuthUrl(state);
  res.json({ url });
});

router.get("/oauth/callback", async (req, res) => {
  const code = String(req.query.code || "");
  const state = String(req.query.state || "");

  if (!code || !state) {
    return res.status(400).send("Missing code or state");
  }

  const { clinicId, userId } = verifyState(state);
  const tokens = await exchangeGoogleEmailCode(code);

  if (!tokens.refresh_token) {
    return res.status(400).send("No refresh token returned. Re-consent may be required.");
  }

  await upsertGoogleEmailConnection({
    clinicId,
    userId,
    emailAddress: process.env.GOOGLE_EMAIL_FROM || "",
    refreshToken: tokens.refresh_token
  });

  res.send("Google email connected successfully.");
});

router.post("/send", authContext, async (req, res) => {
  const user = res.locals.user;

  const data = await sendGoogleEmail({
    clinicId: user.clinicId,
    userId: user.id,
    to: req.body.to,
    subject: req.body.subject,
    body: req.body.body
  });

  res.json({ success: true, data });
});

export default router;

Google’s Gmail API docs describe sending with messages.send, and Google’s web-server OAuth flow is the right pattern when the server needs offline access using a refresh token.

Also, Gmail and Workspace have sending limits, and higher-volume senders to Gmail accounts must meet sender authentication and related requirements, so keep weekly summaries and board packets low-volume and authenticated.

import { db } from "../db";
import { sharedDashboardViews } from "../db/sharedViewsSchema";

export async function createSharedView(payload: {
  clinicId: string;
  createdByUserId: string;
  name: string;
  viewType: string;
  filters: unknown;
}) {
  const [row] = await db.insert(sharedDashboardViews).values(payload).returning();
  return row;
}

export async function listSharedViews(clinicId: string, approvedOnly = false) {
  if (approvedOnly) {
    return db.query.sharedDashboardViews.findMany({
      where: (t, { and, eq }) => and(eq(t.clinicId, clinicId), eq(t.isApproved, true)),
      orderBy: (t, { desc }) => [desc(t.createdAt)]
    });
  }

  return db.query.sharedDashboardViews.findMany({
    where: (t, { eq }) => eq(t.clinicId, clinicId),
    orderBy: (t, { desc }) => [desc(t.createdAt)]
  });
}

export async function approveSharedView(id: number, approvedByUserId: string) {
  const [row] = await db
    .update(sharedDashboardViews)
    .set({
      isApproved: true,
      approvedByUserId,
      approvedAt: new Date()
    })
    .where((t, { eq }) => eq(t.id, id))
    .returning();

  return row;
}

3) Signed board exports
/server/services/signedBoardExports.ts
import crypto from "crypto";

export function canonicalJson(value: unknown) {
  return JSON.stringify(value, Object.keys(value as Record<string, unknown>).sort());
}

export function signBoardPayload(payload: unknown) {
  const serialized = canonicalJson(payload);
  const signature = crypto
    .createHmac("sha256", process.env.BOARD_EXPORT_SIGNING_SECRET || "dev-board-secret")
    .update(serialized)
    .digest("hex");

  return {
    payload,
    signature,
    algorithm: "HMAC-SHA256",
    signedAt: new Date().toISOString()
  };
}

export function verifyBoardSignature(payload: unknown, signature: string) {
  const expected = crypto
    .createHmac("sha256", process.env.BOARD_EXPORT_SIGNING_SECRET || "dev-board-secret")
    .update(canonicalJson(payload))
    .digest("hex");

  return expected === signature;
}

5) Server-side pagination
/server/services/pagination.ts
export type PaginationParams = {
  page?: number;
  pageSize?: number;
};

export function paginate<T>(rows: T[], params: PaginationParams) {
  const page = Math.max(1, params.page || 1);
  const pageSize = Math.min(100, Math.max(1, params.pageSize || 20));
  const start = (page - 1) * pageSize;
  const end = start + pageSize;

  return {
    page,
    pageSize,
    total: rows.length,
    totalPages: Math.ceil(rows.length / pageSize),
    items: rows.slice(start, end)
  };
}

import { useEffect, useState } from "react";

type Props = {
  token: string;
};

export default function SharedViewsApprovalDashboard({ token }: Props) {
  const [rows, setRows] = useState<any[]>([]);

  const load = async () => {
    const res = await fetch("/api/shared-views", {
      headers: { Authorization: `Bearer ${token}` }
    });
    setRows(await res.json());
  };

  useEffect(() => {
    load();
  }, []);

  const approve = async (id: number) => {
    await fetch(`/api/shared-views/${id}/approve`, {
      method: "POST",
      headers: { Authorization: `Bearer ${token}` }
    });
    await load();
  };

  return (
    <div className="p-6 space-y-4">
      <h1 className="text-2xl font-bold">Shared Views Approval</h1>
      {rows.map(row => (
        <div key={row.id} className="border rounded p-4">
          <div className="font-semibold">{row.name}</div>
          <div>Type: {row.viewType}</div>
          <div>Approved: {String(row.isApproved)}</div>
          {!row.isApproved && (
            <button className="border rounded px-3 py-1 mt-2" onClick={() => approve(row.id)}>
              Approve
            </button>
          )}
        </div>
      ))}
    </div>
  );
}
