import { Pool, QueryResult } from "pg";
import { ENV } from "../config/env";

if (!ENV.DATABASE_URL) {
  throw new Error("❌ DATABASE_URL is required");
}

export const pool = new Pool({
  connectionString: ENV.DATABASE_URL,
  ssl: ENV.NODE_ENV === "production" ? { rejectUnauthorized: false } : false,
  max: 20,
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 10000
});

export async function query<T = any>(text: string, params: any[] = []): Promise<QueryResult<T>> {
  return pool.query<T>(text, params);
}

export async function withTransaction<T>(fn: (client: any) => Promise<T>): Promise<T> {
  const client = await pool.connect();
  try {
    await client.query("BEGIN");
    const result = await fn(client);
    await client.query("COMMIT");
    return result;
  } catch (err) {
    await client.query("ROLLBACK");
    throw err;
  } finally {
    client.release();
  }
}

export async function testDbConnection() {
  const result = await query("SELECT 1 as ok");
  return result.rows[0];
}

export async function createRequestRecord(input: {
  clinicId?: string;
  idempotencyKey?: string;
  route: string;
  method: string;
  status: string;
  response?: unknown;
}) {
  const result = await query(
    `INSERT INTO requests (clinic_id, idempotency_key, route, method, status, response)
     VALUES ($1, $2, $3, $4, $5, $6)
     RETURNING *`,
    [
      input.clinicId ?? null,
      input.idempotencyKey ?? null,
      input.route,
      input.method,
      input.status,
      input.response ?? null
    ]
  );

  return result.rows[0];
}

  const result = await query(
    `SELECT * FROM requests WHERE idempotency_key = $1 LIMIT 1`,

  );

  return result.rows[0] ?? null;
}

export async function updateRequestResponse(idempotencyKey: string, status: string, response: unknown) {

     RETURNING *`,

  );

  return result.rows[0] ?? null;

import { Worker } from "bullmq";
import { getRedis } from "../queue/redis";
import { upsertJobRecord } from "../repos/jobRepo";
import { writeAuditEvent } from "../repos/auditRepo";

const connection = getRedis();

export function createTriageWorker() {
  const worker = new Worker(
    "triage",
    async (job) => {
      await upsertJobRecord({
        id: job.id!,
        clinicId: job.data?.clinicId,
        queueName: "triage",
        jobName: job.name,
        status: "processing",
        payload: job.data
      });

      const result = {
        disposition: "review",
        confidence: 0.9
      };

      await upsertJobRecord({
        id: job.id!,
        clinicId: job.data?.clinicId,
        queueName: "triage",
        jobName: job.name,
        status: "completed",
        payload: job.data,
        result
      });

      await writeAuditEvent({
        clinicId: job.data?.clinicId,
        traceId: job.data?.traceId,
        eventType: "triage.job.completed",
        entityType: "job",
        entityId: job.id!,
        data: result
      });

      return result;
    },
    { connection }
  );

  return worker;
}

import type { Worker } from "bullmq";

import { writeAuditEvent } from "../repos/auditRepo";
import { deadLetterQueue } from "../queue/queues";

export function attachWorkerEvents(worker: Worker, queueName: string) {
  worker.on("completed", async (job, result) => {
    if (!job) return;

    await upsertJobRecord({
      id: job.id!,
      clinicId: job.data?.clinicId,
      queueName,
      jobName: job.name,
      status: "completed",
      payload: job.data,
      result
    });

    await writeAuditEvent({
      clinicId: job.data?.clinicId,
      traceId: job.data?.traceId,
      eventType: `${queueName}.job.completed`,
      entityType: "job",
      entityId: job.id!,
      data: result
    });
  });

  worker.on("failed", async (job, err) => {
    if (!job) return;

    await upsertJobRecord({
      id: job.id!,
      clinicId: job.data?.clinicId,
      queueName,
      jobName: job.name,
      status: "failed",
      payload: job.data,
      error: err?.message || "Unknown worker failure"
    });

    await writeAuditEvent({
      clinicId: job.data?.clinicId,
      traceId: job.data?.traceId,
      eventType: `${queueName}.job.failed`,
      entityType: "job",
      entityId: job.id!,
      data: { error: err?.message || "Unknown worker failure" }
    });

    await deadLetterQueue.add(
      `${queueName}-dead-letter`,
      {
        sourceQueue: queueName,
        originalJobId: job.id,
        jobName: job.name,
        clinicId: job.data?.clinicId,
        traceId: job.data?.traceId,
        payload: job.data,
        error: err?.message || "Unknown worker failure"
      },
      {
        removeOnComplete: 1000,
        removeOnFail: 1000
      }
    );
  });
}

import { query } from "../db";

export async function appendSystemEvent(input: {
  clinicId?: string;
  traceId?: string;
  eventName: string;
  severity?: "info" | "warn" | "error";
  source?: string;
  payload?: unknown;
}) {
  const result = await query(
    `INSERT INTO system_events (clinic_id, trace_id, event_name, severity, source, payload)
     VALUES ($1, $2, $3, $4, $5, $6)
     RETURNING *`,
    [
      input.clinicId ?? null,
      input.traceId ?? null,
      input.eventName,
      input.severity ?? "info",
      input.source ?? null,
      input.payload ?? null
    ]
  );

  return result.rows[0];
}

export async function listSystemEvents(limit = 200, severity?: string) {
  const result = severity
    ? await query(
        `SELECT * FROM system_events WHERE severity = $1 ORDER BY created_at DESC LIMIT $2`,
        [severity, limit]
      )
    : await query(
        `SELECT * FROM system_events ORDER BY created_at DESC LIMIT $1`,
        [limit]
      );

  return result.rows;
}

type EnvMode = "development" | "test" | "production";

function getOptional(key: string): string | undefined {
  const value = process.env[key];
  return value && value.trim() !== "" ? value.trim() : undefined;
}

function getRequired(key: string): string {
  const value = getOptional(key);
  if (!value) {
    throw new Error(`Missing environment variable: ${key}`);
  }
  return value;
}

function getBoolean(key: string, fallback = false): boolean {
  const value = getOptional(key);
  if (!value) return fallback;
  return value === "true";
}

function getNumber(key: string, fallback?: number): number {
  const value = getOptional(key);
  if (!value) {
    if (fallback !== undefined) return fallback;
    throw new Error(`Missing numeric environment variable: ${key}`);
  }

  const num = Number(value);
  if (Number.isNaN(num)) {
    throw new Error(`Invalid numeric environment variable: ${key}`);
  }

  return num;
}

export function validateConfig() {
  const errors: string[] = [];

  if (!["development", "test", "production"].includes(ENV.NODE_ENV)) {
    errors.push(`NODE_ENV invalid: ${ENV.NODE_ENV}`);
  }

  if (!Number.isInteger(ENV.PORT) || ENV.PORT <= 0) {
    errors.push("PORT must be a positive integer");
  }

  if (!ENV.DATABASE_URL) {
    errors.push("DATABASE_URL is required");
  }

  if (!ENV.REDIS_URL) {
    errors.push("REDIS_URL is required");
  }

  if (ENV.NODE_ENV === "production") {
    if (!ENV.JWT_SECRET) errors.push("JWT_SECRET is required in production");
    if (!ENV.SESSION_SECRET) errors.push("SESSION_SECRET is required in production");
    if (!ENV.MD_PASSWORD) errors.push("MD_PASSWORD is required in production");
    if (!ENV.CLINICIAN_PASSWORD) errors.push("CLINICIAN_PASSWORD is required in production");
  }

  if (errors.length > 0) {
    throw new Error(`❌ Config validation failed:\n- ${errors.join("\n- ")}`);

import { NodeSDK } from "@opentelemetry/sdk-node";
import { getNodeAutoInstrumentations } from "@opentelemetry/auto-instrumentations-node";
import { OTLPTraceExporter } from "@opentelemetry/exporter-trace-otlp-http";
import { resourceFromAttributes } from "@opentelemetry/resources";
import { SemanticResourceAttributes } from "@opentelemetry/semantic-conventions";

let sdk: NodeSDK | null = null;
let started = false;

export async function startTelemetry(serviceName: string) {
  if (started) return;

  const exporter = new OTLPTraceExporter({
    url: process.env.OTEL_EXPORTER_OTLP_TRACES_ENDPOINT || process.env.OTEL_EXPORTER_OTLP_ENDPOINT
  });

  sdk = new NodeSDK({
    resource: resourceFromAttributes({
      [SemanticResourceAttributes.SERVICE_NAME]: serviceName,
      [SemanticResourceAttributes.DEPLOYMENT_ENVIRONMENT]:
        process.env.NODE_ENV || "development"
    }),
    traceExporter: exporter,
    instrumentations: [getNodeAutoInstrumentations()]
  });

  await sdk.start();
  started = true;
  console.log(`✅ OpenTelemetry started for ${serviceName}`);
}

export async function stopTelemetry() {
  if (!sdk) return;
  await sdk.shutdown();
}

import { v4 as uuid } from "uuid";
import { context, trace } from "@opentelemetry/api";
import { runWithRequestContext } from "../monitoring/requestContext";

export function traceMiddleware(req: any, res: any, next: any) {
  const incoming = req.headers["x-trace-id"];
  const activeSpan = trace.getSpan(context.active());

  const otelTraceId = activeSpan?.spanContext().traceId;
  const traceId =
    (typeof incoming === "string" && incoming.trim()) ||
    otelTraceId ||
    uuid();

  req.traceId = traceId;
  res.setHeader("x-trace-id", traceId);

  runWithRequestContext(
    {
      traceId,
      clinicId: req.clinic?.id,
      userId: req.user?.id
    },
    () => next()
  );
}

import { getRequestContext } from "../monitoring/requestContext";

type LogLevel = "info" | "warn" | "error";

function redactPhone(value?: string) {
  if (!value) return undefined;
  const digits = value.replace(/\D/g, "");
  if (digits.length < 4) return "[REDACTED_PHONE]";
  return `***${digits.slice(-4)}`;
}

function redactText(value?: string) {
  return value ? "[REDACTED_TEXT]" : undefined;
}

function sanitize(data: Record<string, unknown>) {
  const out: Record<string, unknown> = {};

  for (const [key, value] of Object.entries(data)) {
    if (value == null) {
      out[key] = value;
      continue;
    }

    const k = key.toLowerCase();

    if (k.includes("phone") || k === "to" || k === "from") {
      out[key] = redactPhone(String(value));
    } else if (
      k.includes("message") ||
      k.includes("body") ||
      k.includes("text") ||
      k.includes("content") ||
      k.includes("note")
    ) {
      out[key] = redactText(String(value));
    } else if (
      k.includes("patient") ||
      k.includes("payload") ||
      k.includes("request")
    ) {
      out[key] = "[REDACTED_OBJECT]";
    } else {
      out[key] = value;
    }
  }

  return out;
}

function write(level: LogLevel, msg: string, data: Record<string, unknown> = {}) {
  const ctx = getRequestContext();

  const entry = {
    ts: new Date().toISOString(),
    level,
    msg,
    traceId: ctx?.traceId,
    clinicId: ctx?.clinicId,
    userId: ctx?.userId,
    ...sanitize(data)
  };

  const line = JSON.stringify(entry);

  if (level === "error") {
    console.error(line);
  } else {
    console.log(line);
  }
}

export const logger = {
  info: (msg: string, data?: Record<string, unknown>) => write("info", msg, data),
  warn: (msg: string, data?: Record<string, unknown>) => write("warn", msg, data),
  error: (msg: string, data?: Record<string, unknown>) => write("error", msg, data)
};

    checks.queueInfra.ok = pong === "PONG";
    if (pong !== "PONG") {
      checks.redis.error = "Redis ping failed";
      checks.queueInfra.error = "Queue infra unavailable";
    }
  } catch (err: any) {
    const message = err?.message || "Redis failure";
    checks.redis.error = message;
    checks.queueInfra.error = message;
  }

  checks.telemetry.ok = Boolean(
    process.env.OTEL_EXPORTER_OTLP_ENDPOINT ||
      process.env.OTEL_EXPORTER_OTLP_TRACES_ENDPOINT
  );
  if (!checks.telemetry.ok) {
    checks.telemetry.note = "No OTLP exporter configured";
  }

  checks.twilio.ok = Boolean(ENV.TWILIO_ACCOUNT_SID && ENV.TWILIO_AUTH_TOKEN);
  if (!checks.twilio.ok) {
    checks.twilio.note = "Twilio credentials missing";
  }

  res.json(checks);
});

export async function withEngineMetrics<T>(
  engineName: string,
  clinicId: string | undefined,
  fn: () => Promise<T>
): Promise<T> {
  const start = Date.now();

  try {
    const result = await fn();
    await recordEngineSuccess({
      clinicId,
      engineName,
      latencyMs: Date.now() - start
    });
    return result;
  } catch (err: any) {
    await recordEngineError({
      clinicId,
      engineName,
      latencyMs: Date.now() - start,
      error: err?.message || "Unknown engine error"
    });
    throw err;
  }
}

      }
