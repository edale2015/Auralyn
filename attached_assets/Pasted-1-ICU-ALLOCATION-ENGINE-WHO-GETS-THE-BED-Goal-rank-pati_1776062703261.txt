1. ICU ALLOCATION ENGINE (WHO GETS THE BED?)

Goal: rank patients by need + trajectory + benefit of ICU care

Scoring Logic
Acute severity (NEWS2, qSOFA)
Trajectory (worsening vs stable)
Reversibility (who benefits from ICU)
Resource pressure (beds available)
Code
// server/icu/icuAllocator.ts

type ICUScore = {
  patientId: string;
  score: number;
  reasons: string[];
};

export function computeICUPriority(patient: any): ICUScore {
  let score = 0;
  const reasons: string[] = [];

  // Severity
  if (patient.sepsis.news2 >= 7) {
    score += 5;
    reasons.push("High NEWS2");
  }

  if (patient.sepsis.qsofa >= 2) {
    score += 4;
    reasons.push("qSOFA ≥ 2");
  }

  // Trajectory
  if (patient.trends?.worseningVitals) {
    score += 3;
    reasons.push("Worsening vitals");
  }

  if (patient.labs?.lactateTrendUp) {
    score += 3;
    reasons.push("Rising lactate");
  }

  // Reversibility heuristic
  if (patient.age < 75 && !patient.dnr) {
    score += 2;
    reasons.push("High reversibility");
  }

  return {
    patientId: patient.id,
    score,
    reasons,
  };
}

export function rankICUPatients(patients: any[]) {
  return patients
    .map(computeICUPriority)
    .sort((a, b) => b.score - a.score);
}

// server/simulation/digitalTwin.ts

export function simulatePatient(patient: any, intervention: any) {
  const clone = JSON.parse(JSON.stringify(patient));

  for (let hour = 0; hour < 12; hour++) {
    // Apply intervention effects
    if (intervention.type === "fluids") {
      clone.vitals.sbp += 2;
      clone.labs.lactate -= 0.2;
    }

    if (intervention.type === "antibiotics") {
      clone.infectionLoad = Math.max(0, clone.infectionLoad - 0.1);
    }

    // Natural deterioration
    if (clone.infectionLoad > 0.5) {
      clone.vitals.hr += 2;
      clone.vitals.sbp -= 1;
    }
  }

  return clone;
}
Multi-scenario simulation
// server/simulation/scenarioRunner.ts

import { simulatePatient } from "./digitalTwin";
import { calculateSepsisRisk } from "../clinical/sepsisPredictor";

export function runScenarios(patient: any) {
  const scenarios = [
    { name: "No Treatment", intervention: {} },
    { name: "Fluids", intervention: { type: "fluids" } },
    { name: "Antibiotics", intervention: { type: "antibiotics" } },
    {
      name: "Fluids + Antibiotics",
      intervention: { type: "combo" },
    },
  ];

  return scenarios.map((s) => {
    const simulated = simulatePatient(patient, s.intervention);
    const risk = calculateSepsisRisk(simulated);

    return {
      scenario: s.name,
      resultingRisk: risk.risk,
      vitals: simulated.vitals,
    };
  });
}

// server/intervention/interventionEngine.ts

export function generateInterventions(evaluation: any) {
  const actions = [];

  if (evaluation.sepsis.risk === "HIGH") {
    actions.push({
      type: "IV_FLUIDS",
      dose: "30 mL/kg",
      priority: "STAT",
    });

    actions.push({
      type: "BROAD_SPECTRUM_ANTIBIOTICS",
      priority: "STAT",
    });

    actions.push({
      type: "LACTATE_REPEAT",
      timing: "2 hours",
    });
  }

  if (evaluation.council.consensus.confidence > 0.85) {
    actions.push({
      type: "ADMISSION",
      level: "ICU",
    });
  }

  return actions;
}

import crypto from "crypto";

export function logIntervention(event: any) {
  const hash = crypto
    .createHash("sha256")
    .update(JSON.stringify(event))
    .digest("hex");

  return { ...event, hash };
}
