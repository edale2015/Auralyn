import { db } from "../../db";

export async function getPatientHistory(patientId: string) {
  return db
    .select("*")
    .from("patient_history")
    .where({ patient_id: patientId })
    .orderBy("timestamp DESC")
    .limit(20);
}

export function extractPatientPatterns(history: any[]) {
  let antibioticResponseRate = 0;
  let visits = history.length;

  if (visits === 0) return { antibioticResponseRate: 0 };

  const improvedWithAbx = history.filter(h => h.improved_with_antibiotics).length;

  antibioticResponseRate = improvedWithAbx / visits;

  return {
    antibioticResponseRate,
    frequentReturner: visits >= 3
  };
}
🌍 2) POPULATION LEARNING ENGINE
/server/services/learning/populationLearningEngine.ts
interface PopulationStats {
  clinicId: string;
  antibioticSuccessRate: number;
  returnVisitRate: number;
}

const thresholds: Record<string, number> = {};

export async function updatePopulationStats(stats: PopulationStats) {
  const adjustment =
    stats.antibioticSuccessRate < 0.3 ? -0.1 :
    stats.returnVisitRate > 0.2 ? 0.1 :
    0;

  thresholds[stats.clinicId] =
    (thresholds[stats.clinicId] || 0.5) + adjustment;

  thresholds[stats.clinicId] = Math.max(0.3, Math.min(0.7, thresholds[stats.clinicId]));
}

export function getClinicThreshold(clinicId: string): number {
  return thresholds[clinicId] || 0.5;
}
🧬 3) PERSONALIZATION ENGINE
/server/services/learning/personalizationEngine.ts
export interface PersonalizationInput {
  baseProbability: number;
  comorbidities: string[];
  patientPattern: {
    antibioticResponseRate: number;
    frequentReturner: boolean;
  };
}

export function personalizeDecision(input: PersonalizationInput): number {
  let adjusted = input.baseProbability;

  // 🧠 Comorbidity adjustments
  if (input.comorbidities.includes("immunocompromised")) {
    adjusted += 0.2;
  }

  if (input.comorbidities.includes("chronic_lung_disease")) {
    adjusted += 0.1;
  }

  // 🔁 Patient history influence
  if (input.patientPattern.antibioticResponseRate > 0.7) {
    adjusted += 0.1;
  }

  if (input.patientPattern.frequentReturner) {
    adjusted += 0.05;
  }

  return Math.min(adjusted, 0.95);
}

🌐 5) FULL LEARNING ROUTE
/server/routes/learningRoutes.ts
import express from "express";
import { getPatientHistory, extractPatientPatterns } from "../services/learning/patientMemoryService";
import { personalizeDecision } from "../services/learning/personalizationEngine";
import { getClinicThreshold } from "../services/learning/populationLearningEngine";

const router = express.Router();

router.post("/enhanced-decision", async (req, res) => {
  try {
    const { patientId, clinicId, baseProbability, comorbidities } = req.body;

    // 🧠 Patient memory
    const history = await getPatientHistory(patientId);
    const pattern = extractPatientPatterns(history);

    // 🌍 Clinic learning
    const threshold = getClinicThreshold(clinicId);

    // 🧬 Personalization
    const adjustedProbability = personalizeDecision({
      baseProbability,
      comorbidities,
      patientPattern: pattern
    });

    const decision =
      adjustedProbability > threshold
        ? "CONSIDER_ANTIBIOTIC"
        : "NO_ANTIBIOTIC";

    res.json({
      adjustedProbability,
      threshold,
      decision,
      pattern
    });

  } catch (err) {
    res.status(500).json({ error: "Learning pipeline failed" });
  }
});

export default router;
