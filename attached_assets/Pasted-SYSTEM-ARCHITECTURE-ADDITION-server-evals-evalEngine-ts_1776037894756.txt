⚙️ 1. EVAL ENGINE (Core System)
server/evals/evalEngine.ts
import { runEvalCase } from "./evalRunner";
import { compareOutputs } from "./comparator";

export interface EvalCase {
  id: string;
  input: any;
  expected: any;
  tags?: string[];
}

export interface EvalResult {
  id: string;
  passed: boolean;
  score: number;
  withSkill: any;
  withoutSkill: any;
  diff: any;
}

export async function runEvalSuite(
  skillName: string,
  cases: EvalCase[]
): Promise<EvalResult[]> {
  const results: EvalResult[] = [];

  await Promise.all(
    cases.map(async (testCase) => {
      const [withSkill, withoutSkill] = await Promise.all([
        runEvalCase(testCase, true, skillName),
        runEvalCase(testCase, false, skillName),
      ]);

      const comparison = compareOutputs(
        testCase.expected,
        withSkill,
        withoutSkill
      );

      results.push({
        id: testCase.id,
        passed: comparison.passed,
        score: comparison.score,
        withSkill,
        withoutSkill,
        diff: comparison.diff,
      });
    })
  );

  return results;
}

⚖️ 3. BLIND COMPARATOR (Prevents bias)
server/evals/comparator.ts
export function compareOutputs(expected: any, a: any, b: any) {
  const scoreA = scoreOutput(expected, a);
  const scoreB = scoreOutput(expected, b);

  return {
    passed: scoreA >= 0.9,
    score: Math.max(scoreA, scoreB),
    diff: {
      expected,
      withSkill: a,
      withoutSkill: b,
    },
  };
}

function scoreOutput(expected: any, actual: any): number {
  let score = 0;

  if (actual.diagnosis === expected.diagnosis) score += 0.4;
  if (actual.disposition === expected.disposition) score += 0.4;

  if (
    JSON.stringify(actual.orders || []) ===
    JSON.stringify(expected.orders || [])
  ) {
    score += 0.2;
  }

  return score;
}
🚨 4. REGRESSION MONITOR (THIS IS THE BIG ONE)
server/evals/regressionMonitor.ts
import { runEvalSuite } from "./evalEngine";
import { sendAlert } from "../utils/alerts";

export async function runRegressionCheck(skillName: string, cases: any[]) {
  const results = await runEvalSuite(skillName, cases);

  const passRate =
    results.filter((r) => r.passed).length / results.length;

  if (passRate < 0.95) {
    await sendAlert({
      type: "CLINICAL_REGRESSION",
      skill: skillName,
      passRate,
      message: `⚠️ Clinical skill regression detected: ${passRate}`,
    });
  }

  return { passRate, results };
}

🧠 5. TRIGGER OPTIMIZER (Improve Mode)
server/evals/triggerOptimizer.ts
export async function optimizeTriggerDescription(
  skillName: string,
  queries: string[],
  executor: (query: string) => Promise<boolean>
) {
  let successRate = 0;

  for (const q of queries) {
    const triggered = await executor(q);
    if (triggered) successRate++;
  }

  successRate = successRate / queries.length;

  if (successRate < 0.7) {
    return {
      recommendation:
        "Expand trigger description to include common clinical phrasing (e.g. 'review patient', 'evaluate symptoms')",
      successRate,
    };
  }

  return { successRate };
}

import express from "express";
import { runEvalSuite } from "../evals/evalEngine";
import { runRegressionCheck } from "../evals/regressionMonitor";

const router = express.Router();

router.post("/run", async (req, res) => {
  const { skillName, cases } = req.body;

  const results = await runEvalSuite(skillName, cases);

  res.json(results);
});

router.post("/regression", async (req, res) => {
  const { skillName, cases } = req.body;

  const result = await runRegressionCheck(skillName, cases);

  res.json(result);
});

export default router;
