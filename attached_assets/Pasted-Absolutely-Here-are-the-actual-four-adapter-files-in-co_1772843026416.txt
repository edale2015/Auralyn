export function normalizeChatAnswer(raw: string): string | number | boolean {
  const s = String(raw ?? "").trim();
  const lower = s.toLowerCase();

  if (["yes", "y", "yeah", "yep", "true"].includes(lower)) return true;

import { loadComplaintConfig } from "./complaintConfigLoader";
import { evaluateExpr } from "./exprEval";

type ComplaintQuestionRow = {
  Q_ID?: string;
  QUESTION_TEXT?: string;
  ASK_ORDER?: string | number;
  ASK_IF?: string;
  REQUIRED?: string | boolean;
};

export interface PlannedQuestion {
  token?: string;
  text?: string;
  completed: boolean;
}

function tokenFromQid(qid?: string): string {
  if (!qid) return "";
  const parts = qid.split("_");
  if (parts.length >= 3) return parts.slice(2).join("_");
  return qid;
}

function isAnswered(value: unknown): boolean {
  return value !== undefined && value !== null && String(value).trim() !== "";
}

function toBool(v: unknown): boolean {
  if (typeof v === "boolean") return v;
  const s = String(v ?? "").trim().toLowerCase();
  return s === "true" || s === "1" || s === "yes";
}

function buildExprContext(answers: Record<string, unknown>): Record<string, unknown> {
  return {
    answers,
    ...answers
  };
}

export async function planNextQuestion(
  complaintId: string,
  answers: Record<string, unknown>,
  unansweredCriticalQuestions: string[] = []
): Promise<PlannedQuestion> {
  if (unansweredCriticalQuestions.length > 0) {
    const token = unansweredCriticalQuestions[0];
    return {
      token,
      text: `Please answer: ${token.toLowerCase().replace(/_/g, " ")}?`,
      completed: false
    };
  }

  const cfg = await loadComplaintConfig(complaintId);
  const questions: ComplaintQuestionRow[] = [...(cfg.questions ?? [])].sort(
    (a: ComplaintQuestionRow, b: ComplaintQuestionRow) =>
      Number(a.ASK_ORDER ?? 0) - Number(b.ASK_ORDER ?? 0)
  );

  const exprContext = buildExprContext(answers);

  for (const q of questions) {
    const token = tokenFromQid(q.Q_ID);
    if (!token) continue;

    if (isAnswered(answers[token])) continue;

    const askIf = String(q.ASK_IF ?? "").trim();
    if (askIf) {
      try {
        const shouldAsk = evaluateExpr(askIf, exprContext);
        if (!shouldAsk) continue;
      } catch {
        // If ASK_IF is malformed or unsupported here, fail open and still ask.
      }
    }

    const required = q.REQUIRED === undefined ? true : toBool(q.REQUIRED);
    if (!required) {
      // In MVP mode, optional questions can still be asked,
      // but required ones naturally come first due to ordering.
    }

    return {
      token,
      text: q.QUESTION_TEXT || `Please answer: ${token.toLowerCase().replace(/_/g, " ")}?`,
      completed: false
    };
  }

  return { completed: true };
}

import type { ChatEngineRunInput, ChatEngineRunOutput } from "../types/chatEngine";
import type { CaseEngineResult, DxCandidate, ClusterScoreSnapshot, RuleTraceItem } from "../types/case";
import { loadComplaintConfig } from "./complaintConfigLoader";
import { runGenericComplaintV1 } from "../engines/genericComplaintEngineV1";
import { planNextQuestion } from "./chatQuestionPlanner";

type GenericEngineOutput = {
  disposition?: string;
  confidence?: string;
  triggeredRedFlags?: string[];
  winningClusterId?: string;
  likelyDx?: Array<{ id?: string; label?: string; dxId?: string; rank?: number }>;
  dxCandidates?: Array<{ dxId?: string; label?: string; rank?: number; bestClusterId?: string; baseScore?: number }>;
  clusterScores?: Array<{ clusterId?: string; score?: number; reasons?: string[] }>;
  ruleTrace?: Array<{
    table?: string;
    ruleId?: string;
    fired?: boolean;
    expr?: string;
    evidenceLabel?: string;
    points?: number;
    metadata?: Record<string, unknown>;
  }>;
  noteDraft?: string;
  returnPrecautions?: string[];
  render?: Record<string, unknown>;
  unansweredCriticalQuestions?: string[];
};

function normalizeDisposition(v: unknown): CaseEngineResult["recommendedDisposition"] {
  const s = String(v ?? "UNKNOWN").trim().toUpperCase();
  if (s === "ER_SEND" || s === "URGENT" || s === "PCP" || s === "SELF_CARE") return s;
  return "UNKNOWN";
}

function normalizeConfidence(v: unknown): CaseEngineResult["confidence"] {
  const s = String(v ?? "LOW").trim().toUpperCase();
  if (s === "HIGH" || s === "MODERATE" || s === "LOW") return s;
  return s;
}

function mapDxCandidates(out: GenericEngineOutput): DxCandidate[] {
  const source = out.dxCandidates?.length ? out.dxCandidates : out.likelyDx ?? [];

  return source.slice(0, 10).map((d: any, idx: number) => ({
    dxId: d.dxId || d.id || `dx_${idx + 1}`,
    label: d.label || d.dxId || d.id || `Diagnosis ${idx + 1}`,
    bestClusterId: d.bestClusterId,
    baseScore: d.baseScore,
    rank: d.rank ?? idx + 1
  }));
}

function mapClusterScores(out: GenericEngineOutput): ClusterScoreSnapshot[] {
  return (out.clusterScores ?? []).map((c: any) => ({
    clusterId: c.clusterId || "",
    score: Number(c.score ?? 0),
    reasons: c.reasons ?? []
  }));
}

function mapRuleTrace(out: GenericEngineOutput): RuleTraceItem[] {
  return (out.ruleTrace ?? []).map((r: any) => ({
    table: (r.table as RuleTraceItem["table"]) || "OTHER",
    ruleId: r.ruleId || "",
    fired: Boolean(r.fired),
    expr: r.expr,
    evidenceLabel: r.evidenceLabel,
    points: r.points !== undefined ? Number(r.points) : undefined,
    metadata: r.metadata ?? {}
  }));
}

function mapAnswersToEngineInput(answers: Record<string, unknown>) {
  /**
   * This is the single place to adapt your chat-stored answers
   * into whatever your engine expects.
   *
   * If your engine later wants:
   *   { answers, patientContext, messageHistory, modifiers }
   * patch this here without touching chat routes/pages.
   */
  return {
    answers
  };
}

function mapEngineOutputToCaseEngineResult(
  complaintId: string,
  complaintLabel: string | undefined,
  out: GenericEngineOutput
): CaseEngineResult {
  return {
    complaintId,
    complaintLabel,
    recommendedDisposition: normalizeDisposition(out.disposition),
    confidence: normalizeConfidence(out.confidence),
    triggeredRedFlags: out.triggeredRedFlags ?? [],
    winningClusterId: out.winningClusterId,
    dxCandidates: mapDxCandidates(out),
    clusterScores: mapClusterScores(out),
    ruleTrace: mapRuleTrace(out),
    noteDraft: out.noteDraft,
    returnPrecautions: out.returnPrecautions ?? [],
    render: out.render ?? {},
    engineVersion: "GENERIC_V1"
  };
}

export async function runEngineForChatAdapter(
  input: ChatEngineRunInput
): Promise<ChatEngineRunOutput> {
  const { caseRecord } = input;

  const cfg = await loadComplaintConfig(caseRecord.complaintId);

  const engineInput = mapAnswersToEngineInput(caseRecord.answers || {});

  /**
   * IMPORTANT:
   * This assumes your real engine signature is:
   *   runGenericComplaintV1(cfg, engineInput)
   *
   * If your actual signature differs, patch this one line and the input mapper above.
   */
  const rawOutput = (await runGenericComplaintV1(
    cfg as any,
    engineInput as any
  )) as GenericEngineOutput;

  const engineResult = mapEngineOutputToCaseEngineResult(
    caseRecord.complaintId,
    caseRecord.complaintLabel,
    rawOutput
  );

  const unansweredCriticalQuestions = rawOutput.unansweredCriticalQuestions ?? [];

  const next = await planNextQuestion(
    caseRecord.complaintId,
    caseRecord.answers || {},
    unansweredCriticalQuestions
  );

  return {
    engineResult,
    nextQuestionToken: next.token,
    nextQuestionText: next.text,
    unansweredCriticalQuestions,
    completed: next.completed
  };
}

import type {
  TokenAwareNormalizationInput,
  TokenAwareNormalizationResult
} from "../types/chatNormalization";

function clean(raw: string): string {
  return String(raw ?? "").trim();
}

function lower(raw: string): string {
  return clean(raw).toLowerCase();
}

function asResult(
  input: TokenAwareNormalizationInput,
  parsed: string | number | boolean | null,
  confidence: "high" | "medium" | "low",
  normalizedUnit?: string
): TokenAwareNormalizationResult {
  return {
    token: input.token,
    raw: input.raw,
    parsed,
    normalizedUnit,
    confidence
  };
}

function parseBoolean(raw: string): boolean | null {
  const s = lower(raw);

  if (["yes", "y", "yeah", "yep", "true", "correct", "affirmative"].includes(s)) return true;
  if (["no", "n", "nah", "nope", "false", "negative"].includes(s)) return false;

  return null;
}

function parsePlainNumber(raw: string): number | null {
  const s = lower(raw);
  if (/^-?\d+(\.\d+)?$/.test(s)) {
    const n = Number(s);
    if (!Number.isNaN(n)) return n;
  }
  return null;
}

function parseAgeYears(raw: string): number | null {
  const s = lower(raw);

  const direct = parsePlainNumber(s);
  if (direct !== null) return direct;

  const m1 = s.match(/(\d+(\.\d+)?)\s*(year|years|yo|y\/o|yr|yrs)\b/);
  if (m1) return Number(m1[1]);

  return null;
}

function parseSeverity(raw: string): number | string | null {
  const s = lower(raw);

  const direct = parsePlainNumber(s);
  if (direct !== null) return direct;

  if (["mild", "a little", "slight"].includes(s)) return 2;
  if (["moderate", "medium"].includes(s)) return 5;
  if (["severe", "very bad", "intense"].includes(s)) return 8;
  if (["worst", "extreme"].includes(s)) return 10;

  return null;
}

function parseDurationDays(raw: string): number | null {
  const s = lower(raw);

  const direct = parsePlainNumber(s);
  if (direct !== null) return direct;

  const patterns: Array<[RegExp, (m: RegExpMatchArray) => number]> = [
    [/\b(\d+(\.\d+)?)\s*day(s)?\b/, (m) => Number(m[1])],
    [/\b(\d+(\.\d+)?)\s*week(s)?\b/, (m) => Number(m[1]) * 7],
    [/\b(\d+(\.\d+)?)\s*month(s)?\b/, (m) => Number(m[1]) * 30],
    [/\b(\d+(\.\d+)?)\s*hour(s)?\b/, (m) => Number(m[1]) / 24]
  ];

  for (const [re, fn] of patterns) {
    const m = s.match(re);
    if (m) return fn(m);
  }

  if (s === "today") return 1;
  if (s === "yesterday") return 1;
  if (s === "a day" || s === "one day") return 1;
  if (s === "a week" || s === "one week") return 7;
  if (s === "two weeks") return 14;

  if (s.includes("since yesterday")) return 1;
  if (s.includes("for a week")) return 7;
  if (s.includes("for two weeks")) return 14;

  return null;
}

function parseTemperature(raw: string): { value: number; unit: "F" | "C" } | null {
  const s = lower(raw);

  // 102 / 102.4 assumed F if in plausible Fahrenheit fever range
  const direct = parsePlainNumber(s);
  if (direct !== null) {
    if (direct >= 90 && direct <= 110) return { value: direct, unit: "F" };
    if (direct >= 32 && direct <= 43) return { value: direct, unit: "C" };
  }

  const mf = s.match(/(\d+(\.\d+)?)\s*°?\s*f\b/);
  if (mf) return { value: Number(mf[1]), unit: "F" };

  const mc = s.match(/(\d+(\.\d+)?)\s*°?\s*c\b/);
  if (mc) return { value: Number(mc[1]), unit: "C" };

  return null;
}

function parsePainScale(raw: string): number | null {
  const s = lower(raw);

  const direct = parsePlainNumber(s);
  if (direct !== null && direct >= 0 && direct <= 10) return direct;

  const m = s.match(/\b(\d+)\s*\/\s*10\b/);
  if (m) {
    const n = Number(m[1]);
    if (n >= 0 && n <= 10) return n;
  }

  return null;
}

function parseTokenAware(input: TokenAwareNormalizationInput): TokenAwareNormalizationResult {
  const token = String(input.token ?? "").trim().toUpperCase();
  const raw = clean(input.raw);

  if (!raw) {
    return asResult(input, null, "low");
  }

  // universal boolean attempt first for common yes/no symptom questions
  const bool = parseBoolean(raw);

  // Token-specific routing
  if (token === "AGE_Y" || token === "AGE") {
    const age = parseAgeYears(raw);
    if (age !== null) return asResult(input, age, "high", "years");
    return asResult(input, raw, "low");
  }

  if (token === "DURATION_DAYS" || token === "DURATION") {
    const days = parseDurationDays(raw);
    if (days !== null) return asResult(input, days, "high", "days");
    return asResult(input, raw, "low");
  }

  if (token === "TEMP_F" || token === "TEMP_C" || token === "FEVER_TEMP") {
    const temp = parseTemperature(raw);
    if (temp) return asResult(input, temp.value, "high", temp.unit);
    return asResult(input, raw, "low");
  }

  if (token === "SEVERITY" || token === "PAIN_SEVERITY" || token === "PAIN_SCALE") {
    const sev = parsePainScale(raw);
    if (sev !== null) return asResult(input, sev, "high", "0-10");

    const mapped = parseSeverity(raw);
    if (mapped !== null) {
      if (typeof mapped === "number") return asResult(input, mapped, "medium", "0-10");
      return asResult(input, mapped, "medium");
    }

    return asResult(input, raw, "low");
  }

  // Common symptom tokens default to yes/no if answer is boolean-like
  const symptomishPrefixes = [
    "FEVER",
    "COUGH",
    "SOB",
    "EXUDATE",
    "STRIDOR",
    "DIARRHEA",
    "VOMITING",
    "DYSURIA",
    "RUNNY_NOSE",
    "SNEEZING",
    "ITCHY_EYES",
    "FACIAL_PAIN",
    "PLEURITIC",
    "RADIATION",
    "DIAPHORESIS",
    "WEAKNESS",
    "NEURO_DEFICIT",
    "CONFUSION",
    "CHEST_PAIN",
    "FLANK_PAIN"
  ];

  if (symptomishPrefixes.includes(token) || token.endsWith("_PAIN") || token.endsWith("_PRESENT")) {
    if (bool !== null) return asResult(input, bool, "high");
  }

  // Generic yes/no fallback
  if (bool !== null) return asResult(input, bool, "medium");

  // Generic numeric fallback
  const num = parsePlainNumber(raw);
  if (num !== null) return asResult(input, num, "medium");

  // Otherwise preserve raw string
  return asResult(input, raw, "low");
}

/**
 * Backward-compatible simple value return for storage in answers.
 */
export function normalizeChatAnswer(raw: string, token?: string): string | number | boolean | null {
  return parseTokenAware({ token, raw }).parsed;
}

/**
 * Rich result version for future debugging / audit / confidence display.
 */
export function normalizeChatAnswerDetailed(
  input: TokenAwareNormalizationInput
): TokenAwareNormalizationResult {
  return parseTokenAware(input);
}

  }

  return "Please answer the next question.";

This gives you a single place to decide whether the intake should pause immediately and show urgent guidance before continuing.

This does not replace physician review. It just prevents the chat from sleepwalking past danger.

import type { CaseRecord } from "../types/case";

export type ChatEscalationDecision = {
  shouldInterrupt: boolean;
  level: "NONE" | "URGENT" | "ER_SEND";
  message?: string;
  reason?: string;
};

function truthy(v: unknown): boolean {
  if (typeof v === "boolean") return v;
  const s = String(v ?? "").trim().toLowerCase();
  return ["true", "1", "yes", "y", "yeah", "yep"].includes(s);
}

export function evaluateChatEscalation(caseRecord: CaseRecord): ChatEscalationDecision {
  const answers = caseRecord.answers ?? {};
  const redFlags = caseRecord.engineResult?.triggeredRedFlags ?? [];
  const disp = caseRecord.engineResult?.recommendedDisposition ?? "UNKNOWN";

  if (disp === "ER_SEND") {
    return {
      shouldInterrupt: true,
      level: "ER_SEND",
      message: "Your answers suggest this may need emergency evaluation now. Please seek immediate medical care or go to the emergency room.",
      reason: "engine_disposition_er_send"
    };
  }

  if (redFlags.length > 0) {
    return {
      shouldInterrupt: true,
      level: "URGENT",
      message: "Your answers suggest urgent review is needed. Please seek urgent in-person medical evaluation as soon as possible.",
      reason: "triggered_red_flags"
    };
  }

  // lightweight direct-answer safety net
  if (truthy(answers["STRIDOR"]) || truthy(answers["NEURO_DEFICIT"])) {
    return {
      shouldInterrupt: true,
      level: "ER_SEND",
      message: "These symptoms may need emergency evaluation now. Please seek emergency care right away.",
      reason: "critical_symptom_answer"
    };
  }

  if (truthy(answers["SOB"]) && truthy(answers["CHEST_PAIN"])) {
    return {
      shouldInterrupt: true,
      level: "ER_SEND",
      message: "Shortness of breath with chest pain may require emergency evaluation. Please seek immediate care.",
      reason: "sob_plus_chest_pain"
    };
  }

  return {
    shouldInterrupt: false,
    level: "NONE"
  };
}

import type { CaseRecord } from "../types/case";

export type ChatCompletionDecision = {
  completed: boolean;
  reason:
    | "NO_MORE_QUESTIONS"
    | "AWAITING_REVIEW"
    | "ESCALATED"
    | "NEEDS_MORE_INFO"
    | "MISSING_CRITICAL_QUESTIONS";
};

export function decideChatCompletion(input: {
  caseRecord: CaseRecord;
  nextQuestionToken?: string;
  unansweredCriticalQuestions?: string[];
}): ChatCompletionDecision {
  const { caseRecord, nextQuestionToken } = input;
  const unansweredCriticalQuestions = input.unansweredCriticalQuestions ?? [];

  if (caseRecord.status === "AWAITING_REVIEW") {
    return {
      completed: true,
      reason: "AWAITING_REVIEW"
    };
  }

  if (caseRecord.status === "ESCALATED") {
    return {
      completed: true,
      reason: "ESCALATED"
    };
  }

  if (unansweredCriticalQuestions.length > 0) {
    return {
      completed: false,
      reason: "MISSING_CRITICAL_QUESTIONS"
    };
  }

  if (!nextQuestionToken) {
    return {
      completed: true,
      reason: "NO_MORE_QUESTIONS"
    };
  }

  return {
    completed: false,
    reason: "NEEDS_MORE_INFO"
  };
}

type BuildStructuredPromptInput = {
  complaintId?: string;
  complaintLabel?: string;
  token?: string;
  questionText?: string;
  priorAnswers?: Record<string, unknown>;
  shortSummary?: string;
};

function clean(s?: string): string {
  return String(s ?? "").trim();
}

function humanizeToken(token?: string): string {
  return clean(token)
    .toLowerCase()
    .replace(/_/g, " ");
}

function answerSnippet(priorAnswers: Record<string, unknown> = {}, limit = 4): string {
  const entries = Object.entries(priorAnswers)
    .filter(([, v]) => v !== undefined && v !== null && String(v).trim() !== "")
    .slice(0, limit)
    .map(([k, v]) => `${humanizeToken(k)}: ${String(v)}`);

  return entries.join("; ");
}

const TOKEN_HINTS: Record<string, string> = {
  AGE_Y: "A number is fine.",
  DURATION_DAYS: "You can answer in days, like 2 days or 1 week.",
  FEVER_TEMP: "You can answer with a number, like 102 F.",
  TEMP_F: "You can answer with a temperature like 102.4.",
  TEMP_C: "You can answer with a temperature like 39.",
  SEVERITY: "A 0 to 10 number is helpful.",
  PAIN_SEVERITY: "A 0 to 10 number is helpful.",
  PAIN_SCALE: "A 0 to 10 number is helpful.",
  FEVER: "Yes or no is fine.",
  COUGH: "Yes or no is fine.",
  SOB: "Yes or no is fine.",
  DYSURIA: "Yes or no is fine."
};

export function buildStructuredChatPrompt(input: BuildStructuredPromptInput): string {
  const complaintLabel = clean(input.complaintLabel || input.complaintId || "your symptoms");
  const token = clean(input.token).toUpperCase();
  const q = clean(input.questionText) || `Please answer: ${humanizeToken(token)}?`;
  const summary = clean(input.shortSummary);
  const prior = answerSnippet(input.priorAnswers);

  const parts: string[] = [];

  if (summary) {
    parts.push(`So far I understand: ${summary}.`);
  } else if (prior) {
    parts.push(`So far I have: ${prior}.`);
  }

  parts.push(q);

  if (token && TOKEN_HINTS[token]) {
    parts.push(TOKEN_HINTS[token]);
  }

  parts.push(`This helps me assess ${complaintLabel} more accurately.`);

  return parts.join(" ");
}

  messages?: ChatMessage[];

function clean(s?: string): string {
  return String(s ?? "").trim();
}

function humanizeToken(token?: string): string {
  return clean(token)
    .toLowerCase()
    .replace(/_/g, " ");
}

function renderAnswerValue(v: unknown): string {
  if (typeof v === "boolean") return v ? "yes" : "no";
  return String(v);
}

function buildAnswerBullets(answers: Record<string, unknown> = {}, max = 8): string[] {
  return Object.entries(answers)
    .filter(([, v]) => v !== undefined && v !== null && String(v).trim() !== "")
    .slice(0, max)
    .map(([k, v]) => `${humanizeToken(k)} ${renderAnswerValue(v)}`);
}

function extractPatientNarrative(messages: ChatMessage[] = [], max = 3): string[] {
  return messages
    .filter((m) => m.role === "user")
    .map((m) => clean(m.text))
    .filter(Boolean)
    .slice(0, max);
}

export function summarizeChatTranscript(input: BuildTranscriptSummaryInput): {
  shortSummary: string;
  hpiDraft: string;
} {
  const complaintLabel = clean(input.complaintLabel || input.complaintId || "medical concern");
  const identityBits = [
    input.ageYears ? `${input.ageYears}-year-old` : "patient",
    input.sex ? String(input.sex) : ""
  ].filter(Boolean);

  const intro = `${identityBits.join(" ")} with ${complaintLabel}`.trim();

  const answerBullets = buildAnswerBullets(input.answers);
  const patientNarrative = extractPatientNarrative(input.messages);

  const shortParts: string[] = [intro];
  if (answerBullets.length) shortParts.push(answerBullets.slice(0, 4).join("; "));
  const shortSummary = shortParts.join(". ").trim();

  const hpiParts: string[] = [];
  hpiParts.push(`${intro} presenting after structured chat intake.`);

  if (patientNarrative.length) {
    hpiParts.push(`Patient-reported history includes: ${patientNarrative.join(" | ")}.`);
  }

  if (answerBullets.length) {
    hpiParts.push(`Structured responses: ${answerBullets.join("; ")}.`);
  }

  return {
    shortSummary,
    hpiDraft: hpiParts.join(" ")
  };
}
B3) server/services/chatCriticalQuestionDetector.ts

This file identifies must-ask tokens before intake can be considered safe to hand off.

import { loadComplaintConfig } from "./complaintConfigLoader";

type CriticalDetectorInput = {
  complaintId: string;
  answers: Record<string, unknown>;
  triggeredRedFlags?: string[];
  recommendedDisposition?: string;
};

type CriticalQuestionDetectorOutput = {
  criticalTokens: string[];
  reasonsByToken: Record<string, string>;
};

function tokenFromQid(qid?: string): string {
  if (!qid) return "";
  const parts = qid.split("_");
  if (parts.length >= 3) return parts.slice(2).join("_");
  return qid;
}

function isAnswered(v: unknown): boolean {
  return v !== undefined && v !== null && String(v).trim() !== "";
}

const GLOBAL_CRITICAL_TOKENS = ["AGE_Y", "DURATION_DAYS"];

const DISPOSITION_CRITICAL_MAP: Record<string, string[]> = {
  ER_SEND: ["SOB", "CHEST_PAIN", "NEURO_DEFICIT", "STRIDOR"],
  URGENT: ["FEVER", "SEVERITY"]
};

export async function detectCriticalQuestions(
  input: CriticalDetectorInput
): Promise<CriticalQuestionDetectorOutput> {
  const { complaintId, answers, triggeredRedFlags = [], recommendedDisposition = "UNKNOWN" } = input;

  const cfg = await loadComplaintConfig(complaintId);
  const questions = [...(cfg.questions ?? [])];

  const reasonsByToken: Record<string, string> = {};
  const critical = new Set<string>();

  for (const token of GLOBAL_CRITICAL_TOKENS) {
    if (!isAnswered(answers[token])) {
      critical.add(token);
      reasonsByToken[token] = "global_minimum_required";
    }
  }

  for (const token of DISPOSITION_CRITICAL_MAP[recommendedDisposition] ?? []) {
    if (!isAnswered(answers[token])) {
      critical.add(token);
      reasonsByToken[token] = `recommended_disposition_${recommendedDisposition.toLowerCase()}`;
    }
  }

  if (triggeredRedFlags.length > 0) {
    for (const q of questions) {
      const token = tokenFromQid(q.Q_ID);
      if (!token) continue;
      if (!isAnswered(answers[token])) {
        const text = String(q.QUESTION_TEXT ?? "").toLowerCase();
        if (
          text.includes("shortness of breath") ||
          text.includes("chest pain") ||
          text.includes("weakness") ||
          text.includes("stridor") ||
          text.includes("fever")
        ) {
          critical.add(token);
          reasonsByToken[token] = "red_flag_followup";
        }
      }
    }
  }

  return {
    criticalTokens: [...critical],
    reasonsByToken
  };
}
B4) server/services/chatQuestionPriorityRanker.ts

This ranks candidate questions so you ask the most valuable one first.

For MVP, it scores questions by:

critical token status

whether token appears in red-flag-ish language

whether token appears in top dx labels / winning cluster

whether it is still unanswered

type CandidateQuestion = {
  token: string;
  questionText: string;
  askOrder?: number;
};

type RankQuestionInput = {
  candidates: CandidateQuestion[];
  answers: Record<string, unknown>;
  criticalTokens?: string[];
  winningClusterId?: string;
  dxCandidates?: Array<{ dxId?: string; label?: string }>;
  triggeredRedFlags?: string[];
};

type RankedQuestion = CandidateQuestion & {
  priorityScore: number;
  priorityReasons: string[];
};

function isAnswered(v: unknown): boolean {
  return v !== undefined && v !== null && String(v).trim() !== "";
}

function tokenish(s?: string): string {
  return String(s ?? "").trim().toUpperCase();
}

function questionSignal(token: string, text: string): number {
  const t = tokenish(token);
  const q = text.toLowerCase();

  let score = 0;
  if (["SOB", "CHEST_PAIN", "STRIDOR", "NEURO_DEFICIT"].includes(t)) score += 30;
  if (["FEVER", "SEVERITY", "DURATION_DAYS", "AGE_Y"].includes(t)) score += 10;
  if (q.includes("shortness of breath") || q.includes("chest pain") || q.includes("weakness")) score += 15;

  return score;
}

export function rankNextQuestions(input: RankQuestionInput): RankedQuestion[] {
  const critical = new Set((input.criticalTokens ?? []).map(tokenish));
  const dxText = (input.dxCandidates ?? [])
    .map((d) => `${d.label ?? ""} ${d.dxId ?? ""}`.toUpperCase())
    .join(" ");

  const ranked = input.candidates
    .filter((c) => !isAnswered(input.answers[c.token]))
    .map((c) => {
      let priorityScore = 0;
      const priorityReasons: string[] = [];

      if (critical.has(tokenish(c.token))) {
        priorityScore += 100;
        priorityReasons.push("critical_token");
      }

      const signal = questionSignal(c.token, c.questionText);
      if (signal > 0) {
        priorityScore += signal;
        priorityReasons.push("clinical_signal");
      }

      if (dxText.includes(tokenish(c.token))) {
        priorityScore += 8;
        priorityReasons.push("dx_text_overlap");
      }

      if ((input.winningClusterId ?? "").toUpperCase().includes(tokenish(c.token))) {
        priorityScore += 5;
        priorityReasons.push("winning_cluster_overlap");
      }

      priorityScore += Math.max(0, 50 - Number(c.askOrder ?? 999));

      return {
        ...c,
        priorityScore,
        priorityReasons
      };
    })
    .sort((a, b) => b.priorityScore - a.priorityScore || Number(a.askOrder ?? 999) - Number(b.askOrder ?? 999));

  return ranked;
}

import path from "path";

export type ChatAnswerCoercionAuditRow = {
  timestamp: string;
  caseId: string;
  complaintId?: string;
  token?: string;
  rawAnswer: string;
  parsedAnswer: string;
  normalizedUnit?: string;
  confidence: "high" | "medium" | "low";
  notes?: string;
};

const OUT_PATH = path.join(
  process.cwd(),
  "data",
  "complaints",
  "runtime",
  "chat_answer_coercion_audit.csv"
);

const HEADERS = [
  "TIMESTAMP",
  "CASE_ID",
  "CC_ID",
  "TOKEN",
  "RAW_ANSWER",
  "PARSED_ANSWER",
  "NORMALIZED_UNIT",
  "CONFIDENCE",
  "NOTES"
];

function csvEscape(v: string): string {
  if (v.includes(",") || v.includes('"') || v.includes("\n")) {
    return `"${v.replace(/"/g, '""')}"`;
  }
  return v;
}

function ensureFile() {
  if (fs.existsSync(OUT_PATH)) return;
  fs.mkdirSync(path.dirname(OUT_PATH), { recursive: true });
  fs.writeFileSync(OUT_PATH, HEADERS.join(",") + "\n", "utf8");
}

export function logChatAnswerCoercionAudit(row: ChatAnswerCoercionAuditRow) {
  try {
    ensureFile();

    const values = [
      row.timestamp,
      row.caseId,
      row.complaintId ?? "",
      row.token ?? "",
      row.rawAnswer,
      row.parsedAnswer,
      row.normalizedUnit ?? "",
      row.confidence,
      row.notes ?? ""
    ];

    fs.appendFileSync(
      OUT_PATH,
      values.map((v) => csvEscape(String(v))).join(",") + "\n",
      "utf8"
    );
  } catch (err) {
    console.error("[chatAnswerCoercionAudit] failed:", err);
  }
}

/**
 * Recommended policy:
 * - log everything below high confidence
 * - optionally log all critical-token coercions
 */
export function shouldAuditCoercion(confidence: "high" | "medium" | "low", token?: string): boolean {
  const criticalTokens = new Set([
    "AGE_Y",
    "DURATION_DAYS",
    "FEVER_TEMP",
    "TEMP_F",
    "TEMP_C",
    "SEVERITY",
    "PAIN_SEVERITY",
    "SOB",
    "CHEST_PAIN",
    "STRIDOR",
    "NEURO_DEFICIT"
  ]);

  if (confidence !== "high") return true;
  if (token && criticalTokens.has(token.toUpperCase())) return true;
  return false;
}

import type { CaseRecord } from "../types/case";

export type ChatDispositionExplanation = {
  headline: string;
  body: string;
  urgency: "low" | "moderate" | "high";
};

function joinList(items: string[] = []): string {
  if (!items.length) return "";
  if (items.length === 1) return items[0];
  if (items.length === 2) return `${items[0]} and ${items[1]}`;
  return `${items.slice(0, -1).join(", ")}, and ${items[items.length - 1]}`;
}

export function buildChatDispositionExplanation(caseRecord: CaseRecord): ChatDispositionExplanation {
  const disposition = caseRecord.engineResult?.recommendedDisposition ?? "UNKNOWN";
  const redFlags = caseRecord.engineResult?.triggeredRedFlags ?? [];
  const complaint = caseRecord.complaintLabel || caseRecord.complaintId || "your symptoms";

  if (disposition === "ER_SEND") {
    return {
      headline: "Emergency evaluation recommended",
      body: redFlags.length
        ? `Your answers about ${complaint} raised urgent concern, including ${joinList(redFlags)}. Please seek emergency medical care now.`
        : `Your answers about ${complaint} suggest this may need emergency evaluation now. Please seek emergency medical care immediately.`,
      urgency: "high"
    };
  }

  if (disposition === "URGENT") {
    return {
      headline: "Urgent medical evaluation recommended",
      body: redFlags.length
        ? `Your answers suggest urgent in-person review is needed, including concern about ${joinList(redFlags)}. Please seek prompt medical evaluation.`
        : `Your answers suggest ${complaint} should be reviewed urgently in person.`,
      urgency: "high"
    };
  }

  if (caseRecord.status === "AWAITING_REVIEW" || caseRecord.reviewStatus === "PENDING_REVIEW") {
    return {
      headline: "Awaiting clinician review",
      body: `Your intake for ${complaint} has been recorded and is waiting for clinician review. If your symptoms worsen, seek urgent care sooner.`,
      urgency: "moderate"
    };
  }

  if (disposition === "PCP") {
    return {
      headline: "Outpatient follow-up recommended",
      body: `Based on the current intake, ${complaint} may be appropriate for outpatient follow-up. A clinician should still confirm the plan.`,
      urgency: "low"
    };
  }

  if (disposition === "SELF_CARE") {
    return {
      headline: "Lower-acuity pattern so far",
      body: `Based on the current intake, ${complaint} appears lower acuity so far. A clinician may still review the case depending on workflow and symptom changes.`,
      urgency: "low"
    };
  }

  return {
    headline: "Clinical review in progress",
    body: `Your intake for ${complaint} has been recorded. Additional review may still be needed.`,
    urgency: "moderate"
  };
}

function safeText(v?: string): string {
  return String(v ?? "").trim();
}

function answerPreview(answers: Record<string, unknown> = {}, limit = 8): Array<{ token: string; value: string }> {
  return Object.entries(answers)
    .filter(([, v]) => v !== undefined && v !== null && String(v).trim() !== "")
    .slice(0, limit)
    .map(([k, v]) => ({
      token: k,
      value: typeof v === "boolean" ? (v ? "yes" : "no") : String(v)
    }));
}

    caseId: caseRecord.caseId,
    complaintId: caseRecord.complaintId,
    complaintLabel: caseRecord.complaintLabel,
    status: caseRecord.status,
    reviewStatus: caseRecord.reviewStatus,
    sourceChannel: caseRecord.sourceChannel,
    patient: {
      firstName: caseRecord.patientContext?.firstName,
      lastName: caseRecord.patientContext?.lastName,
      ageYears: caseRecord.patientContext?.ageYears,
      sex: caseRecord.patientContext?.sex
    },
    answersPreview: answerPreview(caseRecord.answers ?? {}),
    engine: {
      recommendedDisposition: caseRecord.engineResult?.recommendedDisposition ?? "UNKNOWN",
      confidence: caseRecord.engineResult?.confidence,
      winningClusterId: caseRecord.engineResult?.winningClusterId,
      redFlags: caseRecord.engineResult?.triggeredRedFlags ?? [],

        rank: d.rank

    },

    exportedToEcw: caseRecord.exportedToEcw ?? false,

    physicianSummaryPresent: Boolean(safeText(caseRecord.physicianSummary)),

    updatedAt: caseRecord.updatedAt,
    createdAt: caseRecord.createdAt
  };
}

5) client/src/hooks/useCaseSnapshot.ts

This is the extra frontend hook I strongly recommend now.

It keeps queue/review code cleaner and will save you a lot of repeated fetch boilerplate.

import { useEffect, useState } from "react";
import { useAuth } from "../context/AuthContext";

export function useCaseSnapshot(caseId?: string) {
  const { authFetch } = useAuth();
  const [snapshot, setSnapshot] = useState<any>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState("");

  async function load() {
    if (!caseId) return;

    setLoading(true);
    setError("");

    try {
      const res = await authFetch(`/api/caseSnapshot/${caseId}`);
      const json = await res.json();

      if (!res.ok) {
        throw new Error(json.error || "Failed to load case snapshot");
      }

      setSnapshot(json);
    } catch (err: any) {
      setError(err?.message ?? "Failed to load case snapshot");
    } finally {
      setLoading(false);
    }
  }

  useEffect(() => {
    load();
  }, [caseId]);

  return {
    snapshot,
    loading,
    error,
    reload: load
  };
}
