🧾 DB SCHEMA
CREATE TABLE IF NOT EXISTS robot_devices (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  device_id TEXT,
  type TEXT, -- otoscope | camera | ekg | vitals
  status TEXT,
  last_seen TIMESTAMP
);

CREATE TABLE IF NOT EXISTS robot_commands (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  device_id TEXT,
  command TEXT,
  payload JSONB,
  status TEXT DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT now()
);

CREATE TABLE IF NOT EXISTS robot_results (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  device_id TEXT,
  result_type TEXT,
  data JSONB,
  created_at TIMESTAMP DEFAULT now()
);
⚙️ ROBOT CONTROLLER (SERVER)
// server/robot/robotController.ts
export async function sendRobotCommand(db, device_id, command, payload) {
  const cmd = await db.robot_commands.insert({
    device_id,
    command,
    payload
  })

  return cmd
}

export async function receiveRobotResult(db, data) {
  await db.robot_results.insert(data)
}
⚙️ ROBOT API
// server/routes/robotRoutes.ts
import express from 'express'
const router = express.Router()

router.post('/command', async (req, res) => {
  const { device_id, command, payload } = req.body
  const cmd = await sendRobotCommand(req.db, device_id, command, payload)
  res.json(cmd)
})

router.post('/result', async (req, res) => {
  await receiveRobotResult(req.db, req.body)
  res.json({ ok: true })
})

router.get('/devices', async (req, res) => {
  const devices = await req.db.robot_devices.findAll()
  res.json(devices)
})

export default router

🎛️ ROBOT UI PANEL
// client/components/tower/RobotPanel.tsx
export default function RobotPanel() {
  const send = async (cmd) => {
    await fetch('/api/robot/command', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(cmd)
    })
  }

  return (
    <div className="card">
      <h3>Robotic Exam</h3>

      <button onClick={() => send({ device_id:'cam1', command:'capture_ear' })}>
        Otoscope
      </button>

      <button onClick={() => send({ device_id:'vitals1', command:'measure' })}>
        Vitals
      </button>

      <button onClick={() => send({ device_id:'ekg1', command:'run' })}>
        EKG
      </button>
    </div>
  )
}
