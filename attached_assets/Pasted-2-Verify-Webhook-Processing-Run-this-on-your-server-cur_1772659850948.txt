3) Add the generator script

Create:

server/tools/generateComplaints.ts

import fs from "fs";
import path from "path";
import Papa from "papaparse";
import { z } from "zod";

type RegistryRow = {
  CC_ID: string;
  COMPLAINT_KEY: string;
  DISPLAY_NAME: string;
  SYSTEM: string;
  SEVERITY_MODEL?: string;
  DEFAULT_DISPOSITION?: string;
  NOTES?: string;
};

const RowSchema = z.object({
  CC_ID: z.string().min(1),
  COMPLAINT_KEY: z
    .string()
    .min(1)
    .regex(/^[a-z0-9]+(?:_[a-z0-9]+)*$/, "COMPLAINT_KEY must be snake_case"),
  DISPLAY_NAME: z.string().min(1),
  SYSTEM: z.string().min(1),
  SEVERITY_MODEL: z.string().optional().default("standard"),
  DEFAULT_DISPOSITION: z.string().optional().default("HOME_CARE"),
  NOTES: z.string().optional().default(""),
});

function ensureDir(p: string) {
  fs.mkdirSync(p, { recursive: true });
}

function writeFileIfChanged(filePath: string, content: string) {
  if (fs.existsSync(filePath)) {
    const existing = fs.readFileSync(filePath, "utf8");
    if (existing === content) return;
  }
  fs.writeFileSync(filePath, content, "utf8");
}

function csvString(headers: string[], rows: Array<Record<string, string>>) {
  const lines: string[] = [];
  lines.push(headers.join(","));
  for (const row of rows) {
    lines.push(headers.map((h) => (row[h] ?? "")).join(","));
  }
  return lines.join("\n") + "\n";
}

function nowIso() {
  return new Date().toISOString();
}

function main() {
  const root = process.cwd();
  const inputCsv = path.join(root, "data", "complaints", "complaint_registry_seed.csv");
  const outBase = path.join(root, "server", "complaints", "generated");

  if (!fs.existsSync(inputCsv)) {
    console.error(`Missing input CSV: ${inputCsv}`);
    process.exit(1);
  }

  const raw = fs.readFileSync(inputCsv, "utf8");
  const parsed = Papa.parse<RegistryRow>(raw, {
    header: true,
    skipEmptyLines: true,
  });

  if (parsed.errors?.length) {
    console.error("CSV parse errors:", parsed.errors);
    process.exit(1);
  }

  const rowsRaw = parsed.data.filter((r) => Object.values(r).some(Boolean));
  const rows: Array<z.infer<typeof RowSchema>> = [];

  // Validate rows
  for (const r of rowsRaw) {
    const res = RowSchema.safeParse(r);
    if (!res.success) {
      console.error("Invalid row:", r);
      console.error(res.error.issues);
      process.exit(1);
    }
    rows.push(res.data);
  }

  ensureDir(outBase);

  const barrelExports: string[] = [];
  const mapEntries: string[] = [];

  for (const r of rows) {
    const complaintKey = r.COMPLAINT_KEY;
    const dir = path.join(outBase, complaintKey);
    ensureDir(dir);

    // meta
    const meta = {
      generated_at: nowIso(),
      cc_id: r.CC_ID,
      complaint_key: complaintKey,
      display_name: r.DISPLAY_NAME,
      system: r.SYSTEM,
      severity_model: r.SEVERITY_MODEL ?? "standard",
      default_disposition: r.DEFAULT_DISPOSITION ?? "HOME_CARE",
      notes: r.NOTES ?? "",
      version: "v1",
    };
    writeFileIfChanged(path.join(dir, "complaint.meta.json"), JSON.stringify(meta, null, 2) + "\n");

    // output_templates.csv (stub)
    // Keep headers stable and conservative. Your renderer can ignore unknowns.
    const outputHeaders = [
      "TEMPLATE_ID",
      "CC_ID",
      "COMPLAINT_KEY",
      "STATE",
      "TITLE",
      "BODY_HTML",
      "SEVERITY",
      "VERSION",
    ];
    const outputRows = [
      {
        TEMPLATE_ID: `OT_${r.CC_ID}_TRIAGED_v1`,
        CC_ID: r.CC_ID,
        COMPLAINT_KEY: complaintKey,
        STATE: "TRIAGED",
        TITLE: `${r.DISPLAY_NAME} Result`,
        BODY_HTML:
          "Disposition: <b>{{DISPOSITION}}</b><br/>Possible Diagnoses:<br/>{{DX_LIST}}<br/><br/>{{SAFETY_NET}}",
        SEVERITY: "INFO",
        VERSION: "v1",
      },
      {
        TEMPLATE_ID: `OT_${r.CC_ID}_NEEDS_REVIEW_v1`,
        CC_ID: r.CC_ID,
        COMPLAINT_KEY: complaintKey,
        STATE: "NEEDS_REVIEW",
        TITLE: `${r.DISPLAY_NAME} Needs Review`,
        BODY_HTML:
          "Your answers suggest this may need clinician review.<br/>Disposition: <b>{{DISPOSITION}}</b><br/>{{SAFETY_NET}}",
        SEVERITY: "WARN",
        VERSION: "v1",
      },
    ];
    writeFileIfChanged(path.join(dir, "output_templates.csv"), csvString(outputHeaders, outputRows));

    // disposition_rules.csv (stub)
    const dispHeaders = [
      "RULE_ID",
      "CC_ID",
      "COMPLAINT_KEY",
      "PRIORITY",
      "IF_ALL",
      "DISPOSITION",
      "SET_STATE",
      "REASON",
      "VERSION",
    ];
    const dispRows = [
      {
        RULE_ID: `DR_${r.CC_ID}_RED_FLAG_v1`,
        CC_ID: r.CC_ID,
        COMPLAINT_KEY: complaintKey,
        PRIORITY: "100",
        IF_ALL: "HAS_RED_FLAG=true",
        DISPOSITION: "EMERGENCY",
        SET_STATE: "NEEDS_REVIEW",
        REASON: "Red flag present",
        VERSION: "v1",
      },
      {
        RULE_ID: `DR_${r.CC_ID}_DEFAULT_v1`,
        CC_ID: r.CC_ID,
        COMPLAINT_KEY: complaintKey,
        PRIORITY: "1",
        IF_ALL: "true",
        DISPOSITION: r.DEFAULT_DISPOSITION ?? "HOME_CARE",
        SET_STATE: "TRIAGED",
        REASON: "Default disposition",
        VERSION: "v1",
      },
    ];
    writeFileIfChanged(path.join(dir, "disposition_rules.csv"), csvString(dispHeaders, dispRows));

    // cluster_scoring_rules.csv (stub)
    const clusterHeaders = [
      "RULE_ID",
      "CC_ID",
      "COMPLAINT_KEY",
      "CLUSTER_ID",
      "WEIGHT",
      "IF_ALL",
      "DX_LABEL",
      "VERSION",
    ];
    const clusterRows = [
      {
        RULE_ID: `CSR_${r.CC_ID}_VIRAL_v1`,
        CC_ID: r.CC_ID,
        COMPLAINT_KEY: complaintKey,
        CLUSTER_ID: "CLUSTER_VIRAL_SYNDROME",
        WEIGHT: "1",
        IF_ALL: "true",
        DX_LABEL: "Viral syndrome",
        VERSION: "v1",
      },
    ];
    writeFileIfChanged(path.join(dir, "cluster_scoring_rules.csv"), csvString(clusterHeaders, clusterRows));

    // questions.json (stub)
    const questions = {
      complaint_key: complaintKey,
      version: "v1",
      // You can extend this later to include branching question sets and required modifiers.
      questions: [
        {
          id: "duration_days",
          type: "number",
          prompt: `How many days have you had ${r.DISPLAY_NAME.toLowerCase()}?`,
          required: true,
          min: 0,
          max: 60,
        },
        {
          id: "has_red_flag",
          type: "yesno",
          prompt: "Any severe symptoms like trouble breathing, severe pain, confusion, fainting, or blue lips?",
          required: true,
          maps_to: "HAS_RED_FLAG",
        },
      ],
    };
    writeFileIfChanged(path.join(dir, "questions.json"), JSON.stringify(questions, null, 2) + "\n");

    // index.ts per complaint
    const complaintTs = `/**
 * AUTO-GENERATED. Do not edit by hand.
 * Source: data/complaints/complaint_registry_seed.csv
 */
import fs from "fs";
import path from "path";

export const meta = ${JSON.stringify(meta, null, 2)} as const;

const DIR = __dirname;

export function loadCsv(name: "output_templates.csv" | "disposition_rules.csv" | "cluster_scoring_rules.csv"): string {
  return fs.readFileSync(path.join(DIR, name), "utf8");
}

export function loadQuestionsJson(): any {
  return JSON.parse(fs.readFileSync(path.join(DIR, "questions.json"), "utf8"));
}
`;
    writeFileIfChanged(path.join(dir, "index.ts"), complaintTs);

    barrelExports.push(`export * as ${complaintKey} from "./${complaintKey}";`);
    mapEntries.push(`  "${complaintKey}": ${complaintKey},`);
  }

  // barrel index
  const barrel = `/**
 * AUTO-GENERATED. Do not edit by hand.
 * Generated by: server/tools/generateComplaints.ts
 */
${barrelExports.join("\n")}

export const GENERATED_COMPLAINTS = {
${mapEntries.join("\n")}
} as const;

export type GeneratedComplaintKey = keyof typeof GENERATED_COMPLAINTS;
`;
  writeFileIfChanged(path.join(outBase, "index.ts"), barrel);

  console.log(`✅ Generated ${rows.length} complaint bundles in ${outBase}`);
}

main();
