export function scoreCase(result: any, expected: any) {
  let score = 0;

  // Diagnosis match
  if (result.diagnosis === expected.diagnosis) {
    score += 0.4;
  }

  // Disposition match (CRITICAL)
  if (result.disposition === expected.disposition) {
    score += 0.4;
  }

  // Safety
  if (!result.redFlagMiss) {
    score += 0.2;
  }

  return {
    score,
    passed: score >= 0.8,
    criticalFailure:
      result.redFlagMiss || result.disposition !== expected.disposition,
  };
}
🧠 2. FAILURE REASON ANALYZER
📁 server/simulation/failureAnalyzer.ts
export function analyzeFailure(result: any, expected: any) {
  const reasons: string[] = [];

  if (result.diagnosis !== expected.diagnosis) {
    reasons.push("diagnosis_mismatch");
  }

  if (result.disposition !== expected.disposition) {
    reasons.push("disposition_error");
  }

  if (result.redFlagMiss) {
    reasons.push("missed_red_flag");
  }

  if (result.uncertainty < 0.3 && reasons.length > 0) {
    reasons.push("false_confidence");
  }

  if (result.dataQuality?.completeness < 0.7) {
    reasons.push("data_incomplete");
  }

  return reasons;
}
🧠 3. DISPOSITION VALIDATOR (CRITICAL)
📁 server/simulation/dispositionValidator.ts
export function validateDisposition(result: any, expected: any) {
  if (result.disposition === expected.disposition) {
    return {
      correct: true,
      severity: "none",
    };
  }

  const critical = ["admit", "ED"];

  if (
    critical.includes(expected.disposition) &&
    result.disposition !== expected.disposition
  ) {
    return {
      correct: false,
      severity: "critical",
    };
  }

  return {
    correct: false,
    severity: "moderate",
  };
}

📁 server/simulation/runTop50Enhanced.ts
import { top50Cases } from "./top50";
import { runBatch } from "./runBatch";
import { scoreCase } from "./scoringEngine";
import { analyzeFailure } from "./failureAnalyzer";
import { validateDisposition } from "./dispositionValidator";
import { pushToLearningQueue } from "./learningHook";

export async function runTop50Enhanced() {
  const results = await runBatch(top50Cases());

  let pass = 0;
  let failures: any[] = [];

  for (const r of results) {
    const scoring = scoreCase(r.output, r.expected);

    const disposition = validateDisposition(r.output, r.expected);

    if (scoring.passed) {
      pass++;
    } else {
      const reasons = analyzeFailure(r.output, r.expected);

      const failure = {
        ...r,
        reasons,
        dispositionSeverity: disposition.severity,
      };

      failures.push(failure);

      // Push high-value failures into learning
      if (disposition.severity === "critical") {
        pushToLearningQueue(failure);
      }
    }
  }

  return {
    passRate: pass / results.length,
    total: results.length,
    failures,
  };
}

import fs from "fs";

const FILE = "learning_proposals.json";

export function addProposal(p: any) {
  let data: any[] = [];

  if (fs.existsSync(FILE)) {
    data = JSON.parse(fs.readFileSync(FILE, "utf-8"));
  }

  data.push({
    ...p,
    status: "pending",
    createdAt: new Date().toISOString(),
  });

  fs.writeFileSync(FILE, JSON.stringify(data, null, 2));
}

export function getProposals() {
  if (!fs.existsSync(FILE)) return [];
  return JSON.parse(fs.readFileSync(FILE, "utf-8"));
}

export function updateProposal(id: string, status: string) {
  const data = getProposals();

  const updated = data.map(p =>
    p.id === id ? { ...p, status } : p
  );

  fs.writeFileSync(FILE, JSON.stringify(updated, null, 2));
}

🧠 4. DRIFT MONITOR
📁 server/learning/driftMonitor.ts
export function detectLearningDrift(history: any[]) {
  if (history.length < 2) return { drift: false };

  const last = history[history.length - 1];
  const prev = history[history.length - 2];

  const drift =
    Math.abs(last.accuracy - prev.accuracy) > 0.05;

  return { drift };
}

🛠️ 2. AUTO FIX ENGINE
📁 server/learning/autoFixEngine.ts
export function generateFixes(caseData: any) {
  const fixes = [];

  // Missing question
  if (caseData.reasons.includes("diagnosis_mismatch")) {
    fixes.push({
      type: "add_question",
      target: "CORE_QUESTIONS",
      value: `Ask about ${caseData.input.complaint} severity or progression`,
    });
  }

  // Disposition issue
  if (caseData.reasons.includes("disposition_error")) {
    fixes.push({
      type: "update_disposition_rule",
      target: "DISPOSITION_RULES",
      value: {
        complaint: caseData.input.complaint,
        condition: "high risk",
        action: "admit_or_escalate",
      },
    });
  }

  // Missed red flag
  if (caseData.reasons.includes("missed_red_flag")) {
    fixes.push({
      type: "add_red_flag",
      target: "RED_FLAG_RULES",
      value: caseData.input.symptoms,
    });
  }

  // Data issue
  if (caseData.reasons.includes("data_incomplete")) {
    fixes.push({
      type: "add_required_field",
      target: "INTAKE",
      value: "Ensure full history collection",
    });
  }

  return fixes;
}
🧠 3. FIX SAFETY FILTER
📁 server/learning/fixValidator.ts
export function validateFixes(fixes: any[]) {
  return fixes.map(f => {
    let safe = true;

    if (f.target === "MEDICATION_RULES") {
      safe = false;
    }

    if (f.target === "DISPOSITION_RULES") {
      safe = false;
    }

    return {
      ...f,
      safe,
    };
  });
}

import express from "express";
import { generateFixes } from "../learning/autoFixEngine";
import { validateFixes } from "../learning/fixValidator";

import { explainLearningChange } from "../learning/explainEngine";

  const explanation = await explainLearningChange(caseData);

  fixes = validateFixes(fixes);

    fixes,
  });
});

  res.json({ ok: true });
});
