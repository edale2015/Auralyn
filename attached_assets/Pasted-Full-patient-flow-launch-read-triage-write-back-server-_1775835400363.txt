export async function epicFullFlow(patientId: string, token: string) {

  // 1. Read patient
  const patient = await fetch(`${process.env.FHIR_BASE}/Patient/${patientId}`, {
    headers: { Authorization: `Bearer ${token}` }
  }).then(r => r.json());

  // 2. Run triage
  const triage = await runFinalPipeline({
    patientId,
    rawText: patient?.name?.[0]?.text || "unknown",
    answers: {},
    sessionId: "epic",
    actorId: "system"
  }, "epic-flow");

  // 3. Write disposition note
  const note = {
    resourceType: "Observation",
    status: "final",
    subject: { reference: `Patient/${patientId}` },
    code: { text: "Triage Disposition" },
    valueString: triage.safetyResult.finalDisposition
  };

  await fetch(`${process.env.FHIR_BASE}/Observation`, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${token}`,
      "Content-Type": "application/fhir+json"
    },
    body: JSON.stringify(note)
  });

  return triage;
}
