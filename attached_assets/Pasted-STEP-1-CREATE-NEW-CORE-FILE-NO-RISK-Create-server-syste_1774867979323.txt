import { runFullClinicalFlow } from "../orchestrator/runFullClinicalFlow";
import crypto from "crypto";

export async function runAutonomousPipeline(ctx: any) {
  const start = Date.now();

  try {
    // 1. Run your real engine
    const result = await runFullClinicalFlow(ctx.input || ctx);

    // 2. Add unified metadata layer
    const enriched = {
      ...result,
      _meta: {
        pipeline: "autonomous",
        requestId: crypto.randomUUID(),
        durationMs: Date.now() - start,
        timestamp: Date.now()
      }
    };

    return enriched;

  } catch (err: any) {
    return {
      error: "pipeline_failed",
      message: err.message,
      _meta: {
        pipeline: "autonomous",
        failed: true
      }
    };
  }
}
