export function emitClinicalEvent(caseId:string,event:any){

State projection

Updates state from events.

// server/state/stateProjectionService.ts
import { getClinicalState } from "./clinicalStateStore"

export function projectEvent(caseId:string,event:any){

  const state = getClinicalState(caseId)

  switch(event.type){

    case "SYMPTOMS_RECORDED":
      state.symptoms = event.symptoms
      break

    case "COMPLAINT_IDENTIFIED":
      state.complaint = event.complaint
      break

    case "DIFFERENTIAL_UPDATED":
      state.differential = event.differential
      break

    case "DISPOSITION_SET":
      state.disposition = event.disposition
      break
  }

}

Now every module can read/write the same patient state.

// server/intake/autonomousIntakeEngine.ts

import { emitClinicalEvent } from "../state/clinicalEventBus"

export async function processIntakeMessage(params:{
  caseId:string
  message:string
}){

  const text = params.message.toLowerCase()

  emitClinicalEvent(params.caseId,{
    type:"SYMPTOMS_RECORDED",
    symptoms:text
  })

  const triage = 
    text.includes("chest pain") ? "high"
    : text.includes("fever") ? "moderate"
    : "low"

  return {
    triage,
    nextQuestion:
      triage==="high"
      ? "Are you short of breath?"
      : "How long have you had these symptoms?"
  }
}

// reinforcementPolicyService.ts

import { listReconciliations } from "../services/reconciliationService"

export async function trainDispositionPolicy(){

  const rows = await listReconciliations(5000)

  const policy:any={}

  for(const r of rows){

    const c = r.predictedComplaint

    policy[c] ??= {reward:0,count:0}

    const reward =
      (r.disposition_match?1:0) +
      (r.top_prediction_match?1:0) -
      (r.safety_miss_flag?2:0)

    policy[c].reward += reward
    policy[c].count++
  }

  return Object.entries(policy).map(([complaint,v]:any)=>({

    complaint,
    avgReward:v.reward/v.count

  }))
}

export function predictAdmissionRisk(symptoms:string){

  const s = symptoms.toLowerCase()

  if(s.includes("chest pain"))
    return {risk:"high"}

  if(s.includes("shortness of breath"))
