import { EventEmitter } from "events";

export const simBus = new EventEmitter();

export function startLiveSimulation() {

  let tick = 0;

  setInterval(() => {

    const patients = Math.floor(Math.random() * 40);
    const er = Math.floor(patients * 0.2);
    const critical = Math.floor(er * 0.3);

    const snapshot = {
      timestamp: Date.now(),
      patients,
      er,
      critical,
      load: patients > 30 ? "high" : "normal"
    };

    simBus.emit("update", snapshot);

    tick++;

  }, 1000);
}
📡 WebSocket Streaming Server
server/ws/liveStream.ts
import { WebSocketServer } from "ws";
import { simBus } from "../simulation/liveSimulator";

export function startLiveStream(server: any) {

  const wss = new WebSocketServer({ server });

  wss.on("connection", ws => {

    const handler = (data: any) => {
      ws.send(JSON.stringify(data));
    };

    simBus.on("update", handler);

    ws.on("close", () => {
      simBus.off("update", handler);
    });
  });
}
🖥️ Live Dashboard UI
client/pages/LiveSimulation.tsx
import { useEffect, useState } from "react";

export default function LiveSimulation() {

  const [data, setData] = useState<any[]>([]);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:3000");

    ws.onmessage = (msg) => {
      const d = JSON.parse(msg.data);
      setData(prev => [...prev.slice(-20), d]);
    };
  }, []);

  return (
    <div className="p-4 text-white bg-black">
      <h2>🧪 Live Hospital Simulation</h2>

      {data.map((d, i) => (
        <div key={i}>
          Patients: {d.patients} | ER: {d.er} | Critical: {d.critical}
        </div>
      ))}
    </div>
  );
}

export function selectRegion(clientIp: string) {

  if (clientIp.startsWith("172")) return "us-east";
  if (clientIp.startsWith("10")) return "us-west";

  return "eu-central";
}
🔧 Region Proxy
export async function routeRequest(path: string, req: any) {

  const region = selectRegion(req.ip);

  const url = process.env[`REGION_${region.toUpperCase()}`];

  return fetch(url + path, {
    method: req.method,
    body: JSON.stringify(req.body)
  });
}

🔁 Global Failover Controller
export async function globalFetch(path: string) {

  const regions = ["us-east", "us-west", "eu"];

  for (const r of regions) {
    try {
      const res = await fetch(process.env[`REGION_${r}`] + path);
      if (res.ok) return res.json();
    } catch {}
  }

  throw new Error("Global outage");
}
