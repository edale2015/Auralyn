  }

📊 3) PREDICTIVE ADMISSION RISK SCORING
🧾 Schema
CREATE TABLE IF NOT EXISTS kb_admission_rules (
  id UUID PRIMARY KEY,
  feature_key TEXT,
  weight FLOAT
);
⚙️ Engine
// server/engine/admissionRisk.ts
export async function computeAdmissionRisk(db, input) {
  const rules = await db.kb_admission_rules.findAll()

  let score = 0

  for (const r of rules) {
    if (input[r.feature_key]) {
      score += r.weight
    }
  }

  return Math.min(1, score)
}
🔌 Hook
const risk = await computeAdmissionRisk(db, input)

await db.patient_dashboard_state.update({
  patient_id,
  risk_score: risk
})
