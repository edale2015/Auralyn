// server/knowledge/masterTables.ts
import fs from "fs";
import path from "path";

type Row = Record<string, string>;

function parseCSV(filePath: string): Row[] {
  const raw = fs.readFileSync(filePath, "utf8");
  const lines = raw.split(/\r?\n/).filter(l => l.trim().length > 0);
  const header = lines[0].split(",").map(h => h.trim());
  return lines.slice(1).map(line => {
    const cols = splitCSV(line);
    const row: Row = {};
    header.forEach((h, i) => (row[h] = (cols[i] ?? "").trim()));
    return row;
  });
}

function splitCSV(line: string): string[] {
  // minimal CSV splitter (no multiline). If you already have one, use it.
  const out: string[] = [];
  let buf = "";
  let inQ = false;
  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"' && line[i - 1] !== "\\") inQ = !inQ;
    if (ch === "," && !inQ) {
      out.push(buf);
      buf = "";
    } else buf += ch;
  }
  out.push(buf);
  return out.map(s => s.replace(/^"|"$/g, ""));
}

export class MasterTables {
  clusters: Row[];
  clusterPrimaryDx: Row[];
  urgentCareInterventions: Row[];
  medsMaster: Row[];
  redFlagsMaster: Row[];
  medToCondition: Row[];

  private idxClustersByComplaint = new Map<string, Row[]>();
  private idxDxByCluster = new Map<string, Row[]>();
  private idxInterventionsByKey = new Map<string, Row[]>();
  private idxMedById = new Map<string, Row>();
  private idxRedFlagById = new Map<string, Row[]>();

  constructor(csvDir: string) {
    this.clusters = parseCSV(path.join(csvDir, "GLOBAL_CLUSTER_MASTER.csv"));
    this.clusterPrimaryDx = parseCSV(path.join(csvDir, "CLUSTER_PRIMARY_DIAGNOSIS.csv"));
    this.urgentCareInterventions = parseCSV(path.join(csvDir, "URGENT_CARE_SPOT_INTERVENTION.csv"));
    this.medsMaster = parseCSV(path.join(csvDir, "GLOBAL_MEDICATIONS_MASTER.csv"));
    this.redFlagsMaster = parseCSV(path.join(csvDir, "RED_FLAGS_MASTER.csv"));
    this.medToCondition = parseCSV(path.join(csvDir, "MED_TO_CONDITION_TRIGGERS.csv"));

    this.buildIndexes();
  }

  private buildIndexes() {
    for (const r of this.clusters) {
      const cc = r.CC_ID;
      if (!this.idxClustersByComplaint.has(cc)) this.idxClustersByComplaint.set(cc, []);
      this.idxClustersByComplaint.get(cc)!.push(r);
    }
    for (const r of this.clusterPrimaryDx) {
      const cid = r.CLUSTER_ID;
      if (!this.idxDxByCluster.has(cid)) this.idxDxByCluster.set(cid, []);
      this.idxDxByCluster.get(cid)!.push(r);
    }
    for (const r of this.urgentCareInterventions) {
      const key = `${r.CC_ID}::${r.DISPOSITION_LEVEL}::${r.CLUSTER_ID || ""}`;
      if (!this.idxInterventionsByKey.has(key)) this.idxInterventionsByKey.set(key, []);
      this.idxInterventionsByKey.get(key)!.push(r);
    }
    for (const r of this.medsMaster) {
      const id = r.MED_ID || r.med_id || r.ID;
      if (id) this.idxMedById.set(id, r);
    }
    for (const r of this.redFlagsMaster) {
      const id = r.RF_MASTER_ID;
      if (!this.idxRedFlagById.has(id)) this.idxRedFlagById.set(id, []);
      this.idxRedFlagById.get(id)!.push(r);
    }
  }

  getClustersForComplaint(ccId: string) {
    return this.idxClustersByComplaint.get(ccId) || [];
  }

  getPrimaryDxForCluster(clusterId: string) {
    return this.idxDxByCluster.get(clusterId) || [];
  }

  getSpotInterventions(ccId: string, disposition: string, clusterId?: string) {
    const exact = `${ccId}::${disposition}::${clusterId || ""}`;
    const fallback = `${ccId}::${disposition}::`;
    return (this.idxInterventionsByKey.get(exact) || this.idxInterventionsByKey.get(fallback) || []);
  }

  getMedById(medId: string) {
    return this.idxMedById.get(medId) || null;
  }

  getRedFlagMaster(rfMasterId: string) {
    const arr = this.idxRedFlagById.get(rfMasterId) || [];
    return arr[0] || null;
  }
}

// server/complaints/nodes/clusterSelectNode.ts
import { MasterTables } from "../../knowledge/masterTables";

export function runClusterSelect(state: any, tables: MasterTables) {
  const cc = state.normalizedComplaint;
  const clusters = tables.getClustersForComplaint(cc);

  // For pilot, we score clusters by simple rules keyed on cluster IDs.
  // You can move rules into tables later.
  const scoreByCluster: Record<string, { score: number; evidence: string[] }> = {};

  const A = state.answers || {};
  const centor = Number(state.scores?.centor ?? state.scores?.CENTOR ?? 0);

  function bump(clusterId: string, points: number, evidenceField: string) {
    if (!scoreByCluster[clusterId]) scoreByCluster[clusterId] = { score: 0, evidence: [] };
    scoreByCluster[clusterId].score += points;
    scoreByCluster[clusterId].evidence.push(evidenceField);
  }

  if (cc === "sore_throat") {
    // Viral cluster
    if (A.ST_COUGH === true) bump("CL_ST_VIRAL_URI", 2, "answers.ST_COUGH");
    if (A.ST_RUNNY === true) bump("CL_ST_VIRAL_URI", 2, "answers.ST_RUNNY");
    if (A.ST_HOARSE === true) bump("CL_ST_VIRAL_URI", 1, "answers.ST_HOARSE");

    // GAS cluster
    if (A.ST_FEVER === true) bump("CL_ST_GAS", 1, "answers.ST_FEVER");
    if (A.ST_COUGH === false) bump("CL_ST_GAS", 1, "answers.ST_COUGH");
    if (centor >= 2) bump("CL_ST_GAS", 2, "scores.centor");

    // Mono cluster
    if (A.ST_MONO === true) bump("CL_ST_MONO", 2, "answers.ST_MONO");
  }

  // Rank cluster candidates that exist in master
  const present = new Set(clusters.map(c => c.CLUSTER_ID));
  const ranked = Object.entries(scoreByCluster)
    .filter(([cid]) => present.has(cid))
    .map(([cid, v]) => ({ clusterId: cid, score: v.score, evidence: v.evidence }))
    .sort((a, b) => b.score - a.score)
    .slice(0, 3);

  state.clusterScores = Object.fromEntries(ranked.map(r => [r.clusterId, r.score]));
  state.clusterEvidence = Object.fromEntries(ranked.map(r => [r.clusterId, r.evidence]));
  state.activeClusters = ranked.map(r => r.clusterId);

  // Dx candidates from cluster-primary mapping
  const dxCandidates: any[] = [];
  for (const r of ranked) {
    const dxs = tables.getPrimaryDxForCluster(r.clusterId).slice(0, 2);
    for (const d of dxs) {
      dxCandidates.push({
        diagnosisId: d.DIAGNOSIS_ID,
        label: d.DIAGNOSIS_LABEL || d.LABEL || d.DIAGNOSIS_NAME,
        clusterId: r.clusterId,
        confidence: r === ranked[0] ? "MODERATE" : "LOW",
      });
    }
  }
  state.diagnosisCandidates = dxCandidates.slice(0, 6);

  return { rankedClusters: ranked, dxCandidatesCount: state.diagnosisCandidates.length };
}

// server/complaints/nodes/medSuggestNode.ts
import { MasterTables } from "../../knowledge/masterTables";

export function runMedSuggest(state: any, tables: MasterTables) {
  // Hard gate: no med suggestions if emergent or ER_SEND
  if (state.routing?.state === "EMERGENT_ESCALATION" || state.redFlagGate?.gateResult === "ER_SEND") {
    state.medSuggestions = [];
    return { skipped: true, reason: "emergent" };
  }

  const allergies: string[] = state.modifiers?.allergies || [];
  const pregnant = Boolean(state.demographics?.pregnant);

  // You need a dx→med mapping. If you already have it, wire it here.
  // For now, we’ll assume diagnosisCandidates contain diagnosisId and you have a CSV mapping:
  // DIAGNOSIS_TO_MED.csv with columns: DIAGNOSIS_ID, MED_ID, REASON, PRIORITY
  // If you don’t have it, tell me your existing mapping table name and columns, and I’ll adapt.

  const map = state.__dxToMedMap as Array<any> | undefined;
  if (!map) {
    state.medSuggestions = [];
    return { skipped: true, reason: "no_dx_to_med_map_loaded" };
  }

  const dxIds = (state.diagnosisCandidates || []).map((d: any) => d.diagnosisId);
  const candidates = map
    .filter(r => dxIds.includes(r.DIAGNOSIS_ID))
    .sort((a, b) => Number(a.PRIORITY || 999) - Number(b.PRIORITY || 999))
    .slice(0, 6);

  const out: any[] = [];
  for (const c of candidates) {
    const med = tables.getMedById(c.MED_ID);
    if (!med) continue;

    // Simple allergy gate example: beta-lactam class check
    const medClass = (med.CLASS || "").toLowerCase();
    const betaLactamAllergy = allergies.some(a => String(a).toLowerCase().includes("penicillin") || String(a).toLowerCase().includes("beta"));

    if (betaLactamAllergy && (medClass.includes("penicillin") || medClass.includes("ceph"))) {
      continue;
    }

    // Pregnancy gate (if your med master has flags)
    const pregFlag = String(med.PREGNANCY_FLAG || "").toUpperCase();
    if (pregnant && pregFlag === "AVOID") continue;

    out.push({
      medId: c.MED_ID,
      name: med.NAME,
      rationale: c.REASON || "",
      safetyNotes: [],
    });
  }

  state.medSuggestions = out;
  for (const m of out) {
    state.recommendedActions.push({ type: "MED_SUGGESTION", priority: 60, ...m });
  }

  return { count: out.length, medIds: out.map(x => x.medId) };
}

// server/complaints/nodes/specialistCouncilNode.ts
import { callLLMJson } from "../../llm/callLLMJson"; // adapt to your LLM util

function shouldInvokeCouncil(state: any): boolean {
  const low = state.audit?.steps?.some((s:any)=>s.confidence==="LOW") || state.caseConfidence === "LOW";
  const missingReq = (state.questionQueue || []).some((q:any)=>q.required && !q.answered);
  const conflict = state.__conflictFlag === true; // set this if you detect mismatch
  return (low || missingReq || conflict) && state.routing?.state !== "EMERGENT_ESCALATION";
}

function rank(level: string): number {
  const r: Record<string, number> = { self_care: 1, pcp: 2, urgent_care: 3, er_send: 4 };
  return r[level] ?? 2;
}

export async function runSpecialistCouncilIfNeeded(state: any) {
  if (!process.env.ENABLE_COUNCIL || process.env.ENABLE_COUNCIL !== "1") {
    return { skipped: true };
  }
  if (!shouldInvokeCouncil(state)) return { skipped: true };

  const snapshot = {
    cc: state.normalizedComplaint,
    disposition: state.disposition,
    dispositionReasonCodes: state.dispositionReasonCodes,
    scores: state.scores,
    redFlagGate: state.redFlagGate,
    keyAnswers: pickKeyAnswers(state),
    missingRequired: missingRequiredQuestions(state),
  };

  const system = "You are a cautious clinical safety council. Output STRICT JSON only.";
  const clinician = await callLLMJson(system, { role: "clinician", snapshot });
  const critic = await callLLMJson(system, { role: "safety_critic", snapshot, clinician });
  const arbiter = await callLLMJson(system, { role: "rules_arbiter", snapshot, clinician, critic });

  const proposed = arbiter?.final_disposition || critic?.final_disposition || clinician?.final_disposition || state.disposition;

  // Escalation-only enforcement
  const finalDisp = rank(proposed) >= rank(state.disposition) ? proposed : state.disposition;

  if (finalDisp !== state.disposition) {
    state.disposition = finalDisp;
    state.dispositionReasonCodes = [...(state.dispositionReasonCodes || []), "COUNCIL_ESCALATION"];
  }

  // Add must-ask questions as recommended actions (optional)
  const mustAsk: string[] = arbiter?.must_ask || [];
  for (const qid of mustAsk.slice(0, 5)) {
    state.recommendedActions.push({ type: "ASK_MORE", priority: 20, questionId: qid });
  }

  // Safety net additions
  const safetyAdds: string[] = arbiter?.safety_net_additions || [];
  state.__councilSafetyNet = safetyAdds;

  return {
    invoked: true,
    finalDisposition: finalDisp,
    llmCalls: 3,
    ruleRefs: ["COUNCIL"],
    confidence: "MODERATE",
  };
}

function pickKeyAnswers(state: any) {
  const A = state.answers || {};

  return {
    ST_DUR: A.ST_DUR,
    ST_FEVER: A.ST_FEVER,
    ST_COUGH: A.ST_COUGH,

    ST_TRISMUS: A.ST_TRISMUS,
    ST_ONE_SIDE: A.ST_ONE_SIDE,
    ST_MUFFLED: A.ST_MUFFLED,
  };
}

function missingRequiredQuestions(state: any): string[] {
  return (state.questionQueue || [])
    .filter((q:any)=>q.required && !q.answered)
    .map((q:any)=>q.questionId);
}
