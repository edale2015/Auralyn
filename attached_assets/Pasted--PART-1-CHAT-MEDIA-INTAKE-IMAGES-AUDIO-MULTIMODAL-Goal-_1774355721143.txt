📸 2. Telegram Media Parsing
// server/routes/telegramRoutes.ts (ADD)

function extractTelegramMedia(msg: any) {
  let imageUrl;
  let audioUrl;

  if (msg.photo) {
    const fileId = msg.photo[msg.photo.length - 1].file_id;
    imageUrl = `https://api.telegram.org/file/bot${process.env.TELEGRAM_BOT_TOKEN}/${fileId}`;
  }

  if (msg.voice) {
    audioUrl = msg.voice.file_id;
  }

  return { imageUrl, audioUrl };
}
📸 3. WhatsApp Media Parsing
// server/routes/whatsappRoutes.ts (ADD)

function extractWhatsAppMedia(msg: any) {
  let imageUrl;
  let audioUrl;

  if (msg.type === "image") {
    imageUrl = msg.image?.id;
  }

  if (msg.type === "audio") {
    audioUrl = msg.audio?.id;
  }

  return { imageUrl, audioUrl };
}

export function handlePhysicianReply(text: string, caseId: string) {
  if (text === "1") return approveCase(caseId);
  if (text === "2") return overrideCase(caseId);
  if (text === "3") return escalateCase(caseId);
