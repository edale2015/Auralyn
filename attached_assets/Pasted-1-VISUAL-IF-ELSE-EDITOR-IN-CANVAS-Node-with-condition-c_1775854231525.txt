export default function ConditionNode({ data }: any) {
  const { field, equals } = data.condition || {};

  return (
    <div className="bg-yellow-900 text-white p-2 rounded">
      <div>IF</div>
      <input
        placeholder="field"
        defaultValue={field}
        onChange={e => data.onChange({ ...data, condition: { ...data.condition, field: e.target.value } })}
      />
      <input
        placeholder="equals"
        defaultValue={equals}
        onChange={e => data.onChange({ ...data, condition: { ...data.condition, equals: e.target.value } })}
      />
    </div>
  );
}

🔧 Adapter

export async function sendToECWEncounter(data: any) {

  const payload = {
    patientId: data.patientId,
    note: data.disposition,
    vitals: data.vitals
  };

  const res = await fetch(process.env.ECW_API!, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${process.env.ECW_TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify(payload)
  });

  if (!res.ok) throw new Error("ECW failed");
  return res.json();
}
