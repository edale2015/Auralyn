// server/clinical/safetyGate.ts

export function clinicalSafetyGate(decision: any) {
  if (decision.riskScore > 0.6) {
    return {
      allowed: false,
      reason: "Requires physician review"
    };
  }

  if (decision.uncertainty > 0.3) {
    return {
      allowed: false,
      reason: "Low confidence"
    };
  }

  return { allowed: true };
}

  // 3. safety
