1) TRUE DRAG-AND-DROP CANVAS (NODE-BASED WORKFLOWS)

Uses a lightweight React flow canvas so clinicians/ops can visually wire steps.

client/pages/WorkflowCanvas.tsx
import ReactFlow, { addEdge, useEdgesState, useNodesState } from "reactflow";
import "reactflow/dist/style.css";

const initialNodes = [
  { id: "1", position: { x: 50, y: 50 }, data: { label: "Fast Triage" } },
  { id: "2", position: { x: 250, y: 150 }, data: { label: "Full Triage" } }
];

export default function WorkflowCanvas() {
  const [nodes, setNodes, onNodesChange] = useNodesState(initialNodes);
  const [edges, setEdges, onEdgesChange] = useEdgesState([]);

  function onConnect(params: any) {
    setEdges((eds) => addEdge(params, eds));
  }

  async function save() {
    await fetch("/api/workflows/save", {
      method: "POST",
      body: JSON.stringify({ nodes, edges }),
      headers: { "Content-Type": "application/json" }
    });
    alert("Saved");
  }

  return (
    <div style={{ height: 500 }}>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onConnect={onConnect}
      />
      <button onClick={save}>💾 Save Workflow</button>
    </div>
  );
}
🚨 2) PROMETHEUS ALERTS → SLACK / WHATSAPP
server/monitoring/alerts.ts
export async function sendSlackAlert(msg: string) {
  await fetch(process.env.SLACK_WEBHOOK!, {
    method: "POST",
    body: JSON.stringify({ text: msg })
  });
}

export async function sendWhatsAppAlert(msg: string) {
  await fetch(process.env.TWILIO_URL!, {
    method: "POST",
    body: JSON.stringify({ body: msg })
  });
}

export function evaluateAlerts(metrics: any) {

  if (metrics.safetyMismatchRate > 0.01) {
    sendSlackAlert("🚨 Safety mismatch spike");
  }

  if (metrics.latency > 3000) {
    sendWhatsAppAlert("⚠️ High latency detected");
  }
}
