  const cls =

    <span className={`inline-flex rounded-full px-2.5 py-1 text-xs font-semibold ${cls}`}>
      {label}
    </span>
  );
}

function getSlaBadge(row: any) {
  if (row.blocked) return { label: "Blocked urgent", level: "danger" as const };
  if (!row.dueAt) return { label: "No due time", level: "neutral" as const };

  const diffMin = Math.round((new Date(row.dueAt).getTime() - Date.now()) / 60000);
  if (diffMin < 0) return { label: "Overdue", level: "danger" as const };
  if (diffMin <= 15) return { label: "Due soon", level: "warning" as const };
  return { label: "On time", level: "good" as const };
}

server/services/outcomeCorrelationService.ts
import { db } from "../db";
import { patientOutcomes } from "../db/schema.patientOutcomes";
import { count, desc, sql } from "drizzle-orm";

export async function getOutcomeCorrelationSummary() {
  const rows = await db
    .select({
      total: count(patientOutcomes.id),
      helpful: sql<number>`count(case when ${patientOutcomes.helpful} = true then 1 end)`,
      confusing: sql<number>`count(case when ${patientOutcomes.confusing} = true then 1 end)`,
      escalated: sql<number>`count(case when ${patientOutcomes.escalated} = true then 1 end)`,
      ledToVisit: sql<number>`count(case when ${patientOutcomes.ledToVisit} = true then 1 end)`,
      resolvedWithoutVisit: sql<number>`count(case when ${patientOutcomes.resolvedWithoutVisit} = true then 1 end)`,
      patientResponded: sql<number>`count(case when ${patientOutcomes.patientResponded} = true then 1 end)`,
      avgResponseLatencyMin: sql<number>`avg(${patientOutcomes.responseLatencyMin})`,
    })
    .from(patientOutcomes);

  return rows[0];
}

export async function getOutcomeByComplaint(limit = 20) {
  return db
    .select({
      complaint: patientOutcomes.complaint,
      total: count(patientOutcomes.id),
      helpful: sql<number>`count(case when ${patientOutcomes.helpful} = true then 1 end)`,
      confusing: sql<number>`count(case when ${patientOutcomes.confusing} = true then 1 end)`,
      escalated: sql<number>`count(case when ${patientOutcomes.escalated} = true then 1 end)`,
      ledToVisit: sql<number>`count(case when ${patientOutcomes.ledToVisit} = true then 1 end)`,
      resolvedWithoutVisit: sql<number>`count(case when ${patientOutcomes.resolvedWithoutVisit} = true then 1 end)`,
    })
    .from(patientOutcomes)
    .groupBy(patientOutcomes.complaint)
    .orderBy(desc(count(patientOutcomes.id)))
    .limit(limit);
}

export async function getOutcomeByDisposition(limit = 20) {
  return db
    .select({
      disposition: patientOutcomes.disposition,
      total: count(patientOutcomes.id),
      helpful: sql<number>`count(case when ${patientOutcomes.helpful} = true then 1 end)`,
      confusing: sql<number>`count(case when ${patientOutcomes.confusing} = true then 1 end)`,
      escalated: sql<number>`count(case when ${patientOutcomes.escalated} = true then 1 end)`,
      ledToVisit: sql<number>`count(case when ${patientOutcomes.ledToVisit} = true then 1 end)`,
      resolvedWithoutVisit: sql<number>`count(case when ${patientOutcomes.resolvedWithoutVisit} = true then 1 end)`,
    })
    .from(patientOutcomes)
    .groupBy(patientOutcomes.disposition)
    .orderBy(desc(count(patientOutcomes.id)))
    .limit(limit);
}

export async function getOutcomeByTemplate() {
  return db.execute(sql`
    select
      coalesce(${patientOutcomes.templateKey}, 'unknown') as template_key,
      count(*)::int as total,
      count(case when ${patientOutcomes.helpful} = true then 1 end)::int as helpful,
      count(case when ${patientOutcomes.confusing} = true then 1 end)::int as confusing,
      count(case when ${patientOutcomes.escalated} = true then 1 end)::int as escalated,
      count(case when ${patientOutcomes.ledToVisit} = true then 1 end)::int as led_to_visit,
      count(case when ${patientOutcomes.resolvedWithoutVisit} = true then 1 end)::int as resolved_without_visit
    from ${patientOutcomes}
    group by 1
    order by 2 desc
  `).then(r => r.rows);
}

export async function getOutcomeByTone() {
  return db.execute(sql`
    select
      coalesce(${patientOutcomes.tone}, 'unknown') as tone,
      count(*)::int as total,
      count(case when ${patientOutcomes.helpful} = true then 1 end)::int as helpful,
      count(case when ${patientOutcomes.confusing} = true then 1 end)::int as confusing,
      count(case when ${patientOutcomes.escalated} = true then 1 end)::int as escalated,
      count(case when ${patientOutcomes.ledToVisit} = true then 1 end)::int as led_to_visit,
      count(case when ${patientOutcomes.resolvedWithoutVisit} = true then 1 end)::int as resolved_without_visit
    from ${patientOutcomes}
    group by 1
    order by 2 desc
  `).then(r => r.rows);
}

export async function createNotification(input: {
  draftId?: string;
  caseId?: string;
  threadId?: string;
  physicianId?: string;
  type: string;
  channel: string;
  message: string;
}) {
  const [row] = await db.insert(notifications).values({
    draftId: input.draftId,
    caseId: input.caseId,
    threadId: input.threadId,
    physicianId: input.physicianId,
    type: input.type,
    channel: input.channel,
    message: input.message,
    sent: true,
  }).returning();

  return row;
}

export async function acknowledgeNotification(id: string) {
  const [row] = await db
    .update(notifications)
    .set({ acknowledged: true })
    .where(notifications.id.eq(id))
    .returning();

  return row;
}
2C. Escalation engine
server/services/escalationEngine.ts
import { getPendingDraftQueue } from "./pendingDraftQueueService";
import { createNotification } from "./notificationService";

export async function runEscalationSweep() {
  const rows = await getPendingDraftQueue({ limit: 500, offset: 0 });

  const created: any[] = [];

  for (const row of rows) {
    if (row.blocked) {
      created.push(
        await createNotification({
          draftId: row.draftId,
          caseId: row.caseId,
          threadId: row.threadId,
          physicianId: row.physicianId || undefined,
          type: "blocked_urgent",
          channel: "dashboard",
          message: `Blocked urgent case ${row.caseId} requires immediate manual review.`,
        })
      );
      continue;
    }

    if (row.overdue) {
      created.push(
        await createNotification({
          draftId: row.draftId,
          caseId: row.caseId,
          threadId: row.threadId,
          physicianId: row.physicianId || undefined,
          type: "overdue_draft",
          channel: "dashboard",
          message: `Draft ${row.draftId} for case ${row.caseId} is overdue.`,
        })
      );
    }

    if (row.status === "approved") {
      created.push(
        await createNotification({
          draftId: row.draftId,
          caseId: row.caseId,
          threadId: row.threadId,
          physicianId: row.physicianId || undefined,
          type: "approved_unsent",
          channel: "dashboard",
          message: `Approved draft ${row.draftId} has not been sent yet.`,
        })
      );
    }

    if (row.disposition === "same_day_visit" && !row.overdue) {
      created.push(
        await createNotification({
          draftId: row.draftId,
          caseId: row.caseId,
          threadId: row.threadId,
          physicianId: row.physicianId || undefined,
          type: "same_day_warning",
          channel: "dashboard",
          message: `Same-day visit case ${row.caseId} should be reviewed promptly.`,
        })
      );
    }
  }

  return { createdCount: created.length, created };
}

MODULE 3. Physician auto-learning layer

This learns the doctor’s style from what they keep and what they rewrite.

3A. Auto-learning service
server/services/physicianAutoLearningService.ts
import { db } from "../db";
import { replyDrafts } from "../db/schema.replyDrafts";
import { physicianStylePrefs } from "../db/schema.physicianStylePrefs";
import { avg, count, desc, eq, sql } from "drizzle-orm";

export async function inferPhysicianPreferences(physicianId: string) {
  const rows = await db
    .select({
      tone: replyDrafts.tone,
      total: count(replyDrafts.id),
      unchanged: sql<number>`count(case when ${replyDrafts.rewriteSeverity} = 'unchanged' then 1 end)`,
      light: sql<number>`count(case when ${replyDrafts.rewriteSeverity} = 'light' then 1 end)`,
      avgSimilarity: avg(replyDrafts.editSimilarity),
    })
    .from(replyDrafts)
    .where(eq(replyDrafts.physicianId, physicianId))
    .groupBy(replyDrafts.tone)
    .orderBy(desc(count(replyDrafts.id)));

  const bestTone = rows
    .map(r => ({
      tone: r.tone || "warm",
      score: Number(r.unchanged || 0) * 2 + Number(r.light || 0) + Number(r.avgSimilarity || 0),
    }))
    .sort((a, b) => b.score - a.score)[0]?.tone || "warm";

  const rewriteStats = await db
    .select({
      avgSimilarity: avg(replyDrafts.editSimilarity),
      total: count(replyDrafts.id),
    })
    .from(replyDrafts)
    .where(eq(replyDrafts.physicianId, physicianId));

  const avgSimilarityValue = Number(rewriteStats[0]?.avgSimilarity || 0.75);

  const warmthLevel = bestTone === "warm" ? 8 : bestTone === "clinical" ? 4 : 6;
  const cautionLevel = avgSimilarityValue < 0.75 ? 8 : 6;

  const [row] = await db
    .insert(physicianStylePrefs)
    .values({
      physicianId,
      defaultTone: bestTone,
      prefersBulletStyle: false,
      warmthLevel,
      cautionLevel,
      defaultSignoff: "Please let us know if symptoms worsen.",
    })
    .onConflictDoUpdate({
      target: physicianStylePrefs.physicianId,
      set: {
        defaultTone: bestTone,
        warmthLevel,
        cautionLevel,
        updatedAt: new Date(),
      },
    })
    .returning();

  return {
    learned: row,
    evidence: rows,
  };
}

4A. Heatmap service
server/services/qualityHeatmapService.ts
import { db } from "../db";
import { replyDrafts } from "../db/schema.replyDrafts";
import { patientOutcomes } from "../db/schema.patientOutcomes";
import { sql } from "drizzle-orm";

export async function getComplaintDispositionHeatmap() {
  const rows = await db.execute(sql`
    select
      coalesce(d.complaint, 'unknown') as complaint,
      coalesce(d.disposition, 'unknown') as disposition,
      count(*)::int as total,
      avg(d.edit_similarity)::float as avg_similarity,
      count(case when d.rewrite_severity = 'heavy' then 1 end)::int as heavy_rewrites,
      count(case when d.blocked = true then 1 end)::int as blocked,
      count(case when o.helpful = true then 1 end)::int as helpful,
      count(case when o.confusing = true then 1 end)::int as confusing,
      count(case when o.led_to_visit = true then 1 end)::int as led_to_visit
    from ${replyDrafts} d
    left join ${patientOutcomes} o on o.draft_id = d.id
    group by 1, 2
    order by 1, 2
  `);

  return rows.rows;
}

export async function getComplaintHeatmapSummary() {
  const rows = await db.execute(sql`
    select
      coalesce(${replyDrafts.complaint}, 'unknown') as complaint,
      count(*)::int as total,
      avg(${replyDrafts.editSimilarity})::float as avg_similarity,
      count(case when ${replyDrafts.rewriteSeverity} = 'heavy' then 1 end)::int as heavy_rewrites,
      count(case when ${replyDrafts.blocked} = true then 1 end)::int as blocked
    from ${replyDrafts}
    group by 1
    order by total desc
  `);

  return rows.rows;
}

import { db } from "../db";
import { queueAssignments } from "../db/schema.queueAssignments";
import { getPendingDraftQueue } from "./pendingDraftQueueService";

function choosePhysician(input: {
  complaint?: string | null;
  blocked: boolean;
  disposition?: string | null;
}) {
  if (input.blocked) {
    return { physicianId: "doctor-senior", reason: "Blocked urgent case routed to senior reviewer" };
  }

  if (input.disposition === "same_day_visit") {
    return { physicianId: "doctor-fasttrack", reason: "Same-day case routed to fast-track physician" };
  }

  if (input.complaint && /rash|skin/i.test(input.complaint)) {
    return { physicianId: "doctor-derm", reason: "Skin complaint routed to dermatology-focused physician" };
  }

  if (input.complaint && /uti|urinary|dysuria/i.test(input.complaint)) {
    return { physicianId: "doctor-gu", reason: "GU complaint routed to GU-focused physician" };
  }

  return { physicianId: "doctor-general", reason: "Default general queue assignment" };
}

export async function assignDraft(draft: {
  draftId: string;
  caseId: string;
  threadId: string;
  complaint?: string | null;
  blocked: boolean;
  disposition?: string | null;
  taskPriority?: string | null;
}) {
  const chosen = choosePhysician({
    complaint: draft.complaint,
    blocked: draft.blocked,
    disposition: draft.disposition,
  });

  const [row] = await db.insert(queueAssignments).values({
    draftId: draft.draftId,
    caseId: draft.caseId,
    threadId: draft.threadId,
    assignedPhysicianId: chosen.physicianId,
    assignmentReason: chosen.reason,
    priority: draft.taskPriority || (draft.blocked ? "urgent" : "medium"),
  }).returning();

  return row;
}

export async function runQueueAssignmentSweep() {
  const rows = await getPendingDraftQueue({ limit: 500, offset: 0 });
  const assigned: any[] = [];

  for (const row of rows) {
    assigned.push(await assignDraft(row));
  }

  return {
    assignedCount: assigned.length,
    assigned,
  };
}

import fs from "fs";
import path from "path";
import { db } from "../db";
import { auditExports } from "../db/schema.auditExports";
import { replyDrafts } from "../db/schema.replyDrafts";
import { patientOutcomes } from "../db/schema.patientOutcomes";

const EXPORT_DIR = "/mnt/data";

function csvEscape(value: any) {
  const s = String(value ?? "");
  return `"${s.replace(/"/g, '""')}"`;
}

export async function exportAuditCsv() {
  const rows = await db.execute(sql`
    select
      d.id as draft_id,
      d.case_id,
      d.thread_id,
      d.channel,
      d.complaint,
      d.disposition,
      d.patient_label,
      d.physician_id,
      d.tone,
      d.template_key,
      d.original_patient_message,
      d.assistant_draft,
      d.edited_draft,
      d.final_sent_text,
      d.edit_distance,
      d.edit_similarity,
      d.rewrite_severity,
      d.blocked,
      d.confidence,
      d.status,
      d.created_at,
      d.approved_at,
      d.sent_at,
      o.helpful,
      o.confusing,
      o.escalated,
      o.led_to_visit,
      o.resolved_without_visit,
      o.patient_responded,
      o.response_latency_min
    from ${replyDrafts} d
    left join ${patientOutcomes} o on o.draft_id = d.id
    order by d.created_at desc
  `);

  const headers = Object.keys(rows.rows[0] || {});
  const lines = [
    headers.map(csvEscape).join(","),
    ...rows.rows.map((row: any) => headers.map(h => csvEscape(row[h])).join(",")),
  ];

  const filePath = path.join(EXPORT_DIR, `audit_export_${Date.now()}.csv`);
  fs.writeFileSync(filePath, lines.join("\n"), "utf8");

  const [record] = await db.insert(auditExports).values({
    exportType: "csv",
    requestedBy: "system",
    status: "completed",
    filePath,
  }).returning();

  return { record, filePath };
}

  const [filters, setFilters] = useState<QueueFilters>(DEFAULT_FILTERS);

  const [acceptanceSummary, setAcceptanceSummary] =
    useState<AcceptanceSummary>(DEFAULT_ACCEPTANCE);
  const [slaSummary, setSlaSummary] = useState<SlaSummary>(DEFAULT_SLA);
  const [queueSlaStatus, setQueueSlaStatus] =
    useState<QueueSlaStatus>(DEFAULT_QUEUE_SLA);
  const [outcomeSummary, setOutcomeSummary] =
    useState<OutcomeSummary>(DEFAULT_OUTCOMES);

  const [loadingTop, setLoadingTop] = useState(false);

      priority: searchParams.get("priority") || "",
      complaint: searchParams.get("complaint") || "",
      disposition: searchParams.get("disposition") || "",
      overdueOnly: searchParams.get("overdueOnly") === "true",
    });
  }, []);

    const params = new URLSearchParams();
    if (filters.search) params.set("search", filters.search);
    if (filters.status) params.set("status", filters.status);
    if (filters.channel) params.set("channel", filters.channel);
    if (filters.blocked) params.set("blocked", filters.blocked);
    if (filters.priority) params.set("priority", filters.priority);
    if (filters.complaint) params.set("complaint", filters.complaint);
    if (filters.disposition) params.set("disposition", filters.disposition);
    if (filters.overdueOnly) params.set("overdueOnly", "true");
    params.set("limit", "100");
    return params.toString();
  }, [filters]);

  useEffect(() => {
    const params = new URLSearchParams();
    if (filters.search) params.set("search", filters.search);
    if (filters.status) params.set("status", filters.status);
    if (filters.channel) params.set("channel", filters.channel);
    if (filters.blocked) params.set("blocked", filters.blocked);

        ? undefined
        : String(req.query.acknowledged).toLowerCase() === "true";

      .select()

      .where(whereParts.length ? and(...whereParts) : undefined)

    console.error("notification list error", err);
    res.status(500).json({ ok: false, error: err.message });
  }
});

import React, { useState } from "react";

export default function OutcomeQuickLog({
  draftId,
  caseId,
  threadId,
  physicianId,
  complaint,
  disposition,
  tone,
  templateKey,
  rewriteSeverity,
  onLogged,
}: {
  draftId: string;
  caseId: string;
  threadId: string;
  physicianId?: string | null;
  complaint?: string | null;
  disposition?: string | null;
  tone?: string | null;
  templateKey?: string | null;
  rewriteSeverity?: string | null;
  onLogged?: () => void;
}) {
  const [busy, setBusy] = useState("");

  async function quickLog(payload: Record<string, any>, label: string) {
    setBusy(label);
    try {
      const res = await fetch("/api/patient-outcomes", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          draftId,
          caseId,
          threadId,
          physicianId,
          complaint,
          disposition,
          tone,
          templateKey,
          rewriteSeverity,
          ...payload,
        }),
      }).then((r) => r.json());

      if (!res.ok) throw new Error(res.error || "Outcome log failed");
      onLogged?.();
    } catch (err: any) {
      alert(err.message || "Outcome log failed");
    } finally {
      setBusy("");
    }
  }

  return (
    <div className="rounded-2xl border bg-white p-4 shadow-sm">
      <div className="text-sm font-semibold mb-3">Quick Outcome Log</div>

      <div className="flex flex-wrap gap-2">
        <button
          onClick={() => quickLog({ helpful: true, patientResponded: true }, "helpful")}
          disabled={!!busy}
          className="rounded-xl border px-3 py-2 hover:bg-green-50"
        >
          {busy === "helpful" ? "Saving..." : "Helpful"}
        </button>

        <button
          onClick={() => quickLog({ confusing: true }, "confusing")}
          disabled={!!busy}
          className="rounded-xl border px-3 py-2 hover:bg-amber-50"
        >
          {busy === "confusing" ? "Saving..." : "Confusing"}
        </button>

        <button
          onClick={() => quickLog({ escalated: true, ledToVisit: true }, "escalated")}
          disabled={!!busy}
          className="rounded-xl border px-3 py-2 hover:bg-red-50"
        >
          {busy === "escalated" ? "Saving..." : "Escalated"}
        </button>

        <button
          onClick={() => quickLog({ ledToVisit: true }, "visit")}
          disabled={!!busy}
          className="rounded-xl border px-3 py-2 hover:bg-blue-50"
        >
          {busy === "visit" ? "Saving..." : "Led to Visit"}
        </button>

        <button
          onClick={() => quickLog({ resolvedWithoutVisit: true }, "resolved")}
          disabled={!!busy}
          className="rounded-xl border px-3 py-2 hover:bg-gray-50"
        >
          {busy === "resolved" ? "Saving..." : "Resolved No Visit"}
        </button>
      </div>
    </div>
  );
}

import { and, desc, eq, gte, sql } from "drizzle-orm";
import { db } from "../db";
import { notifications } from "../db/schema.notifications";

const DEFAULT_DEDUPE_MINUTES = 30;

function buildDedupeKey(input: {
  draftId?: string;
  caseId?: string;
  type: string;
  physicianId?: string;
}) {
  return [
    input.type,
    input.draftId || "no-draft",
    input.caseId || "no-case",
    input.physicianId || "no-physician",
  ].join("::");
}

export async function createNotification(input: {
  draftId?: string;
  caseId?: string;
  threadId?: string;
  physicianId?: string;
  type: string;
  channel: string;
  message: string;
  dedupeMinutes?: number;
}) {
  const dedupeKey = buildDedupeKey(input);
  const dedupeMinutes = input.dedupeMinutes ?? DEFAULT_DEDUPE_MINUTES;
  const cutoff = new Date(Date.now() - dedupeMinutes * 60 * 1000);

  const existing = await db
    .select()
    .from(notifications)
    .where(
      and(
        eq(notifications.dedupeKey, dedupeKey),
        eq(notifications.status, "open"),
        gte(notifications.createdAt, cutoff)
      )
    )
    .orderBy(desc(notifications.createdAt))
    .limit(1);

  if (existing.length) {
    return {
      ...existing[0],
      deduped: true,
    };
  }

  const [row] = await db.insert(notifications).values({
    draftId: input.draftId,
    caseId: input.caseId,
    threadId: input.threadId,
    physicianId: input.physicianId,
    type: input.type,
    channel: input.channel,
    message: input.message,
    dedupeKey,
    status: "open",
    sent: true,
  }).returning();

  return row;
}

export async function acknowledgeNotification(id: string) {
  const [row] = await db
    .update(notifications)
    .set({
      acknowledged: true,
      status: "acknowledged",
    })
    .where(eq(notifications.id, id))
    .returning();

  return row;
}

export async function resolveNotification(id: string) {
  const [row] = await db
    .update(notifications)
    .set({
      status: "resolved",
      resolvedAt: new Date(),
      acknowledged: true,
    })
    .where(eq(notifications.id, id))
    .returning();

  return row;
}

export async function resolveNotificationsByDraftAndType(input: {
  draftId: string;
  types?: string[];
}) {
  const whereBase = [eq(notifications.draftId, input.draftId), eq(notifications.status, "open")];
  if (input.types?.length) {
    whereBase.push(sql`${notifications.type} = any(${input.types})`);
  }

  return db
    .update(notifications)
    .set({
      status: "resolved",
      resolvedAt: new Date(),
      acknowledged: true,
    })
    .where(and(...whereBase))
    .returning();
}

import { and, asc, desc, eq, or } from "drizzle-orm";
import { db } from "../db";
import { messageLogs } from "../db/schema.messageLogs";
import { getReplyDraftRecord } from "./replyDraftRepository";

export async function getThreadMessagesByDraftId(draftId: string) {
  const draft = await getReplyDraftRecord(draftId);
  if (!draft) return null;

  const rows = await db
    .select()
    .from(messageLogs)
    .where(
      and(
        eq(messageLogs.caseId, draft.caseId),
        eq(messageLogs.threadId, draft.threadId)
      )
    )
    .orderBy(asc(messageLogs.createdAt));

  return {
    draft,
    messages: rows.map((m) => ({
      id: m.id,
      role:
        m.senderRole === "patient"
          ? "patient"
          : m.senderRole === "doctor"
          ? "doctor"
          : m.senderRole === "assistant"
          ? "assistant"
          : "system",
      text: m.body,
      at: m.createdAt,
      direction: m.direction,
      channel: m.channel,
    })),
  };
}

import { getReplyDraftRecord, updateReplyDraftRecord } from "./replyDraftRepository";

export async function assignNowFromNotification(notification: {
  draftId?: string | null;
  caseId?: string | null;
  threadId?: string | null;
}) {
  if (!notification.draftId || !notification.caseId || !notification.threadId) {
    throw new Error("Notification missing draft/case/thread for assignment");
  }

  const draft = await getReplyDraftRecord(notification.draftId);
  if (!draft) throw new Error("Draft not found");

  return assignDraft({
    draftId: draft.id,
    caseId: draft.caseId,
    threadId: draft.threadId,
    complaint: draft.complaint,
    blocked: !!draft.blocked,
    disposition: draft.disposition,
    taskPriority: draft.blocked ? "urgent" : "medium",
  });
}

export async function sendReminderFromNotification(notification: {
  draftId?: string | null;
  caseId?: string | null;
  threadId?: string | null;
  physicianId?: string | null;
}) {
  if (!notification.draftId) {
    throw new Error("Notification missing draftId");
  }

  const draft = await getReplyDraftRecord(notification.draftId);
  if (!draft) throw new Error("Draft not found");

  return createNotification({
    draftId: draft.id,
    caseId: draft.caseId,
    threadId: draft.threadId,
    physicianId: draft.physicianId || undefined,
    type: "approved_unsent",
    channel: "dashboard",
    message: `Reminder: draft ${draft.id} for case ${draft.caseId} still needs action.`,
    dedupeMinutes: 10,
  });
}

export async function escalateToSeniorFromNotification(notification: {
  draftId?: string | null;
  caseId?: string | null;
  threadId?: string | null;
}) {
  if (!notification.draftId) {
    throw new Error("Notification missing draftId");
  }

  const draft = await getReplyDraftRecord(notification.draftId);
  if (!draft) throw new Error("Draft not found");

    caseId: draft.caseId,
    threadId: draft.threadId,
    physicianId: "doctor-senior",
    type: "blocked_urgent",
    channel: "dashboard",
    message: `Case ${draft.caseId} escalated to senior reviewer.`,
    dedupeMinutes: 5,
  });
}

    "review_draft",

    "ack_notifications",
    "log_outcomes",

    "view_case_context",
  ],
  admin: [
    "view_cockpit",
    "review_draft",
    "override_disposition",
    "assign_queue",
    "ack_notifications",
    "resolve_notifications",
    "escalate_notification",
    "run_sweeps",
    "export_audit",
    "log_outcomes",
    "view_admin_feed",
    "view_case_context",
  ],
};

export type SessionUser = {
  userId: string;
  role: AppRole;
  permissions: AppPermission[];
};

export function getSessionUser(req: Request): SessionUser {
  const userId = String(req.headers["x-user-id"] || "user-default");
  const roleHeader = String(req.headers["x-user-role"] || "doctor").toLowerCase() as AppRole;
  const role: AppRole = ["doctor", "senior", "ops", "admin"].includes(roleHeader)
    ? roleHeader
    : "doctor";

  return {
    userId,
    role,
    permissions: ROLE_PERMISSIONS[role],
  };
}

export function hasPermission(user: SessionUser, permission: AppPermission) {
  return user.permissions.includes(permission);
}

export function requirePermission(permission: AppPermission) {
  return (req: Request, res: Response, next: NextFunction) => {
    const user = getSessionUser(req);
    if (!hasPermission(user, permission)) {
      return res.status(403).json({
        ok: false,
        error: `Missing permission: ${permission}`,
      });
    }
    (req as any).sessionUser = user;
    next();
  };
}

export async function createNotification(input: {
  draftId?: string;
  caseId?: string;
  threadId?: string;
  physicianId?: string;
  type: string;
  channel: string;
  message: string;
  dedupeMinutes?: number;
}) {
  const dedupeKey = buildDedupeKey(input);
  const dedupeMinutes = input.dedupeMinutes ?? DEFAULT_DEDUPE_MINUTES;
  const cutoff = new Date(Date.now() - dedupeMinutes * 60 * 1000);

  const existing = await db
    .select()
    .from(notifications)
    .where(
      and(

        eq(notifications.status, "open"),
        gte(notifications.createdAt, cutoff)
      )
    )
    .orderBy(desc(notifications.createdAt))
    .limit(1);

    caseId: input.caseId,

    physicianId: input.physicianId,
    type: input.type,

  let createdCount = 0;

  if (!detail.blocked) {

    resolvedCount += rows.length;
  } else {
    const row = await createNotification({
      draftId: detail.draftId,
      caseId: detail.caseId,
      threadId: detail.threadId,
      physicianId: detail.physicianId || undefined,
      type: "blocked_urgent",
      channel: "dashboard",
      message: `Blocked urgent case ${detail.caseId} requires immediate manual review.`,
      dedupeMinutes: 10,
    });
    if (!(row as any).deduped) createdCount += 1;
  }

  if (detail.status !== "approved") {
    const rows = await resolveNotificationsByDraftAndType({
      draftId,
      types: ["approved_unsent"],
    });
    resolvedCount += rows.length;
  } else if (detail.status === "approved") {
    const row = await createNotification({
      draftId: detail.draftId,
      caseId: detail.caseId,
      threadId: detail.threadId,
      physicianId: detail.physicianId || undefined,
      type: "approved_unsent",
      channel: "dashboard",
      message: `Approved draft ${detail.draftId} for case ${detail.caseId} still needs to be sent.`,
      dedupeMinutes: 10,
    });
    if (!(row as any).deduped) createdCount += 1;
  }

  if (!detail.overdue) {
    const rows = await resolveNotificationsByDraftAndType({
      draftId,
      types: ["overdue_draft"],
    });
    resolvedCount += rows.length;
  } else {
    const row = await createNotification({

      type: "overdue_draft",

      dedupeMinutes: 15,
    });
    if (!(row as any).deduped) createdCount += 1;
  }

  if (detail.disposition !== "same_day_visit") {
    const rows = await resolveNotificationsByDraftAndType({
      draftId,
      types: ["same_day_warning"],
    });
    resolvedCount += rows.length;
  } else {
    const row = await createNotification({

function computeAgeYears(dob?: string | null, ageYearsFromCase?: number | null) {
  if (ageYearsFromCase != null) return ageYearsFromCase;
  if (!dob) return null;

  const birth = new Date(dob);
  const now = new Date();
  let age = now.getFullYear() - birth.getFullYear();
  const m = now.getMonth() - birth.getMonth();

  if (m < 0 || (m === 0 && now.getDate() < birth.getDate())) age -= 1;
  return age;
}

import { sql } from "drizzle-orm";
import { db } from "../db";
import { replyDrafts } from "../db/schema.replyDrafts";

function isoDay(date: Date) {
  return date.toISOString().slice(0, 10);
}

function buildDenseSeries(
  rows: Array<{ day: string; total: number; blocked: number; heavy: number; sent: number }>,
  daysBack: number
) {
  const map = new Map(rows.map((r) => [r.day, r]));
  const out = [];

  for (let i = daysBack - 1; i >= 0; i--) {
    const d = new Date();
    d.setDate(d.getDate() - i);
    const day = isoDay(d);
    const row = map.get(day);

    out.push({
      day,
      total: row?.total || 0,
      blocked: row?.blocked || 0,
      heavy: row?.heavy || 0,
      sent: row?.sent || 0,
    });
  }

  return out;
}

export async function getCockpitTrendStrip(daysBack = 14) {
  const result = await db.execute(sql`
    select
      to_char(date_trunc('day', ${replyDrafts.createdAt}), 'YYYY-MM-DD') as day,
      count(*)::int as total,
      count(case when ${replyDrafts.blocked} = true then 1 end)::int as blocked,
      count(case when ${replyDrafts.rewriteSeverity} = 'heavy' then 1 end)::int as heavy,
      count(case when ${replyDrafts.status} = 'sent' then 1 end)::int as sent
    from ${replyDrafts}
    where ${replyDrafts.createdAt} >= now() - (${daysBack} || ' days')::interval
    group by 1
    order by 1 asc
  `);

  const dense = buildDenseSeries(result.rows as any, daysBack);

  return {
    series: {
      draftVolume: dense.map((r) => r.total),
      blocked: dense.map((r) => r.blocked),
      heavyRewrites: dense.map((r) => r.heavy),
      sent: dense.map((r) => r.sent),
    },
    labels: dense.map((r) => r.day),
    latest: dense[dense.length - 1] || {
      total: 0,
      blocked: 0,
      heavy: 0,
      sent: 0,
    },
  };
}
