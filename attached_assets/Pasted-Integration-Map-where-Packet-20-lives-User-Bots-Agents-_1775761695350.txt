▶️ 3) Template Runner (Playwright + Healing + Audit)
server/automation/templateRunner.ts
import { chromium } from "playwright";
import { healSelector } from "./selectorHealing";
import { getTemplateById } from "./templateStore";
import { auditStep } from "../audit/auditLogger";

export async function runTemplate(templateId: string, payload: any, traceId: string) {
  const template = await getTemplateById(templateId);

  const browser = await chromium.launch();
  const page = await browser.newPage();

  try {
    await page.goto(template.url);

    for (const step of template.steps) {
      if (step.selector) {
        let selector = step.selector;

        const count = await page.locator(selector).count();
        if (count === 0) {
          const healed = await healSelector(page, selector);
          if (!healed) throw new Error(`Selector failed: ${selector}`);
          selector = healed;
        }

        if (step.type === "click") {
          await page.click(selector);
        }

        if (step.type === "fill") {
          await page.fill(selector, step.value);
        }

        if (step.type === "waitForVisible") {
          await page.waitForSelector(selector);
        }
      }

      if (step.type === "waitForNetworkIdle") {
        await page.waitForLoadState("networkidle");
      }
    }

    await auditStep({
      traceId,
      step: "automation_template_success",
      input: { templateId },
      output: { ok: true },
      metadata: {}
    });

    return { ok: true };

  } catch (err) {

    await auditStep({
      traceId,
      step: "automation_template_failure",
      input: { templateId },
      output: null,
      metadata: { error: String(err) }
    });

    return { ok: false, error: String(err) };

  } finally {
    await browser.close();
  }
}

🧠 5) Oversight Agent Integration

Automation becomes observable.

server/oversight/automationMonitor.ts
export function analyzeAutomationMetrics(runs: any[]) {

  const failures = runs.filter(r => !r.ok).length;
  const rate = failures / runs.length;

  return {
    failureRate: rate,
    alert: rate > 0.1 ? "Automation instability detected" : null
  };
}

Integrate into your Oversight Agent:
