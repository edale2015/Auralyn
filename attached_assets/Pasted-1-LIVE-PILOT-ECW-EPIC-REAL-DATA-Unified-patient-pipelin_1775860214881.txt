import { runFinalPipeline } from "../clinical/finalPipeline";
import { sendToECWEncounter } from "../integrations/ecwAdapter";
import { postObservation } from "../integrations/epic";
import { processRevenue } from "../revenue/fullRevenue";

export async function runLiveSystem(patient: any) {

  // 🧠 Clinical decision
  const triage = await runFinalPipeline(patient, "live");

  const disposition = triage.safetyResult.finalDisposition;

  // 🏥 EHR writes (parallel + safe)
  await Promise.allSettled([
    sendToECWEncounter({
      patientId: patient.patientId,
      disposition
    }),
    postObservation(patient.patientId, process.env.EPIC_TOKEN!, {
      code: "Disposition",
      value: disposition,
      unit: ""
    })
  ]);

  // 💰 Revenue
  const revenue = processRevenue(patient, disposition);

  return {
    disposition,
    revenue
  };
}

export async function submitLiveClaim(claim: any) {

  const payload = {
    patientId: claim.patientId,
    cpt: claim.cpt,
    diagnosis: claim.dx,
    insurance: claim.insurance
  };

  const res = await fetch(process.env.PAYER_API!, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${process.env.PAYER_TOKEN}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify(payload)
  });

  return res.json();
}

const REGIONS = {
  east: process.env.REGION_EAST,
  west: process.env.REGION_WEST,
  eu: process.env.REGION_EU
};

export async function routeGlobal(req: any) {

  const order = ["east", "west", "eu"];

  for (const r of order) {
    try {
      const res = await fetch(REGIONS[r] + "/live/run", {
        method: "POST",
        body: JSON.stringify(req.body),
        headers: { "Content-Type": "application/json" }
      });

      if (res.ok) return res.json();

    } catch {}
  }

  throw new Error("Global failure");
}

export function matchProvider(patient: any) {

  return providers
    .filter(p => p.specialty === patient.complaint)
    .sort((a,b)=>a.load - b.load)[0];
}
