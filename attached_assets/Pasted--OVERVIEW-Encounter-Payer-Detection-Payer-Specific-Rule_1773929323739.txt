🏥 PART 1 — PAYER-SPECIFIC OPTIMIZATION ENGINE

Different payers behave like different “mini-universes” 🌌
This engine adapts coding + documentation per payer.

📁 server/billing/payerEngine.ts
type PayerRule = {
  name: string;
  preferredCPT: string[];
  requiresModifier25?: boolean;
  strictICD?: boolean;
  telehealthPolicy?: "strict" | "lenient";
};

export class PayerEngine {
  private rules: Record<string, PayerRule> = {
    "medicare": {
      name: "Medicare",
      preferredCPT: ["99213", "99214"],
      requiresModifier25: true,
      strictICD: true,
      telehealthPolicy: "strict"
    },
    "aetna": {
      name: "Aetna",
      preferredCPT: ["99213", "99214", "99215"],
      telehealthPolicy: "lenient"
    },
    "united": {
      name: "UnitedHealthcare",
      preferredCPT: ["99213", "99214"],
      strictICD: true
    }
  };

  optimize(encounter: any, payer: string) {
    const rule = this.rules[payer] || this.rules["medicare"];

    const changes: string[] = [];

    // CPT optimization
    if (!rule.preferredCPT.includes(encounter.coding.cpt[0]?.code)) {
      encounter.coding.cpt = [{ code: rule.preferredCPT[0] }];
      changes.push("Adjusted CPT for payer preference");
    }

    // Modifier 25
    if (rule.requiresModifier25) {
      encounter.coding.modifier = "25";
      changes.push("Added modifier 25");
    }

    // ICD specificity
    if (rule.strictICD && encounter.coding.icd10[0]?.code === "R69") {
      changes.push("ICD too vague for payer");
    }

    return { encounter, changes };
  }
}
🤖 PART 2 — DENIAL CLASSIFIER V2 (SMART MODEL)

This replaces simple rules with feature-based scoring.

📁 server/billing/denialClassifierV2.ts
type Features = {
  icd: string;
  cpt: string;
  payer: string;
  complexity: number;
  documentationScore: number;
};

export class DenialClassifierV2 {
  private weights = {
    icdSpecificity: 0.3,
    cptMismatch: 0.25,
    documentation: 0.2,
    payerRisk: 0.25
  };

  private payerRiskMap: Record<string, number> = {
    "medicare": 0.2,
    "aetna": 0.15,
    "united": 0.25
  };

  predict(f: Features) {
    let risk = 0;

    // ICD specificity
    if (f.icd === "R69") risk += this.weights.icdSpecificity;

    // CPT mismatch
    if (f.complexity < 0.5 && f.cpt === "99214") {
      risk += this.weights.cptMismatch;
    }

    // Documentation
    if (f.documentationScore < 0.5) {
      risk += this.weights.documentation;
    }

    // Payer baseline risk
    risk += this.payerRiskMap[f.payer] || 0.2;

    return {
      riskScore: Math.min(risk, 1),
      confidence: 1 - risk
    };
  }
}
🔗 Feature builder
export function buildFeatures(encounter: any, payer: string): Features {
  return {
    icd: encounter.coding.icd10[0]?.code,
    cpt: encounter.coding.cpt[0]?.code,
    payer,
    complexity: encounter.clinical.complexity || 0.5,
    documentationScore: encounter.clinical.reasoning ? 0.8 : 0.3
  };
}
🔧 PART 3 — PAYER-AWARE AUTO-FIX

Now fixes are context-aware per insurer.

📁 server/billing/payerAutoFix.ts
export class PayerAutoFix {
  fix(encounter: any, prediction: any, payer: string) {
    const fixes: string[] = [];

    if (prediction.riskScore > 0.5) {
      // Improve documentation
      if (!encounter.clinical.reasoning) {
        encounter.clinical.reasoning = "Expanded reasoning for payer compliance.";
        fixes.push("Added documentation");
      }

      // Adjust CPT
      if (payer === "medicare") {
        encounter.coding.cpt = [{ code: "99213" }];
        fixes.push("Adjusted CPT for Medicare");
      }

      // Add modifier
      encounter.coding.modifier = "25";
      fixes.push("Added modifier");
    }

    return { encounter, fixes };
  }
}
💰 PART 4 — CONTRACTING INTELLIGENCE ENGINE

This is where you win negotiations with payers.

📁 server/analytics/contractingEngine.ts
type PayerStats = {
  payer: string;
  avgReimbursement: number;
  denialRate: number;
  volume: number;
};

export class ContractingEngine {
  analyze(stats: PayerStats[]) {
    return stats.map(p => {
      let leverage = "low";

      if (p.volume > 1000 && p.denialRate < 0.05) {
        leverage = "high";
      }

      return {
        payer: p.payer,
        leverage,
        recommendation: this.recommend(p)
      };
    });
  }

  private recommend(p: PayerStats) {
    if (p.denialRate > 0.1) {
      return "Negotiate denial reduction terms";
    }

    if (p.avgReimbursement < 80) {
      return "Negotiate higher reimbursement";
    }

    return "Maintain contract";
  }
}

🧠 PART 6 — FULL PIPELINE (PAYER-AWARE + ML)
import { PayerEngine } from "./payerEngine";
import { DenialClassifierV2 } from "./denialClassifierV2";
import { PayerAutoFix } from "./payerAutoFix";

const payerEngine = new PayerEngine();
const classifier = new DenialClassifierV2();
const fixer = new PayerAutoFix();

export function runPayerOptimizedFlow(encounter: any, payer: string) {
  // Step 1 — optimize for payer
  let result = payerEngine.optimize(encounter, payer);

  // Step 2 — build features
  const features = buildFeatures(result.encounter, payer);

  // Step 3 — predict denial
  const prediction = classifier.predict(features);

  // Step 4 — fix if needed
  const fixed = fixer.fix(result.encounter, prediction, payer);

  return {
    encounter: fixed.encounter,
    prediction,
    fixes: [...result.changes, ...fixed.fixes]
  };
}
