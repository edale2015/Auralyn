🔐 3. AUTH + RBAC
📁 /server/auth/authMiddleware.ts
import jwt from "jsonwebtoken";

export function requireAuth(req: any, res: any, next: any) {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) return res.status(401).send("Unauthorized");

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET!);
    req.user = decoded;
    next();
  } catch {
    res.status(401).send("Invalid token");
  }
}
📁 /server/auth/roles.ts
export function requireRole(role: string) {
  return (req: any, res: any, next: any) => {
    if (req.user?.role !== role) {
      return res.status(403).send("Forbidden");
    }
    next();
  };
}
📊 4. CONTROL TOWER DASHBOARD (FRONTEND)
📁 /client/src/pages/ControlTower.tsx
import { useEffect, useState } from "react";

export default function ControlTower() {
  const [patients, setPatients] = useState<any[]>([]);
  const [alerts, setAlerts] = useState<any[]>([]);
  const [goldenStats, setGoldenStats] = useState<any>({});
  const [rlhf, setRlhf] = useState<any[]>([]);

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:3000");

    ws.onmessage = (msg) => {
      const data = JSON.parse(msg.data);
      setAlerts((prev) => [data, ...prev]);
    };

    return () => ws.close();
  }, []);

  const runGolden = async () => {
    const res = await fetch("/api/golden/run-all", { method: "POST" });
    const data = await res.json();
    setGoldenStats(data);
  };

  const loadRLHF = async () => {
    const res = await fetch("/api/rlhf/proposals");
    setRlhf(await res.json());
  };

  return (
    <div style={{ padding: 20 }}>
      <h1>🏥 Control Tower</h1>

      <button onClick={runGolden}>Run Golden Cases</button>
      <button onClick={loadRLHF}>Load RLHF</button>

      <h2>🚨 Live Alerts</h2>
      <pre>{JSON.stringify(alerts, null, 2)}</pre>

      <h2>🧪 Golden Cases</h2>
      <pre>{JSON.stringify(goldenStats, null, 2)}</pre>

      <h2>🧠 RLHF Proposals</h2>
      <pre>{JSON.stringify(rlhf, null, 2)}</pre>
    </div>
  );
}
