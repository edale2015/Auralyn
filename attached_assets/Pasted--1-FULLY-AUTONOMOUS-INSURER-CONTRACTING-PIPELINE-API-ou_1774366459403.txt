📁 server/finance/metricsEngine.ts
type Patient = {
  id: string;
  acquisitionCost: number;
  revenue: number;
};

const patients: Patient[] = [];

export function recordPatient(p: Patient) {
  patients.push(p);
}

export function computeLTV() {
  const totalRevenue = patients.reduce((s, p) => s + p.revenue, 0);
  return totalRevenue / (patients.length || 1);
}

export function computeCAC() {
  const totalCost = patients.reduce((s, p) => s + p.acquisitionCost, 0);
  return totalCost / (patients.length || 1);
}

export function computeMargin() {
  const revenue = patients.reduce((s, p) => s + p.revenue, 0);
  const cost = patients.reduce((s, p) => s + p.acquisitionCost, 0);

  return (revenue - cost) / (revenue || 1);
}
📁 server/finance/dashboardService.ts
import { computeLTV, computeCAC, computeMargin } from "./metricsEngine";

export function getFinanceDashboard() {
  return {
    LTV: computeLTV(),
    CAC: computeCAC(),
    margin: computeMargin(),
    ratio: computeLTV() / (computeCAC() || 1)
  };
}

type License = {
  physicianId: string;
  states: string[];
};

const licenses: License[] = [];

export function registerLicense(l: License) {
  licenses.push(l);
}

export function isLicensed(physicianId: string, state: string) {
  const lic = licenses.find(l => l.physicianId === physicianId);
  return lic?.states.includes(state);
}
📁 server/regulatory/complianceEngine.ts
export function checkCompliance(data: {
  physicianId: string;
  state: string;
  complaint: string;
}) {
  const licensed = isLicensed(data.physicianId, data.state);

  if (!licensed) {
    return {
      allowed: false,
      reason: "Not licensed in this state"
    };
  }

  return { allowed: true };
}

📁 Scheduling Engine
export function generateSchedule() {
  return physicians
    .filter(p => p.active)
    .sort((a, b) => a.hoursWorked - b.hoursWorked)
    .map((p, i) => ({
      physicianId: p.id,
      shift: i % 2 === 0 ? "day" : "night"
    }));
}
📁 Hiring / Firing Logic
export function evaluateWorkforce(load: number) {
  const capacity = physicians.length * 10;

  if (load > capacity * 0.8) {
    return { action: "hire", count: 2 };
  }

  if (load < capacity * 0.3) {
    return { action: "reduce_hours" };
  }

  return { action: "stable" };
}

export function updatePerformance(id: string, score: number) {

  if (p) p.performance = score;
}
