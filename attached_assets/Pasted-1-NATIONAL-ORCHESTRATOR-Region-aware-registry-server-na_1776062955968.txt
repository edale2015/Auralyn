// server/national/router.ts
import { nationalHospitals } from "./registry";
import { haversineKm, etaMinutesKm } from "../network/geo";

export function routeNational(input: {
  location: { lat: number; lon: number };
  triage: { sepsisRisk: "LOW"|"MEDIUM"|"HIGH" };
  preferredPayers: string[];
}) {
  const scored = Object.values(nationalHospitals).map(h => {
    const km = haversineKm(input.location, h.location);
    const eta = etaMinutesKm(km, 70); // faster intercity transport

    let score = 0;

    // Clinical priority
    if (input.triage.sepsisRisk === "HIGH") score += 6;

    // Capacity
    score += (h.capacity.icuAvailable > 0 ? 4 : -4);
    score += (1 - h.capacity.edLoad) * 3;

    // Payer alignment
    const payerMatch = h.payerContracts.some(p => input.preferredPayers.includes(p));
    if (payerMatch) score += 3;

    // Distance penalty
    score -= Math.min(eta / 20, 4);

    return { hospital: h, score, eta };
  });

  scored.sort((a, b) => b.score - a.score);
  return { primary: scored[0], backups: scored.slice(1, 5) };
}

// server/payer/optimizer.ts
import { payerRules } from "./payerRules";

export function optimizeForPayer(patient: any, payerId: string) {
  const rules = payerRules[payerId]?.rules;
  if (!rules) return { suggestions: [] };

  const suggestions = [];

  // Sepsis bundle compliance
  if (rules.sepsisBundleRequired && patient.sepsis.risk === "HIGH") {
    suggestions.push({
      action: "COMPLETE_SEPSIS_BUNDLE",
      steps: [
        "Lactate now",
        `Repeat lactate within ${rules.lactateRepeatWithinMin} min`,
        `Antibiotics within ${rules.antibioticsWithinMin} min`,
      ],
      revenueImpact: "HIGH",
    });
  }

  // ICU justification
  if (
    patient.sepsis.news2 >= rules.icuCriteria.news2 ||
    patient.sepsis.qsofa >= rules.icuCriteria.qsofa
  ) {
    suggestions.push({
      action: "DOCUMENT_ICU_CRITERIA",
      note: "Supports admission + reimbursement",
      revenueImpact: "HIGH",
    });
  }

  return { suggestions };
}

// server/fda/trialSimulator.ts
export function runTrial(patients: any[]) {
  let correct = 0;

  for (const p of patients) {
    const predicted = p.predictedRisk > 0.7;
    const actual = p.actualSepsis;

    if (predicted === actual) correct++;
  }

  return {
    accuracy: correct / patients.length,
    sampleSize: patients.length,
  };
}

// server/fda/auditChain.ts
import crypto from "crypto";

let lastHash = "";

export function appendAudit(record: any) {
  const data = JSON.stringify({ ...record, prev: lastHash });

  const hash = crypto.createHash("sha256").update(data).digest("hex");

  lastHash = hash;

  return { ...record, hash };
}
