1. 🔗 IMMUTABLE AUDIT HASH CHAIN (TAMPER-PROOF CORE)

Every action becomes a block in a chain. Alter one → the whole chain screams.

✅ server/audit/hashChain.ts
import crypto from "crypto";
import { db } from "../db"; // your Postgres client

export interface AuditRecordInput {
  actorId: string;
  actorRole: string;
  clinicId: string;
  action: string;
  resourceType: string;
  resourceId?: string;
  payload?: any;
}

function stableStringify(obj: any): string {
  return JSON.stringify(obj, Object.keys(obj).sort());
}

export async function getLastHash(): Promise<string> {
  const res = await db.query(
    `SELECT hash FROM audit_log ORDER BY id DESC LIMIT 1`
  );
  return res.rows[0]?.hash || "GENESIS";
}

export async function appendAuditRecord(input: AuditRecordInput) {
  const prevHash = await getLastHash();

  const record = {
    ...input,
    timestamp: new Date().toISOString(),
  };

  const dataString = stableStringify(record) + prevHash;

  const hash = crypto
    .createHash("sha256")
    .update(dataString)
    .digest("hex");

  await db.query(
    `INSERT INTO audit_log 
     (actor_id, actor_role, clinic_id, action, resource_type, resource_id, payload, prev_hash, hash)
     VALUES ($1,$2,$3,$4,$5,$6,$7,$8,$9)`,
    [
      input.actorId,
      input.actorRole,
      input.clinicId,
      input.action,
      input.resourceType,
      input.resourceId || null,
      JSON.stringify(input.payload || {}),
      prevHash,
      hash,
    ]
  );

  return hash;
}

import { Request, Response, NextFunction } from "express";
import { appendAuditRecord } from "../audit/hashChain";

export function auditAction(action: string, resourceType: string) {
  return async (req: Request, res: Response, next: NextFunction) => {
    const originalSend = res.send;

    res.send = async function (body: any) {
      try {
        const physician = (req as any).physician;

        if (physician) {
          await appendAuditRecord({
            actorId: physician.id,
            actorRole: physician.role,
            clinicId: physician.clinicId,
            action,
            resourceType,
            resourceId: req.params?.id,
            payload: {
              request: req.body,
              response: body ? JSON.parse(body) : null,
              path: req.path,
              method: req.method,
            },
          });
        }
      } catch (err) {
        console.error("[AUDIT ERROR]", err);
      }

      return originalSend.call(this, body);
    };

    next();
  };
}

export function requireRole(roles: string[]) {
  return (req: any, res: any, next: any) => {
    const user = req.physician;

    if (!user) {
      return res.status(401).json({ error: "Unauthorized" });
    }

    if (!roles.includes(user.role)) {
      return res.status(403).json({
        error: "Forbidden",
        required: roles,
        actual: user.role,
      });
    }

    next();
  };
}

export function validateWeightUpdate(oldWeights: any, newWeights: any) {
  const MAX_DELTA = 0.02;

  for (const key of Object.keys(newWeights)) {
    const delta = Math.abs(newWeights[key] - (oldWeights[key] || 0));

    if (delta > MAX_DELTA) {
      throw new Error(`Unsafe weight change for ${key}`);
    }
  }

  return true;
}

8. 📊 AUDIT VERIFICATION ENDPOINT
✅ routes/governanceRoutes.ts
import { Router } from "express";
import crypto from "crypto";
import { db } from "../db";

const router = Router();

router.get("/verify-chain", async (_req, res) => {
  const rows = await db.query(
    `SELECT * FROM audit_log ORDER BY id ASC`
  );

  let prevHash = "GENESIS";

  for (const row of rows.rows) {
    const data = JSON.stringify({
      actorId: row.actor_id,
      action: row.action,
      timestamp: row.created_at,
    }) + prevHash;

    const expected = crypto.createHash("sha256").update(data).digest("hex");

    if (expected !== row.hash) {
      return res.json({
        valid: false,
        brokenAt: row.id,
      });
    }

    prevHash = row.hash;
  }

  res.json({ valid: true });
});

export default router;
