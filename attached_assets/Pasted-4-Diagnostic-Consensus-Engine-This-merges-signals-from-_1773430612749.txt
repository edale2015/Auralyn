3. Evidence Fusion Engine

This engine combines all signals.

server/diagnostics/evidenceFusionEngine.ts
import { EVIDENCE_WEIGHTS }
from "./evidenceWeightConfig"

export function fuseDiagnosticEvidence(
  signals:any
){

  const votes:Record<string,number> = {}

  function applyVotes(source:any[],weight:number){

    if(!source) return

    for(const s of source){

      const dx =
        s.diagnosis ?? s

      const score =
        (s.score ?? 1) * weight

      votes[dx] =
        (votes[dx] ?? 0) + score

    }

  }

  applyVotes(
    signals.ruleEngine,
    EVIDENCE_WEIGHTS.ruleEngine
  )

  applyVotes(
    signals.similarityEngine,
    EVIDENCE_WEIGHTS.similarityEngine
  )

  applyVotes(
    signals.evidenceGraph,
    EVIDENCE_WEIGHTS.evidenceGraph
  )

  applyVotes(
    signals.epidemiology,
    EVIDENCE_WEIGHTS.epidemiologyEngine
  )

  return Object.entries(votes)
    .map(([diagnosis,score])=>({

      diagnosis,
      score

    }))
    .sort((a,b)=>b.score-a.score)

}

import { TRIAGE_RULES }
from "./triageRiskRules"

export function computeTriagePriority(
  symptoms:string
){

  const s = symptoms.toLowerCase()

  let priority = 0
  let reason = ""

  for(const rule of TRIAGE_RULES){

    let matches = 0

    for(const k of rule.keywords)
      if(s.includes(k))
        matches++

    if(matches >= 2){

      if(rule.score > priority){

        priority = rule.score
        reason = rule.name

      }

    }

  }

  return {

    priority,
    reason,

    level:
      priority >= 0.8
        ? "critical"
        : priority >= 0.5
        ? "urgent"
        : "routine"

  }

}

outcomeCalibrationEngine.ts
import { loadCalibrationData, saveCalibrationData }
from "./calibrationDataStore"

export async function calibrateDiagnosticProbability(
  diagnosis:string,
  predicted:number,
  actual:boolean
){

  const data = await loadCalibrationData()

  data[diagnosis] ??= {
    count:0,
    success:0
  }

  data[diagnosis].count += 1

  if(actual)
    data[diagnosis].success += 1

  await saveCalibrationData(data)

}

export async function adjustProbability(
  diagnosis:string,
  rawProbability:number
){

  const data = await loadCalibrationData()

  const stats = data[diagnosis]

  if(!stats)
    return rawProbability

  const observed =
    stats.success / stats.count

  return (
    rawProbability * 0.7 +
    observed * 0.3
  )

}


This gradually corrects systematic bias in your predictions.

import { computeTriagePriority }
from "../triage/triagePrioritizationEngine"

import { analyzeTemporalProgression }
from "../temporal/temporalSymptomEngine"

import { computePatientRiskProfile }
from "../risk/patientRiskProfileEngine"

import { computeFusedDifferential }
from "../diagnostics/fusedDifferentialService"

import { applyEpidemiologyAdjustment }
from "../epidemiology/epidemiologySignalService"

import { detectContradictions }
from "../diagnostics/differentialContradictionEngine"

import { buildResourcePlan }
from "../resources/resourcePlanService"

import { executeCarePathway }
from "../pathways/carePathwayOrchestrator"

export async function runClinicalBrain(state:any){

  const triage =
    computeTriagePriority(state.symptoms)

  const temporal =
    analyzeTemporalProgression(state.events)

  const risk =
    computePatientRiskProfile(state.patient)

  let differential =
    computeFusedDifferential(state)

  differential =
    await applyEpidemiologyAdjustment(differential)

  const contradictions =
    detectContradictions(
      differential,
      state.symptoms
    )

  const resources =
    buildResourcePlan({
      ...state,
      differential
    })

  const pathway =
    executeCarePathway(
      differential[0]?.diagnosis
    )

  return {

    triage,
    temporal,
    risk,
    differential,
    contradictions,
    resources,
    pathway

  }

}

type Job = {
  type: string
  payload: any
}

const queue: Job[] = []

export function enqueueJob(job: Job) {
  queue.push(job)
}

export function dequeueJob(): Job | undefined {
  return queue.shift()
}

Event Processor
server/pipeline/eventProcessor.ts
import { dequeueJob } from "./eventQueue"
import { runReasoningWorker } from "./reasoningWorker"
import { runLearningWorker } from "./learningWorker"

export async function processEventLoop(){

  while(true){

    const job = dequeueJob()

    if(!job){

      await new Promise(r=>setTimeout(r,50))
      continue

    }

    if(job.type === "reasoning"){
      await runReasoningWorker(job.payload)
    }

    if(job.type === "learning"){
      await runLearningWorker(job.payload)
    }

  }

}
