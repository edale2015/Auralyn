2) Exact computeProposal (paste-ready)
type ComputeCtx = {
  flowId?: string;
  system?: string;
  complaint?: string;
};

type Proposal = {
  disposition: string;
  redFlag: boolean;
  rulePacks: {
    testPack?: string;
    medPack?: string;
    referralPack?: string;
  };
  reasoning: string[];
};

export async function computeProposal(
  answers: Record<string, any>,
  ctx?: ComputeCtx
): Promise<Proposal> {

  // ---------- 0) Resolve flow context ----------
  const flowId = ctx?.flowId || "ENT_FLU_LIKE_V1";

  // ---------- 1) Determine redFlag (flow-aware) ----------
  const redFlagQids = RED_FLAG_MAP[flowId] || [];
  const redFlag = redFlagQids.some(qid => answers[qid] === "Yes");

  // ---------- 2) Load KV rules for this flow ----------
  const rules = await getRulesForFlow(flowId);

  // ---------- 3) Resolve disposition ----------
  // Try flow-specific keys first, then fallbacks
  const redDisp =
    rules[`${flowId}_RED_FLAG_DISPOSITION`] ||
    rules["RED_FLAG_DISPOSITION"] ||
    "urgent_or_ed";

  const nonRedDisp =
    rules[`${flowId}_NON_RED_FLAG_DISPOSITION`] ||
    rules["NON_RED_FLAG_DISPOSITION"] ||
    "routine_or_supportive";

  const disposition = redFlag ? redDisp : nonRedDisp;

  // ---------- 4) Resolve rule packs (optional, non-breaking) ----------
  const testPack =
    rules[`${flowId}_TEST_PACK`] ||
    rules["TEST_PACK"] ||
    undefined;

  const medPack =
    rules[`${flowId}_MED_PACK`] ||
    rules["MED_PACK"] ||
    undefined;

  const referralPack =
    rules[`${flowId}_REFERRAL_PACK`] ||
    rules["REFERRAL_PACK"] ||
    undefined;

  // ---------- 5) Build reasoning (human-readable + audit-safe) ----------
  const reasoning: string[] = [];

  if (redFlag) {
    reasoning.push(
      `Red flag triggered by answers: ${redFlagQids.filter(q => answers[q] === "Yes").join(", ")}`
    );
    reasoning.push(`Disposition set to "${disposition}" per red-flag rules`);
  } else {
    reasoning.push("No red-flag answers detected");
    reasoning.push(`Disposition set to "${disposition}" per non-red-flag rules`);
  }

  if (testPack) reasoning.push(`Test pack suggested: ${testPack}`);
  if (medPack) reasoning.push(`Medication pack suggested: ${medPack}`);
  if (referralPack) reasoning.push(`Referral pack suggested: ${referralPack}`);

  // ---------- 6) Return normalized proposal ----------
  return {
    disposition,
    redFlag,
    rulePacks: {
      testPack,
      medPack,
      referralPack,
    },
    reasoning,
  };
}

// server/rules/kvRuleLoader.ts

import { getSheetRows } from "../sheets/sheetHelper";

export async function getRulesForFlow(flowId: string): Promise<Record<string, string>> {
  const { rowsAsObjects } = await getSheetRows("CLINICAL_RULES");

  const out: Record<string, string> = {};

  for (const r of rowsAsObjects) {
    const fid = String(r.flow_id || "").trim();
    if (fid !== flowId) continue;

    const active = String(r.active || "Y").trim().toUpperCase();
    if (active === "N") continue;

    const key = String(r.rule_key || "").trim();
    if (!key) continue;

    out[key] = String(r.value ?? "").trim();
  }

  return out;
}
