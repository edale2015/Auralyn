1. MULTI-PHYSICIAN ROUTING ENGINE

This assigns cases intelligently based on load, specialty, and risk.

/server/services/physicianRouter.ts
export type PhysicianProfile = {
  id: string;
  clinicId: string;
  name: string;
  role: "physician" | "pa" | "np" | "admin";
  specialties: string[];
  active: boolean;
  currentLoad: number;
  maxConcurrent: number;
  canReviewHighRisk: boolean;
};

export type RoutedCaseInput = {
  clinicId: string;
  complaint: string;
  riskLevel: "LOW" | "MEDIUM" | "HIGH";
  preferredSpecialty?: string;
};

function specialtyScore(physician: PhysicianProfile, preferred?: string) {
  if (!preferred) return 0;
  return physician.specialties.includes(preferred) ? 3 : 0;
}

function highRiskScore(physician: PhysicianProfile, riskLevel: string) {
  if (riskLevel !== "HIGH") return 0;
  return physician.canReviewHighRisk ? 4 : -100;
}

function loadScore(physician: PhysicianProfile) {
  const remaining = physician.maxConcurrent - physician.currentLoad;
  return remaining;
}

export function routeCaseToPhysician(
  physicians: PhysicianProfile[],
  input: RoutedCaseInput
) {
  const candidates = physicians.filter(
    p =>
      p.clinicId === input.clinicId &&
      p.active &&
      p.currentLoad < p.maxConcurrent &&
      p.role !== "admin"
  );

  if (!candidates.length) {
    return {
      assignedPhysicianId: null,
      routingReason: "No available clinician"
    };
  }

  const ranked = candidates
    .map(p => ({
      physician: p,
      score:
        specialtyScore(p, input.preferredSpecialty) +
        highRiskScore(p, input.riskLevel) +
        loadScore(p)
    }))
    .sort((a, b) => b.score - a.score);

  return {
    assignedPhysicianId: ranked[0].physician.id,
    routingReason: `Assigned by score ${ranked[0].score}`
  };
}
2. ESCALATION + SLA QUEUE

High-risk and aging cases should rise like cream. Quietly. Reliably.

/server/services/caseQueue.ts
export type QueueCase = {
  id: string;
  clinicId: string;
  assignedPhysicianId: string | null;
  riskLevel: "LOW" | "MEDIUM" | "HIGH";
  createdAt: number;
  status: "pending" | "reviewed" | "escalated";
  confidence: number;
};

export function calculateCasePriority(qc: QueueCase) {
  let score = 0;

  if (qc.riskLevel === "HIGH") score += 100;
  if (qc.riskLevel === "MEDIUM") score += 40;
  if (qc.confidence < 0.7) score += 30;

  const ageMinutes = Math.floor((Date.now() - qc.createdAt) / 60000);
  score += Math.min(ageMinutes, 60);

  return score;
}

export function sortQueue(cases: QueueCase[]) {
  return [...cases].sort((a, b) => calculateCasePriority(b) - calculateCasePriority(a));
}

export function shouldEscalateCase(qc: QueueCase) {
  const ageMinutes = Math.floor((Date.now() - qc.createdAt) / 60000);

  if (qc.riskLevel === "HIGH" && ageMinutes >= 5) return true;
  if (qc.riskLevel === "MEDIUM" && ageMinutes >= 20) return true;
  if (qc.confidence < 0.5 && ageMinutes >= 10) return true;

  return false;
}

import express from "express";

    escalatedCount: escalated.length,
    escalated
  });

import { Request, Response, NextFunction } from "express";

export function tenantGuard(req: Request, res: Response, next: NextFunction) {
  const clinicId = req.headers["x-clinic-id"];

  if (!clinicId || typeof clinicId !== "string") {
    return res.status(400).json({
      error: "Missing clinic context"
    });
  }

  res.locals.clinicId = clinicId;
  next();
}

4. LIVE OPERATIONS DASHBOARD API

This gives you the pulse of the system.

/server/services/opsMetrics.ts
export type OpsMetricInput = {
  totalCases: number;
  reviewedCases: number;
  escalatedCases: number;
  overrides: number;
  avgReviewSeconds: number;
  satisfactionAverage: number;
};

export function buildOpsSnapshot(input: OpsMetricInput) {
  const pendingCases = input.totalCases - input.reviewedCases;
  const reviewRate = input.totalCases === 0 ? 0 : input.reviewedCases / input.totalCases;
  const overrideRate = input.reviewedCases === 0 ? 0 : input.overrides / input.reviewedCases;
  const escalationRate = input.totalCases === 0 ? 0 : input.escalatedCases / input.totalCases;

  return {
    timestamp: new Date().toISOString(),
    totals: {
      totalCases: input.totalCases,
      reviewedCases: input.reviewedCases,
      pendingCases,
      escalatedCases: input.escalatedCases
    },
    rates: {
      reviewRate,
      overrideRate,
      escalationRate
    },
    performance: {
      avgReviewSeconds: input.avgReviewSeconds,
      satisfactionAverage: input.satisfactionAverage
    }
  };
}

5. OUTCOME DRIFT MONITOR

This watches for silent decay. The kind that arrives in a suit and smiles.

/server/services/driftMonitor.ts
export type OutcomeRecord = {
  timestamp: string;
  correct: boolean;
  complaint: string;
};

export function computeWindowAccuracy(records: OutcomeRecord[]) {
  if (!records.length) return 0;
  return records.filter(r => r.correct).length / records.length;
}

export function detectOutcomeDrift(
  baseline: OutcomeRecord[],
  recent: OutcomeRecord[],
  thresholdDrop = 0.08
) {
  const baselineAccuracy = computeWindowAccuracy(baseline);
  const recentAccuracy = computeWindowAccuracy(recent);
  const delta = baselineAccuracy - recentAccuracy;

  return {
    baselineAccuracy,
    recentAccuracy,
    delta,
    driftDetected: delta >= thresholdDrop,
    recommendedAction:
      delta >= thresholdDrop
        ? "Increase caution, require more physician review, inspect complaint clusters"
        : "Stable"
  };
}

6. PHYSICIAN APPROVAL RULES ENGINE

This determines what can be auto-approved, what must be reviewed, and what gets sirens.

/server/services/approvalRules.ts
export type ApprovalInput = {
  riskLevel: "LOW" | "MEDIUM" | "HIGH";
  confidence: number;
  hasSafetyAlerts: boolean;
  disposition: string;
};

export function evaluateApprovalRule(input: ApprovalInput) {
  if (input.riskLevel === "HIGH") {
    return {
      action: "mandatory_review",
      reason: "High-risk cases require physician review"
    };
  }

  if (input.hasSafetyAlerts) {
    return {
      action: "mandatory_review",
      reason: "Safety alerts present"
    };
  }

  if (input.confidence < 0.75) {
    return {
      action: "mandatory_review",
      reason: "Confidence below threshold"
    };
  }

  if (input.disposition === "er_now" || input.disposition === "urgent_now") {
    return {
      action: "mandatory_review",
      reason: "High-severity disposition"
    };
  }

  return {
    action: "eligible_for_batch_approval",
    reason: "Low-risk, high-confidence, no safety alerts"
  };
}

/server/middleware/roleGuard.ts
import { Request, Response, NextFunction } from "express";

export function roleGuard(allowedRoles: string[]) {
  return (req: Request, res: Response, next: NextFunction) => {
    const role = req.headers["x-user-role"];

    if (!role || typeof role !== "string") {
      return res.status(401).json({ error: "Missing user role" });
    }

    if (!allowedRoles.includes(role)) {
      return res.status(403).json({ error: "Forbidden" });
    }

    next();
  };
}

CODE FOR B: PER-COMPLAINT ANALYTICS
/server/services/complaintAnalytics.ts
export type ComplaintOutcome = {
  complaint: string;
  correct: boolean;
  escalated: boolean;
};

export function buildComplaintAnalytics(rows: ComplaintOutcome[]) {
  const map: Record<string, { total: number; correct: number; escalated: number }> = {};

  for (const row of rows) {
    if (!map[row.complaint]) {
      map[row.complaint] = { total: 0, correct: 0, escalated: 0 };
    }

    map[row.complaint].total += 1;
    if (row.correct) map[row.complaint].correct += 1;
    if (row.escalated) map[row.complaint].escalated += 1;
  }

  return Object.entries(map).map(([complaint, v]) => ({
    complaint,
    total: v.total,
    accuracy: v.total ? v.correct / v.total : 0,
    escalationRate: v.total ? v.escalated / v.total : 0
  }));
}

import crypto from "crypto";

export type AuditEntryInput = {
  timestamp: string;
  caseId: string;
  userId: string;
  action: string;
  payload: unknown;
};

export function hashAuditEntry(entry: AuditEntryInput, previousHash: string) {
  const serialized = JSON.stringify({
    ...entry,
    previousHash
  });

  return crypto.createHash("sha256").update(serialized).digest("hex");
}
