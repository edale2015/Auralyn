import { Queue } from "bullmq";
import { v4 as uuidv4 } from "uuid";

const agentQueue = new Queue("agent-execution", {
  connection: { url: process.env.REDIS_URL }
});

export interface AgentTask {
  id: string;
  type: "diagnosis" | "triage" | "treatment" | "refactor";
  input: any;
  model: string;
}

export async function runAgentFleet(tasks: AgentTask[]) {
  const jobs = tasks.map(task => {
    return agentQueue.add("run-agent", task);
  });

  return Promise.all(jobs);
}

📁 server/agents/runAgent.ts
import OpenAI from "openai";

const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export async function runAgent(task: any) {
  const prompt = buildPrompt(task);

  const response = await client.chat.completions.create({
    model: task.model,
    messages: [{ role: "user", content: prompt }],
    temperature: 0.2
  });

  return {
    taskId: task.id,
    output: response.choices[0].message.content
  };
}

function buildPrompt(task: any) {
  return `
You are a clinical reasoning agent.

Task: ${task.type}
Input: ${JSON.stringify(task.input)}

Return structured JSON:
{
  "diagnosis": [],
  "confidence": number,
  "reasoning": []
}
`;
}
🧠 2. MULTI-AGENT COUNCIL (KEY UPGRADE)
📁 server/agents/agentCouncil.ts
export function aggregateAgentResults(results: any[]) {
  const diagnosisMap: Record<string, number> = {};

  for (const r of results) {
    const parsed = JSON.parse(r.output);

    for (const dx of parsed.diagnosis) {
      diagnosisMap[dx] = (diagnosisMap[dx] || 0) + parsed.confidence;
    }
  }

  const ranked = Object.entries(diagnosisMap)
    .map(([dx, score]) => ({ dx, score }))
    .sort((a, b) => b.score - a.score);

  return ranked;
}

import { execSync } from "child_process";
import path from "path";

export function createWorktree(branchName: string) {
  const repoPath = process.cwd();
  const worktreePath = path.join(repoPath, "worktrees", branchName);

  execSync(`git worktree add ${worktreePath} -b ${branchName}`, {
    stdio: "inherit"
  });

  return worktreePath;
}

export async function bestOfN(input: any) {
  const models = ["gpt-4.1", "gpt-4o", "gpt-5"];

  const tasks = models.map(model => ({
    id: crypto.randomUUID(),
    type: "diagnosis",
    input,
    model
  }));

  const results = await runAgentFleet(tasks);

  return results;
}

🧠 6. AGENT MEMORY (PERSISTENCE)
📁 server/agents/agentMemory.ts
import { db } from "../db";

export async function saveAgentMemory(agentId: string, memory: any) {
  await db.insert("agent_memory").values({
    agentId,
    memory: JSON.stringify(memory),
    createdAt: new Date()
  });
}

export async function getAgentMemory(agentId: string) {
  return db
    .select()
    .from("agent_memory")
    .where({ agentId })
    .orderBy("createdAt DESC")
    .limit(10);
}
