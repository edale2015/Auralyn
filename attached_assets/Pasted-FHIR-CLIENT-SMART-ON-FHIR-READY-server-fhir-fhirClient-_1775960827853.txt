// server/fhir/fhirClient.ts

import fetch from "node-fetch"

const FHIR_BASE = process.env.FHIR_BASE_URL

export async function getPatient(patientId: string) {
  const res = await fetch(`${FHIR_BASE}/Patient/${patientId}`, {
    headers: {
      Authorization: `Bearer ${process.env.FHIR_TOKEN}`
    }
  })

  return res.json()
}
🧾 CREATE ENCOUNTER (BILLABLE VISIT)
// server/fhir/createEncounter.ts

export async function createEncounter(patientId: string) {
  const encounter = {
    resourceType: "Encounter",
    status: "finished",
    subject: {
      reference: `Patient/${patientId}`
    }
  }

  await fetch(`${FHIR_BASE}/Encounter`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${process.env.FHIR_TOKEN}`
    },
    body: JSON.stringify(encounter)
  })
}

export function generateCPT(ctx: any) {
  if (ctx.complexity === "high") {
    return ["99285"] // ER high complexity
  }

  if (ctx.complexity === "moderate") {
    return ["99284"]
  }

  return ["99283"]
}

// server/fda/validationEngine.ts

export function validateCase(result: any, expected: any) {
  const correct = result.diagnosis === expected.diagnosis

  return {
    correct,
    riskLevel: result.risk || "unknown"
  }
}
📊 METRICS AGGREGATOR
// server/fda/metrics.ts

export function computeMetrics(cases: any[]) {
  let correct = 0

  cases.forEach(c => {
    if (c.correct) correct++
  })

  return {
    accuracy: correct / cases.length,
    total: cases.length
  }
}
