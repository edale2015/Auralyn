1. Outcome-Weighted Similarity
server/similarity/outcomeWeightedSimilarity.ts
import { scoreCaseSimilarity } from "./similarityScorer"

function outcomeWeight(row:any){

  if(!row.outcome) return 0.5

  let weight = 1

  if(row.outcome.topPredictionMatch)
    weight += 0.5

  if(row.outcome.dispositionMatch)
    weight += 0.5

  if(row.outcome.safetyMiss)
    weight -= 1

  return Math.max(0,weight)
}

export function scoreOutcomeWeightedSimilarity(
  current:any,
  prior:any
){

  const similarity = scoreCaseSimilarity(current,prior)

  const weight = outcomeWeight(prior)

  return similarity * weight
}

This means cases with good outcomes count more.

2. Differential Vote Aggregator

Each similar case casts a vote for its final diagnosis.

server/similarity/differentialVoteAggregator.ts
export function aggregateDifferentialVotes(similarCases:any[]){

  const votes:Record<string,number> = {}

  for(const c of similarCases){

    const diagnosis =
      c.outcome?.actualDiagnosis ||
      c.differential?.[0]

    if(!diagnosis) continue

    const weight = c.similarityScore ?? 0

    votes[diagnosis] =
      (votes[diagnosis] ?? 0) + weight
  }

  return Object.entries(votes)
    .map(([diagnosis,score])=>({
      diagnosis,
      score
    }))
    .sort((a,b)=>b.score-a.score)
}

import { extractCaseFeatures } from "./caseFeatureExtractor"
import { loadSimilarityIndex } from "./caseSimilarityStore"
import { scoreOutcomeWeightedSimilarity } from "./outcomeWeightedSimilarity"
import { aggregateDifferentialVotes } from "./differentialVoteAggregator"

export async function computeSimilarityWeightedDifferential(state:any){

  const current = extractCaseFeatures(state)

  const index = await loadSimilarityIndex()

  const scored = index
    .map((row:any)=>({

      ...row,

      similarityScore:
        scoreOutcomeWeightedSimilarity(
          current,
          row
        )

    }))
    .sort((a:any,b:any)=>
      b.similarityScore-a.similarityScore
    )
    .slice(0,10)

  const votes =
    aggregateDifferentialVotes(scored)

  return {
    similarCases:scored,
    voteDifferential:votes
  }

}
