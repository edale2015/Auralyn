1) Production server entry
server/index.ts
import http from 'http';
import express from 'express';
import helmet from 'helmet';
import compression from 'compression';
import cors from 'cors';

import diagnosisRoutes from './routes/diagnosisRoutes';
import controlTowerRoutes from './routes/controlTowerRoutes';
import validationRoutes from './routes/validationRoutes';
import controlTowerValidationRoutes from './routes/controlTowerValidationRoutes';
import clinicalAnswerRoute from './routes/clinicalAnswerRoute';
import adminClaudeExportRoutes from './routes/adminClaudeExportRoutes';
import metricsRoutes from './monitoring/metricsRoute';
import { initWebSocket } from './ws/patientStream';
import { validateEnv } from './startup/validateEnv';

validateEnv();

const app = express();

app.use(helmet());
app.use(compression());
app.use(cors({ origin: true, credentials: true }));
app.use(express.json({ limit: '4mb' }));

app.get('/api/healthz', (_req, res) => {
  res.json({
    ok: true,
    service: 'auralyn',
    ts: new Date().toISOString(),
    mode: process.env.NODE_ENV || 'development',
  });
});

app.use('/api/diagnosis', diagnosisRoutes);
app.use('/api/control-tower', controlTowerRoutes);
app.use('/api/validation', validationRoutes);
app.use('/api/control-tower', controlTowerValidationRoutes);
app.use('/api/admin', adminClaudeExportRoutes);
app.use(clinicalAnswerRoute);
app.use(metricsRoutes);

const port = Number(process.env.PORT || 3000);
const server = http.createServer(app);

initWebSocket(server);

server.listen(port, '0.0.0.0', () => {
  console.log(`Auralyn listening on ${port}`);
});
2) Environment validation
server/startup/validateEnv.ts
const REQUIRED_VARS = [
  'DATABASE_URL',
];

const OPTIONAL_WARN_VARS = [
  'FHIR_BASE',
  'FHIR_TOKEN_URL',
  'FHIR_CLIENT_ID',
  'FHIR_SECRET',
  'CLAUDE_API_KEY',
];

export function validateEnv() {
  const missing = REQUIRED_VARS.filter((k) => !process.env[k]);

  if (missing.length) {
    throw new Error(`Missing required env vars: ${missing.join(', ')}`);
  }

  const optionalMissing = OPTIONAL_WARN_VARS.filter((k) => !process.env[k]);
  if (optionalMissing.length) {
    console.warn(`Optional env vars missing: ${optionalMissing.join(', ')}`);
  }
}

10) Production readiness guard

This blocks boot if you forgot critical settings.

server/startup/productionGuard.ts
export function productionGuard() {
  const isProd = process.env.NODE_ENV === 'production';
  if (!isProd) return;

  const failures: string[] = [];

  if (!process.env.DATABASE_URL) failures.push('DATABASE_URL missing');
  if (!process.env.PORT) failures.push('PORT missing');

  if (failures.length) {
    throw new Error(`Production guard failed: ${failures.join('; ')}`);
  }
}

Call it in server/index.ts after validateEnv().

What I think is most helpful here

The most useful deployment pieces are:

one-click cloud configs

That combination keeps deployment from becoming a trapdoor.
