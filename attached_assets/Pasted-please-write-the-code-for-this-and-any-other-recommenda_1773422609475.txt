1. Event Stream (Production-Ready)

Instead of a simple NDJSON file, we add:

append

read by case

read since offset

indexing

server/core/events/eventStream.ts
import fs from "fs/promises"
import path from "path"

const EVENTS_FILE = path.resolve(
  process.cwd(),
  "server/data/runtime/clinical_events.ndjson"
)

export async function appendEvent(event:any){

  await fs.mkdir(
    path.dirname(EVENTS_FILE),
    { recursive:true }
  )

  await fs.appendFile(
    EVENTS_FILE,
    JSON.stringify(event) + "\n"
  )
}

export async function readEvents(caseId?:string){

  const raw = await fs.readFile(EVENTS_FILE,"utf8")

  const rows = raw
    .split("\n")
    .filter(Boolean)
    .map(JSON.parse)

  if(!caseId) return rows

  return rows.filter(e => e.caseId === caseId)
}

const cache:Record<string,any> = {}

export async function getCachedState(caseId:string){

  if(cache[caseId])
    return cache[caseId]

  const state = await rebuildClinicalState(caseId)

  cache[caseId] = state

  return state
}

export function invalidateState(caseId:string){

  delete cache[caseId]

}

Whenever an event is published, invalidate cache.

4. State Projector
server/core/state/clinicalStateProjector.ts
import { readEvents } from "../events/eventStream"

export async function rebuildClinicalState(caseId:string){

  const events = await readEvents(caseId)

  const state:any = {
    caseId,
    symptoms:"",
    differential:[],
    alerts:[]
  }

  for(const e of events){

    switch(e.type){

      case "PATIENT_MESSAGE":

        state.symptoms += " " + e.payload.message
        break

      case "COMPLAINT_IDENTIFIED":

        state.complaint = e.payload.complaint
        break

      case "DIFFERENTIAL_UPDATED":

        state.differential = e.payload.differential
        break

      case "ALERTS_UPDATED":

        state.alerts = e.payload.alerts
        break

      case "MEDICATION_PLAN":

        state.medication = e.payload.medication
        break

      case "DISPOSITION_SET":

        state.disposition = e.payload.disposition
        break

      case "NOTE_READY":

        state.chartNote = e.payload.note
        break
    }

  }

  return state
}
5. Reasoning Worker

Workers consume events and run reasoning.

server/core/workers/reasoningWorker.ts
import { rebuildClinicalState } from "../state/clinicalStateProjector"
import { publishEvent } from "../events/eventPublisher"

import { generateDifferential } from "../../assistant/telemedicineDifferentialService"
import { checkSafetyAlerts } from "../../assistant/telemedicineSafetyService"
import { generatePrescription } from "../../assistant/telemedicinePrescriptionService"

export async function runReasoning(caseId:string){

  const state = await rebuildClinicalState(caseId)

  if(!state.symptoms) return

  const complaint =
    state.symptoms.includes("cough")
      ? "cough"
      : state.symptoms.includes("throat")
      ? "sore_throat"
      : "unknown"

  await publishEvent(
    caseId,
    "COMPLAINT_IDENTIFIED",
    { complaint }
  )

  const diff = generateDifferential({
    complaint,
    symptoms:state.symptoms
  })

  await publishEvent(
    caseId,
    "DIFFERENTIAL_UPDATED",
    { differential:diff }
  )

  const alerts = checkSafetyAlerts(state.symptoms)

  await publishEvent(
    caseId,
    "ALERTS_UPDATED",
    { alerts }
  )

  const med = generatePrescription({ complaint })

  await publishEvent(
    caseId,
    "MEDICATION_PLAN",
    { medication:med }
  )

}

7. Differential Aggregator (Major Intelligence Upgrade)

Instead of a single rule engine, combine multiple signals.

server/assistant/differentialAggregator.ts
export function aggregateDifferential(params:{
  complaint:string
  symptoms:string
}){

  const results:any[] = []

  if(params.complaint==="sore_throat"){

    results.push({
      diagnosis:"Strep Pharyngitis",
      score: params.symptoms.includes("fever") ? 0.7 : 0.4
    })

    results.push({
      diagnosis:"Viral Pharyngitis",
      score: params.symptoms.includes("cough") ? 0.7 : 0.5
    })

  }

  if(params.complaint==="cough"){

    results.push({
      diagnosis:"Viral Bronchitis",
      score:0.6
    })

    results.push({
      diagnosis:"Pneumonia",
      score: params.symptoms.includes("fever") ? 0.6 : 0.3
    })

  }

  return results
    .sort((a,b)=>b.score-a.score)
}

This dramatically improves reasoning quality without ML.

8. Dynamic Question Engine

export function suggestQuestions(state:any){

  const questions:string[] = []

  if(state.complaint==="cough"){

    if(!state.symptoms.includes("shortness"))
      questions.push("Any shortness of breath?")

    if(!state.symptoms.includes("chest pain"))
      questions.push("Any chest pain?")
  }

  if(state.complaint==="sore_throat"){

    questions.push("Any fever?")
    questions.push("Any cough?")
  }

  return questions.slice(0,4)
}
