import { redis } from "../infra/redisClient";

const FAILURE_THRESHOLD  = 5;
const RECOVERY_WINDOW_MS = 60_000;

export interface CircuitState {
  state: "closed" | "open" | "half-open";
  failureCount: number;
  lastFailureAt: number;
  openedAt?: number;
}

function key(agent: string) {
  return `cb:${agent}`;
}

export class RedisCircuitBreaker {

  async getState(agent: string): Promise<CircuitState> {
    const data = await redis.hgetall(key(agent));

    if (!data || Object.keys(data).length === 0) {
      return { state: "closed", failureCount: 0, lastFailureAt: 0 };
    }

    const state: CircuitState = {
      state: data.state as any,
      failureCount: Number(data.failureCount),
      lastFailureAt: Number(data.lastFailureAt),
      openedAt: data.openedAt ? Number(data.openedAt) : undefined,
    };

    // auto half-open transition
    if (
      state.state === "open" &&
      state.openedAt &&
      Date.now() - state.openedAt > RECOVERY_WINDOW_MS
    ) {
      await redis.hset(key(agent), "state", "half-open");
      return { ...state, state: "half-open" };
    }

    return state;
  }

  async recordFailure(agent: string) {
    const k = key(agent);

    const multi = redis.multi();
    multi.hincrby(k, "failureCount", 1);
    multi.hset(k, "lastFailureAt", Date.now());

    const [_, failureCountRaw] = await multi.exec() as any;
    const failureCount = Number(failureCountRaw[1]);

    if (failureCount >= FAILURE_THRESHOLD) {
      await redis.hset(k, {
        state: "open",
        openedAt: Date.now(),
      });

      console.warn(`[CB] OPEN → ${agent}`);
    }
  }

  async recordSuccess(agent: string) {
    const k = key(agent);
    const state = await this.getState(agent);

    if (state.state === "half-open") {
      await redis.del(k); // reset fully
      return;
    }

    if (state.failureCount > 0) {
      await redis.hset(k, "failureCount", 0);
    }
  }

  async forceOpen(agent: string) {
    await redis.hset(key(agent), {
      state: "open",
      failureCount: FAILURE_THRESHOLD,
      openedAt: Date.now(),
    });
  }

  async reset(agent: string) {
    await redis.del(key(agent));
  }
}

export const circuitBreaker = new RedisCircuitBreaker();

export class ReplayEngine {
  constructor(private orchestrator: Orchestrator) {}

  async replay(traceId: string) {
    const trace = await db.query.auditLogs.findMany({
      where: (t, { eq }) => eq(t.traceId, traceId),
      orderBy: (t, { asc }) => asc(t.createdAt),
    });

    if (!trace.length) {
      throw new Error("Trace not found");
    }

    const initialContext = trace[0].input?.context;

    const replayResults = await this.orchestrator.run(
      initialContext,
      `replay-${traceId}`
    );

    return {
      originalTrace: trace,
      replayResults,
    };
  }
}

import { redis } from "../infra/redisClient";

const HEALTH_KEY = "agent:health";

export interface AgentHealth {
  agent: string;
  success: number;
  failures: number;
  timeouts: number;
  lastUpdated: number;
}

export class SelfHealingEngine {

  async recordSuccess(agent: string) {
    await redis.hincrby(`${HEALTH_KEY}:${agent}`, "success", 1);
    await redis.hset(`${HEALTH_KEY}:${agent}`, "lastUpdated", Date.now());
  }

  async recordFailure(agent: string, isTimeout = false) {
    await redis.hincrby(`${HEALTH_KEY}:${agent}`, "failures", 1);
    if (isTimeout) {
      await redis.hincrby(`${HEALTH_KEY}:${agent}`, "timeouts", 1);
    }
    await redis.hset(`${HEALTH_KEY}:${agent}`, "lastUpdated", Date.now());
  }

  async getHealth(agent: string) {
    const data = await redis.hgetall(`${HEALTH_KEY}:${agent}`);
    const success = Number(data.success || 0);
    const failures = Number(data.failures || 0);
    const timeouts = Number(data.timeouts || 0);

    const total = success + failures;
    const successRate = total === 0 ? 1 : success / total;

    const score =
      successRate -
      (timeouts * 0.1) -
      (failures * 0.05);

    return {
      agent,
      success,
      failures,
      timeouts,
      successRate,
      score,
    };
  }

  async getAllAgentsHealth() {
    const keys = await redis.keys(`${HEALTH_KEY}:*`);
    return Promise.all(keys.map(k => this.getHealth(k.split(":")[2])));
  }
}

export const selfHealing = new SelfHealingEngine();

const WEIGHT_KEY = "agent:weight";

export class WeightAdapter {

  async adjust(agent: string) {
    const health = await selfHealing.getHealth(agent);

    let weight = 1;

    if (health.score < 0.5) weight = 0.5;
    if (health.score < 0.2) weight = 0.2;
    if (health.score < 0)   weight = 0.1;

    await redis.hset(WEIGHT_KEY, agent, weight);
  }

  async getWeight(agent: string) {
    const w = await redis.hget(WEIGHT_KEY, agent);
    return Number(w || 1);
  }
}

import { db } from "../db";

export class ValidationEngine {

  constructor(private orchestrator: Orchestrator) {}

  async runGoldenCases(modifiedConfig: any) {
    const cases = await db.query.goldenCases.findMany();

    let passed = 0;

    for (const c of cases) {
      const result = await this.orchestrator.run(c.input, "validation");

      if (JSON.stringify(result.final) === JSON.stringify(c.expected)) {
        passed++;
      }
    }

    const score = passed / cases.length;

    return {
      total: cases.length,
      passed,
      score,
      passedThreshold: score > 0.9,
    };
  }
}

🧠 PART 2 — GRAPH ENGINE CORE
📁 server/reasoning/clinicalGraphEngine.ts
import { ClinicalNode, ClinicalEdge, TraversalState } from "./clinicalGraphTypes";

export class ClinicalGraphEngine {
  private nodes = new Map<string, ClinicalNode>();
  private edges: ClinicalEdge[] = [];

  addNode(node: ClinicalNode) {
    this.nodes.set(node.id, node);
  }

  addEdge(edge: ClinicalEdge) {
    this.edges.push(edge);
  }

  getOutgoing(nodeId: string) {
    return this.edges.filter(e => e.from === nodeId);
  }

  traverse(startNodes: string[], context: any) {
    const results: TraversalState[] = [];

    const dfs = (state: TraversalState) => {
      const current = state.path[state.path.length - 1];

      const outgoing = this.getOutgoing(current);

      if (!outgoing.length) {
        results.push(state);
        return;
      }

      for (const edge of outgoing) {
        if (edge.condition && !edge.condition(context)) continue;

        const nextNode = edge.to;

        if (state.visited.has(nextNode)) continue;

        const nextState: TraversalState = {
          visited: new Set([...state.visited, nextNode]),
          path: [...state.path, nextNode],
          score: state.score + (edge.likelihood || 0),
          riskAccum: state.riskAccum + (edge.risk || 0),
        };

        dfs(nextState);
      }
    };

    for (const start of startNodes) {
      dfs({
        visited: new Set([start]),
        path: [start],
        score: 0,
        riskAccum: 0,
      });
    }

    return results.sort((a, b) => b.score - a.score);
  }
}

🧬 PART 1 — TEMPORAL + BAYESIAN GRAPH CORE

We upgrade your graph into a time-aware probabilistic engine.

📁 server/reasoning/temporalBayesianGraph.ts
export interface TemporalState {
  time: number;                  // timestamp
  node: string;
  probability: number;
}

export interface BayesianUpdate {
  prior: number;
  likelihood: number;
}

function bayesUpdate({ prior, likelihood }: BayesianUpdate) {
  const numerator = likelihood * prior;
  const denominator =
    numerator + (1 - likelihood) * (1 - prior);

  return denominator === 0 ? prior : numerator / denominator;
}

export class TemporalBayesianGraph {
  private transitions: any[] = [];

  addTransition(from: string, to: string, likelihood: number) {
    this.transitions.push({ from, to, likelihood });
  }

  evolve(states: TemporalState[], observations: any, time: number) {
    const nextStates: TemporalState[] = [];

    for (const state of states) {
      const outgoing = this.transitions.filter(t => t.from === state.node);

      for (const t of outgoing) {
        const likelihood = this.adjustLikelihood(t.likelihood, observations);

        const updatedProb = bayesUpdate({
          prior: state.probability,
          likelihood,
        });

        nextStates.push({
          time,
          node: t.to,
          probability: updatedProb,
        });
      }
    }

    return nextStates.sort((a, b) => b.probability - a.probability);
  }

  private adjustLikelihood(base: number, obs: any) {
    // plug in clinical modifiers
    if (obs.fever) return base + 0.1;
    if (obs.immunocompromised) return base + 0.2;
    return base;
  }
}

          patientId: p.id,

        });
      }
    }

    return alerts;
  }
}

export interface PatientVitals {
  hr?: number;
  spo2?: number;
  temp?: number;
  systolic?: number;
  diastolic?: number;
}

export interface PatientTwin {
  patientId: string;
  vitals: PatientVitals;
  lastUpdated: number;
  riskScore: number;
}

export class DigitalTwinEngine {
  private twins = new Map<string, PatientTwin>();

  updateFromDevice(reading: any) {
    let twin = this.twins.get(reading.patientId);

    if (!twin) {
      twin = {
        patientId: reading.patientId,
        vitals: {},
        lastUpdated: Date.now(),
        riskScore: 0,
      };
    }

    switch (reading.deviceType) {
      case "hr":
        twin.vitals.hr = reading.value;
        break;
      case "spo2":
        twin.vitals.spo2 = reading.value;
        break;
      case "temp":
        twin.vitals.temp = reading.value;
        break;
      case "bp":
        twin.vitals.systolic = reading.value.systolic;
        twin.vitals.diastolic = reading.value.diastolic;
        break;
    }

    twin.lastUpdated = Date.now();

    this.twins.set(reading.patientId, twin);
    return twin;
  }

  getTwin(patientId: string) {
    return this.twins.get(patientId);
  }

  getAll() {
    return Array.from(this.twins.values());
  }
}

import { PatientTwin } from "./digitalTwinEngine";

export class ICUPredictionEngine {

  computeRisk(twin: PatientTwin) {
    let risk = 0;

    const v = twin.vitals;

    if (v.spo2 && v.spo2 < 92) risk += 0.4;
    if (v.hr && v.hr > 120) risk += 0.3;
    if (v.temp && v.temp > 39) risk += 0.2;
    if (v.systolic && v.systolic < 90) risk += 0.5;

    // normalize
    risk = Math.min(1, risk);

    return risk;
  }

  classify(risk: number) {
    if (risk > 0.7) return "critical";
    if (risk > 0.4) return "high";
    return "stable";
  }

  evaluate(twin: PatientTwin) {
    const risk = this.computeRisk(twin);
    return {
      ...twin,
      riskScore: risk,
      status: this.classify(risk),
    };
  }
}

export interface TrendResult {
  slope: number;
  direction: "up" | "down" | "stable";
  velocity: number;
}

export class TrendEngine {
  private history = new Map<string, number[]>();

  update(key: string, value: number): TrendResult {
    const arr = this.history.get(key) || [];

    arr.push(value);
    if (arr.length > 10) arr.shift();

    this.history.set(key, arr);

    if (arr.length < 3) {
      return { slope: 0, direction: "stable", velocity: 0 };
    }

    const slope = arr[arr.length - 1] - arr[0];
    const velocity = slope / arr.length;

    let direction: "up" | "down" | "stable" = "stable";
    if (slope > 2) direction = "up";
    if (slope < -2) direction = "down";

    return { slope, direction, velocity };
  }
}

export class WaveformEngine {

  detectArrhythmia(signal: number[]) {
    if (signal.length < 5) return false;

    let irregular = 0;

    for (let i = 1; i < signal.length; i++) {
      const diff = Math.abs(signal[i] - signal[i - 1]);

      if (diff > 20) irregular++;
    }

    return irregular > 2;
  }

  detectTachy(signal: number[]) {
    const avg = signal.reduce((a, b) => a + b, 0) / signal.length;
    return avg > 120;
  }

  analyze(signal: number[]) {
    return {
      arrhythmia: this.detectArrhythmia(signal),
      tachycardia: this.detectTachy(signal),
    };
  }
}

import { trendEngine } from "./trendEngine";

export class MultivariateFusionEngine {

  fuse(twin: any) {
    const v = twin.vitals;

    const hrTrend   = trendEngine.update(`${twin.patientId}:hr`, v.hr || 0);
    const spo2Trend = trendEngine.update(`${twin.patientId}:spo2`, v.spo2 || 0);
    const bpTrend   = trendEngine.update(`${twin.patientId}:bp`, v.systolic || 0);

    let risk = 0;

    // Base vitals
    if (v.spo2 && v.spo2 < 92) risk += 0.3;
    if (v.hr && v.hr > 120) risk += 0.2;
    if (v.systolic && v.systolic < 90) risk += 0.4;

    // Trend amplification (THIS IS THE BIG UPGRADE)
    if (spo2Trend.direction === "down") risk += 0.3;
    if (bpTrend.direction === "down") risk += 0.3;
    if (hrTrend.direction === "up") risk += 0.2;

    // Cross-signal interaction (critical insight)
    if (
      spo2Trend.direction === "down" &&
      bpTrend.direction === "down"
    ) {
      risk += 0.5; // shock-like pattern
    }

    return Math.min(1, risk);
  }
}

export class TimeToEventEngine {

  predict(twin: any, risk: number) {
    const v = twin.vitals;

    let hours = 24;

    if (risk > 0.8) hours = 1;
    else if (risk > 0.6) hours = 3;
    else if (risk > 0.4) hours = 6;

    // refine using vitals
    if (v.spo2 && v.spo2 < 90) hours *= 0.5;
    if (v.systolic && v.systolic < 85) hours *= 0.5;

    return {
      estimatedHoursToEvent: Math.max(0.5, hours),
      confidence: Math.min(1, risk + 0.2),
    };
  }
}

export class BaselineEngine {
  private baselines = new Map<string, any>();

  updateBaseline(patientId: string, vitals: any) {
    let base = this.baselines.get(patientId) || {
      hr: [],
      spo2: [],
      systolic: [],
    };

    base.hr.push(vitals.hr || 0);
    base.spo2.push(vitals.spo2 || 0);
    base.systolic.push(vitals.systolic || 0);

    if (base.hr.length > 50) {
      base.hr.shift();
      base.spo2.shift();
      base.systolic.shift();
    }

    this.baselines.set(patientId, base);
  }

  getDeviation(patientId: string, vitals: any) {
    const base = this.baselines.get(patientId);
    if (!base) return 0;

    const avg = (arr: number[]) =>
      arr.reduce((a, b) => a + b, 0) / arr.length;

    let deviation = 0;

    if (vitals.hr) deviation += Math.abs(vitals.hr - avg(base.hr)) / 100;
    if (vitals.spo2) deviation += Math.abs(vitals.spo2 - avg(base.spo2)) / 100;
    if (vitals.systolic) deviation += Math.abs(vitals.systolic - avg(base.systolic)) / 100;

    return deviation;
  }
}

export class SequenceEngine {
  private history = new Map<string, any[]>();

  update(patientId: string, vitals: any) {
    const arr = this.history.get(patientId) || [];

    arr.push({
      t: Date.now(),
      ...vitals,
    });

    if (arr.length > 20) arr.shift();

    this.history.set(patientId, arr);

    return arr;
  }

  extractFeatures(sequence: any[]) {
    if (sequence.length < 3) return [];

    const first = sequence[0];
    const last  = sequence[sequence.length - 1];

    const slope = (k: string) =>
      (last[k] || 0) - (first[k] || 0);

    return [
      last.hr || 0,
      last.spo2 || 0,
      last.systolic || 0,
      last.temp || 0,

      slope("hr"),
      slope("spo2"),
      slope("systolic"),
    ];
  }
}

import { redis } from "../infra/redisClient";

const KEY = "thresholds";

export class ThresholdEngine {

  async getThreshold() {
    const t = await redis.get(KEY);
    return t ? Number(t) : 0.6;
  }

  async updateThreshold(outcomes: any[]) {
    // outcomes: [{ predicted, actual }]
    let tp = 0, fp = 0;

    for (const o of outcomes) {
      if (o.predicted && o.actual) tp++;
      if (o.predicted && !o.actual) fp++;
    }

    const precision = tp / Math.max(1, tp + fp);

    let threshold = await this.getThreshold();

    if (precision < 0.7) threshold += 0.05; // reduce false positives
    if (precision > 0.9) threshold -= 0.05; // increase sensitivity

    threshold = Math.min(0.9, Math.max(0.3, threshold));

    await redis.set(KEY, threshold);
  }
}

import { db } from "../db";

export interface FeatureRow {
  patientId: string;
  timestamp: number;
  features: number[];
  label?: number;
}

export class FeatureStore {

  async write(row: FeatureRow) {
    await db.insert("feature_store").values({
      patientId: row.patientId,
      timestamp: row.timestamp,
      features: JSON.stringify(row.features),
      label: row.label ?? null,
    });
  }

  async getSequence(patientId: string, limit = 20) {
    const rows = await db.query.feature_store.findMany({
      where: (t, { eq }) => eq(t.patientId, patientId),
      orderBy: (t, { desc }) => desc(t.timestamp),
      limit,
    });

    return rows.map(r => JSON.parse(r.features)).reverse();
  }

  async getTrainingData() {
    const rows = await db.select().from("feature_store");

    return rows
      .filter(r => r.label !== null)
      .map(r => ({
        features: JSON.parse(r.features),
        label: r.label,
      }));
  }
}

  async logPrediction(pred: number, actual?: number) {
    await db.insert("model_metrics").values({
      prediction: pred,
      actual: actual ?? null,
      createdAt: new Date(),
    });

  async getMetrics() {

    let tp = 0, fp = 0, fn = 0;

    for (const r of rows) {
      if (r.actual === null) continue;

      if (r.prediction > 0.5 && r.actual === 1) tp++;
      if (r.prediction > 0.5 && r.actual === 0) fp++;
      if (r.prediction <= 0.5 && r.actual === 1) fn++;
    }

    return {
      precision: tp / Math.max(1, tp + fp),
      recall: tp / Math.max(1, tp + fn),
    };
  }
}

// id="drift_detector_1"
export class DriftDetector {
  private baselineMean = 0;
  private currentMean = 0;

  updateBaseline(values: number[]) {
    this.baselineMean =
      values.reduce((a, b) => a + b, 0) / values.length;
  }

  detect(values: number[]) {
    this.currentMean =
      values.reduce((a, b) => a + b, 0) / values.length;

    const drift = Math.abs(this.currentMean - this.baselineMean);

    return {

// id="ensemble_engine_1"
import { mlPredict } from "./mlClient";
import { lstmPredict } from "./lstmClient";

export class EnsembleEngine {

  async predict(input: {
    features: number[];
    sequence: number[][];
    ruleRisk: number;
  }) {

    const [xgb, lstm] = await Promise.all([
      mlPredict(input.features),
      lstmPredict(input.sequence),
    ]);

    const xgbRisk  = xgb.risk;
    const lstmRisk = lstm.risk;

    // Weighted fusion (tunable / learnable later)
    const final =
      0.4 * xgbRisk +
      0.4 * lstmRisk +
      0.2 * input.ruleRisk;

    return {
      finalRisk: Math.min(1, final),
      components: {
        xgb: xgbRisk,
        lstm: lstmRisk,
        rules: input.ruleRisk,
      }
    };
  }
}

// id="causal_graph_1"
export interface CausalEdge {
  from: string;
  to: string;
  effect: number; // causal strength
}

export class CausalGraph {
  private edges: CausalEdge[] = [];

  addEdge(edge: CausalEdge) {
    this.edges.push(edge);
  }

  intervene(node: string, value: number, state: any) {
    const updated = { ...state, [node]: value };

    for (const e of this.edges) {
      if (e.from === node) {
        updated[e.to] =
          (updated[e.to] || 0) + value * e.effect;
      }
    }

    return updated;
  }
}

📁 server/rl/rlAgent.ts
// id="rl_agent_1"
export class RLAgent {
  private qTable = new Map<string, number>();

  private key(state: any, action: any) {
    return JSON.stringify({ state, action });
  }

  getQ(state: any, action: any) {
    return this.qTable.get(this.key(state, action)) || 0;
  }

  update(state: any, action: any, reward: number, nextState: any) {
    const alpha = 0.1;
    const gamma = 0.9;

    const currentQ = this.getQ(state, action);

    const maxNextQ = Math.max(
      ...["fluids", "oxygen", "antibiotics"].map(a =>
        this.getQ(nextState, { type: a, value: 1 })
      )
    );

    const newQ =
      currentQ +
      alpha * (reward + gamma * maxNextQ - currentQ);

    this.qTable.set(this.key(state, action), newQ);
  }

  chooseAction(state: any) {
    const actions = ["fluids", "oxygen", "antibiotics"];

    return actions
      .map(a => ({
        action: a,
        q: this.getQ(state, { type: a, value: 1 }),
      }))
      .sort((a, b) => b.q - a.q)[0].action;
  }
}

# id="dqn_train_1"
import numpy as np
from dqn_model import build_dqn

ACTIONS = ["fluids", "oxygen", "antibiotics"]

class DQNAgent:
    def __init__(self, state_dim):
        self.model = build_dqn(state_dim, len(ACTIONS))
        self.gamma = 0.95
        self.epsilon = 1.0

    def act(self, state):
        if np.random.rand() < self.epsilon:
            return np.random.randint(len(ACTIONS))

        q_values = self.model.predict(state.reshape(1, -1))
        return np.argmax(q_values[0])

    def train(self, state, action, reward, next_state):
        target = reward + self.gamma * np.max(
            self.model.predict(next_state.reshape(1, -1))
        )

        target_f = self.model.predict(state.reshape(1, -1))
        target_f[0][action] = target

        self.model.fit(state.reshape(1, -1), target_f, epochs=1, verbose=0)

export function computeClinicalReward(prev: any, next: any) {

  if (next.spo2 > prev.spo2) reward += 2;

  // hemodynamic improvement
